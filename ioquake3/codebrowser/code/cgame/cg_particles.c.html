157/*
158===============
159CL_ClearParticles
160===============
161*/
162void CG_ClearParticles (void)
163{
164 int i;
165
166 memset( particles, 0, sizeof(particles) );
167
168 free_particles = &particles[0];
169 active_particles = NULL;
170
171 for (i=0 ;i<cl_numparticles ; i++)
172 {
173 particles[i].next = &particles[i+1];
174 particles[i].type = 0;
175 }
176 particles[cl_numparticles-1].next = NULL;
177
178 oldtime = cg.time;
179
180 // Ridah, init the shaderAnims
181 for (i=0; shaderAnimNames[i]; i++) {
182 int j;
183
184 for (j=0; j<shaderAnimCounts[i]; j++) {
185 shaderAnims[i][j] = trap_R_RegisterShader( va("%s%i", shaderAnimNames[i], j+1) );
186 }
187 }
188 numShaderAnims = i;
189 // done.
190
191 initparticles = qtrue;
192}

854void CG_AddParticles (void)
855{
856 cparticle_t *p, *next;
857 float alpha;
858 float time, time2;

861 vec3_t rotate_ang;

870 vectoangles( cg.refdef.viewaxis[0], rotate_ang );
871 roll += ((cg.time - oldtime) * 0.1) ;
872 rotate_ang[ROLL] += (roll*0.9);
873 AngleVectors ( rotate_ang, rforward, rright, rup);
874
875 oldtime = cg.time;
876
877 active = NULL;
878 tail = NULL;
879
880 for (p=active_particles ; p ; p=next)
881 {
882
883 next = p->next;
884
885 time = (cg.time - p->time)*0.001;
886
887 alpha = p->alpha + time*p->alphavel;
888 if (alpha <= 0)
889 { // faded out
890 p->next = free_particles;
891 free_particles = p;
892 p->type = 0;
893 p->color = 0;
894 p->alpha = 0;
895 continue;
896 }
897
898 if (p->type == P_SMOKE || p->type == P_ANIM || p->type == P_BLEED || p->type == P_SMOKE_IMPACT)

902 p->next = free_particles;
903 free_particles = p;
904 p->type = 0;
905 p->color = 0;
906 p->alpha = 0;
907
908 continue;
909 }
910
911 }
912
913 if (p->type == P_WEATHER_FLURRY)

917 p->next = free_particles;

919 p->type = 0;
920 p->color = 0;
921 p->alpha = 0;
922
923 continue;
924 }
925 }
926
927
928 if (p->type == P_FLAT_SCALEUP_FADE)
929 {
930 if (cg.time > p->endtime)
931 {
932 p->next = free_particles;
933 free_particles = p;
934 p->type = 0;
935 p->color = 0;
936 p->alpha = 0;
937 continue;
938 }
939
940 }
941
942 if ((p->type == P_BAT || p->type == P_SPRITE) && p->endtime < 0) {
943 // temporary sprite
944 CG_AddParticleToScene (p, p->org, alpha);
945 p->next = free_particles;
946 free_particles = p;
947 p->type = 0;
948 p->color = 0;
949 p->alpha = 0;
950 continue;
951 }
952
953 p->next = NULL;
954 if (!tail)
955 active = tail = p;
956 else
957 {
958 tail->next = p;
959 tail = p;
960 }
961
962 if (alpha > 1.0)
963 alpha = 1;
964

970
971 CG_AddParticleToScene (p, org, alpha);
972 }

974 active_particles = active;

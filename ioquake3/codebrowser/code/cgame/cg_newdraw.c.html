58void CG_CheckOrderPending(void) {
59 if (cgs.gametype < GT_CTF) {
60 return;
61 }
62 if (cgs.orderPending) {
63 //clientInfo_t *ci = cgs.clientinfo + sortedTeamPlayers[cg_currentSelectedPlayer.integer];
64 const char *p1, *p2, *b;
65 p1 = p2 = b = NULL;
66 switch (cgs.currentOrder) {
67 case TEAMTASK_OFFENSE:
68 p1 = VOICECHAT_ONOFFENSE;
69 p2 = VOICECHAT_OFFENSE;
70 b = "+button7; wait; -button7";
71 break;
72 case TEAMTASK_DEFENSE:
73 p1 = VOICECHAT_ONDEFENSE;
74 p2 = VOICECHAT_DEFEND;
75 b = "+button8; wait; -button8";
76 break;
77 case TEAMTASK_PATROL:
78 p1 = VOICECHAT_ONPATROL;
79 p2 = VOICECHAT_PATROL;
80 b = "+button9; wait; -button9";
81 break;
82 case TEAMTASK_FOLLOW:
83 p1 = VOICECHAT_ONFOLLOW;
84 p2 = VOICECHAT_FOLLOWME;
85 b = "+button10; wait; -button10";
86 break;
87 case TEAMTASK_CAMP:
88 p1 = VOICECHAT_ONCAMPING;
89 p2 = VOICECHAT_CAMP;
90 break;
91 case TEAMTASK_RETRIEVE:
92 p1 = VOICECHAT_ONGETFLAG;
93 p2 = VOICECHAT_RETURNFLAG;
94 break;
95 case TEAMTASK_ESCORT:
96 p1 = VOICECHAT_ONFOLLOWCARRIER;
97 p2 = VOICECHAT_FOLLOWFLAGCARRIER;
98 break;
99 }
100
101 if (cg_currentSelectedPlayer.integer == numSortedTeamPlayers) {
102 // to everyone
103 trap_SendConsoleCommand(va("cmd vsay_team %s\n", p2));
104 } else {
105 // for the player self
106 if (sortedTeamPlayers[cg_currentSelectedPlayer.integer] == cg.snap->ps.clientNum && p1) {
107 trap_SendConsoleCommand(va("teamtask %i\n", cgs.currentOrder));
108 //trap_SendConsoleCommand(va("cmd say_team %s\n", p2));
109 trap_SendConsoleCommand(va("cmd vsay_team %s\n", p1));
110 } else if (p2) {
111 //trap_SendConsoleCommand(va("cmd say_team %s, %s\n", ci->name,p));
112 trap_SendConsoleCommand(va("cmd vtell %d %s\n", sortedTeamPlayers[cg_currentSelectedPlayer.integer], p2));
113 }
114 }
115 if (b) {
116 trap_SendConsoleCommand(b);
117 }
118 cgs.orderPending = qfalse;
119 }

255static void CG_DrawPlayerHead(rectDef_t *rect, qboolean draw2D) {
256 vec3_t angles;
257 float size, stretch;
258 float frac;
259 float x = rect->x;
260
261 VectorClear( angles );
262
263 if ( cg.damageTime && cg.time - cg.damageTime < DAMAGE_TIME ) {
264 frac = (float)(cg.time - cg.damageTime ) / DAMAGE_TIME;
265 size = rect->w * 1.25 * ( 1.5 - frac * 0.5 );
266
267 stretch = size - rect->w * 1.25;
268 // kick in the direction of damage
269 x -= stretch * 0.5 + cg.damageX * stretch * 0.5;
270
271 cg.headStartYaw = 180 + cg.damageX * 45;
272
273 cg.headEndYaw = 180 + 20 * cos( crandom()*M_PI );
274 cg.headEndPitch = 5 * cos( crandom()*M_PI );
275
276 cg.headStartTime = cg.time;
277 cg.headEndTime = cg.time + 100 + random() * 2000;
278 } else {
279 if ( cg.time >= cg.headEndTime ) {
280 // select a new head angle
281 cg.headStartYaw = cg.headEndYaw;
282 cg.headStartPitch = cg.headEndPitch;
283 cg.headStartTime = cg.headEndTime;
284 cg.headEndTime = cg.time + 100 + random() * 2000;
285
286 cg.headEndYaw = 180 + 20 * cos( crandom()*M_PI );
287 cg.headEndPitch = 5 * cos( crandom()*M_PI );
288 }
289 }
290
291 // if the server was frozen for a while we may have a bad head start time
292 if ( cg.headStartTime > cg.time ) {
293 cg.headStartTime = cg.time;
294 }
295
296 frac = ( cg.time - cg.headStartTime ) / (float)( cg.headEndTime - cg.headStartTime );
297 frac = frac * frac * ( 3 - 2 * frac );
298 angles[YAW] = cg.headStartYaw + ( cg.headEndYaw - cg.headStartYaw ) * frac;
299 angles[PITCH] = cg.headStartPitch + ( cg.headEndPitch - cg.headStartPitch ) * frac;
300
301 CG_DrawHead( x, rect->y, rect->w, rect->h, cg.snap->ps.clientNum, angles );
302}

511static void CG_DrawSelectedPlayerHead( rectDef_t *rect, qboolean draw2D, qboolean voice ) {
512 clipHandle_t cm;
513 clientInfo_t *ci;
514 float len;
515 vec3_t origin;
516 vec3_t mins, maxs, angles;
517
518
519 ci = cgs.clientinfo + ((voice) ? cgs.currentVoiceClient : sortedTeamPlayers[CG_GetSelectedPlayer()]);
520
521 if (ci) {
522 if ( cg_draw3dIcons.integer ) {
523 cm = ci->headModel;
524 if ( !cm ) {
525 return;
526 }
527
528 // offset the origin y and z to center the head
529 trap_R_ModelBounds( cm, mins, maxs );
530
531 origin[2] = -0.5 * ( mins[2] + maxs[2] );
532 origin[1] = 0.5 * ( mins[1] + maxs[1] );
533
534 // calculate distance so the head nearly fills the box
535 // assume heads are taller than wide
536 len = 0.7 * ( maxs[2] - mins[2] );
537 origin[0] = len / 0.268; // len / tan( fov/2 )
538
539 // allow per-model tweaking
540 VectorAdd( origin, ci->headOffset, origin );
541
542 angles[PITCH] = 0;
543 angles[YAW] = 180;
544 angles[ROLL] = 0;
545
546 CG_Draw3DModel( rect->x, rect->y, rect->w, rect->h, ci->headModel, ci->headSkin, origin, angles );
547 } else if ( cg_drawIcons.integer ) {
548 CG_DrawPic( rect->x, rect->y, rect->w, rect->h, ci->modelIcon );
549 }
550
551 // if they are deferred, draw a cross out
552 if ( ci->deferred ) {
553 CG_DrawPic( rect->x, rect->y, rect->w, rect->h, cgs.media.deferShader );
554 }
555 }

804static void CG_DrawAreaPowerUp(rectDef_t *rect, int align, float special, float scale, vec4_t color) {
805 char num[16];
806 int sorted[MAX_POWERUPS];
807 int sortedTime[MAX_POWERUPS];
808 int i, j, k;
809 int active;
810 playerState_t *ps;
811 int t;
812 gitem_t *item;
813 float f;
814 rectDef_t r2;
815 float *inc;
816 r2.x = rect->x;
817 r2.y = rect->y;
818 r2.w = rect->w;
819 r2.h = rect->h;
820
821 inc = (align == HUD_VERTICAL) ? &r2.y : &r2.x;
822
823 ps = &cg.snap->ps;
824
825 if ( ps->stats[STAT_HEALTH] <= 0 ) {
826 return;
827 }
828
829 // sort the list by time remaining
830 active = 0;
831 for ( i = 0 ; i < MAX_POWERUPS ; i++ ) {
832 if ( !ps->powerups[ i ] ) {
833 continue;
834 }
835
836 // ZOID--don't draw if the power up has unlimited time
837 // This is true of the CTF flags
838 if ( ps->powerups[ i ] == INT_MAX ) {
839 continue;
840 }
841
842 t = ps->powerups[ i ] - cg.time;
843 if ( t <= 0 ) {
844 continue;
845 }
846
847 // insert into the list
848 for ( j = 0 ; j < active ; j++ ) {
849 if ( sortedTime[j] >= t ) {
850 for ( k = active - 1 ; k >= j ; k-- ) {
851 sorted[k+1] = sorted[k];
852 sortedTime[k+1] = sortedTime[k];
853 }
854 break;
855 }
856 }
857 sorted[j] = i;
858 sortedTime[j] = t;
859 active++;
860 }
861
862 // draw the icons and timers
863 for ( i = 0 ; i < active ; i++ ) {
864 item = BG_FindItemForPowerup( sorted[i] );
865
866 if (item) {
867 t = ps->powerups[ sorted[i] ];
868 if ( t - cg.time >= POWERUP_BLINKS * POWERUP_BLINK_TIME ) {
869 trap_R_SetColor( NULL );
870 } else {
871 vec4_t modulate;
872
873 f = (float)( t - cg.time ) / POWERUP_BLINK_TIME;
874 f -= (int)f;
875 modulate[0] = modulate[1] = modulate[2] = modulate[3] = f;
876 trap_R_SetColor( modulate );
877 }
878
879 CG_DrawPic( r2.x, r2.y, r2.w * .75, r2.h, trap_R_RegisterShader( item->icon ) );
880
881 Com_sprintf (num, sizeof(num), "%i", sortedTime[i] / 1000);
882 CG_Text_Paint(r2.x + (r2.w * .75) + 3 , r2.y + r2.h, scale, color, num, 0, 0, 0);
883 *inc += r2.w + special;
884 }
885
886 }
887 trap_R_SetColor( NULL );

1192 int len, count;

1228 glyph->imageHeight,
1229 useScale,

1231 glyph->t,
1232 glyph->s2,
1233 glyph->t2,
1234 glyph->glyph);

31static intptr_t (QDECL *syscall)( intptr_t arg, ... ) = (intptr_t (QDECL *)( intptr_t, ...))-1;
32
33
34Q_EXPORT void dllEntry( intptr_t (QDECL *syscallptr)( intptr_t arg,... ) ) {
35 syscall = syscallptr;
36}

39int PASSFLOAT( float x ) {
40 floatint_t fi;
41 fi.f = x;
42 return fi.i;
43}

49sfxHandle_t CG_CustomSound( int clientNum, const char *soundName ) {
50 clientInfo_t *ci;

52

54 return trap_S_RegisterSound( soundName, qfalse );
55 }

59 }

61

64 return ci->sounds[i];

66 }
67
68 CG_Error( "Unknown custom sound: %s", soundName );
69 return 0;
70}

82/*
83======================
84CG_ParseAnimationFile
85
86Read a configuration file containing animation counts and rates
87models/players/visor/animation.cfg, etc
88======================
89*/
90static qboolean CG_ParseAnimationFile( const char *filename, clientInfo_t *ci ) {
91 char *text_p, *prev;
92 int len;
93 int i;
94 char *token;
95 float fps;
96 int skip;
97 char text[20000];
98 fileHandle_t f;
99 animation_t *animations;
100
101 animations = ci->animations;
102
103 // load the file
104 len = trap_FS_FOpenFile( filename, &f, FS_READ );
105 if ( len <= 0 ) {
106 return qfalse;
107 }
108 if ( len >= sizeof( text ) - 1 ) {
109 CG_Printf( "File %s too long\n", filename );
110 trap_FS_FCloseFile( f );
111 return qfalse;
112 }
113 trap_FS_Read( text, len, f );
114 text[len] = 0;
115 trap_FS_FCloseFile( f );
116
117 // parse the text
118 text_p = text;
119 skip = 0; // quite the compiler warning
120
121 ci->footsteps = FOOTSTEP_NORMAL;
122 VectorClear( ci->headOffset );
123 ci->gender = GENDER_MALE;
124 ci->fixedlegs = qfalse;
125 ci->fixedtorso = qfalse;
126
127 // read optional parameters
128 while ( 1 ) {
129 prev = text_p; // so we can unget
130 token = COM_Parse( &text_p );
131 if ( !token[0] ) {
132 break;
133 }
134 if ( !Q_stricmp( token, "footsteps" ) ) {
135 token = COM_Parse( &text_p );
136 if ( !token[0] ) {
137 break;
138 }
139 if ( !Q_stricmp( token, "default" ) || !Q_stricmp( token, "normal" ) ) {
140 ci->footsteps = FOOTSTEP_NORMAL;
141 } else if ( !Q_stricmp( token, "boot" ) ) {
142 ci->footsteps = FOOTSTEP_BOOT;
143 } else if ( !Q_stricmp( token, "flesh" ) ) {
144 ci->footsteps = FOOTSTEP_FLESH;
145 } else if ( !Q_stricmp( token, "mech" ) ) {
146 ci->footsteps = FOOTSTEP_MECH;
147 } else if ( !Q_stricmp( token, "energy" ) ) {
148 ci->footsteps = FOOTSTEP_ENERGY;
149 } else {
150 CG_Printf( "Bad footsteps parm in %s: %s\n", filename, token );
151 }
152 continue;
153 } else if ( !Q_stricmp( token, "headoffset" ) ) {
154 for ( i = 0 ; i < 3 ; i++ ) {
155 token = COM_Parse( &text_p );
156 if ( !token[0] ) {
157 break;
158 }
159 ci->headOffset[i] = atof( token );
160 }
161 continue;
162 } else if ( !Q_stricmp( token, "sex" ) ) {
163 token = COM_Parse( &text_p );
164 if ( !token[0] ) {
165 break;
166 }
167 if ( token[0] == 'f' || token[0] == 'F' ) {
168 ci->gender = GENDER_FEMALE;
169 } else if ( token[0] == 'n' || token[0] == 'N' ) {
170 ci->gender = GENDER_NEUTER;
171 } else {
172 ci->gender = GENDER_MALE;
173 }
174 continue;
175 } else if ( !Q_stricmp( token, "fixedlegs" ) ) {
176 ci->fixedlegs = qtrue;
177 continue;
178 } else if ( !Q_stricmp( token, "fixedtorso" ) ) {
179 ci->fixedtorso = qtrue;
180 continue;
181 }
182
183 // if it is a number, start parsing animations
184 if ( token[0] >= '0' && token[0] <= '9' ) {
185 text_p = prev; // unget the token
186 break;
187 }
188 Com_Printf( "unknown token '%s' in %s\n", token, filename );
189 }
190
191 // read information for each frame
192 for ( i = 0 ; i < MAX_ANIMATIONS ; i++ ) {
193
194 token = COM_Parse( &text_p );
195 if ( !token[0] ) {
196 if( i >= TORSO_GETFLAG && i <= TORSO_NEGATIVE ) {
197 animations[i].firstFrame = animations[TORSO_GESTURE].firstFrame;
198 animations[i].frameLerp = animations[TORSO_GESTURE].frameLerp;
199 animations[i].initialLerp = animations[TORSO_GESTURE].initialLerp;
200 animations[i].loopFrames = animations[TORSO_GESTURE].loopFrames;
201 animations[i].numFrames = animations[TORSO_GESTURE].numFrames;
202 animations[i].reversed = qfalse;
203 animations[i].flipflop = qfalse;
204 continue;
205 }
206 break;
207 }
208 animations[i].firstFrame = atoi( token );
209 // leg only frames are adjusted to not count the upper body only frames
210 if ( i == LEGS_WALKCR ) {
211 skip = animations[LEGS_WALKCR].firstFrame - animations[TORSO_GESTURE].firstFrame;
212 }
213 if ( i >= LEGS_WALKCR && i<TORSO_GETFLAG) {
214 animations[i].firstFrame -= skip;
215 }
216
217 token = COM_Parse( &text_p );
218 if ( !token[0] ) {
219 break;
220 }
221 animations[i].numFrames = atoi( token );
222
223 animations[i].reversed = qfalse;
224 animations[i].flipflop = qfalse;
225 // if numFrames is negative the animation is reversed
226 if (animations[i].numFrames < 0) {
227 animations[i].numFrames = -animations[i].numFrames;
228 animations[i].reversed = qtrue;
229 }
230
231 token = COM_Parse( &text_p );
232 if ( !token[0] ) {
233 break;
234 }
235 animations[i].loopFrames = atoi( token );
236
237 token = COM_Parse( &text_p );
238 if ( !token[0] ) {
239 break;
240 }
241 fps = atof( token );
242 if ( fps == 0 ) {
243 fps = 1;
244 }
245 animations[i].frameLerp = 1000 / fps;
246 animations[i].initialLerp = 1000 / fps;
247 }
248
249 if ( i != MAX_ANIMATIONS ) {
250 CG_Printf( "Error parsing animation file: %s\n", filename );
251 return qfalse;
252 }
253
254 // crouch backward animation
255 memcpy(&animations[LEGS_BACKCR], &animations[LEGS_WALKCR], sizeof(animation_t));
256 animations[LEGS_BACKCR].reversed = qtrue;
257 // walk backward animation
258 memcpy(&animations[LEGS_BACKWALK], &animations[LEGS_WALK], sizeof(animation_t));
259 animations[LEGS_BACKWALK].reversed = qtrue;
260 // flag moving fast
261 animations[FLAG_RUN].firstFrame = 0;
262 animations[FLAG_RUN].numFrames = 16;
263 animations[FLAG_RUN].loopFrames = 16;
264 animations[FLAG_RUN].frameLerp = 1000 / 15;
265 animations[FLAG_RUN].initialLerp = 1000 / 15;
266 animations[FLAG_RUN].reversed = qfalse;
267 // flag not moving or moving slowly
268 animations[FLAG_STAND].firstFrame = 16;
269 animations[FLAG_STAND].numFrames = 5;
270 animations[FLAG_STAND].loopFrames = 0;
271 animations[FLAG_STAND].frameLerp = 1000 / 20;
272 animations[FLAG_STAND].initialLerp = 1000 / 20;
273 animations[FLAG_STAND].reversed = qfalse;
274 // flag speeding up
275 animations[FLAG_STAND2RUN].firstFrame = 16;
276 animations[FLAG_STAND2RUN].numFrames = 5;
277 animations[FLAG_STAND2RUN].loopFrames = 1;
278 animations[FLAG_STAND2RUN].frameLerp = 1000 / 15;
279 animations[FLAG_STAND2RUN].initialLerp = 1000 / 15;
280 animations[FLAG_STAND2RUN].reversed = qtrue;
281 //
282 // new anims changes
283 //
284// animations[TORSO_GETFLAG].flipflop = qtrue;
285// animations[TORSO_GUARDBASE].flipflop = qtrue;
286// animations[TORSO_PATROL].flipflop = qtrue;
287// animations[TORSO_AFFIRMATIVE].flipflop = qtrue;
288// animations[TORSO_NEGATIVE].flipflop = qtrue;
289 //
290 return qtrue;
291}

522static qboolean CG_RegisterClientModelname( clientInfo_t *ci, const char *modelName, const char *skinName, const char *headModelName, const char *headSkinName, const char *teamName ) {
523 char filename[MAX_QPATH];
524 const char *headName;
525 char newTeamName[MAX_QPATH];
526
527 if ( headModelName[0] == '\0' ) {
528 headName = modelName;
529 }
530 else {
531 headName = headModelName;
532 }
533 Com_sprintf( filename, sizeof( filename ), "models/players/%s/lower.md3", modelName );
534 ci->legsModel = trap_R_RegisterModel( filename );
535 if ( !ci->legsModel ) {
536 Com_sprintf( filename, sizeof( filename ), "models/players/characters/%s/lower.md3", modelName );
537 ci->legsModel = trap_R_RegisterModel( filename );
538 if ( !ci->legsModel ) {
539 Com_Printf( "Failed to load model file %s\n", filename );
540 return qfalse;
541 }
542 }
543
544 Com_sprintf( filename, sizeof( filename ), "models/players/%s/upper.md3", modelName );
545 ci->torsoModel = trap_R_RegisterModel( filename );
546 if ( !ci->torsoModel ) {
547 Com_sprintf( filename, sizeof( filename ), "models/players/characters/%s/upper.md3", modelName );
548 ci->torsoModel = trap_R_RegisterModel( filename );
549 if ( !ci->torsoModel ) {
550 Com_Printf( "Failed to load model file %s\n", filename );
551 return qfalse;
552 }
553 }
554
555 if( headName[0] == '*' ) {
556 Com_sprintf( filename, sizeof( filename ), "models/players/heads/%s/%s.md3", &headModelName[1], &headModelName[1] );
557 }
558 else {
559 Com_sprintf( filename, sizeof( filename ), "models/players/%s/head.md3", headName );
560 }
561 ci->headModel = trap_R_RegisterModel( filename );
562 // if the head model could not be found and we didn't load from the heads folder try to load from there
563 if ( !ci->headModel && headName[0] != '*' ) {
564 Com_sprintf( filename, sizeof( filename ), "models/players/heads/%s/%s.md3", headModelName, headModelName );
565 ci->headModel = trap_R_RegisterModel( filename );
566 }
567 if ( !ci->headModel ) {
568 Com_Printf( "Failed to load model file %s\n", filename );
569 return qfalse;
570 }
571
572 // if any skins failed to load, return failure
573 if ( !CG_RegisterClientSkin( ci, teamName, modelName, skinName, headName, headSkinName ) ) {
574 if ( teamName && *teamName) {
575 Com_Printf( "Failed to load skin file: %s : %s : %s, %s : %s\n", teamName, modelName, skinName, headName, headSkinName );
576 if( ci->team == TEAM_BLUE ) {
577 Com_sprintf(newTeamName, sizeof(newTeamName), "%s/", DEFAULT_BLUETEAM_NAME);
578 }
579 else {
580 Com_sprintf(newTeamName, sizeof(newTeamName), "%s/", DEFAULT_REDTEAM_NAME);
581 }
582 if ( !CG_RegisterClientSkin( ci, newTeamName, modelName, skinName, headName, headSkinName ) ) {
583 Com_Printf( "Failed to load skin file: %s : %s : %s, %s : %s\n", newTeamName, modelName, skinName, headName, headSkinName );
584 return qfalse;
585 }
586 } else {
587 Com_Printf( "Failed to load skin file: %s : %s, %s : %s\n", modelName, skinName, headName, headSkinName );
588 return qfalse;
589 }
590 }
591
592 // load the animations
593 Com_sprintf( filename, sizeof( filename ), "models/players/%s/animation.cfg", modelName );
594 if ( !CG_ParseAnimationFile( filename, ci ) ) {
595 Com_sprintf( filename, sizeof( filename ), "models/players/characters/%s/animation.cfg", modelName );
596 if ( !CG_ParseAnimationFile( filename, ci ) ) {
597 Com_Printf( "Failed to load animation file %s\n", filename );
598 return qfalse;
599 }
600 }
601
602 if ( CG_FindClientHeadFile( filename, sizeof(filename), ci, teamName, headName, headSkinName, "icon", "skin" ) ) {
603 ci->modelIcon = trap_R_RegisterShaderNoMip( filename );
604 }
605 else if ( CG_FindClientHeadFile( filename, sizeof(filename), ci, teamName, headName, headSkinName, "icon", "tga" ) ) {
606 ci->modelIcon = trap_R_RegisterShaderNoMip( filename );
607 }
608
609 if ( !ci->modelIcon ) {
610 return qfalse;
611 }
612
613 return qtrue;
614}

617====================
618CG_ColorFromString
619====================
620*/
621static void CG_ColorFromString( const char *v, vec3_t color ) {
622 int val;
623
624 VectorClear( color );
625
626 val = atoi( v );
627
628 if ( val < 1 || val > 7 ) {
629 VectorSet( color, 1, 1, 1 );
630 return;
631 }
632
633 if ( val & 1 ) {
634 color[2] = 1.0f;
635 }
636 if ( val & 2 ) {
637 color[1] = 1.0f;
638 }
639 if ( val & 4 ) {
640 color[0] = 1.0f;
641 }
642}

648Load it now, taking the disk hits.
649This will usually be deferred to a safe time

652static void CG_LoadClientInfo( int clientNum, clientInfo_t *ci ) {
653 const char *dir, *fallback;
654 int i, modelloaded;
655 const char *s;
656 char teamname[MAX_QPATH];
657
658 teamname[0] = 0;
659#ifdef MISSIONPACK

669 }
670#endif
671 modelloaded = qtrue;
672 if ( !CG_RegisterClientModelname( ci, ci->modelName, ci->skinName, ci->headModelName, ci->headSkinName, teamname ) ) {
673 if ( cg_buildScript.integer ) {
674 CG_Error( "CG_RegisterClientModelname( %s, %s, %s, %s %s ) failed", ci->modelName, ci->skinName, ci->headModelName, ci->headSkinName, teamname );
675 }
676
677 // fall back to default team name
678 if( cgs.gametype >= GT_TEAM) {
679 // keep skin name
680 if( ci->team == TEAM_BLUE ) {
681 Q_strncpyz(teamname, DEFAULT_BLUETEAM_NAME, sizeof(teamname) );
682 } else {

1130/*
1131===============
1132CG_RunLerpFrame
1133
1134Sets cg.snap, cg.oldFrame, and cg.backlerp
1135cg.time should be between oldFrameTime and frameTime after exit
1136===============
1137*/
1138static void CG_RunLerpFrame( clientInfo_t *ci, lerpFrame_t *lf, int newAnimation, float speedScale ) {
1139 int f, numFrames;
1140 animation_t *anim;
1141
1142 // debugging tool to get no animations
1143 if ( cg_animSpeed.integer == 0 ) {
1144 lf->oldFrame = lf->frame = lf->backlerp = 0;
1145 return;
1146 }
1147
1148 // see if the animation sequence is switching
1149 if ( newAnimation != lf->animationNumber || !lf->animation ) {
1150 CG_SetLerpFrameAnimation( ci, lf, newAnimation );
1151 }
1152
1153 // if we have passed the current frame, move it to
1154 // oldFrame and calculate a new frame
1155 if ( cg.time >= lf->frameTime ) {
1156 lf->oldFrame = lf->frame;
1157 lf->oldFrameTime = lf->frameTime;
1158
1159 // get the next frame based on the animation
1160 anim = lf->animation;
1161 if ( !anim->frameLerp ) {
1162 return; // shouldn't happen
1163 }
1164 if ( cg.time < lf->animationTime ) {
1165 lf->frameTime = lf->animationTime; // initial lerp
1166 } else {
1167 lf->frameTime = lf->oldFrameTime + anim->frameLerp;
1168 }
1169 f = ( lf->frameTime - lf->animationTime ) / anim->frameLerp;
1170 f *= speedScale; // adjust for haste, etc
1171
1172 numFrames = anim->numFrames;
1173 if (anim->flipflop) {
1174 numFrames *= 2;
1175 }
1176 if ( f >= numFrames ) {
1177 f -= numFrames;
1178 if ( anim->loopFrames ) {
1179 f %= anim->loopFrames;
1180 f += anim->numFrames - anim->loopFrames;
1181 } else {
1182 f = numFrames - 1;
1183 // the animation is stuck at the end, so it
1184 // can immediately transition to another sequence
1185 lf->frameTime = cg.time;
1186 }
1187 }
1188 if ( anim->reversed ) {
1189 lf->frame = anim->firstFrame + anim->numFrames - 1 - f;
1190 }
1191 else if (anim->flipflop && f>=anim->numFrames) {
1192 lf->frame = anim->firstFrame + anim->numFrames - 1 - (f%anim->numFrames);
1193 }
1194 else {
1195 lf->frame = anim->firstFrame + f;
1196 }
1197 if ( cg.time > lf->frameTime ) {
1198 lf->frameTime = cg.time;
1199 if ( cg_debugAnim.integer ) {
1200 CG_Printf( "Clamp lf->frameTime\n");
1201 }
1202 }
1203 }
1204
1205 if ( lf->frameTime > cg.time + 200 ) {
1206 lf->frameTime = cg.time;
1207 }
1208
1209 if ( lf->oldFrameTime > cg.time ) {
1210 lf->oldFrameTime = cg.time;
1211 }
1212 // calculate current lerp value
1213 if ( lf->frameTime == lf->oldFrameTime ) {
1214 lf->backlerp = 0;
1215 } else {
1216 lf->backlerp = 1.0 - (float)( cg.time - lf->oldFrameTime ) / ( lf->frameTime - lf->oldFrameTime );
1217 }
1218}

1285/*
1286==================
1287CG_SwingAngles
1288==================
1289*/
1290static void CG_SwingAngles( float destination, float swingTolerance, float clampTolerance,
1291 float speed, float *angle, qboolean *swinging ) {
1292 float swing;
1293 float move;
1294 float scale;
1295
1296 if ( !*swinging ) {
1297 // see if a swing should be started
1298 swing = AngleSubtract( *angle, destination );
1299 if ( swing > swingTolerance || swing < -swingTolerance ) {
1300 *swinging = qtrue;
1301 }
1302 }
1303
1304 if ( !*swinging ) {
1305 return;
1306 }
1307
1308 // modify the speed depending on the delta
1309 // so it doesn't seem so linear
1310 swing = AngleSubtract( destination, *angle );
1311 scale = fabs( swing );
1312 if ( scale < swingTolerance * 0.5 ) {
1313 scale = 0.5;
1314 } else if ( scale < swingTolerance ) {
1315 scale = 1.0;
1316 } else {
1317 scale = 2.0;
1318 }
1319
1320 // swing towards the destination angle
1321 if ( swing >= 0 ) {
1322 move = cg.frametime * scale * speed;
1323 if ( move >= swing ) {
1324 move = swing;
1325 *swinging = qfalse;
1326 }
1327 *angle = AngleMod( *angle + move );
1328 } else if ( swing < 0 ) {
1329 move = cg.frametime * scale * -speed;
1330 if ( move <= swing ) {
1331 move = swing;
1332 *swinging = qfalse;
1333 }
1334 *angle = AngleMod( *angle + move );
1335 }
1336
1337 // clamp to no more than tolerance
1338 swing = AngleSubtract( destination, *angle );
1339 if ( swing > clampTolerance ) {
1340 *angle = AngleMod( destination - (clampTolerance - 1) );
1341 } else if ( swing < -clampTolerance ) {
1342 *angle = AngleMod( destination + (clampTolerance - 1) );
1343 }
1344}

1384static void CG_PlayerAngles( centity_t *cent, vec3_t legs[3], vec3_t torso[3], vec3_t head[3] ) {
1385 vec3_t legsAngles, torsoAngles, headAngles;
1386 float dest;
1387 static int movementOffsets[8] = { 0, 22, 45, -22, 0, 22, -45, -22 };
1388 vec3_t velocity;
1389 float speed;
1390 int dir, clientNum;
1391 clientInfo_t *ci;
1392
1393 VectorCopy( cent->lerpAngles, headAngles );
1394 headAngles[YAW] = AngleMod( headAngles[YAW] );
1395 VectorClear( legsAngles );
1396 VectorClear( torsoAngles );
1397
1398 // --------- yaw -------------
1399
1400 // allow yaw to drift a bit
1401 if ( ( cent->currentState.legsAnim & ~ANIM_TOGGLEBIT ) != LEGS_IDLE
1402 || ((cent->currentState.torsoAnim & ~ANIM_TOGGLEBIT) != TORSO_STAND
1403 && (cent->currentState.torsoAnim & ~ANIM_TOGGLEBIT) != TORSO_STAND2)) {
1404 // if not standing still, always point all in the same direction
1405 cent->pe.torso.yawing = qtrue; // always center
1406 cent->pe.torso.pitching = qtrue; // always center
1407 cent->pe.legs.yawing = qtrue; // always center
1408 }
1409
1410 // adjust legs for movement dir
1411 if ( cent->currentState.eFlags & EF_DEAD ) {
1412 // don't let dead bodies twitch
1413 dir = 0;
1414 } else {
1415 dir = cent->currentState.angles2[YAW];
1416 if ( dir < 0 || dir > 7 ) {
1417 CG_Error( "Bad player movement angle" );
1418 }
1419 }
1420 legsAngles[YAW] = headAngles[YAW] + movementOffsets[ dir ];
1421 torsoAngles[YAW] = headAngles[YAW] + 0.25 * movementOffsets[ dir ];
1422
1423 // torso
1424 CG_SwingAngles( torsoAngles[YAW], 25, 90, cg_swingSpeed.value, &cent->pe.torso.yawAngle, &cent->pe.torso.yawing );
1425 CG_SwingAngles( legsAngles[YAW], 40, 90, cg_swingSpeed.value, &cent->pe.legs.yawAngle, &cent->pe.legs.yawing );
1426
1427 torsoAngles[YAW] = cent->pe.torso.yawAngle;
1428 legsAngles[YAW] = cent->pe.legs.yawAngle;
1429
1430
1431 // --------- pitch -------------
1432
1433 // only show a fraction of the pitch angle in the torso
1434 if ( headAngles[PITCH] > 180 ) {
1435 dest = (-360 + headAngles[PITCH]) * 0.75f;
1436 } else {
1437 dest = headAngles[PITCH] * 0.75f;
1438 }
1439 CG_SwingAngles( dest, 15, 30, 0.1f, &cent->pe.torso.pitchAngle, &cent->pe.torso.pitching );
1440 torsoAngles[PITCH] = cent->pe.torso.pitchAngle;
1441
1442 //
1443 clientNum = cent->currentState.clientNum;
1444 if ( clientNum >= 0 && clientNum < MAX_CLIENTS ) {
1445 ci = &cgs.clientinfo[ clientNum ];
1446 if ( ci->fixedtorso ) {
1447 torsoAngles[PITCH] = 0.0f;
1448 }
1449 }
1450
1451 // --------- roll -------------
1452
1453
1454 // lean towards the direction of travel
1455 VectorCopy( cent->currentState.pos.trDelta, velocity );
1456 speed = VectorNormalize( velocity );
1457 if ( speed ) {
1458 vec3_t axis[3];
1459 float side;
1460
1461 speed *= 0.05f;
1462
1463 AnglesToAxis( legsAngles, axis );
1464 side = speed * DotProduct( velocity, axis[1] );
1465 legsAngles[ROLL] -= side;
1466
1467 side = speed * DotProduct( velocity, axis[0] );
1468 legsAngles[PITCH] += side;
1469 }
1470
1471 //
1472 clientNum = cent->currentState.clientNum;
1473 if ( clientNum >= 0 && clientNum < MAX_CLIENTS ) {
1474 ci = &cgs.clientinfo[ clientNum ];
1475 if ( ci->fixedlegs ) {
1476 legsAngles[YAW] = torsoAngles[YAW];
1477 legsAngles[PITCH] = 0.0f;
1478 legsAngles[ROLL] = 0.0f;
1479 }
1480 }
1481
1482 // pain twitch
1483 CG_AddPainTwitch( cent, torsoAngles );
1484
1485 // pull the angles back out of the hierarchial chain
1486 AnglesSubtract( headAngles, torsoAngles, headAngles );
1487 AnglesSubtract( torsoAngles, legsAngles, torsoAngles );
1488 AnglesToAxis( legsAngles, legs );
1489 AnglesToAxis( torsoAngles, torso );
1490 AnglesToAxis( headAngles, head );
1491}

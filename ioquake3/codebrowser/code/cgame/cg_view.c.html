28/*
29=============================================================================
30
31 MODEL TESTING
32
33The viewthing and gun positioning tools from Q2 have been integrated and
34enhanced into a single model testing facility.
35
36Model viewing can begin with either "testmodel <modelname>" or "testgun <modelname>".
37
38The names must be the full pathname after the basedir, like
39"models/weapons/v_launch/tris.md3" or "players/male/tris.md3"
40
41Testmodel will create a fake entity 100 units in front of the current view
42position, directly facing the viewer. It will remain immobile, so you can
43move around it to view it from different angles.
44
45Testgun will cause the model to follow the player around and supress the real
46view weapon model. The default frame 0 of most guns is completely off screen,
47so you will probably have to cycle a couple frames to see it.
48
49"nextframe", "prevframe", "nextskin", and "prevskin" commands will change the
50frame or skin of the testmodel. These are bound to F5, F6, F7, and F8 in
51q3default.cfg.
52
53If a gun is being tested, the "gun_x", "gun_y", and "gun_z" variables will let
54you adjust the positioning.
55
56Note that none of the model testing features update while the game is paused, so
57it may be convenient to test with deathmatch set to 1 so that bringing down the
58console doesn't pause the game.
59
60=============================================================================
61*/
62
63/*
64=================
65CG_TestModel_f
66
67Creates an entity in front of the current position, which
68can then be moved around
69=================
70*/
71void CG_TestModel_f (void) {
72 vec3_t angles;
73
74 memset( &cg.testModelEntity, 0, sizeof(cg.testModelEntity) );
75 if ( trap_Argc() < 2 ) {
76 return;
77 }
78
79 Q_strncpyz (cg.testModelName, CG_Argv( 1 ), MAX_QPATH );
80 cg.testModelEntity.hModel = trap_R_RegisterModel( cg.testModelName );
81
82 if ( trap_Argc() == 3 ) {
83 cg.testModelEntity.backlerp = atof( CG_Argv( 2 ) );
84 cg.testModelEntity.frame = 1;
85 cg.testModelEntity.oldframe = 0;
86 }
87 if (! cg.testModelEntity.hModel ) {
88 CG_Printf( "Can't register model\n" );
89 return;
90 }
91
92 VectorMA( cg.refdef.vieworg, 100, cg.refdef.viewaxis[0], cg.testModelEntity.origin );
93
94 angles[PITCH] = 0;
95 angles[YAW] = 180 + cg.refdefViewAngles[1];
96 angles[ROLL] = 0;
97
98 AnglesToAxis( angles, cg.testModelEntity.axis );
99 cg.testGun = qfalse;
100}

142static void CG_AddTestModel (void) {
143 int i;
144
145 // re-register the model, because the level may have changed
146 cg.testModelEntity.hModel = trap_R_RegisterModel( cg.testModelName );
147 if (! cg.testModelEntity.hModel ) {
148 CG_Printf ("Can't register model\n");
149 return;
150 }
151
152 // if testing a gun, set the origin relative to the view origin
153 if ( cg.testGun ) {
154 VectorCopy( cg.refdef.vieworg, cg.testModelEntity.origin );
155 VectorCopy( cg.refdef.viewaxis[0], cg.testModelEntity.axis[0] );
156 VectorCopy( cg.refdef.viewaxis[1], cg.testModelEntity.axis[1] );
157 VectorCopy( cg.refdef.viewaxis[2], cg.testModelEntity.axis[2] );
158
159 // allow the position to be adjusted
160 for (i=0 ; i<3 ; i++) {
161 cg.testModelEntity.origin[i] += cg.refdef.viewaxis[0][i] * cg_gun_x.value;
162 cg.testModelEntity.origin[i] += cg.refdef.viewaxis[1][i] * cg_gun_y.value;
163 cg.testModelEntity.origin[i] += cg.refdef.viewaxis[2][i] * cg_gun_z.value;
164 }
165 }
166
167 trap_R_AddRefEntityToScene( &cg.testModelEntity );
168}

175/*
176=================
177CG_CalcVrect
178
179Sets the coordinates of the rendered window
180=================
181*/
182static void CG_CalcVrect (void) {
183 int size;
184
185 // the intermission should allways be full screen
186 if ( cg.snap->ps.pm_type == PM_INTERMISSION ) {
187 size = 100;
188 } else {
189 // bound normal viewsize
190 if (cg_viewsize.integer < 30) {
191 trap_Cvar_Set ("cg_viewsize","30");
192 size = 30;
193 } else if (cg_viewsize.integer > 100) {
194 trap_Cvar_Set ("cg_viewsize","100");
195 size = 100;
196 } else {
197 size = cg_viewsize.integer;
198 }
199
200 }
201 cg.refdef.width = cgs.glconfig.vidWidth*size/100;
202 cg.refdef.width &= ~1;
203
204 cg.refdef.height = cgs.glconfig.vidHeight*size/100;
205 cg.refdef.height &= ~1;
206
207 cg.refdef.x = (cgs.glconfig.vidWidth - cg.refdef.width)/2;
208 cg.refdef.y = (cgs.glconfig.vidHeight - cg.refdef.height)/2;
209}

214/*
215===============
216CG_OffsetThirdPersonView
217
218===============
219*/
220#define FOCUS_DISTANCE 512
221static void CG_OffsetThirdPersonView( void ) {
222 vec3_t forward, right, up;
223 vec3_t view;
224 vec3_t focusAngles;
225 trace_t trace;
226 static vec3_t mins = { -4, -4, -4 };
227 static vec3_t maxs = { 4, 4, 4 };
228 vec3_t focusPoint;
229 float focusDist;
230 float forwardScale, sideScale;
231
232 cg.refdef.vieworg[2] += cg.predictedPlayerState.viewheight;
233
234 VectorCopy( cg.refdefViewAngles, focusAngles );
235
236 // if dead, look at killer
237 if ( cg.predictedPlayerState.stats[STAT_HEALTH] <= 0 ) {
238 focusAngles[YAW] = cg.predictedPlayerState.stats[STAT_DEAD_YAW];
239 cg.refdefViewAngles[YAW] = cg.predictedPlayerState.stats[STAT_DEAD_YAW];
240 }
241
242 if ( focusAngles[PITCH] > 45 ) {
243 focusAngles[PITCH] = 45; // don't go too far overhead
244 }
245 AngleVectors( focusAngles, forward, NULL, NULL );
246
247 VectorMA( cg.refdef.vieworg, FOCUS_DISTANCE, forward, focusPoint );
248
249 VectorCopy( cg.refdef.vieworg, view );
250
251 view[2] += 8;
252
253 cg.refdefViewAngles[PITCH] *= 0.5;
254
255 AngleVectors( cg.refdefViewAngles, forward, right, up );
256
257 forwardScale = cos( cg_thirdPersonAngle.value / 180 * M_PI );
258 sideScale = sin( cg_thirdPersonAngle.value / 180 * M_PI );
259 VectorMA( view, -cg_thirdPersonRange.value * forwardScale, forward, view );
260 VectorMA( view, -cg_thirdPersonRange.value * sideScale, right, view );
261
262 // trace a ray from the origin to the viewpoint to make sure the view isn't
263 // in a solid block. Use an 8 by 8 block to prevent the view from near clipping anything
264
265 if (!cg_cameraMode.integer) {
266 CG_Trace( &trace, cg.refdef.vieworg, mins, maxs, view, cg.predictedPlayerState.clientNum, MASK_SOLID );
267
268 if ( trace.fraction != 1.0 ) {
269 VectorCopy( trace.endpos, view );
270 view[2] += (1.0 - trace.fraction) * 32;
271 // try another trace to this position, because a tunnel may have the ceiling
272 // close enough that this is poking out
273
274 CG_Trace( &trace, cg.refdef.vieworg, mins, maxs, view, cg.predictedPlayerState.clientNum, MASK_SOLID );
275 VectorCopy( trace.endpos, view );
276 }
277 }
278
279
280 VectorCopy( view, cg.refdef.vieworg );
281
282 // select pitch to look at focus point from vieword
283 VectorSubtract( focusPoint, cg.refdef.vieworg, focusPoint );
284 focusDist = sqrt( focusPoint[0] * focusPoint[0] + focusPoint[1] * focusPoint[1] );
285 if ( focusDist < 1 ) {
286 focusDist = 1; // should never happen
287 }
288 cg.refdefViewAngles[PITCH] = -180 / M_PI * atan2( focusPoint[2], focusDist );
289 cg.refdefViewAngles[YAW] -= cg_thirdPersonAngle.value;
290}

293// this causes a compiler bug on mac MrC compiler
294static void CG_StepOffset( void ) {
295 int timeDelta;
296
297 // smooth out stair climbing
298 timeDelta = cg.time - cg.stepTime;
299 if ( timeDelta < STEP_TIME ) {
300 cg.refdef.vieworg[2] -= cg.stepChange
301 * (STEP_TIME - timeDelta) / STEP_TIME;
302 }
303}

305/*
306===============
307CG_OffsetFirstPersonView
308
309===============
310*/
311static void CG_OffsetFirstPersonView( void ) {
312 float *origin;
313 float *angles;
314 float bob;
315 float ratio;
316 float delta;
317 float speed;
318 float f;
319 vec3_t predictedVelocity;
320 int timeDelta;
321
322 if ( cg.snap->ps.pm_type == PM_INTERMISSION ) {
323 return;
324 }
325
326 origin = cg.refdef.vieworg;
327 angles = cg.refdefViewAngles;
328
329 // if dead, fix the angle and don't add any kick
330 if ( cg.snap->ps.stats[STAT_HEALTH] <= 0 ) {
331 angles[ROLL] = 40;
332 angles[PITCH] = -15;
333 angles[YAW] = cg.snap->ps.stats[STAT_DEAD_YAW];
334 origin[2] += cg.predictedPlayerState.viewheight;
335 return;
336 }
337
338 // add angles based on damage kick
339 if ( cg.damageTime ) {
340 ratio = cg.time - cg.damageTime;
341 if ( ratio < DAMAGE_DEFLECT_TIME ) {
342 ratio /= DAMAGE_DEFLECT_TIME;
343 angles[PITCH] += ratio * cg.v_dmg_pitch;
344 angles[ROLL] += ratio * cg.v_dmg_roll;
345 } else {
346 ratio = 1.0 - ( ratio - DAMAGE_DEFLECT_TIME ) / DAMAGE_RETURN_TIME;
347 if ( ratio > 0 ) {
348 angles[PITCH] += ratio * cg.v_dmg_pitch;
349 angles[ROLL] += ratio * cg.v_dmg_roll;
350 }
351 }
352 }
353
354 // add pitch based on fall kick
355#if 0
356 ratio = ( cg.time - cg.landTime) / FALL_TIME;
357 if (ratio < 0)
358 ratio = 0;
359 angles[PITCH] += ratio * cg.fall_value;
360#endif
361
362 // add angles based on velocity
363 VectorCopy( cg.predictedPlayerState.velocity, predictedVelocity );
364
365 delta = DotProduct ( predictedVelocity, cg.refdef.viewaxis[0]);
366 angles[PITCH] += delta * cg_runpitch.value;
367
368 delta = DotProduct ( predictedVelocity, cg.refdef.viewaxis[1]);
369 angles[ROLL] -= delta * cg_runroll.value;
370
371 // add angles based on bob
372
373 // make sure the bob is visible even at low speeds
374 speed = cg.xyspeed > 200 ? cg.xyspeed : 200;
375
376 delta = cg.bobfracsin * cg_bobpitch.value * speed;
377 if (cg.predictedPlayerState.pm_flags & PMF_DUCKED)
378 delta *= 3; // crouching
379 angles[PITCH] += delta;
380 delta = cg.bobfracsin * cg_bobroll.value * speed;
381 if (cg.predictedPlayerState.pm_flags & PMF_DUCKED)
382 delta *= 3; // crouching accentuates roll
383 if (cg.bobcycle & 1)
384 delta = -delta;
385 angles[ROLL] += delta;
386
387//===================================
388
389 // add view height
390 origin[2] += cg.predictedPlayerState.viewheight;
391
392 // smooth out duck height changes
393 timeDelta = cg.time - cg.duckTime;
394 if ( timeDelta < DUCK_TIME) {
395 cg.refdef.vieworg[2] -= cg.duckChange
396 * (DUCK_TIME - timeDelta) / DUCK_TIME;
397 }
398
399 // add bob height
400 bob = cg.bobfracsin * cg.xyspeed * cg_bobup.value;
401 if (bob > 6) {
402 bob = 6;
403 }
404
405 origin[2] += bob;
406
407
408 // add fall height
409 delta = cg.time - cg.landTime;
410 if ( delta < LAND_DEFLECT_TIME ) {
411 f = delta / LAND_DEFLECT_TIME;
412 cg.refdef.vieworg[2] += cg.landChange * f;
413 } else if ( delta < LAND_DEFLECT_TIME + LAND_RETURN_TIME ) {
414 delta -= LAND_DEFLECT_TIME;
415 f = 1.0 - ( delta / LAND_RETURN_TIME );
416 cg.refdef.vieworg[2] += cg.landChange * f;
417 }
418
419 // add step offset
420 CG_StepOffset();
421
422 // pivot the eye based on a neck length
423#if 0
424 {
425#define NECK_LENGTH 8
426 vec3_t forward, up;
427
428 cg.refdef.vieworg[2] -= NECK_LENGTH;
429 AngleVectors( cg.refdefViewAngles, forward, NULL, up );
430 VectorMA( cg.refdef.vieworg, 3, forward, cg.refdef.vieworg );
431 VectorMA( cg.refdef.vieworg, NECK_LENGTH, up, cg.refdef.vieworg );
432 }
433#endif
434}

455/*
456====================
457CG_CalcFov
458
459Fixed fov at intermissions, otherwise account for fov variable and zooms.
460====================
461*/
462#define WAVE_AMPLITUDE 1
463#define WAVE_FREQUENCY 0.4
464
465static int CG_CalcFov( void ) {
466 float x;
467 float phase;
468 float v;
469 int contents;
470 float fov_x, fov_y;
471 float zoomFov;
472 float f;
473 int inwater;
474
475 if ( cg.predictedPlayerState.pm_type == PM_INTERMISSION ) {
476 // if in intermission, use a fixed value
477 fov_x = 90;
478 } else {
479 // user selectable
480 if ( cgs.dmflags & DF_FIXED_FOV ) {
481 // dmflag to prevent wide fov for all clients
482 fov_x = 90;
483 } else {
484 fov_x = cg_fov.value;
485 if ( fov_x < 1 ) {
486 fov_x = 1;
487 } else if ( fov_x > 160 ) {
488 fov_x = 160;
489 }
490 }
491
492 // account for zooms
493 zoomFov = cg_zoomFov.value;
494 if ( zoomFov < 1 ) {
495 zoomFov = 1;
496 } else if ( zoomFov > 160 ) {
497 zoomFov = 160;
498 }
499
500 if ( cg.zoomed ) {
501 f = ( cg.time - cg.zoomTime ) / (float)ZOOM_TIME;
502 if ( f > 1.0 ) {
503 fov_x = zoomFov;
504 } else {
505 fov_x = fov_x + f * ( zoomFov - fov_x );
506 }
507 } else {
508 f = ( cg.time - cg.zoomTime ) / (float)ZOOM_TIME;
509 if ( f <= 1.0 ) {
510 fov_x = zoomFov + f * ( fov_x - zoomFov );
511 }
512 }
513 }
514
515 x = cg.refdef.width / tan( fov_x / 360 * M_PI );
516 fov_y = atan2( cg.refdef.height, x );
517 fov_y = fov_y * 360 / M_PI;
518
519 // warp if underwater
520 contents = CG_PointContents( cg.refdef.vieworg, -1 );
521 if ( contents & ( CONTENTS_WATER | CONTENTS_SLIME | CONTENTS_LAVA ) ){
522 phase = cg.time / 1000.0 * WAVE_FREQUENCY * M_PI * 2;
523 v = WAVE_AMPLITUDE * sin( phase );
524 fov_x += v;
525 fov_y -= v;
526 inwater = qtrue;
527 }
528 else {
529 inwater = qfalse;
530 }
531
532
533 // set it
534 cg.refdef.fov_x = fov_x;
535 cg.refdef.fov_y = fov_y;
536
537 if ( !cg.zoomed ) {
538 cg.zoomSensitivity = 1;
539 } else {
540 cg.zoomSensitivity = cg.refdef.fov_y / 75.0;
541 }
542
543 return inwater;
544}

548/*
549===============
550CG_DamageBlendBlob
551
552===============
553*/
554static void CG_DamageBlendBlob( void ) {
555 int t;
556 int maxTime;
557 refEntity_t ent;
558
559 if (!cg_blood.integer) {
560 return;
561 }
562
563 if ( !cg.damageValue ) {
564 return;
565 }
566
567 //if (cg.cameraMode) {
568 // return;
569 //}
570
571 // ragePro systems can't fade blends, so don't obscure the screen
572 if ( cgs.glconfig.hardwareType == GLHW_RAGEPRO ) {
573 return;
574 }
575
576 maxTime = DAMAGE_TIME;
577 t = cg.time - cg.damageTime;
578 if ( t <= 0 || t >= maxTime ) {
579 return;
580 }
581
582
583 memset( &ent, 0, sizeof( ent ) );
584 ent.reType = RT_SPRITE;
585 ent.renderfx = RF_FIRST_PERSON;
586
587 VectorMA( cg.refdef.vieworg, 8, cg.refdef.viewaxis[0], ent.origin );
588 VectorMA( ent.origin, cg.damageX * -8, cg.refdef.viewaxis[1], ent.origin );
589 VectorMA( ent.origin, cg.damageY * 8, cg.refdef.viewaxis[2], ent.origin );
590
591 ent.radius = cg.damageValue * 3;
592 ent.customShader = cgs.media.viewBloodShader;
593 ent.shaderRGBA[0] = 255;
594 ent.shaderRGBA[1] = 255;
595 ent.shaderRGBA[2] = 255;
596 ent.shaderRGBA[3] = 200 * ( 1.0 - ((float)t / maxTime) );
597 trap_R_AddRefEntityToScene( &ent );
598}

601/*
602===============
603CG_CalcViewValues
604
605Sets cg.refdef view values
606===============
607*/
608static int CG_CalcViewValues( void ) {
609 playerState_t *ps;
610
611 memset( &cg.refdef, 0, sizeof( cg.refdef ) );
612
613 // strings for in game rendering
614 // Q_strncpyz( cg.refdef.text[0], "Park Ranger", sizeof(cg.refdef.text[0]) );
615 // Q_strncpyz( cg.refdef.text[1], "19", sizeof(cg.refdef.text[1]) );
616
617 // calculate size of 3D view
618 CG_CalcVrect();
619
620 ps = &cg.predictedPlayerState;
621/*
622 if (cg.cameraMode) {
623 vec3_t origin, angles;
624 if (trap_getCameraInfo(cg.time, &origin, &angles)) {
625 VectorCopy(origin, cg.refdef.vieworg);
626 angles[ROLL] = 0;
627 VectorCopy(angles, cg.refdefViewAngles);
628 AnglesToAxis( cg.refdefViewAngles, cg.refdef.viewaxis );
629 return CG_CalcFov();
630 } else {
631 cg.cameraMode = qfalse;
632 }
633 }
634*/
635 // intermission view
636 if ( ps->pm_type == PM_INTERMISSION ) {
637 VectorCopy( ps->origin, cg.refdef.vieworg );
638 VectorCopy( ps->viewangles, cg.refdefViewAngles );
639 AnglesToAxis( cg.refdefViewAngles, cg.refdef.viewaxis );
640 return CG_CalcFov();
641 }
642
643 cg.bobcycle = ( ps->bobCycle & 128 ) >> 7;
644 cg.bobfracsin = fabs( sin( ( ps->bobCycle & 127 ) / 127.0 * M_PI ) );
645 cg.xyspeed = sqrt( ps->velocity[0] * ps->velocity[0] +
646 ps->velocity[1] * ps->velocity[1] );
647
648
649 VectorCopy( ps->origin, cg.refdef.vieworg );
650 VectorCopy( ps->viewangles, cg.refdefViewAngles );
651
652 if (cg_cameraOrbit.integer) {
653 if (cg.time > cg.nextOrbitTime) {
654 cg.nextOrbitTime = cg.time + cg_cameraOrbitDelay.integer;
655 cg_thirdPersonAngle.value += cg_cameraOrbit.value;
656 }
657 }
658 // add error decay
659 if ( cg_errorDecay.value > 0 ) {
660 int t;
661 float f;
662
663 t = cg.time - cg.predictedErrorTime;
664 f = ( cg_errorDecay.value - t ) / cg_errorDecay.value;
665 if ( f > 0 && f < 1 ) {
666 VectorMA( cg.refdef.vieworg, f, cg.predictedError, cg.refdef.vieworg );
667 } else {
668 cg.predictedErrorTime = 0;
669 }
670 }
671
672 if ( cg.renderingThirdPerson ) {
673 // back away from character
674 CG_OffsetThirdPersonView();
675 } else {
676 // offset for local bobbing and kicks
677 CG_OffsetFirstPersonView();
678 }
679
680 // position eye relative to origin
681 AnglesToAxis( cg.refdefViewAngles, cg.refdef.viewaxis );
682
683 if ( cg.hyperspace ) {
684 cg.refdef.rdflags |= RDF_NOWORLDMODEL | RDF_HYPERSPACE;
685 }
686
687 // field of view
688 return CG_CalcFov();
689}

692/*
693=====================
694CG_PowerupTimerSounds
695=====================
696*/
697static void CG_PowerupTimerSounds( void ) {
698 int i;
699 int t;
700
701 // powerup timers going away
702 for ( i = 0 ; i < MAX_POWERUPS ; i++ ) {
703 t = cg.snap->ps.powerups[i];
704 if ( t <= cg.time ) {
705 continue;
706 }
707 if ( t - cg.time >= POWERUP_BLINKS * POWERUP_BLINK_TIME ) {
708 continue;
709 }
710 if ( ( t - cg.time ) / POWERUP_BLINK_TIME != ( t - cg.oldTime ) / POWERUP_BLINK_TIME ) {
711 trap_S_StartSound( NULL, cg.snap->ps.clientNum, CHAN_ITEM, cgs.media.wearOffSound );
712 }
713 }
714}

732=====================
733CG_PlayBufferedSounds
734=====================
735*/
736static void CG_PlayBufferedSounds( void ) {
737 if ( cg.soundTime < cg.time ) {
738 if (cg.soundBufferOut != cg.soundBufferIn && cg.soundBuffer[cg.soundBufferOut]) {
739 trap_S_StartLocalSound(cg.soundBuffer[cg.soundBufferOut], CHAN_ANNOUNCER);
740 cg.soundBuffer[cg.soundBufferOut] = 0;
741 cg.soundBufferOut = (cg.soundBufferOut + 1) % MAX_SOUNDBUFFER;
742 cg.soundTime = cg.time + 750;
743 }
744 }

749/*
750=================
751CG_DrawActiveFrame
752
753Generates and draws a game scene and status information at the given time.
754=================
755*/
756void CG_DrawActiveFrame( int serverTime, stereoFrame_t stereoView, qboolean demoPlayback ) {
757 int inwater;
758
759 cg.time = serverTime;
760 cg.demoPlayback = demoPlayback;
761
762 // update cvars
763 CG_UpdateCvars();
764
765 // if we are only updating the screen as a loading
766 // pacifier, don't even try to read snapshots
767 if ( cg.infoScreenText[0] != 0 ) {
768 CG_DrawInformation();
769 return;
770 }
771
772 // any looped sounds will be respecified as entities
773 // are added to the render list
774 trap_S_ClearLoopingSounds(qfalse);
775
776 // clear all the render lists
777 trap_R_ClearScene();
778
779 // set up cg.snap and possibly cg.nextSnap
780 CG_ProcessSnapshots();
781
782 // if we haven't received any snapshots yet, all
783 // we can draw is the information screen
784 if ( !cg.snap || ( cg.snap->snapFlags & SNAPFLAG_NOT_ACTIVE ) ) {
785 CG_DrawInformation();
786 return;
787 }
788
789 // let the client system know what our weapon and zoom settings are
790 trap_SetUserCmdValue( cg.weaponSelect, cg.zoomSensitivity );
791
792 // this counter will be bumped for every valid scene we generate
793 cg.clientFrame++;
794
795 // update cg.predictedPlayerState
796 CG_PredictPlayerState();
797
798 // decide on third person view
799 cg.renderingThirdPerson = cg.snap->ps.persistant[PERS_TEAM] != TEAM_SPECTATOR
800 && (cg_thirdPerson.integer || (cg.snap->ps.stats[STAT_HEALTH] <= 0));
801
802 // build cg.refdef
803 inwater = CG_CalcViewValues();
804
805 // first person blend blobs, done after AnglesToAxis
806 if ( !cg.renderingThirdPerson ) {
807 CG_DamageBlendBlob();
808 }
809
810 // build the render lists
811 if ( !cg.hyperspace ) {
812 CG_AddPacketEntities(); // adter calcViewValues, so predicted player state is correct
813 CG_AddMarks();
814 CG_AddParticles ();
815 CG_AddLocalEntities();
816 }
817 CG_AddViewWeapon( &cg.predictedPlayerState );
818
819 // add buffered sounds
820 CG_PlayBufferedSounds();
821
822#ifdef MISSIONPACK
823 // play buffered voice chats
824 CG_PlayBufferedVoiceChats();
825#endif
826
827 // finish up the rest of the refdef
828 if ( cg.testModelEntity.hModel ) {
829 CG_AddTestModel();
830 }
831 cg.refdef.time = cg.time;
832 memcpy( cg.refdef.areamask, cg.snap->areamask, sizeof( cg.refdef.areamask ) );
833
834 // warning sounds when powerup is wearing off
835 CG_PowerupTimerSounds();
836
837 // update audio positions
838 trap_S_Respatialize( cg.snap->ps.clientNum, cg.refdef.vieworg, cg.refdef.viewaxis, inwater );
839
840 // make sure the lagometerSample and frame timing isn't done twice when in stereo
841 if ( stereoView != STEREO_RIGHT ) {
842 cg.frametime = cg.time - cg.oldTime;
843 if ( cg.frametime < 0 ) {
844 cg.frametime = 0;
845 }
846 cg.oldTime = cg.time;
847 CG_AddLagometerFrameInfo();
848 }
849 if (cg_timescale.value != cg_timescaleFadeEnd.value) {
850 if (cg_timescale.value < cg_timescaleFadeEnd.value) {
851 cg_timescale.value += cg_timescaleFadeSpeed.value * ((float)cg.frametime) / 1000;
852 if (cg_timescale.value > cg_timescaleFadeEnd.value)
853 cg_timescale.value = cg_timescaleFadeEnd.value;
854 }
855 else {
856 cg_timescale.value -= cg_timescaleFadeSpeed.value * ((float)cg.frametime) / 1000;
857 if (cg_timescale.value < cg_timescaleFadeEnd.value)
858 cg_timescale.value = cg_timescaleFadeEnd.value;
859 }
860 if (cg_timescaleFadeSpeed.value) {
861 trap_Cvar_Set("timescale", va("%f", cg_timescale.value));
862 }
863 }
864
865 // actually issue the rendering calls
866 CG_DrawActive( stereoView );
867
868 if ( cg_stats.integer ) {
869 CG_Printf( "cg.clientFrame:%i\n", cg.clientFrame );
870 }
871
872
873}

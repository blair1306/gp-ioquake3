36static const orderTask_t validOrders[] = {
37 { VOICECHAT_GETFLAG, TEAMTASK_OFFENSE },
38 { VOICECHAT_OFFENSE, TEAMTASK_OFFENSE },
39 { VOICECHAT_DEFEND, TEAMTASK_DEFENSE },
40 { VOICECHAT_DEFENDFLAG, TEAMTASK_DEFENSE },

45 { VOICECHAT_FOLLOWFLAGCARRIER, TEAMTASK_ESCORT }
46};

48static const int numValidOrders = ARRAY_LEN(validOrders);
49
50static int CG_ValidOrder(const char *p) {
51 int i;
52 for (i = 0; i < numValidOrders; i++) {
53 if (Q_stricmp(p, validOrders[i].order) == 0) {
54 return validOrders[i].taskNum;
55 }
56 }
57 return -1;
58}

61/*
62=================
63CG_ParseScores
64
65=================
66*/
67static void CG_ParseScores( void ) {
68 int i, powerups;
69
70 cg.numScores = atoi( CG_Argv( 1 ) );
71 if ( cg.numScores > MAX_CLIENTS ) {
72 cg.numScores = MAX_CLIENTS;
73 }
74
75 cg.teamScores[0] = atoi( CG_Argv( 2 ) );
76 cg.teamScores[1] = atoi( CG_Argv( 3 ) );
77
78 memset( cg.scores, 0, sizeof( cg.scores ) );
79 for ( i = 0 ; i < cg.numScores ; i++ ) {
80 //
81 cg.scores[i].client = atoi( CG_Argv( i * 14 + 4 ) );
82 cg.scores[i].score = atoi( CG_Argv( i * 14 + 5 ) );
83 cg.scores[i].ping = atoi( CG_Argv( i * 14 + 6 ) );
84 cg.scores[i].time = atoi( CG_Argv( i * 14 + 7 ) );
85 cg.scores[i].scoreFlags = atoi( CG_Argv( i * 14 + 8 ) );
86 powerups = atoi( CG_Argv( i * 14 + 9 ) );
87 cg.scores[i].accuracy = atoi(CG_Argv(i * 14 + 10));
88 cg.scores[i].impressiveCount = atoi(CG_Argv(i * 14 + 11));
89 cg.scores[i].excellentCount = atoi(CG_Argv(i * 14 + 12));
90 cg.scores[i].guantletCount = atoi(CG_Argv(i * 14 + 13));
91 cg.scores[i].defendCount = atoi(CG_Argv(i * 14 + 14));
92 cg.scores[i].assistCount = atoi(CG_Argv(i * 14 + 15));
93 cg.scores[i].perfect = atoi(CG_Argv(i * 14 + 16));
94 cg.scores[i].captures = atoi(CG_Argv(i * 14 + 17));
95
96 if ( cg.scores[i].client < 0 || cg.scores[i].client >= MAX_CLIENTS ) {
97 cg.scores[i].client = 0;
98 }
99 cgs.clientinfo[ cg.scores[i].client ].score = cg.scores[i].score;
100 cgs.clientinfo[ cg.scores[i].client ].powerups = powerups;
101
102 cg.scores[i].team = cgs.clientinfo[cg.scores[i].client].team;
103 }
104#ifdef MISSIONPACK
105 CG_SetScoreSelection(NULL);
106#endif
107
108}

110/*
111=================
112CG_ParseTeamInfo
113
114=================
115*/
116static void CG_ParseTeamInfo( void ) {
117 int i;
118 int client;
119
120 numSortedTeamPlayers = atoi( CG_Argv( 1 ) );
121 if( numSortedTeamPlayers < 0 || numSortedTeamPlayers > TEAM_MAXOVERLAY )
122 {
123 CG_Error( "CG_ParseTeamInfo: numSortedTeamPlayers out of range (%d)",
124 numSortedTeamPlayers );
125 return;
126 }
127
128 for ( i = 0 ; i < numSortedTeamPlayers ; i++ ) {
129 client = atoi( CG_Argv( i * 6 + 2 ) );
130 if( client < 0 || client >= MAX_CLIENTS )
131 {
132 CG_Error( "CG_ParseTeamInfo: bad client number: %d", client );
133 return;
134 }
135
136 sortedTeamPlayers[i] = client;
137
138 cgs.clientinfo[ client ].location = atoi( CG_Argv( i * 6 + 3 ) );
139 cgs.clientinfo[ client ].health = atoi( CG_Argv( i * 6 + 4 ) );
140 cgs.clientinfo[ client ].armor = atoi( CG_Argv( i * 6 + 5 ) );
141 cgs.clientinfo[ client ].curWeapon = atoi( CG_Argv( i * 6 + 6 ) );
142 cgs.clientinfo[ client ].powerups = atoi( CG_Argv( i * 6 + 7 ) );
143 }
144}

233/*
234=====================
235CG_ShaderStateChanged
236=====================
237*/
238void CG_ShaderStateChanged(void) {
239 char originalShader[MAX_QPATH];
240 char newShader[MAX_QPATH];
241 char timeOffset[16];
242 const char *o;
243 char *n,*t;
244
245 o = CG_ConfigString( CS_SHADERSTATE );
246 while (o && *o) {
247 n = strstr(o, "=");
248 if (n && *n) {
249 strncpy(originalShader, o, n-o);
250 originalShader[n-o] = 0;
251 n++;
252 t = strstr(n, ":");
253 if (t && *t) {
254 strncpy(newShader, n, t-n);
255 newShader[t-n] = 0;
256 } else {
257 break;
258 }
259 t++;
260 o = strstr(t, "@");
261 if (o) {
262 strncpy(timeOffset, t, o-t);
263 timeOffset[o-t] = 0;
264 o++;
265 trap_R_RemapShader( originalShader, newShader, timeOffset );
266 }
267 } else {
268 break;
269 }
270 }
271}

273/*
274================
275CG_ConfigStringModified
276
277================
278*/
279static void CG_ConfigStringModified( void ) {
280 const char *str;
281 int num;
282
283 num = atoi( CG_Argv( 1 ) );
284
285 // get the gamestate from the client system, which will have the
286 // new configstring already integrated
287 trap_GetGameState( &cgs.gameState );
288
289 // look up the individual string that was modified
290 str = CG_ConfigString( num );
291
292 // do something with it if necessary
293 if ( num == CS_MUSIC ) {
294 CG_StartMusic();
295 } else if ( num == CS_SERVERINFO ) {
296 CG_ParseServerinfo();
297 } else if ( num == CS_WARMUP ) {
298 CG_ParseWarmup();
299 } else if ( num == CS_SCORES1 ) {
300 cgs.scores1 = atoi( str );
301 } else if ( num == CS_SCORES2 ) {
302 cgs.scores2 = atoi( str );
303 } else if ( num == CS_LEVEL_START_TIME ) {
304 cgs.levelStartTime = atoi( str );
305 } else if ( num == CS_VOTE_TIME ) {
306 cgs.voteTime = atoi( str );
307 cgs.voteModified = qtrue;
308 } else if ( num == CS_VOTE_YES ) {
309 cgs.voteYes = atoi( str );
310 cgs.voteModified = qtrue;
311 } else if ( num == CS_VOTE_NO ) {
312 cgs.voteNo = atoi( str );
313 cgs.voteModified = qtrue;
314 } else if ( num == CS_VOTE_STRING ) {
315 Q_strncpyz( cgs.voteString, str, sizeof( cgs.voteString ) );
316#ifdef MISSIONPACK
317 trap_S_StartLocalSound( cgs.media.voteNow, CHAN_ANNOUNCER );
318#endif //MISSIONPACK
319 } else if ( num >= CS_TEAMVOTE_TIME && num <= CS_TEAMVOTE_TIME + 1) {
320 cgs.teamVoteTime[num-CS_TEAMVOTE_TIME] = atoi( str );
321 cgs.teamVoteModified[num-CS_TEAMVOTE_TIME] = qtrue;
322 } else if ( num >= CS_TEAMVOTE_YES && num <= CS_TEAMVOTE_YES + 1) {
323 cgs.teamVoteYes[num-CS_TEAMVOTE_YES] = atoi( str );
324 cgs.teamVoteModified[num-CS_TEAMVOTE_YES] = qtrue;
325 } else if ( num >= CS_TEAMVOTE_NO && num <= CS_TEAMVOTE_NO + 1) {
326 cgs.teamVoteNo[num-CS_TEAMVOTE_NO] = atoi( str );
327 cgs.teamVoteModified[num-CS_TEAMVOTE_NO] = qtrue;
328 } else if ( num >= CS_TEAMVOTE_STRING && num <= CS_TEAMVOTE_STRING + 1) {
329 Q_strncpyz( cgs.teamVoteString[num-CS_TEAMVOTE_STRING], str, sizeof( cgs.teamVoteString[0] ) );
330#ifdef MISSIONPACK
331 trap_S_StartLocalSound( cgs.media.voteNow, CHAN_ANNOUNCER );
332#endif
333 } else if ( num == CS_INTERMISSION ) {
334 cg.intermissionStarted = atoi( str );
335 } else if ( num >= CS_MODELS && num < CS_MODELS+MAX_MODELS ) {
336 cgs.gameModels[ num-CS_MODELS ] = trap_R_RegisterModel( str );
337 } else if ( num >= CS_SOUNDS && num < CS_SOUNDS+MAX_SOUNDS ) {
338 if ( str[0] != '*' ) { // player specific sounds don't register here
339 cgs.gameSounds[ num-CS_SOUNDS] = trap_S_RegisterSound( str, qfalse );
340 }
341 } else if ( num >= CS_PLAYERS && num < CS_PLAYERS+MAX_CLIENTS ) {
342 CG_NewClientInfo( num - CS_PLAYERS );
343 CG_BuildSpectatorString();
344 } else if ( num == CS_FLAGSTATUS ) {
345 if( cgs.gametype == GT_CTF ) {
346 // format is rb where its red/blue, 0 is at base, 1 is taken, 2 is dropped
347 cgs.redflag = str[0] - '0';
348 cgs.blueflag = str[1] - '0';
349 }
350#ifdef MISSIONPACK
351 else if( cgs.gametype == GT_1FCTF ) {
352 cgs.flagStatus = str[0] - '0';
353 }
354#endif
355 }
356 else if ( num == CS_SHADERSTATE ) {
357 CG_ShaderStateChanged();
358 }
359
360}

364=======================
365CG_AddToTeamChat
366
367=======================
368*/
369static void CG_AddToTeamChat( const char *str ) {
370 int len;
371 char *p, *ls;
372 int lastcolor;
373 int chatHeight;
374
375 if (cg_teamChatHeight.integer < TEAMCHAT_HEIGHT) {
376 chatHeight = cg_teamChatHeight.integer;
377 } else {
378 chatHeight = TEAMCHAT_HEIGHT;
379 }
380
381 if (chatHeight <= 0 || cg_teamChatTime.integer <= 0) {
382 // team chat disabled, dump into normal chat
383 cgs.teamChatPos = cgs.teamLastChatPos = 0;
384 return;
385 }
386
387 len = 0;
388
389 p = cgs.teamChatMsgs[cgs.teamChatPos % chatHeight];
390 *p = 0;
391
392 lastcolor = '7';
393
394 ls = NULL;
395 while (*str) {
396 if (len > TEAMCHAT_WIDTH - 1) {
397 if (ls) {
398 str -= (p - ls);
399 str++;
400 p -= (p - ls);
401 }
402 *p = 0;
403
404 cgs.teamChatMsgTimes[cgs.teamChatPos % chatHeight] = cg.time;
405
406 cgs.teamChatPos++;
407 p = cgs.teamChatMsgs[cgs.teamChatPos % chatHeight];
408 *p = 0;
409 *p++ = Q_COLOR_ESCAPE;
410 *p++ = lastcolor;
411 len = 0;
412 ls = NULL;
413 }
414
415 if ( Q_IsColorString( str ) ) {
416 *p++ = *str++;
417 lastcolor = *str;
418 *p++ = *str++;
419 continue;
420 }
421 if (*str == ' ') {
422 ls = p;
423 }
424 *p++ = *str++;
425 len++;
426 }
427 *p = 0;
428
429 cgs.teamChatMsgTimes[cgs.teamChatPos % chatHeight] = cg.time;
430 cgs.teamChatPos++;
431
432 if (cgs.teamChatPos - cgs.teamLastChatPos > chatHeight)
433 cgs.teamLastChatPos = cgs.teamChatPos - chatHeight;
434}

441is completely new and should not be interpolated to.

443A tournement restart will clear everything, but doesn't

449 CG_Printf( "CG_MapRestart\n" );

451
452 CG_InitLocalEntities();
453 CG_InitMarkPolys();
454 CG_ClearParticles ();

461 cg.rewardStack = 0;
462 cg.intermissionStarted = qfalse;

464
465 cgs.voteTime = 0;
466
467 cg.mapRestart = qtrue;
468
469 CG_StartMusic();
470
471 trap_S_ClearLoopingSounds(qtrue);
472
473 // we really should clear more parts of cg here and stop sounds
474
475 // play the "fight" sound if this is a restart without warmup

486 }
487#endif
488 trap_Cvar_Set("cg_thirdPerson", "0");
489}

525/*
526=================
527CG_ParseVoiceChats
528=================
529*/
530int CG_ParseVoiceChats( const char *filename, voiceChatList_t *voiceChatList, int maxVoiceChats ) {
531 int len, i;
532 fileHandle_t f;
533 char buf[MAX_VOICEFILESIZE];
534 char **p, *ptr;
535 char *token;
536 voiceChat_t *voiceChats;
537 qboolean compress;
538 sfxHandle_t sound;
539
540 compress = qtrue;
541 if (cg_buildScript.integer) {
542 compress = qfalse;
543 }
544
545 len = trap_FS_FOpenFile( filename, &f, FS_READ );
546 if ( !f ) {
547 trap_Print( va( S_COLOR_RED "voice chat file not found: %s\n", filename ) );
548 return qfalse;
549 }
550 if ( len >= MAX_VOICEFILESIZE ) {
551 trap_Print( va( S_COLOR_RED "voice chat file too large: %s is %i, max allowed is %i\n", filename, len, MAX_VOICEFILESIZE ) );
552 trap_FS_FCloseFile( f );
553 return qfalse;
554 }
555
556 trap_FS_Read( buf, len, f );
557 buf[len] = 0;
558 trap_FS_FCloseFile( f );
559
560 ptr = buf;
561 p = &ptr;
562
563 Com_sprintf(voiceChatList->name, sizeof(voiceChatList->name), "%s", filename);
564 voiceChats = voiceChatList->voiceChats;
565 for ( i = 0; i < maxVoiceChats; i++ ) {
566 voiceChats[i].id[0] = 0;
567 }
568 token = COM_ParseExt(p, qtrue);
569 if (!token[0]) {
570 return qtrue;
571 }
572 if (!Q_stricmp(token, "female")) {
573 voiceChatList->gender = GENDER_FEMALE;
574 }
575 else if (!Q_stricmp(token, "male")) {
576 voiceChatList->gender = GENDER_MALE;
577 }
578 else if (!Q_stricmp(token, "neuter")) {
579 voiceChatList->gender = GENDER_NEUTER;
580 }
581 else {
582 trap_Print( va( S_COLOR_RED "expected gender not found in voice chat file: %s\n", filename ) );
583 return qfalse;
584 }
585
586 voiceChatList->numVoiceChats = 0;
587 while ( 1 ) {
588 token = COM_ParseExt(p, qtrue);
589 if (!token[0]) {
590 return qtrue;
591 }
592 Com_sprintf(voiceChats[voiceChatList->numVoiceChats].id, sizeof( voiceChats[voiceChatList->numVoiceChats].id ), "%s", token);
593 token = COM_ParseExt(p, qtrue);
594 if (Q_stricmp(token, "{")) {
595 trap_Print( va( S_COLOR_RED "expected { found %s in voice chat file: %s\n", token, filename ) );
596 return qfalse;
597 }
598 voiceChats[voiceChatList->numVoiceChats].numSounds = 0;
599 while(1) {
600 token = COM_ParseExt(p, qtrue);
601 if (!token[0]) {
602 return qtrue;
603 }
604 if (!Q_stricmp(token, "}"))
605 break;
606 sound = trap_S_RegisterSound( token, compress );
607 voiceChats[voiceChatList->numVoiceChats].sounds[voiceChats[voiceChatList->numVoiceChats].numSounds] = sound;
608 token = COM_ParseExt(p, qtrue);
609 if (!token[0]) {
610 return qtrue;
611 }
612 Com_sprintf(voiceChats[voiceChatList->numVoiceChats].chats[
613 voiceChats[voiceChatList->numVoiceChats].numSounds], MAX_CHATSIZE, "%s", token);
614 if (sound)
615 voiceChats[voiceChatList->numVoiceChats].numSounds++;
616 if (voiceChats[voiceChatList->numVoiceChats].numSounds >= MAX_VOICESOUNDS)
617 break;
618 }
619 voiceChatList->numVoiceChats++;
620 if (voiceChatList->numVoiceChats >= maxVoiceChats)
621 return qtrue;
622 }
623 return qtrue;
624}

646/*
647=================
648CG_HeadModelVoiceChats
649=================
650*/
651int CG_HeadModelVoiceChats( char *filename ) {
652 int len, i;
653 fileHandle_t f;
654 char buf[MAX_VOICEFILESIZE];
655 char **p, *ptr;
656 char *token;
657
658 len = trap_FS_FOpenFile( filename, &f, FS_READ );
659 if ( !f ) {
660 //trap_Print( va( "voice chat file not found: %s\n", filename ) );
661 return -1;
662 }
663 if ( len >= MAX_VOICEFILESIZE ) {
664 trap_Print( va( S_COLOR_RED "voice chat file too large: %s is %i, max allowed is %i\n", filename, len, MAX_VOICEFILESIZE ) );
665 trap_FS_FCloseFile( f );
666 return -1;
667 }
668
669 trap_FS_Read( buf, len, f );
670 buf[len] = 0;
671 trap_FS_FCloseFile( f );
672
673 ptr = buf;

676 token = COM_ParseExt(p, qtrue);
677 if ( !token[0] ) {

679 }
680
681 for ( i = 0; i < MAX_VOICEFILES; i++ ) {

683 return i;
684 }
685 }

687 //FIXME: maybe try to load the .voice file which name is stored in token?

689 return -1;

698int CG_GetVoiceChat( voiceChatList_t *voiceChatList, const char *id, sfxHandle_t *snd, char **chat) {
699 int i, rnd;
700
701 for ( i = 0; i < voiceChatList->numVoiceChats; i++ ) {
702 if ( !Q_stricmp( id, voiceChatList->voiceChats[i].id ) ) {
703 rnd = random() * voiceChatList->voiceChats[i].numSounds;
704 *snd = voiceChatList->voiceChats[i].sounds[rnd];
705 *chat = voiceChatList->voiceChats[i].chats[rnd];
706 return qtrue;
707 }
708 }
709 return qfalse;

712/*
713=================
714CG_VoiceChatListForClient
715=================
716*/
717voiceChatList_t *CG_VoiceChatListForClient( int clientNum ) {
718 clientInfo_t *ci;
719 int voiceChatNum, i, j, k, gender;
720 char filename[MAX_QPATH], headModelName[MAX_QPATH];
721
722 if ( clientNum < 0 || clientNum >= MAX_CLIENTS ) {
723 clientNum = 0;
724 }
725 ci = &cgs.clientinfo[ clientNum ];
726
727 for ( k = 0; k < 2; k++ ) {
728 if ( k == 0 ) {
729 if (ci->headModelName[0] == '*') {
730 Com_sprintf( headModelName, sizeof(headModelName), "%s/%s", ci->headModelName+1, ci->headSkinName );
731 }
732 else {
733 Com_sprintf( headModelName, sizeof(headModelName), "%s/%s", ci->headModelName, ci->headSkinName );
734 }
735 }
736 else {
737 if (ci->headModelName[0] == '*') {
738 Com_sprintf( headModelName, sizeof(headModelName), "%s", ci->headModelName+1 );
739 }
740 else {
741 Com_sprintf( headModelName, sizeof(headModelName), "%s", ci->headModelName );
742 }
743 }
744 // find the voice file for the head model the client uses
745 for ( i = 0; i < MAX_HEADMODELS; i++ ) {
746 if (!Q_stricmp(headModelVoiceChat[i].headmodel, headModelName)) {
747 break;
748 }
749 }
750 if (i < MAX_HEADMODELS) {
751 return &voiceChatLists[headModelVoiceChat[i].voiceChatNum];
752 }
753 // find a <headmodelname>.vc file
754 for ( i = 0; i < MAX_HEADMODELS; i++ ) {
755 if (!strlen(headModelVoiceChat[i].headmodel)) {
756 Com_sprintf(filename, sizeof(filename), "scripts/%s.vc", headModelName);
757 voiceChatNum = CG_HeadModelVoiceChats(filename);
758 if (voiceChatNum == -1)
759 break;
760 Com_sprintf(headModelVoiceChat[i].headmodel, sizeof ( headModelVoiceChat[i].headmodel ),
761 "%s", headModelName);
762 headModelVoiceChat[i].voiceChatNum = voiceChatNum;
763 return &voiceChatLists[headModelVoiceChat[i].voiceChatNum];
764 }
765 }
766 }
767 gender = ci->gender;
768 for (k = 0; k < 2; k++) {
769 // just pick the first with the right gender
770 for ( i = 0; i < MAX_VOICEFILES; i++ ) {
771 if (strlen(voiceChatLists[i].name)) {
772 if (voiceChatLists[i].gender == gender) {
773 // store this head model with voice chat for future reference
774 for ( j = 0; j < MAX_HEADMODELS; j++ ) {
775 if (!strlen(headModelVoiceChat[j].headmodel)) {
776 Com_sprintf(headModelVoiceChat[j].headmodel, sizeof ( headModelVoiceChat[j].headmodel ),
777 "%s", headModelName);
778 headModelVoiceChat[j].voiceChatNum = i;
779 break;
780 }
781 }
782 return &voiceChatLists[i];
783 }
784 }
785 }
786 // fall back to male gender because we don't have neuter in the mission pack
787 if (gender == GENDER_MALE)
788 break;
789 gender = GENDER_MALE;
790 }
791 // store this head model with voice chat for future reference
792 for ( j = 0; j < MAX_HEADMODELS; j++ ) {
793 if (!strlen(headModelVoiceChat[j].headmodel)) {
794 Com_sprintf(headModelVoiceChat[j].headmodel, sizeof ( headModelVoiceChat[j].headmodel ),
795 "%s", headModelName);
796 headModelVoiceChat[j].voiceChatNum = 0;
797 break;
798 }
799 }
800 // just return the first voice chat list
801 return &voiceChatLists[0];
802}

890void CG_VoiceChatLocal( int mode, qboolean voiceOnly, int clientNum, int color, const char *cmd ) {
891 char *chat;
892 voiceChatList_t *voiceChatList;
893 clientInfo_t *ci;

901
902 if ( mode == SAY_ALL && cgs.gametype >= GT_TEAM && cg_teamChatsOnly.integer ) {
903 return;

907 clientNum = 0;

915 if ( CG_GetVoiceChat( voiceChatList, cmd, &snd, &chat ) ) {
916 vchat.clientNum = clientNum;
917 vchat.snd = snd;
918 vchat.voiceOnly = voiceOnly;
919 Q_strncpyz(vchat.cmd, cmd, sizeof(vchat.cmd));
920 if ( mode == SAY_TELL ) {
921 Com_sprintf(vchat.message, sizeof(vchat.message), "[%s]: %c%c%s", ci->name, Q_COLOR_ESCAPE, color, chat);
922 }
923 else if ( mode == SAY_TEAM ) {
924 Com_sprintf(vchat.message, sizeof(vchat.message), "(%s): %c%c%s", ci->name, Q_COLOR_ESCAPE, color, chat);
925 }
926 else {
927 Com_sprintf(vchat.message, sizeof(vchat.message), "%s: %c%c%s", ci->name, Q_COLOR_ESCAPE, color, chat);
928 }
929 CG_AddBufferedVoiceChat(&vchat);

979CG_ServerCommand
980
981The string has been tokenized and can be retrieved with
982Cmd_Argc() / Cmd_Argv()
983=================
984*/
985static void CG_ServerCommand( void ) {
986 const char *cmd;
987 char text[MAX_SAY_TEXT];
988
989 cmd = CG_Argv(0);
990
991 if ( !cmd[0] ) {
992 // server claimed the command
993 return;
994 }
995
996 if ( !strcmp( cmd, "cp" ) ) {
997 CG_CenterPrint( CG_Argv(1), SCREEN_HEIGHT * 0.30, BIGCHAR_WIDTH );
998 return;
999 }
1000
1001 if ( !strcmp( cmd, "cs" ) ) {
1002 CG_ConfigStringModified();
1003 return;
1004 }
1005
1006 if ( !strcmp( cmd, "print" ) ) {
1007 CG_Printf( "%s", CG_Argv(1) );
1008#ifdef MISSIONPACK
1009 cmd = CG_Argv(1); // yes, this is obviously a hack, but so is the way we hear about
1010 // votes passing or failing
1011 if ( !Q_stricmpn( cmd, "vote failed", 11 ) || !Q_stricmpn( cmd, "team vote failed", 16 )) {
1012 trap_S_StartLocalSound( cgs.media.voteFailed, CHAN_ANNOUNCER );
1013 } else if ( !Q_stricmpn( cmd, "vote passed", 11 ) || !Q_stricmpn( cmd, "team vote passed", 16 ) ) {
1014 trap_S_StartLocalSound( cgs.media.votePassed, CHAN_ANNOUNCER );
1015 }
1016#endif
1017 return;
1018 }
1019
1020 if ( !strcmp( cmd, "chat" ) ) {
1021 if ( cgs.gametype >= GT_TEAM && cg_teamChatsOnly.integer ) {
1022 return;
1023 }
1024
1025 trap_S_StartLocalSound( cgs.media.talkSound, CHAN_LOCAL_SOUND );
1026 Q_strncpyz( text, CG_Argv(1), MAX_SAY_TEXT );
1027 CG_RemoveChatEscapeChar( text );
1028 CG_Printf( "%s\n", text );
1029 return;
1030 }
1031
1032 if ( !strcmp( cmd, "tchat" ) ) {
1033 trap_S_StartLocalSound( cgs.media.talkSound, CHAN_LOCAL_SOUND );
1034 Q_strncpyz( text, CG_Argv(1), MAX_SAY_TEXT );
1035 CG_RemoveChatEscapeChar( text );
1036 CG_AddToTeamChat( text );
1037 CG_Printf( "%s\n", text );
1038 return;
1039 }
1040
1041#ifdef MISSIONPACK
1042 if ( !strcmp( cmd, "vchat" ) ) {
1043 CG_VoiceChat( SAY_ALL );
1044 return;
1045 }
1046
1047 if ( !strcmp( cmd, "vtchat" ) ) {
1048 CG_VoiceChat( SAY_TEAM );
1049 return;
1050 }
1051
1052 if ( !strcmp( cmd, "vtell" ) ) {
1053 CG_VoiceChat( SAY_TELL );
1054 return;
1055 }
1056#endif
1057
1058 if ( !strcmp( cmd, "scores" ) ) {
1059 CG_ParseScores();
1060 return;
1061 }
1062
1063 if ( !strcmp( cmd, "tinfo" ) ) {
1064 CG_ParseTeamInfo();
1065 return;
1066 }
1067
1068 if ( !strcmp( cmd, "map_restart" ) ) {
1069 CG_MapRestart();
1070 return;
1071 }
1072
1073 if ( Q_stricmp (cmd, "remapShader") == 0 )
1074 {
1075 if (trap_Argc() == 4)
1076 {
1077 char shader1[MAX_QPATH];
1078 char shader2[MAX_QPATH];
1079 char shader3[MAX_QPATH];
1080
1081 Q_strncpyz(shader1, CG_Argv(1), sizeof(shader1));
1082 Q_strncpyz(shader2, CG_Argv(2), sizeof(shader2));
1083 Q_strncpyz(shader3, CG_Argv(3), sizeof(shader3));
1084
1085 trap_R_RemapShader(shader1, shader2, shader3);
1086 }
1087
1088 return;
1089 }
1090
1091 // loaddeferred can be both a servercmd and a consolecmd
1092 if ( !strcmp( cmd, "loaddefered" ) ) { // FIXME: spelled wrong, but not changing for demo
1093 CG_LoadDeferredPlayers();
1094 return;
1095 }
1096
1097 // clientLevelShot is sent before taking a special screenshot for
1098 // the menu system during development
1099 if ( !strcmp( cmd, "clientLevelShot" ) ) {
1100 cg.levelShot = qtrue;
1101 return;
1102 }
1103
1104 CG_Printf( "Unknown client game command: %s\n", cmd );
1105}

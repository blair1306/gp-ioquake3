46Q_EXPORT intptr_t vmMain( int command, int arg0, int arg1, int arg2, int arg3, int arg4, int arg5, int arg6, int arg7, int arg8, int arg9, int arg10, int arg11 ) {
47
48 switch ( command ) {
49 case CG_INIT:
50 CG_Init( arg0, arg1, arg2 );
51 return 0;
52 case CG_SHUTDOWN:
53 CG_Shutdown();
54 return 0;
55 case CG_CONSOLE_COMMAND:
56 return CG_ConsoleCommand();
57 case CG_DRAW_ACTIVE_FRAME:
58 CG_DrawActiveFrame( arg0, arg1, arg2 );
59 return 0;
60 case CG_CROSSHAIR_PLAYER:
61 return CG_CrosshairPlayer();
62 case CG_LAST_ATTACKER:
63 return CG_LastAttacker();
64 case CG_KEY_EVENT:
65 CG_KeyEvent(arg0, arg1);
66 return 0;
67 case CG_MOUSE_EVENT:
68#ifdef MISSIONPACK
69 cgDC.cursorx = cgs.cursorX;
70 cgDC.cursory = cgs.cursorY;
71#endif
72 CG_MouseEvent(arg0, arg1);
73 return 0;
74 case CG_EVENT_HANDLING:
75 CG_EventHandling(arg0);
76 return 0;
77 default:
78 CG_Error( "vmMain: unknown command %i", command );
79 break;
80 }
81 return -1;
82}

371/*
372=================
373CG_UpdateCvars
374=================
375*/
376void CG_UpdateCvars( void ) {
377 int i;
378 cvarTable_t *cv;
379
380 for ( i = 0, cv = cvarTable ; i < cvarTableSize ; i++, cv++ ) {
381 trap_Cvar_Update( cv->vmCvar );
382 }
383
384 // check for modications here
385
386 // If team overlay is on, ask for updates from the server. If it's off,
387 // let the server know so we don't receive it
388 if ( drawTeamOverlayModificationCount != cg_drawTeamOverlay.modificationCount ) {
389 drawTeamOverlayModificationCount = cg_drawTeamOverlay.modificationCount;
390
391 if ( cg_drawTeamOverlay.integer > 0 ) {
392 trap_Cvar_Set( "teamoverlay", "1" );
393 } else {
394 trap_Cvar_Set( "teamoverlay", "0" );
395 }
396 }
397
398 // if force model changed
399 if ( forceModelModificationCount != cg_forceModel.modificationCount ) {
400 forceModelModificationCount = cg_forceModel.modificationCount;
401 CG_ForceModelChange();
402 }

480=================
481CG_RegisterItemSounds
482
483The server says this item is used on this level
484=================
485*/
486static void CG_RegisterItemSounds( int itemNum ) {
487 gitem_t *item;
488 char data[MAX_QPATH];
489 char *s, *start;
490 int len;
491
492 item = &bg_itemlist[ itemNum ];
493
494 if( item->pickup_sound ) {
495 trap_S_RegisterSound( item->pickup_sound, qfalse );
496 }
497
498 // parse the space seperated precache string for other media
499 s = item->sounds;
500 if (!s || !s[0])
501 return;
502
503 while (*s) {
504 start = s;
505 while (*s && *s != ' ') {
506 s++;
507 }
508
509 len = s-start;
510 if (len >= MAX_QPATH || len < 5) {
511 CG_Error( "PrecacheItem: %s has bad precache string",
512 item->classname);
513 return;
514 }
515 memcpy (data, start, len);
516 data[len] = 0;
517 if ( *s ) {
518 s++;
519 }
520
521 if ( !strcmp(data+len-3, "wav" )) {
522 trap_S_RegisterSound( data, qfalse );
523 }
524 }
525}

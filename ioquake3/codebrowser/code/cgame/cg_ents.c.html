28/*
29======================
30CG_PositionEntityOnTag
31
32Modifies the entities position and axis by the given
33tag location
34======================
35*/
36void CG_PositionEntityOnTag( refEntity_t *entity, const refEntity_t *parent,
37 qhandle_t parentModel, char *tagName ) {
38 int i;
39 orientation_t lerped;
40
41 // lerp the tag
42 trap_R_LerpTag( &lerped, parentModel, parent->oldframe, parent->frame,
43 1.0 - parent->backlerp, tagName );
44
45 // FIXME: allow origin offsets along tag?
46 VectorCopy( parent->origin, entity->origin );
47 for ( i = 0 ; i < 3 ; i++ ) {
48 VectorMA( entity->origin, lerped.origin[i], parent->axis[i], entity->origin );
49 }
50
51 // had to cast away the const to avoid compiler problems...
52 MatrixMultiply( lerped.axis, ((refEntity_t *)parent)->axis, entity->axis );
53 entity->backlerp = parent->backlerp;
54}

57/*
58======================
59CG_PositionRotatedEntityOnTag
60
61Modifies the entities position and axis by the given
62tag location
63======================
64*/
65void CG_PositionRotatedEntityOnTag( refEntity_t *entity, const refEntity_t *parent,
66 qhandle_t parentModel, char *tagName ) {
67 int i;
68 orientation_t lerped;
69 vec3_t tempAxis[3];
70
71//AxisClear( entity->axis );
72 // lerp the tag
73 trap_R_LerpTag( &lerped, parentModel, parent->oldframe, parent->frame,
74 1.0 - parent->backlerp, tagName );
75
76 // FIXME: allow origin offsets along tag?
77 VectorCopy( parent->origin, entity->origin );
78 for ( i = 0 ; i < 3 ; i++ ) {
79 VectorMA( entity->origin, lerped.origin[i], parent->axis[i], entity->origin );
80 }
81
82 // had to cast away the const to avoid compiler problems...
83 MatrixMultiply( entity->axis, lerped.axis, tempAxis );
84 MatrixMultiply( tempAxis, ((refEntity_t *)parent)->axis, entity->axis );
85}

117/*
118==================
119CG_EntityEffects
120
121Add continuous entity effects, like local entity emission and lighting
122==================
123*/
124static void CG_EntityEffects( centity_t *cent ) {
125
126 // update sound origins
127 CG_SetEntitySoundPosition( cent );
128
129 // add loop sound
130 if ( cent->currentState.loopSound ) {
131 if (cent->currentState.eType != ET_SPEAKER) {
132 trap_S_AddLoopingSound( cent->currentState.number, cent->lerpOrigin, vec3_origin,
133 cgs.gameSounds[ cent->currentState.loopSound ] );
134 } else {
135 trap_S_AddRealLoopingSound( cent->currentState.number, cent->lerpOrigin, vec3_origin,
136 cgs.gameSounds[ cent->currentState.loopSound ] );
137 }
138 }
139
140
141 // constant light glow
142 if(cent->currentState.constantLight)
143 {
144 int cl;
145 float i, r, g, b;
146
147 cl = cent->currentState.constantLight;
148 r = (float) (cl & 0xFF) / 255.0;
149 g = (float) ((cl >> 8) & 0xFF) / 255.0;
150 b = (float) ((cl >> 16) & 0xFF) / 255.0;
151 i = (float) ((cl >> 24) & 0xFF) * 4.0;
152 trap_R_AddLightToScene(cent->lerpOrigin, i, r, g, b);
153 }
154
155}

158/*
159==================
160CG_General
161==================
162*/
163static void CG_General( centity_t *cent ) {
164 refEntity_t ent;
165 entityState_t *s1;
166
167 s1 = &cent->currentState;
168
169 // if set to invisible, skip
170 if (!s1->modelindex) {
171 return;
172 }
173
174 memset (&ent, 0, sizeof(ent));
175
176 // set frame
177
178 ent.frame = s1->frame;
179 ent.oldframe = ent.frame;
180 ent.backlerp = 0;
181
182 VectorCopy( cent->lerpOrigin, ent.origin);
183 VectorCopy( cent->lerpOrigin, ent.oldorigin);
184
185 ent.hModel = cgs.gameModels[s1->modelindex];
186
187 // player model
188 if (s1->number == cg.snap->ps.clientNum) {
189 ent.renderfx |= RF_THIRD_PERSON; // only draw from mirrors
190 }
191
192 // convert angles to axis
193 AnglesToAxis( cent->lerpAngles, ent.axis );
194
195 // add to refresh list
196 trap_R_AddRefEntityToScene (&ent);
197}

220}

222/*
223==================
224CG_Item
225==================
226*/
227static void CG_Item( centity_t *cent ) {
228 refEntity_t ent;
229 entityState_t *es;
230 gitem_t *item;
231 int msec;
232 float frac;
233 float scale;
234 weaponInfo_t *wi;
235
236 es = &cent->currentState;
237 if ( es->modelindex >= bg_numItems ) {
238 CG_Error( "Bad item index %i on entity", es->modelindex );
239 }
240
241 // if set to invisible, skip
242 if ( !es->modelindex || ( es->eFlags & EF_NODRAW ) ) {
243 return;
244 }
245
246 item = &bg_itemlist[ es->modelindex ];
247 if ( cg_simpleItems.integer && item->giType != IT_TEAM ) {
248 memset( &ent, 0, sizeof( ent ) );
249 ent.reType = RT_SPRITE;
250 VectorCopy( cent->lerpOrigin, ent.origin );
251 ent.radius = 14;
252 ent.customShader = cg_items[es->modelindex].icon;
253 ent.shaderRGBA[0] = 255;
254 ent.shaderRGBA[1] = 255;
255 ent.shaderRGBA[2] = 255;
256 ent.shaderRGBA[3] = 255;
257 trap_R_AddRefEntityToScene(&ent);
258 return;
259 }
260
261 // items bob up and down continuously
262 scale = 0.005 + cent->currentState.number * 0.00001;
263 cent->lerpOrigin[2] += 4 + cos( ( cg.time + 1000 ) * scale ) * 4;
264
265 memset (&ent, 0, sizeof(ent));
266
267 // autorotate at one of two speeds
268 if ( item->giType == IT_HEALTH ) {
269 VectorCopy( cg.autoAnglesFast, cent->lerpAngles );
270 AxisCopy( cg.autoAxisFast, ent.axis );
271 } else {
272 VectorCopy( cg.autoAngles, cent->lerpAngles );
273 AxisCopy( cg.autoAxis, ent.axis );
274 }
275
276 wi = NULL;
277 // the weapons have their origin where they attatch to player
278 // models, so we need to offset them or they will rotate
279 // eccentricly
280 if ( item->giType == IT_WEAPON ) {
281 wi = &cg_weapons[item->giTag];
282 cent->lerpOrigin[0] -=
283 wi->weaponMidpoint[0] * ent.axis[0][0] +
284 wi->weaponMidpoint[1] * ent.axis[1][0] +
285 wi->weaponMidpoint[2] * ent.axis[2][0];
286 cent->lerpOrigin[1] -=
287 wi->weaponMidpoint[0] * ent.axis[0][1] +
288 wi->weaponMidpoint[1] * ent.axis[1][1] +
289 wi->weaponMidpoint[2] * ent.axis[2][1];
290 cent->lerpOrigin[2] -=
291 wi->weaponMidpoint[0] * ent.axis[0][2] +
292 wi->weaponMidpoint[1] * ent.axis[1][2] +
293 wi->weaponMidpoint[2] * ent.axis[2][2];
294
295 cent->lerpOrigin[2] += 8; // an extra height boost
296 }
297
298 if( item->giType == IT_WEAPON && item->giTag == WP_RAILGUN ) {
299 clientInfo_t *ci = &cgs.clientinfo[cg.snap->ps.clientNum];
300 Byte4Copy( ci->c1RGBA, ent.shaderRGBA );
301 }
302
303 ent.hModel = cg_items[es->modelindex].models[0];
304
305 VectorCopy( cent->lerpOrigin, ent.origin);
306 VectorCopy( cent->lerpOrigin, ent.oldorigin);
307
308 ent.nonNormalizedAxes = qfalse;
309
310 // if just respawned, slowly scale up
311 msec = cg.time - cent->miscTime;
312 if ( msec >= 0 && msec < ITEM_SCALEUP_TIME ) {
313 frac = (float)msec / ITEM_SCALEUP_TIME;
314 VectorScale( ent.axis[0], frac, ent.axis[0] );
315 VectorScale( ent.axis[1], frac, ent.axis[1] );
316 VectorScale( ent.axis[2], frac, ent.axis[2] );
317 ent.nonNormalizedAxes = qtrue;
318 } else {
319 frac = 1.0;
320 }
321
322 // items without glow textures need to keep a minimum light value
323 // so they are always visible
324 if ( ( item->giType == IT_WEAPON ) ||
325 ( item->giType == IT_ARMOR ) ) {
326 ent.renderfx |= RF_MINLIGHT;
327 }
328
329 // increase the size of the weapons when they are presented as items
330 if ( item->giType == IT_WEAPON ) {
331 VectorScale( ent.axis[0], 1.5, ent.axis[0] );
332 VectorScale( ent.axis[1], 1.5, ent.axis[1] );
333 VectorScale( ent.axis[2], 1.5, ent.axis[2] );
334 ent.nonNormalizedAxes = qtrue;
335#ifdef MISSIONPACK
336 trap_S_AddLoopingSound( cent->currentState.number, cent->lerpOrigin, vec3_origin, cgs.media.weaponHoverSound );
337#endif
338 }
339
340#ifdef MISSIONPACK
341 if ( item->giType == IT_HOLDABLE && item->giTag == HI_KAMIKAZE ) {
342 VectorScale( ent.axis[0], 2, ent.axis[0] );
343 VectorScale( ent.axis[1], 2, ent.axis[1] );
344 VectorScale( ent.axis[2], 2, ent.axis[2] );
345 ent.nonNormalizedAxes = qtrue;
346 }
347#endif
348
349 // add to refresh list
350 trap_R_AddRefEntityToScene(&ent);
351
352 if ( item->giType == IT_WEAPON && wi && wi->barrelModel ) {
353 refEntity_t barrel;
354 vec3_t angles;
355
356 memset( &barrel, 0, sizeof( barrel ) );
357
358 barrel.hModel = wi->barrelModel;
359
360 VectorCopy( ent.lightingOrigin, barrel.lightingOrigin );
361 barrel.shadowPlane = ent.shadowPlane;
362 barrel.renderfx = ent.renderfx;
363
364 angles[YAW] = 0;
365 angles[PITCH] = 0;
366 angles[ROLL] = 0;
367 AnglesToAxis( angles, barrel.axis );
368
369 CG_PositionRotatedEntityOnTag( &barrel, &ent, wi->weaponModel, "tag_barrel" );
370
371 barrel.nonNormalizedAxes = ent.nonNormalizedAxes;
372
373 trap_R_AddRefEntityToScene( &barrel );
374 }
375
376 // accompanying rings / spheres for powerups
377 if ( !cg_simpleItems.integer )
378 {
379 vec3_t spinAngles;
380
381 VectorClear( spinAngles );
382
383 if ( item->giType == IT_HEALTH || item->giType == IT_POWERUP )
384 {
385 if ( ( ent.hModel = cg_items[es->modelindex].models[1] ) != 0 )
386 {
387 if ( item->giType == IT_POWERUP )
388 {
389 ent.origin[2] += 12;
390 spinAngles[1] = ( cg.time & 1023 ) * 360 / -1024.0f;
391 }
392 AnglesToAxis( spinAngles, ent.axis );
393
394 // scale up if respawning
395 if ( frac != 1.0 ) {
396 VectorScale( ent.axis[0], frac, ent.axis[0] );
397 VectorScale( ent.axis[1], frac, ent.axis[1] );
398 VectorScale( ent.axis[2], frac, ent.axis[2] );
399 ent.nonNormalizedAxes = qtrue;
400 }
401 trap_R_AddRefEntityToScene( &ent );
402 }
403 }
404 }
405}

409/*
410===============
411CG_Missile
412===============
413*/
414static void CG_Missile( centity_t *cent ) {
415 refEntity_t ent;
416 entityState_t *s1;
417 const weaponInfo_t *weapon;
418// int col;
419
420 s1 = &cent->currentState;
421 if ( s1->weapon >= WP_NUM_WEAPONS ) {
422 s1->weapon = 0;
423 }
424 weapon = &cg_weapons[s1->weapon];
425
426 // calculate the axis
427 VectorCopy( s1->angles, cent->lerpAngles);
428
429 // add trails
430 if ( weapon->missileTrailFunc )
431 {
432 weapon->missileTrailFunc( cent, weapon );
433 }
434/*
435 if ( cent->currentState.modelindex == TEAM_RED ) {
436 col = 1;
437 }
438 else if ( cent->currentState.modelindex == TEAM_BLUE ) {
439 col = 2;
440 }
441 else {
442 col = 0;
443 }
444
445 // add dynamic light
446 if ( weapon->missileDlight ) {
447 trap_R_AddLightToScene(cent->lerpOrigin, weapon->missileDlight,
448 weapon->missileDlightColor[col][0], weapon->missileDlightColor[col][1], weapon->missileDlightColor[col][2] );
449 }
450*/
451 // add dynamic light
452 if ( weapon->missileDlight ) {
453 trap_R_AddLightToScene(cent->lerpOrigin, weapon->missileDlight,
454 weapon->missileDlightColor[0], weapon->missileDlightColor[1], weapon->missileDlightColor[2] );
455 }
456
457 // add missile sound
458 if ( weapon->missileSound ) {
459 vec3_t velocity;
460
461 BG_EvaluateTrajectoryDelta( &cent->currentState.pos, cg.time, velocity );
462
463 trap_S_AddLoopingSound( cent->currentState.number, cent->lerpOrigin, velocity, weapon->missileSound );
464 }
465
466 // create the render entity
467 memset (&ent, 0, sizeof(ent));
468 VectorCopy( cent->lerpOrigin, ent.origin);
469 VectorCopy( cent->lerpOrigin, ent.oldorigin);
470
471 if ( cent->currentState.weapon == WP_PLASMAGUN ) {
472 ent.reType = RT_SPRITE;
473 ent.radius = 16;
474 ent.rotation = 0;
475 ent.customShader = cgs.media.plasmaBallShader;
476 trap_R_AddRefEntityToScene( &ent );
477 return;
478 }
479
480 // flicker between two skins
481 ent.skinNum = cg.clientFrame & 1;
482 ent.hModel = weapon->missileModel;
483 ent.renderfx = weapon->missileRenderfx | RF_NOSHADOW;
484
485#ifdef MISSIONPACK
486 if ( cent->currentState.weapon == WP_PROX_LAUNCHER ) {
487 if (s1->generic1 == TEAM_BLUE) {
488 ent.hModel = cgs.media.blueProxMine;
489 }
490 }
491#endif
492
493 // convert direction of travel into axis
494 if ( VectorNormalize2( s1->pos.trDelta, ent.axis[0] ) == 0 ) {
495 ent.axis[0][2] = 1;
496 }
497
498 // spin as it moves
499 if ( s1->pos.trType != TR_STATIONARY ) {
500 RotateAroundDirection( ent.axis, cg.time / 4 );
501 } else {
502#ifdef MISSIONPACK
503 if ( s1->weapon == WP_PROX_LAUNCHER ) {
504 AnglesToAxis( cent->lerpAngles, ent.axis );
505 }
506 else
507#endif
508 {
509 RotateAroundDirection( ent.axis, s1->time );
510 }
511 }
512
513 // add to refresh list, possibly with quad glow
514 CG_AddRefEntityWithPowerups( &ent, s1, TEAM_FREE );
515}

602 trap_R_AddRefEntityToScene(&ent);

636CG_Portal
637===============
638*/
639static void CG_Portal( centity_t *cent ) {
640 refEntity_t ent;
641 entityState_t *s1;
642
643 s1 = &cent->currentState;
644
645 // create the render entity
646 memset (&ent, 0, sizeof(ent));
647 VectorCopy( cent->lerpOrigin, ent.origin );

649 ByteToDir( s1->eventParm, ent.axis[0] );

657 ent.reType = RT_PORTALSURFACE;
658 ent.oldframe = s1->powerups;
659 ent.frame = s1->frame; // rotation speed
660 ent.skinNum = s1->clientNum/256.0 * 360; // roll offset
661
662 // add to refresh list
663 trap_R_AddRefEntityToScene(&ent);
664}

667/*
668=========================
669CG_AdjustPositionForMover
670
671Also called by client movement prediction code
672=========================
673*/
674void CG_AdjustPositionForMover(const vec3_t in, int moverNum, int fromTime, int toTime, vec3_t out, vec3_t angles_in, vec3_t angles_out) {
675 centity_t *cent;
676 vec3_t oldOrigin, origin, deltaOrigin;
677 vec3_t oldAngles, angles, deltaAngles;
678
679 if ( moverNum <= 0 || moverNum >= ENTITYNUM_MAX_NORMAL ) {
680 VectorCopy( in, out );
681 VectorCopy(angles_in, angles_out);
682 return;
683 }
684
685 cent = &cg_entities[ moverNum ];
686 if ( cent->currentState.eType != ET_MOVER ) {
687 VectorCopy( in, out );
688 VectorCopy(angles_in, angles_out);
689 return;
690 }
691
692 BG_EvaluateTrajectory( &cent->currentState.pos, fromTime, oldOrigin );
693 BG_EvaluateTrajectory( &cent->currentState.apos, fromTime, oldAngles );
694
695 BG_EvaluateTrajectory( &cent->currentState.pos, toTime, origin );
696 BG_EvaluateTrajectory( &cent->currentState.apos, toTime, angles );
697
698 VectorSubtract( origin, oldOrigin, deltaOrigin );
699 VectorSubtract( angles, oldAngles, deltaAngles );
700
701 VectorAdd( in, deltaOrigin, out );
702 VectorAdd( angles_in, deltaAngles, angles_out );
703 // FIXME: origin change when on a rotating object
704}

707/*
708=============================
709CG_InterpolateEntityPosition
710=============================
711*/
712static void CG_InterpolateEntityPosition( centity_t *cent ) {
713 vec3_t current, next;
714 float f;
715
716 // it would be an internal error to find an entity that interpolates without
717 // a snapshot ahead of the current one
718 if ( cg.nextSnap == NULL ) {
719 CG_Error( "CG_InterpoateEntityPosition: cg.nextSnap == NULL" );
720 }
721
722 f = cg.frameInterpolation;
723
724 // this will linearize a sine or parabolic curve, but it is important
725 // to not extrapolate player positions if more recent data is available
726 BG_EvaluateTrajectory( &cent->currentState.pos, cg.snap->serverTime, current );
727 BG_EvaluateTrajectory( &cent->nextState.pos, cg.nextSnap->serverTime, next );
728
729 cent->lerpOrigin[0] = current[0] + f * ( next[0] - current[0] );
730 cent->lerpOrigin[1] = current[1] + f * ( next[1] - current[1] );
731 cent->lerpOrigin[2] = current[2] + f * ( next[2] - current[2] );
732
733 BG_EvaluateTrajectory( &cent->currentState.apos, cg.snap->serverTime, current );
734 BG_EvaluateTrajectory( &cent->nextState.apos, cg.nextSnap->serverTime, next );
735
736 cent->lerpAngles[0] = LerpAngle( current[0], next[0], f );
737 cent->lerpAngles[1] = LerpAngle( current[1], next[1], f );
738 cent->lerpAngles[2] = LerpAngle( current[2], next[2], f );
739
740}

746===============
747*/
748static void CG_CalcEntityLerpPositions( centity_t *cent ) {
749
750 // if this player does not want to see extrapolated players
751 if ( !cg_smoothClients.integer ) {
752 // make sure the clients use TR_INTERPOLATE
753 if ( cent->currentState.number < MAX_CLIENTS ) {
754 cent->currentState.pos.trType = TR_INTERPOLATE;
755 cent->nextState.pos.trType = TR_INTERPOLATE;
756 }
757 }
758
759 if ( cent->interpolate && cent->currentState.pos.trType == TR_INTERPOLATE ) {
760 CG_InterpolateEntityPosition( cent );
761 return;
762 }
763
764 // first see if we can interpolate between two snaps for
765 // linear extrapolated clients
766 if ( cent->interpolate && cent->currentState.pos.trType == TR_LINEAR_STOP &&
767 cent->currentState.number < MAX_CLIENTS) {
768 CG_InterpolateEntityPosition( cent );
769 return;
770 }
771
772 // just use the current frame and evaluate as best we can
773 BG_EvaluateTrajectory( &cent->currentState.pos, cg.time, cent->lerpOrigin );
774 BG_EvaluateTrajectory( &cent->currentState.apos, cg.time, cent->lerpAngles );
775
776 // adjust for riding a mover if it wasn't rolled into the predicted
777 // player state
778 if ( cent != &cg.predictedPlayerEntity ) {
779 CG_AdjustPositionForMover( cent->lerpOrigin, cent->currentState.groundEntityNum,
780 cg.snap->serverTime, cg.time, cent->lerpOrigin, cent->lerpAngles, cent->lerpAngles);
781 }
782}

943static void CG_AddCEntity( centity_t *cent ) {
944 // event-only entities will have been dealt with already
945 if ( cent->currentState.eType >= ET_EVENTS ) {
946 return;
947 }
948
949 // calculate the current origin
950 CG_CalcEntityLerpPositions( cent );
951
952 // add automatic effects
953 CG_EntityEffects( cent );
954
955 switch ( cent->currentState.eType ) {
956 default:
957 CG_Error( "Bad entity type: %i", cent->currentState.eType );
958 break;
959 case ET_INVISIBLE:
960 case ET_PUSH_TRIGGER:
961 case ET_TELEPORT_TRIGGER:
962 break;
963 case ET_GENERAL:
964 CG_General( cent );
965 break;
966 case ET_PLAYER:
967 CG_Player( cent );
968 break;
969 case ET_ITEM:
970 CG_Item( cent );
971 break;
972 case ET_MISSILE:
973 CG_Missile( cent );
974 break;
975 case ET_MOVER:
976 CG_Mover( cent );
977 break;
978 case ET_BEAM:
979 CG_Beam( cent );
980 break;
981 case ET_PORTAL:
982 CG_Portal( cent );
983 break;
984 case ET_SPEAKER:
985 CG_Speaker( cent );
986 break;
987 case ET_GRAPPLE:
988 CG_Grapple( cent );
989 break;
990 case ET_TEAM:
991 CG_TeamBase( cent );
992 break;
993 }

1002void CG_AddPacketEntities( void ) {
1003 int num;
1004 centity_t *cent;
1005 playerState_t *ps;
1006
1007 // set cg.frameInterpolation
1008 if ( cg.nextSnap ) {
1009 int delta;
1010
1011 delta = (cg.nextSnap->serverTime - cg.snap->serverTime);
1012 if ( delta == 0 ) {
1013 cg.frameInterpolation = 0;
1014 } else {
1015 cg.frameInterpolation = (float)( cg.time - cg.snap->serverTime ) / delta;
1016 }
1017 } else {
1018 cg.frameInterpolation = 0; // actually, it should never be used, because
1019 // no entities should be marked as interpolating
1020 }
1021
1022 // the auto-rotating items will all have the same axis
1023 cg.autoAngles[0] = 0;
1024 cg.autoAngles[1] = ( cg.time & 2047 ) * 360 / 2048.0;
1025 cg.autoAngles[2] = 0;
1026
1027 cg.autoAnglesFast[0] = 0;
1028 cg.autoAnglesFast[1] = ( cg.time & 1023 ) * 360 / 1024.0f;
1029 cg.autoAnglesFast[2] = 0;
1030
1031 AnglesToAxis( cg.autoAngles, cg.autoAxis );
1032 AnglesToAxis( cg.autoAnglesFast, cg.autoAxisFast );
1033
1034 // generate and add the entity from the playerstate
1035 ps = &cg.predictedPlayerState;
1036 BG_PlayerStateToEntityState( ps, &cg.predictedPlayerEntity.currentState, qfalse );
1037 CG_AddCEntity( &cg.predictedPlayerEntity );
1038
1039 // lerp the non-predicted value for lightning gun origins
1040 CG_CalcEntityLerpPositions( &cg_entities[ cg.snap->ps.clientNum ] );
1041
1042 // add each entity sent over by the server
1043 for ( num = 0 ; num < cg.snap->numEntities ; num++ ) {
1044 cent = &cg_entities[ cg.snap->entities[ num ].number ];
1045 CG_AddCEntity( cent );
1046 }

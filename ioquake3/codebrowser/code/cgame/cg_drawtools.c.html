26/*
27================
28CG_AdjustFrom640
29
30Adjusted for resolution and screen aspect ratio
31================
32*/
33void CG_AdjustFrom640( float *x, float *y, float *w, float *h ) {
34#if 0
35 // adjust for wide screens
36 if ( cgs.glconfig.vidWidth * 480 > cgs.glconfig.vidHeight * 640 ) {
37 *x += 0.5 * ( cgs.glconfig.vidWidth - ( cgs.glconfig.vidHeight * 640 / 480 ) );
38 }
39#endif
40 // scale for screen sizes
41 *x *= cgs.screenXScale;
42 *y *= cgs.screenYScale;
43 *w *= cgs.screenXScale;
44 *h *= cgs.screenYScale;
45}

115/*
116===============
117CG_DrawChar
118
119Coordinates and size in 640*480 virtual screen size
120===============
121*/
122void CG_DrawChar( int x, int y, int width, int height, int ch ) {
123 int row, col;
124 float frow, fcol;
125 float size;
126 float ax, ay, aw, ah;
127
128 ch &= 255;
129
130 if ( ch == ' ' ) {
131 return;
132 }
133
134 ax = x;
135 ay = y;
136 aw = width;
137 ah = height;
138 CG_AdjustFrom640( &ax, &ay, &aw, &ah );
139
140 row = ch>>4;
141 col = ch&15;
142
143 frow = row*0.0625;
144 fcol = col*0.0625;
145 size = 0.0625;
146
147 trap_R_DrawStretchPic( ax, ay, aw, ah,
148 fcol, frow,
149 fcol + size, frow + size,
150 cgs.media.charsetShader );
151}

154/*
155==================
156CG_DrawStringExt
157
158Draws a multi-colored string with a drop shadow, optionally forcing
159to a fixed color.
160
161Coordinates are at 640 by 480 virtual resolution
162==================
163*/
164void CG_DrawStringExt( int x, int y, const char *string, const float *setColor,
165 qboolean forceColor, qboolean shadow, int charWidth, int charHeight, int maxChars ) {
166 vec4_t color;
167 const char *s;
168 int xx;
169 int cnt;
170
171 if (maxChars <= 0)
172 maxChars = 32767; // do them all!
173
174 // draw the drop shadow
175 if (shadow) {
176 color[0] = color[1] = color[2] = 0;
177 color[3] = setColor[3];
178 trap_R_SetColor( color );
179 s = string;
180 xx = x;
181 cnt = 0;
182 while ( *s && cnt < maxChars) {
183 if ( Q_IsColorString( s ) ) {
184 s += 2;
185 continue;
186 }
187 CG_DrawChar( xx + 2, y + 2, charWidth, charHeight, *s );
188 cnt++;
189 xx += charWidth;
190 s++;
191 }
192 }
193
194 // draw the colored text
195 s = string;
196 xx = x;
197 cnt = 0;
198 trap_R_SetColor( setColor );
199 while ( *s && cnt < maxChars) {
200 if ( Q_IsColorString( s ) ) {
201 if ( !forceColor ) {
202 memcpy( color, g_color_table[ColorIndex(*(s+1))], sizeof( color ) );
203 color[3] = setColor[3];
204 trap_R_SetColor( color );
205 }
206 s += 2;
207 continue;
208 }
209 CG_DrawChar( xx, y, charWidth, charHeight, *s );
210 xx += charWidth;
211 cnt++;
212 s++;
213 }
214 trap_R_SetColor( NULL );
215}

284/*
285==============
286CG_TileClear
287
288Clear around a sized down screen
289==============
290*/
291void CG_TileClear( void ) {
292 int top, bottom, left, right;
293 int w, h;
294
295 w = cgs.glconfig.vidWidth;
296 h = cgs.glconfig.vidHeight;
297
298 if ( cg.refdef.x == 0 && cg.refdef.y == 0 &&
299 cg.refdef.width == w && cg.refdef.height == h ) {
300 return; // full screen rendering
301 }
302
303 top = cg.refdef.y;
304 bottom = top + cg.refdef.height-1;
305 left = cg.refdef.x;
306 right = left + cg.refdef.width-1;
307
308 // clear above view screen
309 CG_TileClearBox( 0, 0, w, top, cgs.media.backTileShader );
310
311 // clear below view screen
312 CG_TileClearBox( 0, bottom, w, h - bottom, cgs.media.backTileShader );
313
314 // clear left of view screen
315 CG_TileClearBox( 0, top, left, bottom - top + 1, cgs.media.backTileShader );
316
317 // clear right of view screen
318 CG_TileClearBox( right, top, w - right, bottom - top + 1, cgs.media.backTileShader );
319}

323/*
324================
325CG_FadeColor
326================
327*/
328float *CG_FadeColor( int startMsec, int totalMsec ) {
329 static vec4_t color;
330 int t;
331
332 if ( startMsec == 0 ) {
333 return NULL;
334 }
335
336 t = cg.time - startMsec;
337
338 if ( t >= totalMsec ) {
339 return NULL;
340 }
341
342 // fade out
343 if ( totalMsec - t < FADE_TIME ) {
344 color[3] = ( totalMsec - t ) * 1.0/FADE_TIME;
345 } else {
346 color[3] = 1.0;
347 }
348 color[0] = color[1] = color[2] = 1;
349
350 return color;
351}

379/*
380=================
381CG_GetColorForHealth
382=================
383*/
384void CG_GetColorForHealth( int health, int armor, vec4_t hcolor ) {
385 int count;
386 int max;
387
388 // calculate the total points of damage that can
389 // be sustained at the current health / armor level
390 if ( health <= 0 ) {
391 VectorClear( hcolor ); // black
392 hcolor[3] = 1;
393 return;
394 }
395 count = armor;
396 max = health * ARMOR_PROTECTION / ( 1.0 - ARMOR_PROTECTION );
397 if ( max < count ) {
398 count = max;
399 }
400 health += count;
401
402 // set the color based on health
403 hcolor[0] = 1.0;
404 hcolor[3] = 1.0;
405 if ( health >= 100 ) {
406 hcolor[2] = 1.0;
407 } else if ( health < 66 ) {
408 hcolor[2] = 0;
409 } else {
410 hcolor[2] = ( health - 66 ) / 33.0;
411 }
412
413 if ( health > 60 ) {
414 hcolor[1] = 1.0;
415 } else if ( health < 30 ) {
416 hcolor[1] = 0;
417 } else {
418 hcolor[1] = ( health - 30 ) / 30.0;
419 }
420}

585/*
586=================
587UI_DrawBannerString
588=================
589*/
590static void UI_DrawBannerString2( int x, int y, const char* str, vec4_t color )
591{
592 const char* s;
593 unsigned char ch;
594 float ax;
595 float ay;
596 float aw;
597 float ah;
598 float frow;
599 float fcol;
600 float fwidth;
601 float fheight;
602
603 // draw the colored text
604 trap_R_SetColor( color );
605
606 ax = x * cgs.screenXScale + cgs.screenXBias;
607 ay = y * cgs.screenYScale;
608
609 s = str;
610 while ( *s )
611 {
612 ch = *s & 127;
613 if ( ch == ' ' ) {
614 ax += ((float)PROPB_SPACE_WIDTH + (float)PROPB_GAP_WIDTH)* cgs.screenXScale;
615 }
616 else if ( ch >= 'A' && ch <= 'Z' ) {
617 ch -= 'A';
618 fcol = (float)propMapB[ch][0] / 256.0f;
619 frow = (float)propMapB[ch][1] / 256.0f;
620 fwidth = (float)propMapB[ch][2] / 256.0f;
621 fheight = (float)PROPB_HEIGHT / 256.0f;
622 aw = (float)propMapB[ch][2] * cgs.screenXScale;
623 ah = (float)PROPB_HEIGHT * cgs.screenYScale;
624 trap_R_DrawStretchPic( ax, ay, aw, ah, fcol, frow, fcol+fwidth, frow+fheight, cgs.media.charsetPropB );
625 ax += (aw + (float)PROPB_GAP_WIDTH * cgs.screenXScale);
626 }
627 s++;
628 }
629
630 trap_R_SetColor( NULL );
631}

633void UI_DrawBannerString( int x, int y, const char* str, int style, vec4_t color ) {
634 const char * s;
635 int ch;
636 int width;
637 vec4_t drawcolor;
638
639 // find the width of the drawn text
640 s = str;
641 width = 0;
642 while ( *s ) {
643 ch = *s;
644 if ( ch == ' ' ) {
645 width += PROPB_SPACE_WIDTH;
646 }
647 else if ( ch >= 'A' && ch <= 'Z' ) {
648 width += propMapB[ch - 'A'][2] + PROPB_GAP_WIDTH;
649 }
650 s++;
651 }
652 width -= PROPB_GAP_WIDTH;
653
654 switch( style & UI_FORMATMASK ) {
655 case UI_CENTER:
656 x -= width / 2;
657 break;
658
659 case UI_RIGHT:
660 x -= width;
661 break;
662
663 case UI_LEFT:
664 default:
665 break;
666 }
667
668 if ( style & UI_DROPSHADOW ) {
669 drawcolor[0] = drawcolor[1] = drawcolor[2] = 0;
670 drawcolor[3] = color[3];
671 UI_DrawBannerString2( x+2, y+2, str, drawcolor );
672 }
673
674 UI_DrawBannerString2( x, y, str, color );
675}

678int UI_ProportionalStringWidth( const char* str ) {
679 const char * s;
680 int ch;
681 int charWidth;
682 int width;
683
684 s = str;
685 width = 0;
686 while ( *s ) {
687 ch = *s & 127;
688 charWidth = propMap[ch][2];
689 if ( charWidth != -1 ) {
690 width += charWidth;
691 width += PROP_GAP_WIDTH;
692 }
693 s++;
694 }
695
696 width -= PROP_GAP_WIDTH;
697 return width;
698}

715
716 ax = x * cgs.screenXScale + cgs.screenXBias;
717 ay = y * cgs.screenYScale;
718
719 s = str;

721 {

723 if ( ch == ' ' ) {
724 aw = (float)PROP_SPACE_WIDTH * cgs.screenXScale * sizeScale;
725 } else if ( propMap[ch][2] != -1 ) {
726 fcol = (float)propMap[ch][0] / 256.0f;

733 } else {
734 aw = 0;
735 }
736
737 ax += (aw + (float)PROP_GAP_WIDTH * cgs.screenXScale * sizeScale);
738 s++;
739 }
740
741 trap_R_SetColor( NULL );

758/*
759=================
760UI_DrawProportionalString
761=================
762*/
763void UI_DrawProportionalString( int x, int y, const char* str, int style, vec4_t color ) {
764 vec4_t drawcolor;
765 int width;
766 float sizeScale;
767
768 sizeScale = UI_ProportionalSizeScale( style );
769
770 switch( style & UI_FORMATMASK ) {
771 case UI_CENTER:
772 width = UI_ProportionalStringWidth( str ) * sizeScale;
773 x -= width / 2;
774 break;
775
776 case UI_RIGHT:
777 width = UI_ProportionalStringWidth( str ) * sizeScale;
778 x -= width;
779 break;
780
781 case UI_LEFT:
782 default:
783 break;
784 }
785
786 if ( style & UI_DROPSHADOW ) {
787 drawcolor[0] = drawcolor[1] = drawcolor[2] = 0;
788 drawcolor[3] = color[3];
789 UI_DrawProportionalString2( x+2, y+2, str, drawcolor, sizeScale, cgs.media.charsetProp );
790 }

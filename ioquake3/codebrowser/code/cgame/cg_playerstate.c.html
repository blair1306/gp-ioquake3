30/*
31==============
32CG_CheckAmmo
33
34If the ammo has gone low enough to generate the warning, play a sound
35==============
36*/
37void CG_CheckAmmo( void ) {
38 int i;
39 int total;
40 int previous;
41 int weapons;
42
43 // see about how many seconds of ammo we have remaining
44 weapons = cg.snap->ps.stats[ STAT_WEAPONS ];
45 total = 0;
46 for ( i = WP_MACHINEGUN ; i < WP_NUM_WEAPONS ; i++ ) {
47 if ( ! ( weapons & ( 1 << i ) ) ) {
48 continue;
49 }
50 switch ( i ) {
51 case WP_ROCKET_LAUNCHER:
52 case WP_GRENADE_LAUNCHER:
53 case WP_RAILGUN:
54 case WP_SHOTGUN:
55#ifdef MISSIONPACK
56 case WP_PROX_LAUNCHER:
57#endif
58 total += cg.snap->ps.ammo[i] * 1000;
59 break;
60 default:
61 total += cg.snap->ps.ammo[i] * 200;
62 break;
63 }
64 if ( total >= 5000 ) {
65 cg.lowAmmoWarning = 0;
66 return;
67 }
68 }
69
70 previous = cg.lowAmmoWarning;
71
72 if ( total == 0 ) {
73 cg.lowAmmoWarning = 2;
74 } else {
75 cg.lowAmmoWarning = 1;
76 }
77
78 // play a sound on transitions
79 if ( cg.lowAmmoWarning != previous ) {
80 trap_S_StartLocalSound( cgs.media.noAmmoSound, CHAN_LOCAL_SOUND );
81 }
82}

84/*
85==============
86CG_DamageFeedback
87==============
88*/
89void CG_DamageFeedback( int yawByte, int pitchByte, int damage ) {
90 float left, front, up;
91 float kick;
92 int health;
93 float scale;
94 vec3_t dir;
95 vec3_t angles;
96 float dist;
97 float yaw, pitch;
98
99 // show the attacking player's head and name in corner
100 cg.attackerTime = cg.time;
101
102 // the lower on health you are, the greater the view kick will be
103 health = cg.snap->ps.stats[STAT_HEALTH];
104 if ( health < 40 ) {
105 scale = 1;
106 } else {
107 scale = 40.0 / health;
108 }
109 kick = damage * scale;
110
111 if (kick < 5)
112 kick = 5;
113 if (kick > 10)
114 kick = 10;
115
116 // if yaw and pitch are both 255, make the damage always centered (falling, etc)
117 if ( yawByte == 255 && pitchByte == 255 ) {
118 cg.damageX = 0;
119 cg.damageY = 0;
120 cg.v_dmg_roll = 0;
121 cg.v_dmg_pitch = -kick;
122 } else {
123 // positional
124 pitch = pitchByte / 255.0 * 360;
125 yaw = yawByte / 255.0 * 360;
126
127 angles[PITCH] = pitch;
128 angles[YAW] = yaw;
129 angles[ROLL] = 0;
130
131 AngleVectors( angles, dir, NULL, NULL );
132 VectorSubtract( vec3_origin, dir, dir );
133
134 front = DotProduct (dir, cg.refdef.viewaxis[0] );
135 left = DotProduct (dir, cg.refdef.viewaxis[1] );
136 up = DotProduct (dir, cg.refdef.viewaxis[2] );
137
138 dir[0] = front;
139 dir[1] = left;
140 dir[2] = 0;
141 dist = VectorLength( dir );
142 if ( dist < 0.1 ) {
143 dist = 0.1f;
144 }
145
146 cg.v_dmg_roll = kick * left;
147
148 cg.v_dmg_pitch = -kick * front;
149
150 if ( front <= 0.1 ) {
151 front = 0.1f;
152 }
153 cg.damageX = -left / front;
154 cg.damageY = up / dist;
155 }
156
157 // clamp the position
158 if ( cg.damageX > 1.0 ) {
159 cg.damageX = 1.0;
160 }
161 if ( cg.damageX < - 1.0 ) {
162 cg.damageX = -1.0;
163 }
164
165 if ( cg.damageY > 1.0 ) {
166 cg.damageY = 1.0;
167 }
168 if ( cg.damageY < - 1.0 ) {
169 cg.damageY = -1.0;
170 }
171
172 // don't let the screen flashes vary as much
173 if ( kick > 10 ) {
174 kick = 10;
175 }
176 cg.damageValue = kick;
177 cg.v_dmg_time = cg.time + DAMAGE_TIME;
178 cg.damageTime = cg.snap->serverTime;
179}

204/*
205==============
206CG_CheckPlayerstateEvents
207==============
208*/
209void CG_CheckPlayerstateEvents( playerState_t *ps, playerState_t *ops ) {
210 int i;
211 int event;
212 centity_t *cent;
213
214 if ( ps->externalEvent && ps->externalEvent != ops->externalEvent ) {
215 cent = &cg_entities[ ps->clientNum ];
216 cent->currentState.event = ps->externalEvent;
217 cent->currentState.eventParm = ps->externalEventParm;
218 CG_EntityEvent( cent, cent->lerpOrigin );
219 }
220
221 cent = &cg.predictedPlayerEntity; // cg_entities[ ps->clientNum ];
222 // go through the predictable events buffer
223 for ( i = ps->eventSequence - MAX_PS_EVENTS ; i < ps->eventSequence ; i++ ) {
224 // if we have a new predictable event
225 if ( i >= ops->eventSequence
226 // or the server told us to play another event instead of a predicted event we already issued
227 // or something the server told us changed our prediction causing a different event
228 || (i > ops->eventSequence - MAX_PS_EVENTS && ps->events[i & (MAX_PS_EVENTS-1)] != ops->events[i & (MAX_PS_EVENTS-1)]) ) {
229
230 event = ps->events[ i & (MAX_PS_EVENTS-1) ];
231 cent->currentState.event = event;
232 cent->currentState.eventParm = ps->eventParms[ i & (MAX_PS_EVENTS-1) ];
233 CG_EntityEvent( cent, cent->lerpOrigin );
234
235 cg.predictableEvents[ i & (MAX_PREDICTED_EVENTS-1) ] = event;
236
237 cg.eventSequence++;
238 }
239 }
240}

242/*
243==================
244CG_CheckChangedPredictableEvents
245==================
246*/
247void CG_CheckChangedPredictableEvents( playerState_t *ps ) {
248 int i;
249 int event;
250 centity_t *cent;
251
252 cent = &cg.predictedPlayerEntity;
253 for ( i = ps->eventSequence - MAX_PS_EVENTS ; i < ps->eventSequence ; i++ ) {
254 //
255 if (i >= cg.eventSequence) {
256 continue;
257 }
258 // if this event is not further back in than the maximum predictable events we remember
259 if (i > cg.eventSequence - MAX_PREDICTED_EVENTS) {
260 // if the new playerstate event is different from a previously predicted one
261 if ( ps->events[i & (MAX_PS_EVENTS-1)] != cg.predictableEvents[i & (MAX_PREDICTED_EVENTS-1) ] ) {
262
263 event = ps->events[ i & (MAX_PS_EVENTS-1) ];
264 cent->currentState.event = event;
265 cent->currentState.eventParm = ps->eventParms[ i & (MAX_PS_EVENTS-1) ];
266 CG_EntityEvent( cent, cent->lerpOrigin );
267
268 cg.predictableEvents[ i & (MAX_PREDICTED_EVENTS-1) ] = event;
269
270 if ( cg_showmiss.integer ) {
271 CG_Printf("WARNING: changed predicted event\n");
272 }
273 }
274 }
275 }

279==================
280pushReward
281==================
282*/
283static void pushReward(sfxHandle_t sfx, qhandle_t shader, int rewardCount) {
284 if (cg.rewardStack < (MAX_REWARDSTACK-1)) {
285 cg.rewardStack++;
286 cg.rewardSound[cg.rewardStack] = sfx;
287 cg.rewardShader[cg.rewardStack] = shader;
288 cg.rewardCount[cg.rewardStack] = rewardCount;
289 }

292/*
293==================
294CG_CheckLocalSounds
295==================
296*/
297void CG_CheckLocalSounds( playerState_t *ps, playerState_t *ops ) {
298 int highScore, reward;
299#ifdef MISSIONPACK
300 int health, armor;
301#endif
302 sfxHandle_t sfx;
303
304 // don't play the sounds if the player just changed teams
305 if ( ps->persistant[PERS_TEAM] != ops->persistant[PERS_TEAM] ) {
306 return;
307 }
308
309 // hit changes
310 if ( ps->persistant[PERS_HITS] > ops->persistant[PERS_HITS] ) {
311#ifdef MISSIONPACK
312 armor = ps->persistant[PERS_ATTACKEE_ARMOR] & 0xff;
313 health = ps->persistant[PERS_ATTACKEE_ARMOR] >> 8;
314 if (armor > 50 ) {
315 trap_S_StartLocalSound( cgs.media.hitSoundHighArmor, CHAN_LOCAL_SOUND );
316 } else if (armor || health > 100) {
317 trap_S_StartLocalSound( cgs.media.hitSoundLowArmor, CHAN_LOCAL_SOUND );
318 } else {
319 trap_S_StartLocalSound( cgs.media.hitSound, CHAN_LOCAL_SOUND );
320 }
321#else
322 trap_S_StartLocalSound( cgs.media.hitSound, CHAN_LOCAL_SOUND );
323#endif
324 } else if ( ps->persistant[PERS_HITS] < ops->persistant[PERS_HITS] ) {
325 trap_S_StartLocalSound( cgs.media.hitTeamSound, CHAN_LOCAL_SOUND );
326 }
327
328 // health changes of more than -1 should make pain sounds
329 if ( ps->stats[STAT_HEALTH] < ops->stats[STAT_HEALTH] - 1 ) {
330 if ( ps->stats[STAT_HEALTH] > 0 ) {
331 CG_PainEvent( &cg.predictedPlayerEntity, ps->stats[STAT_HEALTH] );
332 }
333 }
334
335
336 // if we are going into the intermission, don't start any voices
337 if ( cg.intermissionStarted ) {
338 return;
339 }
340
341 // reward sounds
342 reward = qfalse;
343 if (ps->persistant[PERS_CAPTURES] != ops->persistant[PERS_CAPTURES]) {
344 pushReward(cgs.media.captureAwardSound, cgs.media.medalCapture, ps->persistant[PERS_CAPTURES]);
345 reward = qtrue;
346 //Com_Printf("capture\n");
347 }
348 if (ps->persistant[PERS_IMPRESSIVE_COUNT] != ops->persistant[PERS_IMPRESSIVE_COUNT]) {
349#ifdef MISSIONPACK
350 if (ps->persistant[PERS_IMPRESSIVE_COUNT] == 1) {
351 sfx = cgs.media.firstImpressiveSound;
352 } else {
353 sfx = cgs.media.impressiveSound;
354 }
355#else
356 sfx = cgs.media.impressiveSound;
357#endif
358 pushReward(sfx, cgs.media.medalImpressive, ps->persistant[PERS_IMPRESSIVE_COUNT]);
359 reward = qtrue;
360 //Com_Printf("impressive\n");
361 }
362 if (ps->persistant[PERS_EXCELLENT_COUNT] != ops->persistant[PERS_EXCELLENT_COUNT]) {
363#ifdef MISSIONPACK
364 if (ps->persistant[PERS_EXCELLENT_COUNT] == 1) {
365 sfx = cgs.media.firstExcellentSound;
366 } else {
367 sfx = cgs.media.excellentSound;
368 }
369#else
370 sfx = cgs.media.excellentSound;
371#endif
372 pushReward(sfx, cgs.media.medalExcellent, ps->persistant[PERS_EXCELLENT_COUNT]);
373 reward = qtrue;
374 //Com_Printf("excellent\n");
375 }
376 if (ps->persistant[PERS_GAUNTLET_FRAG_COUNT] != ops->persistant[PERS_GAUNTLET_FRAG_COUNT]) {
377#ifdef MISSIONPACK
378 if (ps->persistant[PERS_GAUNTLET_FRAG_COUNT] == 1) {
379 sfx = cgs.media.firstHumiliationSound;
380 } else {
381 sfx = cgs.media.humiliationSound;
382 }
383#else
384 sfx = cgs.media.humiliationSound;
385#endif
386 pushReward(sfx, cgs.media.medalGauntlet, ps->persistant[PERS_GAUNTLET_FRAG_COUNT]);
387 reward = qtrue;
388 //Com_Printf("gauntlet frag\n");
389 }
390 if (ps->persistant[PERS_DEFEND_COUNT] != ops->persistant[PERS_DEFEND_COUNT]) {
391 pushReward(cgs.media.defendSound, cgs.media.medalDefend, ps->persistant[PERS_DEFEND_COUNT]);
392 reward = qtrue;
393 //Com_Printf("defend\n");
394 }
395 if (ps->persistant[PERS_ASSIST_COUNT] != ops->persistant[PERS_ASSIST_COUNT]) {
396 pushReward(cgs.media.assistSound, cgs.media.medalAssist, ps->persistant[PERS_ASSIST_COUNT]);
397 reward = qtrue;
398 //Com_Printf("assist\n");
399 }
400 // if any of the player event bits changed
401 if (ps->persistant[PERS_PLAYEREVENTS] != ops->persistant[PERS_PLAYEREVENTS]) {
402 if ((ps->persistant[PERS_PLAYEREVENTS] & PLAYEREVENT_DENIEDREWARD) !=
403 (ops->persistant[PERS_PLAYEREVENTS] & PLAYEREVENT_DENIEDREWARD)) {
404 trap_S_StartLocalSound( cgs.media.deniedSound, CHAN_ANNOUNCER );
405 }
406 else if ((ps->persistant[PERS_PLAYEREVENTS] & PLAYEREVENT_GAUNTLETREWARD) !=
407 (ops->persistant[PERS_PLAYEREVENTS] & PLAYEREVENT_GAUNTLETREWARD)) {
408 trap_S_StartLocalSound( cgs.media.humiliationSound, CHAN_ANNOUNCER );
409 }
410 else if ((ps->persistant[PERS_PLAYEREVENTS] & PLAYEREVENT_HOLYSHIT) !=
411 (ops->persistant[PERS_PLAYEREVENTS] & PLAYEREVENT_HOLYSHIT)) {
412 trap_S_StartLocalSound( cgs.media.holyShitSound, CHAN_ANNOUNCER );
413 }
414 reward = qtrue;
415 }
416
417 // check for flag pickup
418 if ( cgs.gametype > GT_TEAM ) {
419 if ((ps->powerups[PW_REDFLAG] != ops->powerups[PW_REDFLAG] && ps->powerups[PW_REDFLAG]) ||
420 (ps->powerups[PW_BLUEFLAG] != ops->powerups[PW_BLUEFLAG] && ps->powerups[PW_BLUEFLAG]) ||
421 (ps->powerups[PW_NEUTRALFLAG] != ops->powerups[PW_NEUTRALFLAG] && ps->powerups[PW_NEUTRALFLAG]) )
422 {
423 trap_S_StartLocalSound( cgs.media.youHaveFlagSound, CHAN_ANNOUNCER );
424 }
425 }
426
427 // lead changes
428 if (!reward) {
429 //
430 if ( !cg.warmup ) {
431 // never play lead changes during warmup
432 if ( ps->persistant[PERS_RANK] != ops->persistant[PERS_RANK] ) {
433 if ( cgs.gametype < GT_TEAM) {
434 if ( ps->persistant[PERS_RANK] == 0 ) {
435 CG_AddBufferedSound(cgs.media.takenLeadSound);
436 } else if ( ps->persistant[PERS_RANK] == RANK_TIED_FLAG ) {
437 CG_AddBufferedSound(cgs.media.tiedLeadSound);
438 } else if ( ( ops->persistant[PERS_RANK] & ~RANK_TIED_FLAG ) == 0 ) {
439 CG_AddBufferedSound(cgs.media.lostLeadSound);
440 }
441 }
442 }
443 }
444 }
445
446 // timelimit warnings
447 if ( cgs.timelimit > 0 ) {
448 int msec;
449
450 msec = cg.time - cgs.levelStartTime;
451 if ( !( cg.timelimitWarnings & 4 ) && msec > ( cgs.timelimit * 60 + 2 ) * 1000 ) {
452 cg.timelimitWarnings |= 1 | 2 | 4;
453 trap_S_StartLocalSound( cgs.media.suddenDeathSound, CHAN_ANNOUNCER );
454 }
455 else if ( !( cg.timelimitWarnings & 2 ) && msec > (cgs.timelimit - 1) * 60 * 1000 ) {
456 cg.timelimitWarnings |= 1 | 2;
457 trap_S_StartLocalSound( cgs.media.oneMinuteSound, CHAN_ANNOUNCER );
458 }
459 else if ( cgs.timelimit > 5 && !( cg.timelimitWarnings & 1 ) && msec > (cgs.timelimit - 5) * 60 * 1000 ) {
460 cg.timelimitWarnings |= 1;
461 trap_S_StartLocalSound( cgs.media.fiveMinuteSound, CHAN_ANNOUNCER );
462 }
463 }
464
465 // fraglimit warnings
466 if ( cgs.fraglimit > 0 && cgs.gametype < GT_CTF) {
467 highScore = cgs.scores1;
468
469 if (cgs.gametype == GT_TEAM && cgs.scores2 > highScore) {
470 highScore = cgs.scores2;
471 }
472
473 if ( !( cg.fraglimitWarnings & 4 ) && highScore == (cgs.fraglimit - 1) ) {
474 cg.fraglimitWarnings |= 1 | 2 | 4;
475 CG_AddBufferedSound(cgs.media.oneFragSound);
476 }
477 else if ( cgs.fraglimit > 2 && !( cg.fraglimitWarnings & 2 ) && highScore == (cgs.fraglimit - 2) ) {
478 cg.fraglimitWarnings |= 1 | 2;
479 CG_AddBufferedSound(cgs.media.twoFragSound);
480 }
481 else if ( cgs.fraglimit > 3 && !( cg.fraglimitWarnings & 1 ) && highScore == (cgs.fraglimit - 3) ) {
482 cg.fraglimitWarnings |= 1;
483 CG_AddBufferedSound(cgs.media.threeFragSound);
484 }
485 }
486}

488/*
489===============
490CG_TransitionPlayerState
491
492===============
493*/
494void CG_TransitionPlayerState( playerState_t *ps, playerState_t *ops ) {
495 // check for changing follow mode
496 if ( ps->clientNum != ops->clientNum ) {
497 cg.thisFrameTeleport = qtrue;
498 // make sure we don't get any unwanted transition effects
499 *ops = *ps;
500 }
501
502 // damage events (player is getting wounded)
503 if ( ps->damageEvent != ops->damageEvent && ps->damageCount ) {
504 CG_DamageFeedback( ps->damageYaw, ps->damagePitch, ps->damageCount );
505 }
506
507 // respawning
508 if ( ps->persistant[PERS_SPAWN_COUNT] != ops->persistant[PERS_SPAWN_COUNT] ) {
509 CG_Respawn();
510 }
511
512 if ( cg.mapRestart ) {
513 CG_Respawn();
514 cg.mapRestart = qfalse;
515 }
516
517 if ( cg.snap->ps.pm_type != PM_INTERMISSION
518 && ps->persistant[PERS_TEAM] != TEAM_SPECTATOR ) {
519 CG_CheckLocalSounds( ps, ops );
520 }
521
522 // check for going low on ammo
523 CG_CheckAmmo();
524
525 // run events
526 CG_CheckPlayerstateEvents( ps, ops );
527
528 // smooth the ducking viewheight change
529 if ( ps->viewheight != ops->viewheight ) {
530 cg.duckChange = ps->viewheight - ops->viewheight;
531 cg.duckTime = cg.time;
532 }

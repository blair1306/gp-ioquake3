43/* The audio callback. All the magic happens here. */
44static int dmapos = 0;
45static int dmasize = 0;
46
47/*
48===============
49SNDDMA_AudioCallback
50===============
51*/
52static void SNDDMA_AudioCallback(void *userdata, Uint8 *stream, int len)
53{
54 int pos = (dmapos * (dma.samplebits/8));
55 if (pos >= dmasize)
56 dmapos = pos = 0;
57
58 if (!snd_inited) /* shouldn't happen, but just in case... */
59 {
60 memset(stream, '\0', len);
61 return;
62 }
63 else
64 {
65 int tobufend = dmasize - pos; /* bytes to buffer's end. */
66 int len1 = len;
67 int len2 = 0;
68
69 if (len1 > tobufend)
70 {
71 len1 = tobufend;
72 len2 = len - len1;
73 }
74 memcpy(stream, dma.buffer + pos, len1);
75 if (len2 <= 0)
76 dmapos += (len1 / (dma.samplebits/8));
77 else /* wraparound? */
78 {
79 memcpy(stream+len1, dma.buffer, len2);
80 dmapos = (len2 / (dma.samplebits/8));
81 }
82 }
83
84 if (dmapos >= dmasize)
85 dmapos = 0;
86}

88static struct
89{
90 Uint16 enumFormat;
91 char *stringFormat;
92} formatToStringTable[ ] =
93{
94 { AUDIO_U8, "AUDIO_U8" },
95 { AUDIO_S8, "AUDIO_S8" },
96 { AUDIO_U16LSB, "AUDIO_U16LSB" },
97 { AUDIO_S16LSB, "AUDIO_S16LSB" },
98 { AUDIO_U16MSB, "AUDIO_U16MSB" },
99 { AUDIO_S16MSB, "AUDIO_S16MSB" }
100};
101
102static int formatToStringTableSize = ARRAY_LEN( formatToStringTable );
103
104/*
105===============
106SNDDMA_PrintAudiospec
107===============
108*/
109static void SNDDMA_PrintAudiospec(const char *str, const SDL_AudioSpec *spec)
110{
111 int i;
112 char *fmt = NULL;
113
114 Com_Printf("%s:\n", str);
115
116 for( i = 0; i < formatToStringTableSize; i++ ) {
117 if( spec->format == formatToStringTable[ i ].enumFormat ) {
118 fmt = formatToStringTable[ i ].stringFormat;
119 }
120 }
121
122 if( fmt ) {
123 Com_Printf( " Format: %s\n", fmt );
124 } else {
125 Com_Printf( " Format: " S_COLOR_RED "UNKNOWN\n");
126 }
127
128 Com_Printf( " Freq: %d\n", (int) spec->freq );
129 Com_Printf( " Samples: %d\n", (int) spec->samples );
130 Com_Printf( " Channels: %d\n", (int) spec->channels );
131}

133/*
134===============
135SNDDMA_Init
136===============
137*/
138qboolean SNDDMA_Init(void)
139{
140 SDL_AudioSpec desired;
141 SDL_AudioSpec obtained;
142 int tmp;
143
144 if (snd_inited)
145 return qtrue;
146
147 if (!s_sdlBits) {
148 s_sdlBits = Cvar_Get("s_sdlBits", "16", CVAR_ARCHIVE);
149 s_sdlSpeed = Cvar_Get("s_sdlSpeed", "0", CVAR_ARCHIVE);
150 s_sdlChannels = Cvar_Get("s_sdlChannels", "2", CVAR_ARCHIVE);
151 s_sdlDevSamps = Cvar_Get("s_sdlDevSamps", "0", CVAR_ARCHIVE);
152 s_sdlMixSamps = Cvar_Get("s_sdlMixSamps", "0", CVAR_ARCHIVE);
153 }
154
155 Com_Printf( "SDL_Init( SDL_INIT_AUDIO )... " );
156
157 if (!SDL_WasInit(SDL_INIT_AUDIO))
158 {
159 if (SDL_Init(SDL_INIT_AUDIO) != 0)
160 {
161 Com_Printf( "FAILED (%s)\n", SDL_GetError( ) );
162 return qfalse;
163 }
164 }
165
166 Com_Printf( "OK\n" );
167
168 Com_Printf( "SDL audio driver is \"%s\".\n", SDL_GetCurrentAudioDriver( ) );
169
170 memset(&desired, '\0', sizeof (desired));
171 memset(&obtained, '\0', sizeof (obtained));
172
173 tmp = ((int) s_sdlBits->value);
174 if ((tmp != 16) && (tmp != 8))
175 tmp = 16;
176
177 desired.freq = (int) s_sdlSpeed->value;
178 if(!desired.freq) desired.freq = 22050;
179 desired.format = ((tmp == 16) ? AUDIO_S16SYS : AUDIO_U8);
180
181 // I dunno if this is the best idea, but I'll give it a try...
182 // should probably check a cvar for this...
183 if (s_sdlDevSamps->value)
184 desired.samples = s_sdlDevSamps->value;
185 else
186 {
187 // just pick a sane default.
188 if (desired.freq <= 11025)
189 desired.samples = 256;
190 else if (desired.freq <= 22050)
191 desired.samples = 512;
192 else if (desired.freq <= 44100)
193 desired.samples = 1024;
194 else
195 desired.samples = 2048; // (*shrug*)
196 }
197
198 desired.channels = (int) s_sdlChannels->value;
199 desired.callback = SNDDMA_AudioCallback;
200
201 if (SDL_OpenAudio(&desired, &obtained) == -1)
202 {
203 Com_Printf("SDL_OpenAudio() failed: %s\n", SDL_GetError());
204 SDL_QuitSubSystem(SDL_INIT_AUDIO);
205 return qfalse;
206 }
207
208 SNDDMA_PrintAudiospec("SDL_AudioSpec", &obtained);
209
210 // dma.samples needs to be big, or id's mixer will just refuse to
211 // work at all; we need to keep it significantly bigger than the
212 // amount of SDL callback samples, and just copy a little each time
213 // the callback runs.
214 // 32768 is what the OSS driver filled in here on my system. I don't
215 // know if it's a good value overall, but at least we know it's
216 // reasonable...this is why I let the user override.
217 tmp = s_sdlMixSamps->value;
218 if (!tmp)
219 tmp = (obtained.samples * obtained.channels) * 10;
220
221 if (tmp & (tmp - 1)) // not a power of two? Seems to confuse something.
222 {
223 int val = 1;
224 while (val < tmp)
225 val <<= 1;
226
227 tmp = val;
228 }
229
230 dmapos = 0;
231 dma.samplebits = obtained.format & 0xFF; // first byte of format is bits.
232 dma.channels = obtained.channels;
233 dma.samples = tmp;
234 dma.submission_chunk = 1;
235 dma.speed = obtained.freq;
236 dmasize = (dma.samples * (dma.samplebits/8));
237 dma.buffer = calloc(1, dmasize);
238
239 Com_Printf("Starting SDL audio callback...\n");
240 SDL_PauseAudio(0); // start callback.
241
242 Com_Printf("SDL audio initialized.\n");
243 snd_inited = qtrue;
244 return qtrue;
245}

257/*
258===============
259SNDDMA_Shutdown
260===============
261*/
262void SNDDMA_Shutdown(void)
263{
264 Com_Printf("Closing SDL audio device...\n");
265 SDL_PauseAudio(1);
266 SDL_CloseAudio();
267 SDL_QuitSubSystem(SDL_INIT_AUDIO);
268 free(dma.buffer);
269 dma.buffer = NULL;
270 dmapos = dmasize = 0;
271 snd_inited = qfalse;
272 Com_Printf("SDL audio device shut down.\n");
273}

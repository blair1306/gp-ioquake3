60/*
61===============
62IN_PrintKey
63===============
64*/
65static void IN_PrintKey( const SDL_Keysym *keysym, keyNum_t key, qboolean down )
66{
67 if( down )
68 Com_Printf( "+ " );
69 else
70 Com_Printf( " " );
71
72 Com_Printf( "Scancode: 0x%02x(%s) Sym: 0x%02x(%s)",
73 keysym->scancode, SDL_GetScancodeName( keysym->scancode ),
74 keysym->sym, SDL_GetKeyName( keysym->sym ) );
75
76 if( keysym->mod & KMOD_LSHIFT ) Com_Printf( " KMOD_LSHIFT" );
77 if( keysym->mod & KMOD_RSHIFT ) Com_Printf( " KMOD_RSHIFT" );
78 if( keysym->mod & KMOD_LCTRL ) Com_Printf( " KMOD_LCTRL" );
79 if( keysym->mod & KMOD_RCTRL ) Com_Printf( " KMOD_RCTRL" );
80 if( keysym->mod & KMOD_LALT ) Com_Printf( " KMOD_LALT" );
81 if( keysym->mod & KMOD_RALT ) Com_Printf( " KMOD_RALT" );
82 if( keysym->mod & KMOD_LGUI ) Com_Printf( " KMOD_LGUI" );
83 if( keysym->mod & KMOD_RGUI ) Com_Printf( " KMOD_RGUI" );
84 if( keysym->mod & KMOD_NUM ) Com_Printf( " KMOD_NUM" );
85 if( keysym->mod & KMOD_CAPS ) Com_Printf( " KMOD_CAPS" );
86 if( keysym->mod & KMOD_MODE ) Com_Printf( " KMOD_MODE" );
87 if( keysym->mod & KMOD_RESERVED ) Com_Printf( " KMOD_RESERVED" );
88
89 Com_Printf( " Q:0x%02x(%s)\n", key, Key_KeynumToString( key ) );
90}

94/*
95===============
96IN_IsConsoleKey
97
98TODO: If the SDL_Scancode situation improves, use it instead of
99 both of these methods
100===============
101*/
102static qboolean IN_IsConsoleKey( keyNum_t key, int character )
103{
104 typedef struct consoleKey_s
105 {
106 enum
107 {
108 QUAKE_KEY,
109 CHARACTER
110 } type;
111
112 union
113 {
114 keyNum_t key;
115 int character;
116 } u;
117 } consoleKey_t;
118
119 static consoleKey_t consoleKeys[ MAX_CONSOLE_KEYS ];
120 static int numConsoleKeys = 0;
121 int i;
122
123 // Only parse the variable when it changes
124 if( cl_consoleKeys->modified )
125 {
126 char *text_p, *token;
127
128 cl_consoleKeys->modified = qfalse;
129 text_p = cl_consoleKeys->string;
130 numConsoleKeys = 0;
131
132 while( numConsoleKeys < MAX_CONSOLE_KEYS )
133 {
134 consoleKey_t *c = &consoleKeys[ numConsoleKeys ];
135 int charCode = 0;
136
137 token = COM_Parse( &text_p );
138 if( !token[ 0 ] )
139 break;
140
141 if( strlen( token ) == 4 )
142 charCode = Com_HexStrToInt( token );
143
144 if( charCode > 0 )
145 {
146 c->type = CHARACTER;
147 c->u.character = charCode;
148 }
149 else
150 {
151 c->type = QUAKE_KEY;
152 c->u.key = Key_StringToKeynum( token );
153
154 // 0 isn't a key
155 if( c->u.key <= 0 )
156 continue;
157 }
158
159 numConsoleKeys++;
160 }
161 }
162
163 // If the character is the same as the key, prefer the character
164 if( key == character )
165 key = 0;
166
167 for( i = 0; i < numConsoleKeys; i++ )
168 {
169 consoleKey_t *c = &consoleKeys[ i ];
170
171 switch( c->type )
172 {
173 case QUAKE_KEY:
174 if( key && c->u.key == key )
175 return qtrue;
176 break;
177
178 case CHARACTER:
179 if( c->u.character == character )
180 return qtrue;
181 break;
182 }
183 }
184
185 return qfalse;
186}

188/*
189===============
190IN_TranslateSDLToQ3Key
191===============
192*/
193static keyNum_t IN_TranslateSDLToQ3Key( SDL_Keysym *keysym, qboolean down )
194{
195 keyNum_t key = 0;
196
197 if( keysym->scancode >= SDL_SCANCODE_1 && keysym->scancode <= SDL_SCANCODE_0 )
198 {
199 // Always map the number keys as such even if they actually map
200 // to other characters (eg, "1" is "&" on an AZERTY keyboard).
201 // This is required for SDL before 2.0.6, except on Windows
202 // which already had this behavior.
203 if( keysym->scancode == SDL_SCANCODE_0 )
204 key = '0';
205 else
206 key = '1' + keysym->scancode - SDL_SCANCODE_1;
207 }
208 else if( keysym->sym >= SDLK_SPACE && keysym->sym < SDLK_DELETE )
209 {
210 // These happen to match the ASCII chars
211 key = (int)keysym->sym;
212 }
213 else
214 {
215 switch( keysym->sym )
216 {
217 case SDLK_PAGEUP: key = K_PGUP; break;
218 case SDLK_KP_9: key = K_KP_PGUP; break;
219 case SDLK_PAGEDOWN: key = K_PGDN; break;
220 case SDLK_KP_3: key = K_KP_PGDN; break;
221 case SDLK_KP_7: key = K_KP_HOME; break;
222 case SDLK_HOME: key = K_HOME; break;
223 case SDLK_KP_1: key = K_KP_END; break;
224 case SDLK_END: key = K_END; break;
225 case SDLK_KP_4: key = K_KP_LEFTARROW; break;
226 case SDLK_LEFT: key = K_LEFTARROW; break;
227 case SDLK_KP_6: key = K_KP_RIGHTARROW; break;
228 case SDLK_RIGHT: key = K_RIGHTARROW; break;
229 case SDLK_KP_2: key = K_KP_DOWNARROW; break;
230 case SDLK_DOWN: key = K_DOWNARROW; break;
231 case SDLK_KP_8: key = K_KP_UPARROW; break;
232 case SDLK_UP: key = K_UPARROW; break;
233 case SDLK_ESCAPE: key = K_ESCAPE; break;
234 case SDLK_KP_ENTER: key = K_KP_ENTER; break;
235 case SDLK_RETURN: key = K_ENTER; break;
236 case SDLK_TAB: key = K_TAB; break;
237 case SDLK_F1: key = K_F1; break;
238 case SDLK_F2: key = K_F2; break;
239 case SDLK_F3: key = K_F3; break;
240 case SDLK_F4: key = K_F4; break;
241 case SDLK_F5: key = K_F5; break;
242 case SDLK_F6: key = K_F6; break;
243 case SDLK_F7: key = K_F7; break;
244 case SDLK_F8: key = K_F8; break;
245 case SDLK_F9: key = K_F9; break;
246 case SDLK_F10: key = K_F10; break;
247 case SDLK_F11: key = K_F11; break;
248 case SDLK_F12: key = K_F12; break;
249 case SDLK_F13: key = K_F13; break;
250 case SDLK_F14: key = K_F14; break;
251 case SDLK_F15: key = K_F15; break;
252
253 case SDLK_BACKSPACE: key = K_BACKSPACE; break;
254 case SDLK_KP_PERIOD: key = K_KP_DEL; break;
255 case SDLK_DELETE: key = K_DEL; break;
256 case SDLK_PAUSE: key = K_PAUSE; break;
257
258 case SDLK_LSHIFT:
259 case SDLK_RSHIFT: key = K_SHIFT; break;
260
261 case SDLK_LCTRL:
262 case SDLK_RCTRL: key = K_CTRL; break;
263
264#ifdef __APPLE__
265 case SDLK_RGUI:
266 case SDLK_LGUI: key = K_COMMAND; break;
267#else
268 case SDLK_RGUI:
269 case SDLK_LGUI: key = K_SUPER; break;
270#endif
271
272 case SDLK_RALT:
273 case SDLK_LALT: key = K_ALT; break;
274
275 case SDLK_KP_5: key = K_KP_5; break;
276 case SDLK_INSERT: key = K_INS; break;
277 case SDLK_KP_0: key = K_KP_INS; break;
278 case SDLK_KP_MULTIPLY: key = K_KP_STAR; break;
279 case SDLK_KP_PLUS: key = K_KP_PLUS; break;
280 case SDLK_KP_MINUS: key = K_KP_MINUS; break;
281 case SDLK_KP_DIVIDE: key = K_KP_SLASH; break;
282
283 case SDLK_MODE: key = K_MODE; break;
284 case SDLK_HELP: key = K_HELP; break;
285 case SDLK_PRINTSCREEN: key = K_PRINT; break;
286 case SDLK_SYSREQ: key = K_SYSREQ; break;
287 case SDLK_MENU: key = K_MENU; break;
288 case SDLK_APPLICATION: key = K_MENU; break;
289 case SDLK_POWER: key = K_POWER; break;
290 case SDLK_UNDO: key = K_UNDO; break;
291 case SDLK_SCROLLLOCK: key = K_SCROLLOCK; break;
292 case SDLK_NUMLOCKCLEAR: key = K_KP_NUMLOCK; break;
293 case SDLK_CAPSLOCK: key = K_CAPSLOCK; break;
294
295 default:
296 if( !( keysym->sym & SDLK_SCANCODE_MASK ) && keysym->scancode <= 95 )
297 {
298 // Map Unicode characters to 95 world keys using the key's scan code.
299 // FIXME: There aren't enough world keys to cover all the scancodes.
300 // Maybe create a map of scancode to quake key at start up and on
301 // key map change; allocate world key numbers as needed similar
302 // to SDL 1.2.
303 key = K_WORLD_0 + (int)keysym->scancode;
304 }
305 break;
306 }
307 }
308
309 if( in_keyboardDebug->integer )
310 IN_PrintKey( keysym, key, down );
311
312 if( IN_IsConsoleKey( key, 0 ) )
313 {
314 // Console keys can't be bound or generate characters
315 key = K_CONSOLE;
316 }
317
318 return key;
319}

321/*
322===============
323IN_GobbleMotionEvents
324===============
325*/
326static void IN_GobbleMotionEvents( void )
327{
328 SDL_Event dummy[ 1 ];
329 int val = 0;
330
331 // Gobble any mouse motion events
332 SDL_PumpEvents( );
333 while( ( val = SDL_PeepEvents( dummy, 1, SDL_GETEVENT,
334 SDL_MOUSEMOTION, SDL_MOUSEMOTION ) ) > 0 ) { }
335
336 if ( val < 0 )
337 Com_Printf( "IN_GobbleMotionEvents failed: %s\n", SDL_GetError( ) );
338}

340/*
341===============
342IN_ActivateMouse
343===============
344*/
345static void IN_ActivateMouse( void )
346{
347 if (!mouseAvailable || !SDL_WasInit( SDL_INIT_VIDEO ) )
348 return;
349
350 if( !mouseActive )
351 {
352 SDL_SetRelativeMouseMode( SDL_TRUE );
353 SDL_SetWindowGrab( SDL_window, SDL_TRUE );
354
355 IN_GobbleMotionEvents( );
356 }
357
358 // in_nograb makes no sense in fullscreen mode
359 if( !Cvar_VariableIntegerValue("r_fullscreen") )
360 {
361 if( in_nograb->modified || !mouseActive )
362 {
363 if( in_nograb->integer )
364 SDL_SetWindowGrab( SDL_window, SDL_FALSE );
365 else
366 SDL_SetWindowGrab( SDL_window, SDL_TRUE );
367
368 in_nograb->modified = qfalse;
369 }
370 }
371
372 mouseActive = qtrue;
373}
374
375/*
376===============
377IN_DeactivateMouse
378===============
379*/
380static void IN_DeactivateMouse( void )
381{
382 if( !SDL_WasInit( SDL_INIT_VIDEO ) )
383 return;
384
385 // Always show the cursor when the mouse is disabled,
386 // but not when fullscreen
387 if( !Cvar_VariableIntegerValue("r_fullscreen") )
388 SDL_ShowCursor( 1 );
389
390 if( !mouseAvailable )
391 return;
392
393 if( mouseActive )
394 {
395 IN_GobbleMotionEvents( );
396
397 SDL_SetWindowGrab( SDL_window, SDL_FALSE );
398 SDL_SetRelativeMouseMode( SDL_FALSE );
399
400 // Don't warp the mouse unless the cursor is within the window
401 if( SDL_GetWindowFlags( SDL_window ) & SDL_WINDOW_MOUSE_FOCUS )
402 SDL_WarpMouseInWindow( SDL_window, cls.glconfig.vidWidth / 2, cls.glconfig.vidHeight / 2 );
403
404 mouseActive = qfalse;
405 }
406}

443/*
444===============
445IN_InitJoystick
446===============
447*/
448static void IN_InitJoystick( void )
449{
450 int i = 0;
451 int total = 0;
452 char buf[16384] = "";
453
454 if (gamepad)
455 SDL_GameControllerClose(gamepad);
456
457 if (stick != NULL)
458 SDL_JoystickClose(stick);
459
460 stick = NULL;
461 gamepad = NULL;
462 memset(&stick_state, '\0', sizeof (stick_state));
463
464 // SDL 2.0.4 requires SDL_INIT_JOYSTICK to be initialized separately from
465 // SDL_INIT_GAMECONTROLLER for SDL_JoystickOpen() to work correctly,
466 // despite https://wiki.libsdl.org/SDL_Init (retrieved 2016-08-16)
467 // indicating SDL_INIT_JOYSTICK should be initialized automatically.
468 if (!SDL_WasInit(SDL_INIT_JOYSTICK))
469 {
470 Com_DPrintf("Calling SDL_Init(SDL_INIT_JOYSTICK)...\n");
471 if (SDL_Init(SDL_INIT_JOYSTICK) != 0)
472 {
473 Com_DPrintf("SDL_Init(SDL_INIT_JOYSTICK) failed: %s\n", SDL_GetError());
474 return;
475 }
476 Com_DPrintf("SDL_Init(SDL_INIT_JOYSTICK) passed.\n");
477 }
478
479 if (!SDL_WasInit(SDL_INIT_GAMECONTROLLER))
480 {
481 Com_DPrintf("Calling SDL_Init(SDL_INIT_GAMECONTROLLER)...\n");
482 if (SDL_Init(SDL_INIT_GAMECONTROLLER) != 0)
483 {
484 Com_DPrintf("SDL_Init(SDL_INIT_GAMECONTROLLER) failed: %s\n", SDL_GetError());
485 return;
486 }
487 Com_DPrintf("SDL_Init(SDL_INIT_GAMECONTROLLER) passed.\n");
488 }
489
490 total = SDL_NumJoysticks();
491 Com_DPrintf("%d possible joysticks\n", total);
492
493 // Print list and build cvar to allow ui to select joystick.
494 for (i = 0; i < total; i++)
495 {
496 Q_strcat(buf, sizeof(buf), SDL_JoystickNameForIndex(i));
497 Q_strcat(buf, sizeof(buf), "\n");
498 }
499
500 Cvar_Get( "in_availableJoysticks", buf, CVAR_ROM );
501
502 if( !in_joystick->integer ) {
503 Com_DPrintf( "Joystick is not active.\n" );
504 SDL_QuitSubSystem(SDL_INIT_GAMECONTROLLER);
505 return;
506 }
507
508 in_joystickNo = Cvar_Get( "in_joystickNo", "0", CVAR_ARCHIVE );
509 if( in_joystickNo->integer < 0 || in_joystickNo->integer >= total )
510 Cvar_Set( "in_joystickNo", "0" );
511
512 in_joystickUseAnalog = Cvar_Get( "in_joystickUseAnalog", "0", CVAR_ARCHIVE );
513
514 stick = SDL_JoystickOpen( in_joystickNo->integer );
515
516 if (stick == NULL) {
517 Com_DPrintf( "No joystick opened: %s\n", SDL_GetError() );
518 return;
519 }
520
521 if (SDL_IsGameController(in_joystickNo->integer))
522 gamepad = SDL_GameControllerOpen(in_joystickNo->integer);
523
524 Com_DPrintf( "Joystick %d opened\n", in_joystickNo->integer );
525 Com_DPrintf( "Name: %s\n", SDL_JoystickNameForIndex(in_joystickNo->integer) );
526 Com_DPrintf( "Axes: %d\n", SDL_JoystickNumAxes(stick) );
527 Com_DPrintf( "Hats: %d\n", SDL_JoystickNumHats(stick) );
528 Com_DPrintf( "Buttons: %d\n", SDL_JoystickNumButtons(stick) );
529 Com_DPrintf( "Balls: %d\n", SDL_JoystickNumBalls(stick) );
530 Com_DPrintf( "Use Analog: %s\n", in_joystickUseAnalog->integer ? "Yes" : "No" );
531 Com_DPrintf( "Is gamepad: %s\n", gamepad ? "Yes" : "No" );
532
533 SDL_JoystickEventState(SDL_QUERY);
534 SDL_GameControllerEventState(SDL_QUERY);
535}

537/*
538===============
539IN_ShutdownJoystick
540===============
541*/
542static void IN_ShutdownJoystick( void )
543{
544 if ( !SDL_WasInit( SDL_INIT_GAMECONTROLLER ) )
545 return;
546
547 if ( !SDL_WasInit( SDL_INIT_JOYSTICK ) )
548 return;
549
550 if (gamepad)
551 {
552 SDL_GameControllerClose(gamepad);
553 gamepad = NULL;
554 }
555
556 if (stick)
557 {
558 SDL_JoystickClose(stick);
559 stick = NULL;
560 }
561
562 SDL_QuitSubSystem(SDL_INIT_GAMECONTROLLER);
563 SDL_QuitSubSystem(SDL_INIT_JOYSTICK);
564}

567static qboolean KeyToAxisAndSign(int keynum, int *outAxis, int *outSign)
568{
569 char *bind;
570
571 if (!keynum)
572 return qfalse;
573
574 bind = Key_GetBinding(keynum);
575
576 if (!bind || *bind != '+')
577 return qfalse;
578
579 *outSign = 0;
580
581 if (Q_stricmp(bind, "+forward") == 0)
582 {
583 *outAxis = j_forward_axis->integer;
584 *outSign = j_forward->value > 0.0f ? 1 : -1;
585 }
586 else if (Q_stricmp(bind, "+back") == 0)
587 {
588 *outAxis = j_forward_axis->integer;
589 *outSign = j_forward->value > 0.0f ? -1 : 1;
590 }
591 else if (Q_stricmp(bind, "+moveleft") == 0)
592 {
593 *outAxis = j_side_axis->integer;
594 *outSign = j_side->value > 0.0f ? -1 : 1;
595 }
596 else if (Q_stricmp(bind, "+moveright") == 0)
597 {
598 *outAxis = j_side_axis->integer;
599 *outSign = j_side->value > 0.0f ? 1 : -1;
600 }
601 else if (Q_stricmp(bind, "+lookup") == 0)
602 {
603 *outAxis = j_pitch_axis->integer;
604 *outSign = j_pitch->value > 0.0f ? -1 : 1;
605 }
606 else if (Q_stricmp(bind, "+lookdown") == 0)
607 {
608 *outAxis = j_pitch_axis->integer;
609 *outSign = j_pitch->value > 0.0f ? 1 : -1;
610 }
611 else if (Q_stricmp(bind, "+left") == 0)
612 {
613 *outAxis = j_yaw_axis->integer;
614 *outSign = j_yaw->value > 0.0f ? 1 : -1;
615 }
616 else if (Q_stricmp(bind, "+right") == 0)
617 {
618 *outAxis = j_yaw_axis->integer;
619 *outSign = j_yaw->value > 0.0f ? -1 : 1;
620 }
621 else if (Q_stricmp(bind, "+moveup") == 0)
622 {
623 *outAxis = j_up_axis->integer;
624 *outSign = j_up->value > 0.0f ? 1 : -1;
625 }
626 else if (Q_stricmp(bind, "+movedown") == 0)
627 {
628 *outAxis = j_up_axis->integer;
629 *outSign = j_up->value > 0.0f ? -1 : 1;
630 }
631
632 return *outSign != 0;
633}

635/*
636===============
637IN_GamepadMove
638===============
639*/
640static void IN_GamepadMove( void )
641{
642 int i;
643 int translatedAxes[MAX_JOYSTICK_AXIS];
644 qboolean translatedAxesSet[MAX_JOYSTICK_AXIS];
645
646 SDL_GameControllerUpdate();
647
648 // check buttons
649 for (i = 0; i < SDL_CONTROLLER_BUTTON_MAX; i++)
650 {
651 qboolean pressed = SDL_GameControllerGetButton(gamepad, SDL_CONTROLLER_BUTTON_A + i);
652 if (pressed != stick_state.buttons[i])
653 {
654 Com_QueueEvent(in_eventTime, SE_KEY, K_PAD0_A + i, pressed, 0, NULL);
655 stick_state.buttons[i] = pressed;
656 }
657 }
658
659 // must defer translated axes until all real axes are processed
660 // must be done this way to prevent a later mapped axis from zeroing out a previous one
661 if (in_joystickUseAnalog->integer)
662 {
663 for (i = 0; i < MAX_JOYSTICK_AXIS; i++)
664 {
665 translatedAxes[i] = 0;
666 translatedAxesSet[i] = qfalse;
667 }
668 }
669
670 // check axes
671 for (i = 0; i < SDL_CONTROLLER_AXIS_MAX; i++)
672 {
673 int axis = SDL_GameControllerGetAxis(gamepad, SDL_CONTROLLER_AXIS_LEFTX + i);
674 int oldAxis = stick_state.oldaaxes[i];
675
676 // Smoothly ramp from dead zone to maximum value
677 float f = ((float)abs(axis) / 32767.0f - in_joystickThreshold->value) / (1.0f - in_joystickThreshold->value);
678
679 if (f < 0.0f)
680 f = 0.0f;
681
682 axis = (int)(32767 * ((axis < 0) ? -f : f));
683
684 if (axis != oldAxis)
685 {
686 const int negMap[SDL_CONTROLLER_AXIS_MAX] = { K_PAD0_LEFTSTICK_LEFT, K_PAD0_LEFTSTICK_UP, K_PAD0_RIGHTSTICK_LEFT, K_PAD0_RIGHTSTICK_UP, 0, 0 };
687 const int posMap[SDL_CONTROLLER_AXIS_MAX] = { K_PAD0_LEFTSTICK_RIGHT, K_PAD0_LEFTSTICK_DOWN, K_PAD0_RIGHTSTICK_RIGHT, K_PAD0_RIGHTSTICK_DOWN, K_PAD0_LEFTTRIGGER, K_PAD0_RIGHTTRIGGER };
688
689 qboolean posAnalog = qfalse, negAnalog = qfalse;
690 int negKey = negMap[i];
691 int posKey = posMap[i];
692
693 if (in_joystickUseAnalog->integer)
694 {
695 int posAxis = 0, posSign = 0, negAxis = 0, negSign = 0;
696
697 // get axes and axes signs for keys if available
698 posAnalog = KeyToAxisAndSign(posKey, &posAxis, &posSign);
699 negAnalog = KeyToAxisAndSign(negKey, &negAxis, &negSign);
700
701 // positive to negative/neutral -> keyup if axis hasn't yet been set
702 if (posAnalog && !translatedAxesSet[posAxis] && oldAxis > 0 && axis <= 0)
703 {
704 translatedAxes[posAxis] = 0;
705 translatedAxesSet[posAxis] = qtrue;
706 }
707
708 // negative to positive/neutral -> keyup if axis hasn't yet been set
709 if (negAnalog && !translatedAxesSet[negAxis] && oldAxis < 0 && axis >= 0)
710 {
711 translatedAxes[negAxis] = 0;
712 translatedAxesSet[negAxis] = qtrue;
713 }
714
715 // negative/neutral to positive -> keydown
716 if (posAnalog && axis > 0)
717 {
718 translatedAxes[posAxis] = axis * posSign;
719 translatedAxesSet[posAxis] = qtrue;
720 }
721
722 // positive/neutral to negative -> keydown
723 if (negAnalog && axis < 0)
724 {
725 translatedAxes[negAxis] = -axis * negSign;
726 translatedAxesSet[negAxis] = qtrue;
727 }
728 }
729
730 // keyups first so they get overridden by keydowns later
731
732 // positive to negative/neutral -> keyup
733 if (!posAnalog && posKey && oldAxis > 0 && axis <= 0)
734 Com_QueueEvent(in_eventTime, SE_KEY, posKey, qfalse, 0, NULL);
735
736 // negative to positive/neutral -> keyup
737 if (!negAnalog && negKey && oldAxis < 0 && axis >= 0)
738 Com_QueueEvent(in_eventTime, SE_KEY, negKey, qfalse, 0, NULL);
739
740 // negative/neutral to positive -> keydown
741 if (!posAnalog && posKey && oldAxis <= 0 && axis > 0)
742 Com_QueueEvent(in_eventTime, SE_KEY, posKey, qtrue, 0, NULL);
743
744 // positive/neutral to negative -> keydown
745 if (!negAnalog && negKey && oldAxis >= 0 && axis < 0)
746 Com_QueueEvent(in_eventTime, SE_KEY, negKey, qtrue, 0, NULL);
747
748 stick_state.oldaaxes[i] = axis;
749 }
750 }
751
752 // set translated axes
753 if (in_joystickUseAnalog->integer)
754 {
755 for (i = 0; i < MAX_JOYSTICK_AXIS; i++)
756 {
757 if (translatedAxesSet[i])
758 Com_QueueEvent(in_eventTime, SE_JOYSTICK_AXIS, i, translatedAxes[i], 0, NULL);
759 }
760 }
761}

764/*
765===============
766IN_JoyMove
767===============
768*/
769static void IN_JoyMove( void )
770{
771 unsigned int axes = 0;
772 unsigned int hats = 0;
773 int total = 0;
774 int i = 0;
775
776 if (gamepad)
777 {
778 IN_GamepadMove();
779 return;
780 }
781
782 if (!stick)
783 return;
784
785 SDL_JoystickUpdate();
786
787 // update the ball state.
788 total = SDL_JoystickNumBalls(stick);
789 if (total > 0)
790 {
791 int balldx = 0;
792 int balldy = 0;
793 for (i = 0; i < total; i++)
794 {
795 int dx = 0;
796 int dy = 0;
797 SDL_JoystickGetBall(stick, i, &dx, &dy);
798 balldx += dx;
799 balldy += dy;
800 }
801 if (balldx || balldy)
802 {
803 // !!! FIXME: is this good for stick balls, or just mice?
804 // Scale like the mouse input...
805 if (abs(balldx) > 1)
806 balldx *= 2;
807 if (abs(balldy) > 1)
808 balldy *= 2;
809 Com_QueueEvent( in_eventTime, SE_MOUSE, balldx, balldy, 0, NULL );
810 }
811 }
812
813 // now query the stick buttons...
814 total = SDL_JoystickNumButtons(stick);
815 if (total > 0)
816 {
817 if (total > ARRAY_LEN(stick_state.buttons))
818 total = ARRAY_LEN(stick_state.buttons);
819 for (i = 0; i < total; i++)
820 {
821 qboolean pressed = (SDL_JoystickGetButton(stick, i) != 0);
822 if (pressed != stick_state.buttons[i])
823 {
824 Com_QueueEvent( in_eventTime, SE_KEY, K_JOY1 + i, pressed, 0, NULL );
825 stick_state.buttons[i] = pressed;
826 }
827 }
828 }
829
830 // look at the hats...
831 total = SDL_JoystickNumHats(stick);
832 if (total > 0)
833 {
834 if (total > 4) total = 4;
835 for (i = 0; i < total; i++)
836 {
837 ((Uint8 *)&hats)[i] = SDL_JoystickGetHat(stick, i);
838 }
839 }
840
841 // update hat state
842 if (hats != stick_state.oldhats)
843 {
844 for( i = 0; i < 4; i++ ) {
845 if( ((Uint8 *)&hats)[i] != ((Uint8 *)&stick_state.oldhats)[i] ) {
846 // release event
847 switch( ((Uint8 *)&stick_state.oldhats)[i] ) {
848 case SDL_HAT_UP:
849 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 0], qfalse, 0, NULL );
850 break;
851 case SDL_HAT_RIGHT:
852 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 1], qfalse, 0, NULL );
853 break;
854 case SDL_HAT_DOWN:
855 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 2], qfalse, 0, NULL );
856 break;
857 case SDL_HAT_LEFT:
858 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 3], qfalse, 0, NULL );
859 break;
860 case SDL_HAT_RIGHTUP:
861 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 0], qfalse, 0, NULL );
862 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 1], qfalse, 0, NULL );
863 break;
864 case SDL_HAT_RIGHTDOWN:
865 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 2], qfalse, 0, NULL );
866 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 1], qfalse, 0, NULL );
867 break;
868 case SDL_HAT_LEFTUP:
869 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 0], qfalse, 0, NULL );
870 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 3], qfalse, 0, NULL );
871 break;
872 case SDL_HAT_LEFTDOWN:
873 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 2], qfalse, 0, NULL );
874 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 3], qfalse, 0, NULL );
875 break;
876 default:
877 break;
878 }
879 // press event
880 switch( ((Uint8 *)&hats)[i] ) {
881 case SDL_HAT_UP:
882 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 0], qtrue, 0, NULL );
883 break;
884 case SDL_HAT_RIGHT:
885 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 1], qtrue, 0, NULL );
886 break;
887 case SDL_HAT_DOWN:
888 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 2], qtrue, 0, NULL );
889 break;
890 case SDL_HAT_LEFT:
891 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 3], qtrue, 0, NULL );
892 break;
893 case SDL_HAT_RIGHTUP:
894 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 0], qtrue, 0, NULL );
895 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 1], qtrue, 0, NULL );
896 break;
897 case SDL_HAT_RIGHTDOWN:
898 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 2], qtrue, 0, NULL );
899 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 1], qtrue, 0, NULL );
900 break;
901 case SDL_HAT_LEFTUP:
902 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 0], qtrue, 0, NULL );
903 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 3], qtrue, 0, NULL );
904 break;
905 case SDL_HAT_LEFTDOWN:
906 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 2], qtrue, 0, NULL );
907 Com_QueueEvent( in_eventTime, SE_KEY, hat_keys[4*i + 3], qtrue, 0, NULL );
908 break;
909 default:
910 break;
911 }
912 }
913 }
914 }
915
916 // save hat state
917 stick_state.oldhats = hats;
918
919 // finally, look at the axes...
920 total = SDL_JoystickNumAxes(stick);
921 if (total > 0)
922 {
923 if (in_joystickUseAnalog->integer)
924 {
925 if (total > MAX_JOYSTICK_AXIS) total = MAX_JOYSTICK_AXIS;
926 for (i = 0; i < total; i++)
927 {
928 Sint16 axis = SDL_JoystickGetAxis(stick, i);
929 float f = ( (float) abs(axis) ) / 32767.0f;
930
931 if( f < in_joystickThreshold->value ) axis = 0;
932
933 if ( axis != stick_state.oldaaxes[i] )
934 {
935 Com_QueueEvent( in_eventTime, SE_JOYSTICK_AXIS, i, axis, 0, NULL );
936 stick_state.oldaaxes[i] = axis;
937 }
938 }
939 }
940 else
941 {
942 if (total > 16) total = 16;
943 for (i = 0; i < total; i++)
944 {
945 Sint16 axis = SDL_JoystickGetAxis(stick, i);
946 float f = ( (float) axis ) / 32767.0f;
947 if( f < -in_joystickThreshold->value ) {
948 axes |= ( 1 << ( i * 2 ) );
949 } else if( f > in_joystickThreshold->value ) {
950 axes |= ( 1 << ( ( i * 2 ) + 1 ) );
951 }
952 }
953 }
954 }
955
956 /* Time to update axes state based on old vs. new. */
957 if (axes != stick_state.oldaxes)
958 {
959 for( i = 0; i < 16; i++ ) {
960 if( ( axes & ( 1 << i ) ) && !( stick_state.oldaxes & ( 1 << i ) ) ) {
961 Com_QueueEvent( in_eventTime, SE_KEY, joy_keys[i], qtrue, 0, NULL );
962 }
963
964 if( !( axes & ( 1 << i ) ) && ( stick_state.oldaxes & ( 1 << i ) ) ) {
965 Com_QueueEvent( in_eventTime, SE_KEY, joy_keys[i], qfalse, 0, NULL );
966 }
967 }
968 }
969
970 /* Save for future generations. */
971 stick_state.oldaxes = axes;
972}

974/*
975===============
976IN_ProcessEvents
977===============
978*/
979static void IN_ProcessEvents( void )
980{
981 SDL_Event e;
982 keyNum_t key = 0;
983 static keyNum_t lastKeyDown = 0;
984
985 if( !SDL_WasInit( SDL_INIT_VIDEO ) )
986 return;
987
988 while( SDL_PollEvent( &e ) )
989 {
990 switch( e.type )
991 {
992 case SDL_KEYDOWN:
993 if ( e.key.repeat && Key_GetCatcher( ) == 0 )
994 break;
995
996 if( ( key = IN_TranslateSDLToQ3Key( &e.key.keysym, qtrue ) ) )
997 Com_QueueEvent( in_eventTime, SE_KEY, key, qtrue, 0, NULL );
998
999 if( key == K_BACKSPACE )
1000 Com_QueueEvent( in_eventTime, SE_CHAR, CTRL('h'), 0, 0, NULL );
1001 else if( keys[K_CTRL].down && key >= 'a' && key <= 'z' )
1002 Com_QueueEvent( in_eventTime, SE_CHAR, CTRL(key), 0, 0, NULL );
1003
1004 lastKeyDown = key;
1005 break;
1006
1007 case SDL_KEYUP:
1008 if( ( key = IN_TranslateSDLToQ3Key( &e.key.keysym, qfalse ) ) )
1009 Com_QueueEvent( in_eventTime, SE_KEY, key, qfalse, 0, NULL );
1010
1011 lastKeyDown = 0;
1012 break;
1013
1014 case SDL_TEXTINPUT:
1015 if( lastKeyDown != K_CONSOLE )
1016 {
1017 char *c = e.text.text;
1018
1019 // Quick and dirty UTF-8 to UTF-32 conversion
1020 while( *c )
1021 {
1022 int utf32 = 0;
1023
1024 if( ( *c & 0x80 ) == 0 )
1025 utf32 = *c++;
1026 else if( ( *c & 0xE0 ) == 0xC0 ) // 110x xxxx
1027 {
1028 utf32 |= ( *c++ & 0x1F ) << 6;
1029 utf32 |= ( *c++ & 0x3F );
1030 }
1031 else if( ( *c & 0xF0 ) == 0xE0 ) // 1110 xxxx
1032 {
1033 utf32 |= ( *c++ & 0x0F ) << 12;
1034 utf32 |= ( *c++ & 0x3F ) << 6;
1035 utf32 |= ( *c++ & 0x3F );
1036 }
1037 else if( ( *c & 0xF8 ) == 0xF0 ) // 1111 0xxx
1038 {
1039 utf32 |= ( *c++ & 0x07 ) << 18;
1040 utf32 |= ( *c++ & 0x3F ) << 12;
1041 utf32 |= ( *c++ & 0x3F ) << 6;
1042 utf32 |= ( *c++ & 0x3F );
1043 }
1044 else
1045 {
1046 Com_DPrintf( "Unrecognised UTF-8 lead byte: 0x%x\n", (unsigned int)*c );
1047 c++;
1048 }
1049
1050 if( utf32 != 0 )
1051 {
1052 if( IN_IsConsoleKey( 0, utf32 ) )
1053 {
1054 Com_QueueEvent( in_eventTime, SE_KEY, K_CONSOLE, qtrue, 0, NULL );
1055 Com_QueueEvent( in_eventTime, SE_KEY, K_CONSOLE, qfalse, 0, NULL );
1056 }
1057 else
1058 Com_QueueEvent( in_eventTime, SE_CHAR, utf32, 0, 0, NULL );
1059 }
1060 }
1061 }
1062 break;
1063
1064 case SDL_MOUSEMOTION:
1065 if( mouseActive )
1066 {
1067 if( !e.motion.xrel && !e.motion.yrel )
1068 break;
1069 Com_QueueEvent( in_eventTime, SE_MOUSE, e.motion.xrel, e.motion.yrel, 0, NULL );
1070 }
1071 break;
1072
1073 case SDL_MOUSEBUTTONDOWN:
1074 case SDL_MOUSEBUTTONUP:
1075 {
1076 int b;
1077 switch( e.button.button )
1078 {
1079 case SDL_BUTTON_LEFT: b = K_MOUSE1; break;
1080 case SDL_BUTTON_MIDDLE: b = K_MOUSE3; break;
1081 case SDL_BUTTON_RIGHT: b = K_MOUSE2; break;
1082 case SDL_BUTTON_X1: b = K_MOUSE4; break;
1083 case SDL_BUTTON_X2: b = K_MOUSE5; break;
1084 default: b = K_AUX1 + ( e.button.button - SDL_BUTTON_X2 + 1 ) % 16; break;
1085 }
1086 Com_QueueEvent( in_eventTime, SE_KEY, b,
1087 ( e.type == SDL_MOUSEBUTTONDOWN ? qtrue : qfalse ), 0, NULL );
1088 }
1089 break;
1090
1091 case SDL_MOUSEWHEEL:
1092 if( e.wheel.y > 0 )
1093 {
1094 Com_QueueEvent( in_eventTime, SE_KEY, K_MWHEELUP, qtrue, 0, NULL );
1095 Com_QueueEvent( in_eventTime, SE_KEY, K_MWHEELUP, qfalse, 0, NULL );
1096 }
1097 else if( e.wheel.y < 0 )
1098 {
1099 Com_QueueEvent( in_eventTime, SE_KEY, K_MWHEELDOWN, qtrue, 0, NULL );
1100 Com_QueueEvent( in_eventTime, SE_KEY, K_MWHEELDOWN, qfalse, 0, NULL );
1101 }
1102 break;
1103
1104 case SDL_CONTROLLERDEVICEADDED:
1105 case SDL_CONTROLLERDEVICEREMOVED:
1106 if (in_joystick->integer)
1107 IN_InitJoystick();
1108 break;
1109
1110 case SDL_QUIT:
1111 Cbuf_ExecuteText(EXEC_NOW, "quit Closed window\n");
1112 break;
1113
1114 case SDL_WINDOWEVENT:
1115 switch( e.window.event )
1116 {
1117 case SDL_WINDOWEVENT_RESIZED:
1118 {
1119 int width, height;
1120
1121 width = e.window.data1;
1122 height = e.window.data2;
1123
1124 // check if size actually changed
1125 if( cls.glconfig.vidWidth == width && cls.glconfig.vidHeight == height )
1126 {
1127 break;
1128 }
1129
1130 Cvar_SetValue( "r_customwidth", width );
1131 Cvar_SetValue( "r_customheight", height );
1132 Cvar_Set( "r_mode", "-1" );
1133
1134 // Wait until user stops dragging for 1 second, so
1135 // we aren't constantly recreating the GL context while
1136 // he tries to drag...
1137 vidRestartTime = Sys_Milliseconds( ) + 1000;
1138 }
1139 break;
1140
1141 case SDL_WINDOWEVENT_MINIMIZED: Cvar_SetValue( "com_minimized", 1 ); break;
1142 case SDL_WINDOWEVENT_RESTORED:
1143 case SDL_WINDOWEVENT_MAXIMIZED: Cvar_SetValue( "com_minimized", 0 ); break;
1144 case SDL_WINDOWEVENT_FOCUS_LOST: Cvar_SetValue( "com_unfocused", 1 ); break;
1145 case SDL_WINDOWEVENT_FOCUS_GAINED: Cvar_SetValue( "com_unfocused", 0 ); break;
1146 }
1147 break;
1148
1149 default:
1150 break;
1151 }
1152 }
1153}

1155/*
1156===============
1157IN_Frame
1158===============
1159*/
1160void IN_Frame( void )
1161{
1162 qboolean loading;
1163
1164 IN_JoyMove( );
1165
1166 // If not DISCONNECTED (main menu) or ACTIVE (in game), we're loading
1167 loading = ( clc.state != CA_DISCONNECTED && clc.state != CA_ACTIVE );
1168
1169 if( !cls.glconfig.isFullscreen && ( Key_GetCatcher( ) & KEYCATCH_CONSOLE ) )
1170 {
1171 // Console is down in windowed mode
1172 IN_DeactivateMouse( );
1173 }
1174 else if( !cls.glconfig.isFullscreen && loading )
1175 {
1176 // Loading in windowed mode
1177 IN_DeactivateMouse( );
1178 }
1179 else if( !( SDL_GetWindowFlags( SDL_window ) & SDL_WINDOW_INPUT_FOCUS ) )
1180 {
1181 // Window not got focus
1182 IN_DeactivateMouse( );
1183 }
1184 else
1185 IN_ActivateMouse( );
1186
1187 IN_ProcessEvents( );
1188
1189 // Set event time for next frame to earliest possible time an event could happen
1190 in_eventTime = Sys_Milliseconds( );
1191
1192 // In case we had to delay actual restart of video system
1193 if( ( vidRestartTime != 0 ) && ( vidRestartTime < Sys_Milliseconds( ) ) )
1194 {
1195 vidRestartTime = 0;
1196 Cbuf_AddText( "vid_restart\n" );
1197 }
1198}

1202IN_Init
1203===============
1204*/
1205void IN_Init( void *windowData )
1206{
1207 int appState;
1208
1209 if( !SDL_WasInit( SDL_INIT_VIDEO ) )
1210 {
1211 Com_Error( ERR_FATAL, "IN_Init called before SDL_Init( SDL_INIT_VIDEO )" );
1212 return;
1213 }
1214
1215 SDL_window = (SDL_Window *)windowData;
1216
1217 Com_DPrintf( "\n------- Input Initialization -------\n" );
1218
1219 in_keyboardDebug = Cvar_Get( "in_keyboardDebug", "0", CVAR_ARCHIVE );

1223 in_nograb = Cvar_Get( "in_nograb", "0", CVAR_ARCHIVE );

1230 mouseAvailable = ( in_mouse->value != 0 );
1231 IN_DeactivateMouse( );

1233 appState = SDL_GetWindowFlags( SDL_window );
1234 Cvar_SetValue( "com_unfocused", !( appState & SDL_WINDOW_INPUT_FOCUS ) );
1235 Cvar_SetValue( "com_minimized", appState & SDL_WINDOW_MINIMIZED );

1238 Com_DPrintf( "------------------------------------\n" );

97/*
98===============
99GLimp_CompareModes
100===============
101*/
102static int GLimp_CompareModes( const void *a, const void *b )
103{
104 const float ASPECT_EPSILON = 0.001f;
105 SDL_Rect *modeA = (SDL_Rect *)a;
106 SDL_Rect *modeB = (SDL_Rect *)b;
107 float aspectA = (float)modeA->w / (float)modeA->h;
108 float aspectB = (float)modeB->w / (float)modeB->h;
109 int areaA = modeA->w * modeA->h;
110 int areaB = modeB->w * modeB->h;
111 float aspectDiffA = fabs( aspectA - displayAspect );
112 float aspectDiffB = fabs( aspectB - displayAspect );
113 float aspectDiffsDiff = aspectDiffA - aspectDiffB;
114
115 if( aspectDiffsDiff > ASPECT_EPSILON )
116 return 1;
117 else if( aspectDiffsDiff < -ASPECT_EPSILON )
118 return -1;
119 else
120 return areaA - areaB;
121}

124/*
125===============
126GLimp_DetectAvailableModes
127===============
128*/
129static void GLimp_DetectAvailableModes(void)
130{
131 int i, j;
132 char buf[ MAX_STRING_CHARS ] = { 0 };
133 int numSDLModes;
134 SDL_Rect *modes;
135 int numModes = 0;
136
137 SDL_DisplayMode windowMode;
138 int display = SDL_GetWindowDisplayIndex( SDL_window );
139 if( display < 0 )
140 {
141 ri.Printf( PRINT_WARNING, "Couldn't get window display index, no resolutions detected: %s\n", SDL_GetError() );
142 return;
143 }
144 numSDLModes = SDL_GetNumDisplayModes( display );
145
146 if( SDL_GetWindowDisplayMode( SDL_window, &windowMode ) < 0 || numSDLModes <= 0 )
147 {
148 ri.Printf( PRINT_WARNING, "Couldn't get window display mode, no resolutions detected: %s\n", SDL_GetError() );
149 return;
150 }
151
152 modes = SDL_calloc( (size_t)numSDLModes, sizeof( SDL_Rect ) );
153 if ( !modes )
154 {
155 ri.Error( ERR_FATAL, "Out of memory" );
156 }
157
158 for( i = 0; i < numSDLModes; i++ )
159 {
160 SDL_DisplayMode mode;
161
162 if( SDL_GetDisplayMode( display, i, &mode ) < 0 )
163 continue;
164
165 if( !mode.w || !mode.h )
166 {
167 ri.Printf( PRINT_ALL, "Display supports any resolution\n" );
168 SDL_free( modes );
169 return;
170 }
171
172 if( windowMode.format != mode.format )
173 continue;
174
175 // SDL can give the same resolution with different refresh rates.
176 // Only list resolution once.
177 for( j = 0; j < numModes; j++ )
178 {
179 if( mode.w == modes[ j ].w && mode.h == modes[ j ].h )
180 break;
181 }
182
183 if( j != numModes )
184 continue;
185
186 modes[ numModes ].w = mode.w;
187 modes[ numModes ].h = mode.h;
188 numModes++;
189 }
190
191 if( numModes > 1 )
192 qsort( modes, numModes, sizeof( SDL_Rect ), GLimp_CompareModes );
193
194 for( i = 0; i < numModes; i++ )
195 {
196 const char *newModeString = va( "%ux%u ", modes[ i ].w, modes[ i ].h );
197
198 if( strlen( newModeString ) < (int)sizeof( buf ) - strlen( buf ) )
199 Q_strcat( buf, sizeof( buf ), newModeString );
200 else
201 ri.Printf( PRINT_WARNING, "Skipping mode %ux%u, buffer too small\n", modes[ i ].w, modes[ i ].h );
202 }
203
204 if( *buf )
205 {
206 buf[ strlen( buf ) - 1 ] = 0;
207 ri.Printf( PRINT_ALL, "Available modes: '%s'\n", buf );
208 ri.Cvar_Set( "r_availableModes", buf );
209 }
210 SDL_free( modes );
211}

213/*
214===============
215GLimp_SetMode
216===============
217*/
218static int GLimp_SetMode(int mode, qboolean fullscreen, qboolean noborder, qboolean coreContext)
219{
220 const char *glstring;
221 int perChannelColorBits;
222 int colorBits, depthBits, stencilBits;
223 int samples;
224 int i = 0;
225 SDL_Surface *icon = NULL;
226 Uint32 flags = SDL_WINDOW_SHOWN | SDL_WINDOW_OPENGL;
227 SDL_DisplayMode desktopMode;
228 int display = 0;
229 int x = SDL_WINDOWPOS_UNDEFINED, y = SDL_WINDOWPOS_UNDEFINED;
230
231 ri.Printf( PRINT_ALL, "Initializing OpenGL display\n");
232
233 if ( r_allowResize->integer )
234 flags |= SDL_WINDOW_RESIZABLE;
235
236#ifdef USE_ICON
237 icon = SDL_CreateRGBSurfaceFrom(
238 (void *)CLIENT_WINDOW_ICON.pixel_data,
239 CLIENT_WINDOW_ICON.width,
240 CLIENT_WINDOW_ICON.height,
241 CLIENT_WINDOW_ICON.bytes_per_pixel * 8,
242 CLIENT_WINDOW_ICON.bytes_per_pixel * CLIENT_WINDOW_ICON.width,
243#ifdef Q3_LITTLE_ENDIAN
244 0x000000FF, 0x0000FF00, 0x00FF0000, 0xFF000000
245#else
246 0xFF000000, 0x00FF0000, 0x0000FF00, 0x000000FF
247#endif
248 );
249#endif
250
251 // If a window exists, note its display index
252 if( SDL_window != NULL )
253 {
254 display = SDL_GetWindowDisplayIndex( SDL_window );
255 if( display < 0 )
256 {
257 ri.Printf( PRINT_DEVELOPER, "SDL_GetWindowDisplayIndex() failed: %s\n", SDL_GetError() );
258 }
259 }
260
261 if( display >= 0 && SDL_GetDesktopDisplayMode( display, &desktopMode ) == 0 )
262 {
263 displayAspect = (float)desktopMode.w / (float)desktopMode.h;
264
265 ri.Printf( PRINT_ALL, "Display aspect: %.3f\n", displayAspect );
266 }
267 else
268 {
269 Com_Memset( &desktopMode, 0, sizeof( SDL_DisplayMode ) );
270
271 ri.Printf( PRINT_ALL,
272 "Cannot determine display aspect, assuming 1.333\n" );
273 }
274
275 ri.Printf (PRINT_ALL, "...setting mode %d:", mode );
276
277 if (mode == -2)
278 {
279 // use desktop video resolution
280 if( desktopMode.h > 0 )
281 {
282 glConfig.vidWidth = desktopMode.w;
283 glConfig.vidHeight = desktopMode.h;
284 }
285 else
286 {
287 glConfig.vidWidth = 640;
288 glConfig.vidHeight = 480;
289 ri.Printf( PRINT_ALL,
290 "Cannot determine display resolution, assuming 640x480\n" );
291 }
292
293 glConfig.windowAspect = (float)glConfig.vidWidth / (float)glConfig.vidHeight;
294 }
295 else if ( !R_GetModeInfo( &glConfig.vidWidth, &glConfig.vidHeight, &glConfig.windowAspect, mode ) )
296 {
297 ri.Printf( PRINT_ALL, " invalid mode\n" );
298 return RSERR_INVALID_MODE;
299 }
300 ri.Printf( PRINT_ALL, " %d %d\n", glConfig.vidWidth, glConfig.vidHeight);
301
302 // Center window
303 if( r_centerWindow->integer && !fullscreen )
304 {
305 x = ( desktopMode.w / 2 ) - ( glConfig.vidWidth / 2 );
306 y = ( desktopMode.h / 2 ) - ( glConfig.vidHeight / 2 );
307 }
308
309 // Destroy existing state if it exists
310 if( SDL_glContext != NULL )
311 {
312 SDL_GL_DeleteContext( SDL_glContext );
313 SDL_glContext = NULL;
314 }
315
316 if( SDL_window != NULL )
317 {
318 SDL_GetWindowPosition( SDL_window, &x, &y );
319 ri.Printf( PRINT_DEVELOPER, "Existing window at %dx%d before being destroyed\n", x, y );
320 SDL_DestroyWindow( SDL_window );
321 SDL_window = NULL;
322 }
323
324 if( fullscreen )
325 {
326 flags |= SDL_WINDOW_FULLSCREEN;
327 glConfig.isFullscreen = qtrue;
328 }
329 else
330 {
331 if( noborder )
332 flags |= SDL_WINDOW_BORDERLESS;
333
334 glConfig.isFullscreen = qfalse;
335 }
336
337 colorBits = r_colorbits->value;
338 if ((!colorBits) || (colorBits >= 32))
339 colorBits = 24;
340
341 if (!r_depthbits->value)
342 depthBits = 24;
343 else
344 depthBits = r_depthbits->value;
345
346 stencilBits = r_stencilbits->value;
347 samples = r_ext_multisample->value;
348
349 for (i = 0; i < 16; i++)
350 {
351 int testColorBits, testDepthBits, testStencilBits;
352
353 // 0 - default
354 // 1 - minus colorBits
355 // 2 - minus depthBits
356 // 3 - minus stencil
357 if ((i % 4) == 0 && i)
358 {
359 // one pass, reduce
360 switch (i / 4)
361 {
362 case 2 :
363 if (colorBits == 24)
364 colorBits = 16;
365 break;
366 case 1 :
367 if (depthBits == 24)
368 depthBits = 16;
369 else if (depthBits == 16)
370 depthBits = 8;
371 case 3 :
372 if (stencilBits == 24)
373 stencilBits = 16;
374 else if (stencilBits == 16)
375 stencilBits = 8;
376 }
377 }
378
379 testColorBits = colorBits;
380 testDepthBits = depthBits;
381 testStencilBits = stencilBits;
382
383 if ((i % 4) == 3)
384 { // reduce colorBits
385 if (testColorBits == 24)
386 testColorBits = 16;
387 }
388
389 if ((i % 4) == 2)
390 { // reduce depthBits
391 if (testDepthBits == 24)
392 testDepthBits = 16;
393 else if (testDepthBits == 16)
394 testDepthBits = 8;
395 }
396
397 if ((i % 4) == 1)
398 { // reduce stencilBits
399 if (testStencilBits == 24)
400 testStencilBits = 16;
401 else if (testStencilBits == 16)
402 testStencilBits = 8;
403 else
404 testStencilBits = 0;
405 }
406
407 if (testColorBits == 24)
408 perChannelColorBits = 8;
409 else
410 perChannelColorBits = 4;
411
412#ifdef __sgi /* Fix for SGIs grabbing too many bits of color */
413 if (perChannelColorBits == 4)
414 perChannelColorBits = 0; /* Use minimum size for 16-bit color */
415
416 /* Need alpha or else SGIs choose 36+ bit RGB mode */
417 SDL_GL_SetAttribute( SDL_GL_ALPHA_SIZE, 1);
418#endif
419
420 SDL_GL_SetAttribute( SDL_GL_RED_SIZE, perChannelColorBits );
421 SDL_GL_SetAttribute( SDL_GL_GREEN_SIZE, perChannelColorBits );
422 SDL_GL_SetAttribute( SDL_GL_BLUE_SIZE, perChannelColorBits );
423 SDL_GL_SetAttribute( SDL_GL_DEPTH_SIZE, testDepthBits );
424 SDL_GL_SetAttribute( SDL_GL_STENCIL_SIZE, testStencilBits );
425
426 SDL_GL_SetAttribute( SDL_GL_MULTISAMPLEBUFFERS, samples ? 1 : 0 );
427 SDL_GL_SetAttribute( SDL_GL_MULTISAMPLESAMPLES, samples );
428
429 if(r_stereoEnabled->integer)
430 {
431 glConfig.stereoEnabled = qtrue;
432 SDL_GL_SetAttribute(SDL_GL_STEREO, 1);
433 }
434 else
435 {
436 glConfig.stereoEnabled = qfalse;
437 SDL_GL_SetAttribute(SDL_GL_STEREO, 0);
438 }
439
440 SDL_GL_SetAttribute( SDL_GL_DOUBLEBUFFER, 1 );
441
442#if 0 // if multisampling is enabled on X11, this causes create window to fail.
443 // If not allowing software GL, demand accelerated
444 if( !r_allowSoftwareGL->integer )
445 SDL_GL_SetAttribute( SDL_GL_ACCELERATED_VISUAL, 1 );
446#endif
447
448 if( ( SDL_window = SDL_CreateWindow( CLIENT_WINDOW_TITLE, x, y,
449 glConfig.vidWidth, glConfig.vidHeight, flags ) ) == NULL )
450 {
451 ri.Printf( PRINT_DEVELOPER, "SDL_CreateWindow failed: %s\n", SDL_GetError( ) );
452 continue;
453 }
454
455 if( fullscreen )
456 {
457 SDL_DisplayMode mode;
458
459 switch( testColorBits )
460 {
461 case 16: mode.format = SDL_PIXELFORMAT_RGB565; break;
462 case 24: mode.format = SDL_PIXELFORMAT_RGB24; break;
463 default: ri.Printf( PRINT_DEVELOPER, "testColorBits is %d, can't fullscreen\n", testColorBits ); continue;
464 }
465
466 mode.w = glConfig.vidWidth;
467 mode.h = glConfig.vidHeight;
468 mode.refresh_rate = glConfig.displayFrequency = ri.Cvar_VariableIntegerValue( "r_displayRefresh" );
469 mode.driverdata = NULL;
470
471 if( SDL_SetWindowDisplayMode( SDL_window, &mode ) < 0 )
472 {
473 ri.Printf( PRINT_DEVELOPER, "SDL_SetWindowDisplayMode failed: %s\n", SDL_GetError( ) );
474 continue;
475 }
476 }
477
478 SDL_SetWindowIcon( SDL_window, icon );
479
480 if (coreContext)
481 {
482 int profileMask, majorVersion, minorVersion;
483 SDL_GL_GetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, &profileMask);
484 SDL_GL_GetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, &majorVersion);
485 SDL_GL_GetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, &minorVersion);
486
487 ri.Printf(PRINT_ALL, "Trying to get an OpenGL 3.2 core context\n");
488 SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, SDL_GL_CONTEXT_PROFILE_CORE);
489 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, 3);
490 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, 2);
491 if ((SDL_glContext = SDL_GL_CreateContext(SDL_window)) == NULL)
492 {
493 ri.Printf(PRINT_ALL, "SDL_GL_CreateContext failed: %s\n", SDL_GetError());
494 ri.Printf(PRINT_ALL, "Reverting to default context\n");
495
496 SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, profileMask);
497 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, majorVersion);
498 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, minorVersion);
499 }
500 else
501 {
502 const char *renderer;
503
504 ri.Printf(PRINT_ALL, "SDL_GL_CreateContext succeeded.\n");
505
506 renderer = (const char *)qglGetString(GL_RENDERER);
507 if (renderer && (strstr(renderer, "Software Renderer") || strstr(renderer, "Software Rasterizer")))
508 {
509 ri.Printf(PRINT_ALL, "GL_RENDERER is %s, rejecting context\n", renderer);
510
511 SDL_GL_DeleteContext(SDL_glContext);
512 SDL_glContext = NULL;
513
514 SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, profileMask);
515 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, majorVersion);
516 SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, minorVersion);
517 }
518 }
519 }
520 else
521 {
522 SDL_glContext = NULL;
523 }
524
525 if( !SDL_glContext && ( SDL_glContext = SDL_GL_CreateContext( SDL_window ) ) == NULL )
526 {
527 ri.Printf( PRINT_DEVELOPER, "SDL_GL_CreateContext failed: %s\n", SDL_GetError( ) );
528 continue;
529 }
530
531 qglClearColor( 0, 0, 0, 1 );
532 qglClear( GL_COLOR_BUFFER_BIT );
533 SDL_GL_SwapWindow( SDL_window );
534
535 if( SDL_GL_SetSwapInterval( r_swapInterval->integer ) == -1 )
536 {
537 ri.Printf( PRINT_DEVELOPER, "SDL_GL_SetSwapInterval failed: %s\n", SDL_GetError( ) );
538 }
539
540 glConfig.colorBits = testColorBits;
541 glConfig.depthBits = testDepthBits;
542 glConfig.stencilBits = testStencilBits;
543
544 ri.Printf( PRINT_ALL, "Using %d color bits, %d depth, %d stencil display.\n",
545 glConfig.colorBits, glConfig.depthBits, glConfig.stencilBits );
546 break;
547 }
548
549 SDL_FreeSurface( icon );
550
551 if( !SDL_window )
552 {
553 ri.Printf( PRINT_ALL, "Couldn't get a visual\n" );
554 return RSERR_INVALID_MODE;
555 }
556
557 GLimp_DetectAvailableModes();
558
559 glstring = (char *) qglGetString (GL_RENDERER);
560 ri.Printf( PRINT_ALL, "GL_RENDERER: %s\n", glstring );
561
562 return RSERR_OK;
563}

565/*
566===============
567GLimp_StartDriverAndSetMode
568===============
569*/
570static qboolean GLimp_StartDriverAndSetMode(int mode, qboolean fullscreen, qboolean noborder, qboolean gl3Core)
571{
572 rserr_t err;
573
574 if (!SDL_WasInit(SDL_INIT_VIDEO))
575 {
576 const char *driverName;
577
578 if (SDL_Init(SDL_INIT_VIDEO) != 0)
579 {
580 ri.Printf( PRINT_ALL, "SDL_Init( SDL_INIT_VIDEO ) FAILED (%s)\n", SDL_GetError());
581 return qfalse;
582 }
583
584 driverName = SDL_GetCurrentVideoDriver( );
585 ri.Printf( PRINT_ALL, "SDL using driver \"%s\"\n", driverName );
586 ri.Cvar_Set( "r_sdlDriver", driverName );
587 }
588
589 if (fullscreen && ri.Cvar_VariableIntegerValue( "in_nograb" ) )
590 {
591 ri.Printf( PRINT_ALL, "Fullscreen not allowed with in_nograb 1\n");
592 ri.Cvar_Set( "r_fullscreen", "0" );
593 r_fullscreen->modified = qfalse;
594 fullscreen = qfalse;
595 }
596
597 err = GLimp_SetMode(mode, fullscreen, noborder, gl3Core);
598
599 switch ( err )
600 {
601 case RSERR_INVALID_FULLSCREEN:
602 ri.Printf( PRINT_ALL, "...WARNING: fullscreen unavailable in this mode\n" );
603 return qfalse;
604 case RSERR_INVALID_MODE:
605 ri.Printf( PRINT_ALL, "...WARNING: could not set the given mode (%d)\n", mode );
606 return qfalse;
607 default:
608 break;
609 }
610
611 return qtrue;
612}

615/*
616===============
617GLimp_InitExtensions
618===============
619*/
620static void GLimp_InitExtensions( void )
621{
622 if ( !r_allowExtensions->integer )
623 {
624 ri.Printf( PRINT_ALL, "* IGNORING OPENGL EXTENSIONS *\n" );
625 return;
626 }
627
628 ri.Printf( PRINT_ALL, "Initializing OpenGL extensions\n" );
629
630 glConfig.textureCompression = TC_NONE;
631
632 // GL_EXT_texture_compression_s3tc
633 if ( SDL_GL_ExtensionSupported( "GL_ARB_texture_compression" ) &&
634 SDL_GL_ExtensionSupported( "GL_EXT_texture_compression_s3tc" ) )
635 {
636 if ( r_ext_compressed_textures->value )
637 {
638 glConfig.textureCompression = TC_S3TC_ARB;
639 ri.Printf( PRINT_ALL, "...using GL_EXT_texture_compression_s3tc\n" );
640 }
641 else
642 {
643 ri.Printf( PRINT_ALL, "...ignoring GL_EXT_texture_compression_s3tc\n" );
644 }
645 }
646 else
647 {
648 ri.Printf( PRINT_ALL, "...GL_EXT_texture_compression_s3tc not found\n" );
649 }
650
651 // GL_S3_s3tc ... legacy extension before GL_EXT_texture_compression_s3tc.
652 if (glConfig.textureCompression == TC_NONE)
653 {
654 if ( SDL_GL_ExtensionSupported( "GL_S3_s3tc" ) )
655 {
656 if ( r_ext_compressed_textures->value )
657 {
658 glConfig.textureCompression = TC_S3TC;
659 ri.Printf( PRINT_ALL, "...using GL_S3_s3tc\n" );
660 }
661 else
662 {
663 ri.Printf( PRINT_ALL, "...ignoring GL_S3_s3tc\n" );
664 }
665 }
666 else
667 {
668 ri.Printf( PRINT_ALL, "...GL_S3_s3tc not found\n" );
669 }
670 }
671
672
673 // GL_EXT_texture_env_add
674 glConfig.textureEnvAddAvailable = qfalse;
675 if ( SDL_GL_ExtensionSupported( "GL_EXT_texture_env_add" ) )
676 {
677 if ( r_ext_texture_env_add->integer )
678 {
679 glConfig.textureEnvAddAvailable = qtrue;
680 ri.Printf( PRINT_ALL, "...using GL_EXT_texture_env_add\n" );
681 }
682 else
683 {
684 glConfig.textureEnvAddAvailable = qfalse;
685 ri.Printf( PRINT_ALL, "...ignoring GL_EXT_texture_env_add\n" );
686 }
687 }
688 else
689 {
690 ri.Printf( PRINT_ALL, "...GL_EXT_texture_env_add not found\n" );
691 }
692
693 // GL_ARB_multitexture
694 qglMultiTexCoord2fARB = NULL;
695 qglActiveTextureARB = NULL;
696 qglClientActiveTextureARB = NULL;
697 if ( SDL_GL_ExtensionSupported( "GL_ARB_multitexture" ) )
698 {
699 if ( r_ext_multitexture->value )
700 {
701 qglMultiTexCoord2fARB = SDL_GL_GetProcAddress( "glMultiTexCoord2fARB" );
702 qglActiveTextureARB = SDL_GL_GetProcAddress( "glActiveTextureARB" );
703 qglClientActiveTextureARB = SDL_GL_GetProcAddress( "glClientActiveTextureARB" );
704
705 if ( qglActiveTextureARB )
706 {
707 GLint glint = 0;
708 qglGetIntegerv( GL_MAX_TEXTURE_UNITS_ARB, &glint );
709 glConfig.numTextureUnits = (int) glint;
710 if ( glConfig.numTextureUnits > 1 )
711 {
712 ri.Printf( PRINT_ALL, "...using GL_ARB_multitexture\n" );
713 }
714 else
715 {
716 qglMultiTexCoord2fARB = NULL;
717 qglActiveTextureARB = NULL;
718 qglClientActiveTextureARB = NULL;
719 ri.Printf( PRINT_ALL, "...not using GL_ARB_multitexture, < 2 texture units\n" );
720 }
721 }
722 }
723 else
724 {
725 ri.Printf( PRINT_ALL, "...ignoring GL_ARB_multitexture\n" );
726 }
727 }
728 else
729 {
730 ri.Printf( PRINT_ALL, "...GL_ARB_multitexture not found\n" );
731 }
732
733 // GL_EXT_compiled_vertex_array
734 if ( SDL_GL_ExtensionSupported( "GL_EXT_compiled_vertex_array" ) )
735 {
736 if ( r_ext_compiled_vertex_array->value )
737 {
738 ri.Printf( PRINT_ALL, "...using GL_EXT_compiled_vertex_array\n" );
739 qglLockArraysEXT = ( void ( APIENTRY * )( GLint, GLint ) ) SDL_GL_GetProcAddress( "glLockArraysEXT" );
740 qglUnlockArraysEXT = ( void ( APIENTRY * )( void ) ) SDL_GL_GetProcAddress( "glUnlockArraysEXT" );
741 if (!qglLockArraysEXT || !qglUnlockArraysEXT)
742 {
743 ri.Error (ERR_FATAL, "bad getprocaddress");
744 }
745 }
746 else
747 {
748 ri.Printf( PRINT_ALL, "...ignoring GL_EXT_compiled_vertex_array\n" );
749 }
750 }
751 else
752 {
753 ri.Printf( PRINT_ALL, "...GL_EXT_compiled_vertex_array not found\n" );
754 }
755
756 textureFilterAnisotropic = qfalse;
757 if ( SDL_GL_ExtensionSupported( "GL_EXT_texture_filter_anisotropic" ) )
758 {
759 if ( r_ext_texture_filter_anisotropic->integer ) {
760 qglGetIntegerv( GL_MAX_TEXTURE_MAX_ANISOTROPY_EXT, (GLint *)&maxAnisotropy );
761 if ( maxAnisotropy <= 0 ) {
762 ri.Printf( PRINT_ALL, "...GL_EXT_texture_filter_anisotropic not properly supported!\n" );
763 maxAnisotropy = 0;
764 }
765 else
766 {
767 ri.Printf( PRINT_ALL, "...using GL_EXT_texture_filter_anisotropic (max: %i)\n", maxAnisotropy );
768 textureFilterAnisotropic = qtrue;
769 }
770 }
771 else
772 {
773 ri.Printf( PRINT_ALL, "...ignoring GL_EXT_texture_filter_anisotropic\n" );
774 }
775 }
776 else
777 {
778 ri.Printf( PRINT_ALL, "...GL_EXT_texture_filter_anisotropic not found\n" );
779 }
780}

784/*
785===============
786GLimp_Init
787
788This routine is responsible for initializing the OS specific portions
789of OpenGL
790===============
791*/
792void GLimp_Init( qboolean coreContext)
793{
794 ri.Printf( PRINT_DEVELOPER, "Glimp_Init( )\n" );
795
796 r_allowSoftwareGL = ri.Cvar_Get( "r_allowSoftwareGL", "0", CVAR_LATCH );
797 r_sdlDriver = ri.Cvar_Get( "r_sdlDriver", "", CVAR_ROM );
798 r_allowResize = ri.Cvar_Get( "r_allowResize", "0", CVAR_ARCHIVE | CVAR_LATCH );
799 r_centerWindow = ri.Cvar_Get( "r_centerWindow", "0", CVAR_ARCHIVE | CVAR_LATCH );
800
801 if( ri.Cvar_VariableIntegerValue( "com_abnormalExit" ) )
802 {
803 ri.Cvar_Set( "r_mode", va( "%d", R_MODE_FALLBACK ) );
804 ri.Cvar_Set( "r_fullscreen", "0" );
805 ri.Cvar_Set( "r_centerWindow", "0" );
806 ri.Cvar_Set( "com_abnormalExit", "0" );
807 }
808
809 ri.Sys_GLimpInit( );
810
811 // Create the window and set up the context
812 if(GLimp_StartDriverAndSetMode(r_mode->integer, r_fullscreen->integer, r_noborder->integer, coreContext))
813 goto success;
814
815 // Try again, this time in a platform specific "safe mode"
816 ri.Sys_GLimpSafeInit( );
817
818 if(GLimp_StartDriverAndSetMode(r_mode->integer, r_fullscreen->integer, qfalse, coreContext))
819 goto success;
820
821 // Finally, try the default screen resolution
822 if( r_mode->integer != R_MODE_FALLBACK )
823 {
824 ri.Printf( PRINT_ALL, "Setting r_mode %d failed, falling back on r_mode %d\n",
825 r_mode->integer, R_MODE_FALLBACK );
826
827 if(GLimp_StartDriverAndSetMode(R_MODE_FALLBACK, qfalse, qfalse, coreContext))
828 goto success;
829 }
830
831 // Nothing worked, give up
832 ri.Error( ERR_FATAL, "GLimp_Init() - could not load OpenGL subsystem" );
833
834success:
835 // These values force the UI to disable driver selection
836 glConfig.driverType = GLDRV_ICD;
837 glConfig.hardwareType = GLHW_GENERIC;
838
839 // Only using SDL_SetWindowBrightness to determine if hardware gamma is supported
840 glConfig.deviceSupportsGamma = !r_ignorehwgamma->integer &&
841 SDL_SetWindowBrightness( SDL_window, 1.0f ) >= 0;
842
843 // get our config strings
844 Q_strncpyz( glConfig.vendor_string, (char *) qglGetString (GL_VENDOR), sizeof( glConfig.vendor_string ) );
845 Q_strncpyz( glConfig.renderer_string, (char *) qglGetString (GL_RENDERER), sizeof( glConfig.renderer_string ) );
846 if (*glConfig.renderer_string && glConfig.renderer_string[strlen(glConfig.renderer_string) - 1] == '\n')
847 glConfig.renderer_string[strlen(glConfig.renderer_string) - 1] = 0;
848 Q_strncpyz( glConfig.version_string, (char *) qglGetString (GL_VERSION), sizeof( glConfig.version_string ) );
849 if (qglGetString(GL_EXTENSIONS))
850 Q_strncpyz( glConfig.extensions_string, (char *) qglGetString (GL_EXTENSIONS), sizeof( glConfig.extensions_string ) );
851 else
852 Q_strncpyz( glConfig.extensions_string, "Not available (core context, fixme)", sizeof( glConfig.extensions_string ) );
853
854 // initialize extensions
855 GLimp_InitExtensions( );
856
857 ri.Cvar_Get( "r_availableModes", "", CVAR_ROM );
858
859 // This depends on SDL_INIT_VIDEO, hence having it here
860 ri.IN_Init( SDL_window );
861}

864/*
865===============
866GLimp_EndFrame
867
868Responsible for doing a swapbuffers
869===============
870*/
871void GLimp_EndFrame( void )
872{
873 // don't flip if drawing to front buffer
874 if ( Q_stricmp( r_drawBuffer->string, "GL_FRONT" ) != 0 )
875 {
876 SDL_GL_SwapWindow( SDL_window );
877 }
878
879 if( r_fullscreen->modified )
880 {
881 int fullscreen;
882 qboolean needToToggle;
883 qboolean sdlToggled = qfalse;
884
885 // Find out the current state
886 fullscreen = !!( SDL_GetWindowFlags( SDL_window ) & SDL_WINDOW_FULLSCREEN );
887
888 if( r_fullscreen->integer && ri.Cvar_VariableIntegerValue( "in_nograb" ) )
889 {
890 ri.Printf( PRINT_ALL, "Fullscreen not allowed with in_nograb 1\n");
891 ri.Cvar_Set( "r_fullscreen", "0" );
892 r_fullscreen->modified = qfalse;
893 }
894
895 // Is the state we want different from the current state?
896 needToToggle = !!r_fullscreen->integer != fullscreen;
897
898 if( needToToggle )
899 {
900 sdlToggled = SDL_SetWindowFullscreen( SDL_window, r_fullscreen->integer ) >= 0;
901
902 // SDL_WM_ToggleFullScreen didn't work, so do it the slow way
903 if( !sdlToggled )
904 ri.Cmd_ExecuteText(EXEC_APPEND, "vid_restart\n");
905
906 ri.IN_Restart( );
907 }
908
909 r_fullscreen->modified = qfalse;
910 }

149void UI_LoadBestScores(const char *map, int game)
150{
151 char fileName[MAX_QPATH];
152 fileHandle_t f;
153 postGameInfo_t newInfo;
154 int protocol, protocolLegacy;
155
156 memset(&newInfo, 0, sizeof(postGameInfo_t));
157 Com_sprintf(fileName, MAX_QPATH, "games/%s_%i.game", map, game);
158 if (trap_FS_FOpenFile(fileName, &f, FS_READ) >= 0) {
159 int size = 0;
160 trap_FS_Read(&size, sizeof(int), f);
161 if (size == sizeof(postGameInfo_t)) {
162 trap_FS_Read(&newInfo, sizeof(postGameInfo_t), f);
163 }
164 trap_FS_FCloseFile(f);
165 }
166 UI_SetBestScores(&newInfo, qfalse);
167
168 uiInfo.demoAvailable = qfalse;
169
170 protocolLegacy = trap_Cvar_VariableValue("com_legacyprotocol");
171 protocol = trap_Cvar_VariableValue("com_protocol");
172
173 if(!protocol)
174 protocol = trap_Cvar_VariableValue("protocol");
175 if(protocolLegacy == protocol)
176 protocolLegacy = 0;
177
178 Com_sprintf(fileName, MAX_QPATH, "demos/%s_%d.%s%d", map, game, DEMOEXT, protocol);
179 if(trap_FS_FOpenFile(fileName, &f, FS_READ) >= 0)
180 {
181 uiInfo.demoAvailable = qtrue;
182 trap_FS_FCloseFile(f);
183 }
184 else if(protocolLegacy > 0)
185 {
186 Com_sprintf(fileName, MAX_QPATH, "demos/%s_%d.%s%d", map, game, DEMOEXT, protocolLegacy);
187 if (trap_FS_FOpenFile(fileName, &f, FS_READ) >= 0)
188 {
189 uiInfo.demoAvailable = qtrue;
190 trap_FS_FCloseFile(f);
191 }
192 }
193}

195/*
196===============
197UI_ClearScores
198===============
199*/
200void UI_ClearScores(void) {
201 char gameList[4096];
202 char *gameFile;
203 int i, len, count, size;
204 fileHandle_t f;
205 postGameInfo_t newInfo;
206
207 count = trap_FS_GetFileList( "games", "game", gameList, sizeof(gameList) );
208
209 size = sizeof(postGameInfo_t);
210 memset(&newInfo, 0, size);
211
212 if (count > 0) {
213 gameFile = gameList;
214 for ( i = 0; i < count; i++ ) {
215 len = strlen(gameFile);
216 if (trap_FS_FOpenFile(va("games/%s",gameFile), &f, FS_WRITE) >= 0) {
217 trap_FS_Write(&size, sizeof(int), f);
218 trap_FS_Write(&newInfo, size, f);
219 trap_FS_FCloseFile(f);
220 }
221 gameFile += len + 1;
222 }
223 }
224
225 UI_SetBestScores(&newInfo, qfalse);
226
227}

235/*
236=======================
237UI_CalcPostGameStats
238=======================
239*/
240static void UI_CalcPostGameStats( void ) {
241 char map[MAX_QPATH];
242 char fileName[MAX_QPATH];
243 char info[MAX_INFO_STRING];
244 fileHandle_t f;
245 int size, game, time, adjustedTime;
246 postGameInfo_t oldInfo;
247 postGameInfo_t newInfo;
248 qboolean newHigh = qfalse;
249
250 trap_GetConfigString( CS_SERVERINFO, info, sizeof(info) );
251 Q_strncpyz( map, Info_ValueForKey( info, "mapname" ), sizeof(map) );
252 game = atoi(Info_ValueForKey(info, "g_gametype"));
253
254 // compose file name
255 Com_sprintf(fileName, MAX_QPATH, "games/%s_%i.game", map, game);
256 // see if we have one already
257 memset(&oldInfo, 0, sizeof(postGameInfo_t));
258 if (trap_FS_FOpenFile(fileName, &f, FS_READ) >= 0) {
259 // if so load it
260 size = 0;
261 trap_FS_Read(&size, sizeof(int), f);
262 if (size == sizeof(postGameInfo_t)) {
263 trap_FS_Read(&oldInfo, sizeof(postGameInfo_t), f);
264 }
265 trap_FS_FCloseFile(f);
266 }
267
268 newInfo.accuracy = atoi(UI_Argv(3));
269 newInfo.impressives = atoi(UI_Argv(4));
270 newInfo.excellents = atoi(UI_Argv(5));
271 newInfo.defends = atoi(UI_Argv(6));
272 newInfo.assists = atoi(UI_Argv(7));
273 newInfo.gauntlets = atoi(UI_Argv(8));
274 newInfo.baseScore = atoi(UI_Argv(9));
275 newInfo.perfects = atoi(UI_Argv(10));
276 newInfo.redScore = atoi(UI_Argv(11));
277 newInfo.blueScore = atoi(UI_Argv(12));
278 time = atoi(UI_Argv(13));
279 newInfo.captures = atoi(UI_Argv(14));
280
281 newInfo.time = (time - trap_Cvar_VariableValue("ui_matchStartTime")) / 1000;
282 adjustedTime = uiInfo.mapList[ui_currentMap.integer].timeToBeat[game];
283 if (newInfo.time < adjustedTime) {
284 newInfo.timeBonus = (adjustedTime - newInfo.time) * 10;
285 } else {
286 newInfo.timeBonus = 0;
287 }
288
289 if (newInfo.redScore > newInfo.blueScore && newInfo.blueScore <= 0) {
290 newInfo.shutoutBonus = 100;
291 } else {
292 newInfo.shutoutBonus = 0;
293 }
294
295 newInfo.skillBonus = trap_Cvar_VariableValue("g_spSkill");
296 if (newInfo.skillBonus <= 0) {
297 newInfo.skillBonus = 1;
298 }
299 newInfo.score = newInfo.baseScore + newInfo.shutoutBonus + newInfo.timeBonus;
300 newInfo.score *= newInfo.skillBonus;
301
302 // see if the score is higher for this one
303 newHigh = (newInfo.redScore > newInfo.blueScore && newInfo.score > oldInfo.score);
304
305 if (newHigh) {
306 // if so write out the new one
307 uiInfo.newHighScoreTime = uiInfo.uiDC.realTime + 20000;
308 if (trap_FS_FOpenFile(fileName, &f, FS_WRITE) >= 0) {
309 size = sizeof(postGameInfo_t);
310 trap_FS_Write(&size, sizeof(int), f);
311 trap_FS_Write(&newInfo, sizeof(postGameInfo_t), f);
312 trap_FS_FCloseFile(f);
313 }
314 }
315
316 if (newInfo.time < oldInfo.time) {
317 uiInfo.newBestTime = uiInfo.uiDC.realTime + 20000;
318 }
319
320 // put back all the ui overrides
321 trap_Cvar_Set("capturelimit", UI_Cvar_VariableString("ui_saveCaptureLimit"));
322 trap_Cvar_Set("fraglimit", UI_Cvar_VariableString("ui_saveFragLimit"));
323 trap_Cvar_Set("cg_drawTimer", UI_Cvar_VariableString("ui_drawTimer"));
324 trap_Cvar_Set("g_doWarmup", UI_Cvar_VariableString("ui_doWarmup"));
325 trap_Cvar_Set("g_Warmup", UI_Cvar_VariableString("ui_Warmup"));
326 trap_Cvar_Set("sv_pure", UI_Cvar_VariableString("ui_pure"));
327 trap_Cvar_Set("g_friendlyFire", UI_Cvar_VariableString("ui_friendlyFire"));
328
329 UI_SetBestScores(&newInfo, qtrue);
330 UI_ShowPostGame(newHigh);
331
332
333}

338UI_ConsoleCommand
339=================
340*/
341qboolean UI_ConsoleCommand( int realTime ) {
342 char *cmd;
343
344 uiInfo.uiDC.frameTime = realTime - uiInfo.uiDC.realTime;
345 uiInfo.uiDC.realTime = realTime;
346
347 cmd = UI_Argv( 0 );
348
349 // ensure minimum menu data is available
350 //Menu_Cache();
351
352 if ( Q_stricmp (cmd, "ui_test") == 0 ) {
353 UI_ShowPostGame(qtrue);
354 return qtrue;
355 }
356
357 if ( Q_stricmp (cmd, "ui_report") == 0 ) {
358 UI_Report();
359 return qtrue;
360 }
361
362 if ( Q_stricmp (cmd, "ui_load") == 0 ) {
363 UI_Load();
364 return qtrue;
365 }
366
367 if ( Q_stricmp (cmd, "remapShader") == 0 ) {
368 if (trap_Argc() == 4) {
369 char shader1[MAX_QPATH];
370 char shader2[MAX_QPATH];
371 char shader3[MAX_QPATH];
372
373 Q_strncpyz(shader1, UI_Argv(1), sizeof(shader1));
374 Q_strncpyz(shader2, UI_Argv(2), sizeof(shader2));
375 Q_strncpyz(shader3, UI_Argv(3), sizeof(shader3));
376
377 trap_R_RemapShader(shader1, shader2, shader3);
378 return qtrue;
379 }
380 }
381
382 if ( Q_stricmp (cmd, "postgame") == 0 ) {
383 UI_CalcPostGameStats();
384 return qtrue;
385 }
386
387 if ( Q_stricmp (cmd, "ui_cache") == 0 ) {
388 UI_Cache_f();
389 return qtrue;
390 }
391
392 if ( Q_stricmp (cmd, "ui_teamOrders") == 0 ) {
393 //UI_TeamOrdersMenu_f();
394 return qtrue;
395 }
396
397
398 if ( Q_stricmp (cmd, "ui_cdkey") == 0 ) {
399 //UI_CDKeyMenu_f();
400 return qtrue;
401 }
402
403 return qfalse;
404}

421void UI_AdjustFrom640( float *x, float *y, float *w, float *h ) {

437void UI_DrawHandlePic( float x, float y, float w, float h, qhandle_t hShader ) {
438 float s0;
439 float s1;
440 float t0;
441 float t1;
442
443 if( w < 0 ) { // flip about vertical
444 w = -w;
445 s0 = 1;
446 s1 = 0;
447 }
448 else {
449 s0 = 0;
450 s1 = 1;
451 }
452
453 if( h < 0 ) { // flip about horizontal
454 h = -h;
455 t0 = 1;
456 t1 = 0;
457 }
458 else {
459 t0 = 0;
460 t1 = 1;
461 }
462
463 UI_AdjustFrom640( &x, &y, &w, &h );
464 trap_R_DrawStretchPic( x, y, w, h, s0, t0, s1, t1, hShader );
465}

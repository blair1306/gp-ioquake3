31static intptr_t (QDECL *syscall)( intptr_t arg, ... ) = (intptr_t (QDECL *)( intptr_t, ...))-1;
32
33Q_EXPORT void dllEntry( intptr_t (QDECL *syscallptr)( intptr_t arg,... ) ) {
34 syscall = syscallptr;
35}

37int PASSFLOAT( float x ) {
38 floatint_t fi;
39 fi.f = x;
40 return fi.i;
41}

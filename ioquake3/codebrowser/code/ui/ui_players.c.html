49/*
50===============
51UI_PlayerInfo_SetWeapon
52===============
53*/
54static void UI_PlayerInfo_SetWeapon( playerInfo_t *pi, weapon_t weaponNum ) {
55 gitem_t * item;
56 char path[MAX_QPATH];
57
58 pi->currentWeapon = weaponNum;
59tryagain:
60 pi->realWeapon = weaponNum;
61 pi->weaponModel = 0;
62 pi->barrelModel = 0;
63 pi->flashModel = 0;
64
65 if ( weaponNum == WP_NONE ) {
66 return;
67 }
68
69 for ( item = bg_itemlist + 1; item->classname ; item++ ) {
70 if ( item->giType != IT_WEAPON ) {
71 continue;
72 }
73 if ( item->giTag == weaponNum ) {
74 break;
75 }
76 }
77
78 if ( item->classname ) {
79 pi->weaponModel = trap_R_RegisterModel( item->world_model[0] );
80 }
81
82 if( pi->weaponModel == 0 ) {
83 if( weaponNum == WP_MACHINEGUN ) {
84 weaponNum = WP_NONE;
85 goto tryagain;
86 }
87 weaponNum = WP_MACHINEGUN;
88 goto tryagain;
89 }
90
91 if ( weaponNum == WP_MACHINEGUN || weaponNum == WP_GAUNTLET || weaponNum == WP_BFG ) {
92 COM_StripExtension( item->world_model[0], path, sizeof(path) );
93 Q_strcat( path, sizeof(path), "_barrel.md3" );
94 pi->barrelModel = trap_R_RegisterModel( path );
95 }
96
97 COM_StripExtension( item->world_model[0], path, sizeof(path) );
98 Q_strcat( path, sizeof(path), "_flash.md3" );
99 pi->flashModel = trap_R_RegisterModel( path );
100
101 switch( weaponNum ) {
102 case WP_GAUNTLET:
103 MAKERGB( pi->flashDlightColor, 0.6f, 0.6f, 1 );
104 break;
105
106 case WP_MACHINEGUN:
107 MAKERGB( pi->flashDlightColor, 1, 1, 0 );
108 break;
109
110 case WP_SHOTGUN:
111 MAKERGB( pi->flashDlightColor, 1, 1, 0 );
112 break;
113
114 case WP_GRENADE_LAUNCHER:
115 MAKERGB( pi->flashDlightColor, 1, 0.7f, 0.5f );
116 break;
117
118 case WP_ROCKET_LAUNCHER:
119 MAKERGB( pi->flashDlightColor, 1, 0.75f, 0 );
120 break;
121
122 case WP_LIGHTNING:
123 MAKERGB( pi->flashDlightColor, 0.6f, 0.6f, 1 );
124 break;
125
126 case WP_RAILGUN:
127 MAKERGB( pi->flashDlightColor, 1, 0.5f, 0 );
128 break;
129
130 case WP_PLASMAGUN:
131 MAKERGB( pi->flashDlightColor, 0.6f, 0.6f, 1 );
132 break;
133
134 case WP_BFG:
135 MAKERGB( pi->flashDlightColor, 1, 0.7f, 1 );
136 break;
137
138 case WP_GRAPPLING_HOOK:
139 MAKERGB( pi->flashDlightColor, 0.6f, 0.6f, 1 );
140 break;
141
142 default:
143 MAKERGB( pi->flashDlightColor, 1, 1, 1 );
144 break;
145 }
146}

210/*
211===============
212UI_TorsoSequencing
213===============
214*/
215static void UI_TorsoSequencing( playerInfo_t *pi ) {
216 int currentAnim;
217
218 currentAnim = pi->torsoAnim & ~ANIM_TOGGLEBIT;
219
220 if ( pi->weapon != pi->currentWeapon ) {
221 if ( currentAnim != TORSO_DROP ) {
222 pi->torsoAnimationTimer = UI_TIMER_WEAPON_SWITCH;
223 UI_ForceTorsoAnim( pi, TORSO_DROP );
224 }
225 }
226
227 if ( pi->torsoAnimationTimer > 0 ) {
228 return;
229 }
230
231 if( currentAnim == TORSO_GESTURE ) {
232 UI_SetTorsoAnim( pi, TORSO_STAND );
233 return;
234 }
235
236 if( currentAnim == TORSO_ATTACK || currentAnim == TORSO_ATTACK2 ) {
237 UI_SetTorsoAnim( pi, TORSO_STAND );
238 return;
239 }
240
241 if ( currentAnim == TORSO_DROP ) {
242 UI_PlayerInfo_SetWeapon( pi, pi->weapon );
243 pi->torsoAnimationTimer = UI_TIMER_WEAPON_SWITCH;
244 UI_ForceTorsoAnim( pi, TORSO_RAISE );
245 return;
246 }
247
248 if ( currentAnim == TORSO_RAISE ) {
249 UI_SetTorsoAnim( pi, TORSO_STAND );
250 return;
251 }
252}

255/*
256===============
257UI_LegsSequencing
258===============
259*/
260static void UI_LegsSequencing( playerInfo_t *pi ) {
261 int currentAnim;
262
263 currentAnim = pi->legsAnim & ~ANIM_TOGGLEBIT;
264
265 if ( pi->legsAnimationTimer > 0 ) {
266 if ( currentAnim == LEGS_JUMP ) {
267 jumpHeight = JUMP_HEIGHT * sin( M_PI * ( UI_TIMER_JUMP - pi->legsAnimationTimer ) / UI_TIMER_JUMP );
268 }
269 return;
270 }
271
272 if ( currentAnim == LEGS_JUMP ) {
273 UI_ForceLegsAnim( pi, LEGS_LAND );
274 pi->legsAnimationTimer = UI_TIMER_LAND;
275 jumpHeight = 0;
276 return;
277 }
278
279 if ( currentAnim == LEGS_LAND ) {
280 UI_SetLegsAnim( pi, LEGS_IDLE );
281 return;
282 }
283}

286/*
287======================
288UI_PositionEntityOnTag
289======================
290*/
291static void UI_PositionEntityOnTag( refEntity_t *entity, const refEntity_t *parent,
292 clipHandle_t parentModel, char *tagName ) {
293 int i;
294 orientation_t lerped;
295
296 // lerp the tag
297 trap_CM_LerpTag( &lerped, parentModel, parent->oldframe, parent->frame,
298 1.0 - parent->backlerp, tagName );
299
300 // FIXME: allow origin offsets along tag?
301 VectorCopy( parent->origin, entity->origin );
302 for ( i = 0 ; i < 3 ; i++ ) {
303 VectorMA( entity->origin, lerped.origin[i], parent->axis[i], entity->origin );
304 }
305
306 // cast away const because of compiler problems
307 MatrixMultiply( lerped.axis, ((refEntity_t*)parent)->axis, entity->axis );
308 entity->backlerp = parent->backlerp;
309}

341UI_SetLerpFrameAnimation
342===============
343*/
344static void UI_SetLerpFrameAnimation( playerInfo_t *ci, lerpFrame_t *lf, int newAnimation ) {
345 animation_t *anim;
346
347 lf->animationNumber = newAnimation;
348 newAnimation &= ~ANIM_TOGGLEBIT;
349
350 if ( newAnimation < 0 || newAnimation >= MAX_ANIMATIONS ) {
351 trap_Error( va("Bad animation number: %i", newAnimation) );
352 }
353
354 anim = &ci->animations[ newAnimation ];
355
356 lf->animation = anim;
357 lf->animationTime = lf->frameTime + anim->initialLerp;
358}

361/*
362===============
363UI_RunLerpFrame
364===============
365*/
366static void UI_RunLerpFrame( playerInfo_t *ci, lerpFrame_t *lf, int newAnimation ) {
367 int f;
368 animation_t *anim;
369
370 // see if the animation sequence is switching
371 if ( newAnimation != lf->animationNumber || !lf->animation ) {
372 UI_SetLerpFrameAnimation( ci, lf, newAnimation );
373 }
374
375 // if we have passed the current frame, move it to
376 // oldFrame and calculate a new frame
377 if ( dp_realtime >= lf->frameTime ) {
378 lf->oldFrame = lf->frame;
379 lf->oldFrameTime = lf->frameTime;
380
381 // get the next frame based on the animation
382 anim = lf->animation;
383 if ( dp_realtime < lf->animationTime ) {
384 lf->frameTime = lf->animationTime; // initial lerp
385 } else {
386 lf->frameTime = lf->oldFrameTime + anim->frameLerp;
387 }
388 f = ( lf->frameTime - lf->animationTime ) / anim->frameLerp;
389 if ( f >= anim->numFrames ) {
390 f -= anim->numFrames;
391 if ( anim->loopFrames ) {
392 f %= anim->loopFrames;
393 f += anim->numFrames - anim->loopFrames;
394 } else {
395 f = anim->numFrames - 1;
396 // the animation is stuck at the end, so it
397 // can immediately transition to another sequence
398 lf->frameTime = dp_realtime;
399 }
400 }
401 lf->frame = anim->firstFrame + f;
402 if ( dp_realtime > lf->frameTime ) {
403 lf->frameTime = dp_realtime;
404 }
405 }
406
407 if ( lf->frameTime > dp_realtime + 200 ) {
408 lf->frameTime = dp_realtime;
409 }
410
411 if ( lf->oldFrameTime > dp_realtime ) {
412 lf->oldFrameTime = dp_realtime;
413 }
414 // calculate current lerp value
415 if ( lf->frameTime == lf->oldFrameTime ) {
416 lf->backlerp = 0;
417 } else {
418 lf->backlerp = 1.0 - (float)( dp_realtime - lf->oldFrameTime ) / ( lf->frameTime - lf->oldFrameTime );
419 }
420}

423/*
424===============
425UI_PlayerAnimation
426===============
427*/
428static void UI_PlayerAnimation( playerInfo_t *pi, int *legsOld, int *legs, float *legsBackLerp,
429 int *torsoOld, int *torso, float *torsoBackLerp ) {
430
431 // legs animation
432 pi->legsAnimationTimer -= uiInfo.uiDC.frameTime;
433 if ( pi->legsAnimationTimer < 0 ) {
434 pi->legsAnimationTimer = 0;
435 }
436
437 UI_LegsSequencing( pi );
438
439 if ( pi->legs.yawing && ( pi->legsAnim & ~ANIM_TOGGLEBIT ) == LEGS_IDLE ) {
440 UI_RunLerpFrame( pi, &pi->legs, LEGS_TURN );
441 } else {
442 UI_RunLerpFrame( pi, &pi->legs, pi->legsAnim );
443 }
444 *legsOld = pi->legs.oldFrame;
445 *legs = pi->legs.frame;
446 *legsBackLerp = pi->legs.backlerp;
447
448 // torso animation
449 pi->torsoAnimationTimer -= uiInfo.uiDC.frameTime;
450 if ( pi->torsoAnimationTimer < 0 ) {
451 pi->torsoAnimationTimer = 0;
452 }
453
454 UI_TorsoSequencing( pi );
455
456 UI_RunLerpFrame( pi, &pi->torso, pi->torsoAnim );
457 *torsoOld = pi->torso.oldFrame;
458 *torso = pi->torso.frame;
459 *torsoBackLerp = pi->torso.backlerp;
460}

463/*
464==================
465UI_SwingAngles
466==================
467*/
468static void UI_SwingAngles( float destination, float swingTolerance, float clampTolerance,
469 float speed, float *angle, qboolean *swinging ) {
470 float swing;
471 float move;
472 float scale;
473
474 if ( !*swinging ) {
475 // see if a swing should be started
476 swing = AngleSubtract( *angle, destination );
477 if ( swing > swingTolerance || swing < -swingTolerance ) {
478 *swinging = qtrue;
479 }
480 }
481
482 if ( !*swinging ) {
483 return;
484 }
485
486 // modify the speed depending on the delta
487 // so it doesn't seem so linear
488 swing = AngleSubtract( destination, *angle );
489 scale = fabs( swing );
490 if ( scale < swingTolerance * 0.5 ) {
491 scale = 0.5;
492 } else if ( scale < swingTolerance ) {
493 scale = 1.0;
494 } else {
495 scale = 2.0;
496 }
497
498 // swing towards the destination angle
499 if ( swing >= 0 ) {
500 move = uiInfo.uiDC.frameTime * scale * speed;
501 if ( move >= swing ) {
502 move = swing;
503 *swinging = qfalse;
504 }
505 *angle = AngleMod( *angle + move );
506 } else if ( swing < 0 ) {
507 move = uiInfo.uiDC.frameTime * scale * -speed;
508 if ( move <= swing ) {
509 move = swing;
510 *swinging = qfalse;
511 }
512 *angle = AngleMod( *angle + move );
513 }
514
515 // clamp to no more than tolerance
516 swing = AngleSubtract( destination, *angle );
517 if ( swing > clampTolerance ) {
518 *angle = AngleMod( destination - (clampTolerance - 1) );
519 } else if ( swing < -clampTolerance ) {
520 *angle = AngleMod( destination + (clampTolerance - 1) );
521 }
522}

525/*
526======================
527UI_MovedirAdjustment
528======================
529*/
530static float UI_MovedirAdjustment( playerInfo_t *pi ) {
531 vec3_t relativeAngles;
532 vec3_t moveVector;
533
534 VectorSubtract( pi->viewAngles, pi->moveAngles, relativeAngles );
535 AngleVectors( relativeAngles, moveVector, NULL, NULL );
536 if ( Q_fabs( moveVector[0] ) < 0.01 ) {
537 moveVector[0] = 0.0;
538 }
539 if ( Q_fabs( moveVector[1] ) < 0.01 ) {
540 moveVector[1] = 0.0;
541 }
542
543 if ( moveVector[1] == 0 && moveVector[0] > 0 ) {
544 return 0;
545 }
546 if ( moveVector[1] < 0 && moveVector[0] > 0 ) {
547 return 22;
548 }
549 if ( moveVector[1] < 0 && moveVector[0] == 0 ) {
550 return 45;
551 }
552 if ( moveVector[1] < 0 && moveVector[0] < 0 ) {
553 return -22;
554 }
555 if ( moveVector[1] == 0 && moveVector[0] < 0 ) {
556 return 0;
557 }
558 if ( moveVector[1] > 0 && moveVector[0] < 0 ) {
559 return 22;
560 }
561 if ( moveVector[1] > 0 && moveVector[0] == 0 ) {
562 return -45;
563 }
564
565 return -22;
566}

569/*
570===============
571UI_PlayerAngles
572===============
573*/
574static void UI_PlayerAngles( playerInfo_t *pi, vec3_t legs[3], vec3_t torso[3], vec3_t head[3] ) {
575 vec3_t legsAngles, torsoAngles, headAngles;
576 float dest;
577 float adjust;
578
579 VectorCopy( pi->viewAngles, headAngles );
580 headAngles[YAW] = AngleMod( headAngles[YAW] );
581 VectorClear( legsAngles );
582 VectorClear( torsoAngles );
583
584 // --------- yaw -------------
585
586 // allow yaw to drift a bit
587 if ( ( pi->legsAnim & ~ANIM_TOGGLEBIT ) != LEGS_IDLE
588 || ( pi->torsoAnim & ~ANIM_TOGGLEBIT ) != TORSO_STAND ) {
589 // if not standing still, always point all in the same direction
590 pi->torso.yawing = qtrue; // always center
591 pi->torso.pitching = qtrue; // always center
592 pi->legs.yawing = qtrue; // always center
593 }
594
595 // adjust legs for movement dir
596 adjust = UI_MovedirAdjustment( pi );
597 legsAngles[YAW] = headAngles[YAW] + adjust;
598 torsoAngles[YAW] = headAngles[YAW] + 0.25 * adjust;
599
600
601 // torso
602 UI_SwingAngles( torsoAngles[YAW], 25, 90, SWINGSPEED, &pi->torso.yawAngle, &pi->torso.yawing );
603 UI_SwingAngles( legsAngles[YAW], 40, 90, SWINGSPEED, &pi->legs.yawAngle, &pi->legs.yawing );
604
605 torsoAngles[YAW] = pi->torso.yawAngle;
606 legsAngles[YAW] = pi->legs.yawAngle;
607
608 // --------- pitch -------------
609
610 // only show a fraction of the pitch angle in the torso
611 if ( headAngles[PITCH] > 180 ) {
612 dest = (-360 + headAngles[PITCH]) * 0.75;
613 } else {
614 dest = headAngles[PITCH] * 0.75;
615 }
616 UI_SwingAngles( dest, 15, 30, 0.1f, &pi->torso.pitchAngle, &pi->torso.pitching );
617 torsoAngles[PITCH] = pi->torso.pitchAngle;
618
619 // pull the angles back out of the hierarchial chain
620 AnglesSubtract( headAngles, torsoAngles, headAngles );
621 AnglesSubtract( torsoAngles, legsAngles, torsoAngles );
622 AnglesToAxis( legsAngles, legs );
623 AnglesToAxis( torsoAngles, torso );
624 AnglesToAxis( headAngles, head );
625}

647/*
648======================
649UI_MachinegunSpinAngle
650======================
651*/
652float UI_MachinegunSpinAngle( playerInfo_t *pi ) {
653 int delta;
654 float angle;
655 float speed;
656 int torsoAnim;
657
658 delta = dp_realtime - pi->barrelTime;
659 if ( pi->barrelSpinning ) {
660 angle = pi->barrelAngle + delta * SPIN_SPEED;
661 } else {
662 if ( delta > COAST_TIME ) {
663 delta = COAST_TIME;
664 }
665
666 speed = 0.5 * ( SPIN_SPEED + (float)( COAST_TIME - delta ) / COAST_TIME );
667 angle = pi->barrelAngle + delta * speed;
668 }
669
670 torsoAnim = pi->torsoAnim & ~ANIM_TOGGLEBIT;
671 if( torsoAnim == TORSO_ATTACK2 ) {
672 torsoAnim = TORSO_ATTACK;
673 }
674 if ( pi->barrelSpinning == !(torsoAnim == TORSO_ATTACK) ) {
675 pi->barrelTime = dp_realtime;
676 pi->barrelAngle = AngleMod( angle );
677 pi->barrelSpinning = !!(torsoAnim == TORSO_ATTACK);
678 }
679
680 return angle;
681}

684/*
685===============
686UI_DrawPlayer
687===============
688*/
689void UI_DrawPlayer( float x, float y, float w, float h, playerInfo_t *pi, int time ) {
690 refdef_t refdef;
691 refEntity_t legs = {0};
692 refEntity_t torso = {0};
693 refEntity_t head = {0};
694 refEntity_t gun = {0};
695 refEntity_t barrel = {0};
696 refEntity_t flash = {0};
697 vec3_t origin;
698 int renderfx;
699 vec3_t mins = {-16, -16, -24};
700 vec3_t maxs = {16, 16, 32};
701 float len;
702 float xx;
703
704 if ( !pi->legsModel || !pi->torsoModel || !pi->headModel || !pi->animations[0].numFrames ) {
705 return;
706 }
707
708 // this allows the ui to cache the player model on the main menu
709 if (w == 0 || h == 0) {
710 return;
711 }
712
713 dp_realtime = time;
714
715 if ( pi->pendingWeapon != WP_NUM_WEAPONS && dp_realtime > pi->weaponTimer ) {
716 pi->weapon = pi->pendingWeapon;
717 pi->lastWeapon = pi->pendingWeapon;
718 pi->pendingWeapon = WP_NUM_WEAPONS;
719 pi->weaponTimer = 0;
720 if( pi->currentWeapon != pi->weapon ) {
721 trap_S_StartLocalSound( weaponChangeSound, CHAN_LOCAL );
722 }
723 }
724
725 UI_AdjustFrom640( &x, &y, &w, &h );
726
727 y -= jumpHeight;
728
729 memset( &refdef, 0, sizeof( refdef ) );
730 memset( &legs, 0, sizeof(legs) );
731 memset( &torso, 0, sizeof(torso) );
732 memset( &head, 0, sizeof(head) );
733
734 refdef.rdflags = RDF_NOWORLDMODEL;
735
736 AxisClear( refdef.viewaxis );
737
738 refdef.x = x;
739 refdef.y = y;
740 refdef.width = w;
741 refdef.height = h;
742
743 refdef.fov_x = (int)((float)refdef.width / uiInfo.uiDC.xscale / 640.0f * 90.0f);
744 xx = refdef.width / uiInfo.uiDC.xscale / tan( refdef.fov_x / 360 * M_PI );
745 refdef.fov_y = atan2( refdef.height / uiInfo.uiDC.yscale, xx );
746 refdef.fov_y *= ( 360 / (float)M_PI );
747
748 // calculate distance so the player nearly fills the box
749 len = 0.7 * ( maxs[2] - mins[2] );
750 origin[0] = len / tan( DEG2RAD(refdef.fov_x) * 0.5 );
751 origin[1] = 0.5 * ( mins[1] + maxs[1] );
752 origin[2] = -0.5 * ( mins[2] + maxs[2] );
753
754 refdef.time = dp_realtime;
755
756 trap_R_ClearScene();
757
758 // get the rotation information
759 UI_PlayerAngles( pi, legs.axis, torso.axis, head.axis );
760
761 // get the animation state (after rotation, to allow feet shuffle)
762 UI_PlayerAnimation( pi, &legs.oldframe, &legs.frame, &legs.backlerp,
763 &torso.oldframe, &torso.frame, &torso.backlerp );
764
765 renderfx = RF_LIGHTING_ORIGIN | RF_NOSHADOW;

1013/*
1014======================
1015UI_ParseAnimationFile
1016======================
1017*/
1018static qboolean UI_ParseAnimationFile( const char *filename, animation_t *animations ) {
1019 char *text_p, *prev;
1020 int len;
1021 int i;
1022 char *token;
1023 float fps;
1024 int skip;
1025 char text[20000];
1026 fileHandle_t f;
1027
1028 memset( animations, 0, sizeof( animation_t ) * MAX_ANIMATIONS );
1029
1030 // load the file
1031 len = trap_FS_FOpenFile( filename, &f, FS_READ );
1032 if ( len <= 0 ) {
1033 return qfalse;
1034 }
1035 if ( len >= ( sizeof( text ) - 1 ) ) {
1036 Com_Printf( "File %s too long\n", filename );
1037 trap_FS_FCloseFile( f );
1038 return qfalse;
1039 }
1040 trap_FS_Read( text, len, f );
1041 text[len] = 0;
1042 trap_FS_FCloseFile( f );
1043
1044 COM_Compress(text);
1045
1046 // parse the text
1047 text_p = text;
1048 skip = 0; // quite the compiler warning
1049
1050 // read optional parameters
1051 while ( 1 ) {
1052 prev = text_p; // so we can unget
1053 token = COM_Parse( &text_p );
1054 if ( !token[0] ) {
1055 break;
1056 }
1057 if ( !Q_stricmp( token, "footsteps" ) ) {
1058 token = COM_Parse( &text_p );
1059 if ( !token[0] ) {
1060 break;
1061 }
1062 continue;
1063 } else if ( !Q_stricmp( token, "headoffset" ) ) {
1064 for ( i = 0 ; i < 3 ; i++ ) {
1065 token = COM_Parse( &text_p );
1066 if ( !token[0] ) {
1067 break;
1068 }
1069 }
1070 continue;
1071 } else if ( !Q_stricmp( token, "sex" ) ) {
1072 token = COM_Parse( &text_p );
1073 if ( !token[0] ) {
1074 break;
1075 }
1076 continue;
1077 }
1078
1079 // if it is a number, start parsing animations
1080 if ( token[0] >= '0' && token[0] <= '9' ) {
1081 text_p = prev; // unget the token
1082 break;
1083 }
1084
1085 Com_Printf( "unknown token '%s' in %s\n", token, filename );
1086 }
1087
1088 // read information for each frame
1089 for ( i = 0 ; i < MAX_ANIMATIONS ; i++ ) {
1090
1091 token = COM_Parse( &text_p );
1092 if ( !token[0] ) {
1093 break;
1094 }
1095 animations[i].firstFrame = atoi( token );
1096 // leg only frames are adjusted to not count the upper body only frames
1097 if ( i == LEGS_WALKCR ) {
1098 skip = animations[LEGS_WALKCR].firstFrame - animations[TORSO_GESTURE].firstFrame;
1099 }
1100 if ( i >= LEGS_WALKCR ) {
1101 animations[i].firstFrame -= skip;
1102 }
1103
1104 token = COM_Parse( &text_p );
1105 if ( !token[0] ) {
1106 break;
1107 }
1108 animations[i].numFrames = atoi( token );
1109
1110 token = COM_Parse( &text_p );
1111 if ( !token[0] ) {
1112 break;
1113 }
1114 animations[i].loopFrames = atoi( token );
1115
1116 token = COM_Parse( &text_p );
1117 if ( !token[0] ) {
1118 break;
1119 }
1120 fps = atof( token );
1121 if ( fps == 0 ) {
1122 fps = 1;
1123 }
1124 animations[i].frameLerp = 1000 / fps;
1125 animations[i].initialLerp = 1000 / fps;
1126 }
1127
1128 if ( i != MAX_ANIMATIONS ) {
1129 Com_Printf( "Error parsing animation file: %s\n", filename );
1130 return qfalse;
1131 }
1132
1133 return qtrue;
1134}

1172 } else {
1173 Q_strncpyz( headSkinName, slash + 1, sizeof( skinName ) );
1174 *slash = '\0';
1175 }
1176
1177 // load cmodels before models so filecache works
1178
1179 Com_sprintf( filename, sizeof( filename ), "models/players/%s/lower.md3", modelName );

1208 if ( !pi->headModel && headModelName[0] != '*') {

1219 if ( !UI_RegisterClientSkin( pi, modelName, skinName, headModelName, headSkinName, teamName) ) {

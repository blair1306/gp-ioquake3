22// tr_noise.c
23#include "tr_common.h"
24
25#define NOISE_SIZE 256
26#define NOISE_MASK ( NOISE_SIZE - 1 )
27
28#define VAL( a ) s_noise_perm[ ( a ) & ( NOISE_MASK )]
29#define INDEX( x, y, z, t ) VAL( x + VAL( y + VAL( z + VAL( t ) ) ) )
30
31static float s_noise_table[NOISE_SIZE];
32static int s_noise_perm[NOISE_SIZE];
33
34static float GetNoiseValue( int x, int y, int z, int t )
35{
36 int index = INDEX( ( int ) x, ( int ) y, ( int ) z, ( int ) t );
37
38 return s_noise_table[index];
39}
40
41void R_NoiseInit( void )
42{
43 int i;
44
45 for ( i = 0; i < NOISE_SIZE; i++ )
46 {
47 s_noise_table[i] = ( float ) ( ( ( rand() / ( float ) RAND_MAX ) * 2.0 - 1.0 ) );
48 s_noise_perm[i] = ( unsigned char ) ( rand() / ( float ) RAND_MAX * 255 );
49 }
50}

52float R_NoiseGet4f( float x, float y, float z, double t )
53{
54 int i;
55 int ix, iy, iz, it;
56 float fx, fy, fz, ft;
57 float front[4];
58 float back[4];
59 float fvalue, bvalue, value[2], finalvalue;
60
61 ix = ( int ) floor( x );
62 fx = x - ix;
63 iy = ( int ) floor( y );
64 fy = y - iy;
65 iz = ( int ) floor( z );
66 fz = z - iz;
67 it = ( int ) floor( t );
68 ft = t - it;
69
70 for ( i = 0; i < 2; i++ )
71 {
72 front[0] = GetNoiseValue( ix, iy, iz, it + i );
73 front[1] = GetNoiseValue( ix+1, iy, iz, it + i );
74 front[2] = GetNoiseValue( ix, iy+1, iz, it + i );
75 front[3] = GetNoiseValue( ix+1, iy+1, iz, it + i );
76
77 back[0] = GetNoiseValue( ix, iy, iz + 1, it + i );
78 back[1] = GetNoiseValue( ix+1, iy, iz + 1, it + i );
79 back[2] = GetNoiseValue( ix, iy+1, iz + 1, it + i );
80 back[3] = GetNoiseValue( ix+1, iy+1, iz + 1, it + i );
81
82 fvalue = LERP( LERP( front[0], front[1], fx ), LERP( front[2], front[3], fx ), fy );
83 bvalue = LERP( LERP( back[0], back[1], fx ), LERP( back[2], back[3], fx ), fy );
84
85 value[i] = LERP( fvalue, bvalue, fz );
86 }
87
88 finalvalue = LERP( value[0], value[1], ft );
89
90 return finalvalue;
91}

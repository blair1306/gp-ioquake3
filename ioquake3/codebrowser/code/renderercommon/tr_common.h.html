29typedef enum
30{
31 IMGTYPE_COLORALPHA, // for color, lightmap, diffuse, and specular
32 IMGTYPE_NORMAL,
33 IMGTYPE_NORMALHEIGHT,
34 IMGTYPE_DELUXE, // normals are swizzled, deluxe are not
35} imgType_t;
36
37typedef enum
38{
39 IMGFLAG_NONE = 0x0000,
40 IMGFLAG_MIPMAP = 0x0001,
41 IMGFLAG_PICMIP = 0x0002,
42 IMGFLAG_CUBEMAP = 0x0004,
43 IMGFLAG_NO_COMPRESSION = 0x0010,
44 IMGFLAG_NOLIGHTSCALE = 0x0020,
45 IMGFLAG_CLAMPTOEDGE = 0x0040,
46 IMGFLAG_GENNORMALMAP = 0x0080,
47} imgFlags_t;
48
49typedef struct image_s {
50 char imgName[MAX_QPATH]; // game path, including extension
51 int width, height; // source image
52 int uploadWidth, uploadHeight; // after power of two and picmip but not including clamp to MAX_TEXTURE_SIZE
53 GLuint texnum; // gl texture binding
54
55 int frameUsed; // for texture usage in frame statistics
56
57 int internalFormat;
58 int TMU; // only needed for voodoo2
59
60 imgType_t type;
61 imgFlags_t flags;
62
63 struct image_s* next;
64} image_t;

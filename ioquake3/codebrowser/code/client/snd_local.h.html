35typedef struct {
36 int left; // the final values will be clamped to +/- 0x00ffff00 and shifted down
37 int right;
38} portable_samplepair_t;
39
40typedef struct adpcm_state {
41 short sample; /* Previous output value */
42 char index; /* Index into stepsize table */
43} adpcm_state_t;
44
45typedef struct sndBuffer_s {
46 short sndChunk[SND_CHUNK_SIZE];
47 struct sndBuffer_s *next;
48 int size;
49 adpcm_state_t adpcm;
50} sndBuffer;
51
52typedef struct sfx_s {
53 sndBuffer *soundData;
54 qboolean defaultSound; // couldn't be loaded, so use buzz
55 qboolean inMemory; // not in Memory
56 qboolean soundCompressed; // not in Memory
57 int soundCompressionMethod;
58 int soundLength;
59 int soundChannels;
60 char soundName[MAX_QPATH];
61 int lastTimeUsed;
62 struct sfx_s *next;
63} sfx_t;
64
65typedef struct {
66 int channels;
67 int samples; // mono samples in buffer
68 int submission_chunk; // don't mix less than this #
69 int samplebits;
70 int speed;
71 byte *buffer;
72} dma_t;

80typedef struct loopSound_s {
81 vec3_t origin;
82 vec3_t velocity;
83 sfx_t *sfx;
84 int mergeFrame;
85 qboolean active;
86 qboolean kill;
87 qboolean doppler;
88 float dopplerScale;
89 float oldDopplerScale;
90 int framenum;
91} loopSound_t;
92
93typedef struct
94{
95 int allocTime;
96 int startSample; // START_SAMPLE_IMMEDIATE = set immediately on next mix
97 int entnum; // to allow overriding a specific sound
98 int entchannel; // to allow overriding a specific sound
99 int leftvol; // 0-255 volume after spatialization
100 int rightvol; // 0-255 volume after spatialization
101 int master_vol; // 0-255 volume before spatialization
102 float dopplerScale;
103 float oldDopplerScale;
104 vec3_t origin; // only use if fixed_origin is set
105 qboolean fixed_origin; // use origin instead of fetching entnum's origin
106 sfx_t *thesfx; // sfx structure
107 qboolean doppler;
108 qboolean fullVolume;
109} channel_t;

124// Interface between Q3 sound "api" and the sound backend
125typedef struct
126{
127 void (*Shutdown)(void);
128 void (*StartSound)( vec3_t origin, int entnum, int entchannel, sfxHandle_t sfx );
129 void (*StartLocalSound)( sfxHandle_t sfx, int channelNum );
130 void (*StartBackgroundTrack)( const char *intro, const char *loop );
131 void (*StopBackgroundTrack)( void );
132 void (*RawSamples)(int stream, int samples, int rate, int width, int channels, const byte *data, float volume, int entityNum);
133 void (*StopAllSounds)( void );
134 void (*ClearLoopingSounds)( qboolean killall );
135 void (*AddLoopingSound)( int entityNum, const vec3_t origin, const vec3_t velocity, sfxHandle_t sfx );
136 void (*AddRealLoopingSound)( int entityNum, const vec3_t origin, const vec3_t velocity, sfxHandle_t sfx );
137 void (*StopLoopingSound)(int entityNum );
138 void (*Respatialize)( int entityNum, const vec3_t origin, vec3_t axis[3], int inwater );
139 void (*UpdateEntityPosition)( int entityNum, const vec3_t origin );
140 void (*Update)( void );
141 void (*DisableSounds)( void );
142 void (*BeginRegistration)( void );
143 sfxHandle_t (*RegisterSound)( const char *sample, qboolean compressed );
144 void (*ClearSoundBuffer)( void );
145 void (*SoundInfo)( void );
146 void (*SoundList)( void );
147#ifdef USE_VOIP
148 void (*StartCapture)( void );
149 int (*AvailableCaptureSamples)( void );
150 void (*Capture)( int samples, byte *data );
151 void (*StopCapture)( void );
152 void (*MasterGain)( float gain );
153#endif
154} soundInterface_t;

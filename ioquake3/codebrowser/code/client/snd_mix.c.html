37#if !id386 // if configured not to use asm
38
39void S_WriteLinearBlastStereo16 (void)
40{
41 int i;
42 int val;
43
44 for (i=0 ; i<snd_linear_count ; i+=2)
45 {
46 val = snd_p[i]>>8;
47 if (val > 0x7fff)
48 snd_out[i] = 0x7fff;
49 else if (val < -32768)
50 snd_out[i] = -32768;
51 else
52 snd_out[i] = val;
53
54 val = snd_p[i+1]>>8;
55 if (val > 0x7fff)
56 snd_out[i+1] = 0x7fff;
57 else if (val < -32768)
58 snd_out[i+1] = -32768;
59 else
60 snd_out[i+1] = val;
61 }
62}
63#elif defined(__GNUC__)
64// uses snd_mixa.s
65void S_WriteLinearBlastStereo16 (void);
66#else

68__declspec( naked ) void S_WriteLinearBlastStereo16 (void)
69{
70 __asm {
71
72 push edi
73 push ebx
74 mov ecx,ds:dword ptr[snd_linear_count]
75 mov ebx,ds:dword ptr[snd_p]
76 mov edi,ds:dword ptr[snd_out]
77LWLBLoopTop:
78 mov eax,ds:dword ptr[-8+ebx+ecx*4]
79 sar eax,8
80 cmp eax,07FFFh
81 jg LClampHigh
82 cmp eax,0FFFF8000h
83 jnl LClampDone
84 mov eax,0FFFF8000h
85 jmp LClampDone
86LClampHigh:
87 mov eax,07FFFh
88LClampDone:
89 mov edx,ds:dword ptr[-4+ebx+ecx*4]
90 sar edx,8
91 cmp edx,07FFFh
92 jg LClampHigh2
93 cmp edx,0FFFF8000h
94 jnl LClampDone2
95 mov edx,0FFFF8000h
96 jmp LClampDone2
97LClampHigh2:
98 mov edx,07FFFh
99LClampDone2:
100 shl edx,16
101 and eax,0FFFFh
102 or edx,eax
103 mov ds:dword ptr[-4+edi+ecx*2],edx
104 sub ecx,2
105 jnz LWLBLoopTop
106 pop ebx
107 pop edi
108 ret
109 }
110}

114void S_TransferStereo16 (unsigned long *pbuf, int endtime)
115{
116 int lpos;
117 int ls_paintedtime;
118
119 snd_p = (int *) paintbuffer;
120 ls_paintedtime = s_paintedtime;
121
122 while (ls_paintedtime < endtime)
123 {
124 // handle recirculating buffer issues
125 lpos = ls_paintedtime & ((dma.samples>>1)-1);
126
127 snd_out = (short *) pbuf + (lpos<<1);
128
129 snd_linear_count = (dma.samples>>1) - lpos;
130 if (ls_paintedtime + snd_linear_count > endtime)
131 snd_linear_count = endtime - ls_paintedtime;
132
133 snd_linear_count <<= 1;
134
135 // write a linear blast of samples
136 S_WriteLinearBlastStereo16 ();
137
138 snd_p += snd_linear_count;
139 ls_paintedtime += (snd_linear_count>>1);
140
141 if( CL_VideoRecording( ) )
142 CL_WriteAVIAudioFrame( (byte *)snd_out, snd_linear_count << 1 );
143 }
144}

146/*
147===================
148S_TransferPaintBuffer
149
150===================
151*/
152void S_TransferPaintBuffer(int endtime)
153{
154 int out_idx;
155 int count;
156 int out_mask;
157 int *p;
158 int step;
159 int val;
160 unsigned long *pbuf;
161
162 pbuf = (unsigned long *)dma.buffer;
163
164
165 if ( s_testsound->integer ) {
166 int i;
167
168 // write a fixed sine wave
169 count = (endtime - s_paintedtime);
170 for (i=0 ; i<count ; i++)
171 paintbuffer[i].left = paintbuffer[i].right = sin((s_paintedtime+i)*0.1)*20000*256;
172 }
173
174
175 if (dma.samplebits == 16 && dma.channels == 2)
176 { // optimized case
177 S_TransferStereo16 (pbuf, endtime);
178 }
179 else
180 { // general case
181 p = (int *) paintbuffer;
182 count = (endtime - s_paintedtime) * dma.channels;
183 out_mask = dma.samples - 1;
184 out_idx = s_paintedtime * dma.channels & out_mask;
185 step = 3 - dma.channels;
186
187 if (dma.samplebits == 16)
188 {
189 short *out = (short *) pbuf;
190 while (count--)
191 {
192 val = *p >> 8;
193 p+= step;
194 if (val > 0x7fff)
195 val = 0x7fff;
196 else if (val < -32768)
197 val = -32768;
198 out[out_idx] = val;
199 out_idx = (out_idx + 1) & out_mask;
200 }
201 }
202 else if (dma.samplebits == 8)
203 {
204 unsigned char *out = (unsigned char *) pbuf;
205 while (count--)
206 {
207 val = *p >> 8;
208 p+= step;
209 if (val > 0x7fff)
210 val = 0x7fff;
211 else if (val < -32768)
212 val = -32768;
213 out[out_idx] = (val>>8) + 128;
214 out_idx = (out_idx + 1) & out_mask;
215 }
216 }
217 }
218}

229#if idppc_altivec
230static void S_PaintChannelFrom16_altivec( channel_t *ch, const sfx_t *sc, int count, int sampleOffset, int bufferOffset ) {
231 int data, aoff, boff;
232 int leftvol, rightvol;
233 int i, j;
234 portable_samplepair_t *samp;
235 sndBuffer *chunk;
236 short *samples;
237 float ooff, fdata[2], fdiv, fleftvol, frightvol;
238
239 if (sc->soundChannels <= 0) {
240 return;
241 }
242
243 samp = &paintbuffer[ bufferOffset ];
244
245 if (ch->doppler) {
246 sampleOffset = sampleOffset*ch->oldDopplerScale;
247 }
248
249 if ( sc->soundChannels == 2 ) {
250 sampleOffset *= sc->soundChannels;
251
252 if ( sampleOffset & 1 ) {
253 sampleOffset &= ~1;
254 }
255 }
256
257 chunk = sc->soundData;
258 while (sampleOffset>=SND_CHUNK_SIZE) {
259 chunk = chunk->next;
260 sampleOffset -= SND_CHUNK_SIZE;
261 if (!chunk) {
262 chunk = sc->soundData;
263 }
264 }
265
266 if (!ch->doppler || ch->dopplerScale==1.0f) {
267 vector signed short volume_vec;
268 vector unsigned int volume_shift;
269 int vectorCount, samplesLeft, chunkSamplesLeft;
270 leftvol = ch->leftvol*snd_vol;
271 rightvol = ch->rightvol*snd_vol;
272 samples = chunk->sndChunk;
273 ((short *)&volume_vec)[0] = leftvol;
274 ((short *)&volume_vec)[1] = leftvol;
275 ((short *)&volume_vec)[4] = leftvol;
276 ((short *)&volume_vec)[5] = leftvol;
277 ((short *)&volume_vec)[2] = rightvol;
278 ((short *)&volume_vec)[3] = rightvol;
279 ((short *)&volume_vec)[6] = rightvol;
280 ((short *)&volume_vec)[7] = rightvol;
281 volume_shift = vec_splat_u32(8);
282 i = 0;
283
284 while(i < count) {
285 /* Try to align destination to 16-byte boundary */
286 while(i < count && (((unsigned long)&samp[i] & 0x1f) || ((count-i) < 8) || ((SND_CHUNK_SIZE - sampleOffset) < 8))) {
287 data = samples[sampleOffset++];
288 samp[i].left += (data * leftvol)>>8;
289
290 if ( sc->soundChannels == 2 ) {
291 data = samples[sampleOffset++];
292 }
293 samp[i].right += (data * rightvol)>>8;
294
295 if (sampleOffset == SND_CHUNK_SIZE) {
296 chunk = chunk->next;
297 samples = chunk->sndChunk;
298 sampleOffset = 0;
299 }
300 i++;
301 }
302 /* Destination is now aligned. Process as many 8-sample
303 chunks as we can before we run out of room from the current
304 sound chunk. We do 8 per loop to avoid extra source data reads. */
305 samplesLeft = count - i;
306 chunkSamplesLeft = SND_CHUNK_SIZE - sampleOffset;
307 if(samplesLeft > chunkSamplesLeft)
308 samplesLeft = chunkSamplesLeft;
309
310 vectorCount = samplesLeft / 8;
311
312 if(vectorCount)
313 {
314 vector unsigned char tmp;
315 vector short s0, s1, sampleData0, sampleData1;
316 vector signed int merge0, merge1;
317 vector signed int d0, d1, d2, d3;
318 vector unsigned char samplePermute0 =
319 VECCONST_UINT8(0, 1, 4, 5, 0, 1, 4, 5, 2, 3, 6, 7, 2, 3, 6, 7);
320 vector unsigned char samplePermute1 =
321 VECCONST_UINT8(8, 9, 12, 13, 8, 9, 12, 13, 10, 11, 14, 15, 10, 11, 14, 15);
322 vector unsigned char loadPermute0, loadPermute1;
323
324 // Rather than permute the vectors after we load them to do the sample
325 // replication and rearrangement, we permute the alignment vector so
326 // we do everything in one step below and avoid data shuffling.
327 tmp = vec_lvsl(0,&samples[sampleOffset]);
328 loadPermute0 = vec_perm(tmp,tmp,samplePermute0);
329 loadPermute1 = vec_perm(tmp,tmp,samplePermute1);
330
331 s0 = *(vector short *)&samples[sampleOffset];
332 while(vectorCount)
333 {
334 /* Load up source (16-bit) sample data */
335 s1 = *(vector short *)&samples[sampleOffset+7];
336
337 /* Load up destination sample data */
338 d0 = *(vector signed int *)&samp[i];
339 d1 = *(vector signed int *)&samp[i+2];
340 d2 = *(vector signed int *)&samp[i+4];
341 d3 = *(vector signed int *)&samp[i+6];
342
343 sampleData0 = vec_perm(s0,s1,loadPermute0);
344 sampleData1 = vec_perm(s0,s1,loadPermute1);
345
346 merge0 = vec_mule(sampleData0,volume_vec);
347 merge0 = vec_sra(merge0,volume_shift); /* Shift down to proper range */
348
349 merge1 = vec_mulo(sampleData0,volume_vec);
350 merge1 = vec_sra(merge1,volume_shift);
351
352 d0 = vec_add(merge0,d0);
353 d1 = vec_add(merge1,d1);
354
355 merge0 = vec_mule(sampleData1,volume_vec);
356 merge0 = vec_sra(merge0,volume_shift); /* Shift down to proper range */
357
358 merge1 = vec_mulo(sampleData1,volume_vec);
359 merge1 = vec_sra(merge1,volume_shift);
360
361 d2 = vec_add(merge0,d2);
362 d3 = vec_add(merge1,d3);
363
364 /* Store destination sample data */
365 *(vector signed int *)&samp[i] = d0;
366 *(vector signed int *)&samp[i+2] = d1;
367 *(vector signed int *)&samp[i+4] = d2;
368 *(vector signed int *)&samp[i+6] = d3;
369
370 i += 8;
371 vectorCount--;
372 s0 = s1;
373 sampleOffset += 8;
374 }
375 if (sampleOffset == SND_CHUNK_SIZE) {
376 chunk = chunk->next;
377 samples = chunk->sndChunk;
378 sampleOffset = 0;
379 }
380 }
381 }
382 } else {
383 fleftvol = ch->leftvol*snd_vol;
384 frightvol = ch->rightvol*snd_vol;
385
386 ooff = sampleOffset;
387 samples = chunk->sndChunk;
388
389 for ( i=0 ; i<count ; i++ ) {
390
391 aoff = ooff;
392 ooff = ooff + ch->dopplerScale * sc->soundChannels;
393 boff = ooff;
394 fdata[0] = fdata[1] = 0;
395 for (j=aoff; j<boff; j += sc->soundChannels) {
396 if (j == SND_CHUNK_SIZE) {
397 chunk = chunk->next;
398 if (!chunk) {
399 chunk = sc->soundData;
400 }
401 samples = chunk->sndChunk;
402 ooff -= SND_CHUNK_SIZE;
403 }
404 if ( sc->soundChannels == 2 ) {
405 fdata[0] += samples[j&(SND_CHUNK_SIZE-1)];
406 fdata[1] += samples[(j+1)&(SND_CHUNK_SIZE-1)];
407 } else {
408 fdata[0] += samples[j&(SND_CHUNK_SIZE-1)];
409 fdata[1] += samples[j&(SND_CHUNK_SIZE-1)];
410 }
411 }
412 fdiv = 256 * (boff-aoff) / sc->soundChannels;
413 samp[i].left += (fdata[0] * fleftvol)/fdiv;
414 samp[i].right += (fdata[1] * frightvol)/fdiv;
415 }
416 }
417}

526void S_PaintChannelFromWavelet( channel_t *ch, sfx_t *sc, int count, int sampleOffset, int bufferOffset ) {
527 int data;
528 int leftvol, rightvol;
529 int i;
530 portable_samplepair_t *samp;
531 sndBuffer *chunk;
532 short *samples;
533
534 leftvol = ch->leftvol*snd_vol;
535 rightvol = ch->rightvol*snd_vol;
536
537 i = 0;
538 samp = &paintbuffer[ bufferOffset ];
539 chunk = sc->soundData;
540 while (sampleOffset>=(SND_CHUNK_SIZE_FLOAT*4)) {
541 chunk = chunk->next;
542 sampleOffset -= (SND_CHUNK_SIZE_FLOAT*4);
543 i++;
544 }
545
546 if (i!=sfxScratchIndex || sfxScratchPointer != sc) {
547 S_AdpcmGetSamples( chunk, sfxScratchBuffer );
548 sfxScratchIndex = i;
549 sfxScratchPointer = sc;
550 }
551
552 samples = sfxScratchBuffer;
553
554 for ( i=0 ; i<count ; i++ ) {
555 data = samples[sampleOffset++];
556 samp[i].left += (data * leftvol)>>8;
557 samp[i].right += (data * rightvol)>>8;
558
559 if (sampleOffset == SND_CHUNK_SIZE*2) {
560 chunk = chunk->next;
561 decodeWavelet(chunk, sfxScratchBuffer);
562 sfxScratchIndex++;
563 sampleOffset = 0;
564 }
565 }

615void S_PaintChannelFromMuLaw( channel_t *ch, sfx_t *sc, int count, int sampleOffset, int bufferOffset ) {
616 int data;
617 int leftvol, rightvol;
618 int i;
619 portable_samplepair_t *samp;
620 sndBuffer *chunk;
621 byte *samples;
622 float ooff;
623
624 leftvol = ch->leftvol*snd_vol;
625 rightvol = ch->rightvol*snd_vol;
626
627 samp = &paintbuffer[ bufferOffset ];
628 chunk = sc->soundData;
629 while (sampleOffset>=(SND_CHUNK_SIZE*2)) {
630 chunk = chunk->next;
631 sampleOffset -= (SND_CHUNK_SIZE*2);
632 if (!chunk) {
633 chunk = sc->soundData;
634 }
635 }
636
637 if (!ch->doppler) {
638 samples = (byte *)chunk->sndChunk + sampleOffset;
639 for ( i=0 ; i<count ; i++ ) {
640 data = mulawToShort[*samples];
641 samp[i].left += (data * leftvol)>>8;
642 samp[i].right += (data * rightvol)>>8;
643 samples++;
644 if (chunk != NULL && samples == (byte *)chunk->sndChunk+(SND_CHUNK_SIZE*2)) {
645 chunk = chunk->next;
646 samples = (byte *)chunk->sndChunk;
647 }
648 }
649 } else {
650 ooff = sampleOffset;
651 samples = (byte *)chunk->sndChunk;
652 for ( i=0 ; i<count ; i++ ) {
653 data = mulawToShort[samples[(int)(ooff)]];
654 ooff = ooff + ch->dopplerScale;
655 samp[i].left += (data * leftvol)>>8;
656 samp[i].right += (data * rightvol)>>8;
657 if (ooff >= SND_CHUNK_SIZE*2) {
658 chunk = chunk->next;
659 if (!chunk) {
660 chunk = sc->soundData;
661 }
662 samples = (byte *)chunk->sndChunk;
663 ooff = 0.0;
664 }
665 }
666 }

669/*
670===================
671S_PaintChannels
672===================
673*/
674void S_PaintChannels( int endtime ) {
675 int i;
676 int end;
677 int stream;
678 channel_t *ch;
679 sfx_t *sc;
680 int ltime, count;
681 int sampleOffset;
682
683 if(s_muted->integer)
684 snd_vol = 0;
685 else
686 snd_vol = s_volume->value*255;
687
688//Com_Printf ("%i to %i\n", s_paintedtime, endtime);
689 while ( s_paintedtime < endtime ) {
690 // if paintbuffer is smaller than DMA buffer
691 // we may need to fill it multiple times
692 end = endtime;
693 if ( endtime - s_paintedtime > PAINTBUFFER_SIZE ) {
694 end = s_paintedtime + PAINTBUFFER_SIZE;
695 }
696
697 // clear the paint buffer and mix any raw samples...
698 Com_Memset(paintbuffer, 0, sizeof (paintbuffer));
699 for (stream = 0; stream < MAX_RAW_STREAMS; stream++) {
700 if ( s_rawend[stream] >= s_paintedtime ) {
701 // copy from the streaming sound source
702 const portable_samplepair_t *rawsamples = s_rawsamples[stream];
703 const int stop = (end < s_rawend[stream]) ? end : s_rawend[stream];
704 for ( i = s_paintedtime ; i < stop ; i++ ) {
705 const int s = i&(MAX_RAW_SAMPLES-1);
706 paintbuffer[i-s_paintedtime].left += rawsamples[s].left;
707 paintbuffer[i-s_paintedtime].right += rawsamples[s].right;
708 }
709 }
710 }
711
712 // paint in the channels.
713 ch = s_channels;
714 for ( i = 0; i < MAX_CHANNELS ; i++, ch++ ) {
715 if ( !ch->thesfx || (ch->leftvol<0.25 && ch->rightvol<0.25 )) {
716 continue;
717 }
718
719 ltime = s_paintedtime;
720 sc = ch->thesfx;
721
722 if (sc->soundData==NULL || sc->soundLength==0) {
723 continue;
724 }
725
726 sampleOffset = ltime - ch->startSample;
727 count = end - ltime;
728 if ( sampleOffset + count > sc->soundLength ) {
729 count = sc->soundLength - sampleOffset;
730 }
731
732 if ( count > 0 ) {
733 if( sc->soundCompressionMethod == 1) {
734 S_PaintChannelFromADPCM (ch, sc, count, sampleOffset, ltime - s_paintedtime);
735 } else if( sc->soundCompressionMethod == 2) {
736 S_PaintChannelFromWavelet (ch, sc, count, sampleOffset, ltime - s_paintedtime);
737 } else if( sc->soundCompressionMethod == 3) {
738 S_PaintChannelFromMuLaw (ch, sc, count, sampleOffset, ltime - s_paintedtime);
739 } else {
740 S_PaintChannelFrom16 (ch, sc, count, sampleOffset, ltime - s_paintedtime);
741 }
742 }
743 }

745 // paint in the looped channels.
746 ch = loop_channels;
747 for ( i = 0; i < numLoopChannels ; i++, ch++ ) {
748 if ( !ch->thesfx || (!ch->leftvol && !ch->rightvol )) {
749 continue;
750 }
751
752 ltime = s_paintedtime;
753 sc = ch->thesfx;
754
755 if (sc->soundData==NULL || sc->soundLength==0) {
756 continue;
757 }
758 // we might have to make two passes if it
759 // is a looping sound effect and the end of
760 // the sample is hit
761 do {
762 sampleOffset = (ltime % sc->soundLength);
763
764 count = end - ltime;
765 if ( sampleOffset + count > sc->soundLength ) {
766 count = sc->soundLength - sampleOffset;
767 }
768
769 if ( count > 0 ) {
770 if( sc->soundCompressionMethod == 1) {
771 S_PaintChannelFromADPCM (ch, sc, count, sampleOffset, ltime - s_paintedtime);
772 } else if( sc->soundCompressionMethod == 2) {
773 S_PaintChannelFromWavelet (ch, sc, count, sampleOffset, ltime - s_paintedtime);
774 } else if( sc->soundCompressionMethod == 3) {
775 S_PaintChannelFromMuLaw (ch, sc, count, sampleOffset, ltime - s_paintedtime);
776 } else {
777 S_PaintChannelFrom16 (ch, sc, count, sampleOffset, ltime - s_paintedtime);
778 }
779 ltime += count;
780 }
781 } while ( ltime < end);
782 }
783
784 // transfer out according to DMA format
785 S_TransferPaintBuffer( end );
786 s_paintedtime = end;
787 }

35================
36SCR_DrawNamedPic
37
38Coordinates are 640*480 virtual values
39=================
40*/
41void SCR_DrawNamedPic( float x, float y, float width, float height, const char *picname ) {
42 qhandle_t hShader;
43
44 assert( width != 0 );
45
46 hShader = re.RegisterShader( picname );
47 SCR_AdjustFrom640( &x, &y, &width, &height );
48 re.DrawStretchPic( x, y, width, height, 0, 0, 1, 1, hShader );
49}

52/*
53================
54SCR_AdjustFrom640
55
56Adjusted for resolution and screen aspect ratio
57================
58*/
59void SCR_AdjustFrom640( float *x, float *y, float *w, float *h ) {
60 float xscale;
61 float yscale;
62
63#if 0
64 // adjust for wide screens
65 if ( cls.glconfig.vidWidth * 480 > cls.glconfig.vidHeight * 640 ) {
66 *x += 0.5 * ( cls.glconfig.vidWidth - ( cls.glconfig.vidHeight * 640 / 480 ) );
67 }
68#endif
69
70 // scale for screen sizes
71 xscale = cls.glconfig.vidWidth / 640.0;
72 yscale = cls.glconfig.vidHeight / 480.0;
73 if ( x ) {
74 *x *= xscale;
75 }
76 if ( y ) {
77 *y *= yscale;
78 }
79 if ( w ) {
80 *w *= xscale;
81 }
82 if ( h ) {
83 *h *= yscale;
84 }
85}

118/*
119** SCR_DrawChar
120** chars are drawn at 640*480 virtual screen size
121*/
122static void SCR_DrawChar( int x, int y, float size, int ch ) {
123 int row, col;
124 float frow, fcol;
125 float ax, ay, aw, ah;
126
127 ch &= 255;
128
129 if ( ch == ' ' ) {
130 return;
131 }
132
133 if ( y < -size ) {
134 return;
135 }
136
137 ax = x;
138 ay = y;
139 aw = size;
140 ah = size;
141 SCR_AdjustFrom640( &ax, &ay, &aw, &ah );
142
143 row = ch>>4;
144 col = ch&15;
145
146 frow = row*0.0625;
147 fcol = col*0.0625;
148 size = 0.0625;
149
150 re.DrawStretchPic( ax, ay, aw, ah,
151 fcol, frow,
152 fcol + size, frow + size,
153 cls.charSetShader );
154}

189/*
190==================
191SCR_DrawBigString[Color]
192
193Draws a multi-colored string with a drop shadow, optionally forcing
194to a fixed color.
195
196Coordinates are at 640 by 480 virtual resolution
197==================
198*/
199void SCR_DrawStringExt( int x, int y, float size, const char *string, float *setColor, qboolean forceColor,
200 qboolean noColorEscape ) {
201 vec4_t color;
202 const char *s;
203 int xx;
204
205 // draw the drop shadow
206 color[0] = color[1] = color[2] = 0;
207 color[3] = setColor[3];
208 re.SetColor( color );
209 s = string;
210 xx = x;
211 while ( *s ) {
212 if ( !noColorEscape && Q_IsColorString( s ) ) {
213 s += 2;
214 continue;
215 }
216 SCR_DrawChar( xx+2, y+2, size, *s );
217 xx += size;
218 s++;
219 }
220
221
222 // draw the colored text
223 s = string;
224 xx = x;
225 re.SetColor( setColor );
226 while ( *s ) {
227 if ( Q_IsColorString( s ) ) {
228 if ( !forceColor ) {
229 Com_Memcpy( color, g_color_table[ColorIndex(*(s+1))], sizeof( color ) );
230 color[3] = setColor[3];
231 re.SetColor( color );
232 }
233 if ( !noColorEscape ) {
234 s += 2;
235 continue;
236 }
237 }
238 SCR_DrawChar( xx, y, size, *s );
239 xx += size;
240 s++;
241 }
242 re.SetColor( NULL );
243}

298/*
299** SCR_Strlen -- skips color escape codes
300*/
301static int SCR_Strlen( const char *str ) {
302 const char *s = str;
303 int count = 0;
304
305 while ( *s ) {
306 if ( Q_IsColorString( s ) ) {
307 s += 2;
308 } else {
309 count++;
310 s++;
311 }
312 }
313
314 return count;
315}
316
317/*
318** SCR_GetBigStringWidth
319*/
320int SCR_GetBigStringWidth( const char *str ) {
321 return SCR_Strlen( str ) * BIGCHAR_WIDTH;
322}

328=================
329SCR_DrawDemoRecording
330=================
331*/
332void SCR_DrawDemoRecording( void ) {
333 char string[1024];
334 int pos;
335
336 if ( !clc.demorecording ) {
337 return;
338 }
339 if ( clc.spDemoRecording ) {
340 return;
341 }
342
343 pos = FS_FTell( clc.demofile );
344 sprintf( string, "RECORDING %s: %ik", clc.demoName, pos / 1024 );
345
346 SCR_DrawStringExt( 320 - strlen( string ) * 4, 20, 8, string, g_color_table[7], qtrue, qfalse );
347}

350#ifdef USE_VOIP
351/*
352=================
353SCR_DrawVoipMeter
354=================
355*/
356void SCR_DrawVoipMeter( void ) {
357 char buffer[16];
358 char string[256];
359 int limit, i;
360
361 if (!cl_voipShowMeter->integer)
362 return; // player doesn't want to show meter at all.
363 else if (!cl_voipSend->integer)
364 return; // not recording at the moment.
365 else if (clc.state != CA_ACTIVE)
366 return; // not connected to a server.
367 else if (!clc.voipEnabled)
368 return; // server doesn't support VoIP.
369 else if (clc.demoplaying)
370 return; // playing back a demo.
371 else if (!cl_voip->integer)
372 return; // client has VoIP support disabled.
373
374 limit = (int) (clc.voipPower * 10.0f);
375 if (limit > 10)
376 limit = 10;
377
378 for (i = 0; i < limit; i++)
379 buffer[i] = '*';
380 while (i < 10)
381 buffer[i++] = ' ';
382 buffer[i] = '\0';
383
384 sprintf( string, "VoIP: [%s]", buffer );
385 SCR_DrawStringExt( 320 - strlen( string ) * 4, 10, 8, string, g_color_table[7], qtrue, qfalse );
386}
387#endif

400static int current;
401static float values[1024];
402
403/*
404==============
405SCR_DebugGraph
406==============
407*/
408void SCR_DebugGraph (float value)
409{
410 values[current] = value;
411 current = (current + 1) % ARRAY_LEN(values);
412}
413
414/*
415==============
416SCR_DrawDebugGraph
417==============
418*/
419void SCR_DrawDebugGraph (void)
420{
421 int a, x, y, w, i, h;
422 float v;
423
424 //
425 // draw the graph
426 //
427 w = cls.glconfig.vidWidth;
428 x = 0;
429 y = cls.glconfig.vidHeight;
430 re.SetColor( g_color_table[0] );
431 re.DrawStretchPic(x, y - cl_graphheight->integer,
432 w, cl_graphheight->integer, 0, 0, 0, 0, cls.whiteShader );
433 re.SetColor( NULL );
434
435 for (a=0 ; a<w ; a++)
436 {
437 i = (ARRAY_LEN(values)+current-1-(a % ARRAY_LEN(values))) % ARRAY_LEN(values);
438 v = values[i];
439 v = v * cl_graphscale->integer + cl_graphshift->integer;
440
441 if (v < 0)
442 v += cl_graphheight->integer * (1+(int)(-v / cl_graphheight->integer));
443 h = (int)v % cl_graphheight->integer;
444 re.DrawStretchPic( x+w-1-a, y - h, 1, h, 0, 0, 0, 0, cls.whiteShader );
445 }
446}

468/*
469==================
470SCR_DrawScreenField
471
472This will be called twice if rendering in stereo mode
473==================
474*/
475void SCR_DrawScreenField( stereoFrame_t stereoFrame ) {
476 qboolean uiFullscreen;
477
478 re.BeginFrame( stereoFrame );
479
480 uiFullscreen = (uivm && VM_Call( uivm, UI_IS_FULLSCREEN ));
481
482 // wide aspect ratio screens need to have the sides cleared
483 // unless they are displaying game renderings
484 if ( uiFullscreen || clc.state < CA_LOADING ) {
485 if ( cls.glconfig.vidWidth * 480 > cls.glconfig.vidHeight * 640 ) {
486 re.SetColor( g_color_table[0] );
487 re.DrawStretchPic( 0, 0, cls.glconfig.vidWidth, cls.glconfig.vidHeight, 0, 0, 0, 0, cls.whiteShader );
488 re.SetColor( NULL );
489 }
490 }
491
492 // if the menu is going to cover the entire screen, we
493 // don't need to render anything under it
494 if ( uivm && !uiFullscreen ) {
495 switch( clc.state ) {
496 default:
497 Com_Error( ERR_FATAL, "SCR_DrawScreenField: bad clc.state" );
498 break;
499 case CA_CINEMATIC:
500 SCR_DrawCinematic();
501 break;
502 case CA_DISCONNECTED:
503 // force menu up
504 S_StopAllSounds();
505 VM_Call( uivm, UI_SET_ACTIVE_MENU, UIMENU_MAIN );
506 break;
507 case CA_CONNECTING:
508 case CA_CHALLENGING:
509 case CA_CONNECTED:
510 // connecting clients will only show the connection dialog
511 // refresh to update the time
512 VM_Call( uivm, UI_REFRESH, cls.realtime );
513 VM_Call( uivm, UI_DRAW_CONNECT_SCREEN, qfalse );
514 break;
515 case CA_LOADING:
516 case CA_PRIMED:
517 // draw the game information screen and loading progress
518 CL_CGameRendering(stereoFrame);
519
520 // also draw the connection information, so it doesn't
521 // flash away too briefly on local or lan games
522 // refresh to update the time
523 VM_Call( uivm, UI_REFRESH, cls.realtime );
524 VM_Call( uivm, UI_DRAW_CONNECT_SCREEN, qtrue );
525 break;
526 case CA_ACTIVE:
527 // always supply STEREO_CENTER as vieworg offset is now done by the engine.
528 CL_CGameRendering(stereoFrame);
529 SCR_DrawDemoRecording();
530#ifdef USE_VOIP
531 SCR_DrawVoipMeter();
532#endif
533 break;
534 }
535 }
536
537 // the menu draws next
538 if ( Key_GetCatcher( ) & KEYCATCH_UI && uivm ) {
539 VM_Call( uivm, UI_REFRESH, cls.realtime );
540 }
541
542 // console draws next
543 Con_DrawConsole ();
544
545 // debug graph can be drawn on top of anything
546 if ( cl_debuggraph->integer || cl_timegraph->integer || cl_debugMove->integer ) {
547 SCR_DrawDebugGraph ();
548 }
549}

551/*
552==================
553SCR_UpdateScreen
554
555This is called every frame, and can also be called explicitly to flush
556text to the screen.
557==================
558*/
559void SCR_UpdateScreen( void ) {
560 static int recursive;
561
562 if ( !scr_initialized ) {
563 return; // not initialized yet
564 }
565
566 if ( ++recursive > 2 ) {
567 Com_Error( ERR_FATAL, "SCR_UpdateScreen: recursively called" );
568 }
569 recursive = 1;
570
571 // If there is no VM, there are also no rendering commands issued. Stop the renderer in
572 // that case.
573 if( uivm || com_dedicated->integer )
574 {
575 // XXX
576 int in_anaglyphMode = Cvar_VariableIntegerValue("r_anaglyphMode");
577 // if running in stereo, we need to draw the frame twice
578 if ( cls.glconfig.stereoEnabled || in_anaglyphMode) {
579 SCR_DrawScreenField( STEREO_LEFT );
580 SCR_DrawScreenField( STEREO_RIGHT );
581 } else {
582 SCR_DrawScreenField( STEREO_CENTER );
583 }
584
585 if ( com_speeds->integer ) {
586 re.EndFrame( &time_frontend, &time_backend );
587 } else {
588 re.EndFrame( NULL, NULL );
589 }
590 }
591
592 recursive = 0;
593}

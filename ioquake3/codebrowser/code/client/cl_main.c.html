247void CL_Voip_f( void )
248{
249 const char *cmd = Cmd_Argv(1);
250 const char *reason = NULL;
251
252 if (clc.state != CA_ACTIVE)
253 reason = "Not connected to a server";
254 else if (!clc.voipCodecInitialized)
255 reason = "Voip codec not initialized";
256 else if (!clc.voipEnabled)
257 reason = "Server doesn't support VoIP";
258 else if (!clc.demoplaying && (Cvar_VariableValue( "g_gametype" ) == GT_SINGLE_PLAYER || Cvar_VariableValue("ui_singlePlayerActive")))
259 reason = "running in single-player mode";
260
261 if (reason != NULL) {
262 Com_Printf("VoIP: command ignored: %s\n", reason);
263 return;
264 }
265
266 if (strcmp(cmd, "ignore") == 0) {
267 CL_UpdateVoipIgnore(Cmd_Argv(2), qtrue);
268 } else if (strcmp(cmd, "unignore") == 0) {
269 CL_UpdateVoipIgnore(Cmd_Argv(2), qfalse);
270 } else if (strcmp(cmd, "gain") == 0) {
271 if (Cmd_Argc() > 3) {
272 CL_UpdateVoipGain(Cmd_Argv(2), atof(Cmd_Argv(3)));
273 } else if (Q_isanumber(Cmd_Argv(2))) {
274 int id = atoi(Cmd_Argv(2));
275 if (id >= 0 && id < MAX_CLIENTS) {
276 Com_Printf("VoIP: current gain for player #%d "
277 "is %f\n", id, clc.voipGain[id]);
278 } else {
279 Com_Printf("VoIP: invalid player ID#\n");
280 }
281 } else {
282 Com_Printf("usage: voip gain <playerID#> [value]\n");
283 }
284 } else if (strcmp(cmd, "muteall") == 0) {
285 Com_Printf("VoIP: muting incoming voice\n");
286 CL_AddReliableCommand("voip muteall", qfalse);
287 clc.voipMuteAll = qtrue;
288 } else if (strcmp(cmd, "unmuteall") == 0) {
289 Com_Printf("VoIP: unmuting incoming voice\n");
290 CL_AddReliableCommand("voip unmuteall", qfalse);
291 clc.voipMuteAll = qfalse;
292 } else {
293 Com_Printf("usage: voip [un]ignore <playerID#>\n"
294 " voip [un]muteall\n"
295 " voip gain <playerID#> [value]\n");
296 }
297}

301void CL_VoipNewGeneration(void)
302{
303 // don't have a zero generation so new clients won't match, and don't
304 // wrap to negative so MSG_ReadLong() doesn't "fail."
305 clc.voipOutgoingGeneration++;
306 if (clc.voipOutgoingGeneration <= 0)
307 clc.voipOutgoingGeneration = 1;

309 clc.voipOutgoingSequence = 0;
310
311 opus_encoder_ctl(clc.opusEncoder, OPUS_RESET_STATE);
312}

314/*
315===============
316CL_VoipParseTargets
317
318sets clc.voipTargets according to cl_voipSendTarget
319Generally we don't want who's listening to change during a transmission,
320so this is only called when the key is first pressed
321===============
322*/
323void CL_VoipParseTargets(void)
324{
325 const char *target = cl_voipSendTarget->string;
326 char *end;
327 int val;
328
329 Com_Memset(clc.voipTargets, 0, sizeof(clc.voipTargets));
330 clc.voipFlags &= ~VOIP_SPATIAL;
331
332 while(target)
333 {
334 while(*target == ',' || *target == ' ')
335 target++;
336
337 if(!*target)
338 break;
339
340 if(isdigit(*target))
341 {
342 val = strtol(target, &end, 10);
343 target = end;
344 }
345 else
346 {
347 if(!Q_stricmpn(target, "all", 3))
348 {
349 Com_Memset(clc.voipTargets, ~0, sizeof(clc.voipTargets));
350 return;
351 }
352 if(!Q_stricmpn(target, "spatial", 7))
353 {
354 clc.voipFlags |= VOIP_SPATIAL;
355 target += 7;
356 continue;
357 }
358 else
359 {
360 if(!Q_stricmpn(target, "attacker", 8))
361 {
362 val = VM_Call(cgvm, CG_LAST_ATTACKER);
363 target += 8;
364 }
365 else if(!Q_stricmpn(target, "crosshair", 9))
366 {
367 val = VM_Call(cgvm, CG_CROSSHAIR_PLAYER);
368 target += 9;
369 }
370 else
371 {
372 while(*target && *target != ',' && *target != ' ')
373 target++;
374
375 continue;
376 }
377
378 if(val < 0)
379 continue;
380 }
381 }
382
383 if(val < 0 || val >= MAX_CLIENTS)
384 {
385 Com_Printf(S_COLOR_YELLOW "WARNING: VoIP "
386 "target %d is not a valid client "
387 "number\n", val);
388
389 continue;
390 }
391
392 clc.voipTargets[val / 8] |= 1 << (val % 8);
393 }
394}

396/*
397===============
398CL_CaptureVoip
399
400Record more audio from the hardware if required and encode it into Opus
401 data for later transmission.
402===============
403*/
404static
405void CL_CaptureVoip(void)
406{
407 const float audioMult = cl_voipCaptureMult->value;
408 const qboolean useVad = (cl_voipUseVAD->integer != 0);
409 qboolean initialFrame = qfalse;
410 qboolean finalFrame = qfalse;
411
412#if USE_MUMBLE
413 // if we're using Mumble, don't try to handle VoIP transmission ourselves.
414 if (cl_useMumble->integer)
415 return;
416#endif
417
418 // If your data rate is too low, you'll get Connection Interrupted warnings
419 // when VoIP packets arrive, even if you have a broadband connection.
420 // This might work on rates lower than 25000, but for safety's sake, we'll
421 // just demand it. Who doesn't have at least a DSL line now, anyhow? If
422 // you don't, you don't need VoIP. :)
423 if (cl_voip->modified || cl_rate->modified) {
424 if ((cl_voip->integer) && (cl_rate->integer < 25000)) {
425 Com_Printf(S_COLOR_YELLOW "Your network rate is too slow for VoIP.\n");
426 Com_Printf("Set 'Data Rate' to 'LAN/Cable/xDSL' in 'Setup/System/Network'.\n");
427 Com_Printf("Until then, VoIP is disabled.\n");
428 Cvar_Set("cl_voip", "0");
429 }
430 Cvar_Set("cl_voipProtocol", cl_voip->integer ? "opus" : "");
431 cl_voip->modified = qfalse;
432 cl_rate->modified = qfalse;
433 }
434
435 if (!clc.voipCodecInitialized)
436 return; // just in case this gets called at a bad time.
437
438 if (clc.voipOutgoingDataSize > 0)
439 return; // packet is pending transmission, don't record more yet.
440
441 if (cl_voipUseVAD->modified) {
442 Cvar_Set("cl_voipSend", (useVad) ? "1" : "0");
443 cl_voipUseVAD->modified = qfalse;
444 }
445
446 if ((useVad) && (!cl_voipSend->integer))
447 Cvar_Set("cl_voipSend", "1"); // lots of things reset this.
448
449 if (cl_voipSend->modified) {
450 qboolean dontCapture = qfalse;
451 if (clc.state != CA_ACTIVE)
452 dontCapture = qtrue; // not connected to a server.
453 else if (!clc.voipEnabled)
454 dontCapture = qtrue; // server doesn't support VoIP.
455 else if (clc.demoplaying)
456 dontCapture = qtrue; // playing back a demo.
457 else if ( cl_voip->integer == 0 )
458 dontCapture = qtrue; // client has VoIP support disabled.
459 else if ( audioMult == 0.0f )
460 dontCapture = qtrue; // basically silenced incoming audio.
461
462 cl_voipSend->modified = qfalse;
463
464 if(dontCapture)
465 {
466 Cvar_Set("cl_voipSend", "0");
467 return;
468 }
469
470 if (cl_voipSend->integer) {
471 initialFrame = qtrue;
472 } else {
473 finalFrame = qtrue;
474 }
475 }
476
477 // try to get more audio data from the sound card...
478
479 if (initialFrame) {
480 S_MasterGain(Com_Clamp(0.0f, 1.0f, cl_voipGainDuringCapture->value));
481 S_StartCapture();
482 CL_VoipNewGeneration();
483 CL_VoipParseTargets();
484 }
485
486 if ((cl_voipSend->integer) || (finalFrame)) { // user wants to capture audio?
487 int samples = S_AvailableCaptureSamples();
488 const int packetSamples = (finalFrame) ? VOIP_MAX_FRAME_SAMPLES : VOIP_MAX_PACKET_SAMPLES;
489
490 // enough data buffered in audio hardware to process yet?
491 if (samples >= packetSamples) {
492 // audio capture is always MONO16.
493 static int16_t sampbuffer[VOIP_MAX_PACKET_SAMPLES];
494 float voipPower = 0.0f;
495 int voipFrames;
496 int i, bytes;
497
498 if (samples > VOIP_MAX_PACKET_SAMPLES)
499 samples = VOIP_MAX_PACKET_SAMPLES;
500
501 // !!! FIXME: maybe separate recording from encoding, so voipPower
502 // !!! FIXME: updates faster than 4Hz?
503
504 samples -= samples % VOIP_MAX_FRAME_SAMPLES;
505 if (samples != 120 && samples != 240 && samples != 480 && samples != 960 && samples != 1920 && samples != 2880 ) {
506 Com_Printf("Voip: bad number of samples %d\n", samples);
507 return;
508 }
509 voipFrames = samples / VOIP_MAX_FRAME_SAMPLES;
510
511 S_Capture(samples, (byte *) sampbuffer); // grab from audio card.
512
513 // check the "power" of this packet...
514 for (i = 0; i < samples; i++) {
515 const float flsamp = (float) sampbuffer[i];
516 const float s = fabs(flsamp);
517 voipPower += s * s;
518 sampbuffer[i] = (int16_t) ((flsamp) * audioMult);
519 }
520
521 // encode raw audio samples into Opus data...
522 bytes = opus_encode(clc.opusEncoder, sampbuffer, samples,
523 (unsigned char *) clc.voipOutgoingData,
524 sizeof (clc.voipOutgoingData));
525 if ( bytes <= 0 ) {
526 Com_DPrintf("VoIP: Error encoding %d samples\n", samples);
527 bytes = 0;
528 }
529
530 clc.voipPower = (voipPower / (32768.0f * 32768.0f *
531 ((float) samples))) * 100.0f;
532
533 if ((useVad) && (clc.voipPower < cl_voipVADThreshold->value)) {
534 CL_VoipNewGeneration(); // no "talk" for at least 1/4 second.
535 } else {
536 clc.voipOutgoingDataSize = bytes;
537 clc.voipOutgoingDataFrames = voipFrames;
538
539 Com_DPrintf("VoIP: Send %d frames, %d bytes, %f power\n",
540 voipFrames, bytes, clc.voipPower);
541
542 #if 0
543 static FILE *encio = NULL;
544 if (encio == NULL) encio = fopen("voip-outgoing-encoded.bin", "wb");
545 if (encio != NULL) { fwrite(clc.voipOutgoingData, bytes, 1, encio); fflush(encio); }
546 static FILE *decio = NULL;
547 if (decio == NULL) decio = fopen("voip-outgoing-decoded.bin", "wb");
548 if (decio != NULL) { fwrite(sampbuffer, voipFrames * VOIP_MAX_FRAME_SAMPLES * 2, 1, decio); fflush(decio); }
549 #endif
550 }
551 }
552 }
553
554 // User requested we stop recording, and we've now processed the last of
555 // any previously-buffered data. Pause the capture device, etc.
556 if (finalFrame) {
557 S_StopCapture();
558 S_MasterGain(1.0f);
559 clc.voipPower = 0.0f; // force this value so it doesn't linger.
560 }
561}
562#endif

572/*
573======================
574CL_AddReliableCommand
575
576The given command will be transmitted to the server, and is gauranteed to
577not have future usercmd_t executed before it is executed
578======================
579*/
580void CL_AddReliableCommand(const char *cmd, qboolean isDisconnectCmd)
581{
582 int unacknowledged = clc.reliableSequence - clc.reliableAcknowledge;
583
584 // if we would be losing an old command that hasn't been acknowledged,
585 // we must drop the connection
586 // also leave one slot open for the disconnect command in this case.
587
588 if ((isDisconnectCmd && unacknowledged > MAX_RELIABLE_COMMANDS) ||
589 (!isDisconnectCmd && unacknowledged >= MAX_RELIABLE_COMMANDS))
590 {
591 if(com_errorEntered)
592 return;
593 else
594 Com_Error(ERR_DROP, "Client command overflow");
595 }
596
597 Q_strncpyz(clc.reliableCommands[++clc.reliableSequence & (MAX_RELIABLE_COMMANDS - 1)],
598 cmd, sizeof(*clc.reliableCommands));
599}

617void CL_WriteDemoMessage ( msg_t *msg, int headerBytes ) {

619

623 FS_Write (&swlen, 4, clc.demofile);
624 // skip the packet sequencing information
625 len = msg->cursize - headerBytes;
626 swlen = LittleLong(len);
627 FS_Write (&swlen, 4, clc.demofile);
628 FS_Write ( msg->data + headerBytes, len, clc.demofile );
629}

691void CL_Record_f( void ) {
692 char name[MAX_OSPATH];
693 byte bufData[MAX_MSGLEN];
694 msg_t buf;
695 int i;
696 int len;
697 entityState_t *ent;

708 Com_Printf ("Already recording.\n");
709 }
710 return;
711 }
712
713 if ( clc.state != CA_ACTIVE ) {
714 Com_Printf ("You must be in a level to record.\n");
715 return;
716 }
717
718 // sync 0 doesn't prevent recording, so not forcing it off .. everyone does g_sync 1 ; record ; g_sync 0 ..

720 Com_Printf (S_COLOR_YELLOW "WARNING: You should set 'g_synchronousClients 1' for smoother demo recording\n");

723 if ( Cmd_Argc() == 2 ) {

725 Q_strncpyz( demoName, s, sizeof( demoName ) );
726#ifdef LEGACY_PROTOCOL

729 else
730#endif
731 Com_sprintf(name, sizeof(name), "demos/%s.%s%d", demoName, DEMOEXT, com_protocol->integer);
732 } else {
733 int number;
734
735 // scan for a free demo name
736 for ( number = 0 ; number <= 9999 ; number++ ) {
737 CL_DemoFilename( number, demoName, sizeof( demoName ) );
738#ifdef LEGACY_PROTOCOL
739 if(clc.compat)
740 Com_sprintf(name, sizeof(name), "demos/%s.%s%d", demoName, DEMOEXT, com_legacyprotocol->integer);
741 else
742#endif
743 Com_sprintf(name, sizeof(name), "demos/%s.%s%d", demoName, DEMOEXT, com_protocol->integer);
744
745 if (!FS_FileExists(name))
746 break; // file doesn't exist
747 }
748 }
749
750 // open the demo file
751
752 Com_Printf ("recording to %s.\n", name);
753 clc.demofile = FS_FOpenFileWrite( name );
754 if ( !clc.demofile ) {
755 Com_Printf ("ERROR: couldn't open.\n");
756 return;
757 }
758 clc.demorecording = qtrue;
759 if (Cvar_VariableValue("ui_recordSPDemo")) {
760 clc.spDemoRecording = qtrue;
761 } else {
762 clc.spDemoRecording = qfalse;
763 }
764
765 Q_strncpyz( clc.demoName, demoName, sizeof( clc.demoName ) );
766
767 // don't start saving messages until a non-delta compressed message is received
768 clc.demowaiting = qtrue;
769
770 // write out the gamestate message
771 MSG_Init (&buf, bufData, sizeof(bufData));
772 MSG_Bitstream(&buf);
773
774 // NOTE, MRE: all server->client messages now acknowledge
775 MSG_WriteLong( &buf, clc.reliableSequence );
776
777 MSG_WriteByte (&buf, svc_gamestate);
778 MSG_WriteLong (&buf, clc.serverCommandSequence );
779
780 // configstrings
781 for ( i = 0 ; i < MAX_CONFIGSTRINGS ; i++ ) {
782 if ( !cl.gameState.stringOffsets[i] ) {
783 continue;
784 }
785 s = cl.gameState.stringData + cl.gameState.stringOffsets[i];
786 MSG_WriteByte (&buf, svc_configstring);
787 MSG_WriteShort (&buf, i);
788 MSG_WriteBigString (&buf, s);
789 }
790
791 // baselines
792 Com_Memset (&nullstate, 0, sizeof(nullstate));
793 for ( i = 0; i < MAX_GENTITIES ; i++ ) {
794 ent = &cl.entityBaselines[i];
795 if ( !ent->number ) {
796 continue;
797 }
798 MSG_WriteByte (&buf, svc_baseline);
799 MSG_WriteDeltaEntity (&buf, &nullstate, ent, qtrue );
800 }
801
802 MSG_WriteByte( &buf, svc_EOF );
803
804 // finished writing the gamestate stuff
805
806 // write the client num
807 MSG_WriteLong(&buf, clc.clientNum);
808 // write the checksum feed
809 MSG_WriteLong(&buf, clc.checksumFeed);
810
811 // finished writing the client packet
812 MSG_WriteByte( &buf, svc_EOF );
813
814 // write it to the demo file
815 len = LittleLong( clc.serverMessageSequence - 1 );
816 FS_Write (&len, 4, clc.demofile);
817
818 len = LittleLong (buf.cursize);
819 FS_Write (&len, 4, clc.demofile);
820 FS_Write (buf.data, buf.cursize, clc.demofile);
821
822 // the rest of the demo file will be copied from net messages
823}

833/*
834=================
835CL_DemoFrameDurationSDev
836=================
837*/
838static float CL_DemoFrameDurationSDev( void )
839{
840 int i;
841 int numFrames;
842 float mean = 0.0f;
843 float variance = 0.0f;
844
845 if( ( clc.timeDemoFrames - 1 ) > MAX_TIMEDEMO_DURATIONS )
846 numFrames = MAX_TIMEDEMO_DURATIONS;
847 else
848 numFrames = clc.timeDemoFrames - 1;
849
850 for( i = 0; i < numFrames; i++ )
851 mean += clc.timeDemoDurations[ i ];
852 mean /= numFrames;
853
854 for( i = 0; i < numFrames; i++ )
855 {
856 float x = clc.timeDemoDurations[ i ];
857
858 variance += ( ( x - mean ) * ( x - mean ) );
859 }
860 variance /= numFrames;
861
862 return sqrt( variance );
863}

1555
1556Authorization server protocol
1557-----------------------------
1558
1559All commands are text in Q3 out of band packets (leading 0xff 0xff 0xff 0xff).
1560
1561Whenever the client tries to get a challenge from the server it wants to
1562connect to, it also blindly fires off a packet to the authorize server:
1563
1564getKeyAuthorize <challenge> <cdkey>

1566cdkey may be "demo"
1567
1568
1569#OLD The authorize server returns a:
1570#OLD
1571#OLD keyAthorize <challenge> <accept | deny>
1572#OLD
1573#OLD A client will be accepted if the cdkey is valid and it has not been used by any other IP
1574#OLD address in the last 15 minutes.
1575
1576
1577The server sends a:
1578
1579getIpAuthorize <challenge> <ip>
1580
1581The authorize server returns a:
1582
1583ipAuthorize <challenge> <accept | deny | demo | unknown >
1584
1585A client will be accepted if a valid cdkey was sent by that ip (only) in the last 15 minutes.
1586If no response is received from the authorize server after two tries, the client will be let
1587in anyway.
1588===================
1589*/

2462/*
2463===================
2464CL_ServersResponsePacket
2465===================
2466*/
2467void CL_ServersResponsePacket( const netadr_t* from, msg_t *msg, qboolean extended ) {
2468 int i, j, count, total;
2469 netadr_t addresses[MAX_SERVERSPERPACKET];
2470 int numservers;
2471 byte* buffptr;
2472 byte* buffend;
2473
2474 Com_Printf("CL_ServersResponsePacket from %s\n", NET_AdrToStringwPort(*from));
2475
2476 if (cls.numglobalservers == -1) {
2477 // state to detect lack of servers or lack of response
2478 cls.numglobalservers = 0;
2479 cls.numGlobalServerAddresses = 0;
2480 }
2481
2482 // parse through server response string
2483 numservers = 0;
2484 buffptr = msg->data;
2485 buffend = buffptr + msg->cursize;
2486
2487 // advance to initial token
2488 do
2489 {
2490 if(*buffptr == '\\' || (extended && *buffptr == '/'))
2491 break;
2492
2493 buffptr++;
2494 } while (buffptr < buffend);
2495
2496 while (buffptr + 1 < buffend)
2497 {
2498 // IPv4 address
2499 if (*buffptr == '\\')
2500 {
2501 buffptr++;
2502
2503 if (buffend - buffptr < sizeof(addresses[numservers].ip) + sizeof(addresses[numservers].port) + 1)
2504 break;
2505
2506 for(i = 0; i < sizeof(addresses[numservers].ip); i++)
2507 addresses[numservers].ip[i] = *buffptr++;
2508
2509 addresses[numservers].type = NA_IP;
2510 }
2511 // IPv6 address, if it's an extended response
2512 else if (extended && *buffptr == '/')
2513 {
2514 buffptr++;
2515
2516 if (buffend - buffptr < sizeof(addresses[numservers].ip6) + sizeof(addresses[numservers].port) + 1)
2517 break;
2518
2519 for(i = 0; i < sizeof(addresses[numservers].ip6); i++)
2520 addresses[numservers].ip6[i] = *buffptr++;
2521
2522 addresses[numservers].type = NA_IP6;
2523 addresses[numservers].scope_id = from->scope_id;
2524 }
2525 else
2526 // syntax error!
2527 break;
2528
2529 // parse out port
2530 addresses[numservers].port = (*buffptr++) << 8;
2531 addresses[numservers].port += *buffptr++;
2532 addresses[numservers].port = BigShort( addresses[numservers].port );
2533
2534 // syntax check
2535 if (*buffptr != '\\' && *buffptr != '/')
2536 break;
2537
2538 numservers++;
2539 if (numservers >= MAX_SERVERSPERPACKET)
2540 break;
2541 }
2542
2543 count = cls.numglobalservers;
2544
2545 for (i = 0; i < numservers && count < MAX_GLOBAL_SERVERS; i++) {
2546 // build net address
2547 serverInfo_t *server = &cls.globalServers[count];
2548
2549 // Tequila: It's possible to have sent many master server requests. Then
2550 // we may receive many times the same addresses from the master server.
2551 // We just avoid to add a server if it is still in the global servers list.
2552 for (j = 0; j < count; j++)
2553 {
2554 if (NET_CompareAdr(cls.globalServers[j].adr, addresses[i]))
2555 break;
2556 }
2557
2558 if (j < count)
2559 continue;
2560
2561 CL_InitServerInfo( server, &addresses[i] );
2562 // advance to next slot
2563 count++;
2564 }
2565
2566 // if getting the global list
2567 if ( count >= MAX_GLOBAL_SERVERS && cls.numGlobalServerAddresses < MAX_GLOBAL_SERVERS )
2568 {
2569 // if we couldn't store the servers in the main list anymore
2570 for (; i < numservers && cls.numGlobalServerAddresses < MAX_GLOBAL_SERVERS; i++)
2571 {
2572 // just store the addresses in an additional list
2573 cls.globalServerAddresses[cls.numGlobalServerAddresses++] = addresses[i];
2574 }
2575 }
2576
2577 cls.numglobalservers = count;
2578 total = count + cls.numGlobalServerAddresses;
2579
2580 Com_Printf("%d servers parsed (total %d)\n", numservers, total);
2581}

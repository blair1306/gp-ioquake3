173static void RllSetupTable( void )
174{
175 int z;
176
177 for (z=0;z<128;z++) {

185//-----------------------------------------------------------------------------
186// RllDecodeMonoToMono
187//
188// Decode mono source data into a mono buffer.
189//
190// Parameters: from -> buffer holding encoded data
191// to -> buffer to hold decoded data
192// size = number of bytes of input (= # of shorts of output)
193// signedOutput = 0 for unsigned output, non-zero for signed output
194// flag = flags from asset header
195//
196// Returns: Number of samples placed in output buffer
197//-----------------------------------------------------------------------------
198long RllDecodeMonoToMono(unsigned char *from,short *to,unsigned int size,char signedOutput ,unsigned short flag)
199{
200 unsigned int z;
201 int prev;
202
203 if (signedOutput)
204 prev = flag - 0x8000;
205 else
206 prev = flag;
207
208 for (z=0;z<size;z++) {
209 prev = to[z] = (short)(prev + cin.sqrTable[from[z]]);
210 }
211 return size; //*sizeof(short));
212}

215//-----------------------------------------------------------------------------
216// RllDecodeMonoToStereo
217//
218// Decode mono source data into a stereo buffer. Output is 4 times the number
219// of bytes in the input.
220//
221// Parameters: from -> buffer holding encoded data
222// to -> buffer to hold decoded data
223// size = number of bytes of input (= 1/4 # of bytes of output)
224// signedOutput = 0 for unsigned output, non-zero for signed output
225// flag = flags from asset header
226//
227// Returns: Number of samples placed in output buffer
228//-----------------------------------------------------------------------------
229long RllDecodeMonoToStereo(unsigned char *from,short *to,unsigned int size,char signedOutput,unsigned short flag)
230{
231 unsigned int z;
232 int prev;
233
234 if (signedOutput)
235 prev = flag - 0x8000;
236 else
237 prev = flag;
238
239 for (z = 0; z < size; z++) {
240 prev = (short)(prev + cin.sqrTable[from[z]]);
241 to[z*2+0] = to[z*2+1] = (short)(prev);
242 }
243
244 return size; // * 2 * sizeof(short));
245}

248//-----------------------------------------------------------------------------
249// RllDecodeStereoToStereo
250//
251// Decode stereo source data into a stereo buffer.
252//
253// Parameters: from -> buffer holding encoded data
254// to -> buffer to hold decoded data
255// size = number of bytes of input (= 1/2 # of bytes of output)
256// signedOutput = 0 for unsigned output, non-zero for signed output
257// flag = flags from asset header
258//
259// Returns: Number of samples placed in output buffer
260//-----------------------------------------------------------------------------
261long RllDecodeStereoToStereo(unsigned char *from,short *to,unsigned int size,char signedOutput, unsigned short flag)
262{
263 unsigned int z;
264 unsigned char *zz = from;
265 int prevL, prevR;
266
267 if (signedOutput) {
268 prevL = (flag & 0xff00) - 0x8000;
269 prevR = ((flag & 0x00ff) << 8) - 0x8000;
270 } else {
271 prevL = flag & 0xff00;
272 prevR = (flag & 0x00ff) << 8;
273 }
274
275 for (z=0;z<size;z+=2) {
276 prevL = (short)(prevL + cin.sqrTable[*zz++]);
277 prevR = (short)(prevR + cin.sqrTable[*zz++]);
278 to[z+0] = (short)(prevL);
279 to[z+1] = (short)(prevR);
280 }
281
282 return (size>>1); //*sizeof(short));
283}

286//-----------------------------------------------------------------------------
287// RllDecodeStereoToMono
288//
289// Decode stereo source data into a mono buffer.
290//
291// Parameters: from -> buffer holding encoded data
292// to -> buffer to hold decoded data
293// size = number of bytes of input (= # of bytes of output)
294// signedOutput = 0 for unsigned output, non-zero for signed output
295// flag = flags from asset header
296//
297// Returns: Number of samples placed in output buffer
298//-----------------------------------------------------------------------------
299long RllDecodeStereoToMono(unsigned char *from,short *to,unsigned int size,char signedOutput, unsigned short flag)
300{
301 unsigned int z;
302 int prevL,prevR;
303
304 if (signedOutput) {
305 prevL = (flag & 0xff00) - 0x8000;
306 prevR = ((flag & 0x00ff) << 8) -0x8000;
307 } else {
308 prevL = flag & 0xff00;
309 prevR = (flag & 0x00ff) << 8;
310 }
311
312 for (z=0;z<size;z+=1) {
313 prevL= prevL + cin.sqrTable[from[z*2]];
314 prevR = prevR + cin.sqrTable[from[z*2+1]];
315 to[z] = (short)((prevL + prevR)/2);
316 }
317
318 return size;
319}

414/******************************************************************************
415*
416* Function:
417*
418* Description:
419*
420******************************************************************************/
421
422static void blitVQQuad32fs( byte **status, unsigned char *data )
423{
424unsigned short newd, celdata, code;
425unsigned int index, i;
426int spl;
427
428 newd = 0;
429 celdata = 0;
430 index = 0;
431
432 spl = cinTable[currentHandle].samplesPerLine;
433
434 do {
435 if (!newd) {
436 newd = 7;
437 celdata = data[0] + data[1]*256;
438 data += 2;
439 } else {
440 newd--;
441 }
442
443 code = (unsigned short)(celdata&0xc000);
444 celdata <<= 2;
445
446 switch (code) {
447 case 0x8000: // vq code
448 blit8_32( (byte *)&vq8[(*data)*128], status[index], spl );
449 data++;
450 index += 5;
451 break;
452 case 0xc000: // drop
453 index++; // skip 8x8
454 for(i=0;i<4;i++) {
455 if (!newd) {
456 newd = 7;
457 celdata = data[0] + data[1]*256;
458 data += 2;
459 } else {
460 newd--;
461 }
462
463 code = (unsigned short)(celdata&0xc000); celdata <<= 2;
464
465 switch (code) { // code in top two bits of code
466 case 0x8000: // 4x4 vq code
467 blit4_32( (byte *)&vq4[(*data)*32], status[index], spl );
468 data++;
469 break;
470 case 0xc000: // 2x2 vq code
471 blit2_32( (byte *)&vq2[(*data)*8], status[index], spl );
472 data++;
473 blit2_32( (byte *)&vq2[(*data)*8], status[index]+8, spl );
474 data++;
475 blit2_32( (byte *)&vq2[(*data)*8], status[index]+spl*2, spl );
476 data++;
477 blit2_32( (byte *)&vq2[(*data)*8], status[index]+spl*2+8, spl );
478 data++;
479 break;
480 case 0x4000: // motion compensation
481 move4_32( status[index] + cin.mcomp[(*data)], status[index], spl );
482 data++;
483 break;
484 }
485 index++;
486 }
487 break;
488 case 0x4000: // motion compensation
489 move8_32( status[index] + cin.mcomp[(*data)], status[index], spl );
490 data++;
491 index += 5;
492 break;
493 case 0x0000:
494 index += 5;
495 break;
496 }
497 } while ( status[index] != NULL );
498}

508static void ROQ_GenYUVTables( void )
509{
510 float t_ub,t_vr,t_ug,t_vg;
511 long i;
512
513 t_ub = (1.77200f/2.0f) * (float)(1<<6) + 0.5f;
514 t_vr = (1.40200f/2.0f) * (float)(1<<6) + 0.5f;
515 t_ug = (0.34414f/2.0f) * (float)(1<<6) + 0.5f;
516 t_vg = (0.71414f/2.0f) * (float)(1<<6) + 0.5f;
517 for(i=0;i<256;i++) {
518 float x = (float)(2 * i - 255);
519
520 ROQ_UB_tab[i] = (long)( ( t_ub * x) + (1<<5));
521 ROQ_VR_tab[i] = (long)( ( t_vr * x) + (1<<5));
522 ROQ_UG_tab[i] = (long)( (-t_ug * x) );
523 ROQ_VG_tab[i] = (long)( (-t_vg * x) + (1<<5));
524 ROQ_YY_tab[i] = (long)( (i << 6) | (i >> 2) );
525 }
526}

572static unsigned short yuv_to_rgb( long y, long u, long v )
573{
574 long r,g,b,YY = (long)(ROQ_YY_tab[(y)]);
575
576 r = (YY + ROQ_VR_tab[v]) >> 9;
577 g = (YY + ROQ_UG_tab[u] + ROQ_VG_tab[v]) >> 8;
578 b = (YY + ROQ_UB_tab[u]) >> 9;
579
580 if (r<0) r = 0;
581 if (g<0) g = 0;
582 if (b<0) b = 0;
583 if (r > 31) r = 31;
584 if (g > 63) g = 63;
585 if (b > 31) b = 31;
586
587 return (unsigned short)((r<<11)+(g<<5)+(b));
588}

590/******************************************************************************
591*
592* Function:
593*
594* Description:
595*
596******************************************************************************/
597static unsigned int yuv_to_rgb24( long y, long u, long v )
598{
599 long r,g,b,YY = (long)(ROQ_YY_tab[(y)]);
600
601 r = (YY + ROQ_VR_tab[v]) >> 6;
602 g = (YY + ROQ_UG_tab[u] + ROQ_VG_tab[v]) >> 6;
603 b = (YY + ROQ_UB_tab[u]) >> 6;
604
605 if (r<0) r = 0;
606 if (g<0) g = 0;
607 if (b<0) b = 0;
608 if (r > 255) r = 255;
609 if (g > 255) g = 255;
610 if (b > 255) b = 255;
611
612 return LittleLong ((r)|(g<<8)|(b<<16)|(255<<24));
613}

623static void decodeCodeBook( byte *input, unsigned short roq_flags )
624{
625 long i, j, two, four;
626 unsigned short *aptr, *bptr, *cptr, *dptr;
627 long y0,y1,y2,y3,cr,cb;
628 byte *bbptr, *baptr, *bcptr, *bdptr;
629 union {
630 unsigned int *i;
631 unsigned short *s;
632 } iaptr, ibptr, icptr, idptr;
633
634 if (!roq_flags) {
635 two = four = 256;
636 } else {
637 two = roq_flags>>8;
638 if (!two) two = 256;
639 four = roq_flags&0xff;
640 }
641
642 four *= 2;
643
644 bptr = (unsigned short *)vq2;
645
646 if (!cinTable[currentHandle].half) {
647 if (!cinTable[currentHandle].smootheddouble) {
648//
649// normal height
650//
651 if (cinTable[currentHandle].samplesPerPixel==2) {
652 for(i=0;i<two;i++) {
653 y0 = (long)*input++;
654 y1 = (long)*input++;
655 y2 = (long)*input++;
656 y3 = (long)*input++;
657 cr = (long)*input++;
658 cb = (long)*input++;
659 *bptr++ = yuv_to_rgb( y0, cr, cb );
660 *bptr++ = yuv_to_rgb( y1, cr, cb );
661 *bptr++ = yuv_to_rgb( y2, cr, cb );
662 *bptr++ = yuv_to_rgb( y3, cr, cb );
663 }
664
665 cptr = (unsigned short *)vq4;
666 dptr = (unsigned short *)vq8;
667
668 for(i=0;i<four;i++) {
669 aptr = (unsigned short *)vq2 + (*input++)*4;
670 bptr = (unsigned short *)vq2 + (*input++)*4;
671 for(j=0;j<2;j++)
672 VQ2TO4(aptr,bptr,cptr,dptr);
673 }
674 } else if (cinTable[currentHandle].samplesPerPixel==4) {
675 ibptr.s = bptr;
676 for(i=0;i<two;i++) {
677 y0 = (long)*input++;
678 y1 = (long)*input++;
679 y2 = (long)*input++;
680 y3 = (long)*input++;
681 cr = (long)*input++;
682 cb = (long)*input++;
683 *ibptr.i++ = yuv_to_rgb24( y0, cr, cb );
684 *ibptr.i++ = yuv_to_rgb24( y1, cr, cb );
685 *ibptr.i++ = yuv_to_rgb24( y2, cr, cb );
686 *ibptr.i++ = yuv_to_rgb24( y3, cr, cb );
687 }
688
689 icptr.s = vq4;
690 idptr.s = vq8;
691
692 for(i=0;i<four;i++) {
693 iaptr.s = vq2;
694 iaptr.i += (*input++)*4;
695 ibptr.s = vq2;
696 ibptr.i += (*input++)*4;
697 for(j=0;j<2;j++)
698 VQ2TO4(iaptr.i, ibptr.i, icptr.i, idptr.i);
699 }
700 } else if (cinTable[currentHandle].samplesPerPixel==1) {
701 bbptr = (byte *)bptr;
702 for(i=0;i<two;i++) {
703 *bbptr++ = cinTable[currentHandle].gray[*input++];
704 *bbptr++ = cinTable[currentHandle].gray[*input++];
705 *bbptr++ = cinTable[currentHandle].gray[*input++];
706 *bbptr++ = cinTable[currentHandle].gray[*input]; input +=3;
707 }
708
709 bcptr = (byte *)vq4;
710 bdptr = (byte *)vq8;
711
712 for(i=0;i<four;i++) {
713 baptr = (byte *)vq2 + (*input++)*4;
714 bbptr = (byte *)vq2 + (*input++)*4;
715 for(j=0;j<2;j++)
716 VQ2TO4(baptr,bbptr,bcptr,bdptr);
717 }
718 }
719 } else {
720//
721// double height, smoothed
722//
723 if (cinTable[currentHandle].samplesPerPixel==2) {
724 for(i=0;i<two;i++) {
725 y0 = (long)*input++;
726 y1 = (long)*input++;
727 y2 = (long)*input++;
728 y3 = (long)*input++;
729 cr = (long)*input++;
730 cb = (long)*input++;
731 *bptr++ = yuv_to_rgb( y0, cr, cb );
732 *bptr++ = yuv_to_rgb( y1, cr, cb );
733 *bptr++ = yuv_to_rgb( ((y0*3)+y2)/4, cr, cb );
734 *bptr++ = yuv_to_rgb( ((y1*3)+y3)/4, cr, cb );
735 *bptr++ = yuv_to_rgb( (y0+(y2*3))/4, cr, cb );
736 *bptr++ = yuv_to_rgb( (y1+(y3*3))/4, cr, cb );
737 *bptr++ = yuv_to_rgb( y2, cr, cb );
738 *bptr++ = yuv_to_rgb( y3, cr, cb );
739 }
740
741 cptr = (unsigned short *)vq4;
742 dptr = (unsigned short *)vq8;
743
744 for(i=0;i<four;i++) {
745 aptr = (unsigned short *)vq2 + (*input++)*8;
746 bptr = (unsigned short *)vq2 + (*input++)*8;
747 for(j=0;j<2;j++) {
748 VQ2TO4(aptr,bptr,cptr,dptr);
749 VQ2TO4(aptr,bptr,cptr,dptr);
750 }
751 }
752 } else if (cinTable[currentHandle].samplesPerPixel==4) {
753 ibptr.s = bptr;
754 for(i=0;i<two;i++) {
755 y0 = (long)*input++;
756 y1 = (long)*input++;
757 y2 = (long)*input++;
758 y3 = (long)*input++;
759 cr = (long)*input++;
760 cb = (long)*input++;
761 *ibptr.i++ = yuv_to_rgb24( y0, cr, cb );
762 *ibptr.i++ = yuv_to_rgb24( y1, cr, cb );
763 *ibptr.i++ = yuv_to_rgb24( ((y0*3)+y2)/4, cr, cb );
764 *ibptr.i++ = yuv_to_rgb24( ((y1*3)+y3)/4, cr, cb );
765 *ibptr.i++ = yuv_to_rgb24( (y0+(y2*3))/4, cr, cb );
766 *ibptr.i++ = yuv_to_rgb24( (y1+(y3*3))/4, cr, cb );
767 *ibptr.i++ = yuv_to_rgb24( y2, cr, cb );
768 *ibptr.i++ = yuv_to_rgb24( y3, cr, cb );
769 }
770
771 icptr.s = vq4;

891static void recurseQuad( long startX, long startY, long quadSize, long xOff, long yOff )
892{
893 byte *scroff;
894 long bigx, bigy, lowx, lowy, useY;
895 long offset;

905
906 if ( (startX >= lowx) && (startX+quadSize) <= (bigx) && (startY+quadSize) <= (bigy) && (startY >= lowy) && quadSize <= MAXSIZE) {
907 useY = startY;
908 scroff = cin.linbuf + (useY+((cinTable[currentHandle].CIN_HEIGHT-bigy)>>1)+yOff)*(cinTable[currentHandle].samplesPerLine) + (((startX+xOff))*cinTable[currentHandle].samplesPerPixel);

911 cin.qStatus[1][cinTable[currentHandle].onQuad++] = scroff+offset;
912 }
913
914 if ( quadSize != MINSIZE ) {
915 quadSize >>= 1;
916 recurseQuad( startX, startY , quadSize, xOff, yOff );
917 recurseQuad( startX+quadSize, startY , quadSize, xOff, yOff );
918 recurseQuad( startX, startY+quadSize , quadSize, xOff, yOff );
919 recurseQuad( startX+quadSize, startY+quadSize , quadSize, xOff, yOff );

1091/******************************************************************************
1092*
1093* Function:
1094*
1095* Description:
1096*
1097******************************************************************************/
1098
1099static void RoQInterrupt(void)
1100{
1101 byte *framedata;
1102 short sbuf[32768];
1103 int ssize;
1104
1105 if (currentHandle < 0) return;
1106
1107 FS_Read( cin.file, cinTable[currentHandle].RoQFrameSize+8, cinTable[currentHandle].iFile );
1108 if ( cinTable[currentHandle].RoQPlayed >= cinTable[currentHandle].ROQSize ) {
1109 if (cinTable[currentHandle].holdAtEnd==qfalse) {
1110 if (cinTable[currentHandle].looping) {
1111 RoQReset();
1112 } else {
1113 cinTable[currentHandle].status = FMV_EOF;
1114 }
1115 } else {
1116 cinTable[currentHandle].status = FMV_IDLE;
1117 }
1118 return;
1119 }
1120
1121 framedata = cin.file;
1122//
1123// new frame is ready
1124//
1125redump:
1126 switch(cinTable[currentHandle].roq_id)
1127 {
1128 case ROQ_QUAD_VQ:
1129 if ((cinTable[currentHandle].numQuads&1)) {
1130 cinTable[currentHandle].normalBuffer0 = cinTable[currentHandle].t[1];
1131 RoQPrepMcomp( cinTable[currentHandle].roqF0, cinTable[currentHandle].roqF1 );
1132 cinTable[currentHandle].VQ1( (byte *)cin.qStatus[1], framedata);
1133 cinTable[currentHandle].buf = cin.linbuf + cinTable[currentHandle].screenDelta;
1134 } else {
1135 cinTable[currentHandle].normalBuffer0 = cinTable[currentHandle].t[0];
1136 RoQPrepMcomp( cinTable[currentHandle].roqF0, cinTable[currentHandle].roqF1 );
1137 cinTable[currentHandle].VQ0( (byte *)cin.qStatus[0], framedata );
1138 cinTable[currentHandle].buf = cin.linbuf;
1139 }
1140 if (cinTable[currentHandle].numQuads == 0) { // first frame
1141 Com_Memcpy(cin.linbuf+cinTable[currentHandle].screenDelta, cin.linbuf, cinTable[currentHandle].samplesPerLine*cinTable[currentHandle].ysize);
1142 }
1143 cinTable[currentHandle].numQuads++;
1144 cinTable[currentHandle].dirty = qtrue;
1145 break;
1146 case ROQ_CODEBOOK:
1147 decodeCodeBook( framedata, (unsigned short)cinTable[currentHandle].roq_flags );
1148 break;
1149 case ZA_SOUND_MONO:
1150 if (!cinTable[currentHandle].silent) {
1151 ssize = RllDecodeMonoToStereo( framedata, sbuf, cinTable[currentHandle].RoQFrameSize, 0, (unsigned short)cinTable[currentHandle].roq_flags);
1152 S_RawSamples(0, ssize, 22050, 2, 1, (byte *)sbuf, 1.0f, -1);
1153 }
1154 break;
1155 case ZA_SOUND_STEREO:
1156 if (!cinTable[currentHandle].silent) {
1157 if (cinTable[currentHandle].numQuads == -1) {
1158 S_Update();
1159 s_rawend[0] = s_soundtime;
1160 }
1161 ssize = RllDecodeStereoToStereo( framedata, sbuf, cinTable[currentHandle].RoQFrameSize, 0, (unsigned short)cinTable[currentHandle].roq_flags);
1162 S_RawSamples(0, ssize, 22050, 2, 2, (byte *)sbuf, 1.0f, -1);
1163 }
1164 break;
1165 case ROQ_QUAD_INFO:
1166 if (cinTable[currentHandle].numQuads == -1) {
1167 readQuadInfo( framedata );
1168 setupQuad( 0, 0 );
1169 cinTable[currentHandle].startTime = cinTable[currentHandle].lastTime = CL_ScaledMilliseconds();
1170 }
1171 if (cinTable[currentHandle].numQuads != 1) cinTable[currentHandle].numQuads = 0;
1172 break;
1173 case ROQ_PACKET:
1174 cinTable[currentHandle].inMemory = cinTable[currentHandle].roq_flags;
1175 cinTable[currentHandle].RoQFrameSize = 0; // for header
1176 break;
1177 case ROQ_QUAD_HANG:
1178 cinTable[currentHandle].RoQFrameSize = 0;
1179 break;
1180 case ROQ_QUAD_JPEG:
1181 break;
1182 default:
1183 cinTable[currentHandle].status = FMV_EOF;
1184 break;
1185 }
1186//
1187// read in next frame data
1188//
1189 if ( cinTable[currentHandle].RoQPlayed >= cinTable[currentHandle].ROQSize ) {
1190 if (cinTable[currentHandle].holdAtEnd==qfalse) {
1191 if (cinTable[currentHandle].looping) {
1192 RoQReset();
1193 } else {
1194 cinTable[currentHandle].status = FMV_EOF;
1195 }
1196 } else {
1197 cinTable[currentHandle].status = FMV_IDLE;
1198 }
1199 return;
1200 }
1201
1202 framedata += cinTable[currentHandle].RoQFrameSize;
1203 cinTable[currentHandle].roq_id = framedata[0] + framedata[1]*256;
1204 cinTable[currentHandle].RoQFrameSize = framedata[2] + framedata[3]*256 + framedata[4]*65536;
1205 cinTable[currentHandle].roq_flags = framedata[6] + framedata[7]*256;
1206 cinTable[currentHandle].roqF0 = (signed char)framedata[7];
1207 cinTable[currentHandle].roqF1 = (signed char)framedata[6];
1208
1209 if (cinTable[currentHandle].RoQFrameSize>65536||cinTable[currentHandle].roq_id==0x1084) {
1210 Com_DPrintf("roq_size>65536||roq_id==0x1084\n");
1211 cinTable[currentHandle].status = FMV_EOF;
1212 if (cinTable[currentHandle].looping) {
1213 RoQReset();
1214 }
1215 return;
1216 }
1217 if (cinTable[currentHandle].inMemory && (cinTable[currentHandle].status != FMV_EOF)) { cinTable[currentHandle].inMemory--; framedata += 8; goto redump; }
1218//
1219// one more frame hits the dust
1220//
1221// assert(cinTable[currentHandle].RoQFrameSize <= 65536);
1222// r = FS_Read( cin.file, cinTable[currentHandle].RoQFrameSize+8, cinTable[currentHandle].iFile );
1223 cinTable[currentHandle].RoQPlayed += cinTable[currentHandle].RoQFrameSize+8;
1224}

1327/*
1328==================
1329CIN_RunCinematic
1330
1331Fetch and decompress the pending frame
1332==================
1333*/
1334
1335
1336e_status CIN_RunCinematic (int handle)
1337{
1338 int start = 0;
1339 int thisTime = 0;
1340
1341 if (handle < 0 || handle>= MAX_VIDEO_HANDLES || cinTable[handle].status == FMV_EOF) return FMV_EOF;
1342
1343 if (cin.currentHandle != handle) {
1344 currentHandle = handle;
1345 cin.currentHandle = currentHandle;
1346 cinTable[currentHandle].status = FMV_EOF;
1347 RoQReset();
1348 }
1349
1350 if (cinTable[handle].playonwalls < -1)
1351 {
1352 return cinTable[handle].status;
1353 }
1354
1355 currentHandle = handle;
1356
1357 if (cinTable[currentHandle].alterGameState) {
1358 if ( clc.state != CA_CINEMATIC ) {
1359 return cinTable[currentHandle].status;
1360 }
1361 }
1362
1363 if (cinTable[currentHandle].status == FMV_IDLE) {
1364 return cinTable[currentHandle].status;
1365 }
1366
1367 thisTime = CL_ScaledMilliseconds();
1368 if (cinTable[currentHandle].shader && (abs(thisTime - cinTable[currentHandle].lastTime))>100) {
1369 cinTable[currentHandle].startTime += thisTime - cinTable[currentHandle].lastTime;
1370 }
1371 cinTable[currentHandle].tfps = (((CL_ScaledMilliseconds() - cinTable[currentHandle].startTime)*3)/100);
1372
1373 start = cinTable[currentHandle].startTime;
1374 while( (cinTable[currentHandle].tfps != cinTable[currentHandle].numQuads)
1375 && (cinTable[currentHandle].status == FMV_PLAY) )
1376 {
1377 RoQInterrupt();
1378 if (start != cinTable[currentHandle].startTime) {
1379 cinTable[currentHandle].tfps = (((CL_ScaledMilliseconds() - cinTable[currentHandle].startTime)*3)/100);
1380 start = cinTable[currentHandle].startTime;
1381 }
1382 }
1383
1384 cinTable[currentHandle].lastTime = thisTime;
1385
1386 if (cinTable[currentHandle].status == FMV_LOOPED) {
1387 cinTable[currentHandle].status = FMV_PLAY;
1388 }
1389
1390 if (cinTable[currentHandle].status == FMV_EOF) {
1391 if (cinTable[currentHandle].looping) {
1392 RoQReset();
1393 } else {
1394 RoQShutdown();
1395 return FMV_EOF;
1396 }
1397 }
1398
1399 return cinTable[currentHandle].status;
1400}

1509/*
1510==================
1511CIN_ResampleCinematic
1512
1513Resample cinematic to 256x256 and store in buf2
1514==================
1515*/
1516void CIN_ResampleCinematic(int handle, int *buf2) {
1517 int ix, iy, *buf3, xm, ym, ll;
1518 byte *buf;
1519
1520 buf = cinTable[handle].buf;
1521
1522 xm = cinTable[handle].CIN_WIDTH/256;
1523 ym = cinTable[handle].CIN_HEIGHT/256;
1524 ll = 8;
1525 if (cinTable[handle].CIN_WIDTH==512) {
1526 ll = 9;
1527 }
1528
1529 buf3 = (int*)buf;
1530 if (xm==2 && ym==2) {
1531 byte *bc2, *bc3;
1532 int ic, iiy;
1533
1534 bc2 = (byte *)buf2;
1535 bc3 = (byte *)buf3;
1536 for (iy = 0; iy<256; iy++) {
1537 iiy = iy<<12;
1538 for (ix = 0; ix<2048; ix+=8) {
1539 for(ic = ix;ic<(ix+4);ic++) {
1540 *bc2=(bc3[iiy+ic]+bc3[iiy+4+ic]+bc3[iiy+2048+ic]+bc3[iiy+2048+4+ic])>>2;
1541 bc2++;
1542 }
1543 }
1544 }
1545 } else if (xm==2 && ym==1) {
1546 byte *bc2, *bc3;
1547 int ic, iiy;
1548
1549 bc2 = (byte *)buf2;
1550 bc3 = (byte *)buf3;
1551 for (iy = 0; iy<256; iy++) {
1552 iiy = iy<<11;
1553 for (ix = 0; ix<2048; ix+=8) {
1554 for(ic = ix;ic<(ix+4);ic++) {
1555 *bc2=(bc3[iiy+ic]+bc3[iiy+4+ic])>>1;
1556 bc2++;
1557 }
1558 }
1559 }
1560 } else {
1561 for (iy = 0; iy<256; iy++) {
1562 for (ix = 0; ix<256; ix++) {
1563 buf2[(iy<<8)+ix] = buf3[((iy*ym)<<ll) + (ix*xm)];
1564 }
1565 }
1566 }
1567}

29/*
30===============================================================================
31
32KEY BUTTONS
33
34Continuous button event tracking is complicated by the fact that two different
35input sources (say, mouse button 1 and the control key) can both press the
36same button, but the button should only be released when both of the
37pressing key have been released.
38
39When a key event issues a button command (+forward, +attack, etc), it appends
40its key number as argv(1) so it can be matched up with the release.
41
42argv(2) will be set to the time the event happened, which allows exact
43control even at low framerates when the down and up events may both get qued
44at the same time.
45
46===============================================================================
47*/
48

76void IN_KeyDown( kbutton_t *b ) {
77 int k;
78 char *c;
79
80 c = Cmd_Argv(1);
81 if ( c[0] ) {
82 k = atoi(c);
83 } else {
84 k = -1; // typed manually at the console for continuous down
85 }
86
87 if ( k == b->down[0] || k == b->down[1] ) {
88 return; // repeating key
89 }
90
91 if ( !b->down[0] ) {
92 b->down[0] = k;
93 } else if ( !b->down[1] ) {
94 b->down[1] = k;
95 } else {
96 Com_Printf ("Three keys down for a button!\n");
97 return;
98 }
99
100 if ( b->active ) {
101 return; // still down
102 }
103
104 // save timestamp for partial frame summing
105 c = Cmd_Argv(2);
106 b->downtime = atoi(c);
107
108 b->active = qtrue;
109 b->wasPressed = qtrue;
110}
111
112void IN_KeyUp( kbutton_t *b ) {
113 int k;
114 char *c;
115 unsigned uptime;
116
117 c = Cmd_Argv(1);
118 if ( c[0] ) {
119 k = atoi(c);
120 } else {
121 // typed manually at the console, assume for unsticking, so clear all
122 b->down[0] = b->down[1] = 0;
123 b->active = qfalse;
124 return;
125 }
126
127 if ( b->down[0] == k ) {
128 b->down[0] = 0;
129 } else if ( b->down[1] == k ) {
130 b->down[1] = 0;
131 } else {
132 return; // key up without coresponding down (menu pass through)
133 }
134 if ( b->down[0] || b->down[1] ) {
135 return; // some other key is still holding it down
136 }
137
138 b->active = qfalse;
139
140 // save timestamp for partial frame summing
141 c = Cmd_Argv(2);
142 uptime = atoi(c);
143 if ( uptime ) {
144 b->msec += uptime - b->downtime;
145 } else {
146 b->msec += frame_msec / 2;
147 }
148
149 b->active = qfalse;
150}

154/*
155===============
156CL_KeyState
157
158Returns the fraction of the frame that the key was down
159===============
160*/
161float CL_KeyState( kbutton_t *key ) {
162 float val;
163 int msec;
164
165 msec = key->msec;
166 key->msec = 0;
167
168 if ( key->active ) {
169 // still down
170 if ( !key->downtime ) {
171 msec = com_frameTime;
172 } else {
173 msec += com_frameTime - key->downtime;
174 }
175 key->downtime = com_frameTime;
176 }
177
178#if 0
179 if (msec) {
180 Com_Printf ("%i ", msec);
181 }
182#endif
183
184 val = (float)msec / frame_msec;
185 if ( val < 0 ) {
186 val = 0;
187 }
188 if ( val > 1 ) {
189 val = 1;
190 }
191
192 return val;
193}

285/*
286================
287CL_AdjustAngles
288
289Moves the local angle positions
290================
291*/
292void CL_AdjustAngles( void ) {
293 float speed;
294
295 if ( in_speed.active ) {
296 speed = 0.001 * cls.frametime * cl_anglespeedkey->value;
297 } else {
298 speed = 0.001 * cls.frametime;
299 }
300
301 if ( !in_strafe.active ) {
302 cl.viewangles[YAW] -= speed*cl_yawspeed->value*CL_KeyState (&in_right);
303 cl.viewangles[YAW] += speed*cl_yawspeed->value*CL_KeyState (&in_left);
304 }
305
306 cl.viewangles[PITCH] -= speed*cl_pitchspeed->value * CL_KeyState (&in_lookup);
307 cl.viewangles[PITCH] += speed*cl_pitchspeed->value * CL_KeyState (&in_lookdown);
308}

310/*
311================
312CL_KeyMove
313
314Sets the usercmd_t based on key states
315================
316*/
317void CL_KeyMove( usercmd_t *cmd ) {
318 int movespeed;
319 int forward, side, up;
320
321 //
322 // adjust for speed key / running
323 // the walking flag is to keep animations consistant
324 // even during acceleration and develeration
325 //
326 if ( in_speed.active ^ cl_run->integer ) {
327 movespeed = 127;
328 cmd->buttons &= ~BUTTON_WALKING;
329 } else {
330 cmd->buttons |= BUTTON_WALKING;
331 movespeed = 64;
332 }
333
334 forward = 0;
335 side = 0;
336 up = 0;
337 if ( in_strafe.active ) {
338 side += movespeed * CL_KeyState (&in_right);
339 side -= movespeed * CL_KeyState (&in_left);
340 }
341
342 side += movespeed * CL_KeyState (&in_moveright);
343 side -= movespeed * CL_KeyState (&in_moveleft);
344
345
346 up += movespeed * CL_KeyState (&in_up);
347 up -= movespeed * CL_KeyState (&in_down);
348
349 forward += movespeed * CL_KeyState (&in_forward);
350 forward -= movespeed * CL_KeyState (&in_back);
351
352 cmd->forwardmove = ClampChar( forward );
353 cmd->rightmove = ClampChar( side );
354 cmd->upmove = ClampChar( up );
355}

387/*
388=================
389CL_JoystickMove
390=================
391*/
392void CL_JoystickMove( usercmd_t *cmd ) {
393 float anglespeed;
394
395 float yaw = j_yaw->value * cl.joystickAxis[j_yaw_axis->integer];
396 float right = j_side->value * cl.joystickAxis[j_side_axis->integer];
397 float forward = j_forward->value * cl.joystickAxis[j_forward_axis->integer];
398 float pitch = j_pitch->value * cl.joystickAxis[j_pitch_axis->integer];
399 float up = j_up->value * cl.joystickAxis[j_up_axis->integer];
400
401 if ( !(in_speed.active ^ cl_run->integer) ) {
402 cmd->buttons |= BUTTON_WALKING;
403 }
404
405 if ( in_speed.active ) {
406 anglespeed = 0.001 * cls.frametime * cl_anglespeedkey->value;
407 } else {
408 anglespeed = 0.001 * cls.frametime;
409 }
410
411 if ( !in_strafe.active ) {
412 cl.viewangles[YAW] += anglespeed * yaw;
413 cmd->rightmove = ClampChar( cmd->rightmove + (int)right );
414 } else {
415 cl.viewangles[YAW] += anglespeed * right;
416 cmd->rightmove = ClampChar( cmd->rightmove + (int)yaw );
417 }
418
419 if ( in_mlooking ) {
420 cl.viewangles[PITCH] += anglespeed * forward;
421 cmd->forwardmove = ClampChar( cmd->forwardmove + (int)pitch );
422 } else {
423 cl.viewangles[PITCH] += anglespeed * pitch;
424 cmd->forwardmove = ClampChar( cmd->forwardmove + (int)forward );
425 }
426
427 cmd->upmove = ClampChar( cmd->upmove + (int)up );
428}

430/*
431=================
432CL_MouseMove
433=================
434*/
435
436void CL_MouseMove(usercmd_t *cmd)
437{
438 float mx, my;
439
440 // allow mouse smoothing
441 if (m_filter->integer)
442 {
443 mx = (cl.mouseDx[0] + cl.mouseDx[1]) * 0.5f;
444 my = (cl.mouseDy[0] + cl.mouseDy[1]) * 0.5f;
445 }
446 else
447 {
448 mx = cl.mouseDx[cl.mouseIndex];
449 my = cl.mouseDy[cl.mouseIndex];
450 }
451
452 cl.mouseIndex ^= 1;
453 cl.mouseDx[cl.mouseIndex] = 0;
454 cl.mouseDy[cl.mouseIndex] = 0;
455
456 if (mx == 0.0f && my == 0.0f)
457 return;
458
459 if (cl_mouseAccel->value != 0.0f)
460 {
461 if(cl_mouseAccelStyle->integer == 0)
462 {
463 float accelSensitivity;
464 float rate;
465
466 rate = sqrt(mx * mx + my * my) / (float) frame_msec;
467
468 accelSensitivity = cl_sensitivity->value + rate * cl_mouseAccel->value;
469 mx *= accelSensitivity;
470 my *= accelSensitivity;
471
472 if(cl_showMouseRate->integer)
473 Com_Printf("rate: %f, accelSensitivity: %f\n", rate, accelSensitivity);
474 }
475 else
476 {
477 float rate[2];
478 float power[2];
479
480 // sensitivity remains pretty much unchanged at low speeds
481 // cl_mouseAccel is a power value to how the acceleration is shaped
482 // cl_mouseAccelOffset is the rate for which the acceleration will have doubled the non accelerated amplification
483 // NOTE: decouple the config cvars for independent acceleration setup along X and Y?
484
485 rate[0] = fabs(mx) / (float) frame_msec;
486 rate[1] = fabs(my) / (float) frame_msec;
487 power[0] = powf(rate[0] / cl_mouseAccelOffset->value, cl_mouseAccel->value);
488 power[1] = powf(rate[1] / cl_mouseAccelOffset->value, cl_mouseAccel->value);
489
490 mx = cl_sensitivity->value * (mx + ((mx < 0) ? -power[0] : power[0]) * cl_mouseAccelOffset->value);
491 my = cl_sensitivity->value * (my + ((my < 0) ? -power[1] : power[1]) * cl_mouseAccelOffset->value);
492
493 if(cl_showMouseRate->integer)
494 Com_Printf("ratex: %f, ratey: %f, powx: %f, powy: %f\n", rate[0], rate[1], power[0], power[1]);
495 }
496 }
497 else
498 {
499 mx *= cl_sensitivity->value;
500 my *= cl_sensitivity->value;
501 }
502
503 // ingame FOV
504 mx *= cl.cgameSensitivity;
505 my *= cl.cgameSensitivity;
506
507 // add mouse X/Y movement to cmd
508 if(in_strafe.active)
509 cmd->rightmove = ClampChar(cmd->rightmove + m_side->value * mx);
510 else
511 cl.viewangles[YAW] -= m_yaw->value * mx;
512
513 if ((in_mlooking || cl_freelook->integer) && !in_strafe.active)
514 cl.viewangles[PITCH] += m_pitch->value * my;
515 else
516 cmd->forwardmove = ClampChar(cmd->forwardmove - m_forward->value * my);
517}

520/*
521==============
522CL_CmdButtons
523==============
524*/
525void CL_CmdButtons( usercmd_t *cmd ) {
526 int i;
527
528 //
529 // figure button bits
530 // send a button bit even if the key was pressed and released in
531 // less than a frame
532 //
533 for (i = 0 ; i < 15 ; i++) {
534 if ( in_buttons[i].active || in_buttons[i].wasPressed ) {
535 cmd->buttons |= 1 << i;
536 }
537 in_buttons[i].wasPressed = qfalse;
538 }
539
540 if ( Key_GetCatcher( ) ) {
541 cmd->buttons |= BUTTON_TALK;
542 }
543
544 // allow the game to know if any key at all is
545 // currently pressed, even if it isn't bound to anything
546 if ( anykeydown && Key_GetCatcher( ) == 0 ) {
547 cmd->buttons |= BUTTON_ANY;
548 }
549}

552/*
553==============
554CL_FinishMove
555==============
556*/
557void CL_FinishMove( usercmd_t *cmd ) {
558 int i;
559
560 // copy the state that the cgame is currently sending
561 cmd->weapon = cl.cgameUserCmdValue;
562
563 // send the current server time so the amount of movement
564 // can be determined without allowing cheating
565 cmd->serverTime = cl.serverTime;
566
567 for (i=0 ; i<3 ; i++) {
568 cmd->angles[i] = ANGLE2SHORT(cl.viewangles[i]);
569 }
570}

573/*
574=================
575CL_CreateCmd
576=================
577*/
578usercmd_t CL_CreateCmd( void ) {
579 usercmd_t cmd;
580 vec3_t oldAngles;
581
582 VectorCopy( cl.viewangles, oldAngles );
583
584 // keyboard angle adjustment
585 CL_AdjustAngles ();
586
587 Com_Memset( &cmd, 0, sizeof( cmd ) );
588
589 CL_CmdButtons( &cmd );
590
591 // get basic movement from keyboard
592 CL_KeyMove( &cmd );
593
594 // get basic movement from mouse
595 CL_MouseMove( &cmd );
596
597 // get basic movement from joystick
598 CL_JoystickMove( &cmd );
599
600 // check to make sure the angles haven't wrapped
601 if ( cl.viewangles[PITCH] - oldAngles[PITCH] > 90 ) {
602 cl.viewangles[PITCH] = oldAngles[PITCH] + 90;
603 } else if ( oldAngles[PITCH] - cl.viewangles[PITCH] > 90 ) {
604 cl.viewangles[PITCH] = oldAngles[PITCH] - 90;
605 }
606
607 // store out the final values
608 CL_FinishMove( &cmd );
609
610 // draw debug graphs of turning for mouse testing
611 if ( cl_debugMove->integer ) {
612 if ( cl_debugMove->integer == 1 ) {
613 SCR_DebugGraph( fabs(cl.viewangles[YAW] - oldAngles[YAW]) );
614 }
615 if ( cl_debugMove->integer == 2 ) {
616 SCR_DebugGraph( fabs(cl.viewangles[PITCH] - oldAngles[PITCH]) );
617 }
618 }
619
620 return cmd;
621}

624/*
625=================
626CL_CreateNewCommands
627
628Create a new usercmd_t structure for this frame
629=================
630*/
631void CL_CreateNewCommands( void ) {
632 int cmdNum;
633
634 // no need to create usercmds until we have a gamestate
635 if ( clc.state < CA_PRIMED ) {
636 return;
637 }
638
639 frame_msec = com_frameTime - old_com_frameTime;
640
641 // if running over 1000fps, act as if each frame is 1ms
642 // prevents divisions by zero
643 if ( frame_msec < 1 ) {
644 frame_msec = 1;
645 }
646
647 // if running less than 5fps, truncate the extra time to prevent
648 // unexpected moves after a hitch
649 if ( frame_msec > 200 ) {
650 frame_msec = 200;
651 }
652 old_com_frameTime = com_frameTime;
653
654
655 // generate a command for this frame
656 cl.cmdNumber++;
657 cmdNum = cl.cmdNumber & CMD_MASK;
658 cl.cmds[cmdNum] = CL_CreateCmd ();
659}

661/*
662=================
663CL_ReadyToSendPacket
664
665Returns qfalse if we are over the maxpackets limit
666and should choke back the bandwidth a bit by not sending
667a packet this frame. All the commands will still get
668delivered in the next packet, but saving a header and
669getting more delta compression will reduce total bandwidth.
670=================
671*/
672qboolean CL_ReadyToSendPacket( void ) {
673 int oldPacketNum;
674 int delta;
675
676 // don't send anything if playing back a demo
677 if ( clc.demoplaying || clc.state == CA_CINEMATIC ) {
678 return qfalse;
679 }
680
681 // If we are downloading, we send no less than 50ms between packets
682 if ( *clc.downloadTempName &&
683 cls.realtime - clc.lastPacketSentTime < 50 ) {
684 return qfalse;
685 }
686
687 // if we don't have a valid gamestate yet, only send
688 // one packet a second
689 if ( clc.state != CA_ACTIVE &&
690 clc.state != CA_PRIMED &&
691 !*clc.downloadTempName &&
692 cls.realtime - clc.lastPacketSentTime < 1000 ) {
693 return qfalse;
694 }
695
696 // send every frame for loopbacks
697 if ( clc.netchan.remoteAddress.type == NA_LOOPBACK ) {
698 return qtrue;
699 }
700
701 // send every frame for LAN
702 if ( cl_lanForcePackets->integer && Sys_IsLANAddress( clc.netchan.remoteAddress ) ) {
703 return qtrue;
704 }
705
706 // check for exceeding cl_maxpackets
707 if ( cl_maxpackets->integer < 15 ) {
708 Cvar_Set( "cl_maxpackets", "15" );
709 } else if ( cl_maxpackets->integer > 125 ) {
710 Cvar_Set( "cl_maxpackets", "125" );
711 }
712 oldPacketNum = (clc.netchan.outgoingSequence - 1) & PACKET_MASK;
713 delta = cls.realtime - cl.outPackets[ oldPacketNum ].p_realtime;
714 if ( delta < 1000 / cl_maxpackets->integer ) {
715 // the accumulated commands will go out in the next packet
716 return qfalse;
717 }
718
719 return qtrue;
720}

722/*
723===================
724CL_WritePacket
725
726Create and send the command packet to the server
727Including both the reliable commands and the usercmds
728
729During normal gameplay, a client packet will contain something like:
730
7314 sequence number
7322 qport
7334 serverid
7344 acknowledged sequence number
7354 clc.serverCommandSequence
736<optional reliable commands>
7371 clc_move or clc_moveNoDelta
7381 command count
739<count * usercmds>
740
741===================
742*/
743void CL_WritePacket( void ) {
744 msg_t buf;
745 byte data[MAX_MSGLEN];
746 int i, j;
747 usercmd_t *cmd, *oldcmd;
748 usercmd_t nullcmd;
749 int packetNum;
750 int oldPacketNum;
751 int count, key;
752
753 // don't send anything if playing back a demo
754 if ( clc.demoplaying || clc.state == CA_CINEMATIC ) {
755 return;
756 }
757
758 Com_Memset( &nullcmd, 0, sizeof(nullcmd) );
759 oldcmd = &nullcmd;
760
761 MSG_Init( &buf, data, sizeof(data) );
762
763 MSG_Bitstream( &buf );
764 // write the current serverId so the server
765 // can tell if this is from the current gameState
766 MSG_WriteLong( &buf, cl.serverId );
767
768 // write the last message we received, which can
769 // be used for delta compression, and is also used
770 // to tell if we dropped a gamestate
771 MSG_WriteLong( &buf, clc.serverMessageSequence );
772
773 // write the last reliable message we received
774 MSG_WriteLong( &buf, clc.serverCommandSequence );
775
776 // write any unacknowledged clientCommands
777 for ( i = clc.reliableAcknowledge + 1 ; i <= clc.reliableSequence ; i++ ) {
778 MSG_WriteByte( &buf, clc_clientCommand );
779 MSG_WriteLong( &buf, i );
780 MSG_WriteString( &buf, clc.reliableCommands[ i & (MAX_RELIABLE_COMMANDS-1) ] );
781 }
782
783 // we want to send all the usercmds that were generated in the last
784 // few packet, so even if a couple packets are dropped in a row,
785 // all the cmds will make it to the server
786 if ( cl_packetdup->integer < 0 ) {
787 Cvar_Set( "cl_packetdup", "0" );
788 } else if ( cl_packetdup->integer > 5 ) {
789 Cvar_Set( "cl_packetdup", "5" );
790 }
791 oldPacketNum = (clc.netchan.outgoingSequence - 1 - cl_packetdup->integer) & PACKET_MASK;
792 count = cl.cmdNumber - cl.outPackets[ oldPacketNum ].p_cmdNumber;
793 if ( count > MAX_PACKET_USERCMDS ) {
794 count = MAX_PACKET_USERCMDS;
795 Com_Printf("MAX_PACKET_USERCMDS\n");
796 }
797
798#ifdef USE_VOIP
799 if (clc.voipOutgoingDataSize > 0)
800 {
801 if((clc.voipFlags & VOIP_SPATIAL) || Com_IsVoipTarget(clc.voipTargets, sizeof(clc.voipTargets), -1))
802 {
803 MSG_WriteByte (&buf, clc_voipOpus);
804 MSG_WriteByte (&buf, clc.voipOutgoingGeneration);
805 MSG_WriteLong (&buf, clc.voipOutgoingSequence);
806 MSG_WriteByte (&buf, clc.voipOutgoingDataFrames);
807 MSG_WriteData (&buf, clc.voipTargets, sizeof(clc.voipTargets));
808 MSG_WriteByte(&buf, clc.voipFlags);
809 MSG_WriteShort (&buf, clc.voipOutgoingDataSize);
810 MSG_WriteData (&buf, clc.voipOutgoingData, clc.voipOutgoingDataSize);
811
812 // If we're recording a demo, we have to fake a server packet with
813 // this VoIP data so it gets to disk; the server doesn't send it
814 // back to us, and we might as well eliminate concerns about dropped
815 // and misordered packets here.
816 if(clc.demorecording && !clc.demowaiting)
817 {
818 const int voipSize = clc.voipOutgoingDataSize;
819 msg_t fakemsg;
820 byte fakedata[MAX_MSGLEN];
821 MSG_Init (&fakemsg, fakedata, sizeof (fakedata));
822 MSG_Bitstream (&fakemsg);
823 MSG_WriteLong (&fakemsg, clc.reliableAcknowledge);
824 MSG_WriteByte (&fakemsg, svc_voipOpus);
825 MSG_WriteShort (&fakemsg, clc.clientNum);
826 MSG_WriteByte (&fakemsg, clc.voipOutgoingGeneration);
827 MSG_WriteLong (&fakemsg, clc.voipOutgoingSequence);
828 MSG_WriteByte (&fakemsg, clc.voipOutgoingDataFrames);
829 MSG_WriteShort (&fakemsg, clc.voipOutgoingDataSize );
830 MSG_WriteBits (&fakemsg, clc.voipFlags, VOIP_FLAGCNT);
831 MSG_WriteData (&fakemsg, clc.voipOutgoingData, voipSize);
832 MSG_WriteByte (&fakemsg, svc_EOF);
833 CL_WriteDemoMessage (&fakemsg, 0);
834 }
835
836 clc.voipOutgoingSequence += clc.voipOutgoingDataFrames;
837 clc.voipOutgoingDataSize = 0;
838 clc.voipOutgoingDataFrames = 0;
839 }
840 else
841 {
842 // We have data, but no targets. Silently discard all data
843 clc.voipOutgoingDataSize = 0;
844 clc.voipOutgoingDataFrames = 0;
845 }
846 }
847#endif
848
849 if ( count >= 1 ) {
850 if ( cl_showSend->integer ) {
851 Com_Printf( "(%i)", count );
852 }
853
854 // begin a client move command
855 if ( cl_nodelta->integer || !cl.snap.valid || clc.demowaiting
856 || clc.serverMessageSequence != cl.snap.messageNum ) {
857 MSG_WriteByte (&buf, clc_moveNoDelta);
858 } else {
859 MSG_WriteByte (&buf, clc_move);
860 }
861
862 // write the command count
863 MSG_WriteByte( &buf, count );
864
865 // use the checksum feed in the key
866 key = clc.checksumFeed;
867 // also use the message acknowledge
868 key ^= clc.serverMessageSequence;
869 // also use the last acknowledged server command in the key
870 key ^= MSG_HashKey(clc.serverCommands[ clc.serverCommandSequence & (MAX_RELIABLE_COMMANDS-1) ], 32);
871
872 // write all the commands, including the predicted command
873 for ( i = 0 ; i < count ; i++ ) {
874 j = (cl.cmdNumber - count + i + 1) & CMD_MASK;
875 cmd = &cl.cmds[j];
876 MSG_WriteDeltaUsercmdKey (&buf, key, oldcmd, cmd);
877 oldcmd = cmd;
878 }
879 }
880
881 //
882 // deliver the message
883 //
884 packetNum = clc.netchan.outgoingSequence & PACKET_MASK;
885 cl.outPackets[ packetNum ].p_realtime = cls.realtime;
886 cl.outPackets[ packetNum ].p_serverTime = oldcmd->serverTime;
887 cl.outPackets[ packetNum ].p_cmdNumber = cl.cmdNumber;
888 clc.lastPacketSentTime = cls.realtime;
889
890 if ( cl_showSend->integer ) {
891 Com_Printf( "%i ", buf.cursize );
892 }
893
894 CL_Netchan_Transmit (&clc.netchan, &buf);
895}

897/*
898=================
899CL_SendCmd
900
901Called every frame to builds and sends a command packet to the server.
902=================
903*/
904void CL_SendCmd( void ) {
905 // don't send any message if not connected
906 if ( clc.state < CA_CONNECTED ) {
907 return;
908 }
909
910 // don't send commands if paused
911 if ( com_sv_running->integer && sv_paused->integer && cl_paused->integer ) {
912 return;
913 }
914
915 // we create commands even if a demo is playing,
916 CL_CreateNewCommands();
917
918 // don't send a packet if the last packet was sent too recently
919 if ( !CL_ReadyToSendPacket() ) {
920 if ( cl_showSend->integer ) {
921 Com_Printf( ". " );
922 }
923 return;
924 }
925
926 CL_WritePacket();
927}

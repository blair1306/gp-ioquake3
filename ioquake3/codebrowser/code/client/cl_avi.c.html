77/*
78===============
79SafeFS_Write
80===============
81*/
82static ID_INLINE void SafeFS_Write( const void *buffer, int len, fileHandle_t f )
83{
84 if( FS_Write( buffer, len, f ) < len )
85 Com_Error( ERR_DROP, "Failed to write avi file" );
86}

88/*
89===============
90WRITE_STRING
91===============
92*/
93static ID_INLINE void WRITE_STRING( const char *s )
94{
95 Com_Memcpy( &buffer[ bufIndex ], s, strlen( s ) );
96 bufIndex += strlen( s );
97}
98
99/*
100===============
101WRITE_4BYTES
102===============
103*/
104static ID_INLINE void WRITE_4BYTES( int x )
105{
106 buffer[ bufIndex + 0 ] = (byte)( ( x >> 0 ) & 0xFF );
107 buffer[ bufIndex + 1 ] = (byte)( ( x >> 8 ) & 0xFF );
108 buffer[ bufIndex + 2 ] = (byte)( ( x >> 16 ) & 0xFF );
109 buffer[ bufIndex + 3 ] = (byte)( ( x >> 24 ) & 0xFF );
110 bufIndex += 4;
111}
112
113/*
114===============
115WRITE_2BYTES
116===============
117*/
118static ID_INLINE void WRITE_2BYTES( int x )
119{
120 buffer[ bufIndex + 0 ] = (byte)( ( x >> 0 ) & 0xFF );
121 buffer[ bufIndex + 1 ] = (byte)( ( x >> 8 ) & 0xFF );
122 bufIndex += 2;
123}
124
125/*
126===============
127START_CHUNK
128===============
129*/
130static ID_INLINE void START_CHUNK( const char *s )
131{
132 if( afd.chunkStackTop == MAX_RIFF_CHUNKS )
133 {
134 Com_Error( ERR_DROP, "ERROR: Top of chunkstack breached" );
135 }
136
137 afd.chunkStack[ afd.chunkStackTop ] = bufIndex;
138 afd.chunkStackTop++;
139 WRITE_STRING( s );
140 WRITE_4BYTES( 0 );
141}
142
143/*
144===============
145END_CHUNK
146===============
147*/
148static ID_INLINE void END_CHUNK( void )
149{
150 int endIndex = bufIndex;
151
152 if( afd.chunkStackTop <= 0 )
153 {
154 Com_Error( ERR_DROP, "ERROR: Bottom of chunkstack breached" );
155 }
156
157 afd.chunkStackTop--;
158 bufIndex = afd.chunkStack[ afd.chunkStackTop ];
159 bufIndex += 4;
160 WRITE_4BYTES( endIndex - bufIndex - 4 );
161 bufIndex = endIndex;
162 bufIndex = PAD( bufIndex, 2 );
163}

165/*
166===============
167CL_WriteAVIHeader
168===============
169*/
170void CL_WriteAVIHeader( void )
171{
172 bufIndex = 0;
173 afd.chunkStackTop = 0;
174
175 START_CHUNK( "RIFF" );
176 {
177 WRITE_STRING( "AVI " );
178 {
179 START_CHUNK( "LIST" );
180 {
181 WRITE_STRING( "hdrl" );
182 WRITE_STRING( "avih" );
183 WRITE_4BYTES( 56 ); //"avih" "chunk" size
184 WRITE_4BYTES( afd.framePeriod ); //dwMicroSecPerFrame
185 WRITE_4BYTES( afd.maxRecordSize *
186 afd.frameRate ); //dwMaxBytesPerSec
187 WRITE_4BYTES( 0 ); //dwReserved1
188 WRITE_4BYTES( 0x110 ); //dwFlags bits HAS_INDEX and IS_INTERLEAVED
189 WRITE_4BYTES( afd.numVideoFrames ); //dwTotalFrames
190 WRITE_4BYTES( 0 ); //dwInitialFrame
191
192 if( afd.audio ) //dwStreams
193 WRITE_4BYTES( 2 );
194 else
195 WRITE_4BYTES( 1 );
196
197 WRITE_4BYTES( afd.maxRecordSize ); //dwSuggestedBufferSize
198 WRITE_4BYTES( afd.width ); //dwWidth
199 WRITE_4BYTES( afd.height ); //dwHeight
200 WRITE_4BYTES( 0 ); //dwReserved[ 0 ]
201 WRITE_4BYTES( 0 ); //dwReserved[ 1 ]
202 WRITE_4BYTES( 0 ); //dwReserved[ 2 ]
203 WRITE_4BYTES( 0 ); //dwReserved[ 3 ]
204
205 START_CHUNK( "LIST" );
206 {
207 WRITE_STRING( "strl" );
208 WRITE_STRING( "strh" );
209 WRITE_4BYTES( 56 ); //"strh" "chunk" size
210 WRITE_STRING( "vids" );
211
212 if( afd.motionJpeg )
213 WRITE_STRING( "MJPG" );
214 else
215 WRITE_4BYTES( 0 ); // BI_RGB
216
217 WRITE_4BYTES( 0 ); //dwFlags
218 WRITE_4BYTES( 0 ); //dwPriority
219 WRITE_4BYTES( 0 ); //dwInitialFrame
220
221 WRITE_4BYTES( 1 ); //dwTimescale
222 WRITE_4BYTES( afd.frameRate ); //dwDataRate
223 WRITE_4BYTES( 0 ); //dwStartTime
224 WRITE_4BYTES( afd.numVideoFrames ); //dwDataLength
225
226 WRITE_4BYTES( afd.maxRecordSize ); //dwSuggestedBufferSize
227 WRITE_4BYTES( -1 ); //dwQuality
228 WRITE_4BYTES( 0 ); //dwSampleSize
229 WRITE_2BYTES( 0 ); //rcFrame
230 WRITE_2BYTES( 0 ); //rcFrame
231 WRITE_2BYTES( afd.width ); //rcFrame
232 WRITE_2BYTES( afd.height ); //rcFrame
233
234 WRITE_STRING( "strf" );
235 WRITE_4BYTES( 40 ); //"strf" "chunk" size
236 WRITE_4BYTES( 40 ); //biSize
237 WRITE_4BYTES( afd.width ); //biWidth
238 WRITE_4BYTES( afd.height ); //biHeight
239 WRITE_2BYTES( 1 ); //biPlanes
240 WRITE_2BYTES( 24 ); //biBitCount
241
242 if( afd.motionJpeg ) //biCompression
243 {
244 WRITE_STRING( "MJPG" );
245 WRITE_4BYTES( afd.width *
246 afd.height ); //biSizeImage
247 }
248 else
249 {
250 WRITE_4BYTES( 0 ); // BI_RGB
251 WRITE_4BYTES( afd.width *
252 afd.height * 3 ); //biSizeImage
253 }
254
255 WRITE_4BYTES( 0 ); //biXPelsPetMeter
256 WRITE_4BYTES( 0 ); //biYPelsPetMeter
257 WRITE_4BYTES( 0 ); //biClrUsed
258 WRITE_4BYTES( 0 ); //biClrImportant
259 }
260 END_CHUNK( );
261
262 if( afd.audio )
263 {
264 START_CHUNK( "LIST" );
265 {
266 WRITE_STRING( "strl" );
267 WRITE_STRING( "strh" );
268 WRITE_4BYTES( 56 ); //"strh" "chunk" size
269 WRITE_STRING( "auds" );
270 WRITE_4BYTES( 0 ); //FCC
271 WRITE_4BYTES( 0 ); //dwFlags
272 WRITE_4BYTES( 0 ); //dwPriority
273 WRITE_4BYTES( 0 ); //dwInitialFrame
274
275 WRITE_4BYTES( afd.a.sampleSize ); //dwTimescale
276 WRITE_4BYTES( afd.a.sampleSize *
277 afd.a.rate ); //dwDataRate
278 WRITE_4BYTES( 0 ); //dwStartTime
279 WRITE_4BYTES( afd.a.totalBytes /
280 afd.a.sampleSize ); //dwDataLength
281
282 WRITE_4BYTES( 0 ); //dwSuggestedBufferSize
283 WRITE_4BYTES( -1 ); //dwQuality
284 WRITE_4BYTES( afd.a.sampleSize ); //dwSampleSize
285 WRITE_2BYTES( 0 ); //rcFrame
286 WRITE_2BYTES( 0 ); //rcFrame
287 WRITE_2BYTES( 0 ); //rcFrame
288 WRITE_2BYTES( 0 ); //rcFrame
289
290 WRITE_STRING( "strf" );
291 WRITE_4BYTES( 18 ); //"strf" "chunk" size
292 WRITE_2BYTES( afd.a.format ); //wFormatTag
293 WRITE_2BYTES( afd.a.channels ); //nChannels
294 WRITE_4BYTES( afd.a.rate ); //nSamplesPerSec
295 WRITE_4BYTES( afd.a.sampleSize *
296 afd.a.rate ); //nAvgBytesPerSec
297 WRITE_2BYTES( afd.a.sampleSize ); //nBlockAlign
298 WRITE_2BYTES( afd.a.bits ); //wBitsPerSample
299 WRITE_2BYTES( 0 ); //cbSize
300 }
301 END_CHUNK( );
302 }
303 }
304 END_CHUNK( );
305
306 afd.moviOffset = bufIndex;
307
308 START_CHUNK( "LIST" );
309 {
310 WRITE_STRING( "movi" );
311 }
312 }
313 }

316/*
317===============
318CL_OpenAVIForWriting
319
320Creates an AVI file and gets it into a state where
321writing the actual data can begin
322===============
323*/
324qboolean CL_OpenAVIForWriting( const char *fileName )
325{
326 if( afd.fileOpen )
327 return qfalse;
328
329 Com_Memset( &afd, 0, sizeof( aviFileData_t ) );
330
331 // Don't start if a framerate has not been chosen
332 if( cl_aviFrameRate->integer <= 0 )
333 {
334 Com_Printf( S_COLOR_RED "cl_aviFrameRate must be >= 1\n" );
335 return qfalse;
336 }
337
338 if( ( afd.f = FS_FOpenFileWrite( fileName ) ) <= 0 )
339 return qfalse;
340
341 if( ( afd.idxF = FS_FOpenFileWrite(
342 va( "%s" INDEX_FILE_EXTENSION, fileName ) ) ) <= 0 )
343 {
344 FS_FCloseFile( afd.f );
345 return qfalse;
346 }
347
348 Q_strncpyz( afd.fileName, fileName, MAX_QPATH );
349
350 afd.frameRate = cl_aviFrameRate->integer;
351 afd.framePeriod = (int)( 1000000.0f / afd.frameRate );
352 afd.width = cls.glconfig.vidWidth;
353 afd.height = cls.glconfig.vidHeight;
354
355 if( cl_aviMotionJpeg->integer )
356 afd.motionJpeg = qtrue;
357 else
358 afd.motionJpeg = qfalse;
359
360 // Buffers only need to store RGB pixels.
361 // Allocate a bit more space for the capture buffer to account for possible
362 // padding at the end of pixel lines, and padding for alignment
363 #define MAX_PACK_LEN 16
364 afd.cBuffer = Z_Malloc((afd.width * 3 + MAX_PACK_LEN - 1) * afd.height + MAX_PACK_LEN - 1);
365 // raw avi files have pixel lines start on 4-byte boundaries
366 afd.eBuffer = Z_Malloc(PAD(afd.width * 3, AVI_LINE_PADDING) * afd.height);
367
368 afd.a.rate = dma.speed;
369 afd.a.format = WAV_FORMAT_PCM;
370 afd.a.channels = dma.channels;
371 afd.a.bits = dma.samplebits;
372 afd.a.sampleSize = ( afd.a.bits / 8 ) * afd.a.channels;
373
374 if( afd.a.rate % afd.frameRate )
375 {
376 int suggestRate = afd.frameRate;
377
378 while( ( afd.a.rate % suggestRate ) && suggestRate >= 1 )
379 suggestRate--;
380
381 Com_Printf( S_COLOR_YELLOW "WARNING: cl_aviFrameRate is not a divisor "
382 "of the audio rate, suggest %d\n", suggestRate );
383 }
384
385 if( !Cvar_VariableIntegerValue( "s_initsound" ) )
386 {
387 afd.audio = qfalse;
388 }
389 else if( Q_stricmp( Cvar_VariableString( "s_backend" ), "OpenAL" ) )
390 {
391 if( afd.a.bits != 16 || afd.a.channels != 2 )
392 {
393 Com_Printf( S_COLOR_YELLOW "WARNING: Audio format of %d bit/%d channels not supported",
394 afd.a.bits, afd.a.channels );
395 afd.audio = qfalse;
396 }
397 else
398 afd.audio = qtrue;
399 }
400 else
401 {
402 afd.audio = qfalse;
403 Com_Printf( S_COLOR_YELLOW "WARNING: Audio capture is not supported "
404 "with OpenAL. Set s_useOpenAL to 0 for audio capture\n" );
405 }
406
407 // This doesn't write a real header, but allocates the
408 // correct amount of space at the beginning of the file
409 CL_WriteAVIHeader( );
410
411 SafeFS_Write( buffer, bufIndex, afd.f );
412 afd.fileSize = bufIndex;
413
414 bufIndex = 0;
415 START_CHUNK( "idx1" );
416 SafeFS_Write( buffer, bufIndex, afd.idxF );
417
418 afd.moviSize = 4; // For the "movi"
419 afd.fileOpen = qtrue;
420
421 return qtrue;
422}

424/*
425===============
426CL_CheckFileSize
427===============
428*/
429static qboolean CL_CheckFileSize( int bytesToAdd )
430{
431 unsigned int newFileSize;
432
433 newFileSize =
434 afd.fileSize + // Current file size
435 bytesToAdd + // What we want to add
436 ( afd.numIndices * 16 ) + // The index
437 4; // The index size
438
439 // I assume all the operating systems
440 // we target can handle a 2Gb file
441 if( newFileSize > INT_MAX )
442 {
443 // Close the current file...
444 CL_CloseAVI( );
445
446 // ...And open a new one
447 CL_OpenAVIForWriting( va( "%s_", afd.fileName ) );
448
449 return qtrue;
450 }
451
452 return qfalse;
453}

455/*
456===============
457CL_WriteAVIVideoFrame
458===============
459*/
460void CL_WriteAVIVideoFrame( const byte *imageBuffer, int size )
461{
462 int chunkOffset = afd.fileSize - afd.moviOffset - 8;
463 int chunkSize = 8 + size;
464 int paddingSize = PADLEN(size, 2);
465 byte padding[ 4 ] = { 0 };
466
467 if( !afd.fileOpen )
468 return;
469
470 // Chunk header + contents + padding
471 if( CL_CheckFileSize( 8 + size + 2 ) )
472 return;
473
474 bufIndex = 0;
475 WRITE_STRING( "00dc" );
476 WRITE_4BYTES( size );
477
478 SafeFS_Write( buffer, 8, afd.f );
479 SafeFS_Write( imageBuffer, size, afd.f );
480 SafeFS_Write( padding, paddingSize, afd.f );
481 afd.fileSize += ( chunkSize + paddingSize );
482
483 afd.numVideoFrames++;
484 afd.moviSize += ( chunkSize + paddingSize );
485
486 if( size > afd.maxRecordSize )
487 afd.maxRecordSize = size;
488
489 // Index
490 bufIndex = 0;
491 WRITE_STRING( "00dc" ); //dwIdentifier
492 WRITE_4BYTES( 0x00000010 ); //dwFlags (all frames are KeyFrames)
493 WRITE_4BYTES( chunkOffset ); //dwOffset
494 WRITE_4BYTES( size ); //dwLength
495 SafeFS_Write( buffer, 16, afd.idxF );
496
497 afd.numIndices++;
498}

500#define PCM_BUFFER_SIZE 44100
501
502/*
503===============
504CL_WriteAVIAudioFrame
505===============
506*/
507void CL_WriteAVIAudioFrame( const byte *pcmBuffer, int size )
508{
509 static byte pcmCaptureBuffer[ PCM_BUFFER_SIZE ] = { 0 };
510 static int bytesInBuffer = 0;
511
512 if( !afd.audio )
513 return;
514
515 if( !afd.fileOpen )
516 return;
517
518 // Chunk header + contents + padding
519 if( CL_CheckFileSize( 8 + bytesInBuffer + size + 2 ) )
520 return;
521
522 if( bytesInBuffer + size > PCM_BUFFER_SIZE )
523 {
524 Com_Printf( S_COLOR_YELLOW
525 "WARNING: Audio capture buffer overflow -- truncating\n" );
526 size = PCM_BUFFER_SIZE - bytesInBuffer;
527 }
528
529 Com_Memcpy( &pcmCaptureBuffer[ bytesInBuffer ], pcmBuffer, size );
530 bytesInBuffer += size;
531
532 // Only write if we have a frame's worth of audio
533 if( bytesInBuffer >= (int)ceil( (float)afd.a.rate / (float)afd.frameRate ) *
534 afd.a.sampleSize )
535 {
536 int chunkOffset = afd.fileSize - afd.moviOffset - 8;
537 int chunkSize = 8 + bytesInBuffer;
538 int paddingSize = PADLEN(bytesInBuffer, 2);
539 byte padding[ 4 ] = { 0 };
540
541 bufIndex = 0;
542 WRITE_STRING( "01wb" );
543 WRITE_4BYTES( bytesInBuffer );
544
545 SafeFS_Write( buffer, 8, afd.f );
546 SafeFS_Write( pcmCaptureBuffer, bytesInBuffer, afd.f );
547 SafeFS_Write( padding, paddingSize, afd.f );
548 afd.fileSize += ( chunkSize + paddingSize );
549
550 afd.numAudioFrames++;
551 afd.moviSize += ( chunkSize + paddingSize );
552 afd.a.totalBytes += bytesInBuffer;
553
554 // Index
555 bufIndex = 0;
556 WRITE_STRING( "01wb" ); //dwIdentifier
557 WRITE_4BYTES( 0 ); //dwFlags
558 WRITE_4BYTES( chunkOffset ); //dwOffset
559 WRITE_4BYTES( bytesInBuffer ); //dwLength
560 SafeFS_Write( buffer, 16, afd.idxF );
561
562 afd.numIndices++;
563
564 bytesInBuffer = 0;
565 }

583/*
584===============
585CL_CloseAVI
586
587Closes the AVI file and writes an index chunk
588===============
589*/
590qboolean CL_CloseAVI( void )
591{
592 int indexRemainder;
593 int indexSize = afd.numIndices * 16;
594 const char *idxFileName = va( "%s" INDEX_FILE_EXTENSION, afd.fileName );
595
596 // AVI file isn't open
597 if( !afd.fileOpen )
598 return qfalse;
599
600 afd.fileOpen = qfalse;
601
602 FS_Seek( afd.idxF, 4, FS_SEEK_SET );
603 bufIndex = 0;
604 WRITE_4BYTES( indexSize );
605 SafeFS_Write( buffer, bufIndex, afd.idxF );
606 FS_FCloseFile( afd.idxF );
607
608 // Write index
609
610 // Open the temp index file
611 if( ( indexSize = FS_FOpenFileRead( idxFileName,
612 &afd.idxF, qtrue ) ) <= 0 )
613 {
614 FS_FCloseFile( afd.f );
615 return qfalse;
616 }
617
618 indexRemainder = indexSize;
619
620 // Append index to end of avi file
621 while( indexRemainder > MAX_AVI_BUFFER )
622 {
623 FS_Read( buffer, MAX_AVI_BUFFER, afd.idxF );
624 SafeFS_Write( buffer, MAX_AVI_BUFFER, afd.f );
625 afd.fileSize += MAX_AVI_BUFFER;
626 indexRemainder -= MAX_AVI_BUFFER;
627 }
628 FS_Read( buffer, indexRemainder, afd.idxF );
629 SafeFS_Write( buffer, indexRemainder, afd.f );
630 afd.fileSize += indexRemainder;
631 FS_FCloseFile( afd.idxF );
632
633 // Remove temp index file
634 FS_HomeRemove( idxFileName );
635
636 // Write the real header
637 FS_Seek( afd.f, 0, FS_SEEK_SET );
638 CL_WriteAVIHeader( );
639
640 bufIndex = 4;
641 WRITE_4BYTES( afd.fileSize - 8 ); // "RIFF" size
642
643 bufIndex = afd.moviOffset + 4; // Skip "LIST"
644 WRITE_4BYTES( afd.moviSize );
645
646 SafeFS_Write( buffer, bufIndex, afd.f );
647
648 Z_Free( afd.cBuffer );
649 Z_Free( afd.eBuffer );
650 FS_FCloseFile( afd.f );
651
652 Com_Printf( "Wrote %d:%d frames to %s\n", afd.numVideoFrames, afd.numAudioFrames, afd.fileName );
653
654 return qtrue;
655}

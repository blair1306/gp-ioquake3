45/*
46====================
47LAN_LoadCachedServers
48====================
49*/
50void LAN_LoadCachedServers( void ) {
51 int size;
52 fileHandle_t fileIn;
53 cls.numglobalservers = cls.numfavoriteservers = 0;
54 cls.numGlobalServerAddresses = 0;
55 if (FS_SV_FOpenFileRead("servercache.dat", &fileIn)) {
56 FS_Read(&cls.numglobalservers, sizeof(int), fileIn);
57 FS_Read(&cls.numfavoriteservers, sizeof(int), fileIn);
58 FS_Read(&size, sizeof(int), fileIn);
59 if (size == sizeof(cls.globalServers) + sizeof(cls.favoriteServers)) {
60 FS_Read(&cls.globalServers, sizeof(cls.globalServers), fileIn);
61 FS_Read(&cls.favoriteServers, sizeof(cls.favoriteServers), fileIn);
62 } else {
63 cls.numglobalservers = cls.numfavoriteservers = 0;
64 cls.numGlobalServerAddresses = 0;
65 }
66 FS_FCloseFile(fileIn);
67 }
68}

120/*
121====================
122LAN_AddServer
123====================
124*/
125static int LAN_AddServer(int source, const char *name, const char *address) {
126 int max, *count, i;
127 netadr_t adr;
128 serverInfo_t *servers = NULL;
129 max = MAX_OTHER_SERVERS;
130 count = NULL;
131
132 switch (source) {
133 case AS_LOCAL :
134 count = &cls.numlocalservers;
135 servers = &cls.localServers[0];
136 break;
137 case AS_MPLAYER:
138 case AS_GLOBAL :
139 max = MAX_GLOBAL_SERVERS;
140 count = &cls.numglobalservers;
141 servers = &cls.globalServers[0];
142 break;
143 case AS_FAVORITES :
144 count = &cls.numfavoriteservers;
145 servers = &cls.favoriteServers[0];
146 break;
147 }
148 if (servers && *count < max) {
149 NET_StringToAdr( address, &adr, NA_UNSPEC );
150 for ( i = 0; i < *count; i++ ) {
151 if (NET_CompareAdr(servers[i].adr, adr)) {
152 break;
153 }
154 }
155 if (i >= *count) {
156 servers[*count].adr = adr;
157 Q_strncpyz(servers[*count].hostName, name, sizeof(servers[*count].hostName));
158 servers[*count].visible = qtrue;
159 (*count)++;
160 return 1;
161 }
162 return 0;
163 }
164 return -1;
165}

167/*
168====================
169LAN_RemoveServer
170====================
171*/
172static void LAN_RemoveServer(int source, const char *addr) {
173 int *count, i;
174 serverInfo_t *servers = NULL;
175 count = NULL;
176 switch (source) {
177 case AS_LOCAL :
178 count = &cls.numlocalservers;
179 servers = &cls.localServers[0];
180 break;
181 case AS_MPLAYER:
182 case AS_GLOBAL :
183 count = &cls.numglobalservers;
184 servers = &cls.globalServers[0];
185 break;
186 case AS_FAVORITES :
187 count = &cls.numfavoriteservers;
188 servers = &cls.favoriteServers[0];
189 break;
190 }
191 if (servers) {
192 netadr_t comp;
193 NET_StringToAdr( addr, &comp, NA_UNSPEC );
194 for (i = 0; i < *count; i++) {
195 if (NET_CompareAdr( comp, servers[i].adr)) {
196 int j = i;
197 while (j < *count - 1) {
198 Com_Memcpy(&servers[j], &servers[j+1], sizeof(servers[j]));
199 j++;
200 }
201 (*count)--;
202 break;
203 }
204 }
205 }
206}

369/*
370====================
371LAN_CompareServers
372====================
373*/
374static int LAN_CompareServers( int source, int sortKey, int sortDir, int s1, int s2 ) {
375 int res;
376 serverInfo_t *server1, *server2;
377 int clients1, clients2;
378
379 server1 = LAN_GetServerPtr(source, s1);
380 server2 = LAN_GetServerPtr(source, s2);
381 if (!server1 || !server2) {
382 return 0;
383 }
384
385 res = 0;
386 switch( sortKey ) {
387 case SORT_HOST:
388 res = Q_stricmp( server1->hostName, server2->hostName );
389 break;
390
391 case SORT_MAP:
392 res = Q_stricmp( server1->mapName, server2->mapName );
393 break;
394 case SORT_CLIENTS:
395 // sub sort by max clients
396 if ( server1->clients == server2->clients ) {
397 clients1 = server1->maxClients;
398 clients2 = server2->maxClients;
399 } else {
400 clients1 = server1->clients;
401 clients2 = server2->clients;
402 }
403
404 if (clients1 < clients2) {
405 res = -1;
406 }
407 else if (clients1 > clients2) {
408 res = 1;
409 }
410 else {
411 res = 0;
412 }
413 break;
414 case SORT_GAME:
415 if (server1->gameType < server2->gameType) {
416 res = -1;
417 }
418 else if (server1->gameType > server2->gameType) {
419 res = 1;
420 }
421 else {
422 res = 0;
423 }
424 break;
425 case SORT_PING:
426 if (server1->ping < server2->ping) {
427 res = -1;
428 }
429 else if (server1->ping > server2->ping) {
430 res = 1;
431 }
432 else {
433 res = 0;
434 }
435 break;
436 }
437
438 if (sortDir) {
439 if (res < 0)
440 return 1;
441 if (res > 0)
442 return -1;
443 return 0;
444 }
445 return res;
446}

1099/*
1100====================
1101CL_InitUI
1102====================
1103*/
1104#define UI_OLD_API_VERSION 4
1105
1106void CL_InitUI( void ) {
1107 int v;
1108 vmInterpret_t interpret;
1109
1110 // load the dll or bytecode
1111 interpret = Cvar_VariableValue("vm_ui");
1112 if(cl_connectedToPureServer)
1113 {
1114 // if sv_pure is set we only allow qvms to be loaded
1115 if(interpret != VMI_COMPILED && interpret != VMI_BYTECODE)
1116 interpret = VMI_COMPILED;
1117 }
1118
1119 uivm = VM_Create( "ui", CL_UISystemCalls, interpret );
1120 if ( !uivm ) {
1121 Com_Error( ERR_FATAL, "VM_Create on UI failed" );
1122 }
1123
1124 // sanity check
1125 v = VM_Call( uivm, UI_GETAPIVERSION );
1126 if (v == UI_OLD_API_VERSION) {
1127// Com_Printf(S_COLOR_YELLOW "WARNING: loading old Quake III Arena User Interface version %d\n", v );
1128 // init for this gamestate
1129 VM_Call( uivm, UI_INIT, (clc.state >= CA_AUTHORIZING && clc.state < CA_ACTIVE));
1130 }
1131 else if (v != UI_API_VERSION) {
1132 // Free uivm now, so UI_SHUTDOWN doesn't get called later.
1133 VM_Free( uivm );
1134 uivm = NULL;
1135
1136 Com_Error( ERR_DROP, "User Interface is version %d, expected %d", v, UI_API_VERSION );
1137 cls.uiStarted = qfalse;
1138 }
1139 else {
1140 // init for this gamestate
1141 VM_Call( uivm, UI_INIT, (clc.state >= CA_AUTHORIZING && clc.state < CA_ACTIVE) );
1142 }
1143}

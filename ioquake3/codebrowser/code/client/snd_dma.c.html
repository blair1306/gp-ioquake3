201void S_ChannelSetup( void ) {
202 channel_t *p, *q;
203
204 // clear all the sounds so they don't
205 Com_Memset( s_channels, 0, sizeof( s_channels ) );
206
207 p = s_channels;;
208 q = p + MAX_CHANNELS;
209 while (--q > p) {
210 *(channel_t **)q = q-1;
211 }
212
213 *(channel_t **)q = NULL;
214 freelist = p + MAX_CHANNELS - 1;
215 Com_DPrintf("Channel memory manager started\n");
216}

224/*
225================
226return a hash value for the sfx name
227================
228*/
229static long S_HashSFXName(const char *name) {
230 int i;
231 long hash;
232 char letter;
233
234 hash = 0;
235 i = 0;
236 while (name[i] != '\0') {
237 letter = tolower(name[i]);
238 if (letter =='.') break; // don't include extension
239 if (letter =='\\') letter = '/'; // damn path names
240 hash+=(long)(letter)*(i+119);
241 i++;
242 }
243 hash &= (LOOP_HASH-1);
244 return hash;
245}
246
247/*
248==================
249S_FindName
250
251Will allocate a new sfx if it isn't found
252==================
253*/
254static sfx_t *S_FindName( const char *name ) {
255 int i;
256 int hash;
257
258 sfx_t *sfx;
259
260 if (!name) {
261 Com_Error(ERR_FATAL, "Sound name is NULL");
262 }
263
264 if (!name[0]) {
265 Com_Printf( S_COLOR_YELLOW "WARNING: Sound name is empty\n" );
266 return NULL;
267 }
268
269 if (strlen(name) >= MAX_QPATH) {
270 Com_Printf( S_COLOR_YELLOW "WARNING: Sound name is too long: %s\n", name );
271 return NULL;
272 }
273
274 if (name[0] == '*') {
275 Com_Printf( S_COLOR_YELLOW "WARNING: Tried to load player sound directly: %s\n", name );
276 return NULL;
277 }
278
279 hash = S_HashSFXName(name);
280
281 sfx = sfxHash[hash];
282 // see if already loaded
283 while (sfx) {
284 if (!Q_stricmp(sfx->soundName, name) ) {
285 return sfx;
286 }
287 sfx = sfx->next;
288 }
289
290 // find a free sfx
291 for (i=0 ; i < s_numSfx ; i++) {
292 if (!s_knownSfx[i].soundName[0]) {
293 break;
294 }
295 }
296
297 if (i == s_numSfx) {
298 if (s_numSfx == MAX_SFX) {
299 Com_Error (ERR_FATAL, "S_FindName: out of sfx_t");
300 }
301 s_numSfx++;
302 }
303
304 sfx = &s_knownSfx[i];
305 Com_Memset (sfx, 0, sizeof(*sfx));
306 strcpy (sfx->soundName, name);
307
308 sfx->next = sfxHash[hash];
309 sfxHash[hash] = sfx;
310
311 return sfx;
312}

347/*
348==================
349S_RegisterSound
350
351Creates a default buzz sound if the file can't be loaded
352==================
353*/
354sfxHandle_t S_Base_RegisterSound( const char *name, qboolean compressed ) {
355 sfx_t *sfx;
356
357 compressed = qfalse;
358 if (!s_soundStarted) {
359 return 0;
360 }
361
362 sfx = S_FindName( name );
363 if ( !sfx ) {
364 return 0;
365 }
366
367 if ( sfx->soundData ) {
368 if ( sfx->defaultSound ) {
369 Com_Printf( S_COLOR_YELLOW "WARNING: could not find %s - using default\n", sfx->soundName );
370 return 0;
371 }
372 return sfx - s_knownSfx;
373 }
374
375 sfx->inMemory = qfalse;
376 sfx->soundCompressed = compressed;
377
378 S_memoryLoad(sfx);
379
380 if ( sfx->defaultSound ) {
381 Com_Printf( S_COLOR_YELLOW "WARNING: could not find %s - using default\n", sfx->soundName );
382 return 0;
383 }
384
385 return sfx - s_knownSfx;
386}

418/*
419=================
420S_SpatializeOrigin
421
422Used for spatializing s_channels
423=================
424*/
425void S_SpatializeOrigin (vec3_t origin, int master_vol, int *left_vol, int *right_vol)
426{
427 vec_t dot;
428 vec_t dist;
429 vec_t lscale, rscale, scale;
430 vec3_t source_vec;
431 vec3_t vec;
432
433 const float dist_mult = SOUND_ATTENUATE;
434
435 // calculate stereo seperation and distance attenuation
436 VectorSubtract(origin, listener_origin, source_vec);
437
438 dist = VectorNormalize(source_vec);
439 dist -= SOUND_FULLVOLUME;
440 if (dist < 0)
441 dist = 0; // close enough to be at full volume
442 dist *= dist_mult; // different attenuation levels
443
444 VectorRotate( source_vec, listener_axis, vec );
445
446 dot = -vec[1];
447
448 if (dma.channels == 1)
449 { // no attenuation = no spatialization
450 rscale = 1.0;
451 lscale = 1.0;
452 }
453 else
454 {
455 rscale = 0.5 * (1.0 + dot);
456 lscale = 0.5 * (1.0 - dot);
457 if ( rscale < 0 ) {
458 rscale = 0;
459 }
460 if ( lscale < 0 ) {
461 lscale = 0;
462 }
463 }
464
465 // add in distance effect
466 scale = (1.0 - dist) * rscale;
467 *right_vol = (master_vol * scale);
468 if (*right_vol < 0)
469 *right_vol = 0;
470
471 scale = (1.0 - dist) * lscale;
472 *left_vol = (master_vol * scale);
473 if (*left_vol < 0)
474 *left_vol = 0;
475}

481/*
482=================
483S_Base_HearingThroughEntity
484
485Also see S_AL_HearingThroughEntity
486=================
487*/
488static qboolean S_Base_HearingThroughEntity( int entityNum, vec3_t origin )
489{
490 float distanceSq;
491 vec3_t sorigin;
492
493 if (origin)
494 VectorCopy(origin, sorigin);
495 else
496 VectorCopy(loopSounds[entityNum].origin, sorigin);
497
498 if( listener_number == entityNum )
499 {
500 // This is an outrageous hack to detect
501 // whether or not the player is rendering in third person or not. We can't
502 // ask the renderer because the renderer has no notion of entities and we
503 // can't ask cgame since that would involve changing the API and hence mod
504 // compatibility. I don't think there is any way around this, but I'll leave
505 // the FIXME just in case anyone has a bright idea.
506 distanceSq = DistanceSquared(
507 sorigin,
508 listener_origin );
509
510 if( distanceSq > THIRD_PERSON_THRESHOLD_SQ )
511 return qfalse; //we're the player, but third person
512 else
513 return qtrue; //we're the player
514 }
515 else
516 return qfalse; //not the player
517}

519/*
520====================
521S_Base_StartSoundEx
522
523Validates the parms and ques the sound up
524if origin is NULL, the sound will be dynamically sourced from the entity
525Entchannel 0 will never override a playing sound
526====================
527*/
528static void S_Base_StartSoundEx( vec3_t origin, int entityNum, int entchannel, sfxHandle_t sfxHandle, qboolean localSound ) {
529 channel_t *ch;
530 sfx_t *sfx;
531 int i, oldest, chosen, time;
532 int inplay, allowed;
533 qboolean fullVolume;
534
535 if ( !s_soundStarted || s_soundMuted ) {
536 return;
537 }
538
539 if ( !origin && ( entityNum < 0 || entityNum >= MAX_GENTITIES ) ) {
540 Com_Error( ERR_DROP, "S_StartSound: bad entitynum %i", entityNum );
541 }
542
543 if ( sfxHandle < 0 || sfxHandle >= s_numSfx ) {
544 Com_Printf( S_COLOR_YELLOW "S_StartSound: handle %i out of range\n", sfxHandle );
545 return;
546 }
547
548 sfx = &s_knownSfx[ sfxHandle ];
549
550 if (sfx->inMemory == qfalse) {
551 S_memoryLoad(sfx);
552 }
553
554 if ( s_show->integer == 1 ) {
555 Com_Printf( "%i : %s\n", s_paintedtime, sfx->soundName );
556 }
557
558 time = Com_Milliseconds();
559
560// Com_Printf("playing %s\n", sfx->soundName);
561 // pick a channel to play on
562
563 allowed = 4;
564 if (entityNum == listener_number) {
565 allowed = 8;
566 }
567
568 fullVolume = qfalse;
569 if (localSound || S_Base_HearingThroughEntity(entityNum, origin)) {
570 fullVolume = qtrue;
571 }
572
573 ch = s_channels;
574 inplay = 0;
575 for ( i = 0; i < MAX_CHANNELS ; i++, ch++ ) {
576 if (ch->entnum == entityNum && ch->thesfx == sfx) {
577 if (time - ch->allocTime < 50) {
578// if (Cvar_VariableValue( "cg_showmiss" )) {
579// Com_Printf("double sound start\n");
580// }
581 return;
582 }
583 inplay++;
584 }
585 }
586
587 if (inplay>allowed) {
588 return;
589 }
590
591 sfx->lastTimeUsed = time;
592
593 ch = S_ChannelMalloc(); // entityNum, entchannel);
594 if (!ch) {
595 ch = s_channels;
596
597 oldest = sfx->lastTimeUsed;
598 chosen = -1;
599 for ( i = 0 ; i < MAX_CHANNELS ; i++, ch++ ) {
600 if (ch->entnum != listener_number && ch->entnum == entityNum && ch->allocTime<oldest && ch->entchannel != CHAN_ANNOUNCER) {
601 oldest = ch->allocTime;
602 chosen = i;
603 }
604 }
605 if (chosen == -1) {
606 ch = s_channels;
607 for ( i = 0 ; i < MAX_CHANNELS ; i++, ch++ ) {
608 if (ch->entnum != listener_number && ch->allocTime<oldest && ch->entchannel != CHAN_ANNOUNCER) {
609 oldest = ch->allocTime;
610 chosen = i;
611 }
612 }
613 if (chosen == -1) {
614 ch = s_channels;
615 if (ch->entnum == listener_number) {
616 for ( i = 0 ; i < MAX_CHANNELS ; i++, ch++ ) {
617 if (ch->allocTime<oldest) {
618 oldest = ch->allocTime;
619 chosen = i;
620 }
621 }
622 }
623 if (chosen == -1) {
624 Com_Printf("dropping sound\n");
625 return;
626 }
627 }
628 }
629 ch = &s_channels[chosen];
630 ch->allocTime = sfx->lastTimeUsed;
631 }
632
633 if (origin) {
634 VectorCopy (origin, ch->origin);
635 ch->fixed_origin = qtrue;
636 } else {
637 ch->fixed_origin = qfalse;
638 }
639
640 ch->master_vol = 127;
641 ch->entnum = entityNum;
642 ch->thesfx = sfx;
643 ch->startSample = START_SAMPLE_IMMEDIATE;
644 ch->entchannel = entchannel;
645 ch->leftvol = ch->master_vol; // these will get calced at next spatialize
646 ch->rightvol = ch->master_vol; // unless the game isn't running
647 ch->doppler = qfalse;
648 ch->fullVolume = fullVolume;
649}

681/*
682==================
683S_ClearSoundBuffer
684
685If we are about to perform file access, clear the buffer
686so sound doesn't stutter.
687==================
688*/
689void S_Base_ClearSoundBuffer( void ) {
690 int clear;
691
692 if (!s_soundStarted)
693 return;
694
695 // stop looping sounds
696 Com_Memset(loopSounds, 0, MAX_GENTITIES*sizeof(loopSound_t));
697 Com_Memset(loop_channels, 0, MAX_CHANNELS*sizeof(channel_t));
698 numLoopChannels = 0;
699
700 S_ChannelSetup();
701
702 Com_Memset(s_rawend, '\0', sizeof (s_rawend));
703
704 if (dma.samplebits == 8)
705 clear = 0x80;
706 else
707 clear = 0;
708
709 SNDDMA_BeginPainting ();
710 if (dma.buffer)
711 Com_Memset(dma.buffer, clear, dma.samples * dma.samplebits/8);
712 SNDDMA_Submit ();
713}

761/*
762==================
763S_AddLoopingSound
764
765Called during entity generation for a frame
766Include velocity in case I get around to doing doppler...
767==================
768*/
769void S_Base_AddLoopingSound( int entityNum, const vec3_t origin, const vec3_t velocity, sfxHandle_t sfxHandle ) {
770 sfx_t *sfx;
771
772 if ( !s_soundStarted || s_soundMuted ) {
773 return;
774 }
775
776 if ( sfxHandle < 0 || sfxHandle >= s_numSfx ) {
777 Com_Printf( S_COLOR_YELLOW "S_AddLoopingSound: handle %i out of range\n", sfxHandle );
778 return;
779 }
780
781 sfx = &s_knownSfx[ sfxHandle ];
782
783 if (sfx->inMemory == qfalse) {
784 S_memoryLoad(sfx);
785 }
786
787 if ( !sfx->soundLength ) {
788 Com_Error( ERR_DROP, "%s has length 0", sfx->soundName );
789 }
790
791 VectorCopy( origin, loopSounds[entityNum].origin );
792 VectorCopy( velocity, loopSounds[entityNum].velocity );
793 loopSounds[entityNum].active = qtrue;
794 loopSounds[entityNum].kill = qtrue;
795 loopSounds[entityNum].doppler = qfalse;
796 loopSounds[entityNum].oldDopplerScale = 1.0;
797 loopSounds[entityNum].dopplerScale = 1.0;
798 loopSounds[entityNum].sfx = sfx;
799
800 if (s_doppler->integer && VectorLengthSquared(velocity)>0.0) {
801 vec3_t out;
802 float lena, lenb;
803
804 loopSounds[entityNum].doppler = qtrue;
805 lena = DistanceSquared(loopSounds[listener_number].origin, loopSounds[entityNum].origin);
806 VectorAdd(loopSounds[entityNum].origin, loopSounds[entityNum].velocity, out);
807 lenb = DistanceSquared(loopSounds[listener_number].origin, out);
808 if ((loopSounds[entityNum].framenum+1) != cls.framecount) {
809 loopSounds[entityNum].oldDopplerScale = 1.0;
810 } else {
811 loopSounds[entityNum].oldDopplerScale = loopSounds[entityNum].dopplerScale;
812 }
813 loopSounds[entityNum].dopplerScale = lenb/(lena*100);
814 if (loopSounds[entityNum].dopplerScale<=1.0) {
815 loopSounds[entityNum].doppler = qfalse; // don't bother doing the math
816 } else if (loopSounds[entityNum].dopplerScale>MAX_DOPPLER_SCALE) {
817 loopSounds[entityNum].dopplerScale = MAX_DOPPLER_SCALE;
818 }
819 }
820
821 loopSounds[entityNum].framenum = cls.framecount;
822}

832void S_Base_AddRealLoopingSound( int entityNum, const vec3_t origin, const vec3_t velocity, sfxHandle_t sfxHandle ) {

838
839 if ( sfxHandle < 0 || sfxHandle >= s_numSfx ) {
840 Com_Printf( S_COLOR_YELLOW "S_AddRealLoopingSound: handle %i out of range\n", sfxHandle );
841 return;
842 }
843
844 sfx = &s_knownSfx[ sfxHandle ];
845
846 if (sfx->inMemory == qfalse) {
847 S_memoryLoad(sfx);
848 }
849
850 if ( !sfx->soundLength ) {
851 Com_Error( ERR_DROP, "%s has length 0", sfx->soundName );
852 }
853 VectorCopy( origin, loopSounds[entityNum].origin );
854 VectorCopy( velocity, loopSounds[entityNum].velocity );
855 loopSounds[entityNum].sfx = sfx;
856 loopSounds[entityNum].active = qtrue;
857 loopSounds[entityNum].kill = qfalse;
858 loopSounds[entityNum].doppler = qfalse;
859}

863/*
864==================
865S_AddLoopSounds
866
867Spatialize all of the looping sounds.
868All sounds are on the same cycle, so any duplicates can just
869sum up the channel multipliers.
870==================
871*/
872void S_AddLoopSounds (void) {
873 int i, j, time;
874 int left_total, right_total, left, right;
875 channel_t *ch;
876 loopSound_t *loop, *loop2;
877 static int loopFrame;
878
879
880 numLoopChannels = 0;
881
882 time = Com_Milliseconds();
883
884 loopFrame++;
885 for ( i = 0 ; i < MAX_GENTITIES ; i++) {
886 loop = &loopSounds[i];
887 if ( !loop->active || loop->mergeFrame == loopFrame ) {
888 continue; // already merged into an earlier sound
889 }
890
891 if (loop->kill) {
892 S_SpatializeOrigin( loop->origin, 127, &left_total, &right_total); // 3d
893 } else {
894 S_SpatializeOrigin( loop->origin, 90, &left_total, &right_total); // sphere
895 }
896
897 loop->sfx->lastTimeUsed = time;
898
899 for (j=(i+1); j< MAX_GENTITIES ; j++) {
900 loop2 = &loopSounds[j];
901 if ( !loop2->active || loop2->doppler || loop2->sfx != loop->sfx) {
902 continue;
903 }
904 loop2->mergeFrame = loopFrame;
905
906 if (loop2->kill) {
907 S_SpatializeOrigin( loop2->origin, 127, &left, &right); // 3d
908 } else {
909 S_SpatializeOrigin( loop2->origin, 90, &left, &right); // sphere
910 }
911
912 loop2->sfx->lastTimeUsed = time;
913 left_total += left;
914 right_total += right;
915 }
916 if (left_total == 0 && right_total == 0) {
917 continue; // not audible
918 }
919
920 // allocate a channel
921 ch = &loop_channels[numLoopChannels];
922
923 if (left_total > 255) {
924 left_total = 255;
925 }
926 if (right_total > 255) {
927 right_total = 255;
928 }
929
930 ch->master_vol = 127;
931 ch->leftvol = left_total;
932 ch->rightvol = right_total;
933 ch->thesfx = loop->sfx;
934 ch->doppler = loop->doppler;
935 ch->dopplerScale = loop->dopplerScale;
936 ch->oldDopplerScale = loop->oldDopplerScale;
937 ch->fullVolume = qfalse;
938 numLoopChannels++;
939 if (numLoopChannels == MAX_CHANNELS) {
940 return;
941 }
942 }
943}

947/*
948=================
949S_ByteSwapRawSamples
950
951If raw data has been loaded in little endien binary form, this must be done.
952If raw data was calculated, as with ADPCM, this should not be called.
953=================
954*/
955void S_ByteSwapRawSamples( int samples, int width, int s_channels, const byte *data ) {
956 int i;
957
958 if ( width != 2 ) {
959 return;
960 }
961 if ( LittleShort( 256 ) == 256 ) {
962 return;
963 }
964
965 if ( s_channels == 2 ) {
966 samples <<= 1;
967 }
968 for ( i = 0 ; i < samples ; i++ ) {
969 ((short *)data)[i] = LittleShort( ((short *)data)[i] );
970 }
971}

973/*
974============
975S_Base_RawSamples
976
977Music streaming
978============
979*/
980void S_Base_RawSamples( int stream, int samples, int rate, int width, int s_channels, const byte *data, float volume, int entityNum)
981{
982 int i;
983 int src, dst;
984 float scale;
985 int intVolumeLeft, intVolumeRight;
986 portable_samplepair_t *rawsamples;
987
988 if ( !s_soundStarted || s_soundMuted ) {
989 return;
990 }
991
992 if ( (stream < 0) || (stream >= MAX_RAW_STREAMS) ) {
993 return;
994 }
995
996 rawsamples = s_rawsamples[stream];
997
998 if ( s_muted->integer ) {
999 intVolumeLeft = intVolumeRight = 0;
1000 } else {
1001 int leftvol, rightvol;
1002
1003 if ( entityNum >= 0 && entityNum < MAX_GENTITIES ) {
1004 // support spatialized raw streams, e.g. for VoIP
1005 S_SpatializeOrigin( loopSounds[ entityNum ].origin, 256, &leftvol, &rightvol );
1006 } else {
1007 leftvol = rightvol = 256;
1008 }
1009
1010 intVolumeLeft = leftvol * volume * s_volume->value;
1011 intVolumeRight = rightvol * volume * s_volume->value;
1012 }
1013
1014 if ( s_rawend[stream] < s_soundtime ) {
1015 Com_DPrintf( "S_Base_RawSamples: resetting minimum: %i < %i\n", s_rawend[stream], s_soundtime );
1016 s_rawend[stream] = s_soundtime;
1017 }
1018
1019 scale = (float)rate / dma.speed;
1020
1021//Com_Printf ("%i < %i < %i\n", s_soundtime, s_paintedtime, s_rawend[stream]);
1022 if (s_channels == 2 && width == 2)
1023 {
1024 if (scale == 1.0)
1025 { // optimized case
1026 for (i=0 ; i<samples ; i++)
1027 {
1028 dst = s_rawend[stream]&(MAX_RAW_SAMPLES-1);
1029 s_rawend[stream]++;
1030 rawsamples[dst].left = ((short *)data)[i*2] * intVolumeLeft;
1031 rawsamples[dst].right = ((short *)data)[i*2+1] * intVolumeRight;
1032 }
1033 }
1034 else
1035 {
1036 for (i=0 ; ; i++)
1037 {
1038 src = i*scale;
1039 if (src >= samples)
1040 break;
1041 dst = s_rawend[stream]&(MAX_RAW_SAMPLES-1);
1042 s_rawend[stream]++;
1043 rawsamples[dst].left = ((short *)data)[src*2] * intVolumeLeft;
1044 rawsamples[dst].right = ((short *)data)[src*2+1] * intVolumeRight;
1045 }
1046 }
1047 }
1048 else if (s_channels == 1 && width == 2)
1049 {
1050 for (i=0 ; ; i++)
1051 {
1052 src = i*scale;
1053 if (src >= samples)
1054 break;
1055 dst = s_rawend[stream]&(MAX_RAW_SAMPLES-1);
1056 s_rawend[stream]++;
1057 rawsamples[dst].left = ((short *)data)[src] * intVolumeLeft;
1058 rawsamples[dst].right = ((short *)data)[src] * intVolumeRight;
1059 }
1060 }
1061 else if (s_channels == 2 && width == 1)
1062 {
1063 intVolumeLeft *= 256;
1064 intVolumeRight *= 256;
1065
1066 for (i=0 ; ; i++)
1067 {
1068 src = i*scale;
1069 if (src >= samples)
1070 break;
1071 dst = s_rawend[stream]&(MAX_RAW_SAMPLES-1);
1072 s_rawend[stream]++;
1073 rawsamples[dst].left = ((char *)data)[src*2] * intVolumeLeft;
1074 rawsamples[dst].right = ((char *)data)[src*2+1] * intVolumeRight;
1075 }
1076 }
1077 else if (s_channels == 1 && width == 1)
1078 {
1079 intVolumeLeft *= 256;
1080 intVolumeRight *= 256;
1081
1082 for (i=0 ; ; i++)
1083 {
1084 src = i*scale;
1085 if (src >= samples)
1086 break;
1087 dst = s_rawend[stream]&(MAX_RAW_SAMPLES-1);
1088 s_rawend[stream]++;
1089 rawsamples[dst].left = (((byte *)data)[src]-128) * intVolumeLeft;
1090 rawsamples[dst].right = (((byte *)data)[src]-128) * intVolumeRight;
1091 }
1092 }
1093
1094 if ( s_rawend[stream] > s_soundtime + MAX_RAW_SAMPLES ) {
1095 Com_DPrintf( "S_Base_RawSamples: overflowed %i > %i\n", s_rawend[stream], s_soundtime );
1096 }
1097}

1241void S_GetSoundtime(void)
1242{
1243 int samplepos;
1244 static int buffers;
1245 static int oldsamplepos;
1246 int fullsamples;
1247
1248 fullsamples = dma.samples / dma.channels;
1249
1250 if( CL_VideoRecording( ) )
1251 {
1252 float fps = MIN(cl_aviFrameRate->value, 1000.0f);
1253 float frameDuration = MAX(dma.speed / fps, 1.0f) + clc.aviSoundFrameRemainder;
1254
1255 int msec = (int)frameDuration;
1256 s_soundtime += msec;
1257 clc.aviSoundFrameRemainder = frameDuration - msec;
1258
1259 return;
1260 }
1261
1262 // it is possible to miscount buffers if it has wrapped twice between
1263 // calls to S_Update. Oh well.
1264 samplepos = SNDDMA_GetDMAPos();
1265 if (samplepos < oldsamplepos)
1266 {
1267 buffers++; // buffer wrapped
1268
1269 if (s_paintedtime > 0x40000000)
1270 { // time to chop things off to avoid 32 bit limits
1271 buffers = 0;
1272 s_paintedtime = fullsamples;
1273 S_Base_StopAllSounds ();
1274 }
1275 }
1276 oldsamplepos = samplepos;
1277
1278 s_soundtime = buffers*fullsamples + samplepos/dma.channels;
1279
1280#if 0
1281// check to make sure that we haven't overshot
1282 if (s_paintedtime < s_soundtime)
1283 {
1284 Com_DPrintf ("S_Update_ : overflow\n");
1285 s_paintedtime = s_soundtime;
1286 }
1287#endif
1288
1289 if ( dma.submission_chunk < 256 ) {
1290 s_paintedtime = s_soundtime + s_mixPreStep->value * dma.speed;
1291 } else {
1292 s_paintedtime = s_soundtime + dma.submission_chunk;
1293 }
1294}

1296
1297void S_Update_(void) {
1298 unsigned endtime;
1299 int samps;
1300 static float lastTime = 0.0f;
1301 float ma, op;
1302 float thisTime, sane;
1303 static int ot = -1;
1304
1305 if ( !s_soundStarted || s_soundMuted ) {
1306 return;
1307 }
1308
1309 thisTime = Com_Milliseconds();
1310
1311 // Updates s_soundtime
1312 S_GetSoundtime();
1313
1314 if (s_soundtime == ot) {
1315 return;
1316 }
1317 ot = s_soundtime;
1318
1319 // clear any sound effects that end before the current time,
1320 // and start any new sounds
1321 S_ScanChannelStarts();
1322
1323 sane = thisTime - lastTime;
1324 if (sane<11) {
1325 sane = 11; // 85hz
1326 }
1327
1328 ma = s_mixahead->value * dma.speed;
1329 op = s_mixPreStep->value + sane*dma.speed*0.01;
1330
1331 if (op < ma) {
1332 ma = op;
1333 }
1334
1335 // mix ahead of current position
1336 endtime = s_soundtime + ma;
1337
1338 // mix to an even submission block size
1339 endtime = (endtime + dma.submission_chunk-1)
1340 & ~(dma.submission_chunk-1);
1341
1342 // never mix more than the complete buffer
1343 samps = dma.samples >> (dma.channels-1);
1344 if (endtime - s_soundtime > samps)
1345 endtime = s_soundtime + samps;
1346
1347
1348
1349 SNDDMA_BeginPainting ();
1350
1351 S_PaintChannels (endtime);
1352
1353 SNDDMA_Submit ();
1354
1355 lastTime = thisTime;
1356}

1432/*
1433======================
1434S_UpdateBackgroundTrack
1435======================
1436*/
1437void S_UpdateBackgroundTrack( void ) {
1438 int bufferSamples;
1439 int fileSamples;
1440 byte raw[30000]; // just enough to fit in a mac stack frame
1441 int fileBytes;
1442 int r;
1443
1444 if(!s_backgroundStream) {
1445 return;
1446 }
1447
1448 // don't bother playing anything if musicvolume is 0
1449 if ( s_musicVolume->value <= 0 ) {
1450 return;
1451 }
1452
1453 // see how many samples should be copied into the raw buffer
1454 if ( s_rawend[0] < s_soundtime ) {
1455 s_rawend[0] = s_soundtime;
1456 }
1457
1458 while ( s_rawend[0] < s_soundtime + MAX_RAW_SAMPLES ) {
1459 bufferSamples = MAX_RAW_SAMPLES - (s_rawend[0] - s_soundtime);
1460
1461 // decide how much data needs to be read from the file
1462 fileSamples = bufferSamples * s_backgroundStream->info.rate / dma.speed;
1463
1464 if (!fileSamples)
1465 return;
1466
1467 // our max buffer size
1468 fileBytes = fileSamples * (s_backgroundStream->info.width * s_backgroundStream->info.channels);
1469 if ( fileBytes > sizeof(raw) ) {
1470 fileBytes = sizeof(raw);
1471 fileSamples = fileBytes / (s_backgroundStream->info.width * s_backgroundStream->info.channels);
1472 }
1473
1474 // Read
1475 r = S_CodecReadStream(s_backgroundStream, fileBytes, raw);
1476 if(r < fileBytes)
1477 {
1478 fileSamples = r / (s_backgroundStream->info.width * s_backgroundStream->info.channels);
1479 }
1480
1481 if(r > 0)
1482 {
1483 // add to raw buffer
1484 S_Base_RawSamples(0, fileSamples, s_backgroundStream->info.rate,
1485 s_backgroundStream->info.width, s_backgroundStream->info.channels, raw, s_musicVolume->value, -1);
1486 }
1487 else
1488 {
1489 // loop
1490 if(s_backgroundLoop[0])
1491 {
1492 S_OpenBackgroundStream( s_backgroundLoop );
1493 if(!s_backgroundStream)
1494 return;
1495 }
1496 else
1497 {
1498 S_Base_StopBackgroundTrack();
1499 return;
1500 }
1501 }
1502
1503 }
1504}

39/*
40=================
41S_ValidateInterface
42=================
43*/
44static qboolean S_ValidSoundInterface( soundInterface_t *si )
45{
46 if( !si->Shutdown ) return qfalse;
47 if( !si->StartSound ) return qfalse;
48 if( !si->StartLocalSound ) return qfalse;
49 if( !si->StartBackgroundTrack ) return qfalse;
50 if( !si->StopBackgroundTrack ) return qfalse;
51 if( !si->RawSamples ) return qfalse;
52 if( !si->StopAllSounds ) return qfalse;
53 if( !si->ClearLoopingSounds ) return qfalse;
54 if( !si->AddLoopingSound ) return qfalse;
55 if( !si->AddRealLoopingSound ) return qfalse;
56 if( !si->StopLoopingSound ) return qfalse;
57 if( !si->Respatialize ) return qfalse;
58 if( !si->UpdateEntityPosition ) return qfalse;
59 if( !si->Update ) return qfalse;
60 if( !si->DisableSounds ) return qfalse;
61 if( !si->BeginRegistration ) return qfalse;
62 if( !si->RegisterSound ) return qfalse;
63 if( !si->ClearSoundBuffer ) return qfalse;
64 if( !si->SoundInfo ) return qfalse;
65 if( !si->SoundList ) return qfalse;
66
67#ifdef USE_VOIP
68 if( !si->StartCapture ) return qfalse;
69 if( !si->AvailableCaptureSamples ) return qfalse;
70 if( !si->Capture ) return qfalse;
71 if( !si->StopCapture ) return qfalse;
72 if( !si->MasterGain ) return qfalse;
73#endif
74
75 return qtrue;
76}

225/*
226=================
227S_Update
228=================
229*/
230void S_Update( void )
231{
232 if(s_muted->integer)
233 {
234 if(!(s_muteWhenMinimized->integer && com_minimized->integer) &&
235 !(s_muteWhenUnfocused->integer && com_unfocused->integer))
236 {
237 s_muted->integer = qfalse;
238 s_muted->modified = qtrue;
239 }
240 }
241 else
242 {
243 if((s_muteWhenMinimized->integer && com_minimized->integer) ||
244 (s_muteWhenUnfocused->integer && com_unfocused->integer))
245 {
246 s_muted->integer = qtrue;
247 s_muted->modified = qtrue;
248 }
249 }
250
251 if( si.Update ) {
252 si.Update( );
253 }
254}

396/*
397=================
398S_Play_f
399=================
400*/
401void S_Play_f( void ) {
402 int i;
403 int c;
404 sfxHandle_t h;
405
406 if( !si.RegisterSound || !si.StartLocalSound ) {
407 return;
408 }
409
410 c = Cmd_Argc();
411
412 if( c < 2 ) {
413 Com_Printf ("Usage: play <sound filename> [sound filename] [sound filename] ...\n");
414 return;
415 }
416
417 for( i = 1; i < c; i++ ) {
418 h = si.RegisterSound( Cmd_Argv(i), qfalse );
419
420 if( h ) {
421 si.StartLocalSound( h, CHAN_LOCAL_SOUND );
422 }
423 }
424}

426/*
427=================
428S_Music_f
429=================
430*/
431void S_Music_f( void ) {
432 int c;
433
434 if( !si.StartBackgroundTrack ) {
435 return;
436 }
437
438 c = Cmd_Argc();
439
440 if ( c == 2 ) {
441 si.StartBackgroundTrack( Cmd_Argv(1), NULL );
442 } else if ( c == 3 ) {
443 si.StartBackgroundTrack( Cmd_Argv(1), Cmd_Argv(2) );
444 } else {
445 Com_Printf ("Usage: music <musicfile> [loopfile]\n");
446 return;
447 }
448
449}

467/*
468=================
469S_Init
470=================
471*/
472void S_Init( void )
473{
474 cvar_t *cv;
475 qboolean started = qfalse;
476
477 Com_Printf( "------ Initializing Sound ------\n" );
478
479 s_volume = Cvar_Get( "s_volume", "0.8", CVAR_ARCHIVE );
480 s_musicVolume = Cvar_Get( "s_musicvolume", "0.25", CVAR_ARCHIVE );
481 s_muted = Cvar_Get("s_muted", "0", CVAR_ROM);
482 s_doppler = Cvar_Get( "s_doppler", "1", CVAR_ARCHIVE );
483 s_backend = Cvar_Get( "s_backend", "", CVAR_ROM );
484 s_muteWhenMinimized = Cvar_Get( "s_muteWhenMinimized", "0", CVAR_ARCHIVE );
485 s_muteWhenUnfocused = Cvar_Get( "s_muteWhenUnfocused", "0", CVAR_ARCHIVE );
486
487 cv = Cvar_Get( "s_initsound", "1", 0 );
488 if( !cv->integer ) {
489 Com_Printf( "Sound disabled.\n" );
490 } else {
491
492 S_CodecInit( );
493
494 Cmd_AddCommand( "play", S_Play_f );
495 Cmd_AddCommand( "music", S_Music_f );
496 Cmd_AddCommand( "stopmusic", S_StopMusic_f );
497 Cmd_AddCommand( "s_list", S_SoundList );
498 Cmd_AddCommand( "s_stop", S_StopAllSounds );
499 Cmd_AddCommand( "s_info", S_SoundInfo );
500
501 cv = Cvar_Get( "s_useOpenAL", "1", CVAR_ARCHIVE );
502 if( cv->integer ) {
503 //OpenAL
504 started = S_AL_Init( &si );
505 Cvar_Set( "s_backend", "OpenAL" );
506 }
507
508 if( !started ) {
509 started = S_Base_Init( &si );
510 Cvar_Set( "s_backend", "base" );
511 }
512
513 if( started ) {
514 if( !S_ValidSoundInterface( &si ) ) {
515 Com_Error( ERR_FATAL, "Sound interface invalid" );
516 }
517
518 S_SoundInfo( );
519 Com_Printf( "Sound initialization successful.\n" );
520 } else {
521 Com_Printf( "Sound initialization failed.\n" );
522 }
523 }
524
525 Com_Printf( "--------------------------------\n");
526}

59static void *cURLLib = NULL;

61/*
62=================
63GPA
64=================
65*/
66static void *GPA(char *str)
67{
68 void *rv;
69
70 rv = Sys_LoadFunction(cURLLib, str);
71 if(!rv)
72 {
73 Com_Printf("Can't load symbol %s\n", str);
74 clc.cURLEnabled = qfalse;
75 return NULL;
76 }
77 else
78 {
79 Com_DPrintf("Loaded symbol %s (0x%p)\n", str, rv);
80 return rv;
81 }
82}
83#endif /* USE_CURL_DLOPEN */

85/*
86=================
87CL_cURL_Init
88=================
89*/
90qboolean CL_cURL_Init()
91{
92#ifdef USE_CURL_DLOPEN
93 if(cURLLib)
94 return qtrue;
95
96
97 Com_Printf("Loading \"%s\"...", cl_cURLLib->string);
98 if(!(cURLLib = Sys_LoadDll(cl_cURLLib->string, qtrue)))
99 {
100#ifdef ALTERNATE_CURL_LIB
101 // On some linux distributions there is no libcurl.so.3, but only libcurl.so.4. That one works too.
102 if(!(cURLLib = Sys_LoadDll(ALTERNATE_CURL_LIB, qtrue)))
103#endif
104 return qfalse;
105 }
106
107 clc.cURLEnabled = qtrue;
108
109 qcurl_version = GPA("curl_version");
110
111 qcurl_easy_init = GPA("curl_easy_init");
112 qcurl_easy_setopt = GPA("curl_easy_setopt");
113 qcurl_easy_perform = GPA("curl_easy_perform");
114 qcurl_easy_cleanup = GPA("curl_easy_cleanup");
115 qcurl_easy_getinfo = GPA("curl_easy_getinfo");
116 qcurl_easy_duphandle = GPA("curl_easy_duphandle");
117 qcurl_easy_reset = GPA("curl_easy_reset");
118 qcurl_easy_strerror = GPA("curl_easy_strerror");
119
120 qcurl_multi_init = GPA("curl_multi_init");
121 qcurl_multi_add_handle = GPA("curl_multi_add_handle");
122 qcurl_multi_remove_handle = GPA("curl_multi_remove_handle");
123 qcurl_multi_fdset = GPA("curl_multi_fdset");
124 qcurl_multi_perform = GPA("curl_multi_perform");
125 qcurl_multi_cleanup = GPA("curl_multi_cleanup");
126 qcurl_multi_info_read = GPA("curl_multi_info_read");
127 qcurl_multi_strerror = GPA("curl_multi_strerror");
128
129 if(!clc.cURLEnabled)
130 {
131 CL_cURL_Shutdown();
132 Com_Printf("FAIL One or more symbols not found\n");
133 return qfalse;
134 }
135 Com_Printf("OK\n");
136
137 return qtrue;
138#else
139 clc.cURLEnabled = qtrue;
140 return qtrue;
141#endif /* USE_CURL_DLOPEN */
142}

172 qcurl_multi_info_read = NULL;
173 qcurl_multi_strerror = NULL;
174#endif /* USE_CURL_DLOPEN */

177void CL_cURL_Cleanup(void)
178{
179 if(clc.downloadCURLM) {
180 CURLMcode result;
181
182 if(clc.downloadCURL) {
183 result = qcurl_multi_remove_handle(clc.downloadCURLM,
184 clc.downloadCURL);
185 if(result != CURLM_OK) {
186 Com_DPrintf("qcurl_multi_remove_handle failed: %s\n", qcurl_multi_strerror(result));
187 }
188 qcurl_easy_cleanup(clc.downloadCURL);
189 }
190 result = qcurl_multi_cleanup(clc.downloadCURLM);
191 if(result != CURLM_OK) {
192 Com_DPrintf("CL_cURL_Cleanup: qcurl_multi_cleanup failed: %s\n", qcurl_multi_strerror(result));
193 }
194 clc.downloadCURLM = NULL;
195 clc.downloadCURL = NULL;
196 }
197 else if(clc.downloadCURL) {
198 qcurl_easy_cleanup(clc.downloadCURL);
199 clc.downloadCURL = NULL;
200 }
201}

203static int CL_cURL_CallbackProgress( void *dummy, double dltotal, double dlnow,
204 double ultotal, double ulnow )

213static size_t CL_cURL_CallbackWrite(void *buffer, size_t size, size_t nmemb,
214 void *stream)
215{
216 FS_Write( buffer, size*nmemb, ((fileHandle_t*)stream)[0] );
217 return size*nmemb;

220CURLcode qcurl_easy_setopt_warn(CURL *curl, CURLoption option, ...)
221{
222 CURLcode result;
223
224 va_list argp;
225 va_start(argp, option);
226
227 if(option < CURLOPTTYPE_OBJECTPOINT) {
228 long longValue = va_arg(argp, long);
229 result = qcurl_easy_setopt(curl, option, longValue);
230 } else if(option < CURLOPTTYPE_OFF_T) {
231 void *pointerValue = va_arg(argp, void *);
232 result = qcurl_easy_setopt(curl, option, pointerValue);
233 } else {
234 curl_off_t offsetValue = va_arg(argp, curl_off_t);
235 result = qcurl_easy_setopt(curl, option, offsetValue);
236 }
237
238 if(result != CURLE_OK) {
239 Com_DPrintf("qcurl_easy_setopt failed: %s\n", qcurl_easy_strerror(result));
240 }
241 va_end(argp);
242
243 return result;
244}

246void CL_cURL_BeginDownload( const char *localName, const char *remoteURL )
247{
248 CURLMcode result;
249
250 clc.cURLUsed = qtrue;
251 Com_Printf("URL: %s\n", remoteURL);
252 Com_DPrintf("***** CL_cURL_BeginDownload *****\n"
253 "Localname: %s\n"
254 "RemoteURL: %s\n"
255 "****************************\n", localName, remoteURL);
256 CL_cURL_Cleanup();
257 Q_strncpyz(clc.downloadURL, remoteURL, sizeof(clc.downloadURL));
258 Q_strncpyz(clc.downloadName, localName, sizeof(clc.downloadName));
259 Com_sprintf(clc.downloadTempName, sizeof(clc.downloadTempName),
260 "%s.tmp", localName);
261
262 // Set so UI gets access to it
263 Cvar_Set("cl_downloadName", localName);
264 Cvar_Set("cl_downloadSize", "0");
265 Cvar_Set("cl_downloadCount", "0");
266 Cvar_SetValue("cl_downloadTime", cls.realtime);
267
268 clc.downloadBlock = 0; // Starting new file
269 clc.downloadCount = 0;
270
271 clc.downloadCURL = qcurl_easy_init();
272 if(!clc.downloadCURL) {
273 Com_Error(ERR_DROP, "CL_cURL_BeginDownload: qcurl_easy_init() "
274 "failed");
275 return;
276 }
277 clc.download = FS_SV_FOpenFileWrite(clc.downloadTempName);
278 if(!clc.download) {
279 Com_Error(ERR_DROP, "CL_cURL_BeginDownload: failed to open "
280 "%s for writing", clc.downloadTempName);
281 return;
282 }
283
284 if(com_developer->integer)
285 qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_VERBOSE, 1);
286 qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_URL, clc.downloadURL);
287 qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_TRANSFERTEXT, 0);
288 qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_REFERER, va("ioQ3://%s",
289 NET_AdrToString(clc.serverAddress)));
290 qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_USERAGENT, va("%s %s",
291 Q3_VERSION, qcurl_version()));
292 qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_WRITEFUNCTION,
293 CL_cURL_CallbackWrite);
294 qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_WRITEDATA, &clc.download);
295 qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_NOPROGRESS, 0);
296 qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_PROGRESSFUNCTION,
297 CL_cURL_CallbackProgress);
298 qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_PROGRESSDATA, NULL);
299 qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_FAILONERROR, 1);
300 qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_FOLLOWLOCATION, 1);
301 qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_MAXREDIRS, 5);
302 qcurl_easy_setopt_warn(clc.downloadCURL, CURLOPT_PROTOCOLS,
303 CURLPROTO_HTTP | CURLPROTO_HTTPS | CURLPROTO_FTP | CURLPROTO_FTPS);
304 clc.downloadCURLM = qcurl_multi_init();
305 if(!clc.downloadCURLM) {
306 qcurl_easy_cleanup(clc.downloadCURL);
307 clc.downloadCURL = NULL;
308 Com_Error(ERR_DROP, "CL_cURL_BeginDownload: qcurl_multi_init() "
309 "failed");
310 return;
311 }
312 result = qcurl_multi_add_handle(clc.downloadCURLM, clc.downloadCURL);
313 if(result != CURLM_OK) {
314 qcurl_easy_cleanup(clc.downloadCURL);
315 clc.downloadCURL = NULL;
316 Com_Error(ERR_DROP,"CL_cURL_BeginDownload: qcurl_multi_add_handle() failed: %s", qcurl_multi_strerror(result));
317 return;
318 }
319
320 if(!(clc.sv_allowDownload & DLF_NO_DISCONNECT) &&
321 !clc.cURLDisconnected) {
322
323 CL_AddReliableCommand("disconnect", qtrue);
324 CL_WritePacket();
325 CL_WritePacket();
326 CL_WritePacket();
327 clc.cURLDisconnected = qtrue;
328 }
329}

331void CL_cURL_PerformDownload(void)
332{
333 CURLMcode res;
334 CURLMsg *msg;
335 int c;
336 int i = 0;
337
338 res = qcurl_multi_perform(clc.downloadCURLM, &c);
339 while(res == CURLM_CALL_MULTI_PERFORM && i < 100) {
340 res = qcurl_multi_perform(clc.downloadCURLM, &c);
341 i++;
342 }
343 if(res == CURLM_CALL_MULTI_PERFORM)
344 return;
345 msg = qcurl_multi_info_read(clc.downloadCURLM, &c);
346 if(msg == NULL) {
347 return;
348 }
349 FS_FCloseFile(clc.download);
350 if(msg->msg == CURLMSG_DONE && msg->data.result == CURLE_OK) {
351 FS_SV_Rename(clc.downloadTempName, clc.downloadName, qfalse);
352 clc.downloadRestart = qtrue;
353 }
354 else {
355 long code;
356
357 qcurl_easy_getinfo(msg->easy_handle, CURLINFO_RESPONSE_CODE,
358 &code);
359 Com_Error(ERR_DROP, "Download Error: %s Code: %ld URL: %s",
360 qcurl_easy_strerror(msg->data.result),
361 code, clc.downloadURL);
362 }
363
364 CL_NextDownload();
365}

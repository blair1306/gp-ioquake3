47// snapshots are a view of the server at a given time
48typedef struct {
49 qboolean valid; // cleared if delta parsing was invalid
50 int snapFlags; // rate delayed and dropped commands
51
52 int serverTime; // server time the message is valid for (in msec)
53
54 int messageNum; // copied from netchan->incoming_sequence
55 int deltaNum; // messageNum the delta is from
56 int ping; // time from when cmdNum-1 was sent to time packet was reeceived
57 byte areamask[MAX_MAP_AREA_BYTES]; // portalarea visibility bits
58
59 int cmdNum; // the next cmdNum the server is expecting
60 playerState_t ps; // complete information about the current player at this time
61
62 int numEntities; // all of the entities that need to be presented
63 int parseEntitiesNum; // at the time of this snapshot
64
65 int serverCommandNum; // execute all commands up to this before
66 // making the snapshot current
67} clSnapshot_t;

80typedef struct {

86// the parseEntities array must be large enough to hold PACKET_BACKUP frames of
87// entities, so that when a delta compressed message arives from the server

91extern int g_console_field_width;

93typedef struct {
94 int timeoutcount; // it requres several frames in a timeout condition
95 // to disconnect, preventing debugging breaks from
96 // causing immediate disconnects on continue
97 clSnapshot_t snap; // latest received from server
98
99 int serverTime; // may be paused during play
100 int oldServerTime; // to prevent time from flowing bakcwards
101 int oldFrameServerTime; // to check tournament restarts
102 int serverTimeDelta; // cl.serverTime = cls.realtime + cl.serverTimeDelta
103 // this value changes as net lag varies
104 qboolean extrapolatedSnapshot; // set if any cgame frame has been forced to extrapolate
105 // cleared when CL_AdjustTimeDelta looks at it
106 qboolean newSnapshots; // set on parse of any valid packet
107
108 gameState_t gameState; // configstrings
109 char mapname[MAX_QPATH]; // extracted from CS_SERVERINFO
110
111 int parseEntitiesNum; // index (not anded off) into cl_parse_entities[]
112
113 int mouseDx[2], mouseDy[2]; // added to by mouse events
114 int mouseIndex;
115 int joystickAxis[MAX_JOYSTICK_AXIS]; // set by joystick events
116
117 // cgame communicates a few values to the client system
118 int cgameUserCmdValue; // current weapon to add to usercmd_t
119 float cgameSensitivity;
120
121 // cmds[cmdNumber] is the predicted command, [cmdNumber-1] is the last
122 // properly generated command
123 usercmd_t cmds[CMD_BACKUP]; // each mesage will send several old cmds
124 int cmdNumber; // incremented each frame, because multiple
125 // frames may need to be packed into a single packet
126
127 outPacket_t outPackets[PACKET_BACKUP]; // information about each packet we have sent out
128
129 // the client maintains its own idea of view angles, which are
130 // sent to the server each frame. It is cleared to 0 upon entering each level.
131 // the server sends a delta each frame which is added to the locally
132 // tracked view angles to account for standing on rotating objects,
133 // and teleport direction changes
134 vec3_t viewangles;
135
136 int serverId; // included in each client message so the server
137 // can tell if it is for a prior map_restart
138 // big stuff at end of structure so most offsets are 15 bits or less
139 clSnapshot_t snapshots[PACKET_BACKUP];
140
141 entityState_t entityBaselines[MAX_GENTITIES]; // for delta compression when not in previous frame
142
143 entityState_t parseEntities[MAX_PARSE_ENTITIES];
144} clientActive_t;

331/*
332===================
333Field_Draw
334
335Handles horizontal scrolling and cursor blinking
336x, y, and width are in pixels
337===================
338*/
339void Field_VariableSizeDraw( field_t *edit, int x, int y, int width, int size, qboolean showCursor,
340 qboolean noColorEscape ) {
341 int len;
342 int drawLen;
343 int prestep;
344 int cursorChar;
345 char str[MAX_STRING_CHARS];
346 int i;
347
348 drawLen = edit->widthInChars - 1; // - 1 so there is always a space for the cursor
349 len = strlen( edit->buffer );
350
351 // guarantee that cursor will be visible
352 if ( len <= drawLen ) {
353 prestep = 0;
354 } else {
355 if ( edit->scroll + drawLen > len ) {
356 edit->scroll = len - drawLen;
357 if ( edit->scroll < 0 ) {
358 edit->scroll = 0;
359 }
360 }
361 prestep = edit->scroll;
362 }
363
364 if ( prestep + drawLen > len ) {
365 drawLen = len - prestep;
366 }
367
368 // extract <drawLen> characters from the field at <prestep>
369 if ( drawLen >= MAX_STRING_CHARS ) {
370 Com_Error( ERR_DROP, "drawLen >= MAX_STRING_CHARS" );
371 }
372
373 Com_Memcpy( str, edit->buffer + prestep, drawLen );
374 str[ drawLen ] = 0;
375
376 // draw it
377 if ( size == SMALLCHAR_WIDTH ) {
378 float color[4];
379
380 color[0] = color[1] = color[2] = color[3] = 1.0;
381 SCR_DrawSmallStringExt( x, y, str, color, qfalse, noColorEscape );
382 } else {
383 // draw big string with drop shadow
384 SCR_DrawBigString( x, y, str, 1.0, noColorEscape );
385 }
386
387 // draw the cursor
388 if ( showCursor ) {
389 if ( (int)( cls.realtime >> 8 ) & 1 ) {
390 return; // off blink
391 }
392
393 if ( key_overstrikeMode ) {
394 cursorChar = 11;
395 } else {
396 cursorChar = 10;
397 }
398
399 i = drawLen - strlen( str );
400
401 if ( size == SMALLCHAR_WIDTH ) {
402 SCR_DrawSmallChar( x + ( edit->cursor - prestep - i ) * size, y, cursorChar );
403 } else {
404 str[0] = cursorChar;
405 str[1] = 0;
406 SCR_DrawBigString( x + ( edit->cursor - prestep - i ) * size, y, str, 1.0, qfalse );
407
408 }
409 }
410}

412void Field_Draw( field_t *edit, int x, int y, int width, qboolean showCursor, qboolean noColorEscape )
413{
414 Field_VariableSizeDraw( edit, x, y, width, SMALLCHAR_WIDTH, showCursor, noColorEscape );
415}

417void Field_BigDraw( field_t *edit, int x, int y, int width, qboolean showCursor, qboolean noColorEscape )
418{
419 Field_VariableSizeDraw( edit, x, y, width, BIGCHAR_WIDTH, showCursor, noColorEscape );
420}

422/*
423================
424Field_Paste
425================
426*/
427void Field_Paste( field_t *edit ) {
428 char *cbd;
429 int pasteLen, i;
430
431 cbd = Sys_GetClipboardData();
432
433 if ( !cbd ) {
434 return;
435 }
436
437 // send as if typed, so insert / overstrike works properly
438 pasteLen = strlen( cbd );
439 for ( i = 0 ; i < pasteLen ; i++ ) {
440 Field_CharEvent( edit, cbd[i] );
441 }
442
443 Z_Free( cbd );
444}

446/*
447=================
448Field_KeyDownEvent
449
450Performs the basic line editing functions for the console,
451in-game talk, and menu fields
452
453Key events are used for non-printable characters, others are gotten from char events.
454=================
455*/
456void Field_KeyDownEvent( field_t *edit, int key ) {
457 int len;
458
459 // shift-insert is paste
460 if ( ( ( key == K_INS ) || ( key == K_KP_INS ) ) && keys[K_SHIFT].down ) {
461 Field_Paste( edit );
462 return;
463 }
464
465 key = tolower( key );
466 len = strlen( edit->buffer );
467
468 switch ( key ) {
469 case K_DEL:
470 if ( edit->cursor < len ) {
471 memmove( edit->buffer + edit->cursor,
472 edit->buffer + edit->cursor + 1, len - edit->cursor );
473 }
474 break;
475
476 case K_RIGHTARROW:
477 if ( edit->cursor < len ) {
478 edit->cursor++;
479 }
480 break;
481
482 case K_LEFTARROW:
483 if ( edit->cursor > 0 ) {
484 edit->cursor--;
485 }
486 break;
487
488 case K_HOME:
489 edit->cursor = 0;
490 break;
491
492 case K_END:
493 edit->cursor = len;
494 break;
495
496 case K_INS:
497 key_overstrikeMode = !key_overstrikeMode;
498 break;
499
500 default:
501 break;
502 }
503
504 // Change scroll if cursor is no longer visible
505 if ( edit->cursor < edit->scroll ) {
506 edit->scroll = edit->cursor;
507 } else if ( edit->cursor >= edit->scroll + edit->widthInChars && edit->cursor <= len ) {
508 edit->scroll = edit->cursor - edit->widthInChars + 1;
509 }
510}

512/*
513==================
514Field_CharEvent
515==================
516*/
517void Field_CharEvent( field_t *edit, int ch ) {
518 int len;
519
520 if ( ch == 'v' - 'a' + 1 ) { // ctrl-v is paste
521 Field_Paste( edit );
522 return;
523 }
524
525 if ( ch == 'c' - 'a' + 1 ) { // ctrl-c clears the field
526 Field_Clear( edit );
527 return;
528 }
529
530 len = strlen( edit->buffer );
531
532 if ( ch == 'h' - 'a' + 1 ) { // ctrl-h is backspace
533 if ( edit->cursor > 0 ) {
534 memmove( edit->buffer + edit->cursor - 1,
535 edit->buffer + edit->cursor, len + 1 - edit->cursor );
536 edit->cursor--;
537 if ( edit->cursor < edit->scroll )
538 {
539 edit->scroll--;
540 }
541 }
542 return;
543 }
544
545 if ( ch == 'a' - 'a' + 1 ) { // ctrl-a is home
546 edit->cursor = 0;
547 edit->scroll = 0;
548 return;
549 }
550
551 if ( ch == 'e' - 'a' + 1 ) { // ctrl-e is end
552 edit->cursor = len;
553 edit->scroll = edit->cursor - edit->widthInChars;
554 return;
555 }
556
557 //
558 // ignore any other non printable chars
559 //
560 if ( ch < 32 ) {
561 return;
562 }
563
564 if ( key_overstrikeMode ) {
565 // - 2 to leave room for the leading slash and trailing \0
566 if ( edit->cursor == MAX_EDIT_LINE - 2 )
567 return;
568 edit->buffer[edit->cursor] = ch;
569 edit->cursor++;
570 } else { // insert mode
571 // - 2 to leave room for the leading slash and trailing \0
572 if ( len == MAX_EDIT_LINE - 2 ) {
573 return; // all full
574 }
575 memmove( edit->buffer + edit->cursor + 1,
576 edit->buffer + edit->cursor, len + 1 - edit->cursor );
577 edit->buffer[edit->cursor] = ch;
578 edit->cursor++;
579 }
580
581
582 if ( edit->cursor >= edit->widthInChars ) {
583 edit->scroll++;
584 }
585
586 if ( edit->cursor == len + 1) {
587 edit->buffer[edit->cursor] = 0;
588 }
589}

599/*
600====================
601Console_Key
602
603Handles history and console scrollback
604====================
605*/
606void Console_Key (int key) {
607 // ctrl-L clears screen
608 if ( key == 'l' && keys[K_CTRL].down ) {
609 Cbuf_AddText ("clear\n");
610 return;
611 }
612
613 // enter finishes the line
614 if ( key == K_ENTER || key == K_KP_ENTER ) {
615 // if not in the game explicitly prepend a slash if needed
616 if ( clc.state != CA_ACTIVE && con_autochat->integer &&
617 g_consoleField.buffer[0] &&
618 g_consoleField.buffer[0] != '\\' &&
619 g_consoleField.buffer[0] != '/' ) {
620 char temp[MAX_EDIT_LINE-1];
621
622 Q_strncpyz( temp, g_consoleField.buffer, sizeof( temp ) );
623 Com_sprintf( g_consoleField.buffer, sizeof( g_consoleField.buffer ), "\\%s", temp );
624 g_consoleField.cursor++;
625 }
626
627 Com_Printf ( "]%s\n", g_consoleField.buffer );
628
629 // leading slash is an explicit command
630 if ( g_consoleField.buffer[0] == '\\' || g_consoleField.buffer[0] == '/' ) {
631 Cbuf_AddText( g_consoleField.buffer+1 ); // valid command
632 Cbuf_AddText ("\n");
633 } else {
634 // other text will be chat messages
635 if ( !g_consoleField.buffer[0] ) {
636 return; // empty lines just scroll the console without adding to history
637 } else {
638 if ( con_autochat->integer ) {
639 Cbuf_AddText ("cmd say ");
640 }
641
642 Cbuf_AddText( g_consoleField.buffer );
643 Cbuf_AddText ("\n");
644 }
645 }
646
647 // copy line to history buffer
648 historyEditLines[nextHistoryLine % COMMAND_HISTORY] = g_consoleField;
649 nextHistoryLine++;
650 historyLine = nextHistoryLine;
651
652 Field_Clear( &g_consoleField );
653
654 g_consoleField.widthInChars = g_console_field_width;
655
656 CL_SaveConsoleHistory( );
657
658 if ( clc.state == CA_DISCONNECTED ) {
659 SCR_UpdateScreen (); // force an update, because the command
660 } // may take some time
661 return;
662 }
663
664 // command completion
665
666 if (key == K_TAB) {
667 Field_AutoComplete(&g_consoleField);
668 return;
669 }
670
671 // command history (ctrl-p ctrl-n for unix style)
672
673 if ( (key == K_MWHEELUP && keys[K_SHIFT].down) || ( key == K_UPARROW ) || ( key == K_KP_UPARROW ) ||
674 ( ( tolower(key) == 'p' ) && keys[K_CTRL].down ) ) {
675 if ( nextHistoryLine - historyLine < COMMAND_HISTORY
676 && historyLine > 0 ) {
677 historyLine--;
678 }
679 g_consoleField = historyEditLines[ historyLine % COMMAND_HISTORY ];
680 return;
681 }
682
683 if ( (key == K_MWHEELDOWN && keys[K_SHIFT].down) || ( key == K_DOWNARROW ) || ( key == K_KP_DOWNARROW ) ||
684 ( ( tolower(key) == 'n' ) && keys[K_CTRL].down ) ) {
685 historyLine++;
686 if (historyLine >= nextHistoryLine) {
687 historyLine = nextHistoryLine;
688 Field_Clear( &g_consoleField );
689 g_consoleField.widthInChars = g_console_field_width;
690 return;
691 }
692 g_consoleField = historyEditLines[ historyLine % COMMAND_HISTORY ];
693 return;
694 }
695
696 // console scrolling
697 if ( key == K_PGUP ) {
698 Con_PageUp();
699 return;
700 }
701
702 if ( key == K_PGDN) {
703 Con_PageDown();
704 return;
705 }
706
707 if ( key == K_MWHEELUP) { //----(SA) added some mousewheel functionality to the console
708 Con_PageUp();
709 if(keys[K_CTRL].down) { // hold <ctrl> to accelerate scrolling
710 Con_PageUp();
711 Con_PageUp();
712 }
713 return;
714 }
715
716 if ( key == K_MWHEELDOWN) { //----(SA) added some mousewheel functionality to the console
717 Con_PageDown();
718 if(keys[K_CTRL].down) { // hold <ctrl> to accelerate scrolling
719 Con_PageDown();
720 Con_PageDown();
721 }
722 return;
723 }
724
725 // ctrl-home = top of console
726 if ( key == K_HOME && keys[K_CTRL].down ) {
727 Con_Top();
728 return;
729 }
730
731 // ctrl-end = bottom of console
732 if ( key == K_END && keys[K_CTRL].down ) {

813/*
814===================
815Key_StringToKeynum
816
817Returns a key number to be used to index keys[] by looking at
818the given string. Single ascii characters return themselves, while
819the K_* names are matched up.
820
8210x11 will be interpreted as raw hex, which will allow new controlers
822
823to be configured even if they don't have defined names.
824===================
825*/
826int Key_StringToKeynum( char *str ) {
827 keyname_t *kn;
828
829 if ( !str || !str[0] ) {
830 return -1;
831 }
832 if ( !str[1] ) {
833 return tolower( str[0] );
834 }
835
836 // check for hex code
837 if ( strlen( str ) == 4 ) {
838 int n = Com_HexStrToInt( str );
839
840 if ( n >= 0 ) {
841 return n;
842 }
843 }
844
845 // scan for a text match
846 for ( kn=keynames ; kn->name ; kn++ ) {
847 if ( !Q_stricmp( str,kn->name ) )
848 return kn->keynum;
849 }
850
851 return -1;
852}
853
854/*
855===================
856Key_KeynumToString
857
858Returns a string (either a single ascii char, a K_* name, or a 0x11 hex string) for the
859given keynum.
860===================
861*/
862char *Key_KeynumToString( int keynum ) {
863 keyname_t *kn;
864 static char tinystr[5];
865 int i, j;
866
867 if ( keynum == -1 ) {
868 return "<KEY NOT FOUND>";
869 }
870
871 if ( keynum < 0 || keynum >= MAX_KEYS ) {
872 return "<OUT OF RANGE>";
873 }
874
875 // check for printable ascii (don't use quote)
876 if ( keynum > 32 && keynum < 127 && keynum != '"' && keynum != ';' ) {
877 tinystr[0] = keynum;
878 tinystr[1] = 0;
879 return tinystr;
880 }
881
882 // check for a key string
883 for ( kn=keynames ; kn->name ; kn++ ) {
884 if (keynum == kn->keynum) {
885 return kn->name;
886 }
887 }
888
889 // make a hex string
890 i = keynum >> 4;
891 j = keynum & 15;
892
893 tinystr[0] = '0';
894 tinystr[1] = 'x';
895 tinystr[2] = i > 9 ? i - 10 + 'a' : i + '0';
896 tinystr[3] = j > 9 ? j - 10 + 'a' : j + '0';
897 tinystr[4] = 0;
898
899 return tinystr;
900}

1172/*
1173===================
1174CL_ParseBinding
1175
1176Execute the commands in the bind string
1177===================
1178*/
1179void CL_ParseBinding( int key, qboolean down, unsigned time )
1180{
1181 char buf[ MAX_STRING_CHARS ], *p = buf, *end;
1182 qboolean allCommands, allowUpCmds;
1183
1184 if( clc.state == CA_DISCONNECTED && Key_GetCatcher( ) == 0 )
1185 return;
1186 if( !keys[key].binding || !keys[key].binding[0] )
1187 return;
1188 Q_strncpyz( buf, keys[key].binding, sizeof( buf ) );
1189
1190 // run all bind commands if console, ui, etc aren't reading keys
1191 allCommands = ( Key_GetCatcher( ) == 0 );
1192
1193 // allow button up commands if in game even if key catcher is set
1194 allowUpCmds = ( clc.state != CA_DISCONNECTED );
1195
1196 while( 1 )
1197 {
1198 while( isspace( *p ) )
1199 p++;
1200 end = strchr( p, ';' );
1201 if( end )
1202 *end = '\0';
1203 if( *p == '+' )
1204 {
1205 // button commands add keynum and time as parameters
1206 // so that multiple sources can be discriminated and
1207 // subframe corrected
1208 if ( allCommands || ( allowUpCmds && !down ) ) {
1209 char cmd[1024];
1210 Com_sprintf( cmd, sizeof( cmd ), "%c%s %d %d\n",
1211 ( down ) ? '+' : '-', p + 1, key, time );
1212 Cbuf_AddText( cmd );
1213 }
1214 }
1215 else if( down )
1216 {
1217 // normal commands only execute on key press
1218 if ( allCommands || CL_BindUICommand( p ) ) {
1219 Cbuf_AddText( p );
1220 Cbuf_AddText( "\n" );
1221 }
1222 }
1223 if( !end )
1224 break;
1225 p = end + 1;
1226 }
1227}

1229/*
1230===================
1231CL_KeyDownEvent
1232
1233Called by CL_KeyEvent to handle a keypress
1234===================
1235*/
1236void CL_KeyDownEvent( int key, unsigned time )
1237{
1238 keys[key].down = qtrue;
1239 keys[key].repeats++;
1240 if( keys[key].repeats == 1 )
1241 anykeydown++;
1242
1243 if( keys[K_ALT].down && key == K_ENTER )
1244 {
1245 // don't repeat fullscreen toggle when keys are held down
1246 if ( keys[K_ENTER].repeats > 1 ) {
1247 return;
1248 }
1249
1250 Cvar_SetValue( "r_fullscreen",
1251 !Cvar_VariableIntegerValue( "r_fullscreen" ) );
1252 return;
1253 }
1254
1255 // console key is hardcoded, so the user can never unbind it
1256 if( key == K_CONSOLE || ( keys[K_SHIFT].down && key == K_ESCAPE ) )
1257 {
1258 Con_ToggleConsole_f ();
1259 Key_ClearStates ();
1260 return;
1261 }
1262
1263
1264 // keys can still be used for bound actions
1265 if ( ( key < 128 || key == K_MOUSE1 ) &&
1266 ( clc.demoplaying || clc.state == CA_CINEMATIC ) && Key_GetCatcher( ) == 0 ) {
1267
1268 if (Cvar_VariableValue ("com_cameraMode") == 0) {
1269 Cvar_Set ("nextdemo","");
1270 key = K_ESCAPE;
1271 }
1272 }
1273
1274 // escape is always handled special
1275 if ( key == K_ESCAPE ) {
1276 if ( Key_GetCatcher( ) & KEYCATCH_MESSAGE ) {
1277 // clear message mode
1278 Message_Key( key );
1279 return;
1280 }
1281
1282 // escape always gets out of CGAME stuff
1283 if (Key_GetCatcher( ) & KEYCATCH_CGAME) {
1284 Key_SetCatcher( Key_GetCatcher( ) & ~KEYCATCH_CGAME );
1285 VM_Call (cgvm, CG_EVENT_HANDLING, CGAME_EVENT_NONE);
1286 return;
1287 }
1288
1289 if ( !( Key_GetCatcher( ) & KEYCATCH_UI ) ) {
1290 if ( clc.state == CA_ACTIVE && !clc.demoplaying ) {
1291 VM_Call( uivm, UI_SET_ACTIVE_MENU, UIMENU_INGAME );
1292 }
1293 else if ( clc.state != CA_DISCONNECTED ) {
1294 CL_Disconnect_f();
1295 S_StopAllSounds();
1296 VM_Call( uivm, UI_SET_ACTIVE_MENU, UIMENU_MAIN );
1297 }
1298 return;
1299 }
1300
1301 VM_Call( uivm, UI_KEY_EVENT, key, qtrue );
1302 return;
1303 }
1304
1305 // send the bound action
1306 CL_ParseBinding( key, qtrue, time );
1307
1308 // distribute the key down event to the apropriate handler
1309 if ( Key_GetCatcher( ) & KEYCATCH_CONSOLE ) {
1310 Console_Key( key );
1311 } else if ( Key_GetCatcher( ) & KEYCATCH_UI ) {
1312 if ( uivm ) {
1313 VM_Call( uivm, UI_KEY_EVENT, key, qtrue );
1314 }
1315 } else if ( Key_GetCatcher( ) & KEYCATCH_CGAME ) {
1316 if ( cgvm ) {
1317 VM_Call( cgvm, CG_KEY_EVENT, key, qtrue );
1318 }
1319 } else if ( Key_GetCatcher( ) & KEYCATCH_MESSAGE ) {
1320 Message_Key( key );
1321 } else if ( clc.state == CA_DISCONNECTED ) {
1322 Console_Key( key );
1323 }
1324}

1326/*
1327===================
1328CL_KeyUpEvent
1329
1330Called by CL_KeyEvent to handle a keyrelease
1331===================
1332*/
1333void CL_KeyUpEvent( int key, unsigned time )
1334{
1335 keys[key].repeats = 0;
1336 keys[key].down = qfalse;
1337 anykeydown--;
1338
1339 if (anykeydown < 0) {
1340 anykeydown = 0;
1341 }
1342
1343 // don't process key-up events for the console key
1344 if ( key == K_CONSOLE || ( key == K_ESCAPE && keys[K_SHIFT].down ) )
1345 return;
1346
1347 //
1348 // key up events only perform actions if the game key binding is
1349 // a button command (leading + sign). These will be processed even in
1350 // console mode and menu mode, to keep the character from continuing
1351 // an action started before a mode switch.
1352 //
1353 CL_ParseBinding( key, qfalse, time );
1354
1355 if ( Key_GetCatcher( ) & KEYCATCH_UI && uivm ) {
1356 VM_Call( uivm, UI_KEY_EVENT, key, qfalse );
1357 } else if ( Key_GetCatcher( ) & KEYCATCH_CGAME && cgvm ) {
1358 VM_Call( cgvm, CG_KEY_EVENT, key, qfalse );
1359 }
1360}

1455// This must not exceed MAX_CMD_LINE
1456#define MAX_CONSOLE_SAVE_BUFFER 1024
1457#define CONSOLE_HISTORY_FILE "q3history"
1458static char consoleSaveBuffer[ MAX_CONSOLE_SAVE_BUFFER ];
1459static int consoleSaveBufferSize = 0;
1460
1461/*
1462================
1463CL_LoadConsoleHistory
1464
1465Load the console history from cl_consoleHistory
1466================
1467*/
1468void CL_LoadConsoleHistory( void )
1469{
1470 char *token, *text_p;
1471 int i, numChars, numLines = 0;
1472 fileHandle_t f;
1473
1474 consoleSaveBufferSize = FS_FOpenFileRead( CONSOLE_HISTORY_FILE, &f, qfalse );
1475 if( !f )
1476 {
1477 Com_Printf( "Couldn't read %s.\n", CONSOLE_HISTORY_FILE );
1478 return;
1479 }
1480
1481 if( consoleSaveBufferSize <= MAX_CONSOLE_SAVE_BUFFER &&
1482 FS_Read( consoleSaveBuffer, consoleSaveBufferSize, f ) == consoleSaveBufferSize )
1483 {
1484 text_p = consoleSaveBuffer;
1485
1486 for( i = COMMAND_HISTORY - 1; i >= 0; i-- )
1487 {
1488 if( !*( token = COM_Parse( &text_p ) ) )
1489 break;
1490
1491 historyEditLines[ i ].cursor = atoi( token );
1492
1493 if( !*( token = COM_Parse( &text_p ) ) )
1494 break;
1495
1496 historyEditLines[ i ].scroll = atoi( token );
1497
1498 if( !*( token = COM_Parse( &text_p ) ) )
1499 break;
1500
1501 numChars = atoi( token );
1502 text_p++;
1503 if( numChars > ( strlen( consoleSaveBuffer ) - ( text_p - consoleSaveBuffer ) ) )
1504 {
1505 Com_DPrintf( S_COLOR_YELLOW "WARNING: probable corrupt history\n" );
1506 break;
1507 }
1508 Com_Memcpy( historyEditLines[ i ].buffer,
1509 text_p, numChars );
1510 historyEditLines[ i ].buffer[ numChars ] = '\0';
1511 text_p += numChars;
1512
1513 numLines++;
1514 }
1515
1516 memmove( &historyEditLines[ 0 ], &historyEditLines[ i + 1 ],
1517 numLines * sizeof( field_t ) );
1518 for( i = numLines; i < COMMAND_HISTORY; i++ )
1519 Field_Clear( &historyEditLines[ i ] );
1520
1521 historyLine = nextHistoryLine = numLines;
1522 }
1523 else
1524 Com_Printf( "Couldn't read %s.\n", CONSOLE_HISTORY_FILE );
1525
1526 FS_FCloseFile( f );
1527}

1530================
1531CL_SaveConsoleHistory
1532
1533Save the console history into the cvar cl_consoleHistory
1534so that it persists across invocations of q3
1535================
1536*/
1537void CL_SaveConsoleHistory( void )
1538{
1539 int i;
1540 int lineLength, saveBufferLength, additionalLength;
1541 fileHandle_t f;
1542
1543 consoleSaveBuffer[ 0 ] = '\0';
1544
1545 i = ( nextHistoryLine - 1 ) % COMMAND_HISTORY;
1546 do
1547 {
1548 if( historyEditLines[ i ].buffer[ 0 ] )
1549 {
1550 lineLength = strlen( historyEditLines[ i ].buffer );
1551 saveBufferLength = strlen( consoleSaveBuffer );
1552
1553 //ICK
1554 additionalLength = lineLength + strlen( "999 999 999 " );
1555
1556 if( saveBufferLength + additionalLength < MAX_CONSOLE_SAVE_BUFFER )
1557 {
1558 Q_strcat( consoleSaveBuffer, MAX_CONSOLE_SAVE_BUFFER,
1559 va( "%d %d %d %s ",

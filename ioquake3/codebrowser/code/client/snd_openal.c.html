54/*
55=================
56S_AL_Format
57=================
58*/
59static
60ALuint S_AL_Format(int width, int channels)
61{
62 ALuint format = AL_FORMAT_MONO16;
63
64 // Work out format
65 if(width == 1)
66 {
67 if(channels == 1)
68 format = AL_FORMAT_MONO8;
69 else if(channels == 2)
70 format = AL_FORMAT_STEREO8;
71 }
72 else if(width == 2)
73 {
74 if(channels == 1)
75 format = AL_FORMAT_MONO16;
76 else if(channels == 2)
77 format = AL_FORMAT_STEREO16;
78 }
79
80 return format;
81}

96 case AL_INVALID_ENUM:

98 case AL_INVALID_VALUE:
99 return "Invalid value";

102 case AL_OUT_OF_MEMORY:

159S_AL_BufferFindFree
160
161Find a free handle

164static sfxHandle_t S_AL_BufferFindFree( void )
165{
166 int i;
167
168 for(i = 0; i < MAX_SFX; i++)
169 {
170 // Got one
171 if(knownSfx[i].filename[0] == '\0')
172 {
173 if(i >= numSfx)
174 numSfx = i + 1;
175 return i;
176 }
177 }
178
179 // Shit...
180 Com_Error(ERR_FATAL, "S_AL_BufferFindFree: No free sound handles");
181 return -1;
182}

184/*
185=================
186S_AL_BufferFind
187
188Find a sound effect if loaded, set up a handle otherwise
189=================
190*/
191static sfxHandle_t S_AL_BufferFind(const char *filename)
192{
193 // Look it up in the table
194 sfxHandle_t sfx = -1;
195 int i;
196
197 if ( !filename ) {
198 Com_Error( ERR_FATAL, "Sound name is NULL" );
199 }
200
201 if ( !filename[0] ) {
202 Com_Printf( S_COLOR_YELLOW "WARNING: Sound name is empty\n" );
203 return 0;
204 }
205
206 if ( strlen( filename ) >= MAX_QPATH ) {
207 Com_Printf( S_COLOR_YELLOW "WARNING: Sound name is too long: %s\n", filename );
208 return 0;
209 }
210
211 if ( filename[0] == '*' ) {
212 Com_Printf( S_COLOR_YELLOW "WARNING: Tried to load player sound directly: %s\n", filename );
213 return 0;
214 }
215
216 for(i = 0; i < numSfx; i++)
217 {
218 if(!Q_stricmp(knownSfx[i].filename, filename))
219 {
220 sfx = i;
221 break;
222 }
223 }
224
225 // Not found in table?
226 if(sfx == -1)
227 {
228 alSfx_t *ptr;
229
230 sfx = S_AL_BufferFindFree();
231
232 // Clear and copy the filename over
233 ptr = &knownSfx[sfx];
234 memset(ptr, 0, sizeof(*ptr));
235 ptr->masterLoopSrc = -1;
236 strcpy(ptr->filename, filename);
237 }
238
239 // Return the handle
240 return sfx;
241}

258/*
259=================
260S_AL_BufferUnload
261=================
262*/
263static void S_AL_BufferUnload(sfxHandle_t sfx)
264{
265 if(knownSfx[sfx].filename[0] == '\0')
266 return;
267
268 if(!knownSfx[sfx].inMemory)
269 return;
270
271 // Delete it
272 S_AL_ClearError( qfalse );
273 qalDeleteBuffers(1, &knownSfx[sfx].buffer);
274 if(qalGetError() != AL_NO_ERROR)
275 Com_Printf( S_COLOR_RED "ERROR: Can't delete sound buffer for %s\n",
276 knownSfx[sfx].filename);
277
278 knownSfx[sfx].inMemory = qfalse;
279}

281/*
282=================
283S_AL_BufferEvict
284=================
285*/
286static qboolean S_AL_BufferEvict( void )
287{
288 int i, oldestBuffer = -1;
289 int oldestTime = Sys_Milliseconds( );
290
291 for( i = 0; i < numSfx; i++ )
292 {
293 if( !knownSfx[ i ].filename[ 0 ] )
294 continue;
295
296 if( !knownSfx[ i ].inMemory )
297 continue;
298
299 if( knownSfx[ i ].lastUsedTime < oldestTime )
300 {
301 oldestTime = knownSfx[ i ].lastUsedTime;
302 oldestBuffer = i;
303 }
304 }
305
306 if( oldestBuffer >= 0 )
307 {
308 S_AL_BufferUnload( oldestBuffer );
309 return qtrue;
310 }
311 else
312 return qfalse;
313}

315/*
316=================
317S_AL_GenBuffers
318=================
319*/
320static qboolean S_AL_GenBuffers(ALsizei numBuffers, ALuint *buffers, const char *name)
321{
322 ALenum error;
323
324 S_AL_ClearError( qfalse );
325 qalGenBuffers( numBuffers, buffers );
326 error = qalGetError();
327
328 // If we ran out of buffers, start evicting the least recently used sounds
329 while( error == AL_INVALID_VALUE )
330 {
331 if( !S_AL_BufferEvict( ) )
332 {
333 Com_Printf( S_COLOR_RED "ERROR: Out of audio buffers\n");
334 return qfalse;
335 }
336
337 // Try again
338 S_AL_ClearError( qfalse );
339 qalGenBuffers( numBuffers, buffers );
340 error = qalGetError();
341 }
342
343 if( error != AL_NO_ERROR )
344 {
345 Com_Printf( S_COLOR_RED "ERROR: Can't create a sound buffer for %s - %s\n",
346 name, S_AL_ErrorMsg(error));
347 return qfalse;
348 }
349
350 return qtrue;
351}

353/*
354=================
355S_AL_BufferLoad
356=================
357*/
358static void S_AL_BufferLoad(sfxHandle_t sfx, qboolean cache)
359{
360 ALenum error;
361 ALuint format;
362
363 void *data;
364 snd_info_t info;
365 alSfx_t *curSfx = &knownSfx[sfx];
366
367 // Nothing?
368 if(curSfx->filename[0] == '\0')
369 return;
370
371 // Already done?
372 if((curSfx->inMemory) || (curSfx->isDefault) || (!cache && curSfx->isDefaultChecked))
373 return;
374
375 // Try to load
376 data = S_CodecLoad(curSfx->filename, &info);
377 if(!data)
378 {
379 S_AL_BufferUseDefault(sfx);
380 return;
381 }
382
383 curSfx->isDefaultChecked = qtrue;
384
385 if (!cache)
386 {
387 // Don't create AL cache
388 Hunk_FreeTempMemory(data);
389 return;
390 }
391
392 format = S_AL_Format(info.width, info.channels);
393
394 // Create a buffer
395 if (!S_AL_GenBuffers(1, &curSfx->buffer, curSfx->filename))
396 {
397 S_AL_BufferUseDefault(sfx);
398 Hunk_FreeTempMemory(data);
399 return;
400 }
401
402 // Fill the buffer
403 if( info.size == 0 )
404 {
405 // We have no data to buffer, so buffer silence
406 byte dummyData[ 2 ] = { 0 };
407
408 qalBufferData(curSfx->buffer, AL_FORMAT_MONO16, (void *)dummyData, 2, 22050);
409 }
410 else
411 qalBufferData(curSfx->buffer, format, data, info.size, info.rate);
412
413 error = qalGetError();
414
415 // If we ran out of memory, start evicting the least recently used sounds
416 while(error == AL_OUT_OF_MEMORY)
417 {
418 if( !S_AL_BufferEvict( ) )
419 {
420 qalDeleteBuffers(1, &curSfx->buffer);
421 S_AL_BufferUseDefault(sfx);
422 Hunk_FreeTempMemory(data);
423 Com_Printf( S_COLOR_RED "ERROR: Out of memory loading %s\n", curSfx->filename);
424 return;
425 }
426
427 // Try load it again
428 qalBufferData(curSfx->buffer, format, data, info.size, info.rate);
429 error = qalGetError();
430 }
431
432 // Some other error condition
433 if(error != AL_NO_ERROR)
434 {
435 qalDeleteBuffers(1, &curSfx->buffer);
436 S_AL_BufferUseDefault(sfx);
437 Hunk_FreeTempMemory(data);
438 Com_Printf( S_COLOR_RED "ERROR: Can't fill sound buffer for %s - %s\n",
439 curSfx->filename, S_AL_ErrorMsg(error));
440 return;
441 }
442
443 curSfx->info = info;
444
445 // Free the memory
446 Hunk_FreeTempMemory(data);
447
448 // Woo!
449 curSfx->inMemory = qtrue;
450}

643/*
644=================
645S_AL_ScaleGain
646Adapt the gain if necessary to get a quicker fadeout when the source is too far away.
647=================
648*/
649
650static void S_AL_ScaleGain(src_t *chksrc, vec3_t origin)
651{
652 float distance;
653
654 if(!chksrc->local)
655 distance = Distance(origin, lastListenerOrigin);
656
657 // If we exceed a certain distance, scale the gain linearly until the sound
658 // vanishes into nothingness.
659 if(!chksrc->local && (distance -= s_alMaxDistance->value) > 0)
660 {
661 float scaleFactor;
662
663 if(distance >= s_alGraceDistance->value)
664 scaleFactor = 0.0f;
665 else
666 scaleFactor = 1.0f - distance / s_alGraceDistance->value;
667
668 scaleFactor *= chksrc->curGain;
669
670 if(chksrc->scaleGain != scaleFactor)
671 {
672 chksrc->scaleGain = scaleFactor;
673 S_AL_Gain(chksrc->alSource, chksrc->scaleGain);
674 }
675 }
676 else if(chksrc->scaleGain != chksrc->curGain)
677 {
678 chksrc->scaleGain = chksrc->curGain;
679 S_AL_Gain(chksrc->alSource, chksrc->scaleGain);
680 }
681}

683/*
684=================
685S_AL_HearingThroughEntity
686
687Also see S_Base_HearingThroughEntity
688=================
689*/
690static qboolean S_AL_HearingThroughEntity( int entityNum )
691{
692 float distanceSq;
693
694 if( lastListenerNumber == entityNum )
695 {
696 // This is an outrageous hack to detect
697 // whether or not the player is rendering in third person or not. We can't
698 // ask the renderer because the renderer has no notion of entities and we
699 // can't ask cgame since that would involve changing the API and hence mod
700 // compatibility. I don't think there is any way around this, but I'll leave
701 // the FIXME just in case anyone has a bright idea.
702 distanceSq = DistanceSquared(
703 entityList[ entityNum ].origin,
704 lastListenerOrigin );
705
706 if( distanceSq > THIRD_PERSON_THRESHOLD_SQ )
707 return qfalse; //we're the player, but third person
708 else
709 return qtrue; //we're the player
710 }
711 else
712 return qfalse; //not the player
713}

715/*
716=================
717S_AL_SrcInit
718=================
719*/
720static
721qboolean S_AL_SrcInit( void )
722{
723 int i;
724 int limit;
725
726 // Clear the sources data structure
727 memset(srcList, 0, sizeof(srcList));
728 srcCount = 0;
729 srcActiveCnt = 0;
730
731 // Cap s_alSources to MAX_SRC
732 limit = s_alSources->integer;
733 if(limit > MAX_SRC)
734 limit = MAX_SRC;
735 else if(limit < 16)
736 limit = 16;
737
738 S_AL_ClearError( qfalse );
739 // Allocate as many sources as possible
740 for(i = 0; i < limit; i++)
741 {
742 qalGenSources(1, &srcList[i].alSource);
743 if(qalGetError() != AL_NO_ERROR)
744 break;
745 srcCount++;
746 }
747
748 // All done. Print this for informational purposes
749 Com_Printf( "Allocated %d sources.\n", srcCount);
750 alSourcesInitialised = qtrue;
751 return qtrue;
752}

760void S_AL_SrcShutdown( void )
761{
762 int i;
763 src_t *curSource;
764

766 return;

776 if(curSource->entity > 0)
777 entityList[curSource->entity].srcAllocated = qfalse;

780 qalDeleteSources(1, &srcList[i].alSource);
781 }
782
783 memset(srcList, 0, sizeof(srcList));
784
785 alSourcesInitialised = qfalse;
786}

788/*
789=================
790S_AL_SrcSetup
791=================
792*/
793static void S_AL_SrcSetup(srcHandle_t src, sfxHandle_t sfx, alSrcPriority_t priority,
794 int entity, int channel, qboolean local)
795{
796 src_t *curSource;
797
798 // Set up src struct
799 curSource = &srcList[src];
800
801 curSource->lastUsedTime = Sys_Milliseconds();
802 curSource->sfx = sfx;
803 curSource->priority = priority;
804 curSource->entity = entity;
805 curSource->channel = channel;
806 curSource->isPlaying = qfalse;
807 curSource->isLocked = qfalse;
808 curSource->isLooping = qfalse;
809 curSource->isTracking = qfalse;
810 curSource->isStream = qfalse;
811 curSource->curGain = s_alGain->value * s_volume->value;
812 curSource->scaleGain = curSource->curGain;
813 curSource->local = local;
814
815 // Set up OpenAL source
816 if(sfx >= 0)
817 {
818 // Mark the SFX as used, and grab the raw AL buffer
819 S_AL_BufferUse(sfx);
820 qalSourcei(curSource->alSource, AL_BUFFER, S_AL_BufferGet(sfx));
821 }
822
823 qalSourcef(curSource->alSource, AL_PITCH, 1.0f);
824 S_AL_Gain(curSource->alSource, curSource->curGain);
825 qalSourcefv(curSource->alSource, AL_POSITION, vec3_origin);
826 qalSourcefv(curSource->alSource, AL_VELOCITY, vec3_origin);
827 qalSourcei(curSource->alSource, AL_LOOPING, AL_FALSE);
828 qalSourcef(curSource->alSource, AL_REFERENCE_DISTANCE, s_alMinDistance->value);
829
830 if(local)
831 {
832 qalSourcei(curSource->alSource, AL_SOURCE_RELATIVE, AL_TRUE);
833 qalSourcef(curSource->alSource, AL_ROLLOFF_FACTOR, 0.0f);
834 }
835 else
836 {
837 qalSourcei(curSource->alSource, AL_SOURCE_RELATIVE, AL_FALSE);
838 qalSourcef(curSource->alSource, AL_ROLLOFF_FACTOR, s_alRolloff->value);
839 }
840}

842/*
843=================
844S_AL_SaveLoopPos
845Remove given source as loop master if it is the master and hand off master status to another source in this case.
846=================
847*/
848
849static void S_AL_SaveLoopPos(src_t *dest, ALuint alSource)
850{
851 int error;
852
853 S_AL_ClearError(qfalse);
854
855 qalGetSourcef(alSource, AL_SEC_OFFSET, &dest->lastTimePos);
856 if((error = qalGetError()) != AL_NO_ERROR)
857 {
858 // Old OpenAL implementations don't support AL_SEC_OFFSET
859
860 if(error != AL_INVALID_ENUM)
861 {
862 Com_Printf(S_COLOR_YELLOW "WARNING: Could not get time offset for alSource %d: %s\n",
863 alSource, S_AL_ErrorMsg(error));
864 }
865
866 dest->lastTimePos = -1;
867 }
868 else
869 dest->lastSampleTime = Sys_Milliseconds();

872/*
873=================
874S_AL_NewLoopMaster
875Remove given source as loop master if it is the master and hand off master status to another source in this case.
876=================
877*/
878
879static void S_AL_NewLoopMaster(src_t *rmSource, qboolean iskilled)
880{
881 int index;
882 src_t *curSource = NULL;
883 alSfx_t *curSfx;
884
885 curSfx = &knownSfx[rmSource->sfx];
886
887 if(rmSource->isPlaying)
888 curSfx->loopActiveCnt--;
889 if(iskilled)
890 curSfx->loopCnt--;
891
892 if(curSfx->loopCnt)
893 {
894 if(rmSource->priority == SRCPRI_ENTITY)
895 {
896 if(!iskilled && rmSource->isPlaying)
897 {
898 // only sync ambient loops...
899 // It makes more sense to have sounds for weapons/projectiles unsynced
900 S_AL_SaveLoopPos(rmSource, rmSource->alSource);
901 }
902 }
903 else if(rmSource == &srcList[curSfx->masterLoopSrc])
904 {
905 int firstInactive = -1;
906
907 // Only if rmSource was the master and if there are still playing loops for
908 // this sound will we need to find a new master.
909
910 if(iskilled || curSfx->loopActiveCnt)
911 {
912 for(index = 0; index < srcCount; index++)
913 {
914 curSource = &srcList[index];
915
916 if(curSource->sfx == rmSource->sfx && curSource != rmSource &&
917 curSource->isActive && curSource->isLooping && curSource->priority == SRCPRI_AMBIENT)
918 {
919 if(curSource->isPlaying)
920 {
921 curSfx->masterLoopSrc = index;
922 break;
923 }
924 else if(firstInactive < 0)
925 firstInactive = index;
926 }
927 }
928 }
929
930 if(!curSfx->loopActiveCnt)
931 {
932 if(firstInactive < 0)
933 {
934 if(iskilled)
935 {
936 curSfx->masterLoopSrc = -1;
937 return;
938 }
939 else
940 curSource = rmSource;
941 }
942 else
943 curSource = &srcList[firstInactive];
944
945 if(rmSource->isPlaying)
946 {
947 // this was the last not stopped source, save last sample position + time
948 S_AL_SaveLoopPos(curSource, rmSource->alSource);
949 }
950 else
951 {
952 // second case: all loops using this sound have stopped due to listener being of of range,
953 // and now the inactive master gets deleted. Just move over the soundpos settings to the
954 // new master.
955 curSource->lastTimePos = rmSource->lastTimePos;
956 curSource->lastSampleTime = rmSource->lastSampleTime;
957 }
958 }
959 }
960 }
961 else
962 curSfx->masterLoopSrc = -1;
963}

965/*
966=================
967S_AL_SrcKill
968=================
969*/
970static void S_AL_SrcKill(srcHandle_t src)
971{
972 src_t *curSource = &srcList[src];
973
974 // I'm not touching it. Unlock it first.
975 if(curSource->isLocked)
976 return;
977
978 // Remove the entity association and loop master status
979 if(curSource->isLooping)
980 {
981 curSource->isLooping = qfalse;
982
983 if(curSource->entity != -1)
984 {
985 sentity_t *curEnt = &entityList[curSource->entity];
986
987 curEnt->srcAllocated = qfalse;
988 curEnt->srcIndex = -1;
989 curEnt->loopAddedThisFrame = qfalse;
990 curEnt->startLoopingSound = qfalse;
991 }
992
993 S_AL_NewLoopMaster(curSource, qtrue);
994 }
995
996 // Stop it if it's playing
997 if(curSource->isPlaying)
998 {
999 qalSourceStop(curSource->alSource);
1000 curSource->isPlaying = qfalse;
1001 }
1002
1003 // Detach any buffers
1004 qalSourcei(curSource->alSource, AL_BUFFER, 0);
1005
1006 curSource->sfx = 0;
1007 curSource->lastUsedTime = 0;
1008 curSource->priority = 0;
1009 curSource->entity = -1;
1010 curSource->channel = -1;
1011 if(curSource->isActive)
1012 {
1013 curSource->isActive = qfalse;
1014 srcActiveCnt--;
1015 }
1016 curSource->isLocked = qfalse;
1017 curSource->isTracking = qfalse;
1018 curSource->local = qfalse;
1019}

1021/*
1022=================
1023S_AL_SrcAlloc
1024=================
1025*/
1026static
1027srcHandle_t S_AL_SrcAlloc( alSrcPriority_t priority, int entnum, int channel )
1028{
1029 int i;
1030 int empty = -1;
1031 int weakest = -1;
1032 int weakest_time = Sys_Milliseconds();
1033 int weakest_pri = 999;
1034 float weakest_gain = 1000.0;
1035 qboolean weakest_isplaying = qtrue;
1036 int weakest_numloops = 0;
1037 src_t *curSource;
1038
1039 for(i = 0; i < srcCount; i++)
1040 {
1041 curSource = &srcList[i];
1042
1043 // If it's locked, we aren't even going to look at it
1044 if(curSource->isLocked)
1045 continue;
1046
1047 // Is it empty or not?
1048 if(!curSource->isActive)
1049 {
1050 empty = i;
1051 break;
1052 }
1053
1054 if(curSource->isPlaying)
1055 {
1056 if(weakest_isplaying && curSource->priority < priority &&
1057 (curSource->priority < weakest_pri ||
1058 (!curSource->isLooping && (curSource->scaleGain < weakest_gain || curSource->lastUsedTime < weakest_time))))
1059 {
1060 // If it has lower priority, is fainter or older, flag it as weak
1061 // the last two values are only compared if it's not a looping sound, because we want to prevent two
1062 // loops (loops are added EVERY frame) fighting for a slot
1063 weakest_pri = curSource->priority;
1064 weakest_time = curSource->lastUsedTime;
1065 weakest_gain = curSource->scaleGain;
1066 weakest = i;
1067 }
1068 }
1069 else
1070 {
1071 weakest_isplaying = qfalse;
1072
1073 if(weakest < 0 ||
1074 knownSfx[curSource->sfx].loopCnt > weakest_numloops ||
1075 curSource->priority < weakest_pri ||
1076 curSource->lastUsedTime < weakest_time)
1077 {
1078 // Sources currently not playing of course have lowest priority
1079 // also try to always keep at least one loop master for every loop sound
1080 weakest_pri = curSource->priority;
1081 weakest_time = curSource->lastUsedTime;
1082 weakest_numloops = knownSfx[curSource->sfx].loopCnt;
1083 weakest = i;
1084 }
1085 }
1086
1087 // The channel system is not actually adhered to by baseq3, and not
1088 // implemented in snd_dma.c, so while the following is strictly correct, it
1089 // causes incorrect behaviour versus defacto baseq3
1090#if 0
1091 // Is it an exact match, and not on channel 0?
1092 if((curSource->entity == entnum) && (curSource->channel == channel) && (channel != 0))
1093 {
1094 S_AL_SrcKill(i);
1095 return i;
1096 }
1097#endif
1098 }
1099
1100 if(empty == -1)
1101 empty = weakest;
1102
1103 if(empty >= 0)
1104 {
1105 S_AL_SrcKill(empty);
1106 srcList[empty].isActive = qtrue;
1107 srcActiveCnt++;
1108 }
1109
1110 return empty;
1111}

1233

1310/*
1311=================
1312S_AL_SrcLoop
1313=================
1314*/
1315static void S_AL_SrcLoop( alSrcPriority_t priority, sfxHandle_t sfx,
1316 const vec3_t origin, const vec3_t velocity, int entityNum )
1317{
1318 int src;
1319 sentity_t *sent = &entityList[ entityNum ];
1320 src_t *curSource;
1321 vec3_t sorigin, svelocity;
1322
1323 if(S_AL_CheckInput(entityNum, sfx))
1324 return;
1325
1326 // Do we need to allocate a new source for this entity
1327 if( !sent->srcAllocated )
1328 {
1329 // Try to get a channel
1330 src = S_AL_SrcAlloc( priority, entityNum, -1 );
1331 if( src == -1 )
1332 {
1333 Com_DPrintf( S_COLOR_YELLOW "WARNING: Failed to allocate source "
1334 "for loop sfx %d on entity %d\n", sfx, entityNum );
1335 return;
1336 }
1337
1338 curSource = &srcList[src];
1339
1340 sent->startLoopingSound = qtrue;
1341
1342 curSource->lastTimePos = -1.0;
1343 curSource->lastSampleTime = Sys_Milliseconds();
1344 }
1345 else
1346 {
1347 src = sent->srcIndex;
1348 curSource = &srcList[src];
1349 }
1350
1351 sent->srcAllocated = qtrue;
1352 sent->srcIndex = src;
1353
1354 sent->loopPriority = priority;
1355 sent->loopSfx = sfx;
1356
1357 // If this is not set then the looping sound is stopped.
1358 sent->loopAddedThisFrame = qtrue;
1359
1360 // UGH
1361 // These lines should be called via S_AL_SrcSetup, but we
1362 // can't call that yet as it buffers sfxes that may change
1363 // with subsequent calls to S_AL_SrcLoop
1364 curSource->entity = entityNum;
1365 curSource->isLooping = qtrue;
1366
1367 if( S_AL_HearingThroughEntity( entityNum ) )
1368 {
1369 curSource->local = qtrue;
1370
1371 VectorClear(sorigin);
1372
1373 qalSourcefv(curSource->alSource, AL_POSITION, sorigin);
1374 qalSourcefv(curSource->alSource, AL_VELOCITY, vec3_origin);
1375 }
1376 else
1377 {
1378 curSource->local = qfalse;
1379
1380 if(origin)
1381 VectorCopy(origin, sorigin);
1382 else
1383 VectorCopy(sent->origin, sorigin);
1384
1385 S_AL_SanitiseVector(sorigin);
1386
1387 VectorCopy(sorigin, curSource->loopSpeakerPos);
1388
1389 if(velocity)
1390 {
1391 VectorCopy(velocity, svelocity);
1392 S_AL_SanitiseVector(svelocity);
1393 }
1394 else
1395 VectorClear(svelocity);
1396
1397 qalSourcefv(curSource->alSource, AL_POSITION, (ALfloat *) sorigin);
1398 qalSourcefv(curSource->alSource, AL_VELOCITY, (ALfloat *) svelocity);
1399 }
1400}

1434/*
1435=================
1436S_AL_SrcUpdate
1437
1438Update state (move things around, manage sources, and so on)
1439=================
1440*/
1441static
1442void S_AL_SrcUpdate( void )
1443{
1444 int i;
1445 int entityNum;
1446 ALint state;
1447 src_t *curSource;
1448
1449 for(i = 0; i < srcCount; i++)
1450 {
1451 entityNum = srcList[i].entity;
1452 curSource = &srcList[i];
1453
1454 if(curSource->isLocked)
1455 continue;
1456
1457 if(!curSource->isActive)
1458 continue;
1459
1460 // Update source parameters
1461 if((s_alGain->modified) || (s_volume->modified))
1462 curSource->curGain = s_alGain->value * s_volume->value;
1463 if((s_alRolloff->modified) && (!curSource->local))
1464 qalSourcef(curSource->alSource, AL_ROLLOFF_FACTOR, s_alRolloff->value);
1465 if(s_alMinDistance->modified)
1466 qalSourcef(curSource->alSource, AL_REFERENCE_DISTANCE, s_alMinDistance->value);
1467
1468 if(curSource->isLooping)
1469 {
1470 sentity_t *sent = &entityList[ entityNum ];
1471
1472 // If a looping effect hasn't been touched this frame, pause or kill it
1473 if(sent->loopAddedThisFrame)
1474 {
1475 alSfx_t *curSfx;
1476
1477 // The sound has changed without an intervening removal
1478 if(curSource->isActive && !sent->startLoopingSound &&
1479 curSource->sfx != sent->loopSfx)
1480 {
1481 S_AL_NewLoopMaster(curSource, qtrue);
1482
1483 curSource->isPlaying = qfalse;
1484 qalSourceStop(curSource->alSource);
1485 qalSourcei(curSource->alSource, AL_BUFFER, 0);
1486 sent->startLoopingSound = qtrue;
1487 }
1488
1489 // The sound hasn't been started yet
1490 if(sent->startLoopingSound)
1491 {
1492 S_AL_SrcSetup(i, sent->loopSfx, sent->loopPriority,
1493 entityNum, -1, curSource->local);
1494 curSource->isLooping = qtrue;
1495
1496 knownSfx[curSource->sfx].loopCnt++;
1497 sent->startLoopingSound = qfalse;
1498 }
1499
1500 curSfx = &knownSfx[curSource->sfx];
1501
1502 S_AL_ScaleGain(curSource, curSource->loopSpeakerPos);
1503 if(!curSource->scaleGain)
1504 {
1505 if(curSource->isPlaying)
1506 {
1507 // Sound is mute, stop playback until we are in range again
1508 S_AL_NewLoopMaster(curSource, qfalse);
1509 qalSourceStop(curSource->alSource);
1510 curSource->isPlaying = qfalse;
1511 }
1512 else if(!curSfx->loopActiveCnt && curSfx->masterLoopSrc < 0)
1513 curSfx->masterLoopSrc = i;
1514
1515 continue;
1516 }
1517
1518 if(!curSource->isPlaying)
1519 {
1520 qalSourcei(curSource->alSource, AL_LOOPING, AL_TRUE);
1521 curSource->isPlaying = qtrue;
1522 qalSourcePlay(curSource->alSource);
1523
1524 if(curSource->priority == SRCPRI_AMBIENT)
1525 {
1526 // If there are other ambient looping sources with the same sound,
1527 // make sure the sound of these sources are in sync.
1528
1529 if(curSfx->loopActiveCnt)
1530 {
1531 int offset, error;
1532
1533 // we already have a master loop playing, get buffer position.
1534 S_AL_ClearError(qfalse);
1535 qalGetSourcei(srcList[curSfx->masterLoopSrc].alSource, AL_SAMPLE_OFFSET, &offset);
1536 if((error = qalGetError()) != AL_NO_ERROR)
1537 {
1538 if(error != AL_INVALID_ENUM)
1539 {
1540 Com_Printf(S_COLOR_YELLOW "WARNING: Cannot get sample offset from source %d: "
1541 "%s\n", i, S_AL_ErrorMsg(error));
1542 }
1543 }
1544 else
1545 qalSourcei(curSource->alSource, AL_SAMPLE_OFFSET, offset);
1546 }
1547 else if(curSfx->loopCnt && curSfx->masterLoopSrc >= 0)
1548 {
1549 float secofs;
1550
1551 src_t *master = &srcList[curSfx->masterLoopSrc];
1552 // This loop sound used to be played, but all sources are stopped. Use last sample position/time
1553 // to calculate offset so the player thinks the sources continued playing while they were inaudible.
1554
1555 if(master->lastTimePos >= 0)
1556 {
1557 secofs = master->lastTimePos + (Sys_Milliseconds() - master->lastSampleTime) / 1000.0f;
1558 secofs = fmodf(secofs, (float) curSfx->info.samples / curSfx->info.rate);
1559
1560 qalSourcef(curSource->alSource, AL_SEC_OFFSET, secofs);
1561 }
1562
1563 // I be the master now
1564 curSfx->masterLoopSrc = i;
1565 }
1566 else
1567 curSfx->masterLoopSrc = i;
1568 }
1569 else if(curSource->lastTimePos >= 0)
1570 {
1571 float secofs;
1572
1573 // For unsynced loops (SRCPRI_ENTITY) just carry on playing as if the sound was never stopped
1574
1575 secofs = curSource->lastTimePos + (Sys_Milliseconds() - curSource->lastSampleTime) / 1000.0f;
1576 secofs = fmodf(secofs, (float) curSfx->info.samples / curSfx->info.rate);
1577 qalSourcef(curSource->alSource, AL_SEC_OFFSET, secofs);
1578 }
1579
1580 curSfx->loopActiveCnt++;
1581 }
1582
1583 // Update locality
1584 if(curSource->local)
1585 {
1586 qalSourcei(curSource->alSource, AL_SOURCE_RELATIVE, AL_TRUE);
1587 qalSourcef(curSource->alSource, AL_ROLLOFF_FACTOR, 0.0f);
1588 }
1589 else
1590 {
1591 qalSourcei(curSource->alSource, AL_SOURCE_RELATIVE, AL_FALSE);
1592 qalSourcef(curSource->alSource, AL_ROLLOFF_FACTOR, s_alRolloff->value);
1593 }
1594
1595 }
1596 else if(curSource->priority == SRCPRI_AMBIENT)
1597 {
1598 if(curSource->isPlaying)
1599 {
1600 S_AL_NewLoopMaster(curSource, qfalse);
1601 qalSourceStop(curSource->alSource);
1602 curSource->isPlaying = qfalse;
1603 }
1604 }
1605 else
1606 S_AL_SrcKill(i);
1607
1608 continue;
1609 }
1610
1611 if(!curSource->isStream)
1612 {
1613 // Check if it's done, and flag it
1614 qalGetSourcei(curSource->alSource, AL_SOURCE_STATE, &state);
1615 if(state == AL_STOPPED)
1616 {
1617 curSource->isPlaying = qfalse;
1618 S_AL_SrcKill(i);
1619 continue;
1620 }
1621 }
1622
1623 // Query relativity of source, don't move if it's true
1624 qalGetSourcei(curSource->alSource, AL_SOURCE_RELATIVE, &state);
1625
1626 // See if it needs to be moved
1627 if(curSource->isTracking && !state)
1628 {
1629 qalSourcefv(curSource->alSource, AL_POSITION, entityList[entityNum].origin);
1630 S_AL_ScaleGain(curSource, entityList[entityNum].origin);
1631 }
1632 }
1633}

1759/*
1760=================
1761S_AL_RawSamples
1762=================
1763*/
1764static
1765void S_AL_RawSamples(int stream, int samples, int rate, int width, int channels, const byte *data, float volume, int entityNum)
1766{
1767 int numBuffers;
1768 ALuint buffer;
1769 ALuint format;
1770
1771 if ((stream < 0) || (stream >= MAX_RAW_STREAMS))
1772 return;
1773
1774 format = S_AL_Format( width, channels );
1775
1776 // Create the streamSource if necessary
1777 if(streamSourceHandles[stream] == -1)
1778 {
1779 S_AL_AllocateStreamChannel(stream, entityNum);
1780
1781 // Failed?
1782 if(streamSourceHandles[stream] == -1)
1783 {
1784 Com_Printf( S_COLOR_RED "ERROR: Can't allocate streaming streamSource\n");
1785 return;
1786 }
1787 }
1788
1789 qalGetSourcei(streamSources[stream], AL_BUFFERS_QUEUED, &numBuffers);
1790
1791 if (numBuffers == MAX_STREAM_BUFFERS)
1792 {
1793 Com_DPrintf(S_COLOR_RED"WARNING: Steam dropping raw samples, reached MAX_STREAM_BUFFERS\n");
1794 return;
1795 }
1796
1797 // Allocate a new AL buffer if needed
1798 if (numBuffers == streamNumBuffers[stream])
1799 {
1800 ALuint oldBuffers[MAX_STREAM_BUFFERS];
1801 int i;
1802
1803 if (!S_AL_GenBuffers(1, &buffer, "stream"))
1804 return;
1805
1806 Com_Memcpy(oldBuffers, &streamBuffers[stream], sizeof (oldBuffers));
1807
1808 // Reorder buffer array in order of oldest to newest
1809 for ( i = 0; i < streamNumBuffers[stream]; ++i )
1810 streamBuffers[stream][i] = oldBuffers[(streamBufIndex[stream] + i) % streamNumBuffers[stream]];
1811
1812 // Add the new buffer to end
1813 streamBuffers[stream][streamNumBuffers[stream]] = buffer;
1814 streamBufIndex[stream] = streamNumBuffers[stream];
1815 streamNumBuffers[stream]++;
1816 }
1817
1818 // Select next buffer in loop
1819 buffer = streamBuffers[stream][ streamBufIndex[stream] ];
1820 streamBufIndex[stream] = (streamBufIndex[stream] + 1) % streamNumBuffers[stream];
1821
1822 // Fill buffer
1823 qalBufferData(buffer, format, (ALvoid *)data, (samples * width * channels), rate);
1824
1825 // Shove the data onto the streamSource
1826 qalSourceQueueBuffers(streamSources[stream], 1, &buffer);
1827
1828 if(entityNum < 0)
1829 {
1830 // Volume
1831 S_AL_Gain (streamSources[stream], volume * s_volume->value * s_alGain->value);
1832 }
1833
1834 // Start stream
1835 if(!streamPlaying[stream])
1836 {
1837 qalSourcePlay( streamSources[stream] );
1838 streamPlaying[stream] = qtrue;
1839 }
1840}

1842/*
1843=================
1844S_AL_StreamUpdate
1845=================
1846*/
1847static
1848void S_AL_StreamUpdate( int stream )
1849{
1850 int numBuffers;
1851 ALint state;
1852
1853 if ((stream < 0) || (stream >= MAX_RAW_STREAMS))
1854 return;
1855
1856 if(streamSourceHandles[stream] == -1)
1857 return;
1858
1859 // Un-queue any buffers
1860 qalGetSourcei( streamSources[stream], AL_BUFFERS_PROCESSED, &numBuffers );
1861 while( numBuffers-- )
1862 {
1863 ALuint buffer;
1864 qalSourceUnqueueBuffers(streamSources[stream], 1, &buffer);
1865 }
1866
1867 // Start the streamSource playing if necessary
1868 qalGetSourcei( streamSources[stream], AL_BUFFERS_QUEUED, &numBuffers );
1869
1870 qalGetSourcei(streamSources[stream], AL_SOURCE_STATE, &state);
1871 if(state == AL_STOPPED)
1872 {
1873 streamPlaying[stream] = qfalse;
1874
1875 // If there are no buffers queued up, release the streamSource
1876 if( !numBuffers )
1877 S_AL_FreeStreamChannel( stream );
1878 }
1879
1880 if( !streamPlaying[stream] && numBuffers )
1881 {
1882 qalSourcePlay( streamSources[stream] );
1883 streamPlaying[stream] = qtrue;
1884 }
1885}

2486{
2487#ifdef USE_OPENAL
2488 const char* device = NULL;
2489 const char* inputdevice = NULL;
2490 int i;
2491
2492 if( !si ) {
2493 return qfalse;
2494 }
2495
2496 for (i = 0; i < MAX_RAW_STREAMS; i++) {
2497 streamSourceHandles[i] = -1;
2498 streamPlaying[i] = qfalse;
2499 streamSources[i] = 0;
2500 streamNumBuffers[i] = 0;
2501 streamBufIndex[i] = 0;
2502 }
2503
2504 // New console variables
2505 s_alPrecache = Cvar_Get( "s_alPrecache", "1", CVAR_ARCHIVE );
2506 s_alGain = Cvar_Get( "s_alGain", "1.0", CVAR_ARCHIVE );
2507 s_alSources = Cvar_Get( "s_alSources", "96", CVAR_ARCHIVE );
2508 s_alDopplerFactor = Cvar_Get( "s_alDopplerFactor", "1.0", CVAR_ARCHIVE );
2509 s_alDopplerSpeed = Cvar_Get( "s_alDopplerSpeed", "9000", CVAR_ARCHIVE );
2510 s_alMinDistance = Cvar_Get( "s_alMinDistance", "120", CVAR_CHEAT );
2511 s_alMaxDistance = Cvar_Get("s_alMaxDistance", "1024", CVAR_CHEAT);
2512 s_alRolloff = Cvar_Get( "s_alRolloff", "2", CVAR_CHEAT);
2513 s_alGraceDistance = Cvar_Get("s_alGraceDistance", "512", CVAR_CHEAT);
2514
2515 s_alDriver = Cvar_Get( "s_alDriver", ALDRIVER_DEFAULT, CVAR_ARCHIVE | CVAR_LATCH | CVAR_PROTECTED );
2516
2517 s_alInputDevice = Cvar_Get( "s_alInputDevice", "", CVAR_ARCHIVE | CVAR_LATCH );
2518 s_alDevice = Cvar_Get("s_alDevice", "", CVAR_ARCHIVE | CVAR_LATCH);
2519
2520 // Load QAL
2521 if( !QAL_Init( s_alDriver->string ) )
2522 {
2523 Com_Printf( "Failed to load library: \"%s\".\n", s_alDriver->string );
2524 if( !Q_stricmp( s_alDriver->string, ALDRIVER_DEFAULT ) || !QAL_Init( ALDRIVER_DEFAULT ) ) {
2525 return qfalse;
2526 }
2527 }
2528
2529 device = s_alDevice->string;
2530 if(device && !*device)
2531 device = NULL;
2532
2533 inputdevice = s_alInputDevice->string;
2534 if(inputdevice && !*inputdevice)
2535 inputdevice = NULL;
2536
2537
2538 // Device enumeration support
2539 enumeration_all_ext = qalcIsExtensionPresent(NULL, "ALC_ENUMERATE_ALL_EXT");
2540 enumeration_ext = qalcIsExtensionPresent(NULL, "ALC_ENUMERATION_EXT");
2541
2542 if(enumeration_ext || enumeration_all_ext)
2543 {
2544 char devicenames[16384] = "";
2545 const char *devicelist;
2546#ifdef _WIN32
2547 const char *defaultdevice;
2548#endif
2549 int curlen;
2550
2551 // get all available devices + the default device name.
2552 if(enumeration_all_ext)
2553 {
2554 devicelist = qalcGetString(NULL, ALC_ALL_DEVICES_SPECIFIER);
2555#ifdef _WIN32

2558 }
2559 else

2562 devicelist = qalcGetString(NULL, ALC_DEVICE_SPECIFIER);
2563#ifdef _WIN32
2564 defaultdevice = qalcGetString(NULL, ALC_DEFAULT_DEVICE_SPECIFIER);
2565#endif
2566 enumeration_ext = qtrue;
2567 }
2568
2569#ifdef _WIN32
2570 // check whether the default device is generic hardware. If it is, change to
2571 // Generic Software as that one works more reliably with various sound systems.
2572 // If it's not, use OpenAL's default selection as we don't want to ignore
2573 // native hardware acceleration.
2574 if(!device && defaultdevice && !strcmp(defaultdevice, "Generic Hardware"))
2575 device = "Generic Software";
2576#endif
2577
2578 // dump a list of available devices to a cvar for the user to see.
2579
2580 if(devicelist)
2581 {

2583 {

2585 Q_strcat(devicenames, sizeof(devicenames), "\n");

2596 {

2599 }

2602 {

2606 }

2609 alContext = qalcCreateContext( alDevice, NULL );
2610 if( !alContext )
2611 {
2612 QAL_Shutdown( );

2615 return qfalse;
2616 }
2617 qalcMakeContextCurrent( alContext );
2618

2620 S_AL_BufferInit( );
2621 S_AL_SrcInit( );
2622

2629 // !!! FIXME: some of these alcCaptureOpenDevice() values should be cvars.

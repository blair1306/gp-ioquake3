110static void *OpenALLib = NULL;
111
112static qboolean alinit_fail = qfalse;
113
114/*
115=================
116GPA
117=================
118*/
119static void *GPA(char *str)
120{
121 void *rv;
122
123 rv = Sys_LoadFunction(OpenALLib, str);
124 if(!rv)
125 {
126 Com_Printf( " Can't load symbol %s\n", str);
127 alinit_fail = qtrue;
128 return NULL;
129 }
130 else
131 {
132 Com_DPrintf( " Loaded symbol %s (%p)\n", str, rv);
133 return rv;
134 }
135}

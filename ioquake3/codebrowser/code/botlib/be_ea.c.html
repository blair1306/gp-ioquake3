214//===========================================================================
215void EA_Jump(int client)
216{
217 bot_input_t *bi;
218
219 bi = &botinputs[client];
220
221 if (bi->actionflags & ACTION_JUMPEDLASTFRAME)
222 {
223 bi->actionflags &= ~ACTION_JUMP;
224 } //end if
225 else
226 {
227 bi->actionflags |= ACTION_JUMP;
228 } //end if
229} //end of the function EA_Jump

379// Parameter: -
380// Returns: -
381// Changes Globals: -
382//===========================================================================
383void EA_Move(int client, vec3_t dir, float speed)
384{
385 bot_input_t *bi;
386
387 bi = &botinputs[client];
388
389 VectorCopy(dir, bi->dir);
390 //cap speed
391 if (speed > MAX_USERMOVE) speed = MAX_USERMOVE;
392 else if (speed < -MAX_USERMOVE) speed = -MAX_USERMOVE;
393 bi->speed = speed;
394} //end of the function EA_Move

432//===========================================================================
433//
434// Parameter: -
435// Returns: -
436// Changes Globals: -
437//===========================================================================
438void EA_ResetInput(int client)
439{
440 bot_input_t *bi;
441 int jumped = qfalse;
442
443 bi = &botinputs[client];
444
445 bi->thinktime = 0;
446 VectorClear(bi->dir);
447 bi->speed = 0;
448 jumped = bi->actionflags & ACTION_JUMP;
449 bi->actionflags = 0;
450 if (jumped) bi->actionflags |= ACTION_JUMPEDLASTFRAME;
451} //end of the function EA_ResetInput

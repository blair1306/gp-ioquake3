176void PC_PushIndent(source_t *source, int type, int skip)
177{
178 indent_t *indent;
179
180 indent = (indent_t *) GetMemory(sizeof(indent_t));
181 indent->type = type;

185 indent->next = source->indentstack;
186 source->indentstack = indent;
187} //end of the function PC_PushIndent

194void PC_PopIndent(source_t *source, int *type, int *skip)
195{
196 indent_t *indent;
197
198 *type = 0;
199 *skip = 0;
200
201 indent = source->indentstack;
202 if (!indent) return;
203
204 //must be an indent from the current script
205 if (source->indentstack->script != source->scriptstack) return;
206
207 *type = indent->type;
208 *skip = indent->skip;
209 source->indentstack = source->indentstack->next;
210 source->skip -= indent->skip;
211 FreeMemory(indent);
212} //end of the function PC_PopIndent

304int PC_ReadSourceToken(source_t *source, token_t *token)
305{
306 token_t *t;
307 script_t *script;
308 int type, skip;
309
310 //if there's no token already available
311 while(!source->tokens)
312 {
313 //if there's a token to read from the script
314 if (PS_ReadToken(source->scriptstack, token)) return qtrue;
315 //if at the end of the script
316 if (EndOfScript(source->scriptstack))
317 {
318 //remove all indents of the script
319 while(source->indentstack &&
320 source->indentstack->script == source->scriptstack)
321 {
322 SourceWarning(source, "missing #endif");
323 PC_PopIndent(source, &type, &skip);
324 } //end if
325 } //end if
326 //if this was the initial script
327 if (!source->scriptstack->next) return qfalse;
328 //remove the script and return to the last one
329 script = source->scriptstack;
330 source->scriptstack = source->scriptstack->next;
331 FreeScript(script);
332 } //end while
333 //copy the already available token
334 Com_Memcpy(token, source->tokens, sizeof(token_t));
335 //free the read token
336 t = source->tokens;
337 source->tokens = source->tokens->next;
338 PC_FreeToken(t);
339 return qtrue;
340} //end of the function PC_ReadSourceToken

362int PC_ReadDefineParms(source_t *source, define_t *define, token_t **parms, int maxparms)
363{
364 token_t token, *t, *last;
365 int i, done, lastcomma, numparms, indent;
366
367 if (!PC_ReadSourceToken(source, &token))
368 {
369 SourceError(source, "define %s missing parms", define->name);
370 return qfalse;
371 } //end if
372 //
373 if (define->numparms > maxparms)
374 {
375 SourceError(source, "define with more than %d parameters", maxparms);
376 return qfalse;
377 } //end if
378 //
379 for (i = 0; i < define->numparms; i++) parms[i] = NULL;
380 //if no leading "("
381 if (strcmp(token.string, "("))
382 {
383 PC_UnreadSourceToken(source, &token);
384 SourceError(source, "define %s missing parms", define->name);
385 return qfalse;
386 } //end if
387 //read the define parameters
388 for (done = 0, numparms = 0, indent = 0; !done;)
389 {
390 if (numparms >= maxparms)
391 {
392 SourceError(source, "define %s with too many parms", define->name);
393 return qfalse;
394 } //end if
395 if (numparms >= define->numparms)
396 {
397 SourceWarning(source, "define %s has too many parms", define->name);
398 return qfalse;
399 } //end if
400 parms[numparms] = NULL;
401 lastcomma = 1;
402 last = NULL;
403 while(!done)
404 {
405 //
406 if (!PC_ReadSourceToken(source, &token))
407 {
408 SourceError(source, "define %s incomplete", define->name);
409 return qfalse;
410 } //end if
411 //
412 if (!strcmp(token.string, ","))
413 {
414 if (indent <= 0)
415 {
416 if (lastcomma) SourceWarning(source, "too many comma's");
417 break;
418 } //end if
419 } //end if
420 lastcomma = 0;
421 //
422 if (!strcmp(token.string, "("))
423 {
424 indent++;
425 continue;
426 } //end if
427 else if (!strcmp(token.string, ")"))
428 {
429 if (--indent <= 0)
430 {
431 if (!parms[define->numparms-1])
432 {
433 SourceWarning(source, "too few define parms");
434 } //end if
435 done = 1;
436 break;
437 } //end if
438 } //end if
439 //
440 if (numparms < define->numparms)
441 {
442 //
443 t = PC_CopyToken(&token);
444 t->next = NULL;
445 if (last) last->next = t;
446 else parms[numparms] = t;
447 last = t;
448 } //end if
449 } //end while
450 numparms++;
451 } //end for
452 return qtrue;
453} //end of the function PC_ReadDefineParms

704int PC_ExpandBuiltinDefine(source_t *source, token_t *deftoken, define_t *define,
705 token_t **firsttoken, token_t **lasttoken)
706{
707 token_t *token;
708 time_t t;
709
710 char *curtime;
711
712 token = PC_CopyToken(deftoken);
713 switch(define->builtin)
714 {
715 case BUILTIN_LINE:
716 {
717 sprintf(token->string, "%d", deftoken->line);
718#ifdef NUMBERVALUE
719 token->intvalue = deftoken->line;
720 token->floatvalue = deftoken->line;
721#endif //NUMBERVALUE
722 token->type = TT_NUMBER;
723 token->subtype = TT_DECIMAL | TT_INTEGER;
724 *firsttoken = token;
725 *lasttoken = token;
726 break;
727 } //end case
728 case BUILTIN_FILE:
729 {
730 strcpy(token->string, source->scriptstack->filename);
731 token->type = TT_NAME;
732 token->subtype = strlen(token->string);
733 *firsttoken = token;
734 *lasttoken = token;
735 break;
736 } //end case
737 case BUILTIN_DATE:
738 {
739 t = time(NULL);
740 curtime = ctime(&t);
741 strcpy(token->string, "\"");
742 strncat(token->string, curtime+4, 7);
743 strncat(token->string+7, curtime+20, 4);
744 strcat(token->string, "\"");
745 free(curtime);
746 token->type = TT_NAME;
747 token->subtype = strlen(token->string);
748 *firsttoken = token;
749 *lasttoken = token;
750 break;
751 } //end case
752 case BUILTIN_TIME:
753 {
754 t = time(NULL);
755 curtime = ctime(&t);
756 strcpy(token->string, "\"");
757 strncat(token->string, curtime+11, 8);
758 strcat(token->string, "\"");
759 free(curtime);
760 token->type = TT_NAME;
761 token->subtype = strlen(token->string);
762 *firsttoken = token;
763 *lasttoken = token;
764 break;
765 } //end case
766 case BUILTIN_STDC:
767 default:
768 {
769 *firsttoken = NULL;
770 *lasttoken = NULL;
771 break;
772 } //end case
773 } //end switch
774 return qtrue;
775} //end of the function PC_ExpandBuiltinDefine

782int PC_ExpandDefine(source_t *source, token_t *deftoken, define_t *define,

791 {

793 } //end if
794 //if the define has parameters
795 if (define->numparms)
796 {
797 if (!PC_ReadDefineParms(source, define, parms, MAX_DEFINEPARMS)) return qfalse;
798#ifdef DEBUG_EVAL
799 for (i = 0; i < define->numparms; i++)
800 {

806 } //end for
807#endif //DEBUG_EVAL
808 } //end if
809 //empty list at first
810 first = NULL;
811 last = NULL;
812 //create a list with tokens of the expanded define
813 for (dt = define->tokens; dt; dt = dt->next)
814 {
815 parmnum = -1;
816 //if the token is a name, it could be a define parameter
817 if (dt->type == TT_NAME)
818 {
819 parmnum = PC_FindDefineParm(define, dt->string);
820 } //end if
821 //if it is a define parameter
822 if (parmnum >= 0)
823 {
824 for (pt = parms[parmnum]; pt; pt = pt->next)
825 {
826 t = PC_CopyToken(pt);
827 //add the token to the list
828 t->next = NULL;
829 if (last) last->next = t;
830 else first = t;
831 last = t;
832 } //end for
833 } //end if
834 else
835 {
836 //if stringizing operator
837 if (dt->string[0] == '#' && dt->string[1] == '\0')
838 {
839 //the stringizing operator must be followed by a define parameter
840 if (dt->next) parmnum = PC_FindDefineParm(define, dt->next->string);
841 else parmnum = -1;
842 //
843 if (parmnum >= 0)
844 {
845 //step over the stringizing operator
846 dt = dt->next;
847 //stringize the define parameter tokens
848 if (!PC_StringizeTokens(parms[parmnum], &token))
849 {
850 SourceError(source, "can't stringize tokens");
851 return qfalse;
852 } //end if
853 t = PC_CopyToken(&token);
854 } //end if
855 else
856 {
857 SourceWarning(source, "stringizing operator without define parameter");
858 continue;
859 } //end if
860 } //end if
861 else
862 {
863 t = PC_CopyToken(dt);
864 } //end else
865 //add the token to the list
866 t->next = NULL;
867 if (last) last->next = t;
868 else first = t;
869 last = t;
870 } //end else
871 } //end for
872 //check for the merging operator
873 for (t = first; t; )
874 {
875 if (t->next)
876 {
877 //if the merging operator
878 if (t->next->string[0] == '#' && t->next->string[1] == '#')
879 {
880 t1 = t;
881 t2 = t->next->next;
882 if (t2)
883 {
884 if (!PC_MergeTokens(t1, t2))
885 {
886 SourceError(source, "can't merge %s with %s", t1->string, t2->string);
887 return qfalse;
888 } //end if
889 PC_FreeToken(t1->next);
890 t1->next = t2->next;
891 if (t2 == last) last = t1;
892 PC_FreeToken(t2);
893 continue;
894 } //end if
895 } //end if
896 } //end if
897 t = t->next;
898 } //end for
899 //store the first and last token of the list
900 *firsttoken = first;
901 *lasttoken = last;
902 //free all the parameter tokens
903 for (i = 0; i < define->numparms; i++)
904 {
905 for (pt = parms[i]; pt; pt = nextpt)
906 {
907 nextpt = pt->next;
908 PC_FreeToken(pt);
909 } //end for
910 } //end for
911 //
912 return qtrue;
913} //end of the function PC_ExpandDefine

1194 //check if the define already exists
1195#if DEFINEHASHING

1197#else

1199#endif //DEFINEHASHING
1200 if (define)
1201 {
1202 if (define->flags & DEFINE_FIXED)
1203 {
1204 SourceError(source, "can't redefine %s", token.string);
1205 return qfalse;
1206 } //end if
1207 SourceWarning(source, "redefinition of %s", token.string);
1208 //unread the define name before executing the #undef directive
1209 PC_UnreadSourceToken(source, &token);
1210 if (!PC_Directive_undef(source)) return qfalse;
1211 } //end if

1213 define = (define_t *) GetMemory(sizeof(define_t));

1220#else //DEFINEHASHING

1223#endif //DEFINEHASHING
1224 //if nothing is defined, just return
1225 if (!PC_ReadLine(source, &token)) return qtrue;
1226 //if it is a define with parameters
1227 if (!PC_WhiteSpaceBeforeToken(&token) && !strcmp(token.string, "("))
1228 {
1229 //read the define parameters
1230 last = NULL;

1232 {

1234 {
1235 if (!PC_ReadLine(source, &token))
1236 {
1237 SourceError(source, "expected define parameter");

1239 } //end if
1240 //if it isn't a name
1241 if (token.type != TT_NAME)
1242 {
1243 SourceError(source, "invalid define parameter");
1244 return qfalse;
1245 } //end if
1246 //
1247 if (PC_FindDefineParm(define, token.string) >= 0)
1248 {
1249 SourceError(source, "two the same define parameters");
1250 return qfalse;
1251 } //end if
1252 //add the define parm

1254 PC_ClearTokenWhiteSpace(t);
1255 t->next = NULL;
1256 if (last) last->next = t;
1257 else define->parms = t;
1258 last = t;
1259 define->numparms++;
1260 //read next token
1261 if (!PC_ReadLine(source, &token))
1262 {
1263 SourceError(source, "define parameters not terminated");
1264 return qfalse;
1265 } //end if
1266 //
1267 if (!strcmp(token.string, ")")) break;
1268 //then it must be a comma
1269 if (strcmp(token.string, ","))
1270 {
1271 SourceError(source, "define not terminated");
1272 return qfalse;
1273 } //end if
1274 } //end while
1275 } //end if
1276 if (!PC_ReadLine(source, &token)) return qtrue;
1277 } //end if
1278 //read the defined stuff
1279 last = NULL;

1281 {

1287 } //end if

1293 } while(PC_ReadLine(source, &token));

1295 if (last)

1299 !strcmp(last->string, "##"))
1300 {
1301 SourceError(source, "define with misplaced ##");
1302 return qfalse;
1303 } //end if
1304 } //end if
1305 return qtrue;

2756 }
2757 } //end if

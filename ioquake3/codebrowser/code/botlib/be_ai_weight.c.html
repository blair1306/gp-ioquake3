59int ReadValue(source_t *source, float *value)
60{
61 token_t token;
62
63 if (!PC_ExpectAnyToken(source, &token)) return qfalse;
64 if (!strcmp(token.string, "-"))
65 {
66 SourceWarning(source, "negative value set to zero");
67
68 if(!PC_ExpectAnyToken(source, &token))
69 {
70 SourceError(source, "Missing return value");
71 return qfalse;
72 }
73 }
74
75 if (token.type != TT_NUMBER)
76 {
77 SourceError(source, "invalid return value %s", token.string);
78 return qfalse;
79 }
80
81 *value = token.floatvalue;
82 return qtrue;
83} //end of the function ReadValue

90int ReadFuzzyWeight(source_t *source, fuzzyseperator_t *fs)
91{
92 if (PC_CheckTokenString(source, "balance"))
93 {
94 fs->type = WT_BALANCE;
95 if (!PC_ExpectTokenString(source, "(")) return qfalse;
96 if (!ReadValue(source, &fs->weight)) return qfalse;
97 if (!PC_ExpectTokenString(source, ",")) return qfalse;
98 if (!ReadValue(source, &fs->minweight)) return qfalse;
99 if (!PC_ExpectTokenString(source, ",")) return qfalse;
100 if (!ReadValue(source, &fs->maxweight)) return qfalse;
101 if (!PC_ExpectTokenString(source, ")")) return qfalse;
102 } //end if
103 else
104 {
105 fs->type = 0;
106 if (!ReadValue(source, &fs->weight)) return qfalse;
107 fs->minweight = fs->weight;
108 fs->maxweight = fs->weight;
109 } //end if
110 if (!PC_ExpectTokenString(source, ";")) return qfalse;
111 return qtrue;
112} //end of the function ReadFuzzyWeight

160fuzzyseperator_t *ReadFuzzySeperators_r(source_t *source)
161{
162 int newindent, index, def, founddefault;
163 token_t token;
164 fuzzyseperator_t *fs, *lastfs, *firstfs;
165
166 founddefault = qfalse;
167 firstfs = NULL;
168 lastfs = NULL;
169 if (!PC_ExpectTokenString(source, "(")) return NULL;
170 if (!PC_ExpectTokenType(source, TT_NUMBER, TT_INTEGER, &token)) return NULL;
171 index = token.intvalue;
172 if (!PC_ExpectTokenString(source, ")")) return NULL;
173 if (!PC_ExpectTokenString(source, "{")) return NULL;
174 if (!PC_ExpectAnyToken(source, &token)) return NULL;
175 do
176 {
177 def = !strcmp(token.string, "default");
178 if (def || !strcmp(token.string, "case"))
179 {
180 fs = (fuzzyseperator_t *) GetClearedMemory(sizeof(fuzzyseperator_t));
181 fs->index = index;
182 if (lastfs) lastfs->next = fs;
183 else firstfs = fs;
184 lastfs = fs;
185 if (def)
186 {
187 if (founddefault)
188 {
189 SourceError(source, "switch already has a default");
190 FreeFuzzySeperators_r(firstfs);
191 return NULL;
192 } //end if
193 fs->value = MAX_INVENTORYVALUE;
194 founddefault = qtrue;
195 } //end if
196 else
197 {
198 if (!PC_ExpectTokenType(source, TT_NUMBER, TT_INTEGER, &token))
199 {
200 FreeFuzzySeperators_r(firstfs);
201 return NULL;
202 } //end if
203 fs->value = token.intvalue;
204 } //end else
205 if (!PC_ExpectTokenString(source, ":") || !PC_ExpectAnyToken(source, &token))
206 {
207 FreeFuzzySeperators_r(firstfs);
208 return NULL;
209 } //end if
210 newindent = qfalse;
211 if (!strcmp(token.string, "{"))
212 {
213 newindent = qtrue;
214 if (!PC_ExpectAnyToken(source, &token))
215 {
216 FreeFuzzySeperators_r(firstfs);
217 return NULL;
218 } //end if
219 } //end if
220 if (!strcmp(token.string, "return"))
221 {
222 if (!ReadFuzzyWeight(source, fs))
223 {
224 FreeFuzzySeperators_r(firstfs);
225 return NULL;
226 } //end if
227 } //end if
228 else if (!strcmp(token.string, "switch"))
229 {
230 fs->child = ReadFuzzySeperators_r(source);
231 if (!fs->child)
232 {
233 FreeFuzzySeperators_r(firstfs);
234 return NULL;
235 } //end if
236 } //end else if
237 else
238 {
239 SourceError(source, "invalid name %s", token.string);
240 return NULL;
241 } //end else
242 if (newindent)
243 {
244 if (!PC_ExpectTokenString(source, "}"))
245 {
246 FreeFuzzySeperators_r(firstfs);
247 return NULL;
248 } //end if
249 } //end if
250 } //end if
251 else
252 {
253 FreeFuzzySeperators_r(firstfs);
254 SourceError(source, "invalid name %s", token.string);
255 return NULL;
256 } //end else
257 if (!PC_ExpectAnyToken(source, &token))
258 {
259 FreeFuzzySeperators_r(firstfs);
260 return NULL;
261 } //end if
262 } while(strcmp(token.string, "}"));
263 //
264 if (!founddefault)
265 {
266 SourceWarning(source, "switch without default");
267 fs = (fuzzyseperator_t *) GetClearedMemory(sizeof(fuzzyseperator_t));
268 fs->index = index;
269 fs->value = MAX_INVENTORYVALUE;
270 fs->weight = 0;
271 fs->next = NULL;
272 fs->child = NULL;
273 if (lastfs) lastfs->next = fs;
274 else firstfs = fs;
275 } //end if
276 //
277 return firstfs;
278} //end of the function ReadFuzzySeperators_r

285weightconfig_t *ReadWeightConfig(char *filename)
286{
287 int newindent, avail = 0, n;
288 token_t token;
289 source_t *source;
290 fuzzyseperator_t *fs;
291 weightconfig_t *config = NULL;
292#ifdef DEBUG
293 int starttime;
294
295 starttime = Sys_MilliSeconds();
296#endif //DEBUG
297
298 if (!LibVarGetValue("bot_reloadcharacters"))
299 {
300 avail = -1;
301 for( n = 0; n < MAX_WEIGHT_FILES; n++ )
302 {
303 config = weightFileList[n];
304 if( !config )
305 {
306 if( avail == -1 )
307 {
308 avail = n;
309 } //end if
310 continue;
311 } //end if
312 if( strcmp( filename, config->filename ) == 0 )
313 {
314 //botimport.Print( PRT_MESSAGE, "retained %s\n", filename );
315 return config;
316 } //end if
317 } //end for
318
319 if( avail == -1 )
320 {
321 botimport.Print( PRT_ERROR, "weightFileList was full trying to load %s\n", filename );
322 return NULL;
323 } //end if
324 } //end if
325
326 PC_SetBaseFolder(BOTFILESBASEFOLDER);
327 source = LoadSourceFile(filename);
328 if (!source)
329 {
330 botimport.Print(PRT_ERROR, "counldn't load %s\n", filename);
331 return NULL;
332 } //end if
333 //
334 config = (weightconfig_t *) GetClearedMemory(sizeof(weightconfig_t));
335 config->numweights = 0;
336 Q_strncpyz( config->filename, filename, sizeof(config->filename) );
337 //parse the item config file
338 while(PC_ReadToken(source, &token))
339 {
340 if (!strcmp(token.string, "weight"))
341 {
342 if (config->numweights >= MAX_WEIGHTS)
343 {
344 SourceWarning(source, "too many fuzzy weights");
345 break;
346 } //end if
347 if (!PC_ExpectTokenType(source, TT_STRING, 0, &token))
348 {
349 FreeWeightConfig(config);
350 FreeSource(source);
351 return NULL;
352 } //end if
353 StripDoubleQuotes(token.string);
354 config->weights[config->numweights].name = (char *) GetClearedMemory(strlen(token.string) + 1);
355 strcpy(config->weights[config->numweights].name, token.string);
356 if (!PC_ExpectAnyToken(source, &token))
357 {
358 FreeWeightConfig(config);
359 FreeSource(source);
360 return NULL;
361 } //end if
362 newindent = qfalse;
363 if (!strcmp(token.string, "{"))
364 {
365 newindent = qtrue;
366 if (!PC_ExpectAnyToken(source, &token))
367 {
368 FreeWeightConfig(config);
369 FreeSource(source);
370 return NULL;
371 } //end if
372 } //end if
373 if (!strcmp(token.string, "switch"))
374 {
375 fs = ReadFuzzySeperators_r(source);
376 if (!fs)
377 {
378 FreeWeightConfig(config);
379 FreeSource(source);
380 return NULL;
381 } //end if
382 config->weights[config->numweights].firstseperator = fs;
383 } //end if
384 else if (!strcmp(token.string, "return"))
385 {
386 fs = (fuzzyseperator_t *) GetClearedMemory(sizeof(fuzzyseperator_t));
387 fs->index = 0;
388 fs->value = MAX_INVENTORYVALUE;
389 fs->next = NULL;
390 fs->child = NULL;
391 if (!ReadFuzzyWeight(source, fs))
392 {
393 FreeMemory(fs);
394 FreeWeightConfig(config);
395 FreeSource(source);
396 return NULL;
397 } //end if
398 config->weights[config->numweights].firstseperator = fs;
399 } //end else if
400 else
401 {
402 SourceError(source, "invalid name %s", token.string);
403 FreeWeightConfig(config);
404 FreeSource(source);
405 return NULL;
406 } //end else
407 if (newindent)
408 {
409 if (!PC_ExpectTokenString(source, "}"))
410 {
411 FreeWeightConfig(config);
412 FreeSource(source);
413 return NULL;
414 } //end if
415 } //end if
416 config->numweights++;
417 } //end if
418 else
419 {
420 SourceError(source, "invalid name %s", token.string);
421 FreeWeightConfig(config);
422 FreeSource(source);
423 return NULL;
424 } //end else
425 } //end while
426 //free the source at the end of a pass
427 FreeSource(source);
428 //if the file was located in a pak file
429 botimport.Print(PRT_MESSAGE, "loaded %s\n", filename);
430#ifdef DEBUG
431 if (botDeveloper)
432 {
433 botimport.Print(PRT_MESSAGE, "weights loaded in %d msec\n", Sys_MilliSeconds() - starttime);
434 } //end if
435#endif //DEBUG
436 //
437 if (!LibVarGetValue("bot_reloadcharacters"))
438 {
439 weightFileList[avail] = config;
440 } //end if
441 //
442 return config;
443} //end of the function ReadWeightConfig

477qboolean WriteFuzzySeperators_r(FILE *fp, fuzzyseperator_t *fs, int indent)
478{
479 if (!WriteIndent(fp, indent)) return qfalse;
480 if (fprintf(fp, "switch(%d)\n", fs->index) < 0) return qfalse;
481 if (!WriteIndent(fp, indent)) return qfalse;
482 if (fprintf(fp, "{\n") < 0) return qfalse;
483 indent++;
484 do
485 {
486 if (!WriteIndent(fp, indent)) return qfalse;
487 if (fs->next)
488 {
489 if (fprintf(fp, "case %d:", fs->value) < 0) return qfalse;
490 } //end if
491 else
492 {
493 if (fprintf(fp, "default:") < 0) return qfalse;
494 } //end else
495 if (fs->child)
496 {
497 if (fprintf(fp, "\n") < 0) return qfalse;
498 if (!WriteIndent(fp, indent)) return qfalse;
499 if (fprintf(fp, "{\n") < 0) return qfalse;
500 if (!WriteFuzzySeperators_r(fp, fs->child, indent + 1)) return qfalse;
501 if (!WriteIndent(fp, indent)) return qfalse;
502 if (fs->next)
503 {
504 if (fprintf(fp, "} //end case\n") < 0) return qfalse;
505 } //end if
506 else
507 {
508 if (fprintf(fp, "} //end default\n") < 0) return qfalse;
509 } //end else
510 } //end if
511 else
512 {
513 if (!WriteFuzzyWeight(fp, fs)) return qfalse;
514 } //end else
515 fs = fs->next;
516 } while(fs);
517 indent--;
518 if (!WriteIndent(fp, indent)) return qfalse;
519 if (fprintf(fp, "} //end switch\n") < 0) return qfalse;
520 return qtrue;
521} //end of the function WriteItemFuzzyWeights_r

582float FuzzyWeight_r(int *inventory, fuzzyseperator_t *fs)
583{
584 float scale, w1, w2;
585
586 if (inventory[fs->index] < fs->value)
587 {
588 if (fs->child) return FuzzyWeight_r(inventory, fs->child);
589 else return fs->weight;
590 } //end if
591 else if (fs->next)
592 {
593 if (inventory[fs->index] < fs->next->value)
594 {
595 //first weight
596 if (fs->child) w1 = FuzzyWeight_r(inventory, fs->child);
597 else w1 = fs->weight;
598 //second weight
599 if (fs->next->child) w2 = FuzzyWeight_r(inventory, fs->next->child);
600 else w2 = fs->next->weight;
601 //the scale factor
602 if(fs->next->value == MAX_INVENTORYVALUE) // is fs->next the default case?
603 return w2; // can't interpolate, return default weight
604 else
605 scale = (float) (inventory[fs->index] - fs->value) / (fs->next->value - fs->value);
606 //scale between the two weights
607 return (1 - scale) * w1 + scale * w2;
608 } //end if
609 return FuzzyWeight_r(inventory, fs->next);
610 } //end else if
611 return fs->weight;
612} //end of the function FuzzyWeight_r

619float FuzzyWeightUndecided_r(int *inventory, fuzzyseperator_t *fs)
620{
621 float scale, w1, w2;
622
623 if (inventory[fs->index] < fs->value)
624 {
625 if (fs->child) return FuzzyWeightUndecided_r(inventory, fs->child);
626 else return fs->minweight + random() * (fs->maxweight - fs->minweight);
627 } //end if
628 else if (fs->next)
629 {
630 if (inventory[fs->index] < fs->next->value)
631 {
632 //first weight
633 if (fs->child) w1 = FuzzyWeightUndecided_r(inventory, fs->child);
634 else w1 = fs->minweight + random() * (fs->maxweight - fs->minweight);
635 //second weight
636 if (fs->next->child) w2 = FuzzyWeight_r(inventory, fs->next->child);
637 else w2 = fs->next->minweight + random() * (fs->next->maxweight - fs->next->minweight);
638 //the scale factor
639 if(fs->next->value == MAX_INVENTORYVALUE) // is fs->next the default case?
640 return w2; // can't interpolate, return default weight
641 else
642 scale = (float) (inventory[fs->index] - fs->value) / (fs->next->value - fs->value);
643 //scale between the two weights
644 return (1 - scale) * w1 + scale * w2;
645 } //end if
646 return FuzzyWeightUndecided_r(inventory, fs->next);
647 } //end else if
648 return fs->weight;
649} //end of the function FuzzyWeightUndecided_r

718void EvolveFuzzySeperator_r(fuzzyseperator_t *fs)
719{
720 if (fs->child)
721 {
722 EvolveFuzzySeperator_r(fs->child);
723 } //end if
724 else if (fs->type == WT_BALANCE)
725 {
726 //every once in a while an evolution leap occurs, mutation
727 if (random() < 0.01) fs->weight += crandom() * (fs->maxweight - fs->minweight);
728 else fs->weight += crandom() * (fs->maxweight - fs->minweight) * 0.5;
729 //modify bounds if necesary because of mutation
730 if (fs->weight < fs->minweight) fs->minweight = fs->weight;
731 else if (fs->weight > fs->maxweight) fs->maxweight = fs->weight;
732 } //end else if
733 if (fs->next) EvolveFuzzySeperator_r(fs->next);
734} //end of the function EvolveFuzzySeperator_r

842 fuzzyseperator_t *fsout)
843{
844 if (fs1->child)
845 {
846 if (!fs2->child || !fsout->child)
847 {
848 botimport.Print(PRT_ERROR, "cannot interbreed weight configs, unequal child\n");

850 } //end if
851 if (!InterbreedFuzzySeperator_r(fs2->child, fs2->child, fsout->child))

854 } //end if
855 } //end if
856 else if (fs1->type == WT_BALANCE)
857 {
858 if (fs2->type != WT_BALANCE || fsout->type != WT_BALANCE)
859 {
860 botimport.Print(PRT_ERROR, "cannot interbreed weight configs, unequal balance\n");
861 return qfalse;
862 } //end if
863 fsout->weight = (fs1->weight + fs2->weight) / 2;
864 if (fsout->weight > fsout->maxweight) fsout->maxweight = fsout->weight;
865 if (fsout->weight > fsout->minweight) fsout->minweight = fsout->weight;
866 } //end else if
867 if (fs1->next)
868 {
869 if (!fs2->next || !fsout->next)
870 {
871 botimport.Print(PRT_ERROR, "cannot interbreed weight configs, unequal next\n");
872 return qfalse;
873 } //end if
874 if (!InterbreedFuzzySeperator_r(fs1->next, fs2->next, fsout->next))
875 {
876 return qfalse;
877 } //end if
878 } //end if
879 return qtrue;
880} //end of the function InterbreedFuzzySeperator_r

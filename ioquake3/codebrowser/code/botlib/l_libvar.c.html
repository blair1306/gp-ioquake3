36//list with library variables
37libvar_t *libvarlist = NULL;
38
39//===========================================================================
40//
41// Parameter: -
42// Returns: -
43// Changes Globals: -
44//===========================================================================
45float LibVarStringValue(const char *string)
46{
47 int dotfound = 0;
48 float value = 0;
49
50 while(*string)
51 {
52 if (*string < '0' || *string > '9')
53 {
54 if (dotfound || *string != '.')
55 {
56 return 0;
57 } //end if
58 else
59 {
60 dotfound = 10;
61 string++;
62 } //end if
63 } //end if
64 if (dotfound)
65 {
66 value = value + (float) (*string - '0') / (float) dotfound;
67 dotfound *= 10;
68 } //end if
69 else
70 {
71 value = value * 10.0 + (float) (*string - '0');
72 } //end else
73 string++;
74 } //end while
75 return value;
76} //end of the function LibVarStringValue
77//===========================================================================
78//
79// Parameter: -
80// Returns: -
81// Changes Globals: -
82//===========================================================================
83libvar_t *LibVarAlloc(const char *var_name)
84{
85 libvar_t *v;
86
87 v = (libvar_t *) GetMemory(sizeof(libvar_t));
88 Com_Memset(v, 0, sizeof(libvar_t));
89 v->name = (char *) GetMemory(strlen(var_name)+1);
90 strcpy(v->name, var_name);
91 //add the variable in the list
92 v->next = libvarlist;
93 libvarlist = v;
94 return v;
95} //end of the function LibVarAlloc
96//===========================================================================
97//
98// Parameter: -
99// Returns: -
100// Changes Globals: -
101//===========================================================================
102void LibVarDeAlloc(libvar_t *v)
103{
104 if (v->string) FreeMemory(v->string);
105 FreeMemory(v->name);
106 FreeMemory(v);
107} //end of the function LibVarDeAlloc
108//===========================================================================
109//
110// Parameter: -
111// Returns: -
112// Changes Globals: -
113//===========================================================================
114void LibVarDeAllocAll(void)
115{
116 libvar_t *v;
117
118 for (v = libvarlist; v; v = libvarlist)
119 {
120 libvarlist = libvarlist->next;
121 LibVarDeAlloc(v);
122 } //end for
123 libvarlist = NULL;
124} //end of the function LibVarDeAllocAll
125//===========================================================================
126//
127// Parameter: -
128// Returns: -
129// Changes Globals: -
130//===========================================================================
131libvar_t *LibVarGet(const char *var_name)
132{
133 libvar_t *v;
134
135 for (v = libvarlist; v; v = v->next)
136 {
137 if (!Q_stricmp(v->name, var_name))
138 {
139 return v;
140 } //end if
141 } //end for
142 return NULL;
143} //end of the function LibVarGet

184//===========================================================================
185//
186// Parameter: -
187// Returns: -
188// Changes Globals: -
189//===========================================================================
190libvar_t *LibVar(const char *var_name, const char *value)
191{
192 libvar_t *v;
193 v = LibVarGet(var_name);
194 if (v) return v;
195 //create new variable
196 v = LibVarAlloc(var_name);
197 //variable string
198 v->string = (char *) GetMemory(strlen(value) + 1);
199 strcpy(v->string, value);
200 //the value
201 v->value = LibVarStringValue(v->string);
202 //variable is modified
203 v->modified = qtrue;
204 //
205 return v;
206} //end of the function LibVar

233//===========================================================================
234//
235// Parameter: -
236// Returns: -
237// Changes Globals: -
238//===========================================================================
239void LibVarSet(const char *var_name, const char *value)
240{
241 libvar_t *v;
242
243 v = LibVarGet(var_name);
244 if (v)
245 {
246 FreeMemory(v->string);
247 } //end if
248 else
249 {
250 v = LibVarAlloc(var_name);
251 } //end else
252 //variable string
253 v->string = (char *) GetMemory(strlen(value) + 1);
254 strcpy(v->string, value);
255 //the value
256 v->value = LibVarStringValue(v->string);
257 //variable is modified
258 v->modified = qtrue;
259} //end of the function LibVarSet

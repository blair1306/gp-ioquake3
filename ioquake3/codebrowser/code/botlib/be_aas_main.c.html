105//
106// Parameter: -
107// Returns: -
108// Changes Globals: -
109//===========================================================================
110void AAS_ContinueInit(float time)
111{
112 //if no AAS file loaded
113 if (!aasworld.loaded) return;
114 //if AAS is already initialized
115 if (aasworld.initialized) return;
116 //calculate reachability, if not finished return
117 if (AAS_ContinueInitReachability(time)) return;
118 //initialize clustering for the new map
119 AAS_InitClustering();
120 //if reachability has been calculated and an AAS file should be written
121 //or there is a forced data optimization
122 if (aasworld.savefile || ((int)LibVarGetValue("forcewrite")))
123 {
124 //optimize the AAS data
125 if ((int)LibVarValue("aasoptimize", "0")) AAS_Optimize();
126 //save the AAS file
127 if (AAS_WriteAASFile(aasworld.filename))
128 {
129 botimport.Print(PRT_MESSAGE, "%s written successfully\n", aasworld.filename);
130 } //end if
131 else
132 {
133 botimport.Print(PRT_ERROR, "couldn't write %s\n", aasworld.filename);
134 } //end else
135 } //end if
136 //initialize the routing
137 AAS_InitRouting();
138 //at this point AAS is initialized
139 AAS_SetInitialized();
140} //end of the function AAS_ContinueInit
141//===========================================================================
142// called at the start of every frame
143//
144// Parameter: -
145// Returns: -
146// Changes Globals: -
147//===========================================================================
148int AAS_StartFrame(float time)
149{
150 aasworld.time = time;
151 //unlink all entities that were not updated last frame
152 AAS_UnlinkInvalidEntities();
153 //invalidate the entities
154 AAS_InvalidateEntities();
155 //initialize AAS
156 AAS_ContinueInit(time);
157 //
158 aasworld.frameroutingupdates = 0;
159 //
160 if (botDeveloper)
161 {
162 if (LibVarGetValue("showcacheupdates"))
163 {
164 AAS_RoutingInfo();
165 LibVarSet("showcacheupdates", "0");
166 } //end if
167 if (LibVarGetValue("showmemoryusage"))
168 {
169 PrintUsedMemorySize();
170 LibVarSet("showmemoryusage", "0");
171 } //end if
172 if (LibVarGetValue("memorydump"))
173 {
174 PrintMemoryLabels();
175 LibVarSet("memorydump", "0");
176 } //end if
177 } //end if
178 //
179 if (saveroutingcache->value)
180 {
181 AAS_WriteRouteCache();
182 LibVarSet("saveroutingcache", "0");
183 } //end if
184 //
185 aasworld.numframes++;
186 return BLERR_NOERROR;
187} //end of the function AAS_StartFrame

204void AAS_ProjectPointOntoVector( vec3_t point, vec3_t vStart, vec3_t vEnd, vec3_t vProj )
205{
206 vec3_t pVec, vec;
207
208 VectorSubtract( point, vStart, pVec );
209 VectorSubtract( vEnd, vStart, vec );
210 VectorNormalize( vec );
211 // project onto the directional vector for this segment
212 VectorMA( vStart, DotProduct( pVec, vec ), vec, vProj );
213} //end of the function AAS_ProjectPointOntoVector

220int AAS_LoadFiles(const char *mapname)
221{
222 int errnum;
223 char aasfile[MAX_PATH];
224// char bspfile[MAX_PATH];
225
226 strcpy(aasworld.mapname, mapname);
227 //NOTE: first reset the entity links into the AAS areas and BSP leaves
228 // the AAS link heap and BSP link heap are reset after respectively the
229 // AAS file and BSP file are loaded
230 AAS_ResetEntityLinks();
231 // load bsp info
232 AAS_LoadBSPFile();
233
234 //load the aas file
235 Com_sprintf(aasfile, MAX_PATH, "maps/%s.aas", mapname);
236 errnum = AAS_LoadAASFile(aasfile);
237 if (errnum != BLERR_NOERROR)
238 return errnum;
239
240 botimport.Print(PRT_MESSAGE, "loaded %s\n", aasfile);
241 Q_strncpyz(aasworld.filename, aasfile, sizeof(aasworld.filename));
242 return BLERR_NOERROR;
243} //end of the function AAS_LoadFiles
244//===========================================================================
245// called everytime a map changes
246//
247// Parameter: -
248// Returns: -
249// Changes Globals: -
250//===========================================================================
251int AAS_LoadMap(const char *mapname)
252{
253 int errnum;
254
255 //if no mapname is provided then the string indexes are updated
256 if (!mapname)
257 {
258 return 0;
259 } //end if
260 //
261 aasworld.initialized = qfalse;
262 //NOTE: free the routing caches before loading a new map because
263 // to free the caches the old number of areas, number of clusters
264 // and number of areas in a clusters must be available
265 AAS_FreeRoutingCaches();
266 //load the map
267 errnum = AAS_LoadFiles(mapname);
268 if (errnum != BLERR_NOERROR)
269 {
270 aasworld.loaded = qfalse;
271 return errnum;
272 } //end if
273 //
274 AAS_InitSettings();
275 //initialize the AAS link heap for the new map
276 AAS_InitAASLinkHeap();
277 //initialize the AAS linked entities for the new map
278 AAS_InitAASLinkedEntities();
279 //initialize reachability for the new map
280 AAS_InitReachability();
281 //initialize the alternative routing
282 AAS_InitAlternativeRouting();
283 //everything went ok
284 return 0;
285} //end of the function AAS_LoadMap

293int AAS_Setup(void)
294{
295 aasworld.maxclients = (int) LibVarValue("maxclients", "128");
296 aasworld.maxentities = (int) LibVarValue("maxentities", "1024");
297 // as soon as it's set to 1 the routing cache will be saved
298 saveroutingcache = LibVar("saveroutingcache", "0");
299 //allocate memory for the entities
300 if (aasworld.entities) FreeMemory(aasworld.entities);
301 aasworld.entities = (aas_entity_t *) GetClearedHunkMemory(aasworld.maxentities * sizeof(aas_entity_t));
302 //invalidate all the entities
303 AAS_InvalidateEntities();
304 //force some recalculations
305 //LibVarSet("forceclustering", "1"); //force clustering calculation
306 //LibVarSet("forcereachability", "1"); //force reachability calculation
307 aasworld.numframes = 0;
308 return BLERR_NOERROR;
309} //end of the function AAS_Setup
310//===========================================================================
311//
312// Parameter: -
313// Returns: -
314// Changes Globals: -
315//===========================================================================
316void AAS_Shutdown(void)
317{
318 AAS_ShutdownAlternativeRouting();
319 //
320 AAS_DumpBSPData();
321 //free routing caches
322 AAS_FreeRoutingCaches();
323 //free aas link heap
324 AAS_FreeAASLinkHeap();
325 //free aas linked entities
326 AAS_FreeAASLinkedEntities();
327 //free the aas data
328 AAS_DumpAASData();
329 //free the entities
330 if (aasworld.entities) FreeMemory(aasworld.entities);
331 //clear the aasworld structure
332 Com_Memset(&aasworld, 0, sizeof(aas_t));
333 //aas has not been initialized
334 aasworld.initialized = qfalse;
335 //NOTE: as soon as a new .bsp file is loaded the .bsp file memory is
336 // freed and reallocated, so there's no need to free that memory here
337 //print shutdown
338 botimport.Print(PRT_MESSAGE, "AAS shutdown.\n");
339} //end of the function AAS_Shutdown

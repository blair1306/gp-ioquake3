46//structure to link entities to areas and areas to entities
47typedef struct aas_link_s
48{
49 int entnum;
50 int areanum;
51 struct aas_link_s *next_ent, *prev_ent;
52 struct aas_link_s *next_area, *prev_area;
53} aas_link_t;
54
55//structure to link entities to leaves and leaves to entities
56typedef struct bsp_link_s
57{
58 int entnum;
59 int leafnum;
60 struct bsp_link_s *next_ent, *prev_ent;
61 struct bsp_link_s *next_leaf, *prev_leaf;
62} bsp_link_t;

72} bsp_entdata_t;

126#define CACHETYPE_PORTAL 0

129//routing cache
130typedef struct aas_routingcache_s
131{
132 byte type; //portal or area cache
133 float time; //last time accessed or updated
134 int size; //size of the routing cache
135 int cluster; //cluster the cache is for
136 int areanum; //area the cache is created for
137 vec3_t origin; //origin within the area
138 float starttraveltime; //travel time to start with
139 int travelflags; //combinations of the travel flags
140 struct aas_routingcache_s *prev, *next;
141 struct aas_routingcache_s *time_prev, *time_next;
142 unsigned char *reachabilities; //reachabilities used for routing
143 unsigned short int traveltimes[1]; //travel time for every area (variable sized)
144} aas_routingcache_t;
145
146//fields for the routing algorithm
147typedef struct aas_routingupdate_s
148{
149 int cluster;
150 int areanum; //area number of the update
151 vec3_t start; //start point the area was entered
152 unsigned short int tmptraveltime; //temporary travel time
153 unsigned short int *areatraveltimes; //travel times within the area
154 qboolean inlist; //true if the update is in the list
155 struct aas_routingupdate_s *next;
156 struct aas_routingupdate_s *prev;
157} aas_routingupdate_t;
158
159//reversed reachability link
160typedef struct aas_reversedlink_s
161{
162 int linknum; //the aas_areareachability_t
163 int areanum; //reachable from this area
164 struct aas_reversedlink_s *next; //next link
165} aas_reversedlink_t;
166
167//reversed area reachability
168typedef struct aas_reversedreachability_s
169{
170 int numlinks;
171 aas_reversedlink_t *first;
172} aas_reversedreachability_t;
173
174//areas a reachability goes through
175typedef struct aas_reachabilityareas_s
176{
177 int firstarea, numareas;
178} aas_reachabilityareas_t;

180typedef struct aas_s
181{
182 int loaded; //true when an AAS file is loaded
183 int initialized; //true when AAS has been initialized
184 int savefile; //set true when file should be saved
185 int bspchecksum;
186 //current time
187 float time;
188 int numframes;
189 //name of the aas file
190 char filename[MAX_PATH];
191 char mapname[MAX_PATH];
192 //bounding boxes
193 int numbboxes;
194 aas_bbox_t *bboxes;
195 //vertexes
196 int numvertexes;
197 aas_vertex_t *vertexes;
198 //planes
199 int numplanes;
200 aas_plane_t *planes;
201 //edges
202 int numedges;
203 aas_edge_t *edges;
204 //edge index
205 int edgeindexsize;
206 aas_edgeindex_t *edgeindex;
207 //faces
208 int numfaces;
209 aas_face_t *faces;
210 //face index
211 int faceindexsize;
212 aas_faceindex_t *faceindex;
213 //convex areas
214 int numareas;
215 aas_area_t *areas;
216 //convex area settings
217 int numareasettings;
218 aas_areasettings_t *areasettings;
219 //reachablity list
220 int reachabilitysize;
221 aas_reachability_t *reachability;
222 //nodes of the bsp tree
223 int numnodes;
224 aas_node_t *nodes;
225 //cluster portals
226 int numportals;
227 aas_portal_t *portals;
228 //cluster portal index
229 int portalindexsize;
230 aas_portalindex_t *portalindex;
231 //clusters
232 int numclusters;
233 aas_cluster_t *clusters;
234 //
235 int numreachabilityareas;
236 float reachabilitytime;
237 //enities linked in the areas
238 aas_link_t *linkheap; //heap with link structures
239 int linkheapsize; //size of the link heap
240 aas_link_t *freelinks; //first free link
241 aas_link_t **arealinkedentities; //entities linked into areas
242 //entities
243 int maxentities;
244 int maxclients;
245 aas_entity_t *entities;
246 //index to retrieve travel flag for a travel type
247 int travelflagfortype[MAX_TRAVELTYPES];
248 //travel flags for each area based on contents
249 int *areacontentstravelflags;
250 //routing update
251 aas_routingupdate_t *areaupdate;
252 aas_routingupdate_t *portalupdate;
253 //number of routing updates during a frame (reset every frame)
254 int frameroutingupdates;
255 //reversed reachability links
256 aas_reversedreachability_t *reversedreachability;
257 //travel times within the areas
258 unsigned short ***areatraveltimes;
259 //array of size numclusters with cluster cache
260 aas_routingcache_t ***clusterareacache;
261 aas_routingcache_t **portalcache;
262 //cache list sorted on time
263 aas_routingcache_t *oldestcache; // start of cache list sorted on time
264 aas_routingcache_t *newestcache; // end of cache list sorted on time
265 //maximum travel time through portal areas
266 int *portalmaxtraveltimes;
267 //areas the reachabilities go through
268 int *reachabilityareaindex;
269 aas_reachabilityareas_t *reachabilityareas;
270} aas_t;

33#define MAX_STRINGFIELD 80
34//field types
35#define FT_CHAR 1 // char
36#define FT_INT 2 // int
37#define FT_FLOAT 3 // float
38#define FT_STRING 4 // char [MAX_STRINGFIELD]
39#define FT_STRUCT 6 // struct (sub structure)
40//type only mask
41#define FT_TYPE 0x00FF // only type, clear subtype
42//sub types
43#define FT_ARRAY 0x0100 // array of type
44#define FT_BOUNDED 0x0200 // bounded value
45#define FT_UNSIGNED 0x0400
46
47//structure field definition
48typedef struct fielddef_s
49{
50 char *name; //name of the field
51 int offset; //offset in the structure
52 int type; //type of the field
53 //type specific fields
54 int maxarray; //maximum array size
55 float floatmin, floatmax; //float min and max
56 struct structdef_s *substruct; //sub structure
57} fielddef_t;
58
59//structure definition
60typedef struct structdef_s
61{
62 int size;
63 fielddef_t *fields;
64} structdef_t;
65
66//read a structure from a script
67int ReadStructure(source_t *source, structdef_t *def, char *structure);
68//write a structure to a file
69int WriteStructure(FILE *fp, structdef_t *def, char *structure);
70//writes indents
71int WriteIndent(FILE *fp, int indent);
72//writes a float without traling zeros
73int WriteFloat(FILE *fp, float value);

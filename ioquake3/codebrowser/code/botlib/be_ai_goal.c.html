267itemconfig_t *LoadItemConfig(char *filename)
268{
269 int max_iteminfo;
270 token_t token;
271 char path[MAX_PATH];
272 source_t *source;
273 itemconfig_t *ic;
274 iteminfo_t *ii;
275
276 max_iteminfo = (int) LibVarValue("max_iteminfo", "256");
277 if (max_iteminfo < 0)
278 {
279 botimport.Print(PRT_ERROR, "max_iteminfo = %d\n", max_iteminfo);
280 max_iteminfo = 256;
281 LibVarSet( "max_iteminfo", "256" );
282 }
283
284 Q_strncpyz(path, filename, sizeof(path));
285 PC_SetBaseFolder(BOTFILESBASEFOLDER);
286 source = LoadSourceFile( path );
287 if( !source ) {
288 botimport.Print( PRT_ERROR, "counldn't load %s\n", path );
289 return NULL;
290 } //end if
291 //initialize item config
292 ic = (itemconfig_t *) GetClearedHunkMemory(sizeof(itemconfig_t) +
293 max_iteminfo * sizeof(iteminfo_t));
294 ic->iteminfo = (iteminfo_t *) ((char *) ic + sizeof(itemconfig_t));
295 ic->numiteminfo = 0;
296 //parse the item config file
297 while(PC_ReadToken(source, &token))
298 {
299 if (!strcmp(token.string, "iteminfo"))
300 {
301 if (ic->numiteminfo >= max_iteminfo)
302 {
303 SourceError(source, "more than %d item info defined", max_iteminfo);
304 FreeMemory(ic);
305 FreeSource(source);
306 return NULL;
307 } //end if
308 ii = &ic->iteminfo[ic->numiteminfo];
309 Com_Memset(ii, 0, sizeof(iteminfo_t));
310 if (!PC_ExpectTokenType(source, TT_STRING, 0, &token))
311 {
312 FreeMemory(ic);
313 FreeSource(source);
314 return NULL;
315 } //end if
316 StripDoubleQuotes(token.string);
317 Q_strncpyz(ii->classname, token.string, sizeof(ii->classname));
318 if (!ReadStructure(source, &iteminfo_struct, (char *) ii))
319 {
320 FreeMemory(ic);
321 FreeSource(source);
322 return NULL;
323 } //end if
324 ii->number = ic->numiteminfo;
325 ic->numiteminfo++;
326 } //end if
327 else
328 {
329 SourceError(source, "unknown definition %s", token.string);
330 FreeMemory(ic);
331 FreeSource(source);
332 return NULL;
333 } //end else
334 } //end while
335 FreeSource(source);
336 //
337 if (!ic->numiteminfo) botimport.Print(PRT_WARNING, "no item info loaded\n");
338 botimport.Print(PRT_MESSAGE, "loaded %s\n", path);
339 return ic;
340} //end of the function LoadItemConfig

475void BotInitInfoEntities(void)
476{
477 char classname[MAX_EPAIRKEY];
478 maplocation_t *ml;
479 campspot_t *cs;
480 int ent, numlocations, numcampspots;
481
482 BotFreeInfoEntities();
483 //
484 numlocations = 0;
485 numcampspots = 0;
486 for (ent = AAS_NextBSPEntity(0); ent; ent = AAS_NextBSPEntity(ent))
487 {
488 if (!AAS_ValueForBSPEpairKey(ent, "classname", classname, MAX_EPAIRKEY)) continue;
489
490 //map locations
491 if (!strcmp(classname, "target_location"))
492 {
493 ml = (maplocation_t *) GetClearedMemory(sizeof(maplocation_t));
494 AAS_VectorForBSPEpairKey(ent, "origin", ml->origin);
495 AAS_ValueForBSPEpairKey(ent, "message", ml->name, sizeof(ml->name));
496 ml->areanum = AAS_PointAreaNum(ml->origin);
497 ml->next = maplocations;
498 maplocations = ml;
499 numlocations++;
500 } //end if
501 //camp spots
502 else if (!strcmp(classname, "info_camp"))
503 {
504 cs = (campspot_t *) GetClearedMemory(sizeof(campspot_t));
505 AAS_VectorForBSPEpairKey(ent, "origin", cs->origin);
506 //cs->origin[2] += 16;
507 AAS_ValueForBSPEpairKey(ent, "message", cs->name, sizeof(cs->name));
508 AAS_FloatForBSPEpairKey(ent, "range", &cs->range);
509 AAS_FloatForBSPEpairKey(ent, "weight", &cs->weight);
510 AAS_FloatForBSPEpairKey(ent, "wait", &cs->wait);
511 AAS_FloatForBSPEpairKey(ent, "random", &cs->random);
512 cs->areanum = AAS_PointAreaNum(cs->origin);
513 if (!cs->areanum)
514 {
515 botimport.Print(PRT_MESSAGE, "camp spot at %1.1f %1.1f %1.1f in solid\n", cs->origin[0], cs->origin[1], cs->origin[2]);
516 FreeMemory(cs);
517 continue;
518 } //end if
519 cs->next = campspots;
520 campspots = cs;
521 //AAS_DrawPermanentCross(cs->origin, 4, LINECOLOR_YELLOW);
522 numcampspots++;
523 } //end else if
524 } //end for
525 if (botDeveloper)
526 {
527 botimport.Print(PRT_MESSAGE, "%d map locations\n", numlocations);
528 botimport.Print(PRT_MESSAGE, "%d camp spots\n", numcampspots);
529 } //end if
530} //end of the function BotInitInfoEntities

537void BotInitLevelItems(void)
538{
539 int i, spawnflags, value;
540 char classname[MAX_EPAIRKEY];
541 vec3_t origin, end;
542 int ent, goalareanum;
543 itemconfig_t *ic;
544 levelitem_t *li;
545 bsp_trace_t trace;
546
547 //initialize the map locations and camp spots
548 BotInitInfoEntities();
549
550 //initialize the level item heap
551 InitLevelItemHeap();
552 levelitems = NULL;
553 numlevelitems = 0;
554 //
555 ic = itemconfig;
556 if (!ic) return;
557
558 //if there's no AAS file loaded
559 if (!AAS_Loaded()) return;
560
561 //validate the modelindexes of the item info
562 for (i = 0; i < ic->numiteminfo; i++)
563 {
564 if (!ic->iteminfo[i].modelindex)
565 {
566 Log_Write("item %s has modelindex 0", ic->iteminfo[i].classname);
567 } //end if
568 } //end for
569
570 for (ent = AAS_NextBSPEntity(0); ent; ent = AAS_NextBSPEntity(ent))
571 {
572 if (!AAS_ValueForBSPEpairKey(ent, "classname", classname, MAX_EPAIRKEY)) continue;
573 //
574 spawnflags = 0;
575 AAS_IntForBSPEpairKey(ent, "spawnflags", &spawnflags);
576 //
577 for (i = 0; i < ic->numiteminfo; i++)
578 {
579 if (!strcmp(classname, ic->iteminfo[i].classname)) break;
580 } //end for
581 if (i >= ic->numiteminfo)
582 {
583 Log_Write("entity %s unknown item\r\n", classname);
584 continue;
585 } //end if
586 //get the origin of the item
587 if (!AAS_VectorForBSPEpairKey(ent, "origin", origin))
588 {
589 botimport.Print(PRT_ERROR, "item %s without origin\n", classname);
590 continue;
591 } //end else
592 //
593 goalareanum = 0;
594 //if it is a floating item
595 if (spawnflags & 1)
596 {
597 //if the item is not floating in water
598 if (!(AAS_PointContents(origin) & CONTENTS_WATER))
599 {
600 VectorCopy(origin, end);
601 end[2] -= 32;
602 trace = AAS_Trace(origin, ic->iteminfo[i].mins, ic->iteminfo[i].maxs, end, -1, CONTENTS_SOLID|CONTENTS_PLAYERCLIP);
603 //if the item not near the ground
604 if (trace.fraction >= 1)
605 {
606 //if the item is not reachable from a jumppad
607 goalareanum = AAS_BestReachableFromJumpPadArea(origin, ic->iteminfo[i].mins, ic->iteminfo[i].maxs);
608 Log_Write("item %s reachable from jumppad area %d\r\n", ic->iteminfo[i].classname, goalareanum);
609 //botimport.Print(PRT_MESSAGE, "item %s reachable from jumppad area %d\r\n", ic->iteminfo[i].classname, goalareanum);
610 if (!goalareanum) continue;
611 } //end if
612 } //end if
613 } //end if
614
615 li = AllocLevelItem();
616 if (!li) return;
617 //
618 li->number = ++numlevelitems;
619 li->timeout = 0;
620 li->entitynum = 0;
621 //
622 li->flags = 0;
623 AAS_IntForBSPEpairKey(ent, "notfree", &value);
624 if (value) li->flags |= IFL_NOTFREE;
625 AAS_IntForBSPEpairKey(ent, "notteam", &value);
626 if (value) li->flags |= IFL_NOTTEAM;
627 AAS_IntForBSPEpairKey(ent, "notsingle", &value);
628 if (value) li->flags |= IFL_NOTSINGLE;
629 AAS_IntForBSPEpairKey(ent, "notbot", &value);
630 if (value) li->flags |= IFL_NOTBOT;
631 if (!strcmp(classname, "item_botroam"))
632 {
633 li->flags |= IFL_ROAM;
634 AAS_FloatForBSPEpairKey(ent, "weight", &li->weight);
635 } //end if
636 //if not a stationary item
637 if (!(spawnflags & 1))
638 {
639 if (!AAS_DropToFloor(origin, ic->iteminfo[i].mins, ic->iteminfo[i].maxs))
640 {
641 botimport.Print(PRT_MESSAGE, "%s in solid at (%1.1f %1.1f %1.1f)\n",
642 classname, origin[0], origin[1], origin[2]);
643 } //end if
644 } //end if
645 //item info of the level item
646 li->iteminfo = i;
647 //origin of the item
648 VectorCopy(origin, li->origin);
649 //
650 if (goalareanum)
651 {
652 li->goalareanum = goalareanum;
653 VectorCopy(origin, li->goalorigin);
654 } //end if
655 else
656 {
657 //get the item goal area and goal origin
658 li->goalareanum = AAS_BestReachableArea(origin,
659 ic->iteminfo[i].mins, ic->iteminfo[i].maxs,
660 li->goalorigin);
661 if (!li->goalareanum)
662 {
663 botimport.Print(PRT_MESSAGE, "%s not reachable for bots at (%1.1f %1.1f %1.1f)\n",
664 classname, origin[0], origin[1], origin[2]);
665 } //end if
666 } //end else
667 //
668 AddLevelItemToList(li);
669 } //end for
670 botimport.Print(PRT_MESSAGE, "found %d level items\n", numlevelitems);
671} //end of the function BotInitLevelItems

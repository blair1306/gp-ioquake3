123int AAS_AgainstLadder(vec3_t origin)
124{
125 int areanum, i, facenum, side;
126 vec3_t org;
127 aas_plane_t *plane;
128 aas_face_t *face;
129 aas_area_t *area;
130
131 VectorCopy(origin, org);
132 areanum = AAS_PointAreaNum(org);
133 if (!areanum)
134 {
135 org[0] += 1;
136 areanum = AAS_PointAreaNum(org);
137 if (!areanum)
138 {
139 org[1] += 1;
140 areanum = AAS_PointAreaNum(org);
141 if (!areanum)
142 {
143 org[0] -= 2;
144 areanum = AAS_PointAreaNum(org);
145 if (!areanum)
146 {
147 org[1] -= 2;
148 areanum = AAS_PointAreaNum(org);
149 } //end if
150 } //end if
151 } //end if
152 } //end if
153 //if in solid... wrrr shouldn't happen
154 if (!areanum) return qfalse;
155 //if not in a ladder area
156 if (!(aasworld.areasettings[areanum].areaflags & AREA_LADDER)) return qfalse;
157 //if a crouch only area
158 if (!(aasworld.areasettings[areanum].presencetype & PRESENCE_NORMAL)) return qfalse;
159 //
160 area = &aasworld.areas[areanum];
161 for (i = 0; i < area->numfaces; i++)
162 {
163 facenum = aasworld.faceindex[area->firstface + i];
164 side = facenum < 0;
165 face = &aasworld.faces[abs(facenum)];
166 //if the face isn't a ladder face
167 if (!(face->faceflags & FACE_LADDER)) continue;
168 //get the plane the face is in
169 plane = &aasworld.planes[face->planenum ^ side];
170 //if the origin is pretty close to the plane
171 if (fabsf(DotProduct(plane->normal, origin) - plane->dist) < 3)
172 {
173 if (AAS_PointInsideFace(abs(facenum), origin, 0.1f)) return qtrue;
174 } //end if
175 } //end for
176 return qfalse;
177} //end of the function AAS_AgainstLadder

185int AAS_OnGround(vec3_t origin, int presencetype, int passent)
186{
187 aas_trace_t trace;
188 vec3_t end, up = {0, 0, 1};
189 aas_plane_t *plane;
190
191 VectorCopy(origin, end);
192 end[2] -= 10;
193
194 trace = AAS_TraceClientBBox(origin, end, presencetype, passent);
195
196 //if in solid
197 if (trace.startsolid) return qfalse;
198 //if nothing hit at all
199 if (trace.fraction >= 1.0) return qfalse;
200 //if too far from the hit plane
201 if (origin[2] - trace.endpos[2] > 10) return qfalse;
202 //check if the plane isn't too steep
203 plane = AAS_PlaneFromNum(trace.planenum);
204 if (DotProduct(plane->normal, up) < aassettings.phys_maxsteepness) return qfalse;
205 //the bot is on the ground
206 return qtrue;
207} //end of the function AAS_OnGround

290float AAS_WeaponJumpZVelocity(vec3_t origin, float radiusdamage)
291{
292 vec3_t kvel, v, start, end, forward, right, viewangles, dir;
293 float mass, knockback, points;
294 vec3_t rocketoffset = {8, 8, -8};
295 vec3_t botmins = {-16, -16, -24};
296 vec3_t botmaxs = {16, 16, 32};
297 bsp_trace_t bsptrace;
298
299 //look down (90 degrees)
300 viewangles[PITCH] = 90;
301 viewangles[YAW] = 0;
302 viewangles[ROLL] = 0;
303 //get the start point shooting from
304 VectorCopy(origin, start);
305 start[2] += 8; //view offset Z
306 AngleVectors(viewangles, forward, right, NULL);
307 start[0] += forward[0] * rocketoffset[0] + right[0] * rocketoffset[1];
308 start[1] += forward[1] * rocketoffset[0] + right[1] * rocketoffset[1];
309 start[2] += forward[2] * rocketoffset[0] + right[2] * rocketoffset[1] + rocketoffset[2];
310 //end point of the trace
311 VectorMA(start, 500, forward, end);
312 //trace a line to get the impact point
313 bsptrace = AAS_Trace(start, NULL, NULL, end, 1, CONTENTS_SOLID);
314 //calculate the damage the bot will get from the rocket impact
315 VectorAdd(botmins, botmaxs, v);
316 VectorMA(origin, 0.5, v, v);
317 VectorSubtract(bsptrace.endpos, v, v);
318 //
319 points = radiusdamage - 0.5 * VectorLength(v);
320 if (points < 0) points = 0;
321 //the owner of the rocket gets half the damage
322 points *= 0.5;
323 //mass of the bot (p_client.c: PutClientInServer)
324 mass = 200;
325 //knockback is the same as the damage points
326 knockback = points;
327 //direction of the damage (from trace.endpos to bot origin)
328 VectorSubtract(origin, bsptrace.endpos, dir);
329 VectorNormalize(dir);
330 //damage velocity
331 VectorScale(dir, 1600.0 * (float)knockback / mass, kvel); //the rocket jump hack...
332 //rocket impact velocity + jump velocity
333 return kvel[2] + aassettings.phys_jumpvel;
334} //end of the function AAS_WeaponJumpZVelocity

364void AAS_Accelerate(vec3_t velocity, float frametime, vec3_t wishdir, float wishspeed, float accel)
365{
366 // q2 style
367 int i;
368 float addspeed, accelspeed, currentspeed;
369
370 currentspeed = DotProduct(velocity, wishdir);
371 addspeed = wishspeed - currentspeed;
372 if (addspeed <= 0) {
373 return;
374 }
375 accelspeed = accel*frametime*wishspeed;
376 if (accelspeed > addspeed) {
377 accelspeed = addspeed;
378 }
379
380 for (i=0 ; i<3 ; i++) {
381 velocity[i] += accelspeed*wishdir[i];
382 }
383} //end of the function AAS_Accelerate
384//===========================================================================
385// applies ground friction to the given velocity
386//
387// Parameter: -
388// Returns: -
389// Changes Globals: -
390//===========================================================================
391void AAS_ApplyFriction(vec3_t vel, float friction, float stopspeed,
392 float frametime)
393{
394 float speed, control, newspeed;
395
396 //horizontal speed
397 speed = sqrt(vel[0] * vel[0] + vel[1] * vel[1]);
398 if (speed)
399 {
400 control = speed < stopspeed ? stopspeed : speed;
401 newspeed = speed - frametime * control * friction;
402 if (newspeed < 0) newspeed = 0;
403 newspeed /= speed;
404 vel[0] *= newspeed;
405 vel[1] *= newspeed;
406 } //end if
407} //end of the function AAS_ApplyFriction

414int AAS_ClipToBBox(aas_trace_t *trace, vec3_t start, vec3_t end, int presencetype, vec3_t mins, vec3_t maxs)
415{
416 int i, j, side;
417 float front, back, frac, planedist;
418 vec3_t bboxmins, bboxmaxs, absmins, absmaxs, dir, mid;
419
420 AAS_PresenceTypeBoundingBox(presencetype, bboxmins, bboxmaxs);
421 VectorSubtract(mins, bboxmaxs, absmins);
422 VectorSubtract(maxs, bboxmins, absmaxs);
423 //
424 VectorCopy(end, trace->endpos);
425 trace->fraction = 1;
426 for (i = 0; i < 3; i++)
427 {
428 if (start[i] < absmins[i] && end[i] < absmins[i]) return qfalse;
429 if (start[i] > absmaxs[i] && end[i] > absmaxs[i]) return qfalse;
430 } //end for
431 //check bounding box collision
432 VectorSubtract(end, start, dir);
433 frac = 1;
434 for (i = 0; i < 3; i++)
435 {
436 //get plane to test collision with for the current axis direction
437 if (dir[i] > 0) planedist = absmins[i];
438 else planedist = absmaxs[i];
439 //calculate collision fraction
440 front = start[i] - planedist;
441 back = end[i] - planedist;
442 frac = front / (front-back);
443 //check if between bounding planes of next axis
444 side = i + 1;
445 if (side > 2) side = 0;
446 mid[side] = start[side] + dir[side] * frac;
447 if (mid[side] > absmins[side] && mid[side] < absmaxs[side])
448 {
449 //check if between bounding planes of next axis
450 side++;
451 if (side > 2) side = 0;
452 mid[side] = start[side] + dir[side] * frac;
453 if (mid[side] > absmins[side] && mid[side] < absmaxs[side])
454 {
455 mid[i] = planedist;
456 break;
457 } //end if
458 } //end if
459 } //end for
460 //if there was a collision
461 if (i != 3)
462 {
463 trace->startsolid = qfalse;
464 trace->fraction = frac;
465 trace->ent = 0;
466 trace->planenum = 0;
467 trace->area = 0;
468 trace->lastarea = 0;
469 //trace endpos
470 for (j = 0; j < 3; j++) trace->endpos[j] = start[j] + dir[j] * frac;
471 return qtrue;
472 } //end if
473 return qfalse;
474} //end of the function AAS_ClipToBBox
475//===========================================================================
476// predicts the movement
477// assumes regular bounding box sizes
478// NOTE: out of water jumping is not included
479// NOTE: grappling hook is not included
480//
481// Parameter: origin : origin to start with
482// presencetype : presence type to start with
483// velocity : velocity to start with
484// cmdmove : client command movement
485// cmdframes : number of frame cmdmove is valid
486// maxframes : maximum number of predicted frames
487// frametime : duration of one predicted frame
488// stopevent : events that stop the prediction
489// stopareanum : stop as soon as entered this area
490// Returns: aas_clientmove_t
491// Changes Globals: -
492//===========================================================================
493int AAS_ClientMovementPrediction(struct aas_clientmove_s *move,
494 int entnum, vec3_t origin,
495 int presencetype, int onground,
496 vec3_t velocity, vec3_t cmdmove,
497 int cmdframes,
498 int maxframes, float frametime,
499 int stopevent, int stopareanum,
500 vec3_t mins, vec3_t maxs, int visualize)
501{
502 float phys_friction, phys_stopspeed, phys_gravity, phys_waterfriction;
503 float phys_watergravity;
504 float phys_walkaccelerate, phys_airaccelerate, phys_swimaccelerate;
505 float phys_maxwalkvelocity, phys_maxcrouchvelocity, phys_maxswimvelocity;
506 float phys_maxstep, phys_maxsteepness, phys_jumpvel, friction;
507 float gravity, delta, maxvel, wishspeed, accelerate;
508 //float velchange, newvel;
509 //int ax;
510 int n, i, j, pc, step, swimming, crouch, event, jump_frame, areanum;
511 int areas[20], numareas;
512 vec3_t points[20];
513 vec3_t org, end, feet, start, stepend, lastorg, wishdir;
514 vec3_t frame_test_vel, old_frame_test_vel, left_test_vel;
515 vec3_t up = {0, 0, 1};
516 aas_plane_t *plane, *plane2;
517 aas_trace_t trace, steptrace;
518
519 if (frametime <= 0) frametime = 0.1f;
520 //
521 phys_friction = aassettings.phys_friction;
522 phys_stopspeed = aassettings.phys_stopspeed;
523 phys_gravity = aassettings.phys_gravity;
524 phys_waterfriction = aassettings.phys_waterfriction;
525 phys_watergravity = aassettings.phys_watergravity;
526 phys_maxwalkvelocity = aassettings.phys_maxwalkvelocity;// * frametime;
527 phys_maxcrouchvelocity = aassettings.phys_maxcrouchvelocity;// * frametime;
528 phys_maxswimvelocity = aassettings.phys_maxswimvelocity;// * frametime;
529 phys_walkaccelerate = aassettings.phys_walkaccelerate;
530 phys_airaccelerate = aassettings.phys_airaccelerate;
531 phys_swimaccelerate = aassettings.phys_swimaccelerate;
532 phys_maxstep = aassettings.phys_maxstep;
533 phys_maxsteepness = aassettings.phys_maxsteepness;
534 phys_jumpvel = aassettings.phys_jumpvel * frametime;
535 //
536 Com_Memset(move, 0, sizeof(aas_clientmove_t));
537 Com_Memset(&trace, 0, sizeof(aas_trace_t));
538 //start at the current origin
539 VectorCopy(origin, org);
540 org[2] += 0.25;
541 //velocity to test for the first frame
542 VectorScale(velocity, frametime, frame_test_vel);
543 //
544 jump_frame = -1;
545 //predict a maximum of 'maxframes' ahead
546 for (n = 0; n < maxframes; n++)
547 {
548 swimming = AAS_Swimming(org);
549 //get gravity depending on swimming or not
550 gravity = swimming ? phys_watergravity : phys_gravity;
551 //apply gravity at the START of the frame
552 frame_test_vel[2] = frame_test_vel[2] - (gravity * 0.1 * frametime);
553 //if on the ground or swimming
554 if (onground || swimming)
555 {
556 friction = swimming ? phys_waterfriction : phys_friction;
557 //apply friction
558 VectorScale(frame_test_vel, 1/frametime, frame_test_vel);
559 AAS_ApplyFriction(frame_test_vel, friction, phys_stopspeed, frametime);
560 VectorScale(frame_test_vel, frametime, frame_test_vel);
561 } //end if
562 crouch = qfalse;
563 //apply command movement
564 if (n < cmdframes)
565 {
566 //ax = 0;
567 maxvel = phys_maxwalkvelocity;
568 accelerate = phys_airaccelerate;
569 VectorCopy(cmdmove, wishdir);
570 if (onground)
571 {
572 if (cmdmove[2] < -300)
573 {
574 crouch = qtrue;
575 maxvel = phys_maxcrouchvelocity;
576 } //end if
577 //if not swimming and upmove is positive then jump
578 if (!swimming && cmdmove[2] > 1)
579 {
580 //jump velocity minus the gravity for one frame + 5 for safety
581 frame_test_vel[2] = phys_jumpvel - (gravity * 0.1 * frametime) + 5;
582 jump_frame = n;
583 //jumping so air accelerate
584 accelerate = phys_airaccelerate;
585 } //end if
586 else
587 {
588 accelerate = phys_walkaccelerate;
589 } //end else
590 //ax = 2;
591 } //end if
592 if (swimming)
593 {
594 maxvel = phys_maxswimvelocity;
595 accelerate = phys_swimaccelerate;
596 //ax = 3;
597 } //end if
598 else
599 {
600 wishdir[2] = 0;
601 } //end else
602 //
603 wishspeed = VectorNormalize(wishdir);
604 if (wishspeed > maxvel) wishspeed = maxvel;
605 VectorScale(frame_test_vel, 1/frametime, frame_test_vel);
606 AAS_Accelerate(frame_test_vel, frametime, wishdir, wishspeed, accelerate);
607 VectorScale(frame_test_vel, frametime, frame_test_vel);
608 /*
609 for (i = 0; i < ax; i++)
610 {
611 velchange = (cmdmove[i] * frametime) - frame_test_vel[i];
612 if (velchange > phys_maxacceleration) velchange = phys_maxacceleration;
613 else if (velchange < -phys_maxacceleration) velchange = -phys_maxacceleration;
614 newvel = frame_test_vel[i] + velchange;
615 //
616 if (frame_test_vel[i] <= maxvel && newvel > maxvel) frame_test_vel[i] = maxvel;
617 else if (frame_test_vel[i] >= -maxvel && newvel < -maxvel) frame_test_vel[i] = -maxvel;
618 else frame_test_vel[i] = newvel;
619 } //end for
620 */
621 } //end if
622 if (crouch)
623 {
624 presencetype = PRESENCE_CROUCH;
625 } //end if

1040//===========================================================================
1041// calculates the horizontal velocity needed to perform a jump from start
1042// to end
1043//
1044// Parameter: zvel : z velocity for jump
1045// start : start position of jump
1046// end : end position of jump
1047// *speed : returned speed for jump
1048// Returns: qfalse if too high or too far from start to end
1049// Changes Globals: -
1050//===========================================================================
1051int AAS_HorizontalVelocityForJump(float zvel, vec3_t start, vec3_t end, float *velocity)
1052{
1053 float phys_gravity, phys_maxvelocity;
1054 float maxjump, height2fall, t, top;
1055 vec3_t dir;
1056
1057 phys_gravity = aassettings.phys_gravity;
1058 phys_maxvelocity = aassettings.phys_maxvelocity;
1059
1060 //maximum height a player can jump with the given initial z velocity
1061 maxjump = 0.5 * phys_gravity * (zvel / phys_gravity) * (zvel / phys_gravity);
1062 //top of the parabolic jump
1063 top = start[2] + maxjump;
1064 //height the bot will fall from the top
1065 height2fall = top - end[2];
1066 //if the goal is to high to jump to
1067 if (height2fall < 0)
1068 {
1069 *velocity = phys_maxvelocity;
1070 return 0;
1071 } //end if
1072 //time a player takes to fall the height
1073 t = sqrt(height2fall / (0.5 * phys_gravity));
1074 //direction from start to end
1075 VectorSubtract(end, start, dir);
1076 //
1077 if ( (t + zvel / phys_gravity) == 0.0f ) {
1078 *velocity = phys_maxvelocity;
1079 return 0;
1080 }
1081 //calculate horizontal speed
1082 *velocity = sqrt(dir[0]*dir[0] + dir[1]*dir[1]) / (t + zvel / phys_gravity);
1083 //the horizontal speed must be lower than the max speed
1084 if (*velocity > phys_maxvelocity)
1085 {
1086 *velocity = phys_maxvelocity;
1087 return 0;
1088 } //end if
1089 return 1;
1090} //end of the function AAS_HorizontalVelocityForJump

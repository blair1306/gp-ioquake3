87qboolean ValidClientNumber(int num, char *str)
88{
89 if (num < 0 || num > botlibglobals.maxclients)
90 {
91 //weird: the disabled stuff results in a crash
92 botimport.Print(PRT_ERROR, "%s: invalid client number %d, [0, %d]\n",
93 str, num, botlibglobals.maxclients);
94 return qfalse;
95 } //end if
96 return qtrue;
97} //end of the function BotValidateClientNumber

136int Export_BotLibSetup(void)
137{
138 int errnum;
139
140 botDeveloper = LibVarGetValue("bot_developer");
141 memset( &botlibglobals, 0, sizeof(botlibglobals) );
142 //initialize byte swapping (litte endian etc.)
143// Swap_Init();
144
145 if(botDeveloper)
146 {
147 char *homedir, *gamedir, *basegame;
148 char logfilename[MAX_OSPATH];
149
150 homedir = LibVarGetString("homedir");
151 gamedir = LibVarGetString("gamedir");
152 basegame = LibVarGetString("basegame");
153
154 if (*homedir)
155 {
156 if(*gamedir)
157 Com_sprintf(logfilename, sizeof(logfilename), "%s%c%s%cbotlib.log", homedir, PATH_SEP, gamedir, PATH_SEP);
158 else if(*basegame)
159 Com_sprintf(logfilename, sizeof(logfilename), "%s%c%s%cbotlib.log", homedir, PATH_SEP, basegame, PATH_SEP);
160 else
161 Com_sprintf(logfilename, sizeof(logfilename), "%s%c" BASEGAME "%cbotlib.log", homedir, PATH_SEP, PATH_SEP);
162 }
163 else
164 Com_sprintf(logfilename, sizeof(logfilename), "botlib.log");
165
166 Log_Open(logfilename);
167 }
168
169 botimport.Print(PRT_MESSAGE, "------- BotLib Initialization -------\n");
170
171 botlibglobals.maxclients = (int) LibVarValue("maxclients", "128");
172 botlibglobals.maxentities = (int) LibVarValue("maxentities", "1024");
173
174 errnum = AAS_Setup(); //be_aas_main.c
175 if (errnum != BLERR_NOERROR) return errnum;
176 errnum = EA_Setup(); //be_ea.c
177 if (errnum != BLERR_NOERROR) return errnum;
178 errnum = BotSetupWeaponAI(); //be_ai_weap.c
179 if (errnum != BLERR_NOERROR)return errnum;
180 errnum = BotSetupGoalAI(); //be_ai_goal.c
181 if (errnum != BLERR_NOERROR) return errnum;
182 errnum = BotSetupChatAI(); //be_ai_chat.c
183 if (errnum != BLERR_NOERROR) return errnum;
184 errnum = BotSetupMoveAI(); //be_ai_move.c
185 if (errnum != BLERR_NOERROR) return errnum;
186
187 botlibsetup = qtrue;
188 botlibglobals.botlibsetup = qtrue;
189
190 return BLERR_NOERROR;
191} //end of the function Export_BotLibSetup

198int Export_BotLibShutdown(void)
199{
200 if (!BotLibSetup("BotLibShutdown")) return BLERR_LIBRARYNOTSETUP;
201#ifndef DEMO
202 //DumpFileCRCs();
203#endif //DEMO
204 //
205 BotShutdownChatAI(); //be_ai_chat.c
206 BotShutdownMoveAI(); //be_ai_move.c
207 BotShutdownGoalAI(); //be_ai_goal.c
208 BotShutdownWeaponAI(); //be_ai_weap.c
209 BotShutdownWeights(); //be_ai_weight.c
210 BotShutdownCharacters(); //be_ai_char.c
211 //shud down aas
212 AAS_Shutdown();
213 //shut down bot elemantary actions
214 EA_Shutdown();
215 //free all libvars
216 LibVarDeAllocAll();
217 //remove all global defines from the pre compiler
218 PC_RemoveAllGlobalDefines();
219
220 //dump all allocated memory
221// DumpMemory();
222#ifdef DEBUG
223 PrintMemoryLabels();
224#endif
225 //shut down library log file
226 Log_Shutdown();
227 //
228 botlibsetup = qfalse;
229 botlibglobals.botlibsetup = qfalse;
230 // print any files still open
231 PC_CheckOpenSourceHandles();
232 //
233 return BLERR_NOERROR;
234} //end of the function Export_BotLibShutdown

199void AAS_UnlinkCache(aas_routingcache_t *cache)
200{
201 if (cache->time_next) cache->time_next->time_prev = cache->time_prev;
202 else aasworld.newestcache = cache->time_prev;
203 if (cache->time_prev) cache->time_prev->time_next = cache->time_next;
204 else aasworld.oldestcache = cache->time_next;
205 cache->time_next = NULL;
206 cache->time_prev = NULL;
207} //end of the function AAS_UnlinkCache
208//===========================================================================
209//
210// Parameter: -
211// Returns: -
212// Changes Globals: -
213//===========================================================================
214void AAS_LinkCache(aas_routingcache_t *cache)
215{
216 if (aasworld.newestcache)
217 {
218 aasworld.newestcache->time_next = cache;
219 cache->time_prev = aasworld.newestcache;
220 } //end if
221 else
222 {
223 aasworld.oldestcache = cache;
224 cache->time_prev = NULL;
225 } //end else
226 cache->time_next = NULL;
227 aasworld.newestcache = cache;
228} //end of the function AAS_LinkCache

272void AAS_RemoveRoutingCacheUsingArea( int areanum )
273{
274 int i, clusternum;
275 aas_routingcache_t *cache, *nextcache;
276
277 clusternum = aasworld.areasettings[areanum].cluster;
278 if (clusternum > 0)
279 {
280 //remove all the cache in the cluster the area is in
281 AAS_RemoveRoutingCacheInCluster( clusternum );
282 } //end if
283 else
284 {
285 // if this is a portal remove all cache in both the front and back cluster
286 AAS_RemoveRoutingCacheInCluster( aasworld.portals[-clusternum].frontcluster );
287 AAS_RemoveRoutingCacheInCluster( aasworld.portals[-clusternum].backcluster );
288 } //end else
289 // remove all portal cache
290 for (i = 0; i < aasworld.numareas; i++)
291 {
292 //refresh portal cache
293 for (cache = aasworld.portalcache[i]; cache; cache = nextcache)
294 {
295 nextcache = cache->next;
296 AAS_FreeRoutingCache(cache);
297 } //end for
298 aasworld.portalcache[i] = NULL;
299 } //end for
300} //end of the function AAS_RemoveRoutingCacheUsingArea

307int AAS_EnableRoutingArea(int areanum, int enable)
308{
309 int flags;
310
311 if (areanum <= 0 || areanum >= aasworld.numareas)
312 {
313 if (botDeveloper)
314 {
315 botimport.Print(PRT_ERROR, "AAS_EnableRoutingArea: areanum %d out of range\n", areanum);
316 } //end if
317 return 0;
318 } //end if
319 flags = aasworld.areasettings[areanum].areaflags & AREA_DISABLED;
320 if (enable < 0)
321 return !flags;
322
323 if (enable)
324 aasworld.areasettings[areanum].areaflags &= ~AREA_DISABLED;
325 else
326 aasworld.areasettings[areanum].areaflags |= AREA_DISABLED;
327 // if the status of the area changed
328 if ( (flags & AREA_DISABLED) != (aasworld.areasettings[areanum].areaflags & AREA_DISABLED) )
329 {
330 //remove all routing cache involving this area
331 AAS_RemoveRoutingCacheUsingArea( areanum );
332 } //end if
333 return !flags;
334} //end of the function AAS_EnableRoutingArea

447 //create reversed links for the reachabilities

449 {
450 //reachability link

452 //

458 revlink->next = aasworld.reversedreachability[reach->areanum].first;

460 aasworld.reversedreachability[reach->areanum].numlinks++;
461 } //end for
462 } //end for
463#ifdef DEBUG

473unsigned short int AAS_AreaTravelTime(int areanum, vec3_t start, vec3_t end)
474{
475 int intdist;
476 float dist;
477 vec3_t dir;
478
479 VectorSubtract(start, end, dir);
480 dist = VectorLength(dir);
481 //if crouch only area
482 if (AAS_AreaCrouch(areanum)) dist *= DISTANCEFACTOR_CROUCH;
483 //if swim area
484 else if (AAS_AreaSwim(areanum)) dist *= DISTANCEFACTOR_SWIM;
485 //normal walk area
486 else dist *= DISTANCEFACTOR_WALK;
487 //
488 intdist = (int) dist;
489 //make sure the distance isn't zero
490 if (intdist <= 0) intdist = 1;
491 return intdist;
492} //end of the function AAS_AreaTravelTime

533 for (i = 0; i < aasworld.numareas; i++)
534 {

537 //settings of the area

539 //

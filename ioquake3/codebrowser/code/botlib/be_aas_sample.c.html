72void AAS_PresenceTypeBoundingBox(int presencetype, vec3_t mins, vec3_t maxs)
73{
74 int index;
75 //bounding box size for each presence type
76 vec3_t boxmins[3] = {{0, 0, 0}, {-15, -15, -24}, {-15, -15, -24}};
77 vec3_t boxmaxs[3] = {{0, 0, 0}, { 15, 15, 32}, { 15, 15, 8}};
78
79 if (presencetype == PRESENCE_NORMAL) index = 1;
80 else if (presencetype == PRESENCE_CROUCH) index = 2;
81 else
82 {
83 botimport.Print(PRT_FATAL, "AAS_PresenceTypeBoundingBox: unknown presence type\n");
84 index = 2;
85 } //end if
86 VectorCopy(boxmins[index], mins);
87 VectorCopy(boxmaxs[index], maxs);
88} //end of the function AAS_PresenceTypeBoundingBox

95void AAS_InitAASLinkHeap(void)
96{
97 int i, max_aaslinks;
98
99 max_aaslinks = aasworld.linkheapsize;
100 //if there's no link heap present
101 if (!aasworld.linkheap)
102 {
103#ifdef BSPC
104 max_aaslinks = 6144;
105#else
106 max_aaslinks = (int) LibVarValue("max_aaslinks", "6144");
107#endif
108 if (max_aaslinks < 0) max_aaslinks = 0;
109 aasworld.linkheapsize = max_aaslinks;
110 aasworld.linkheap = (aas_link_t *) GetHunkMemory(max_aaslinks * sizeof(aas_link_t));
111 } //end if
112 //link the links on the heap
113 aasworld.linkheap[0].prev_ent = NULL;
114 aasworld.linkheap[0].next_ent = &aasworld.linkheap[1];
115 for (i = 1; i < max_aaslinks-1; i++)
116 {
117 aasworld.linkheap[i].prev_ent = &aasworld.linkheap[i - 1];
118 aasworld.linkheap[i].next_ent = &aasworld.linkheap[i + 1];
119 } //end for
120 aasworld.linkheap[max_aaslinks-1].prev_ent = &aasworld.linkheap[max_aaslinks-2];
121 aasworld.linkheap[max_aaslinks-1].next_ent = NULL;
122 //pointer to the first free link
123 aasworld.freelinks = &aasworld.linkheap[0];
124 //
125 numaaslinks = max_aaslinks;
126} //end of the function AAS_InitAASLinkHeap

212int AAS_PointAreaNum(vec3_t point)
213{
214 int nodenum;
215 vec_t dist;
216 aas_node_t *node;
217 aas_plane_t *plane;
218
219 if (!aasworld.loaded)
220 {
221 botimport.Print(PRT_ERROR, "AAS_PointAreaNum: aas not loaded\n");
222 return 0;
223 } //end if
224
225 //start with node 1 because node zero is a dummy used for solid leafs
226 nodenum = 1;
227 while (nodenum > 0)
228 {
229// botimport.Print(PRT_MESSAGE, "[%d]", nodenum);
230#ifdef AAS_SAMPLE_DEBUG
231 if (nodenum >= aasworld.numnodes)
232 {
233 botimport.Print(PRT_ERROR, "nodenum = %d >= aasworld.numnodes = %d\n", nodenum, aasworld.numnodes);
234 return 0;
235 } //end if
236#endif //AAS_SAMPLE_DEBUG
237 node = &aasworld.nodes[nodenum];
238#ifdef AAS_SAMPLE_DEBUG
239 if (node->planenum < 0 || node->planenum >= aasworld.numplanes)
240 {
241 botimport.Print(PRT_ERROR, "node->planenum = %d >= aasworld.numplanes = %d\n", node->planenum, aasworld.numplanes);
242 return 0;
243 } //end if
244#endif //AAS_SAMPLE_DEBUG
245 plane = &aasworld.planes[node->planenum];
246 dist = DotProduct(point, plane->normal) - plane->dist;
247 if (dist > 0) nodenum = node->children[0];
248 else nodenum = node->children[1];
249 } //end while
250 if (!nodenum)
251 {
252#ifdef AAS_SAMPLE_DEBUG
253 botimport.Print(PRT_MESSAGE, "in solid\n");
254#endif //AAS_SAMPLE_DEBUG
255 return 0;
256 } //end if
257 return -nodenum;
258} //end of the function AAS_PointAreaNum

265int AAS_PointReachabilityAreaIndex( vec3_t origin )
266{
267 int areanum, cluster, i, index;
268
269 if (!aasworld.initialized)
270 return 0;
271
272 if ( !origin )
273 {
274 index = 0;
275 for (i = 0; i < aasworld.numclusters; i++)
276 {
277 index += aasworld.clusters[i].numreachabilityareas;
278 } //end for
279 return index;
280 } //end if
281
282 areanum = AAS_PointAreaNum( origin );
283 if ( !areanum || !AAS_AreaReachability(areanum) )
284 return 0;
285 cluster = aasworld.areasettings[areanum].cluster;
286 areanum = aasworld.areasettings[areanum].clusterareanum;
287 if (cluster < 0)
288 {
289 cluster = aasworld.portals[-cluster].frontcluster;
290 areanum = aasworld.portals[-cluster].clusterareanum[0];
291 } //end if
292
293 index = 0;
294 for (i = 0; i < cluster; i++)
295 {
296 index += aasworld.clusters[i].numreachabilityareas;
297 } //end for
298 index += areanum;
299 return index;
300} //end of the function AAS_PointReachabilityAreaIndex

350//===========================================================================
351// calculates the minimum distance between the origin of the box and the
352// given plane when both will collide on the given side of the plane
353//
354// normal = normal vector of plane to calculate distance from
355// mins = minimums of box relative to origin
356// maxs = maximums of box relative to origin
357// side = side of the plane we want to calculate the distance from
358// 0 normal vector side
359// 1 not normal vector side
360//
361// Parameter: -
362// Returns: -
363// Changes Globals: -
364//===========================================================================
365vec_t AAS_BoxOriginDistanceFromPlane(vec3_t normal, vec3_t mins, vec3_t maxs, int side)
366{
367 vec3_t v1, v2;
368 int i;
369
370 //swap maxs and mins when on the other side of the plane
371 if (side)
372 {
373 //get a point of the box that would be one of the first
374 //to collide with the plane
375 for (i = 0; i < 3; i++)
376 {
377 if (normal[i] > BBOX_NORMAL_EPSILON) v1[i] = maxs[i];
378 else if (normal[i] < -BBOX_NORMAL_EPSILON) v1[i] = mins[i];
379 else v1[i] = 0;
380 } //end for
381 } //end if
382 else
383 {
384 //get a point of the box that would be one of the first
385 //to collide with the plane
386 for (i = 0; i < 3; i++)
387 {
388 if (normal[i] > BBOX_NORMAL_EPSILON) v1[i] = mins[i];
389 else if (normal[i] < -BBOX_NORMAL_EPSILON) v1[i] = maxs[i];
390 else v1[i] = 0;
391 } //end for
392 } //end else
393 //
394 VectorCopy(normal, v2);
395 VectorInverse(v2);
396// VectorNegate(normal, v2);
397 return DotProduct(v1, v2);
398} //end of the function AAS_BoxOriginDistanceFromPlane

405qboolean AAS_AreaEntityCollision(int areanum, vec3_t start, vec3_t end,
406 int presencetype, int passent, aas_trace_t *trace)
407{
408 int collision;
409 vec3_t boxmins, boxmaxs;
410 aas_link_t *link;
411 bsp_trace_t bsptrace;
412
413 AAS_PresenceTypeBoundingBox(presencetype, boxmins, boxmaxs);
414
415 Com_Memset(&bsptrace, 0, sizeof(bsp_trace_t)); //make compiler happy
416 //assume no collision
417 bsptrace.fraction = 1;
418 collision = qfalse;
419 for (link = aasworld.arealinkedentities[areanum]; link; link = link->next_ent)
420 {
421 //ignore the pass entity
422 if (link->entnum == passent) continue;
423 //
424 if (AAS_EntityCollision(link->entnum, start, boxmins, boxmaxs, end,
425 CONTENTS_SOLID|CONTENTS_PLAYERCLIP, &bsptrace))
426 {
427 collision = qtrue;
428 } //end if
429 } //end for
430 if (collision)
431 {
432 trace->startsolid = bsptrace.startsolid;
433 trace->ent = bsptrace.ent;
434 VectorCopy(bsptrace.endpos, trace->endpos);
435 trace->area = 0;
436 trace->planenum = 0;
437 return qtrue;
438 } //end if
439 return qfalse;
440} //end of the function AAS_AreaEntityCollision
441//===========================================================================
442// recursive subdivision of the line by the BSP tree.
443//
444// Parameter: -
445// Returns: -
446// Changes Globals: -
447//===========================================================================
448aas_trace_t AAS_TraceClientBBox(vec3_t start, vec3_t end, int presencetype,
449 int passent)
450{
451 int side, nodenum, tmpplanenum;
452 float front, back, frac;
453 vec3_t cur_start, cur_end, cur_mid, v1, v2;
454 aas_tracestack_t tracestack[127];
455 aas_tracestack_t *tstack_p;
456 aas_node_t *aasnode;
457 aas_plane_t *plane;
458 aas_trace_t trace;
459
460 //clear the trace structure
461 Com_Memset(&trace, 0, sizeof(aas_trace_t));
462
463 if (!aasworld.loaded) return trace;
464
465 tstack_p = tracestack;
466 //we start with the whole line on the stack
467 VectorCopy(start, tstack_p->start);
468 VectorCopy(end, tstack_p->end);
469 tstack_p->planenum = 0;
470 //start with node 1 because node zero is a dummy for a solid leaf
471 tstack_p->nodenum = 1; //starting at the root of the tree
472 tstack_p++;
473
474 while (1)
475 {
476 //pop up the stack
477 tstack_p--;
478 //if the trace stack is empty (ended up with a piece of the
479 //line to be traced in an area)
480 if (tstack_p < tracestack)
481 {
482 tstack_p++;
483 //nothing was hit
484 trace.startsolid = qfalse;
485 trace.fraction = 1.0;
486 //endpos is the end of the line
487 VectorCopy(end, trace.endpos);
488 //nothing hit
489 trace.ent = 0;
490 trace.area = 0;
491 trace.planenum = 0;
492 return trace;
493 } //end if
494 //number of the current node to test the line against
495 nodenum = tstack_p->nodenum;
496 //if it is an area
497 if (nodenum < 0)
498 {
499#ifdef AAS_SAMPLE_DEBUG
500 if (-nodenum > aasworld.numareasettings)
501 {
502 botimport.Print(PRT_ERROR, "AAS_TraceBoundingBox: -nodenum out of range\n");
503 return trace;
504 } //end if
505#endif //AAS_SAMPLE_DEBUG
506 //botimport.Print(PRT_MESSAGE, "areanum = %d, must be %d\n", -nodenum, AAS_PointAreaNum(start));
507 //if can't enter the area because it hasn't got the right presence type
508 if (!(aasworld.areasettings[-nodenum].presencetype & presencetype))
509 {
510 //if the start point is still the initial start point
511 //NOTE: no need for epsilons because the points will be
512 //exactly the same when they're both the start point
513 if (tstack_p->start[0] == start[0] &&
514 tstack_p->start[1] == start[1] &&
515 tstack_p->start[2] == start[2])
516 {
517 trace.startsolid = qtrue;
518 trace.fraction = 0.0;
519 VectorClear(v1);
520 } //end if
521 else
522 {
523 trace.startsolid = qfalse;
524 VectorSubtract(end, start, v1);
525 VectorSubtract(tstack_p->start, start, v2);
526 trace.fraction = VectorLength(v2) / VectorNormalize(v1);
527 VectorMA(tstack_p->start, -0.125, v1, tstack_p->start);
528 } //end else
529 VectorCopy(tstack_p->start, trace.endpos);
530 trace.ent = 0;
531 trace.area = -nodenum;
532// VectorSubtract(end, start, v1);
533 trace.planenum = tstack_p->planenum;
534 //always take the plane with normal facing towards the trace start
535 plane = &aasworld.planes[trace.planenum];
536 if (DotProduct(v1, plane->normal) > 0) trace.planenum ^= 1;
537 return trace;
538 } //end if
539 else
540 {
541 if (passent >= 0)
542 {
543 if (AAS_AreaEntityCollision(-nodenum, tstack_p->start,
544 tstack_p->end, presencetype, passent,
545 &trace))
546 {
547 if (!trace.startsolid)
548 {
549 VectorSubtract(end, start, v1);
550 VectorSubtract(trace.endpos, start, v2);
551 trace.fraction = VectorLength(v2) / VectorLength(v1);
552 } //end if
553 return trace;
554 } //end if
555 } //end if
556 } //end else
557 trace.lastarea = -nodenum;
558 continue;
559 } //end if
560 //if it is a solid leaf
561 if (!nodenum)
562 {
563 //if the start point is still the initial start point
564 //NOTE: no need for epsilons because the points will be
565 //exactly the same when they're both the start point
566 if (tstack_p->start[0] == start[0] &&
567 tstack_p->start[1] == start[1] &&
568 tstack_p->start[2] == start[2])
569 {
570 trace.startsolid = qtrue;
571 trace.fraction = 0.0;
572 VectorClear(v1);
573 } //end if
574 else
575 {
576 trace.startsolid = qfalse;
577 VectorSubtract(end, start, v1);
578 VectorSubtract(tstack_p->start, start, v2);
579 trace.fraction = VectorLength(v2) / VectorNormalize(v1);
580 VectorMA(tstack_p->start, -0.125, v1, tstack_p->start);
581 } //end else
582 VectorCopy(tstack_p->start, trace.endpos);
583 trace.ent = 0;
584 trace.area = 0; //hit solid leaf
585// VectorSubtract(end, start, v1);
586 trace.planenum = tstack_p->planenum;
587 //always take the plane with normal facing towards the trace start
588 plane = &aasworld.planes[trace.planenum];
589 if (DotProduct(v1, plane->normal) > 0) trace.planenum ^= 1;
590 return trace;
591 } //end if
592#ifdef AAS_SAMPLE_DEBUG
593 if (nodenum > aasworld.numnodes)
594 {
595 botimport.Print(PRT_ERROR, "AAS_TraceBoundingBox: nodenum out of range\n");
596 return trace;
597 } //end if
598#endif //AAS_SAMPLE_DEBUG
599 //the node to test against
600 aasnode = &aasworld.nodes[nodenum];
601 //start point of current line to test against node
602 VectorCopy(tstack_p->start, cur_start);
603 //end point of the current line to test against node
604 VectorCopy(tstack_p->end, cur_end);
605 //the current node plane
606 plane = &aasworld.planes[aasnode->planenum];
607
608 switch(plane->type)
609 {/*FIXME: wtf doesn't this work? obviously the axial node planes aren't always facing positive!!!
610 //check for axial planes
611 case PLANE_X:
612 {
613 front = cur_start[0] - plane->dist;
614 back = cur_end[0] - plane->dist;
615 break;
616 } //end case
617 case PLANE_Y:
618 {
619 front = cur_start[1] - plane->dist;
620 back = cur_end[1] - plane->dist;
621 break;
622 } //end case
623 case PLANE_Z:
624 {
625 front = cur_start[2] - plane->dist;
626 back = cur_end[2] - plane->dist;
627 break;
628 } //end case*/
629 default: //gee it's not an axial plane
630 {
631 front = DotProduct(cur_start, plane->normal) - plane->dist;
632 back = DotProduct(cur_end, plane->normal) - plane->dist;
633 break;
634 } //end default
635 } //end switch
636 // bk010221 - old location of FPE hack and divide by zero expression
637 //if the whole to be traced line is totally at the front of this node
638 //only go down the tree with the front child
639 if ((front >= -ON_EPSILON && back >= -ON_EPSILON))
640 {
641 //keep the current start and end point on the stack
642 //and go down the tree with the front child
643 tstack_p->nodenum = aasnode->children[0];
644 tstack_p++;
645 if (tstack_p >= &tracestack[127])
646 {
647 botimport.Print(PRT_ERROR, "AAS_TraceBoundingBox: stack overflow\n");
648 return trace;
649 } //end if
650 } //end if
651 //if the whole to be traced line is totally at the back of this node
652 //only go down the tree with the back child
653 else if ((front < ON_EPSILON && back < ON_EPSILON))
654 {
655 //keep the current start and end point on the stack
656 //and go down the tree with the back child
657 tstack_p->nodenum = aasnode->children[1];
658 tstack_p++;
659 if (tstack_p >= &tracestack[127])
660 {
661 botimport.Print(PRT_ERROR, "AAS_TraceBoundingBox: stack overflow\n");
662 return trace;
663 } //end if
664 } //end if
665 //go down the tree both at the front and back of the node
666 else
667 {
668 tmpplanenum = tstack_p->planenum;
669 // bk010221 - new location of divide by zero (see above)
670 if ( front == back ) front -= 0.001f; // bk0101022 - hack/FPE
671 //calculate the hitpoint with the node (split point of the line)
672 //put the crosspoint TRACEPLANE_EPSILON pixels on the near side
673 if (front < 0) frac = (front + TRACEPLANE_EPSILON)/(front-back);
674 else frac = (front - TRACEPLANE_EPSILON)/(front-back); // bk010221
675 //
676 if (frac < 0)
677 frac = 0.001f; //0
678 else if (frac > 1)
679 frac = 0.999f; //1
680 //frac = front / (front-back);
681 //
682 cur_mid[0] = cur_start[0] + (cur_end[0] - cur_start[0]) * frac;
683 cur_mid[1] = cur_start[1] + (cur_end[1] - cur_start[1]) * frac;
684 cur_mid[2] = cur_start[2] + (cur_end[2] - cur_start[2]) * frac;
685
686// AAS_DrawPlaneCross(cur_mid, plane->normal, plane->dist, plane->type, LINECOLOR_RED);
687 //side the front part of the line is on
688 side = front < 0;

916//===========================================================================
917// tests if the given point is within the face boundaries
918//
919// Parameter: face : face to test if the point is in it
920// pnormal : normal of the plane to use for the face
921// point : point to test if inside face boundaries
922// Returns: qtrue if the point is within the face boundaries
923// Changes Globals: -
924//===========================================================================
925qboolean AAS_InsideFace(aas_face_t *face, vec3_t pnormal, vec3_t point, float epsilon)
926{
927 int i, firstvertex, edgenum;
928 vec3_t v0;
929 vec3_t edgevec, pointvec, sepnormal;
930 aas_edge_t *edge;
931#ifdef AAS_SAMPLE_DEBUG
932 int lastvertex = 0;
933#endif //AAS_SAMPLE_DEBUG
934
935 if (!aasworld.loaded) return qfalse;
936
937 for (i = 0; i < face->numedges; i++)
938 {
939 edgenum = aasworld.edgeindex[face->firstedge + i];
940 edge = &aasworld.edges[abs(edgenum)];
941 //get the first vertex of the edge
942 firstvertex = edgenum < 0;
943 VectorCopy(aasworld.vertexes[edge->v[firstvertex]], v0);
944 //edge vector
945 VectorSubtract(aasworld.vertexes[edge->v[!firstvertex]], v0, edgevec);
946 //
947#ifdef AAS_SAMPLE_DEBUG
948 if (lastvertex && lastvertex != edge->v[firstvertex])
949 {
950 botimport.Print(PRT_MESSAGE, "winding not counter clockwise\n");
951 } //end if
952 lastvertex = edge->v[!firstvertex];
953#endif //AAS_SAMPLE_DEBUG
954 //vector from first edge point to point possible in face
955 VectorSubtract(point, v0, pointvec);
956 //get a vector pointing inside the face orthogonal to both the
957 //edge vector and the normal vector of the plane the face is in
958 //this vector defines a plane through the origin (first vertex of
959 //edge) and through both the edge vector and the normal vector
960 //of the plane
961 AAS_OrthogonalToVectors(edgevec, pnormal, sepnormal);
962 //check on wich side of the above plane the point is
963 //this is done by checking the sign of the dot product of the
964 //vector orthogonal vector from above and the vector from the
965 //origin (first vertex of edge) to the point
966 //if the dotproduct is smaller than zero the point is outside the face
967 if (DotProduct(pointvec, sepnormal) < -epsilon) return qfalse;
968 } //end for
969 return qtrue;
970} //end of the function AAS_InsideFace

1223{

1226 aas_linkstack_t *lstack_p;
1227 aas_node_t *aasnode;
1228 aas_plane_t *plane;

1241 //start with node 1 because node zero is a dummy used for solid leafs

1246 {

1248 lstack_p--;

1296 } //end if

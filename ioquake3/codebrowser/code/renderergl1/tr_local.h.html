771/*
772
773the drawsurf sort data is packed into a single 32 bit value so it can be
774compared quickly during the qsorting process
775
776the bits are allocated as follows:
777
7780 - 1 : dlightmap index
779//2 : used to be clipped flag REMOVED - 03.21.00 rad
7802 - 6 : fog index
78111 - 20 : entity index
78221 - 31 : sorted shader index
783
784 TTimo - 1.32
7850-1 : dlightmap index
7862-6 : fog index
7877-16 : entity index
78817-30 : sorted shader index
789*/
790#define QSORT_FOGNUM_SHIFT 2
791#define QSORT_REFENTITYNUM_SHIFT 7
792#define QSORT_SHADERNUM_SHIFT (QSORT_REFENTITYNUM_SHIFT+REFENTITYNUM_BITS)

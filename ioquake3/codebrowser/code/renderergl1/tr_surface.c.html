46/*
47==============
48RB_CheckOverflow
49==============
50*/
51void RB_CheckOverflow( int verts, int indexes ) {
52 if (tess.numVertexes + verts < SHADER_MAX_VERTEXES
53 && tess.numIndexes + indexes < SHADER_MAX_INDEXES) {
54 return;
55 }
56
57 RB_EndSurface();
58
59 if ( verts >= SHADER_MAX_VERTEXES ) {
60 ri.Error(ERR_DROP, "RB_CheckOverflow: verts > MAX (%d > %d)", verts, SHADER_MAX_VERTEXES );
61 }
62 if ( indexes >= SHADER_MAX_INDEXES ) {
63 ri.Error(ERR_DROP, "RB_CheckOverflow: indices > MAX (%d > %d)", indexes, SHADER_MAX_INDEXES );
64 }
65
66 RB_BeginSurface(tess.shader, tess.fogNum );
67}

70/*
71==============
72RB_AddQuadStampExt
73==============
74*/
75void RB_AddQuadStampExt( vec3_t origin, vec3_t left, vec3_t up, byte *color, float s1, float t1, float s2, float t2 ) {
76 vec3_t normal;
77 int ndx;
78
79 RB_CHECKOVERFLOW( 4, 6 );
80
81 ndx = tess.numVertexes;
82
83 // triangle indexes for a simple quad
84 tess.indexes[ tess.numIndexes ] = ndx;
85 tess.indexes[ tess.numIndexes + 1 ] = ndx + 1;
86 tess.indexes[ tess.numIndexes + 2 ] = ndx + 3;
87
88 tess.indexes[ tess.numIndexes + 3 ] = ndx + 3;
89 tess.indexes[ tess.numIndexes + 4 ] = ndx + 1;
90 tess.indexes[ tess.numIndexes + 5 ] = ndx + 2;
91
92 tess.xyz[ndx][0] = origin[0] + left[0] + up[0];
93 tess.xyz[ndx][1] = origin[1] + left[1] + up[1];
94 tess.xyz[ndx][2] = origin[2] + left[2] + up[2];
95
96 tess.xyz[ndx+1][0] = origin[0] - left[0] + up[0];
97 tess.xyz[ndx+1][1] = origin[1] - left[1] + up[1];
98 tess.xyz[ndx+1][2] = origin[2] - left[2] + up[2];
99
100 tess.xyz[ndx+2][0] = origin[0] - left[0] - up[0];
101 tess.xyz[ndx+2][1] = origin[1] - left[1] - up[1];
102 tess.xyz[ndx+2][2] = origin[2] - left[2] - up[2];
103
104 tess.xyz[ndx+3][0] = origin[0] + left[0] - up[0];
105 tess.xyz[ndx+3][1] = origin[1] + left[1] - up[1];
106 tess.xyz[ndx+3][2] = origin[2] + left[2] - up[2];
107
108
109 // constant normal all the way around
110 VectorSubtract( vec3_origin, backEnd.viewParms.or.axis[0], normal );
111
112 tess.normal[ndx][0] = tess.normal[ndx+1][0] = tess.normal[ndx+2][0] = tess.normal[ndx+3][0] = normal[0];
113 tess.normal[ndx][1] = tess.normal[ndx+1][1] = tess.normal[ndx+2][1] = tess.normal[ndx+3][1] = normal[1];
114 tess.normal[ndx][2] = tess.normal[ndx+1][2] = tess.normal[ndx+2][2] = tess.normal[ndx+3][2] = normal[2];
115
116 // standard square texture coordinates
117 tess.texCoords[ndx][0][0] = tess.texCoords[ndx][1][0] = s1;
118 tess.texCoords[ndx][0][1] = tess.texCoords[ndx][1][1] = t1;
119
120 tess.texCoords[ndx+1][0][0] = tess.texCoords[ndx+1][1][0] = s2;
121 tess.texCoords[ndx+1][0][1] = tess.texCoords[ndx+1][1][1] = t1;
122
123 tess.texCoords[ndx+2][0][0] = tess.texCoords[ndx+2][1][0] = s2;
124 tess.texCoords[ndx+2][0][1] = tess.texCoords[ndx+2][1][1] = t2;
125
126 tess.texCoords[ndx+3][0][0] = tess.texCoords[ndx+3][1][0] = s1;
127 tess.texCoords[ndx+3][0][1] = tess.texCoords[ndx+3][1][1] = t2;
128
129 // constant color all the way around
130 // should this be identity and let the shader specify from entity?
131 * ( unsigned int * ) &tess.vertexColors[ndx] =
132 * ( unsigned int * ) &tess.vertexColors[ndx+1] =
133 * ( unsigned int * ) &tess.vertexColors[ndx+2] =
134 * ( unsigned int * ) &tess.vertexColors[ndx+3] =
135 * ( unsigned int * )color;
136
137
138 tess.numVertexes += 4;
139 tess.numIndexes += 6;
140}

151/*
152==============
153RB_SurfaceSprite
154==============
155*/
156static void RB_SurfaceSprite( void ) {
157 vec3_t left, up;
158 float radius;
159
160 // calculate the xyz locations for the four corners
161 radius = backEnd.currentEntity->e.radius;
162 if ( backEnd.currentEntity->e.rotation == 0 ) {
163 VectorScale( backEnd.viewParms.or.axis[1], radius, left );
164 VectorScale( backEnd.viewParms.or.axis[2], radius, up );
165 } else {
166 float s, c;
167 float ang;
168
169 ang = M_PI * backEnd.currentEntity->e.rotation / 180;
170 s = sin( ang );
171 c = cos( ang );
172
173 VectorScale( backEnd.viewParms.or.axis[1], c * radius, left );
174 VectorMA( left, -s * radius, backEnd.viewParms.or.axis[2], left );
175
176 VectorScale( backEnd.viewParms.or.axis[2], c * radius, up );
177 VectorMA( up, s * radius, backEnd.viewParms.or.axis[1], up );
178 }
179 if ( backEnd.viewParms.isMirror ) {
180 VectorSubtract( vec3_origin, left, left );
181 }
182
183 RB_AddQuadStamp( backEnd.currentEntity->e.origin, left, up, backEnd.currentEntity->e.shaderRGBA );
184}

187/*
188=============
189RB_SurfacePolychain
190=============
191*/
192static void RB_SurfacePolychain( srfPoly_t *p ) {
193 int i;
194 int numv;
195
196 RB_CHECKOVERFLOW( p->numVerts, 3*(p->numVerts - 2) );
197
198 // fan triangles into the tess array
199 numv = tess.numVertexes;
200 for ( i = 0; i < p->numVerts; i++ ) {
201 VectorCopy( p->verts[i].xyz, tess.xyz[numv] );
202 tess.texCoords[numv][0][0] = p->verts[i].st[0];
203 tess.texCoords[numv][0][1] = p->verts[i].st[1];
204 *(int *)&tess.vertexColors[numv] = *(int *)p->verts[ i ].modulate;
205
206 numv++;
207 }
208
209 // generate fan indexes into the tess array
210 for ( i = 0; i < p->numVerts-2; i++ ) {
211 tess.indexes[tess.numIndexes + 0] = tess.numVertexes;
212 tess.indexes[tess.numIndexes + 1] = tess.numVertexes + i + 1;
213 tess.indexes[tess.numIndexes + 2] = tess.numVertexes + i + 2;
214 tess.numIndexes += 3;
215 }
216
217 tess.numVertexes = numv;
218}

221/*
222=============
223RB_SurfaceTriangles
224=============
225*/
226static void RB_SurfaceTriangles( srfTriangles_t *srf ) {
227 int i;
228 drawVert_t *dv;
229 float *xyz, *normal, *texCoords;
230 byte *color;
231 int dlightBits;
232 qboolean needsNormal;
233
234 dlightBits = srf->dlightBits;
235 tess.dlightBits |= dlightBits;
236
237 RB_CHECKOVERFLOW( srf->numVerts, srf->numIndexes );
238
239 for ( i = 0 ; i < srf->numIndexes ; i += 3 ) {
240 tess.indexes[ tess.numIndexes + i + 0 ] = tess.numVertexes + srf->indexes[ i + 0 ];
241 tess.indexes[ tess.numIndexes + i + 1 ] = tess.numVertexes + srf->indexes[ i + 1 ];
242 tess.indexes[ tess.numIndexes + i + 2 ] = tess.numVertexes + srf->indexes[ i + 2 ];
243 }
244 tess.numIndexes += srf->numIndexes;
245
246 dv = srf->verts;
247 xyz = tess.xyz[ tess.numVertexes ];
248 normal = tess.normal[ tess.numVertexes ];
249 texCoords = tess.texCoords[ tess.numVertexes ][0];
250 color = tess.vertexColors[ tess.numVertexes ];
251 needsNormal = tess.shader->needsNormal;
252
253 for ( i = 0 ; i < srf->numVerts ; i++, dv++, xyz += 4, normal += 4, texCoords += 4, color += 4 ) {
254 xyz[0] = dv->xyz[0];
255 xyz[1] = dv->xyz[1];
256 xyz[2] = dv->xyz[2];
257
258 if ( needsNormal ) {
259 normal[0] = dv->normal[0];
260 normal[1] = dv->normal[1];
261 normal[2] = dv->normal[2];
262 }
263
264 texCoords[0] = dv->st[0];
265 texCoords[1] = dv->st[1];
266
267 texCoords[2] = dv->lightmap[0];
268 texCoords[3] = dv->lightmap[1];
269
270 *(int *)color = *(int *)dv->color;
271 }
272
273 for ( i = 0 ; i < srf->numVerts ; i++ ) {
274 tess.vertexDlightBits[ tess.numVertexes + i] = dlightBits;
275 }
276
277 tess.numVertexes += srf->numVerts;
278}

282/*
283==============
284RB_SurfaceBeam
285==============
286*/
287static void RB_SurfaceBeam( void )
288{
289#define NUM_BEAM_SEGS 6
290 refEntity_t *e;
291 int i;
292 vec3_t perpvec;
293 vec3_t direction, normalized_direction;
294 vec3_t start_points[NUM_BEAM_SEGS], end_points[NUM_BEAM_SEGS];
295 vec3_t oldorigin, origin;
296
297 e = &backEnd.currentEntity->e;
298
299 oldorigin[0] = e->oldorigin[0];
300 oldorigin[1] = e->oldorigin[1];
301 oldorigin[2] = e->oldorigin[2];
302
303 origin[0] = e->origin[0];
304 origin[1] = e->origin[1];
305 origin[2] = e->origin[2];
306
307 normalized_direction[0] = direction[0] = oldorigin[0] - origin[0];
308 normalized_direction[1] = direction[1] = oldorigin[1] - origin[1];
309 normalized_direction[2] = direction[2] = oldorigin[2] - origin[2];
310
311 if ( VectorNormalize( normalized_direction ) == 0 )
312 return;
313
314 PerpendicularVector( perpvec, normalized_direction );
315
316 VectorScale( perpvec, 4, perpvec );
317
318 for ( i = 0; i < NUM_BEAM_SEGS ; i++ )
319 {
320 RotatePointAroundVector( start_points[i], normalized_direction, perpvec, (360.0/NUM_BEAM_SEGS)*i );
321// VectorAdd( start_points[i], origin, start_points[i] );
322 VectorAdd( start_points[i], direction, end_points[i] );
323 }
324
325 GL_Bind( tr.whiteImage );
326
327 GL_State( GLS_SRCBLEND_ONE | GLS_DSTBLEND_ONE );
328
329 qglColor3f( 1, 0, 0 );
330
331 qglBegin( GL_TRIANGLE_STRIP );
332 for ( i = 0; i <= NUM_BEAM_SEGS; i++ ) {
333 qglVertex3fv( start_points[ i % NUM_BEAM_SEGS] );
334 qglVertex3fv( end_points[ i % NUM_BEAM_SEGS] );
335 }
336 qglEnd();
337}

341static void DoRailCore( const vec3_t start, const vec3_t end, const vec3_t up, float len, float spanWidth )
342{
343 float spanWidth2;
344 int vbase;
345 float t = len / 256.0f;
346
347 RB_CHECKOVERFLOW( 4, 6 );
348
349 vbase = tess.numVertexes;
350
351 spanWidth2 = -spanWidth;
352
353 // FIXME: use quad stamp?
354 VectorMA( start, spanWidth, up, tess.xyz[tess.numVertexes] );
355 tess.texCoords[tess.numVertexes][0][0] = 0;
356 tess.texCoords[tess.numVertexes][0][1] = 0;
357 tess.vertexColors[tess.numVertexes][0] = backEnd.currentEntity->e.shaderRGBA[0] * 0.25;
358 tess.vertexColors[tess.numVertexes][1] = backEnd.currentEntity->e.shaderRGBA[1] * 0.25;
359 tess.vertexColors[tess.numVertexes][2] = backEnd.currentEntity->e.shaderRGBA[2] * 0.25;
360 tess.numVertexes++;
361
362 VectorMA( start, spanWidth2, up, tess.xyz[tess.numVertexes] );
363 tess.texCoords[tess.numVertexes][0][0] = 0;
364 tess.texCoords[tess.numVertexes][0][1] = 1;
365 tess.vertexColors[tess.numVertexes][0] = backEnd.currentEntity->e.shaderRGBA[0];
366 tess.vertexColors[tess.numVertexes][1] = backEnd.currentEntity->e.shaderRGBA[1];
367 tess.vertexColors[tess.numVertexes][2] = backEnd.currentEntity->e.shaderRGBA[2];
368 tess.numVertexes++;
369
370 VectorMA( end, spanWidth, up, tess.xyz[tess.numVertexes] );
371
372 tess.texCoords[tess.numVertexes][0][0] = t;
373 tess.texCoords[tess.numVertexes][0][1] = 0;
374 tess.vertexColors[tess.numVertexes][0] = backEnd.currentEntity->e.shaderRGBA[0];
375 tess.vertexColors[tess.numVertexes][1] = backEnd.currentEntity->e.shaderRGBA[1];
376 tess.vertexColors[tess.numVertexes][2] = backEnd.currentEntity->e.shaderRGBA[2];
377 tess.numVertexes++;
378
379 VectorMA( end, spanWidth2, up, tess.xyz[tess.numVertexes] );
380 tess.texCoords[tess.numVertexes][0][0] = t;
381 tess.texCoords[tess.numVertexes][0][1] = 1;
382 tess.vertexColors[tess.numVertexes][0] = backEnd.currentEntity->e.shaderRGBA[0];
383 tess.vertexColors[tess.numVertexes][1] = backEnd.currentEntity->e.shaderRGBA[1];
384 tess.vertexColors[tess.numVertexes][2] = backEnd.currentEntity->e.shaderRGBA[2];
385 tess.numVertexes++;
386
387 tess.indexes[tess.numIndexes++] = vbase;
388 tess.indexes[tess.numIndexes++] = vbase + 1;
389 tess.indexes[tess.numIndexes++] = vbase + 2;
390
391 tess.indexes[tess.numIndexes++] = vbase + 2;
392 tess.indexes[tess.numIndexes++] = vbase + 1;
393 tess.indexes[tess.numIndexes++] = vbase + 3;
394}

396static void DoRailDiscs( int numSegs, const vec3_t start, const vec3_t dir, const vec3_t right, const vec3_t up )
397{
398 int i;
399 vec3_t pos[4];
400 vec3_t v;
401 int spanWidth = r_railWidth->integer;
402 float c, s;
403 float scale;
404
405 if ( numSegs > 1 )
406 numSegs--;
407 if ( !numSegs )
408 return;
409
410 scale = 0.25;
411
412 for ( i = 0; i < 4; i++ )
413 {
414 c = cos( DEG2RAD( 45 + i * 90 ) );
415 s = sin( DEG2RAD( 45 + i * 90 ) );
416 v[0] = ( right[0] * c + up[0] * s ) * scale * spanWidth;
417 v[1] = ( right[1] * c + up[1] * s ) * scale * spanWidth;
418 v[2] = ( right[2] * c + up[2] * s ) * scale * spanWidth;
419 VectorAdd( start, v, pos[i] );
420
421 if ( numSegs > 1 )
422 {
423 // offset by 1 segment if we're doing a long distance shot
424 VectorAdd( pos[i], dir, pos[i] );
425 }
426 }
427
428 for ( i = 0; i < numSegs; i++ )
429 {
430 int j;
431
432 RB_CHECKOVERFLOW( 4, 6 );
433
434 for ( j = 0; j < 4; j++ )
435 {
436 VectorCopy( pos[j], tess.xyz[tess.numVertexes] );
437 tess.texCoords[tess.numVertexes][0][0] = ( j < 2 );
438 tess.texCoords[tess.numVertexes][0][1] = ( j && j != 3 );
439 tess.vertexColors[tess.numVertexes][0] = backEnd.currentEntity->e.shaderRGBA[0];
440 tess.vertexColors[tess.numVertexes][1] = backEnd.currentEntity->e.shaderRGBA[1];
441 tess.vertexColors[tess.numVertexes][2] = backEnd.currentEntity->e.shaderRGBA[2];
442 tess.numVertexes++;
443
444 VectorAdd( pos[j], dir, pos[j] );
445 }
446
447 tess.indexes[tess.numIndexes++] = tess.numVertexes - 4 + 0;
448 tess.indexes[tess.numIndexes++] = tess.numVertexes - 4 + 1;
449 tess.indexes[tess.numIndexes++] = tess.numVertexes - 4 + 3;
450 tess.indexes[tess.numIndexes++] = tess.numVertexes - 4 + 3;
451 tess.indexes[tess.numIndexes++] = tess.numVertexes - 4 + 1;
452 tess.indexes[tess.numIndexes++] = tess.numVertexes - 4 + 2;
453 }
454}

456/*
457** RB_SurfaceRailRinges
458*/
459static void RB_SurfaceRailRings( void ) {
460 refEntity_t *e;
461 int numSegs;
462 int len;
463 vec3_t vec;
464 vec3_t right, up;
465 vec3_t start, end;
466
467 e = &backEnd.currentEntity->e;
468
469 VectorCopy( e->oldorigin, start );
470 VectorCopy( e->origin, end );
471
472 // compute variables
473 VectorSubtract( end, start, vec );
474 len = VectorNormalize( vec );
475 MakeNormalVectors( vec, right, up );
476 numSegs = ( len ) / r_railSegmentLength->value;
477 if ( numSegs <= 0 ) {
478 numSegs = 1;
479 }
480
481 VectorScale( vec, r_railSegmentLength->value, vec );
482
483 DoRailDiscs( numSegs, start, vec, right, up );
484}

488*/
489static void RB_SurfaceRailCore( void ) {
490 refEntity_t *e;
491 int len;
492 vec3_t right;
493 vec3_t vec;
494 vec3_t start, end;

509 VectorNormalize( v2 );
510 CrossProduct( v1, v2, right );
511 VectorNormalize( right );
512
513 DoRailCore( start, end, right, len, r_railCoreWidth->integer );
514}

554/*
555** VectorArrayNormalize
556*
557* The inputs to this routing seem to always be close to length = 1.0 (about 0.6 to 2.0)
558* This means that we don't have to worry about zero length or enormously long vectors.
559*/
560static void VectorArrayNormalize(vec4_t *normals, unsigned int count)
561{
562// assert(count);
563
564#if idppc
565 {
566 float half = 0.5;
567 float one = 1.0;
568 float *components = (float *)normals;
569
570 // Vanilla PPC code, but since PPC has a reciprocal square root estimate instruction,
571 // runs *much* faster than calling sqrt(). We'll use a single Newton-Raphson
572 // refinement step to get a little more precision. This seems to yeild results
573 // that are correct to 3 decimal places and usually correct to at least 4 (sometimes 5).
574 // (That is, for the given input range of about 0.6 to 2.0).
575 do {
576 float x, y, z;
577 float B, y0, y1;
578
579 x = components[0];
580 y = components[1];
581 z = components[2];
582 components += 4;
583 B = x*x + y*y + z*z;
584
585#ifdef __GNUC__
586 asm("frsqrte %0,%1" : "=f" (y0) : "f" (B));
587#else
588 y0 = __frsqrte(B);
589#endif
590 y1 = y0 + half*y0*(one - B*y0*y0);
591
592 x = x * y1;
593 y = y * y1;
594 components[-4] = x;
595 z = z * y1;
596 components[-3] = y;
597 components[-2] = z;
598 } while(count--);
599 }
600#else // No assembly version for this architecture, or C_ONLY defined
601 // given the input, it's safe to call VectorNormalizeFast
602 while (count--) {
603 VectorNormalizeFast(normals[0]);
604 normals++;
605 }
606#endif
607
608}

612/*
613** LerpMeshVertexes
614*/
615#if idppc_altivec
616static void LerpMeshVertexes_altivec(md3Surface_t *surf, float backlerp)
617{
618 short *oldXyz, *newXyz, *oldNormals, *newNormals;
619 float *outXyz, *outNormal;
620 float oldXyzScale QALIGN(16);
621 float newXyzScale QALIGN(16);
622 float oldNormalScale QALIGN(16);
623 float newNormalScale QALIGN(16);
624 int vertNum;
625 unsigned lat, lng;
626 int numVerts;
627
628 outXyz = tess.xyz[tess.numVertexes];
629 outNormal = tess.normal[tess.numVertexes];
630
631 newXyz = (short *)((byte *)surf + surf->ofsXyzNormals)
632 + (backEnd.currentEntity->e.frame * surf->numVerts * 4);
633 newNormals = newXyz + 3;
634

647 vector unsigned char newNormalsStorePermute;
648 vector float zero;

651 newXyzScaleVec = *(vector float *)&newXyzScale;
652 newXyzScaleVec = vec_perm(newXyzScaleVec,newXyzScaleVec,newNormalsStorePermute);
653 newXyzScaleVec = vec_splat(newXyzScaleVec,0);
654 newNormalsLoadPermute = vec_lvsl(0,newXyz);
655 newNormalsStorePermute = vec_lvsr(0,outXyz);
656 zero = (vector float)vec_splat_s8(0);

658 // just copy the vertexes
659 //

672 newNormalsFloatVec = vec_perm(newNormalsFloatVec,newNormalsFloatVec,newNormalsStorePermute);
673 //outXyz[0] = newXyz[0] * newXyzScale;

678 lng = ( newNormals[0] & 0xff );
679 lat *= (FUNCTABLE_SIZE/256);
680 lng *= (FUNCTABLE_SIZE/256);
681
682 // decode X as cos( lat ) * sin( long )
683 // decode Y as sin( lat ) * sin( long )
684 // decode Z as cos( long )

686 outNormal[0] = tr.sinTable[(lat+(FUNCTABLE_SIZE/4))&FUNCTABLE_MASK] * tr.sinTable[lng];
687 outNormal[1] = tr.sinTable[lat] * tr.sinTable[lng];
688 outNormal[2] = tr.sinTable[(lng+(FUNCTABLE_SIZE/4))&FUNCTABLE_MASK];

690 vec_ste(newNormalsFloatVec,0,outXyz);

692 vec_ste(newNormalsFloatVec,8,outXyz);
693 }
694 } else {
695 //
696 // interpolate and copy the vertex and normal
697 //
698 oldXyz = (short *)((byte *)surf + surf->ofsXyzNormals)
699 + (backEnd.currentEntity->e.oldframe * surf->numVerts * 4);
700 oldNormals = oldXyz + 3;
701
702 oldXyzScale = MD3_XYZ_SCALE * backlerp;
703 oldNormalScale = backlerp;
704
705 for (vertNum=0 ; vertNum < numVerts ; vertNum++,
706 oldXyz += 4, newXyz += 4, oldNormals += 4, newNormals += 4,
707 outXyz += 4, outNormal += 4)
708 {
709 vec3_t uncompressedOldNormal, uncompressedNewNormal;
710
711 // interpolate the xyz
712 outXyz[0] = oldXyz[0] * oldXyzScale + newXyz[0] * newXyzScale;
713 outXyz[1] = oldXyz[1] * oldXyzScale + newXyz[1] * newXyzScale;
714 outXyz[2] = oldXyz[2] * oldXyzScale + newXyz[2] * newXyzScale;
715
716 // FIXME: interpolate lat/long instead?
717 lat = ( newNormals[0] >> 8 ) & 0xff;
718 lng = ( newNormals[0] & 0xff );
719 lat *= 4;
720 lng *= 4;
721 uncompressedNewNormal[0] = tr.sinTable[(lat+(FUNCTABLE_SIZE/4))&FUNCTABLE_MASK] * tr.sinTable[lng];
722 uncompressedNewNormal[1] = tr.sinTable[lat] * tr.sinTable[lng];
723 uncompressedNewNormal[2] = tr.sinTable[(lng+(FUNCTABLE_SIZE/4))&FUNCTABLE_MASK];
724
725 lat = ( oldNormals[0] >> 8 ) & 0xff;
726 lng = ( oldNormals[0] & 0xff );
727 lat *= 4;
728 lng *= 4;
729
730 uncompressedOldNormal[0] = tr.sinTable[(lat+(FUNCTABLE_SIZE/4))&FUNCTABLE_MASK] * tr.sinTable[lng];

861static void RB_SurfaceMesh(md3Surface_t *surface) {
862 int j;
863 float backlerp;
864 int *triangles;

878 LerpMeshVertexes (surface, backlerp);
879

881 indexes = surface->numTriangles * 3;
882 Bob = tess.numIndexes;
883 Doug = tess.numVertexes;
884 for (j = 0 ; j < indexes ; j++) {
885 tess.indexes[Bob + j] = Doug + triangles[j];
886 }

888
889 texCoords = (float *) ((byte *)surface + surface->ofsSt);
890
891 numVerts = surface->numVerts;
892 for ( j = 0; j < numVerts; j++ ) {
893 tess.texCoords[Doug + j][0][0] = texCoords[j*2+0];
894 tess.texCoords[Doug + j][0][1] = texCoords[j*2+1];
895 // FIXME: fill in lightmapST for completeness?
896 }
897
898 tess.numVertexes += surface->numVerts;

958static float LodErrorForVolume( vec3_t local, float radius ) {
959 vec3_t world;
960 float d;
961
962 // never let it go negative
963 if ( r_lodCurveError->value < 0 ) {
964 return 0;
965 }
966
967 world[0] = local[0] * backEnd.or.axis[0][0] + local[1] * backEnd.or.axis[1][0] +
968 local[2] * backEnd.or.axis[2][0] + backEnd.or.origin[0];
969 world[1] = local[0] * backEnd.or.axis[0][1] + local[1] * backEnd.or.axis[1][1] +
970 local[2] * backEnd.or.axis[2][1] + backEnd.or.origin[1];
971 world[2] = local[0] * backEnd.or.axis[0][2] + local[1] * backEnd.or.axis[1][2] +
972 local[2] * backEnd.or.axis[2][2] + backEnd.or.origin[2];
973
974 VectorSubtract( world, backEnd.viewParms.or.origin, world );
975 d = DotProduct( world, backEnd.viewParms.or.axis[0] );
976
977 if ( d < 0 ) {
978 d = -d;
979 }
980 d -= radius;
981 if ( d < 1 ) {
982 d = 1;
983 }
984
985 return r_lodCurveError->value / d;
986}

988/*
989=============
990RB_SurfaceGrid
991
992Just copy the grid of points and triangulate
993=============
994*/
995static void RB_SurfaceGrid( srfGridMesh_t *cv ) {
996 int i, j;
997 float *xyz;
998 float *texCoords;
999 float *normal;
1000 unsigned char *color;
1001 drawVert_t *dv;
1002 int rows, irows, vrows;
1003 int used;
1004 int widthTable[MAX_GRID_SIZE];
1005 int heightTable[MAX_GRID_SIZE];
1006 float lodError;
1007 int lodWidth, lodHeight;
1008 int numVertexes;
1009 int dlightBits;
1010 int *vDlightBits;
1011 qboolean needsNormal;
1012
1013 dlightBits = cv->dlightBits;
1014 tess.dlightBits |= dlightBits;
1015
1016 // determine the allowable discrepance
1017 lodError = LodErrorForVolume( cv->lodOrigin, cv->lodRadius );
1018
1019 // determine which rows and columns of the subdivision
1020 // we are actually going to use
1021 widthTable[0] = 0;
1022 lodWidth = 1;
1023 for ( i = 1 ; i < cv->width-1 ; i++ ) {
1024 if ( cv->widthLodError[i] <= lodError ) {
1025 widthTable[lodWidth] = i;
1026 lodWidth++;
1027 }
1028 }
1029 widthTable[lodWidth] = cv->width-1;
1030 lodWidth++;
1031
1032 heightTable[0] = 0;
1033 lodHeight = 1;
1034 for ( i = 1 ; i < cv->height-1 ; i++ ) {
1035 if ( cv->heightLodError[i] <= lodError ) {
1036 heightTable[lodHeight] = i;
1037 lodHeight++;
1038 }
1039 }
1040 heightTable[lodHeight] = cv->height-1;
1041 lodHeight++;
1042
1043
1044 // very large grids may have more points or indexes than can be fit
1045 // in the tess structure, so we may have to issue it in multiple passes
1046
1047 used = 0;
1048 while ( used < lodHeight - 1 ) {
1049 // see how many rows of both verts and indexes we can add without overflowing
1050 do {
1051 vrows = ( SHADER_MAX_VERTEXES - tess.numVertexes ) / lodWidth;
1052 irows = ( SHADER_MAX_INDEXES - tess.numIndexes ) / ( lodWidth * 6 );
1053
1054 // if we don't have enough space for at least one strip, flush the buffer
1055 if ( vrows < 2 || irows < 1 ) {
1056 RB_EndSurface();
1057 RB_BeginSurface(tess.shader, tess.fogNum );
1058 } else {
1059 break;
1060 }
1061 } while ( 1 );
1062
1063 rows = irows;
1064 if ( vrows < irows + 1 ) {
1065 rows = vrows - 1;
1066 }
1067 if ( used + rows > lodHeight ) {
1068 rows = lodHeight - used;
1069 }
1070
1071 numVertexes = tess.numVertexes;
1072
1073 xyz = tess.xyz[numVertexes];
1074 normal = tess.normal[numVertexes];
1075 texCoords = tess.texCoords[numVertexes][0];
1076 color = ( unsigned char * ) &tess.vertexColors[numVertexes];
1077 vDlightBits = &tess.vertexDlightBits[numVertexes];
1078 needsNormal = tess.shader->needsNormal;
1079
1080 for ( i = 0 ; i < rows ; i++ ) {
1081 for ( j = 0 ; j < lodWidth ; j++ ) {
1082 dv = cv->verts + heightTable[ used + i ] * cv->width
1083 + widthTable[ j ];
1084
1085 xyz[0] = dv->xyz[0];
1086 xyz[1] = dv->xyz[1];
1087 xyz[2] = dv->xyz[2];
1088 texCoords[0] = dv->st[0];
1089 texCoords[1] = dv->st[1];
1090 texCoords[2] = dv->lightmap[0];
1091 texCoords[3] = dv->lightmap[1];
1092 if ( needsNormal ) {
1093 normal[0] = dv->normal[0];
1094 normal[1] = dv->normal[1];
1095 normal[2] = dv->normal[2];
1096 }
1097 * ( unsigned int * ) color = * ( unsigned int * ) dv->color;
1098 *vDlightBits++ = dlightBits;
1099 xyz += 4;
1100 normal += 4;
1101 texCoords += 4;
1102 color += 4;
1103 }
1104 }
1105
1106
1107 // add the indexes
1108 {
1109 int numIndexes;
1110 int w, h;
1111
1112 h = rows - 1;
1113 w = lodWidth - 1;
1114 numIndexes = tess.numIndexes;
1115 for (i = 0 ; i < h ; i++) {
1116 for (j = 0 ; j < w ; j++) {
1117 int v1, v2, v3, v4;
1118
1119 // vertex order to be reckognized as tristrips
1120 v1 = numVertexes + i*lodWidth + j + 1;
1121 v2 = v1 - 1;
1122 v3 = v2 + lodWidth;
1123 v4 = v3 + 1;
1124
1125 tess.indexes[numIndexes] = v2;
1126 tess.indexes[numIndexes+1] = v3;
1127 tess.indexes[numIndexes+2] = v1;
1128
1129 tess.indexes[numIndexes+3] = v1;
1130 tess.indexes[numIndexes+4] = v3;
1131 tess.indexes[numIndexes+5] = v4;
1132 numIndexes += 6;
1133 }
1134 }
1135
1136 tess.numIndexes = numIndexes;
1137 }
1138
1139 tess.numVertexes += rows * lodWidth;
1140
1141 used += rows - 1;
1142 }
1143}

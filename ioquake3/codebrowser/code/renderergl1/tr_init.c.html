171/*
172** InitOpenGL
173**
174** This function is responsible for initializing a valid OpenGL subsystem. This
175** is done by calling GLimp_Init (which gives us a working OGL subsystem) then
176** setting variables, checking GL constants, and reporting the gfx system config
177** to the user.
178*/
179static void InitOpenGL( void )
180{
181 char renderer_buffer[1024];
182
183 //
184 // initialize OS specific portions of the renderer
185 //
186 // GLimp_Init directly or indirectly references the following cvars:
187 // - r_fullscreen
188 // - r_mode
189 // - r_(color|depth|stencil)bits
190 // - r_ignorehwgamma
191 // - r_gamma
192 //
193
194 if ( glConfig.vidWidth == 0 )
195 {
196 GLint temp;
197
198 GLimp_Init( qfalse );
199
200 strcpy( renderer_buffer, glConfig.renderer_string );
201 Q_strlwr( renderer_buffer );
202
203 // OpenGL driver constants
204 qglGetIntegerv( GL_MAX_TEXTURE_SIZE, &temp );
205 glConfig.maxTextureSize = temp;
206
207 // stubbed or broken drivers may have reported 0...
208 if ( glConfig.maxTextureSize <= 0 )
209 {
210 glConfig.maxTextureSize = 0;
211 }
212 }
213
214 // set default state
215 GL_SetDefaultState();
216}

218/*
219==================
220GL_CheckErrors
221==================
222*/
223void GL_CheckErrors( void ) {
224 int err;
225 char s[64];
226
227 err = qglGetError();
228 if ( err == GL_NO_ERROR ) {
229 return;
230 }
231 if ( r_ignoreGLErrors->integer ) {
232 return;
233 }
234 switch( err ) {
235 case GL_INVALID_ENUM:
236 strcpy( s, "GL_INVALID_ENUM" );
237 break;
238 case GL_INVALID_VALUE:
239 strcpy( s, "GL_INVALID_VALUE" );
240 break;
241 case GL_INVALID_OPERATION:
242 strcpy( s, "GL_INVALID_OPERATION" );
243 break;
244 case GL_STACK_OVERFLOW:
245 strcpy( s, "GL_STACK_OVERFLOW" );
246 break;
247 case GL_STACK_UNDERFLOW:
248 strcpy( s, "GL_STACK_UNDERFLOW" );
249 break;
250 case GL_OUT_OF_MEMORY:
251 strcpy( s, "GL_OUT_OF_MEMORY" );
252 break;
253 default:
254 Com_sprintf( s, sizeof(s), "%i", err);
255 break;
256 }
257
258 ri.Error( ERR_FATAL, "GL_CheckErrors: %s", s );
259}

265typedef struct vidmode_s
266{
267 const char *description;
268 int width, height;
269 float pixelAspect; // pixel width / height
270} vidmode_t;
271
272vidmode_t r_vidModes[] =
273{
274 { "Mode 0: 320x240", 320, 240, 1 },
275 { "Mode 1: 400x300", 400, 300, 1 },
276 { "Mode 2: 512x384", 512, 384, 1 },
277 { "Mode 3: 640x480", 640, 480, 1 },
278 { "Mode 4: 800x600", 800, 600, 1 },
279 { "Mode 5: 960x720", 960, 720, 1 },
280 { "Mode 6: 1024x768", 1024, 768, 1 },
281 { "Mode 7: 1152x864", 1152, 864, 1 },
282 { "Mode 8: 1280x1024", 1280, 1024, 1 },
283 { "Mode 9: 1600x1200", 1600, 1200, 1 },
284 { "Mode 10: 2048x1536", 2048, 1536, 1 },
285 { "Mode 11: 856x480 (wide)",856, 480, 1 }
286};
287static int s_numVidModes = ARRAY_LEN( r_vidModes );
288
289qboolean R_GetModeInfo( int *width, int *height, float *windowAspect, int mode ) {
290 vidmode_t *vm;
291 float pixelAspect;
292
293 if ( mode < -1 ) {
294 return qfalse;
295 }
296 if ( mode >= s_numVidModes ) {
297 return qfalse;
298 }
299
300 if ( mode == -1 ) {
301 *width = r_customwidth->integer;
302 *height = r_customheight->integer;
303 pixelAspect = r_customPixelAspect->value;
304 } else {
305 vm = &r_vidModes[mode];
306
307 *width = vm->width;
308 *height = vm->height;
309 pixelAspect = vm->pixelAspect;
310 }
311
312 *windowAspect = (float)*width / ( *height * pixelAspect );
313
314 return qtrue;
315}

352/*
353==================
354RB_ReadPixels
355
356Reads an image but takes care of alignment issues for reading RGB images.
357
358Reads a minimum offset for where the RGB data starts in the image from
359integer stored at pointer offset. When the function has returned the actual
360offset was written back to address offset. This address will always have an
361alignment of packAlign to ensure efficient copying.
362
363Stores the length of padding after a line of pixels to address padlen
364
365Return value must be freed with ri.Hunk_FreeTempMemory()
366==================
367*/
368
369byte *RB_ReadPixels(int x, int y, int width, int height, size_t *offset, int *padlen)
370{
371 byte *buffer, *bufstart;
372 int padwidth, linelen;
373 GLint packAlign;
374
375 qglGetIntegerv(GL_PACK_ALIGNMENT, &packAlign);
376
377 linelen = width * 3;
378 padwidth = PAD(linelen, packAlign);
379
380 // Allocate a few more bytes so that we can choose an alignment we like
381 buffer = ri.Hunk_AllocateTempMemory(padwidth * height + *offset + packAlign - 1);
382
383 bufstart = PADP((intptr_t) buffer + *offset, packAlign);
384 qglReadPixels(x, y, width, height, GL_RGB, GL_UNSIGNED_BYTE, bufstart);
385
386 *offset = bufstart - buffer;
387 *padlen = padwidth - linelen;
388
389 return buffer;
390}

392/*
393==================
394RB_TakeScreenshot
395==================
396*/
397void RB_TakeScreenshot(int x, int y, int width, int height, char *fileName)
398{
399 byte *allbuf, *buffer;
400 byte *srcptr, *destptr;
401 byte *endline, *endmem;
402 byte temp;
403
404 int linelen, padlen;
405 size_t offset = 18, memcount;
406
407 allbuf = RB_ReadPixels(x, y, width, height, &offset, &padlen);
408 buffer = allbuf + offset - 18;
409
410 Com_Memset (buffer, 0, 18);
411 buffer[2] = 2; // uncompressed type
412 buffer[12] = width & 255;
413 buffer[13] = width >> 8;
414 buffer[14] = height & 255;
415 buffer[15] = height >> 8;
416 buffer[16] = 24; // pixel size
417
418 // swap rgb to bgr and remove padding from line endings
419 linelen = width * 3;
420
421 srcptr = destptr = allbuf + offset;
422 endmem = srcptr + (linelen + padlen) * height;
423
424 while(srcptr < endmem)
425 {
426 endline = srcptr + linelen;
427
428 while(srcptr < endline)
429 {
430 temp = srcptr[0];
431 *destptr++ = srcptr[2];
432 *destptr++ = srcptr[1];
433 *destptr++ = temp;
434
435 srcptr += 3;
436 }
437
438 // Skip the pad
439 srcptr += padlen;
440 }
441
442 memcount = linelen * height;
443
444 // gamma correct
445 if(glConfig.deviceSupportsGamma)
446 R_GammaCorrect(allbuf + offset, memcount);
447
448 ri.FS_WriteFile(fileName, buffer, memcount + 18);
449
450 ri.Hunk_FreeTempMemory(allbuf);
451}

568/*
569====================
570R_LevelShot
571
572levelshots are specialized 128*128 thumbnails for
573the menu system, sampled down from full screen distorted images
574====================
575*/
576void R_LevelShot( void ) {
577 char checkname[MAX_OSPATH];
578 byte *buffer;
579 byte *source, *allsource;
580 byte *src, *dst;
581 size_t offset = 0;
582 int padlen;
583 int x, y;
584 int r, g, b;
585 float xScale, yScale;
586 int xx, yy;
587
588 Com_sprintf(checkname, sizeof(checkname), "levelshots/%s.tga", tr.world->baseName);
589
590 allsource = RB_ReadPixels(0, 0, glConfig.vidWidth, glConfig.vidHeight, &offset, &padlen);
591 source = allsource + offset;
592
593 buffer = ri.Hunk_AllocateTempMemory(128 * 128*3 + 18);
594 Com_Memset (buffer, 0, 18);
595 buffer[2] = 2; // uncompressed type
596 buffer[12] = 128;
597 buffer[14] = 128;
598 buffer[16] = 24; // pixel size
599
600 // resample from source
601 xScale = glConfig.vidWidth / 512.0f;
602 yScale = glConfig.vidHeight / 384.0f;
603 for ( y = 0 ; y < 128 ; y++ ) {
604 for ( x = 0 ; x < 128 ; x++ ) {
605 r = g = b = 0;
606 for ( yy = 0 ; yy < 3 ; yy++ ) {
607 for ( xx = 0 ; xx < 4 ; xx++ ) {
608 src = source + (3 * glConfig.vidWidth + padlen) * (int)((y*3 + yy) * yScale) +
609 3 * (int) ((x*4 + xx) * xScale);
610 r += src[0];
611 g += src[1];
612 b += src[2];
613 }
614 }
615 dst = buffer + 18 + 3 * ( y * 128 + x );
616 dst[0] = b / 12;
617 dst[1] = g / 12;
618 dst[2] = r / 12;
619 }
620 }
621
622 // gamma correct
623 if ( glConfig.deviceSupportsGamma ) {
624 R_GammaCorrect( buffer + 18, 128 * 128 * 3 );
625 }
626
627 ri.FS_WriteFile( checkname, buffer, 128 * 128*3 + 18 );
628
629 ri.Hunk_FreeTempMemory(buffer);
630 ri.Hunk_FreeTempMemory(allsource);
631
632 ri.Printf( PRINT_ALL, "Wrote %s\n", checkname );
633}

635/*
636==================
637R_ScreenShot_f
638
639screenshot
640screenshot [silent]
641screenshot [levelshot]
642screenshot [filename]
643
644Doesn't print the pacifier message if there is a second arg
645==================
646*/
647void R_ScreenShot_f (void) {
648 char checkname[MAX_OSPATH];
649 static int lastNumber = -1;
650 qboolean silent;
651
652 if ( !strcmp( ri.Cmd_Argv(1), "levelshot" ) ) {
653 R_LevelShot();
654 return;
655 }
656
657 if ( !strcmp( ri.Cmd_Argv(1), "silent" ) ) {
658 silent = qtrue;
659 } else {
660 silent = qfalse;
661 }
662
663 if ( ri.Cmd_Argc() == 2 && !silent ) {
664 // explicit filename
665 Com_sprintf( checkname, MAX_OSPATH, "screenshots/%s.tga", ri.Cmd_Argv( 1 ) );
666 } else {
667 // scan for a free filename
668
669 // if we have saved a previous screenshot, don't scan
670 // again, because recording demo avis can involve
671 // thousands of shots
672 if ( lastNumber == -1 ) {
673 lastNumber = 0;
674 }
675 // scan for a free number
676 for ( ; lastNumber <= 9999 ; lastNumber++ ) {
677 R_ScreenshotFilename( lastNumber, checkname );
678
679 if (!ri.FS_FileExists( checkname ))
680 {
681 break; // file doesn't exist
682 }
683 }
684
685 if ( lastNumber >= 9999 ) {
686 ri.Printf (PRINT_ALL, "ScreenShot: Couldn't create a file\n");
687 return;
688 }
689
690 lastNumber++;
691 }
692
693 R_TakeScreenshot( 0, 0, glConfig.vidWidth, glConfig.vidHeight, checkname, qfalse );
694
695 if ( !silent ) {
696 ri.Printf (PRINT_ALL, "Wrote %s\n", checkname);
697 }
698}

755/*
756==================
757RB_TakeVideoFrameCmd
758==================
759*/
760const void *RB_TakeVideoFrameCmd( const void *data )
761{
762 const videoFrameCommand_t *cmd;
763 byte *cBuf;
764 size_t memcount, linelen;
765 int padwidth, avipadwidth, padlen, avipadlen;
766 GLint packAlign;
767
768 cmd = (const videoFrameCommand_t *)data;
769
770 qglGetIntegerv(GL_PACK_ALIGNMENT, &packAlign);
771
772 linelen = cmd->width * 3;
773
774 // Alignment stuff for glReadPixels
775 padwidth = PAD(linelen, packAlign);
776 padlen = padwidth - linelen;
777 // AVI line padding
778 avipadwidth = PAD(linelen, AVI_LINE_PADDING);
779 avipadlen = avipadwidth - linelen;
780
781 cBuf = PADP(cmd->captureBuffer, packAlign);
782
783 qglReadPixels(0, 0, cmd->width, cmd->height, GL_RGB,
784 GL_UNSIGNED_BYTE, cBuf);
785
786 memcount = padwidth * cmd->height;
787
788 // gamma correct
789 if(glConfig.deviceSupportsGamma)
790 R_GammaCorrect(cBuf, memcount);
791
792 if(cmd->motionJpeg)
793 {
794 memcount = RE_SaveJPGToBuffer(cmd->encodeBuffer, linelen * cmd->height,
795 r_aviMotionJpegQuality->integer,
796 cmd->width, cmd->height, cBuf, padlen);
797 ri.CL_WriteAVIVideoFrame(cmd->encodeBuffer, memcount);
798 }
799 else
800 {
801 byte *lineend, *memend;
802 byte *srcptr, *destptr;
803
804 srcptr = cBuf;
805 destptr = cmd->encodeBuffer;
806 memend = srcptr + memcount;
807
808 // swap R and B and remove line paddings
809 while(srcptr < memend)
810 {
811 lineend = srcptr + linelen;
812 while(srcptr < lineend)
813 {
814 *destptr++ = srcptr[2];
815 *destptr++ = srcptr[1];
816 *destptr++ = srcptr[0];
817 srcptr += 3;
818 }
819
820 Com_Memset(destptr, '\0', avipadlen);
821 destptr += avipadlen;
822
823 srcptr += padlen;
824 }
825
826 ri.CL_WriteAVIVideoFrame(cmd->encodeBuffer, avipadwidth * cmd->height);
827 }
828
829 return (const void *)(cmd + 1);
830}

834/*
835** GL_SetDefaultState
836*/
837void GL_SetDefaultState( void )
838{
839 qglClearDepth( 1.0f );
840
841 qglCullFace(GL_FRONT);
842
843 qglColor4f (1,1,1,1);
844
845 // initialize downstream texture unit if we're running
846 // in a multitexture environment
847 if ( qglActiveTextureARB ) {
848 GL_SelectTexture( 1 );
849 GL_TextureMode( r_textureMode->string );
850 GL_TexEnv( GL_MODULATE );
851 qglDisable( GL_TEXTURE_2D );
852 GL_SelectTexture( 0 );
853 }
854
855 qglEnable(GL_TEXTURE_2D);
856 GL_TextureMode( r_textureMode->string );
857 GL_TexEnv( GL_MODULATE );
858
859 qglShadeModel( GL_SMOOTH );
860 qglDepthFunc( GL_LEQUAL );
861
862 // the vertex array is always enabled, but the color and texture
863 // arrays are enabled and disabled around the compiled vertex array call
864 qglEnableClientState (GL_VERTEX_ARRAY);
865
866 //
867 // make sure our GL state vector is set correctly
868 //
869 glState.glStateBits = GLS_DEPTHTEST_DISABLE | GLS_DEPTHMASK_TRUE;
870
871 qglPolygonMode (GL_FRONT_AND_BACK, GL_FILL);
872 qglDepthMask( GL_TRUE );
873 qglDisable( GL_DEPTH_TEST );
874 qglEnable( GL_SCISSOR_TEST );
875 qglDisable( GL_CULL_FACE );
876 qglDisable( GL_BLEND );
877}

1157/*
1158===============
1159R_Init
1160===============
1161*/
1162void R_Init( void ) {
1163 int err;
1164 int i;
1165 byte *ptr;
1166
1167 ri.Printf( PRINT_ALL, "----- R_Init -----\n" );
1168
1169 // clear all our internal state
1170 Com_Memset( &tr, 0, sizeof( tr ) );
1171 Com_Memset( &backEnd, 0, sizeof( backEnd ) );
1172 Com_Memset( &tess, 0, sizeof( tess ) );
1173
1174 if(sizeof(glconfig_t) != 11332)
1175 ri.Error( ERR_FATAL, "Mod ABI incompatible: sizeof(glconfig_t) == %u != 11332", (unsigned int) sizeof(glconfig_t));
1176
1177// Swap_Init();
1178
1179 if ( (intptr_t)tess.xyz & 15 ) {
1180 ri.Printf( PRINT_WARNING, "tess.xyz not 16 byte aligned\n" );
1181 }
1182 Com_Memset( tess.constantColor255, 255, sizeof( tess.constantColor255 ) );
1183
1184 //
1185 // init function tables
1186 //
1187 for ( i = 0; i < FUNCTABLE_SIZE; i++ )
1188 {
1189 tr.sinTable[i] = sin( DEG2RAD( i * 360.0f / ( ( float ) ( FUNCTABLE_SIZE - 1 ) ) ) );
1190 tr.squareTable[i] = ( i < FUNCTABLE_SIZE/2 ) ? 1.0f : -1.0f;
1191 tr.sawToothTable[i] = (float)i / FUNCTABLE_SIZE;
1192 tr.inverseSawToothTable[i] = 1.0f - tr.sawToothTable[i];
1193
1194 if ( i < FUNCTABLE_SIZE / 2 )
1195 {
1196 if ( i < FUNCTABLE_SIZE / 4 )
1197 {
1198 tr.triangleTable[i] = ( float ) i / ( FUNCTABLE_SIZE / 4 );
1199 }
1200 else
1201 {
1202 tr.triangleTable[i] = 1.0f - tr.triangleTable[i-FUNCTABLE_SIZE / 4];
1203 }
1204 }
1205 else
1206 {
1207 tr.triangleTable[i] = -tr.triangleTable[i-FUNCTABLE_SIZE/2];
1208 }
1209 }
1210
1211 R_InitFogTable();
1212
1213 R_NoiseInit();
1214
1215 R_Register();
1216
1217 max_polys = r_maxpolys->integer;
1218 if (max_polys < MAX_POLYS)
1219 max_polys = MAX_POLYS;
1220
1221 max_polyverts = r_maxpolyverts->integer;
1222 if (max_polyverts < MAX_POLYVERTS)
1223 max_polyverts = MAX_POLYVERTS;
1224
1225 ptr = ri.Hunk_Alloc( sizeof( *backEndData ) + sizeof(srfPoly_t) * max_polys + sizeof(polyVert_t) * max_polyverts, h_low);
1226 backEndData = (backEndData_t *) ptr;
1227 backEndData->polys = (srfPoly_t *) ((char *) ptr + sizeof( *backEndData ));
1228 backEndData->polyVerts = (polyVert_t *) ((char *) ptr + sizeof( *backEndData ) + sizeof(srfPoly_t) * max_polys);
1229 R_InitNextFrame();
1230
1231 InitOpenGL();
1232
1233 R_InitImages();
1234
1235 R_InitShaders();
1236
1237 R_InitSkins();
1238
1239 R_ModelInit();
1240
1241 R_InitFreeType();
1242
1243
1244 err = qglGetError();
1245 if ( err != GL_NO_ERROR )
1246 ri.Printf (PRINT_ALL, "glGetError() = 0x%x\n", err);
1247
1248 // print info
1249 GfxInfo_f();
1250 ri.Printf( PRINT_ALL, "----- finished R_Init -----\n" );
1251}

36/*
37================
38R_ArrayElementDiscrete
39
40This is just for OpenGL conformance testing, it should never be the fastest
41================
42*/
43static void APIENTRY R_ArrayElementDiscrete( GLint index ) {
44 qglColor4ubv( tess.svars.colors[ index ] );
45 if ( glState.currenttmu ) {
46 qglMultiTexCoord2fARB( 0, tess.svars.texcoords[ 0 ][ index ][0], tess.svars.texcoords[ 0 ][ index ][1] );
47 qglMultiTexCoord2fARB( 1, tess.svars.texcoords[ 1 ][ index ][0], tess.svars.texcoords[ 1 ][ index ][1] );
48 } else {
49 qglTexCoord2fv( tess.svars.texcoords[ 0 ][ index ] );
50 }
51 qglVertex3fv( tess.xyz[ index ] );
52}

54/*
55===================
56R_DrawStripElements
57
58===================
59*/
60static int c_vertexes; // for seeing how long our average strips are
61static int c_begins;
62static void R_DrawStripElements( int numIndexes, const glIndex_t *indexes, void ( APIENTRY *element )(GLint) ) {
63 int i;
64 int last[3] = { -1, -1, -1 };
65 qboolean even;
66
67 c_begins++;
68
69 if ( numIndexes <= 0 ) {
70 return;
71 }
72
73 qglBegin( GL_TRIANGLE_STRIP );
74
75 // prime the strip
76 element( indexes[0] );
77 element( indexes[1] );
78 element( indexes[2] );
79 c_vertexes += 3;
80
81 last[0] = indexes[0];
82 last[1] = indexes[1];
83 last[2] = indexes[2];
84
85 even = qfalse;
86
87 for ( i = 3; i < numIndexes; i += 3 )
88 {
89 // odd numbered triangle in potential strip
90 if ( !even )
91 {
92 // check previous triangle to see if we're continuing a strip
93 if ( ( indexes[i+0] == last[2] ) && ( indexes[i+1] == last[1] ) )
94 {
95 element( indexes[i+2] );
96 c_vertexes++;
97 assert( indexes[i+2] < tess.numVertexes );
98 even = qtrue;
99 }
100 // otherwise we're done with this strip so finish it and start
101 // a new one
102 else
103 {
104 qglEnd();
105
106 qglBegin( GL_TRIANGLE_STRIP );
107 c_begins++;
108
109 element( indexes[i+0] );
110 element( indexes[i+1] );
111 element( indexes[i+2] );
112
113 c_vertexes += 3;
114
115 even = qfalse;
116 }
117 }
118 else
119 {
120 // check previous triangle to see if we're continuing a strip
121 if ( ( last[2] == indexes[i+1] ) && ( last[0] == indexes[i+0] ) )
122 {
123 element( indexes[i+2] );
124 c_vertexes++;
125
126 even = qfalse;
127 }
128 // otherwise we're done with this strip so finish it and start
129 // a new one
130 else
131 {
132 qglEnd();
133
134 qglBegin( GL_TRIANGLE_STRIP );
135 c_begins++;
136
137 element( indexes[i+0] );
138 element( indexes[i+1] );
139 element( indexes[i+2] );
140 c_vertexes += 3;
141
142 even = qfalse;
143 }
144 }
145
146 // cache the last three vertices
147 last[0] = indexes[i+0];
148 last[1] = indexes[i+1];
149 last[2] = indexes[i+2];
150 }
151
152 qglEnd();
153}

157/*
158==================
159R_DrawElements
160
161Optionally performs our own glDrawElements that looks for strip conditions
162instead of using the single glDrawElements call that may be inefficient
163without compiled vertex arrays.
164==================
165*/
166static void R_DrawElements( int numIndexes, const glIndex_t *indexes ) {
167 int primitives;
168
169 primitives = r_primitives->integer;
170
171 // default is to use triangles if compiled vertex arrays are present
172 if ( primitives == 0 ) {
173 if ( qglLockArraysEXT ) {
174 primitives = 2;
175 } else {
176 primitives = 1;
177 }
178 }
179
180
181 if ( primitives == 2 ) {
182 qglDrawElements( GL_TRIANGLES,
183 numIndexes,
184 GL_INDEX_TYPE,
185 indexes );
186 return;
187 }
188
189 if ( primitives == 1 ) {
190 R_DrawStripElements( numIndexes, indexes, qglArrayElement );
191 return;
192 }
193
194 if ( primitives == 3 ) {
195 R_DrawStripElements( numIndexes, indexes, R_ArrayElementDiscrete );
196 return;
197 }
198
199 // anything else will cause no drawing
200}

214/*
215=================
216R_BindAnimatedImage
217
218=================
219*/
220static void R_BindAnimatedImage( textureBundle_t *bundle ) {
221 int64_t index;
222
223 if ( bundle->isVideoMap ) {
224 ri.CIN_RunCinematic(bundle->videoMapHandle);
225 ri.CIN_UploadCinematic(bundle->videoMapHandle);
226 return;
227 }
228
229 if ( bundle->numImageAnimations <= 1 ) {
230 GL_Bind( bundle->image[0] );
231 return;
232 }
233
234 // it is necessary to do this messy calc to make sure animations line up
235 // exactly with waveforms of the same frequency
236 index = tess.shaderTime * bundle->imageAnimationSpeed * FUNCTABLE_SIZE;
237 index >>= FUNCTABLE_SIZE2;
238
239 if ( index < 0 ) {
240 index = 0; // may happen with shader time offsets
241 }
242 index %= bundle->numImageAnimations;
243
244 GL_Bind( bundle->image[ index ] );
245}

247/*
248================
249DrawTris
250
251Draws triangle outlines for debugging
252================
253*/
254static void DrawTris (shaderCommands_t *input) {
255 GL_Bind( tr.whiteImage );
256 qglColor3f (1,1,1);
257
258 GL_State( GLS_POLYMODE_LINE | GLS_DEPTHMASK_TRUE );
259 qglDepthRange( 0, 0 );
260
261 qglDisableClientState (GL_COLOR_ARRAY);
262 qglDisableClientState (GL_TEXTURE_COORD_ARRAY);
263
264 qglVertexPointer (3, GL_FLOAT, 16, input->xyz); // padded for SIMD
265
266 if (qglLockArraysEXT) {
267 qglLockArraysEXT(0, input->numVertexes);
268 GLimp_LogComment( "glLockArraysEXT\n" );
269 }
270
271 R_DrawElements( input->numIndexes, input->indexes );
272
273 if (qglUnlockArraysEXT) {
274 qglUnlockArraysEXT();
275 GLimp_LogComment( "glUnlockArraysEXT\n" );
276 }
277 qglDepthRange( 0, 1 );
278}

283DrawNormals
284
285Draws vertex normals for debugging

287*/
288static void DrawNormals (shaderCommands_t *input) {
289 int i;
290 vec3_t temp;
291
292 GL_Bind( tr.whiteImage );
293 qglColor3f (1,1,1);
294 qglDepthRange( 0, 0 ); // never occluded
295 GL_State( GLS_POLYMODE_LINE | GLS_DEPTHMASK_TRUE );
296
297 qglBegin (GL_LINES);
298 for (i = 0 ; i < input->numVertexes ; i++) {
299 qglVertex3fv (input->xyz[i]);
300 VectorMA (input->xyz[i], 2, input->normal[i], temp);
301 qglVertex3fv (temp);
302 }
303 qglEnd ();
304
305 qglDepthRange( 0, 1 );
306}

308/*
309==============
310RB_BeginSurface
311
312We must set some things up before beginning any tesselation,
313because a surface may be forced to perform a RB_End due
314to overflow.
315==============
316*/
317void RB_BeginSurface( shader_t *shader, int fogNum ) {
318
319 shader_t *state = (shader->remappedShader) ? shader->remappedShader : shader;
320
321 tess.numIndexes = 0;
322 tess.numVertexes = 0;
323 tess.shader = state;
324 tess.fogNum = fogNum;
325 tess.dlightBits = 0; // will be OR'd in by surface functions
326 tess.xstages = state->stages;
327 tess.numPasses = state->numUnfoggedPasses;
328 tess.currentStageIteratorFunc = state->optimalStageIteratorFunc;
329
330 tess.shaderTime = backEnd.refdef.floatTime - tess.shader->timeOffset;
331 if (tess.shader->clampTime && tess.shaderTime >= tess.shader->clampTime) {
332 tess.shaderTime = tess.shader->clampTime;
333 }
334
335
336}

338/*
339===================
340DrawMultitextured
341
342output = t0 * t1 or t0 + t1
343
344t0 = most upstream according to spec
345t1 = most downstream according to spec
346===================
347*/
348static void DrawMultitextured( shaderCommands_t *input, int stage ) {
349 shaderStage_t *pStage;
350
351 pStage = tess.xstages[stage];
352
353 GL_State( pStage->stateBits );
354
355 // this is an ugly hack to work around a GeForce driver
356 // bug with multitexture and clip planes
357 if ( backEnd.viewParms.isPortal ) {
358 qglPolygonMode( GL_FRONT_AND_BACK, GL_FILL );
359 }
360
361 //
362 // base
363 //
364 GL_SelectTexture( 0 );
365 qglTexCoordPointer( 2, GL_FLOAT, 0, input->svars.texcoords[0] );
366 R_BindAnimatedImage( &pStage->bundle[0] );
367
368 //
369 // lightmap/secondary pass
370 //
371 GL_SelectTexture( 1 );
372 qglEnable( GL_TEXTURE_2D );
373 qglEnableClientState( GL_TEXTURE_COORD_ARRAY );
374
375 if ( r_lightmap->integer ) {
376 GL_TexEnv( GL_REPLACE );
377 } else {
378 GL_TexEnv( tess.shader->multitextureEnv );
379 }
380
381 qglTexCoordPointer( 2, GL_FLOAT, 0, input->svars.texcoords[1] );
382
383 R_BindAnimatedImage( &pStage->bundle[1] );
384
385 R_DrawElements( input->numIndexes, input->indexes );
386
387 //
388 // disable texturing on TEXTURE1, then select TEXTURE0
389 //
390 //qglDisableClientState( GL_TEXTURE_COORD_ARRAY );
391 qglDisable( GL_TEXTURE_2D );
392
393 GL_SelectTexture( 0 );
394}

398/*
399===================
400ProjectDlightTexture
401
402Perform dynamic lighting with another rendering pass
403===================
404*/
405#if idppc_altivec
406static void ProjectDlightTexture_altivec( void ) {
407 int i, l;
408 vec_t origin0, origin1, origin2;
409 float texCoords0, texCoords1;
410 vector float floatColorVec0, floatColorVec1;
411 vector float modulateVec, colorVec, zero;
412 vector short colorShort;
413 vector signed int colorInt;
414 vector unsigned char floatColorVecPerm, modulatePerm, colorChar;
415 vector unsigned char vSel = VECCONST_UINT8(0x00, 0x00, 0x00, 0xff,
416 0x00, 0x00, 0x00, 0xff,
417 0x00, 0x00, 0x00, 0xff,
418 0x00, 0x00, 0x00, 0xff);
419 float *texCoords;
420 byte *colors;
421 byte clipBits[SHADER_MAX_VERTEXES];
422 float texCoordsArray[SHADER_MAX_VERTEXES][2];
423 byte colorArray[SHADER_MAX_VERTEXES][4];
424 glIndex_t hitIndexes[SHADER_MAX_INDEXES];
425 int numIndexes;
426 float scale;
427 float radius;
428 vec3_t floatColor;
429 float modulate = 0.0f;
430
431 if ( !backEnd.refdef.num_dlights ) {
432 return;
433 }
434
435 // There has to be a better way to do this so that floatColor
436 // and/or modulate are already 16-byte aligned.
437 floatColorVecPerm = vec_lvsl(0,(float *)floatColor);
438 modulatePerm = vec_lvsl(0,(float *)&modulate);
439 modulatePerm = (vector unsigned char)vec_splat((vector unsigned int)modulatePerm,0);
440 zero = (vector float)vec_splat_s8(0);
441
442 for ( l = 0 ; l < backEnd.refdef.num_dlights ; l++ ) {
443 dlight_t *dl;
444
445 if ( !( tess.dlightBits & ( 1 << l ) ) ) {
446 continue; // this surface definately doesn't have any of this light
447 }
448 texCoords = texCoordsArray[0];
449 colors = colorArray[0];
450
451 dl = &backEnd.refdef.dlights[l];
452 origin0 = dl->transformed[0];
453 origin1 = dl->transformed[1];
454 origin2 = dl->transformed[2];
455 radius = dl->radius;
456 scale = 1.0f / radius;
457
458 if(r_greyscale->integer)
459 {
460 float luminance;
461
462 luminance = LUMA(dl->color[0], dl->color[1], dl->color[2]) * 255.0f;
463 floatColor[0] = floatColor[1] = floatColor[2] = luminance;
464 }
465 else if(r_greyscale->value)
466 {
467 float luminance;
468
469 luminance = LUMA(dl->color[0], dl->color[1], dl->color[2]) * 255.0f;
470 floatColor[0] = LERP(dl->color[0] * 255.0f, luminance, r_greyscale->value);
471 floatColor[1] = LERP(dl->color[1] * 255.0f, luminance, r_greyscale->value);
472 floatColor[2] = LERP(dl->color[2] * 255.0f, luminance, r_greyscale->value);
473 }
474 else
475 {
476 floatColor[0] = dl->color[0] * 255.0f;
477 floatColor[1] = dl->color[1] * 255.0f;
478 floatColor[2] = dl->color[2] * 255.0f;
479 }
480 floatColorVec0 = vec_ld(0, floatColor);
481 floatColorVec1 = vec_ld(11, floatColor);
482 floatColorVec0 = vec_perm(floatColorVec0,floatColorVec0,floatColorVecPerm);
483 for ( i = 0 ; i < tess.numVertexes ; i++, texCoords += 2, colors += 4 ) {
484 int clip = 0;
485 vec_t dist0, dist1, dist2;
486
487 dist0 = origin0 - tess.xyz[i][0];
488 dist1 = origin1 - tess.xyz[i][1];
489 dist2 = origin2 - tess.xyz[i][2];
490
491 backEnd.pc.c_dlightVertexes++;
492
493 texCoords0 = 0.5f + dist0 * scale;
494 texCoords1 = 0.5f + dist1 * scale;
495
496 if( !r_dlightBacks->integer &&
497 // dist . tess.normal[i]
498 ( dist0 * tess.normal[i][0] +
499 dist1 * tess.normal[i][1] +
500 dist2 * tess.normal[i][2] ) < 0.0f ) {
501 clip = 63;
502 } else {
503 if ( texCoords0 < 0.0f ) {
504 clip |= 1;
505 } else if ( texCoords0 > 1.0f ) {
506 clip |= 2;
507 }
508 if ( texCoords1 < 0.0f ) {
509 clip |= 4;
510 } else if ( texCoords1 > 1.0f ) {
511 clip |= 8;
512 }
513 texCoords[0] = texCoords0;
514 texCoords[1] = texCoords1;
515
516 // modulate the strength based on the height and color
517 if ( dist2 > radius ) {
518 clip |= 16;
519 modulate = 0.0f;
520 } else if ( dist2 < -radius ) {
521 clip |= 32;
522 modulate = 0.0f;
523 } else {
524 dist2 = Q_fabs(dist2);
525 if ( dist2 < radius * 0.5f ) {
526 modulate = 1.0f;
527 } else {
528 modulate = 2.0f * (radius - dist2) * scale;
529 }
530 }
531 }
532 clipBits[i] = clip;
533
534 modulateVec = vec_ld(0,(float *)&modulate);
535 modulateVec = vec_perm(modulateVec,modulateVec,modulatePerm);
536 colorVec = vec_madd(floatColorVec0,modulateVec,zero);
537 colorInt = vec_cts(colorVec,0); // RGBx
538 colorShort = vec_pack(colorInt,colorInt); // RGBxRGBx
539 colorChar = vec_packsu(colorShort,colorShort); // RGBxRGBxRGBxRGBx
540 colorChar = vec_sel(colorChar,vSel,vSel); // RGBARGBARGBARGBA replace alpha with 255
541 vec_ste((vector unsigned int)colorChar,0,(unsigned int *)colors); // store color
542 }
543
544 // build a list of triangles that need light
545 numIndexes = 0;
546 for ( i = 0 ; i < tess.numIndexes ; i += 3 ) {
547 int a, b, c;
548
549 a = tess.indexes[i];
550 b = tess.indexes[i+1];
551 c = tess.indexes[i+2];
552 if ( clipBits[a] & clipBits[b] & clipBits[c] ) {
553 continue; // not lighted
554 }
555 hitIndexes[numIndexes] = a;
556 hitIndexes[numIndexes+1] = b;
557 hitIndexes[numIndexes+2] = c;
558 numIndexes += 3;
559 }
560
561 if ( !numIndexes ) {
562 continue;
563 }
564
565 qglEnableClientState( GL_TEXTURE_COORD_ARRAY );
566 qglTexCoordPointer( 2, GL_FLOAT, 0, texCoordsArray[0] );
567
568 qglEnableClientState( GL_COLOR_ARRAY );
569 qglColorPointer( 4, GL_UNSIGNED_BYTE, 0, colorArray );
570
571 GL_Bind( tr.dlightImage );
572 // include GLS_DEPTHFUNC_EQUAL so alpha tested surfaces don't add light
573 // where they aren't rendered
574 if ( dl->additive ) {
575 GL_State( GLS_SRCBLEND_ONE | GLS_DSTBLEND_ONE | GLS_DEPTHFUNC_EQUAL );
576 }
577 else {
578 GL_State( GLS_SRCBLEND_DST_COLOR | GLS_DSTBLEND_ONE | GLS_DEPTHFUNC_EQUAL );
579 }
580 R_DrawElements( numIndexes, hitIndexes );
581 backEnd.pc.c_totalIndexes += numIndexes;
582 backEnd.pc.c_dlightIndexes += numIndexes;
583 }
584}
585#endif

588static void ProjectDlightTexture_scalar( void ) {
589 int i, l;
590 vec3_t origin;
591 float *texCoords;
592 byte *colors;
593 byte clipBits[SHADER_MAX_VERTEXES];
594 float texCoordsArray[SHADER_MAX_VERTEXES][2];
595 byte colorArray[SHADER_MAX_VERTEXES][4];
596 glIndex_t hitIndexes[SHADER_MAX_INDEXES];
597 int numIndexes;
598 float scale;
599 float radius;
600 vec3_t floatColor;
601 float modulate = 0.0f;
602
603 if ( !backEnd.refdef.num_dlights ) {
604 return;
605 }
606
607 for ( l = 0 ; l < backEnd.refdef.num_dlights ; l++ ) {
608 dlight_t *dl;
609
610 if ( !( tess.dlightBits & ( 1 << l ) ) ) {
611 continue; // this surface definately doesn't have any of this light
612 }
613 texCoords = texCoordsArray[0];
614 colors = colorArray[0];
615
616 dl = &backEnd.refdef.dlights[l];
617 VectorCopy( dl->transformed, origin );
618 radius = dl->radius;
619 scale = 1.0f / radius;
620
621 if(r_greyscale->integer)
622 {
623 float luminance;
624
625 luminance = LUMA(dl->color[0], dl->color[1], dl->color[2]) * 255.0f;
626 floatColor[0] = floatColor[1] = floatColor[2] = luminance;
627 }
628 else if(r_greyscale->value)
629 {
630 float luminance;
631
632 luminance = LUMA(dl->color[0], dl->color[1], dl->color[2]) * 255.0f;
633 floatColor[0] = LERP(dl->color[0] * 255.0f, luminance, r_greyscale->value);
634 floatColor[1] = LERP(dl->color[1] * 255.0f, luminance, r_greyscale->value);
635 floatColor[2] = LERP(dl->color[2] * 255.0f, luminance, r_greyscale->value);
636 }
637 else
638 {
639 floatColor[0] = dl->color[0] * 255.0f;
640 floatColor[1] = dl->color[1] * 255.0f;
641 floatColor[2] = dl->color[2] * 255.0f;
642 }
643
644 for ( i = 0 ; i < tess.numVertexes ; i++, texCoords += 2, colors += 4 ) {
645 int clip = 0;
646 vec3_t dist;
647
648 VectorSubtract( origin, tess.xyz[i], dist );
649
650 backEnd.pc.c_dlightVertexes++;
651
652 texCoords[0] = 0.5f + dist[0] * scale;
653 texCoords[1] = 0.5f + dist[1] * scale;
654
655 if( !r_dlightBacks->integer &&
656 // dist . tess.normal[i]
657 ( dist[0] * tess.normal[i][0] +
658 dist[1] * tess.normal[i][1] +
659 dist[2] * tess.normal[i][2] ) < 0.0f ) {
660 clip = 63;
661 } else {
662 if ( texCoords[0] < 0.0f ) {
663 clip |= 1;
664 } else if ( texCoords[0] > 1.0f ) {
665 clip |= 2;
666 }
667 if ( texCoords[1] < 0.0f ) {
668 clip |= 4;
669 } else if ( texCoords[1] > 1.0f ) {
670 clip |= 8;
671 }
672 texCoords[0] = texCoords[0];
673 texCoords[1] = texCoords[1];
674
675 // modulate the strength based on the height and color
676 if ( dist[2] > radius ) {
677 clip |= 16;
678 modulate = 0.0f;
679 } else if ( dist[2] < -radius ) {
680 clip |= 32;
681 modulate = 0.0f;
682 } else {
683 dist[2] = Q_fabs(dist[2]);
684 if ( dist[2] < radius * 0.5f ) {
685 modulate = 1.0f;
686 } else {
687 modulate = 2.0f * (radius - dist[2]) * scale;
688 }
689 }
690 }
691 clipBits[i] = clip;
692 colors[0] = ri.ftol(floatColor[0] * modulate);
693 colors[1] = ri.ftol(floatColor[1] * modulate);
694 colors[2] = ri.ftol(floatColor[2] * modulate);
695 colors[3] = 255;
696 }
697
698 // build a list of triangles that need light
699 numIndexes = 0;
700 for ( i = 0 ; i < tess.numIndexes ; i += 3 ) {
701 int a, b, c;
702
703 a = tess.indexes[i];
704 b = tess.indexes[i+1];
705 c = tess.indexes[i+2];
706 if ( clipBits[a] & clipBits[b] & clipBits[c] ) {
707 continue; // not lighted
708 }
709 hitIndexes[numIndexes] = a;
710 hitIndexes[numIndexes+1] = b;
711 hitIndexes[numIndexes+2] = c;
712 numIndexes += 3;
713 }
714
715 if ( !numIndexes ) {
716 continue;
717 }
718
719 qglEnableClientState( GL_TEXTURE_COORD_ARRAY );
720 qglTexCoordPointer( 2, GL_FLOAT, 0, texCoordsArray[0] );
721
722 qglEnableClientState( GL_COLOR_ARRAY );
723 qglColorPointer( 4, GL_UNSIGNED_BYTE, 0, colorArray );
724
725 GL_Bind( tr.dlightImage );
726 // include GLS_DEPTHFUNC_EQUAL so alpha tested surfaces don't add light
727 // where they aren't rendered
728 if ( dl->additive ) {
729 GL_State( GLS_SRCBLEND_ONE | GLS_DSTBLEND_ONE | GLS_DEPTHFUNC_EQUAL );
730 }
731 else {
732 GL_State( GLS_SRCBLEND_DST_COLOR | GLS_DSTBLEND_ONE | GLS_DEPTHFUNC_EQUAL );
733 }
734 R_DrawElements( numIndexes, hitIndexes );
735 backEnd.pc.c_totalIndexes += numIndexes;
736 backEnd.pc.c_dlightIndexes += numIndexes;
737 }
738}

752/*
753===================
754RB_FogPass
755
756Blends a fog texture on top of everything else
757===================
758*/
759static void RB_FogPass( void ) {
760 fog_t *fog;
761 int i;
762
763 qglEnableClientState( GL_COLOR_ARRAY );
764 qglColorPointer( 4, GL_UNSIGNED_BYTE, 0, tess.svars.colors );
765
766 qglEnableClientState( GL_TEXTURE_COORD_ARRAY);
767 qglTexCoordPointer( 2, GL_FLOAT, 0, tess.svars.texcoords[0] );
768
769 fog = tr.world->fogs + tess.fogNum;
770
771 for ( i = 0; i < tess.numVertexes; i++ ) {
772 * ( int * )&tess.svars.colors[i] = fog->colorInt;
773 }
774
775 RB_CalcFogTexCoords( ( float * ) tess.svars.texcoords[0] );
776
777 GL_Bind( tr.fogImage );
778
779 if ( tess.shader->fogPass == FP_EQUAL ) {
780 GL_State( GLS_SRCBLEND_SRC_ALPHA | GLS_DSTBLEND_ONE_MINUS_SRC_ALPHA | GLS_DEPTHFUNC_EQUAL );
781 } else {
782 GL_State( GLS_SRCBLEND_SRC_ALPHA | GLS_DSTBLEND_ONE_MINUS_SRC_ALPHA );
783 }
784
785 R_DrawElements( tess.numIndexes, tess.indexes );
786}

788/*
789===============
790ComputeColors
791===============
792*/
793static void ComputeColors( shaderStage_t *pStage )
794{
795 int i;
796
797 //
798 // rgbGen
799 //
800 switch ( pStage->rgbGen )
801 {
802 case CGEN_IDENTITY:
803 Com_Memset( tess.svars.colors, 0xff, tess.numVertexes * 4 );
804 break;
805 default:
806 case CGEN_IDENTITY_LIGHTING:
807 Com_Memset( tess.svars.colors, tr.identityLightByte, tess.numVertexes * 4 );
808 break;
809 case CGEN_LIGHTING_DIFFUSE:
810 RB_CalcDiffuseColor( ( unsigned char * ) tess.svars.colors );
811 break;
812 case CGEN_EXACT_VERTEX:
813 Com_Memcpy( tess.svars.colors, tess.vertexColors, tess.numVertexes * sizeof( tess.vertexColors[0] ) );
814 break;
815 case CGEN_CONST:
816 for ( i = 0; i < tess.numVertexes; i++ ) {
817 *(int *)tess.svars.colors[i] = *(int *)pStage->constantColor;
818 }
819 break;
820 case CGEN_VERTEX:
821 if ( tr.identityLight == 1 )
822 {
823 Com_Memcpy( tess.svars.colors, tess.vertexColors, tess.numVertexes * sizeof( tess.vertexColors[0] ) );
824 }
825 else
826 {
827 for ( i = 0; i < tess.numVertexes; i++ )
828 {
829 tess.svars.colors[i][0] = tess.vertexColors[i][0] * tr.identityLight;
830 tess.svars.colors[i][1] = tess.vertexColors[i][1] * tr.identityLight;
831 tess.svars.colors[i][2] = tess.vertexColors[i][2] * tr.identityLight;
832 tess.svars.colors[i][3] = tess.vertexColors[i][3];
833 }
834 }
835 break;
836 case CGEN_ONE_MINUS_VERTEX:
837 if ( tr.identityLight == 1 )
838 {
839 for ( i = 0; i < tess.numVertexes; i++ )
840 {
841 tess.svars.colors[i][0] = 255 - tess.vertexColors[i][0];
842 tess.svars.colors[i][1] = 255 - tess.vertexColors[i][1];
843 tess.svars.colors[i][2] = 255 - tess.vertexColors[i][2];
844 }
845 }
846 else
847 {
848 for ( i = 0; i < tess.numVertexes; i++ )
849 {
850 tess.svars.colors[i][0] = ( 255 - tess.vertexColors[i][0] ) * tr.identityLight;
851 tess.svars.colors[i][1] = ( 255 - tess.vertexColors[i][1] ) * tr.identityLight;
852 tess.svars.colors[i][2] = ( 255 - tess.vertexColors[i][2] ) * tr.identityLight;
853 }
854 }
855 break;
856 case CGEN_FOG:
857 {
858 fog_t *fog;
859
860 fog = tr.world->fogs + tess.fogNum;
861
862 for ( i = 0; i < tess.numVertexes; i++ ) {
863 * ( int * )&tess.svars.colors[i] = fog->colorInt;
864 }
865 }
866 break;
867 case CGEN_WAVEFORM:
868 RB_CalcWaveColor( &pStage->rgbWave, ( unsigned char * ) tess.svars.colors );
869 break;
870 case CGEN_ENTITY:
871 RB_CalcColorFromEntity( ( unsigned char * ) tess.svars.colors );
872 break;
873 case CGEN_ONE_MINUS_ENTITY:
874 RB_CalcColorFromOneMinusEntity( ( unsigned char * ) tess.svars.colors );
875 break;
876 }
877
878 //
879 // alphaGen
880 //
881 switch ( pStage->alphaGen )
882 {
883 case AGEN_SKIP:
884 break;
885 case AGEN_IDENTITY:
886 if ( pStage->rgbGen != CGEN_IDENTITY ) {
887 if ( ( pStage->rgbGen == CGEN_VERTEX && tr.identityLight != 1 ) ||
888 pStage->rgbGen != CGEN_VERTEX ) {
889 for ( i = 0; i < tess.numVertexes; i++ ) {
890 tess.svars.colors[i][3] = 0xff;
891 }
892 }
893 }
894 break;
895 case AGEN_CONST:
896 if ( pStage->rgbGen != CGEN_CONST ) {
897 for ( i = 0; i < tess.numVertexes; i++ ) {
898 tess.svars.colors[i][3] = pStage->constantColor[3];
899 }
900 }
901 break;
902 case AGEN_WAVEFORM:
903 RB_CalcWaveAlpha( &pStage->alphaWave, ( unsigned char * ) tess.svars.colors );
904 break;
905 case AGEN_LIGHTING_SPECULAR:
906 RB_CalcSpecularAlpha( ( unsigned char * ) tess.svars.colors );
907 break;
908 case AGEN_ENTITY:
909 RB_CalcAlphaFromEntity( ( unsigned char * ) tess.svars.colors );
910 break;
911 case AGEN_ONE_MINUS_ENTITY:
912 RB_CalcAlphaFromOneMinusEntity( ( unsigned char * ) tess.svars.colors );
913 break;
914 case AGEN_VERTEX:
915 if ( pStage->rgbGen != CGEN_VERTEX ) {
916 for ( i = 0; i < tess.numVertexes; i++ ) {
917 tess.svars.colors[i][3] = tess.vertexColors[i][3];
918 }
919 }
920 break;
921 case AGEN_ONE_MINUS_VERTEX:
922 for ( i = 0; i < tess.numVertexes; i++ )
923 {
924 tess.svars.colors[i][3] = 255 - tess.vertexColors[i][3];
925 }
926 break;
927 case AGEN_PORTAL:
928 {
929 unsigned char alpha;
930
931 for ( i = 0; i < tess.numVertexes; i++ )
932 {
933 float len;
934 vec3_t v;
935
936 VectorSubtract( tess.xyz[i], backEnd.viewParms.or.origin, v );
937 len = VectorLength( v );
938
939 len /= tess.shader->portalRange;
940
941 if ( len < 0 )
942 {
943 alpha = 0;
944 }
945 else if ( len > 1 )
946 {
947 alpha = 0xff;
948 }
949 else
950 {
951 alpha = len * 0xff;
952 }
953
954 tess.svars.colors[i][3] = alpha;
955 }
956 }
957 break;
958 }
959
960 //
961 // fog adjustment for colors to fade out as fog increases
962 //
963 if ( tess.fogNum )
964 {
965 switch ( pStage->adjustColorsForFog )
966 {
967 case ACFF_MODULATE_RGB:
968 RB_CalcModulateColorsByFog( ( unsigned char * ) tess.svars.colors );
969 break;
970 case ACFF_MODULATE_ALPHA:
971 RB_CalcModulateAlphasByFog( ( unsigned char * ) tess.svars.colors );
972 break;
973 case ACFF_MODULATE_RGBA:
974 RB_CalcModulateRGBAsByFog( ( unsigned char * ) tess.svars.colors );
975 break;
976 case ACFF_NONE:
977 break;
978 }
979 }
980
981 // if in greyscale rendering mode turn all color values into greyscale.
982 if(r_greyscale->integer)
983 {
984 int scale;
985 for(i = 0; i < tess.numVertexes; i++)
986 {
987 scale = LUMA(tess.svars.colors[i][0], tess.svars.colors[i][1], tess.svars.colors[i][2]);
988 tess.svars.colors[i][0] = tess.svars.colors[i][1] = tess.svars.colors[i][2] = scale;
989 }
990 }
991 else if(r_greyscale->value)
992 {
993 float scale;
994
995 for(i = 0; i < tess.numVertexes; i++)
996 {
997 scale = LUMA(tess.svars.colors[i][0], tess.svars.colors[i][1], tess.svars.colors[i][2]);
998 tess.svars.colors[i][0] = LERP(tess.svars.colors[i][0], scale, r_greyscale->value);
999 tess.svars.colors[i][1] = LERP(tess.svars.colors[i][1], scale, r_greyscale->value);
1000 tess.svars.colors[i][2] = LERP(tess.svars.colors[i][2], scale, r_greyscale->value);
1001 }
1002 }
1003}

1005/*
1006===============
1007ComputeTexCoords
1008===============
1009*/
1010static void ComputeTexCoords( shaderStage_t *pStage ) {
1011 int i;
1012 int b;
1013
1014 for ( b = 0; b < NUM_TEXTURE_BUNDLES; b++ ) {
1015 int tm;
1016
1017 //
1018 // generate the texture coordinates
1019 //
1020 switch ( pStage->bundle[b].tcGen )
1021 {
1022 case TCGEN_IDENTITY:
1023 Com_Memset( tess.svars.texcoords[b], 0, sizeof( float ) * 2 * tess.numVertexes );
1024 break;
1025 case TCGEN_TEXTURE:
1026 for ( i = 0 ; i < tess.numVertexes ; i++ ) {
1027 tess.svars.texcoords[b][i][0] = tess.texCoords[i][0][0];
1028 tess.svars.texcoords[b][i][1] = tess.texCoords[i][0][1];
1029 }
1030 break;
1031 case TCGEN_LIGHTMAP:
1032 for ( i = 0 ; i < tess.numVertexes ; i++ ) {
1033 tess.svars.texcoords[b][i][0] = tess.texCoords[i][1][0];
1034 tess.svars.texcoords[b][i][1] = tess.texCoords[i][1][1];
1035 }
1036 break;
1037 case TCGEN_VECTOR:
1038 for ( i = 0 ; i < tess.numVertexes ; i++ ) {
1039 tess.svars.texcoords[b][i][0] = DotProduct( tess.xyz[i], pStage->bundle[b].tcGenVectors[0] );
1040 tess.svars.texcoords[b][i][1] = DotProduct( tess.xyz[i], pStage->bundle[b].tcGenVectors[1] );
1041 }
1042 break;
1043 case TCGEN_FOG:
1044 RB_CalcFogTexCoords( ( float * ) tess.svars.texcoords[b] );
1045 break;
1046 case TCGEN_ENVIRONMENT_MAPPED:
1047 RB_CalcEnvironmentTexCoords( ( float * ) tess.svars.texcoords[b] );
1048 break;
1049 case TCGEN_BAD:
1050 return;
1051 }
1052
1053 //
1054 // alter texture coordinates
1055 //
1056 for ( tm = 0; tm < pStage->bundle[b].numTexMods ; tm++ ) {
1057 switch ( pStage->bundle[b].texMods[tm].type )
1058 {
1059 case TMOD_NONE:
1060 tm = TR_MAX_TEXMODS; // break out of for loop
1061 break;
1062
1063 case TMOD_TURBULENT:
1064 RB_CalcTurbulentTexCoords( &pStage->bundle[b].texMods[tm].wave,
1065 ( float * ) tess.svars.texcoords[b] );
1066 break;
1067
1068 case TMOD_ENTITY_TRANSLATE:
1069 RB_CalcScrollTexCoords( backEnd.currentEntity->e.shaderTexCoord,
1070 ( float * ) tess.svars.texcoords[b] );
1071 break;
1072
1073 case TMOD_SCROLL:
1074 RB_CalcScrollTexCoords( pStage->bundle[b].texMods[tm].scroll,
1075 ( float * ) tess.svars.texcoords[b] );
1076 break;
1077
1078 case TMOD_SCALE:
1079 RB_CalcScaleTexCoords( pStage->bundle[b].texMods[tm].scale,
1080 ( float * ) tess.svars.texcoords[b] );
1081 break;
1082
1083 case TMOD_STRETCH:
1084 RB_CalcStretchTexCoords( &pStage->bundle[b].texMods[tm].wave,
1085 ( float * ) tess.svars.texcoords[b] );
1086 break;
1087
1088 case TMOD_TRANSFORM:
1089 RB_CalcTransformTexCoords( &pStage->bundle[b].texMods[tm],
1090 ( float * ) tess.svars.texcoords[b] );
1091 break;
1092
1093 case TMOD_ROTATE:
1094 RB_CalcRotateTexCoords( pStage->bundle[b].texMods[tm].rotateSpeed,
1095 ( float * ) tess.svars.texcoords[b] );
1096 break;
1097
1098 default:
1099 ri.Error( ERR_DROP, "ERROR: unknown texmod '%d' in shader '%s'", pStage->bundle[b].texMods[tm].type, tess.shader->name );
1100 break;
1101 }
1102 }
1103 }
1104}

1106/*
1107** RB_IterateStagesGeneric
1108*/
1109static void RB_IterateStagesGeneric( shaderCommands_t *input )
1110{
1111 int stage;
1112
1113 for ( stage = 0; stage < MAX_SHADER_STAGES; stage++ )
1114 {
1115 shaderStage_t *pStage = tess.xstages[stage];
1116
1117 if ( !pStage )
1118 {
1119 break;
1120 }
1121
1122 ComputeColors( pStage );
1123 ComputeTexCoords( pStage );
1124
1125 if ( !setArraysOnce )
1126 {
1127 qglEnableClientState( GL_COLOR_ARRAY );
1128 qglColorPointer( 4, GL_UNSIGNED_BYTE, 0, input->svars.colors );
1129 }
1130
1131 //
1132 // do multitexture
1133 //
1134 if ( pStage->bundle[1].image[0] != 0 )
1135 {
1136 DrawMultitextured( input, stage );
1137 }
1138 else
1139 {
1140 if ( !setArraysOnce )
1141 {
1142 qglTexCoordPointer( 2, GL_FLOAT, 0, input->svars.texcoords[0] );
1143 }
1144
1145 //
1146 // set state
1147 //
1148 R_BindAnimatedImage( &pStage->bundle[0] );
1149
1150 GL_State( pStage->stateBits );
1151
1152 //
1153 // draw
1154 //
1155 R_DrawElements( input->numIndexes, input->indexes );
1156 }
1157 // allow skipping out to show just lightmaps during development
1158 if ( r_lightmap->integer && ( pStage->bundle[0].isLightmap || pStage->bundle[1].isLightmap ) )
1159 {
1160 break;
1161 }
1162 }
1163}

1166/*
1167** RB_StageIteratorGeneric
1168*/
1169void RB_StageIteratorGeneric( void )
1170{
1171 shaderCommands_t *input;
1172 shader_t *shader;
1173
1174 input = &tess;
1175 shader = input->shader;
1176
1177 RB_DeformTessGeometry();
1178
1179 //
1180 // log this call
1181 //
1182 if ( r_logFile->integer )
1183 {
1184 // don't just call LogComment, or we will get
1185 // a call to va() every frame!
1186 GLimp_LogComment( va("--- RB_StageIteratorGeneric( %s ) ---\n", tess.shader->name) );
1187 }
1188
1189 //
1190 // set face culling appropriately
1191 //
1192 GL_Cull( shader->cullType );
1193
1194 // set polygon offset if necessary
1195 if ( shader->polygonOffset )
1196 {
1197 qglEnable( GL_POLYGON_OFFSET_FILL );
1198 qglPolygonOffset( r_offsetFactor->value, r_offsetUnits->value );
1199 }
1200
1201 //
1202 // if there is only a single pass then we can enable color
1203 // and texture arrays before we compile, otherwise we need
1204 // to avoid compiling those arrays since they will change
1205 // during multipass rendering
1206 //
1207 if ( tess.numPasses > 1 || shader->multitextureEnv )
1208 {
1209 setArraysOnce = qfalse;
1210 qglDisableClientState (GL_COLOR_ARRAY);
1211 qglDisableClientState (GL_TEXTURE_COORD_ARRAY);
1212 }
1213 else
1214 {
1215 setArraysOnce = qtrue;
1216
1217 qglEnableClientState( GL_COLOR_ARRAY);
1218 qglColorPointer( 4, GL_UNSIGNED_BYTE, 0, tess.svars.colors );
1219
1220 qglEnableClientState( GL_TEXTURE_COORD_ARRAY);
1221 qglTexCoordPointer( 2, GL_FLOAT, 0, tess.svars.texcoords[0] );
1222 }
1223
1224 //
1225 // lock XYZ
1226 //
1227 qglVertexPointer (3, GL_FLOAT, 16, input->xyz); // padded for SIMD
1228 if (qglLockArraysEXT)
1229 {
1230 qglLockArraysEXT(0, input->numVertexes);
1231 GLimp_LogComment( "glLockArraysEXT\n" );
1232 }
1233
1234 //
1235 // enable color and texcoord arrays after the lock if necessary
1236 //
1237 if ( !setArraysOnce )
1238 {
1239 qglEnableClientState( GL_TEXTURE_COORD_ARRAY );
1240 qglEnableClientState( GL_COLOR_ARRAY );
1241 }
1242
1243 //
1244 // call shader function
1245 //
1246 RB_IterateStagesGeneric( input );
1247
1248 //
1249 // now do any dynamic lighting needed
1250 //
1251 if ( tess.dlightBits && tess.shader->sort <= SS_OPAQUE
1252 && !(tess.shader->surfaceFlags & (SURF_NODLIGHT | SURF_SKY) ) ) {
1253 ProjectDlightTexture();
1254 }
1255
1256 //
1257 // now do fog
1258 //
1259 if ( tess.fogNum && tess.shader->fogPass ) {
1260 RB_FogPass();
1261 }
1262
1263 //
1264 // unlock arrays
1265 //
1266 if (qglUnlockArraysEXT)
1267 {
1268 qglUnlockArraysEXT();
1269 GLimp_LogComment( "glUnlockArraysEXT\n" );
1270 }
1271
1272 //
1273 // reset polygon offset
1274 //
1275 if ( shader->polygonOffset )
1276 {
1277 qglDisable( GL_POLYGON_OFFSET_FILL );
1278 }
1279}

1468*/
1469void RB_EndSurface( void ) {
1470 shaderCommands_t *input;
1471
1472 input = &tess;
1473
1474 if (input->numIndexes == 0) {
1475 return;
1476 }
1477
1478 if (input->indexes[SHADER_MAX_INDEXES-1] != 0) {
1479 ri.Error (ERR_DROP, "RB_EndSurface() - SHADER_MAX_INDEXES hit");
1480 }
1481 if (input->xyz[SHADER_MAX_VERTEXES-1][0] != 0) {
1482 ri.Error (ERR_DROP, "RB_EndSurface() - SHADER_MAX_VERTEXES hit");
1483 }
1484
1485 if ( tess.shader == tr.shadowShader ) {
1486 RB_ShadowTessEnd();
1487 return;
1488 }
1489
1490 // for debugging of sort order issues, stop rendering after a given sort value
1491 if ( r_debugSort->integer && r_debugSort->integer < tess.shader->sort ) {
1492 return;
1493 }
1494
1495 //
1496 // update performance counters
1497 //
1498 backEnd.pc.c_shaders++;
1499 backEnd.pc.c_vertexes += tess.numVertexes;
1500 backEnd.pc.c_indexes += tess.numIndexes;
1501 backEnd.pc.c_totalIndexes += tess.numIndexes * tess.numPasses;
1502
1503 //
1504 // call off to shader specific tess end function
1505 //
1506 tess.currentStageIteratorFunc();
1507
1508 //
1509 // draw debugging stuff
1510 //
1511 if ( r_showtris->integer ) {
1512 DrawTris (input);
1513 }
1514 if ( r_shownormals->integer ) {
1515 DrawNormals (input);
1516 }
1517 // clear shader so we can tell we don't have any unclosed surfaces
1518 tess.numIndexes = 0;
1519
1520 GLimp_LogComment( "----------\n" );
1521}

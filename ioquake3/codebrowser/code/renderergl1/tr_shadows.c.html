25/*
26
27 for a projection shadow:
28
29 point[x] += light vector * ( z - shadow plane )
30 point[y] +=
31 point[z] = shadow plane
32
33 1 0 light[x] / light[z]
34
35*/
36
37typedef struct {
38 int i2;
39 int facing;
40} edgeDef_t;
41
42#define MAX_EDGE_DEFS 32
43
44static edgeDef_t edgeDefs[SHADER_MAX_VERTEXES][MAX_EDGE_DEFS];
45static int numEdgeDefs[SHADER_MAX_VERTEXES];
46static int facing[SHADER_MAX_INDEXES/3];
47static vec3_t shadowXyz[SHADER_MAX_VERTEXES];
48
49void R_AddEdgeDef( int i1, int i2, int facing ) {
50 int c;
51
52 c = numEdgeDefs[ i1 ];
53 if ( c == MAX_EDGE_DEFS ) {
54 return; // overflow
55 }
56 edgeDefs[ i1 ][ c ].i2 = i2;
57 edgeDefs[ i1 ][ c ].facing = facing;
58
59 numEdgeDefs[ i1 ]++;
60}

64

143/*
144=================
145RB_ShadowTessEnd
146
147triangleFromEdge[ v1 ][ v2 ]
148
149
150 set triangle from edge( v1, v2, tri )
151 if ( facing[ triangleFromEdge[ v1 ][ v2 ] ] && !facing[ triangleFromEdge[ v2 ][ v1 ] ) {
152 }
153=================
154*/
155void RB_ShadowTessEnd( void ) {
156 int i;
157 int numTris;
158 vec3_t lightDir;
159 GLboolean rgba[4];
160
161 if ( glConfig.stencilBits < 4 ) {
162 return;
163 }
164
165 VectorCopy( backEnd.currentEntity->lightDir, lightDir );
166
167 // project vertexes away from light direction
168 for ( i = 0 ; i < tess.numVertexes ; i++ ) {
169 VectorMA( tess.xyz[i], -512, lightDir, shadowXyz[i] );
170 }
171
172 // decide which triangles face the light
173 Com_Memset( numEdgeDefs, 0, 4 * tess.numVertexes );
174
175 numTris = tess.numIndexes / 3;
176 for ( i = 0 ; i < numTris ; i++ ) {
177 int i1, i2, i3;
178 vec3_t d1, d2, normal;
179 float *v1, *v2, *v3;
180 float d;
181
182 i1 = tess.indexes[ i*3 + 0 ];
183 i2 = tess.indexes[ i*3 + 1 ];
184 i3 = tess.indexes[ i*3 + 2 ];
185
186 v1 = tess.xyz[ i1 ];
187 v2 = tess.xyz[ i2 ];
188 v3 = tess.xyz[ i3 ];
189
190 VectorSubtract( v2, v1, d1 );
191 VectorSubtract( v3, v1, d2 );
192 CrossProduct( d1, d2, normal );
193
194 d = DotProduct( normal, lightDir );
195 if ( d > 0 ) {
196 facing[ i ] = 1;
197 } else {
198 facing[ i ] = 0;
199 }
200
201 // create the edges
202 R_AddEdgeDef( i1, i2, facing[ i ] );
203 R_AddEdgeDef( i2, i3, facing[ i ] );
204 R_AddEdgeDef( i3, i1, facing[ i ] );
205 }
206
207 // draw the silhouette edges
208
209 GL_Bind( tr.whiteImage );
210 GL_State( GLS_SRCBLEND_ONE | GLS_DSTBLEND_ZERO );
211 qglColor3f( 0.2f, 0.2f, 0.2f );
212
213 // don't write to the color buffer
214 qglGetBooleanv(GL_COLOR_WRITEMASK, rgba);
215 qglColorMask( GL_FALSE, GL_FALSE, GL_FALSE, GL_FALSE );
216
217 qglEnable( GL_STENCIL_TEST );
218 qglStencilFunc( GL_ALWAYS, 1, 255 );
219
220 GL_Cull( CT_BACK_SIDED );
221 qglStencilOp( GL_KEEP, GL_KEEP, GL_INCR );
222
223 R_RenderShadowEdges();
224
225 GL_Cull( CT_FRONT_SIDED );
226 qglStencilOp( GL_KEEP, GL_KEEP, GL_DECR );
227
228 R_RenderShadowEdges();
229
230
231 // reenable writing to the color buffer
232 qglColorMask(rgba[0], rgba[1], rgba[2], rgba[3]);
233}

236/*
237=================
238RB_ShadowFinish
239
240Darken everything that is is a shadow volume.
241We have to delay this until everything has been shadowed,
242because otherwise shadows from different body parts would
243overlap and double darken.
244=================
245*/
246void RB_ShadowFinish( void ) {
247 if ( r_shadows->integer != 2 ) {
248 return;
249 }
250 if ( glConfig.stencilBits < 4 ) {
251 return;
252 }
253 qglEnable( GL_STENCIL_TEST );
254 qglStencilFunc( GL_NOTEQUAL, 0, 255 );
255
256 qglDisable (GL_CLIP_PLANE0);
257 GL_Cull( CT_TWO_SIDED );
258
259 GL_Bind( tr.whiteImage );
260
261 qglLoadIdentity ();
262
263 qglColor3f( 0.6f, 0.6f, 0.6f );
264 GL_State( GLS_DEPTHMASK_TRUE | GLS_SRCBLEND_DST_COLOR | GLS_DSTBLEND_ZERO );
265
266// qglColor3f( 1, 0, 0 );
267// GL_State( GLS_DEPTHMASK_TRUE | GLS_SRCBLEND_ONE | GLS_DSTBLEND_ZERO );
268
269 qglBegin( GL_QUADS );
270 qglVertex3f( -100, 100, -10 );
271 qglVertex3f( 100, 100, -10 );
272 qglVertex3f( 100, -100, -10 );
273 qglVertex3f( -100, -100, -10 );
274 qglEnd ();
275
276 qglColor4f(1,1,1,1);
277 qglDisable( GL_STENCIL_TEST );
278}

281/*
282=================
283RB_ProjectionShadowDeform
284
285=================
286*/
287void RB_ProjectionShadowDeform( void ) {
288 float *xyz;
289 int i;
290 float h;
291 vec3_t ground;
292 vec3_t light;
293 float groundDist;
294 float d;
295 vec3_t lightDir;
296
297 xyz = ( float * ) tess.xyz;
298
299 ground[0] = backEnd.or.axis[0][2];
300 ground[1] = backEnd.or.axis[1][2];
301 ground[2] = backEnd.or.axis[2][2];
302
303 groundDist = backEnd.or.origin[2] - backEnd.currentEntity->e.shadowPlane;
304
305 VectorCopy( backEnd.currentEntity->lightDir, lightDir );
306 d = DotProduct( lightDir, ground );
307 // don't let the shadows get too long or go negative
308 if ( d < 0.5 ) {
309 VectorMA( lightDir, (0.5 - d), ground, lightDir );
310 d = DotProduct( lightDir, ground );
311 }
312 d = 1.0 / d;
313
314 light[0] = lightDir[0] * d;
315 light[1] = lightDir[1] * d;
316 light[2] = lightDir[2] * d;
317
318 for ( i = 0; i < tess.numVertexes; i++, xyz += 4 ) {
319 h = DotProduct( xyz, ground ) + groundDist;
320
321 xyz[0] -= light[0] * h;
322 xyz[1] -= light[1] * h;
323 xyz[2] -= light[2] * h;
324 }

59/*
60================
61return a hash value for the filename
62================
63*/
64static long generateHashValue( const char *fname ) {
65 int i;
66 long hash;
67 char letter;
68
69 hash = 0;
70 i = 0;
71 while (fname[i] != '\0') {
72 letter = tolower(fname[i]);
73 if (letter =='.') break; // don't include extension
74 if (letter =='\\') letter = '/'; // damn path names
75 hash+=(long)(letter)*(i+119);
76 i++;
77 }
78 hash &= (FILE_HASH_SIZE-1);
79 return hash;
80}

82/*
83===============
84GL_TextureMode
85===============
86*/
87void GL_TextureMode( const char *string ) {
88 int i;
89 image_t *glt;
90
91 for ( i=0 ; i< 6 ; i++ ) {
92 if ( !Q_stricmp( modes[i].name, string ) ) {
93 break;
94 }
95 }
96
97 // hack to prevent trilinear from being set on voodoo,
98 // because their driver freaks...
99 if ( i == 5 && glConfig.hardwareType == GLHW_3DFX_2D3D ) {
100 ri.Printf( PRINT_ALL, "Refusing to set trilinear on a voodoo.\n" );
101 i = 3;
102 }
103
104
105 if ( i == 6 ) {
106 ri.Printf (PRINT_ALL, "bad filter name\n");
107 return;
108 }
109
110 gl_filter_min = modes[i].minimize;
111 gl_filter_max = modes[i].maximize;
112
113 // change all the existing mipmap texture objects
114 for ( i = 0 ; i < tr.numImages ; i++ ) {
115 glt = tr.images[ i ];
116 if ( glt->flags & IMGFLAG_MIPMAP ) {
117 GL_Bind (glt);
118 qglTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, gl_filter_min);
119 qglTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, gl_filter_max);
120 }
121 }
122}

143/*
144===============
145R_ImageList_f
146===============
147*/
148void R_ImageList_f( void ) {
149 int i;
150 int estTotalSize = 0;
151
152 ri.Printf(PRINT_ALL, "\n -w-- -h-- type -size- --name-------\n");
153
154 for ( i = 0 ; i < tr.numImages ; i++ )
155 {
156 image_t *image = tr.images[i];
157 char *format = "???? ";
158 char *sizeSuffix;
159 int estSize;
160 int displaySize;
161
162 estSize = image->uploadHeight * image->uploadWidth;
163
164 switch(image->internalFormat)
165 {
166 case GL_COMPRESSED_SRGB_ALPHA_S3TC_DXT1_EXT:
167 format = "sDXT1";
168 // 64 bits per 16 pixels, so 4 bits per pixel
169 estSize /= 2;
170 break;
171 case GL_COMPRESSED_SRGB_ALPHA_S3TC_DXT5_EXT:
172 format = "sDXT5";
173 // 128 bits per 16 pixels, so 1 byte per pixel
174 break;
175 case GL_COMPRESSED_SRGB_ALPHA_BPTC_UNORM_ARB:
176 format = "sBPTC";
177 // 128 bits per 16 pixels, so 1 byte per pixel
178 break;
179 case GL_COMPRESSED_LUMINANCE_ALPHA_LATC2_EXT:
180 format = "LATC ";
181 // 128 bits per 16 pixels, so 1 byte per pixel
182 break;
183 case GL_COMPRESSED_RGBA_S3TC_DXT1_EXT:
184 format = "DXT1 ";
185 // 64 bits per 16 pixels, so 4 bits per pixel
186 estSize /= 2;
187 break;
188 case GL_COMPRESSED_RGBA_S3TC_DXT5_EXT:
189 format = "DXT5 ";
190 // 128 bits per 16 pixels, so 1 byte per pixel
191 break;
192 case GL_COMPRESSED_RGBA_BPTC_UNORM_ARB:
193 format = "BPTC ";
194 // 128 bits per 16 pixels, so 1 byte per pixel
195 break;
196 case GL_RGB4_S3TC:
197 format = "S3TC ";
198 // same as DXT1?
199 estSize /= 2;
200 break;
201 case GL_RGBA4:
202 case GL_RGBA8:
203 case GL_RGBA:
204 format = "RGBA ";
205 // 4 bytes per pixel
206 estSize *= 4;
207 break;
208 case GL_LUMINANCE8:
209 case GL_LUMINANCE16:
210 case GL_LUMINANCE:
211 format = "L ";
212 // 1 byte per pixel?
213 break;
214 case GL_RGB5:
215 case GL_RGB8:
216 case GL_RGB:
217 format = "RGB ";
218 // 3 bytes per pixel?
219 estSize *= 3;
220 break;
221 case GL_LUMINANCE8_ALPHA8:
222 case GL_LUMINANCE16_ALPHA16:
223 case GL_LUMINANCE_ALPHA:
224 format = "LA ";
225 // 2 bytes per pixel?
226 estSize *= 2;
227 break;
228 case GL_SRGB_EXT:
229 case GL_SRGB8_EXT:
230 format = "sRGB ";
231 // 3 bytes per pixel?
232 estSize *= 3;
233 break;
234 case GL_SRGB_ALPHA_EXT:
235 case GL_SRGB8_ALPHA8_EXT:
236 format = "sRGBA";
237 // 4 bytes per pixel?
238 estSize *= 4;
239 break;
240 case GL_SLUMINANCE_EXT:
241 case GL_SLUMINANCE8_EXT:
242 format = "sL ";
243 // 1 byte per pixel?
244 break;
245 case GL_SLUMINANCE_ALPHA_EXT:
246 case GL_SLUMINANCE8_ALPHA8_EXT:
247 format = "sLA ";
248 // 2 byte per pixel?
249 estSize *= 2;
250 break;
251 }
252
253 // mipmap adds about 50%
254 if (image->flags & IMGFLAG_MIPMAP)
255 estSize += estSize / 2;
256
257 sizeSuffix = "b ";
258 displaySize = estSize;
259
260 if (displaySize > 1024)
261 {
262 displaySize /= 1024;
263 sizeSuffix = "kb";
264 }
265
266 if (displaySize > 1024)
267 {
268 displaySize /= 1024;
269 sizeSuffix = "Mb";
270 }
271
272 if (displaySize > 1024)
273 {
274 displaySize /= 1024;
275 sizeSuffix = "Gb";
276 }
277
278 ri.Printf(PRINT_ALL, "%4i: %4ix%4i %s %4i%s %s\n", i, image->uploadWidth, image->uploadHeight, format, displaySize, sizeSuffix, image->imgName);
279 estTotalSize += estSize;
280 }
281
282 ri.Printf (PRINT_ALL, " ---------\n");
283 ri.Printf (PRINT_ALL, " approx %i bytes\n", estTotalSize);
284 ri.Printf (PRINT_ALL, " %i total images\n\n", tr.numImages );
285}

289/*
290================
291ResampleTexture
292
293Used to resample images in a more general than quartering fashion.
294
295This will only be filtered properly if the resampled size
296is greater than half the original size.
297
298If a larger shrinking is needed, use the mipmap function
299before or after.
300================
301*/
302static void ResampleTexture( unsigned *in, int inwidth, int inheight, unsigned *out,
303 int outwidth, int outheight ) {
304 int i, j;
305 unsigned *inrow, *inrow2;
306 unsigned frac, fracstep;
307 unsigned p1[2048], p2[2048];
308 byte *pix1, *pix2, *pix3, *pix4;
309
310 if (outwidth>2048)
311 ri.Error(ERR_DROP, "ResampleTexture: max width");
312
313 fracstep = inwidth*0x10000/outwidth;
314
315 frac = fracstep>>2;
316 for ( i=0 ; i<outwidth ; i++ ) {
317 p1[i] = 4*(frac>>16);
318 frac += fracstep;
319 }
320 frac = 3*(fracstep>>2);
321 for ( i=0 ; i<outwidth ; i++ ) {
322 p2[i] = 4*(frac>>16);
323 frac += fracstep;
324 }
325
326 for (i=0 ; i<outheight ; i++, out += outwidth) {
327 inrow = in + inwidth*(int)((i+0.25)*inheight/outheight);
328 inrow2 = in + inwidth*(int)((i+0.75)*inheight/outheight);
329 for (j=0 ; j<outwidth ; j++) {
330 pix1 = (byte *)inrow + p1[j];
331 pix2 = (byte *)inrow + p2[j];
332 pix3 = (byte *)inrow2 + p1[j];
333 pix4 = (byte *)inrow2 + p2[j];
334 ((byte *)(out+j))[0] = (pix1[0] + pix2[0] + pix3[0] + pix4[0])>>2;
335 ((byte *)(out+j))[1] = (pix1[1] + pix2[1] + pix3[1] + pix4[1])>>2;
336 ((byte *)(out+j))[2] = (pix1[2] + pix2[2] + pix3[2] + pix4[2])>>2;
337 ((byte *)(out+j))[3] = (pix1[3] + pix2[3] + pix3[3] + pix4[3])>>2;
338 }
339 }
340}

342/*
343================
344R_LightScaleTexture
345
346Scale up the pixel values in a texture to increase the
347lighting range
348================
349*/
350void R_LightScaleTexture (unsigned *in, int inwidth, int inheight, qboolean only_gamma )
351{
352 if ( only_gamma )
353 {
354 if ( !glConfig.deviceSupportsGamma )
355 {
356 int i, c;
357 byte *p;
358
359 p = (byte *)in;
360
361 c = inwidth*inheight;
362 for (i=0 ; i<c ; i++, p+=4)
363 {
364 p[0] = s_gammatable[p[0]];
365 p[1] = s_gammatable[p[1]];
366 p[2] = s_gammatable[p[2]];
367 }
368 }
369 }
370 else
371 {
372 int i, c;
373 byte *p;
374
375 p = (byte *)in;
376
377 c = inwidth*inheight;
378
379 if ( glConfig.deviceSupportsGamma )
380 {
381 for (i=0 ; i<c ; i++, p+=4)
382 {
383 p[0] = s_intensitytable[p[0]];
384 p[1] = s_intensitytable[p[1]];
385 p[2] = s_intensitytable[p[2]];
386 }
387 }
388 else
389 {
390 for (i=0 ; i<c ; i++, p+=4)
391 {
392 p[0] = s_gammatable[s_intensitytable[p[0]]];
393 p[1] = s_gammatable[s_intensitytable[p[1]]];
394 p[2] = s_gammatable[s_intensitytable[p[2]]];
395 }
396 }
397 }

401/*
402================
403R_MipMap2
404
405Operates in place, quartering the size of the texture
406Proper linear filter
407================
408*/
409static void R_MipMap2( unsigned *in, int inWidth, int inHeight ) {
410 int i, j, k;
411 byte *outpix;
412 int inWidthMask, inHeightMask;
413 int total;
414 int outWidth, outHeight;
415 unsigned *temp;
416
417 outWidth = inWidth >> 1;
418 outHeight = inHeight >> 1;
419 temp = ri.Hunk_AllocateTempMemory( outWidth * outHeight * 4 );
420
421 inWidthMask = inWidth - 1;
422 inHeightMask = inHeight - 1;
423
424 for ( i = 0 ; i < outHeight ; i++ ) {
425 for ( j = 0 ; j < outWidth ; j++ ) {
426 outpix = (byte *) ( temp + i * outWidth + j );
427 for ( k = 0 ; k < 4 ; k++ ) {
428 total =
429 1 * ((byte *)&in[ ((i*2-1)&inHeightMask)*inWidth + ((j*2-1)&inWidthMask) ])[k] +
430 2 * ((byte *)&in[ ((i*2-1)&inHeightMask)*inWidth + ((j*2)&inWidthMask) ])[k] +
431 2 * ((byte *)&in[ ((i*2-1)&inHeightMask)*inWidth + ((j*2+1)&inWidthMask) ])[k] +
432 1 * ((byte *)&in[ ((i*2-1)&inHeightMask)*inWidth + ((j*2+2)&inWidthMask) ])[k] +
433
434 2 * ((byte *)&in[ ((i*2)&inHeightMask)*inWidth + ((j*2-1)&inWidthMask) ])[k] +
435 4 * ((byte *)&in[ ((i*2)&inHeightMask)*inWidth + ((j*2)&inWidthMask) ])[k] +
436 4 * ((byte *)&in[ ((i*2)&inHeightMask)*inWidth + ((j*2+1)&inWidthMask) ])[k] +
437 2 * ((byte *)&in[ ((i*2)&inHeightMask)*inWidth + ((j*2+2)&inWidthMask) ])[k] +
438
439 2 * ((byte *)&in[ ((i*2+1)&inHeightMask)*inWidth + ((j*2-1)&inWidthMask) ])[k] +
440 4 * ((byte *)&in[ ((i*2+1)&inHeightMask)*inWidth + ((j*2)&inWidthMask) ])[k] +
441 4 * ((byte *)&in[ ((i*2+1)&inHeightMask)*inWidth + ((j*2+1)&inWidthMask) ])[k] +
442 2 * ((byte *)&in[ ((i*2+1)&inHeightMask)*inWidth + ((j*2+2)&inWidthMask) ])[k] +
443
444 1 * ((byte *)&in[ ((i*2+2)&inHeightMask)*inWidth + ((j*2-1)&inWidthMask) ])[k] +
445 2 * ((byte *)&in[ ((i*2+2)&inHeightMask)*inWidth + ((j*2)&inWidthMask) ])[k] +
446 2 * ((byte *)&in[ ((i*2+2)&inHeightMask)*inWidth + ((j*2+1)&inWidthMask) ])[k] +
447 1 * ((byte *)&in[ ((i*2+2)&inHeightMask)*inWidth + ((j*2+2)&inWidthMask) ])[k];
448 outpix[k] = total / 36;
449 }
450 }
451 }
452
453 Com_Memcpy( in, temp, outWidth * outHeight * 4 );
454 ri.Hunk_FreeTempMemory( temp );
455}

457/*
458================
459R_MipMap
460
461Operates in place, quartering the size of the texture
462================
463*/
464static void R_MipMap (byte *in, int width, int height) {
465 int i, j;
466 byte *out;
467 int row;
468
469 if ( !r_simpleMipMaps->integer ) {
470 R_MipMap2( (unsigned *)in, width, height );
471 return;
472 }
473
474 if ( width == 1 && height == 1 ) {
475 return;
476 }
477
478 row = width * 4;
479 out = in;
480 width >>= 1;
481 height >>= 1;
482
483 if ( width == 0 || height == 0 ) {
484 width += height; // get largest
485 for (i=0 ; i<width ; i++, out+=4, in+=8 ) {
486 out[0] = ( in[0] + in[4] )>>1;
487 out[1] = ( in[1] + in[5] )>>1;
488 out[2] = ( in[2] + in[6] )>>1;
489 out[3] = ( in[3] + in[7] )>>1;
490 }
491 return;
492 }
493
494 for (i=0 ; i<height ; i++, in+=row) {
495 for (j=0 ; j<width ; j++, out+=4, in+=8) {
496 out[0] = (in[0] + in[4] + in[row+0] + in[row+4])>>2;
497 out[1] = (in[1] + in[5] + in[row+1] + in[row+5])>>2;
498 out[2] = (in[2] + in[6] + in[row+2] + in[row+6])>>2;
499 out[3] = (in[3] + in[7] + in[row+3] + in[row+7])>>2;
500 }
501 }
502}

505/*
506==================
507R_BlendOverTexture
508
509Apply a color blend over a set of pixels
510==================
511*/
512static void R_BlendOverTexture( byte *data, int pixelCount, byte blend[4] ) {
513 int i;
514 int inverseAlpha;
515 int premult[3];
516
517 inverseAlpha = 255 - blend[3];
518 premult[0] = blend[0] * blend[3];
519 premult[1] = blend[1] * blend[3];
520 premult[2] = blend[2] * blend[3];
521
522 for ( i = 0 ; i < pixelCount ; i++, data+=4 ) {
523 data[0] = ( data[0] * inverseAlpha + premult[0] ) >> 9;
524 data[1] = ( data[1] * inverseAlpha + premult[1] ) >> 9;
525 data[2] = ( data[2] * inverseAlpha + premult[2] ) >> 9;
526 }
527}

549/*
550===============
551Upload32
552
553===============
554*/
555static void Upload32( unsigned *data,
556 int width, int height,
557 qboolean mipmap,
558 qboolean picmip,
559 qboolean lightMap,
560 qboolean allowCompression,
561 int *format,
562 int *pUploadWidth, int *pUploadHeight )
563{
564 int samples;
565 unsigned *scaledBuffer = NULL;
566 unsigned *resampledBuffer = NULL;
567 int scaled_width, scaled_height;
568 int i, c;
569 byte *scan;
570 GLenum internalFormat = GL_RGB;
571 float rMax = 0, gMax = 0, bMax = 0;
572
573 //
574 // convert to exact power of 2 sizes
575 //
576 for (scaled_width = 1 ; scaled_width < width ; scaled_width<<=1)
577 ;
578 for (scaled_height = 1 ; scaled_height < height ; scaled_height<<=1)
579 ;
580 if ( r_roundImagesDown->integer && scaled_width > width )
581 scaled_width >>= 1;
582 if ( r_roundImagesDown->integer && scaled_height > height )
583 scaled_height >>= 1;
584
585 if ( scaled_width != width || scaled_height != height ) {
586 resampledBuffer = ri.Hunk_AllocateTempMemory( scaled_width * scaled_height * 4 );
587 ResampleTexture (data, width, height, resampledBuffer, scaled_width, scaled_height);
588 data = resampledBuffer;
589 width = scaled_width;
590 height = scaled_height;
591 }
592
593 //
594 // perform optional picmip operation
595 //
596 if ( picmip ) {
597 scaled_width >>= r_picmip->integer;
598 scaled_height >>= r_picmip->integer;
599 }
600
601 //
602 // clamp to minimum size
603 //
604 if (scaled_width < 1) {
605 scaled_width = 1;
606 }
607 if (scaled_height < 1) {
608 scaled_height = 1;
609 }
610
611 //
612 // clamp to the current upper OpenGL limit
613 // scale both axis down equally so we don't have to
614 // deal with a half mip resampling
615 //
616 while ( scaled_width > glConfig.maxTextureSize
617 || scaled_height > glConfig.maxTextureSize ) {
618 scaled_width >>= 1;
619 scaled_height >>= 1;
620 }
621
622 scaledBuffer = ri.Hunk_AllocateTempMemory( sizeof( unsigned ) * scaled_width * scaled_height );
623
624 //
625 // scan the texture for each channel's max values
626 // and verify if the alpha channel is being used or not
627 //
628 c = width*height;
629 scan = ((byte *)data);
630 samples = 3;
631
632 if( r_greyscale->integer )
633 {
634 for ( i = 0; i < c; i++ )
635 {
636 byte luma = LUMA(scan[i*4], scan[i*4 + 1], scan[i*4 + 2]);
637 scan[i*4] = luma;
638 scan[i*4 + 1] = luma;
639 scan[i*4 + 2] = luma;
640 }
641 }
642 else if( r_greyscale->value )
643 {
644 for ( i = 0; i < c; i++ )
645 {
646 float luma = LUMA(scan[i*4], scan[i*4 + 1], scan[i*4 + 2]);
647 scan[i*4] = LERP(scan[i*4], luma, r_greyscale->value);
648 scan[i*4 + 1] = LERP(scan[i*4 + 1], luma, r_greyscale->value);
649 scan[i*4 + 2] = LERP(scan[i*4 + 2], luma, r_greyscale->value);
650 }
651 }
652
653 if(lightMap)
654 {
655 if(r_greyscale->integer)
656 internalFormat = GL_LUMINANCE;
657 else
658 internalFormat = GL_RGB;
659 }
660 else
661 {
662 for ( i = 0; i < c; i++ )
663 {
664 if ( scan[i*4+0] > rMax )
665 {
666 rMax = scan[i*4+0];
667 }
668 if ( scan[i*4+1] > gMax )
669 {
670 gMax = scan[i*4+1];
671 }
672 if ( scan[i*4+2] > bMax )
673 {
674 bMax = scan[i*4+2];
675 }
676 if ( scan[i*4 + 3] != 255 )
677 {
678 samples = 4;
679 break;
680 }
681 }
682 // select proper internal format
683 if ( samples == 3 )
684 {
685 if(r_greyscale->integer)
686 {
687 if(r_texturebits->integer == 16)
688 internalFormat = GL_LUMINANCE8;
689 else if(r_texturebits->integer == 32)
690 internalFormat = GL_LUMINANCE16;
691 else
692 internalFormat = GL_LUMINANCE;
693 }
694 else
695 {
696 if ( allowCompression && glConfig.textureCompression == TC_S3TC_ARB )
697 {
698 internalFormat = GL_COMPRESSED_RGBA_S3TC_DXT1_EXT;
699 }
700 else if ( allowCompression && glConfig.textureCompression == TC_S3TC )
701 {
702 internalFormat = GL_RGB4_S3TC;
703 }
704 else if ( r_texturebits->integer == 16 )
705 {
706 internalFormat = GL_RGB5;
707 }
708 else if ( r_texturebits->integer == 32 )
709 {
710 internalFormat = GL_RGB8;
711 }
712 else
713 {
714 internalFormat = GL_RGB;
715 }
716 }
717 }
718 else if ( samples == 4 )
719 {
720 if(r_greyscale->integer)
721 {
722 if(r_texturebits->integer == 16)
723 internalFormat = GL_LUMINANCE8_ALPHA8;
724 else if(r_texturebits->integer == 32)
725 internalFormat = GL_LUMINANCE16_ALPHA16;
726 else
727 internalFormat = GL_LUMINANCE_ALPHA;
728 }
729 else
730 {
731 if ( r_texturebits->integer == 16 )
732 {
733 internalFormat = GL_RGBA4;
734 }
735 else if ( r_texturebits->integer == 32 )
736 {
737 internalFormat = GL_RGBA8;
738 }
739 else
740 {
741 internalFormat = GL_RGBA;
742 }
743 }
744 }
745 }
746
747 // copy or resample data as appropriate for first MIP level
748 if ( ( scaled_width == width ) &&
749 ( scaled_height == height ) ) {
750 if (!mipmap)
751 {
752 qglTexImage2D (GL_TEXTURE_2D, 0, internalFormat, scaled_width, scaled_height, 0, GL_RGBA, GL_UNSIGNED_BYTE, data);
753 *pUploadWidth = scaled_width;
754 *pUploadHeight = scaled_height;
755 *format = internalFormat;
756
757 goto done;
758 }
759 Com_Memcpy (scaledBuffer, data, width*height*4);
760 }
761 else
762 {
763 // use the normal mip-mapping function to go down from here
764 while ( width > scaled_width || height > scaled_height ) {
765 R_MipMap( (byte *)data, width, height );
766 width >>= 1;
767 height >>= 1;
768 if ( width < 1 ) {
769 width = 1;
770 }
771 if ( height < 1 ) {
772 height = 1;
773 }
774 }
775 Com_Memcpy( scaledBuffer, data, width * height * 4 );
776 }
777
778 R_LightScaleTexture (scaledBuffer, scaled_width, scaled_height, !mipmap );
779
780 *pUploadWidth = scaled_width;
781 *pUploadHeight = scaled_height;
782 *format = internalFormat;
783
784 qglTexImage2D (GL_TEXTURE_2D, 0, internalFormat, scaled_width, scaled_height, 0, GL_RGBA, GL_UNSIGNED_BYTE, scaledBuffer );
785
786 if (mipmap)
787 {
788 int miplevel;
789
790 miplevel = 0;
791 while (scaled_width > 1 || scaled_height > 1)
792 {
793 R_MipMap( (byte *)scaledBuffer, scaled_width, scaled_height );
794 scaled_width >>= 1;
795 scaled_height >>= 1;
796 if (scaled_width < 1)
797 scaled_width = 1;
798 if (scaled_height < 1)
799 scaled_height = 1;
800 miplevel++;
801
802 if ( r_colorMipLevels->integer ) {
803 R_BlendOverTexture( (byte *)scaledBuffer, scaled_width * scaled_height, mipBlendColors[miplevel] );
804 }
805
806 qglTexImage2D (GL_TEXTURE_2D, miplevel, internalFormat, scaled_width, scaled_height, 0, GL_RGBA, GL_UNSIGNED_BYTE, scaledBuffer );
807 }
808 }
809done:
810
811 if (mipmap)
812 {
813 if ( textureFilterAnisotropic )
814 qglTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_MAX_ANISOTROPY_EXT,
815 (GLint)Com_Clamp( 1, maxAnisotropy, r_ext_max_anisotropy->integer ) );
816
817 qglTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, gl_filter_min);
818 qglTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, gl_filter_max);
819 }
820 else
821 {
822 if ( textureFilterAnisotropic )
823 qglTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_MAX_ANISOTROPY_EXT, 1 );
824
825 qglTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR );
826 qglTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR );
827 }
828
829 GL_CheckErrors();
830
831 if ( scaledBuffer != 0 )
832 ri.Hunk_FreeTempMemory( scaledBuffer );
833 if ( resampledBuffer != 0 )
834 ri.Hunk_FreeTempMemory( resampledBuffer );
835}

838/*
839================
840R_CreateImage
841
842This is the only way any image_t are created
843================
844*/
845image_t *R_CreateImage( const char *name, byte *pic, int width, int height,
846 imgType_t type, imgFlags_t flags, int internalFormat ) {
847 image_t *image;
848 qboolean isLightmap = qfalse;
849 long hash;
850 int glWrapClampMode;
851
852 if (strlen(name) >= MAX_QPATH ) {
853 ri.Error (ERR_DROP, "R_CreateImage: \"%s\" is too long", name);
854 }
855 if ( !strncmp( name, "*lightmap", 9 ) ) {
856 isLightmap = qtrue;
857 }
858
859 if ( tr.numImages == MAX_DRAWIMAGES ) {
860 ri.Error( ERR_DROP, "R_CreateImage: MAX_DRAWIMAGES hit");
861 }
862
863 image = tr.images[tr.numImages] = ri.Hunk_Alloc( sizeof( image_t ), h_low );
864 image->texnum = 1024 + tr.numImages;
865 tr.numImages++;
866
867 image->type = type;
868 image->flags = flags;
869
870 strcpy (image->imgName, name);
871
872 image->width = width;
873 image->height = height;
874 if (flags & IMGFLAG_CLAMPTOEDGE)
875 glWrapClampMode = GL_CLAMP_TO_EDGE;
876 else
877 glWrapClampMode = GL_REPEAT;
878
879 // lightmaps are always allocated on TMU 1
880 if ( qglActiveTextureARB && isLightmap ) {
881 image->TMU = 1;
882 } else {
883 image->TMU = 0;
884 }
885
886 if ( qglActiveTextureARB ) {
887 GL_SelectTexture( image->TMU );
888 }
889
890 GL_Bind(image);
891
892 Upload32( (unsigned *)pic, image->width, image->height,
893 image->flags & IMGFLAG_MIPMAP,
894 image->flags & IMGFLAG_PICMIP,
895 isLightmap,
896 !(image->flags & IMGFLAG_NO_COMPRESSION),
897 &image->internalFormat,
898 &image->uploadWidth,
899 &image->uploadHeight );
900
901 qglTexParameterf( GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, glWrapClampMode );
902 qglTexParameterf( GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, glWrapClampMode );
903
904 glState.currenttextures[glState.currenttmu] = 0;
905 qglBindTexture( GL_TEXTURE_2D, 0 );
906
907 if ( image->TMU == 1 ) {
908 GL_SelectTexture( 0 );
909 }
910
911 hash = generateHashValue(name);
912 image->next = hashTable[hash];
913 hashTable[hash] = image;
914
915 return image;
916}

921{
922 char *ext;
923 void (*ImageLoader)( const char *, unsigned char **, int *, int * );
924} imageExtToLoaderMap_t;

926// Note that the ordering indicates the order of preference used
927// when there are multiple images of different formats available
928static imageExtToLoaderMap_t imageLoaders[ ] =
929{
930 { "tga", R_LoadTGA },
931 { "jpg", R_LoadJPG },
932 { "jpeg", R_LoadJPG },
933 { "png", R_LoadPNG },
934 { "pcx", R_LoadPCX },
935 { "bmp", R_LoadBMP }
936};

940/*
941=================
942R_LoadImage
943
944Loads any of the supported image types into a cannonical
94532 bit format.
946=================
947*/
948void R_LoadImage( const char *name, byte **pic, int *width, int *height )
949{
950 qboolean orgNameFailed = qfalse;
951 int orgLoader = -1;
952 int i;
953 char localName[ MAX_QPATH ];
954 const char *ext;
955 char *altName;
956
957 *pic = NULL;
958 *width = 0;
959 *height = 0;
960
961 Q_strncpyz( localName, name, MAX_QPATH );
962
963 ext = COM_GetExtension( localName );
964
965 if( *ext )
966 {
967 // Look for the correct loader and use it
968 for( i = 0; i < numImageLoaders; i++ )
969 {
970 if( !Q_stricmp( ext, imageLoaders[ i ].ext ) )
971 {
972 // Load
973 imageLoaders[ i ].ImageLoader( localName, pic, width, height );
974 break;
975 }
976 }
977
978 // A loader was found
979 if( i < numImageLoaders )
980 {
981 if( *pic == NULL )
982 {
983 // Loader failed, most likely because the file isn't there;
984 // try again without the extension
985 orgNameFailed = qtrue;
986 orgLoader = i;
987 COM_StripExtension( name, localName, MAX_QPATH );
988 }
989 else
990 {
991 // Something loaded
992 return;
993 }
994 }
995 }
996
997 // Try and find a suitable match using all
998 // the image formats supported
999 for( i = 0; i < numImageLoaders; i++ )
1000 {
1001 if (i == orgLoader)
1002 continue;
1003
1004 altName = va( "%s.%s", localName, imageLoaders[ i ].ext );
1005
1006 // Load
1007 imageLoaders[ i ].ImageLoader( altName, pic, width, height );
1008
1009 if( *pic )
1010 {
1011 if( orgNameFailed )
1012 {
1013 ri.Printf( PRINT_DEVELOPER, "WARNING: %s not present, using %s instead\n",
1014 name, altName );
1015 }
1016
1017 break;
1018 }
1019 }
1020}

1024===============
1025R_FindImageFile
1026
1027Finds or loads the given image.
1028Returns NULL if it fails, not a default image.
1029==============
1030*/
1031image_t *R_FindImageFile( const char *name, imgType_t type, imgFlags_t flags )
1032{
1033 image_t *image;
1034 int width, height;
1035 byte *pic;
1036 long hash;
1037
1038 if (!name) {
1039 return NULL;
1040 }
1041
1042 hash = generateHashValue(name);
1043
1044 //
1045 // see if the image is already loaded
1046 //
1047 for (image=hashTable[hash]; image; image=image->next) {
1048 if ( !strcmp( name, image->imgName ) ) {
1049 // the white image can be used with any set of parms, but other mismatches are errors
1050 if ( strcmp( name, "*white" ) ) {
1051 if ( image->flags != flags ) {
1052 ri.Printf( PRINT_DEVELOPER, "WARNING: reused image %s with mixed flags (%i vs %i)\n", name, image->flags, flags );
1053 }
1054 }
1055 return image;
1056 }
1057 }
1058
1059 //
1060 // load the pic from disk
1061 //
1062 R_LoadImage( name, &pic, &width, &height );
1063 if ( pic == NULL ) {
1064 return NULL;
1065 }
1066
1067 image = R_CreateImage( ( char * ) name, pic, width, height, type, flags, 0 );
1068 ri.Free( pic );
1069 return image;
1070}

1126/*
1127================
1128R_FogFactor
1129
1130Returns a 0.0 to 1.0 fog density value
1131This is called for each texel of the fog texture on startup
1132and for each vertex of transparent shaders in fog dynamically
1133================
1134*/
1135float R_FogFactor( float s, float t ) {
1136 float d;
1137
1138 s -= 1.0/512;
1139 if ( s < 0 ) {
1140 return 0;
1141 }
1142 if ( t < 1.0/32 ) {
1143 return 0;
1144 }
1145 if ( t < 31.0/32 ) {
1146 s *= (t - 1.0f/32.0f) / (30.0f/32.0f);
1147 }
1148
1149 // we need to leave a lot of clamp range
1150 s *= 8;
1151
1152 if ( s > 1.0 ) {
1153 s = 1.0;
1154 }
1155
1156 d = tr.fogTable[ (int)(s * (FOG_TABLE_SIZE-1)) ];
1157
1158 return d;
1159}

1265/*
1266===============
1267R_SetColorMappings
1268===============
1269*/
1270void R_SetColorMappings( void ) {
1271 int i, j;
1272 float g;
1273 int inf;
1274 int shift;
1275
1276 // setup the overbright lighting
1277 tr.overbrightBits = r_overBrightBits->integer;
1278 if ( !glConfig.deviceSupportsGamma ) {
1279 tr.overbrightBits = 0; // need hardware gamma for overbright
1280 }
1281
1282 // never overbright in windowed mode
1283 if ( !glConfig.isFullscreen )
1284 {
1285 tr.overbrightBits = 0;
1286 }
1287
1288 // allow 2 overbright bits in 24 bit, but only 1 in 16 bit
1289 if ( glConfig.colorBits > 16 ) {
1290 if ( tr.overbrightBits > 2 ) {
1291 tr.overbrightBits = 2;
1292 }
1293 } else {
1294 if ( tr.overbrightBits > 1 ) {
1295 tr.overbrightBits = 1;
1296 }
1297 }
1298 if ( tr.overbrightBits < 0 ) {
1299 tr.overbrightBits = 0;
1300 }
1301
1302 tr.identityLight = 1.0f / ( 1 << tr.overbrightBits );
1303 tr.identityLightByte = 255 * tr.identityLight;
1304
1305
1306 if ( r_intensity->value <= 1 ) {
1307 ri.Cvar_Set( "r_intensity", "1" );
1308 }
1309
1310 if ( r_gamma->value < 0.5f ) {
1311 ri.Cvar_Set( "r_gamma", "0.5" );
1312 } else if ( r_gamma->value > 3.0f ) {
1313 ri.Cvar_Set( "r_gamma", "3.0" );
1314 }
1315
1316 g = r_gamma->value;
1317
1318 shift = tr.overbrightBits;
1319
1320 for ( i = 0; i < 256; i++ ) {
1321 if ( g == 1 ) {
1322 inf = i;
1323 } else {
1324 inf = 255 * pow ( i/255.0f, 1.0f / g ) + 0.5f;
1325 }
1326 inf <<= shift;
1327 if (inf < 0) {
1328 inf = 0;
1329 }
1330 if (inf > 255) {
1331 inf = 255;
1332 }
1333 s_gammatable[i] = inf;
1334 }
1335
1336 for (i=0 ; i<256 ; i++) {
1337 j = i * r_intensity->value;
1338 if (j > 255) {
1339 j = 255;
1340 }
1341 s_intensitytable[i] = j;
1342 }
1343
1344 if ( glConfig.deviceSupportsGamma )
1345 {
1346 GLimp_SetGamma( s_gammatable, s_gammatable, s_gammatable );
1347 }
1348}

1398/*
1399==================
1400CommaParse
1401
1402This is unfortunate, but the skin files aren't
1403compatable with our normal parsing rules.
1404==================
1405*/
1406static char *CommaParse( char **data_p ) {
1407 int c = 0, len;
1408 char *data;
1409 static char com_token[MAX_TOKEN_CHARS];
1410
1411 data = *data_p;
1412 len = 0;
1413 com_token[0] = 0;
1414
1415 // make sure incoming data is valid
1416 if ( !data ) {
1417 *data_p = NULL;
1418 return com_token;
1419 }
1420
1421 while ( 1 ) {
1422 // skip whitespace
1423 while( (c = *data) <= ' ') {
1424 if( !c ) {
1425 break;
1426 }
1427 data++;
1428 }
1429
1430
1431 c = *data;
1432
1433 // skip double slash comments
1434 if ( c == '/' && data[1] == '/' )
1435 {
1436 data += 2;
1437 while (*data && *data != '\n') {
1438 data++;
1439 }
1440 }
1441 // skip /* */ comments
1442 else if ( c=='/' && data[1] == '*' )
1443 {
1444 data += 2;
1445 while ( *data && ( *data != '*' || data[1] != '/' ) )
1446 {
1447 data++;
1448 }
1449 if ( *data )
1450 {
1451 data += 2;
1452 }
1453 }
1454 else
1455 {
1456 break;
1457 }
1458 }
1459
1460 if ( c == 0 ) {
1461 return "";
1462 }
1463
1464 // handle quoted strings
1465 if (c == '\"')
1466 {
1467 data++;
1468 while (1)
1469 {
1470 c = *data++;
1471 if (c=='\"' || !c)
1472 {
1473 com_token[len] = 0;
1474 *data_p = ( char * ) data;
1475 return com_token;
1476 }
1477 if (len < MAX_TOKEN_CHARS - 1)
1478 {
1479 com_token[len] = c;
1480 len++;
1481 }
1482 }
1483 }
1484
1485 // parse a regular word
1486 do
1487 {
1488 if (len < MAX_TOKEN_CHARS - 1)
1489 {
1490 com_token[len] = c;
1491 len++;
1492 }
1493 data++;
1494 c = *data;
1495 } while (c>32 && c != ',' );
1496
1497 com_token[len] = 0;
1498
1499 *data_p = ( char * ) data;
1500 return com_token;
1501}

1506RE_RegisterSkin
1507
1508===============
1509*/
1510qhandle_t RE_RegisterSkin( const char *name ) {
1511 skinSurface_t parseSurfaces[MAX_SKIN_SURFACES];
1512 qhandle_t hSkin;
1513 skin_t *skin;
1514 skinSurface_t *surf;
1515 union {
1516 char *c;
1517 void *v;
1518 } text;
1519 char *text_p;
1520 char *token;
1521 char surfName[MAX_QPATH];
1522 int totalSurfaces;

1534
1535 // see if the skin is already loaded
1536 for ( hSkin = 1; hSkin < tr.numSkins ; hSkin++ ) {
1537 skin = tr.skins[hSkin];
1538 if ( !Q_stricmp( skin->name, name ) ) {
1539 if( skin->numSurfaces == 0 ) {
1540 return 0; // default skin
1541 }
1542 return hSkin;
1543 }
1544 }
1545
1546 // allocate a new skin
1547 if ( tr.numSkins == MAX_SKINS ) {
1548 ri.Printf( PRINT_WARNING, "WARNING: RE_RegisterSkin( '%s' ) MAX_SKINS hit\n", name );
1549 return 0;
1550 }
1551 tr.numSkins++;
1552 skin = ri.Hunk_Alloc( sizeof( skin_t ), h_low );
1553 tr.skins[hSkin] = skin;
1554 Q_strncpyz( skin->name, name, sizeof( skin->name ) );
1555 skin->numSurfaces = 0;
1556
1557 R_IssuePendingRenderCommands();
1558
1559 // If not a .skin file, load as a single shader
1560 if ( strcmp( name + strlen( name ) - 5, ".skin" ) ) {
1561 skin->numSurfaces = 1;
1562 skin->surfaces = ri.Hunk_Alloc( sizeof( skinSurface_t ), h_low );
1563 skin->surfaces[0].shader = R_FindShader( name, LIGHTMAP_NONE, qtrue );
1564 return hSkin;
1565 }
1566
1567 // load and parse the skin file
1568 ri.FS_ReadFile( name, &text.v );
1569 if ( !text.c ) {
1570 return 0;
1571 }
1572
1573 totalSurfaces = 0;
1574 text_p = text.c;
1575 while ( text_p && *text_p ) {
1576 // get surface name
1577 token = CommaParse( &text_p );
1578 Q_strncpyz( surfName, token, sizeof( surfName ) );
1579
1580 if ( !token[0] ) {
1581 break;
1582 }
1583 // lowercase the surface name so skin compares are faster
1584 Q_strlwr( surfName );
1585
1586 if ( *text_p == ',' ) {
1587 text_p++;
1588 }
1589
1590 if ( strstr( token, "tag_" ) ) {
1591 continue;
1592 }
1593
1594 // parse the shader name
1595 token = CommaParse( &text_p );
1596
1597 if ( skin->numSurfaces < MAX_SKIN_SURFACES ) {
1598 surf = &parseSurfaces[skin->numSurfaces];
1599 Q_strncpyz( surf->name, surfName, sizeof( surf->name ) );
1600 surf->shader = R_FindShader( token, LIGHTMAP_NONE, qtrue );
1601 skin->numSurfaces++;
1602 }
1603
1604 totalSurfaces++;
1605 }
1606
1607 ri.FS_FreeFile( text.v );
1608
1609 if ( totalSurfaces > MAX_SKIN_SURFACES ) {
1610 ri.Printf( PRINT_WARNING, "WARNING: Ignoring excess surfaces (found %d, max is %d) in skin '%s'!\n",
1611 totalSurfaces, MAX_SKIN_SURFACES, name );
1612 }
1613
1614 // never let a skin have 0 shaders
1615 if ( skin->numSurfaces == 0 ) {
1616 return 0; // use default skin
1617 }
1618
1619 // copy surfaces to skin
1620 skin->surfaces = ri.Hunk_Alloc( skin->numSurfaces * sizeof( skinSurface_t ), h_low );
1621 memcpy( skin->surfaces, parseSurfaces, skin->numSurfaces * sizeof( skinSurface_t ) );
1622
1623 return hSkin;
1624}

44static void HSVtoRGB( float h, float s, float v, float rgb[3] )
45{
46 int i;
47 float f;
48 float p, q, t;
49
50 h *= 5;
51
52 i = floor( h );
53 f = h - i;
54
55 p = v * ( 1 - s );
56 q = v * ( 1 - s * f );
57 t = v * ( 1 - s * ( 1 - f ) );
58
59 switch ( i )
60 {
61 case 0:
62 rgb[0] = v;
63 rgb[1] = t;
64 rgb[2] = p;
65 break;
66 case 1:
67 rgb[0] = q;
68 rgb[1] = v;
69 rgb[2] = p;
70 break;
71 case 2:
72 rgb[0] = p;
73 rgb[1] = v;
74 rgb[2] = t;
75 break;
76 case 3:
77 rgb[0] = p;
78 rgb[1] = q;
79 rgb[2] = v;
80 break;
81 case 4:
82 rgb[0] = t;
83 rgb[1] = p;
84 rgb[2] = v;
85 break;
86 case 5:
87 rgb[0] = v;
88 rgb[1] = p;
89 rgb[2] = q;
90 break;
91 }
92}

94/*
95===============
96R_ColorShiftLightingBytes
97
98===============
99*/
100static void R_ColorShiftLightingBytes( byte in[4], byte out[4] ) {
101 int shift, r, g, b;
102
103 // shift the color data based on overbright range
104 shift = r_mapOverBrightBits->integer - tr.overbrightBits;
105
106 // shift the data based on overbright range
107 r = in[0] << shift;
108 g = in[1] << shift;
109 b = in[2] << shift;
110
111 // normalize by color instead of saturating to white
112 if ( ( r | g | b ) > 255 ) {
113 int max;
114
115 max = r > g ? r : g;
116 max = max > b ? max : b;
117 r = r * 255 / max;
118 g = g * 255 / max;
119 b = b * 255 / max;
120 }
121
122 out[0] = r;
123 out[1] = g;
124 out[2] = b;
125 out[3] = in[3];
126}

128/*
129===============
130R_LoadLightmaps
131
132===============
133*/
134#define LIGHTMAP_SIZE 128
135static void R_LoadLightmaps( lump_t *l ) {
136 byte *buf, *buf_p;
137 int len;
138 byte image[LIGHTMAP_SIZE*LIGHTMAP_SIZE*4];
139 int i, j;
140 float maxIntensity = 0;
141 double sumIntensity = 0;
142
143 len = l->filelen;
144 if ( !len ) {
145 return;
146 }
147 buf = fileBase + l->fileofs;
148
149 // we are about to upload textures
150 R_IssuePendingRenderCommands();
151
152 // create all the lightmaps
153 tr.numLightmaps = len / (LIGHTMAP_SIZE * LIGHTMAP_SIZE * 3);
154 if ( tr.numLightmaps == 1 ) {
155 //FIXME: HACK: maps with only one lightmap turn up fullbright for some reason.
156 //this avoids this, but isn't the correct solution.
157 tr.numLightmaps++;
158 }
159
160 // if we are in r_vertexLight mode, we don't need the lightmaps at all
161 if ( r_vertexLight->integer || glConfig.hardwareType == GLHW_PERMEDIA2 ) {
162 return;
163 }
164
165 tr.lightmaps = ri.Hunk_Alloc( tr.numLightmaps * sizeof(image_t *), h_low );
166 for ( i = 0 ; i < tr.numLightmaps ; i++ ) {
167 // expand the 24 bit on-disk to 32 bit
168 buf_p = buf + i * LIGHTMAP_SIZE*LIGHTMAP_SIZE * 3;
169
170 if ( r_lightmap->integer == 2 )
171 { // color code by intensity as development tool (FIXME: check range)
172 for ( j = 0; j < LIGHTMAP_SIZE * LIGHTMAP_SIZE; j++ )
173 {
174 float r = buf_p[j*3+0];
175 float g = buf_p[j*3+1];
176 float b = buf_p[j*3+2];
177 float intensity;
178 float out[3] = {0.0, 0.0, 0.0};
179
180 intensity = 0.33f * r + 0.685f * g + 0.063f * b;
181
182 if ( intensity > 255 )
183 intensity = 1.0f;
184 else
185 intensity /= 255.0f;
186
187 if ( intensity > maxIntensity )
188 maxIntensity = intensity;
189
190 HSVtoRGB( intensity, 1.00, 0.50, out );
191
192 image[j*4+0] = out[0] * 255;
193 image[j*4+1] = out[1] * 255;
194 image[j*4+2] = out[2] * 255;
195 image[j*4+3] = 255;
196
197 sumIntensity += intensity;
198 }
199 } else {
200 for ( j = 0 ; j < LIGHTMAP_SIZE * LIGHTMAP_SIZE; j++ ) {
201 R_ColorShiftLightingBytes( &buf_p[j*3], &image[j*4] );
202 image[j*4+3] = 255;
203 }
204 }
205 tr.lightmaps[i] = R_CreateImage( va("*lightmap%d",i), image,
206 LIGHTMAP_SIZE, LIGHTMAP_SIZE, IMGTYPE_COLORALPHA,
207 IMGFLAG_NOLIGHTSCALE | IMGFLAG_NO_COMPRESSION | IMGFLAG_CLAMPTOEDGE, 0 );
208 }
209
210 if ( r_lightmap->integer == 2 ) {
211 ri.Printf( PRINT_ALL, "Brightest lightmap value: %d\n", ( int ) ( maxIntensity * 255 ) );
212 }
213}

268===============
269ShaderForShaderNum
270===============
271*/
272static shader_t *ShaderForShaderNum( int shaderNum, int lightmapNum ) {
273 shader_t *shader;
274 dshader_t *dsh;
275
276 int _shaderNum = LittleLong( shaderNum );
277 if ( _shaderNum < 0 || _shaderNum >= s_worldData.numShaders ) {
278 ri.Error( ERR_DROP, "ShaderForShaderNum: bad num %i", _shaderNum );
279 }
280 dsh = &s_worldData.shaders[ _shaderNum ];
281
282 if ( r_vertexLight->integer || glConfig.hardwareType == GLHW_PERMEDIA2 ) {
283 lightmapNum = LIGHTMAP_BY_VERTEX;
284 }
285
286 if ( r_fullbright->integer ) {
287 lightmapNum = LIGHTMAP_WHITEIMAGE;
288 }
289
290 shader = R_FindShader( dsh->shader, lightmapNum, qtrue );
291
292 // if the shader had errors, just use default shader
293 if ( shader->defaultShader ) {
294 return tr.defaultShader;
295 }
296
297 return shader;
298}

300/*
301===============
302ParseFace
303===============
304*/
305static void ParseFace( dsurface_t *ds, drawVert_t *verts, msurface_t *surf, int *indexes ) {
306 int i, j;
307 srfSurfaceFace_t *cv;
308 int numPoints, numIndexes;
309 int lightmapNum;
310 int sfaceSize, ofsIndexes;
311
312 lightmapNum = LittleLong( ds->lightmapNum );
313
314 // get fog volume
315 surf->fogIndex = LittleLong( ds->fogNum ) + 1;
316
317 // get shader value
318 surf->shader = ShaderForShaderNum( ds->shaderNum, lightmapNum );
319 if ( r_singleShader->integer && !surf->shader->isSky ) {
320 surf->shader = tr.defaultShader;
321 }
322
323 numPoints = LittleLong( ds->numVerts );
324 if (numPoints > MAX_FACE_POINTS) {
325 ri.Printf( PRINT_WARNING, "WARNING: MAX_FACE_POINTS exceeded: %i\n", numPoints);
326 numPoints = MAX_FACE_POINTS;
327 surf->shader = tr.defaultShader;
328 }
329
330 numIndexes = LittleLong( ds->numIndexes );
331
332 // create the srfSurfaceFace_t
333 sfaceSize = ( size_t ) &((srfSurfaceFace_t *)0)->points[numPoints];
334 ofsIndexes = sfaceSize;
335 sfaceSize += sizeof( int ) * numIndexes;
336
337 cv = ri.Hunk_Alloc( sfaceSize, h_low );
338 cv->surfaceType = SF_FACE;
339 cv->numPoints = numPoints;
340 cv->numIndices = numIndexes;
341 cv->ofsIndices = ofsIndexes;
342
343 verts += LittleLong( ds->firstVert );
344 for ( i = 0 ; i < numPoints ; i++ ) {
345 for ( j = 0 ; j < 3 ; j++ ) {
346 cv->points[i][j] = LittleFloat( verts[i].xyz[j] );
347 }
348 for ( j = 0 ; j < 2 ; j++ ) {
349 cv->points[i][3+j] = LittleFloat( verts[i].st[j] );
350 cv->points[i][5+j] = LittleFloat( verts[i].lightmap[j] );
351 }
352 R_ColorShiftLightingBytes( verts[i].color, (byte *)&cv->points[i][7] );
353 }
354
355 indexes += LittleLong( ds->firstIndex );
356 for ( i = 0 ; i < numIndexes ; i++ ) {
357 ((int *)((byte *)cv + cv->ofsIndices ))[i] = LittleLong( indexes[ i ] );
358 }
359
360 // take the plane information from the lightmap vector
361 for ( i = 0 ; i < 3 ; i++ ) {
362 cv->plane.normal[i] = LittleFloat( ds->lightmapVecs[2][i] );
363 }
364 cv->plane.dist = DotProduct( cv->points[0], cv->plane.normal );
365 SetPlaneSignbits( &cv->plane );
366 cv->plane.type = PlaneTypeForNormal( cv->plane.normal );
367
368 surf->data = (surfaceType_t *)cv;
369}

372/*
373===============
374ParseMesh
375===============
376*/
377static void ParseMesh ( dsurface_t *ds, drawVert_t *verts, msurface_t *surf ) {
378 srfGridMesh_t *grid;
379 int i, j;
380 int width, height, numPoints;
381 drawVert_t points[MAX_PATCH_SIZE*MAX_PATCH_SIZE];
382 int lightmapNum;
383 vec3_t bounds[2];
384 vec3_t tmpVec;
385 static surfaceType_t skipData = SF_SKIP;
386
387 lightmapNum = LittleLong( ds->lightmapNum );
388
389 // get fog volume
390 surf->fogIndex = LittleLong( ds->fogNum ) + 1;
391
392 // get shader value
393 surf->shader = ShaderForShaderNum( ds->shaderNum, lightmapNum );
394 if ( r_singleShader->integer && !surf->shader->isSky ) {
395 surf->shader = tr.defaultShader;
396 }
397
398 // we may have a nodraw surface, because they might still need to
399 // be around for movement clipping
400 if ( s_worldData.shaders[ LittleLong( ds->shaderNum ) ].surfaceFlags & SURF_NODRAW ) {
401 surf->data = &skipData;
402 return;
403 }
404
405 width = LittleLong( ds->patchWidth );
406 height = LittleLong( ds->patchHeight );
407
408 verts += LittleLong( ds->firstVert );
409 numPoints = width * height;
410 for ( i = 0 ; i < numPoints ; i++ ) {
411 for ( j = 0 ; j < 3 ; j++ ) {
412 points[i].xyz[j] = LittleFloat( verts[i].xyz[j] );
413 points[i].normal[j] = LittleFloat( verts[i].normal[j] );
414 }
415 for ( j = 0 ; j < 2 ; j++ ) {
416 points[i].st[j] = LittleFloat( verts[i].st[j] );
417 points[i].lightmap[j] = LittleFloat( verts[i].lightmap[j] );
418 }
419 R_ColorShiftLightingBytes( verts[i].color, points[i].color );
420 }
421
422 // pre-tesseleate
423 grid = R_SubdividePatchToGrid( width, height, points );
424 surf->data = (surfaceType_t *)grid;
425
426 // copy the level of detail origin, which is the center
427 // of the group of all curves that must subdivide the same
428 // to avoid cracking
429 for ( i = 0 ; i < 3 ; i++ ) {
430 bounds[0][i] = LittleFloat( ds->lightmapVecs[0][i] );
431 bounds[1][i] = LittleFloat( ds->lightmapVecs[1][i] );
432 }
433 VectorAdd( bounds[0], bounds[1], bounds[1] );
434 VectorScale( bounds[1], 0.5f, grid->lodOrigin );
435 VectorSubtract( bounds[0], grid->lodOrigin, tmpVec );
436 grid->lodRadius = VectorLength( tmpVec );
437}

439/*
440===============
441ParseTriSurf
442===============
443*/
444static void ParseTriSurf( dsurface_t *ds, drawVert_t *verts, msurface_t *surf, int *indexes ) {
445 srfTriangles_t *tri;
446 int i, j;
447 int numVerts, numIndexes;
448
449 // get fog volume
450 surf->fogIndex = LittleLong( ds->fogNum ) + 1;
451
452 // get shader
453 surf->shader = ShaderForShaderNum( ds->shaderNum, LIGHTMAP_BY_VERTEX );
454 if ( r_singleShader->integer && !surf->shader->isSky ) {
455 surf->shader = tr.defaultShader;
456 }
457
458 numVerts = LittleLong( ds->numVerts );
459 numIndexes = LittleLong( ds->numIndexes );
460
461 tri = ri.Hunk_Alloc( sizeof( *tri ) + numVerts * sizeof( tri->verts[0] )
462 + numIndexes * sizeof( tri->indexes[0] ), h_low );
463 tri->surfaceType = SF_TRIANGLES;
464 tri->numVerts = numVerts;
465 tri->numIndexes = numIndexes;
466 tri->verts = (drawVert_t *)(tri + 1);
467 tri->indexes = (int *)(tri->verts + tri->numVerts );
468
469 surf->data = (surfaceType_t *)tri;
470
471 // copy vertexes
472 ClearBounds( tri->bounds[0], tri->bounds[1] );
473 verts += LittleLong( ds->firstVert );
474 for ( i = 0 ; i < numVerts ; i++ ) {
475 for ( j = 0 ; j < 3 ; j++ ) {
476 tri->verts[i].xyz[j] = LittleFloat( verts[i].xyz[j] );
477 tri->verts[i].normal[j] = LittleFloat( verts[i].normal[j] );
478 }
479 AddPointToBounds( tri->verts[i].xyz, tri->bounds[0], tri->bounds[1] );
480 for ( j = 0 ; j < 2 ; j++ ) {
481 tri->verts[i].st[j] = LittleFloat( verts[i].st[j] );
482 tri->verts[i].lightmap[j] = LittleFloat( verts[i].lightmap[j] );
483 }
484
485 R_ColorShiftLightingBytes( verts[i].color, tri->verts[i].color );
486 }
487
488 // copy indexes
489 indexes += LittleLong( ds->firstIndex );
490 for ( i = 0 ; i < numIndexes ; i++ ) {
491 tri->indexes[i] = LittleLong( indexes[i] );
492 if ( tri->indexes[i] < 0 || tri->indexes[i] >= numVerts ) {
493 ri.Error( ERR_DROP, "Bad index in triangle surface" );
494 }
495 }
496}

529/*
530=================
531R_MergedWidthPoints
532
533returns true if there are grid points merged on a width edge
534=================
535*/
536int R_MergedWidthPoints(srfGridMesh_t *grid, int offset) {
537 int i, j;
538
539 for (i = 1; i < grid->width-1; i++) {
540 for (j = i + 1; j < grid->width-1; j++) {
541 if ( fabs(grid->verts[i + offset].xyz[0] - grid->verts[j + offset].xyz[0]) > .1) continue;
542 if ( fabs(grid->verts[i + offset].xyz[1] - grid->verts[j + offset].xyz[1]) > .1) continue;
543 if ( fabs(grid->verts[i + offset].xyz[2] - grid->verts[j + offset].xyz[2]) > .1) continue;
544 return qtrue;
545 }
546 }
547 return qfalse;
548}

557int R_MergedHeightPoints(srfGridMesh_t *grid, int offset) {

565 return qtrue;
566 }
567 }
568 return qfalse;
569}

571/*
572=================
573R_FixSharedVertexLodError_r
574
575NOTE: never sync LoD through grid edges with merged points!
576
577FIXME: write generalized version that also avoids cracks between a patch and one that meets half way?
578=================
579*/
580void R_FixSharedVertexLodError_r( int start, srfGridMesh_t *grid1 ) {
581 int j, k, l, m, n, offset1, offset2, touch;
582 srfGridMesh_t *grid2;
583
584 for ( j = start; j < s_worldData.numsurfaces; j++ ) {
585 //
586 grid2 = (srfGridMesh_t *) s_worldData.surfaces[j].data;
587 // if this surface is not a grid
588 if ( grid2->surfaceType != SF_GRID ) continue;
589 // if the LOD errors are already fixed for this patch
590 if ( grid2->lodFixed == 2 ) continue;
591 // grids in the same LOD group should have the exact same lod radius
592 if ( grid1->lodRadius != grid2->lodRadius ) continue;
593 // grids in the same LOD group should have the exact same lod origin
594 if ( grid1->lodOrigin[0] != grid2->lodOrigin[0] ) continue;
595 if ( grid1->lodOrigin[1] != grid2->lodOrigin[1] ) continue;
596 if ( grid1->lodOrigin[2] != grid2->lodOrigin[2] ) continue;
597 //
598 touch = qfalse;
599 for (n = 0; n < 2; n++) {
600 //
601 if (n) offset1 = (grid1->height-1) * grid1->width;
602 else offset1 = 0;
603 if (R_MergedWidthPoints(grid1, offset1)) continue;
604 for (k = 1; k < grid1->width-1; k++) {
605 for (m = 0; m < 2; m++) {
606
607 if (m) offset2 = (grid2->height-1) * grid2->width;
608 else offset2 = 0;
609 if (R_MergedWidthPoints(grid2, offset2)) continue;
610 for ( l = 1; l < grid2->width-1; l++) {
611 //
612 if ( fabs(grid1->verts[k + offset1].xyz[0] - grid2->verts[l + offset2].xyz[0]) > .1) continue;
613 if ( fabs(grid1->verts[k + offset1].xyz[1] - grid2->verts[l + offset2].xyz[1]) > .1) continue;
614 if ( fabs(grid1->verts[k + offset1].xyz[2] - grid2->verts[l + offset2].xyz[2]) > .1) continue;
615 // ok the points are equal and should have the same lod error
616 grid2->widthLodError[l] = grid1->widthLodError[k];
617 touch = qtrue;
618 }
619 }
620 for (m = 0; m < 2; m++) {
621
622 if (m) offset2 = grid2->width-1;
623 else offset2 = 0;
624 if (R_MergedHeightPoints(grid2, offset2)) continue;
625 for ( l = 1; l < grid2->height-1; l++) {
626 //
627 if ( fabs(grid1->verts[k + offset1].xyz[0] - grid2->verts[grid2->width * l + offset2].xyz[0]) > .1) continue;
628 if ( fabs(grid1->verts[k + offset1].xyz[1] - grid2->verts[grid2->width * l + offset2].xyz[1]) > .1) continue;
629 if ( fabs(grid1->verts[k + offset1].xyz[2] - grid2->verts[grid2->width * l + offset2].xyz[2]) > .1) continue;
630 // ok the points are equal and should have the same lod error
631 grid2->heightLodError[l] = grid1->widthLodError[k];
632 touch = qtrue;
633 }
634 }
635 }
636 }
637 for (n = 0; n < 2; n++) {
638 //
639 if (n) offset1 = grid1->width-1;
640 else offset1 = 0;
641 if (R_MergedHeightPoints(grid1, offset1)) continue;
642 for (k = 1; k < grid1->height-1; k++) {
643 for (m = 0; m < 2; m++) {
644
645 if (m) offset2 = (grid2->height-1) * grid2->width;
646 else offset2 = 0;
647 if (R_MergedWidthPoints(grid2, offset2)) continue;
648 for ( l = 1; l < grid2->width-1; l++) {
649 //
650 if ( fabs(grid1->verts[grid1->width * k + offset1].xyz[0] - grid2->verts[l + offset2].xyz[0]) > .1) continue;
651 if ( fabs(grid1->verts[grid1->width * k + offset1].xyz[1] - grid2->verts[l + offset2].xyz[1]) > .1) continue;
652 if ( fabs(grid1->verts[grid1->width * k + offset1].xyz[2] - grid2->verts[l + offset2].xyz[2]) > .1) continue;
653 // ok the points are equal and should have the same lod error
654 grid2->widthLodError[l] = grid1->heightLodError[k];
655 touch = qtrue;
656 }
657 }
658 for (m = 0; m < 2; m++) {
659
660 if (m) offset2 = grid2->width-1;
661 else offset2 = 0;
662 if (R_MergedHeightPoints(grid2, offset2)) continue;
663 for ( l = 1; l < grid2->height-1; l++) {
664 //
665 if ( fabs(grid1->verts[grid1->width * k + offset1].xyz[0] - grid2->verts[grid2->width * l + offset2].xyz[0]) > .1) continue;
666 if ( fabs(grid1->verts[grid1->width * k + offset1].xyz[1] - grid2->verts[grid2->width * l + offset2].xyz[1]) > .1) continue;
667 if ( fabs(grid1->verts[grid1->width * k + offset1].xyz[2] - grid2->verts[grid2->width * l + offset2].xyz[2]) > .1) continue;
668 // ok the points are equal and should have the same lod error
669 grid2->heightLodError[l] = grid1->heightLodError[k];
670 touch = qtrue;

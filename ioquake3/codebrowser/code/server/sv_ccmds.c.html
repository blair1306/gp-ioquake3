35/*
36==================
37SV_GetPlayerByHandle
38
39Returns the player with player id or name from Cmd_Argv(1)
40==================
41*/
42static client_t *SV_GetPlayerByHandle( void ) {
43 client_t *cl;
44 int i;
45 char *s;
46 char cleanName[64];
47
48 // make sure server is running
49 if ( !com_sv_running->integer ) {
50 return NULL;
51 }
52
53 if ( Cmd_Argc() < 2 ) {
54 Com_Printf( "No player specified.\n" );
55 return NULL;
56 }
57
58 s = Cmd_Argv(1);
59
60 // Check whether this is a numeric player handle
61 for(i = 0; s[i] >= '0' && s[i] <= '9'; i++);
62
63 if(!s[i])
64 {
65 int plid = atoi(s);
66
67 // Check for numeric playerid match
68 if(plid >= 0 && plid < sv_maxclients->integer)
69 {
70 cl = &svs.clients[plid];
71
72 if(cl->state)
73 return cl;
74 }
75 }
76
77 // check for a name match
78 for ( i=0, cl=svs.clients ; i < sv_maxclients->integer ; i++,cl++ ) {
79 if ( !cl->state ) {
80 continue;
81 }
82 if ( !Q_stricmp( cl->name, s ) ) {
83 return cl;
84 }
85
86 Q_strncpyz( cleanName, cl->name, sizeof(cleanName) );
87 Q_CleanStr( cleanName );
88 if ( !Q_stricmp( cleanName, s ) ) {
89 return cl;
90 }
91 }
92
93 Com_Printf( "Player %s is not on the server\n", s );
94
95 return NULL;
96}

98/*
99==================
100SV_GetPlayerByNum
101
102Returns the player with idnum from Cmd_Argv(1)
103==================
104*/
105static client_t *SV_GetPlayerByNum( void ) {
106 client_t *cl;
107 int i;
108 int idnum;
109 char *s;
110
111 // make sure server is running
112 if ( !com_sv_running->integer ) {
113 return NULL;
114 }
115
116 if ( Cmd_Argc() < 2 ) {
117 Com_Printf( "No player specified.\n" );
118 return NULL;
119 }
120
121 s = Cmd_Argv(1);
122
123 for (i = 0; s[i]; i++) {
124 if (s[i] < '0' || s[i] > '9') {
125 Com_Printf( "Bad slot number: %s\n", s);
126 return NULL;
127 }
128 }
129 idnum = atoi( s );
130 if ( idnum < 0 || idnum >= sv_maxclients->integer ) {
131 Com_Printf( "Bad client slot: %i\n", idnum );
132 return NULL;
133 }
134
135 cl = &svs.clients[idnum];
136 if ( !cl->state ) {
137 Com_Printf( "Client %i is not active\n", idnum );
138 return NULL;
139 }
140 return cl;
141}

146/*
147==================
148SV_Map_f
149
150Restart the server on a different map
151==================
152*/
153static void SV_Map_f( void ) {
154 char *cmd;
155 char *map;
156 qboolean killBots, cheat;
157 char expanded[MAX_QPATH];
158 char mapname[MAX_QPATH];
159
160 map = Cmd_Argv(1);
161 if ( !map ) {
162 return;
163 }
164
165 // make sure the level exists before trying to change, so that
166 // a typo at the server console won't end the game
167 Com_sprintf (expanded, sizeof(expanded), "maps/%s.bsp", map);
168 if ( FS_ReadFile (expanded, NULL) == -1 ) {
169 Com_Printf ("Can't find map %s\n", expanded);
170 return;
171 }
172
173 // force latched values to get set
174 Cvar_Get ("g_gametype", "0", CVAR_SERVERINFO | CVAR_USERINFO | CVAR_LATCH );
175
176 cmd = Cmd_Argv(0);
177 if( Q_stricmpn( cmd, "sp", 2 ) == 0 ) {
178 Cvar_SetValue( "g_gametype", GT_SINGLE_PLAYER );
179 Cvar_SetValue( "g_doWarmup", 0 );
180 // may not set sv_maxclients directly, always set latched
181 Cvar_SetLatched( "sv_maxclients", "8" );
182 cmd += 2;
183 if (!Q_stricmp( cmd, "devmap" ) ) {
184 cheat = qtrue;
185 } else {
186 cheat = qfalse;
187 }
188 killBots = qtrue;
189 }
190 else {
191 if ( !Q_stricmp( cmd, "devmap" ) ) {
192 cheat = qtrue;
193 killBots = qtrue;
194 } else {
195 cheat = qfalse;
196 killBots = qfalse;
197 }
198 if( sv_gametype->integer == GT_SINGLE_PLAYER ) {
199 Cvar_SetValue( "g_gametype", GT_FFA );
200 }
201 }
202
203 // save the map name here cause on a map restart we reload the q3config.cfg
204 // and thus nuke the arguments of the map command
205 Q_strncpyz(mapname, map, sizeof(mapname));
206
207 // start up the map
208 SV_SpawnServer( mapname, killBots );
209
210 // set the cheat value
211 // if the level was started with "map <levelname>", then
212 // cheats will not be allowed. If started with "devmap <levelname>"
213 // then cheats will be allowed
214 if ( cheat ) {
215 Cvar_Set( "sv_cheats", "1" );
216 } else {
217 Cvar_Set( "sv_cheats", "0" );
218 }
219}

221/*
222================
223SV_MapRestart_f
224
225Completely restarts a level, but doesn't send a new gamestate to the clients.
226This allows fair starts with variable load times.
227================
228*/
229static void SV_MapRestart_f( void ) {
230 int i;
231 client_t *client;
232 char *denied;
233 qboolean isBot;
234 int delay;
235
236 // make sure we aren't restarting twice in the same frame
237 if ( com_frameTime == sv.serverId ) {
238 return;
239 }
240
241 // make sure server is running
242 if ( !com_sv_running->integer ) {
243 Com_Printf( "Server is not running.\n" );
244 return;
245 }
246
247 if ( sv.restartTime ) {
248 return;
249 }
250
251 if (Cmd_Argc() > 1 ) {
252 delay = atoi( Cmd_Argv(1) );
253 }
254 else {
255 delay = 5;
256 }
257 if( delay && !Cvar_VariableValue("g_doWarmup") ) {
258 sv.restartTime = sv.time + delay * 1000;
259 SV_SetConfigstring( CS_WARMUP, va("%i", sv.restartTime) );
260 return;
261 }
262
263 // check for changes in variables that can't just be restarted
264 // check for maxclients change
265 if ( sv_maxclients->modified || sv_gametype->modified ) {
266 char mapname[MAX_QPATH];
267
268 Com_Printf( "variable change -- restarting.\n" );
269 // restart the map the slow way
270 Q_strncpyz( mapname, Cvar_VariableString( "mapname" ), sizeof( mapname ) );
271
272 SV_SpawnServer( mapname, qfalse );
273 return;
274 }
275
276 // toggle the server bit so clients can detect that a
277 // map_restart has happened
278 svs.snapFlagServerBit ^= SNAPFLAG_SERVERCOUNT;
279
280 // generate a new serverid
281 // TTimo - don't update restartedserverId there, otherwise we won't deal correctly with multiple map_restart
282 sv.serverId = com_frameTime;
283 Cvar_Set( "sv_serverid", va("%i", sv.serverId ) );
284
285 // if a map_restart occurs while a client is changing maps, we need
286 // to give them the correct time so that when they finish loading
287 // they don't violate the backwards time check in cl_cgame.c
288 for (i=0 ; i<sv_maxclients->integer ; i++) {
289 if (svs.clients[i].state == CS_PRIMED) {
290 svs.clients[i].oldServerTime = sv.restartTime;
291 }
292 }
293
294 // reset all the vm data in place without changing memory allocation
295 // note that we do NOT set sv.state = SS_LOADING, so configstrings that
296 // had been changed from their default values will generate broadcast updates
297 sv.state = SS_LOADING;
298 sv.restarting = qtrue;
299
300 SV_RestartGameProgs();
301
302 // run a few frames to allow everything to settle
303 for (i = 0; i < 3; i++)
304 {
305 VM_Call (gvm, GAME_RUN_FRAME, sv.time);
306 sv.time += 100;
307 svs.time += 100;
308 }
309
310 sv.state = SS_GAME;
311 sv.restarting = qfalse;
312
313 // connect and begin all the clients
314 for (i=0 ; i<sv_maxclients->integer ; i++) {
315 client = &svs.clients[i];
316
317 // send the new gamestate to all connected clients
318 if ( client->state < CS_CONNECTED) {
319 continue;
320 }
321
322 if ( client->netchan.remoteAddress.type == NA_BOT ) {
323 isBot = qtrue;
324 } else {
325 isBot = qfalse;
326 }
327
328 // add the map_restart command
329 SV_AddServerCommand( client, "map_restart\n" );
330
331 // connect the client again, without the firstTime flag
332 denied = VM_ExplicitArgPtr( gvm, VM_Call( gvm, GAME_CLIENT_CONNECT, i, qfalse, isBot ) );
333 if ( denied ) {
334 // this generally shouldn't happen, because the client
335 // was connected before the level change
336 SV_DropClient( client, denied );
337 Com_Printf( "SV_MapRestart_f(%d): dropped client %i - denied!\n", delay, i );
338 continue;
339 }
340
341 if(client->state == CS_ACTIVE)
342 SV_ClientEnterWorld(client, &client->lastUsercmd);
343 else
344 {
345 // If we don't reset client->lastUsercmd and are restarting during map load,
346 // the client will hang because we'll use the last Usercmd from the previous map,
347 // which is wrong obviously.
348 SV_ClientEnterWorld(client, NULL);
349 }
350 }
351
352 // run another frame to allow things to look at all the players
353 VM_Call (gvm, GAME_RUN_FRAME, sv.time);
354 sv.time += 100;
355 svs.time += 100;
356}

360/*
361==================
362SV_Kick_f
363
364Kick a user off of the server
365==================
366*/
367static void SV_Kick_f( void ) {
368 client_t *cl;
369 int i;
370
371 // make sure server is running
372 if ( !com_sv_running->integer ) {
373 Com_Printf( "Server is not running.\n" );
374 return;
375 }
376
377 if ( Cmd_Argc() != 2 ) {
378 Com_Printf ("Usage: kick <player name>\nkick all = kick everyone\nkick allbots = kick all bots\n");
379 return;
380 }
381
382 cl = SV_GetPlayerByHandle();
383 if ( !cl ) {
384 if ( !Q_stricmp(Cmd_Argv(1), "all") ) {
385 for ( i=0, cl=svs.clients ; i < sv_maxclients->integer ; i++,cl++ ) {
386 if ( !cl->state ) {
387 continue;
388 }
389 if( cl->netchan.remoteAddress.type == NA_LOOPBACK ) {
390 continue;
391 }
392 SV_DropClient( cl, "was kicked" );
393 cl->lastPacketTime = svs.time; // in case there is a funny zombie
394 }
395 }
396 else if ( !Q_stricmp(Cmd_Argv(1), "allbots") ) {
397 for ( i=0, cl=svs.clients ; i < sv_maxclients->integer ; i++,cl++ ) {
398 if ( !cl->state ) {
399 continue;
400 }
401 if( cl->netchan.remoteAddress.type != NA_BOT ) {
402 continue;
403 }
404 SV_DropClient( cl, "was kicked" );
405 cl->lastPacketTime = svs.time; // in case there is a funny zombie
406 }
407 }
408 return;
409 }
410 if( cl->netchan.remoteAddress.type == NA_LOOPBACK ) {
411 Com_Printf("Cannot kick host player\n");
412 return;
413 }
414
415 SV_DropClient( cl, "was kicked" );
416 cl->lastPacketTime = svs.time; // in case there is a funny zombie
417}

520

527
528 // make sure server is running
529 if ( !com_sv_running->integer ) {
530 Com_Printf( "Server is not running.\n" );
531 return;
532 }
533
534 if ( Cmd_Argc() != 2 ) {
535 Com_Printf ("Usage: banUser <player name>\n");

537 }
538
539 cl = SV_GetPlayerByHandle();
540
541 if (!cl) {
542 return;
543 }
544
545 if( cl->netchan.remoteAddress.type == NA_LOOPBACK ) {
546 Com_Printf("Cannot kick host player\n");
547 return;
548 }
549
550 // look up the authorize server's IP
551 if ( !svs.authorizeAddress.ip[0] && svs.authorizeAddress.type != NA_BAD ) {
552 Com_Printf( "Resolving %s\n", AUTHORIZE_SERVER_NAME );
553 if ( !NET_StringToAdr( AUTHORIZE_SERVER_NAME, &svs.authorizeAddress, NA_IP ) ) {
554 Com_Printf( "Couldn't resolve address\n" );
555 return;
556 }
557 svs.authorizeAddress.port = BigShort( PORT_AUTHORIZE );
558 Com_Printf( "%s resolved to %i.%i.%i.%i:%i\n", AUTHORIZE_SERVER_NAME,
559 svs.authorizeAddress.ip[0], svs.authorizeAddress.ip[1],
560 svs.authorizeAddress.ip[2], svs.authorizeAddress.ip[3],
561 BigShort( svs.authorizeAddress.port ) );
562 }
563
564 // otherwise send their ip to the authorize server
565 if ( svs.authorizeAddress.type != NA_BAD ) {
566 NET_OutOfBandPrint( NS_SERVER, svs.authorizeAddress,
567 "banUser %i.%i.%i.%i", cl->netchan.remoteAddress.ip[0], cl->netchan.remoteAddress.ip[1],
568 cl->netchan.remoteAddress.ip[2], cl->netchan.remoteAddress.ip[3] );
569 Com_Printf("%s was banned from coming back\n", cl->name);
570 }
571}

628/*
629==================
630SV_RehashBans_f
631
632Load saved bans from file.
633==================
634*/
635static void SV_RehashBans_f(void)
636{
637 int index, filelen;
638 fileHandle_t readfrom;
639 char *textbuf, *curpos, *maskpos, *newlinepos, *endpos;
640 char filepath[MAX_QPATH];
641
642 // make sure server is running
643 if ( !com_sv_running->integer ) {
644 return;
645 }
646
647 serverBansCount = 0;
648
649 if(!sv_banFile->string || !*sv_banFile->string)
650 return;
651
652 Com_sprintf(filepath, sizeof(filepath), "%s/%s", FS_GetCurrentGameDir(), sv_banFile->string);
653
654 if((filelen = FS_SV_FOpenFileRead(filepath, &readfrom)) >= 0)
655 {
656 if(filelen < 2)
657 {
658 // Don't bother if file is too short.
659 FS_FCloseFile(readfrom);
660 return;
661 }
662
663 curpos = textbuf = Z_Malloc(filelen);
664
665 filelen = FS_Read(textbuf, filelen, readfrom);
666 FS_FCloseFile(readfrom);
667
668 endpos = textbuf + filelen;
669
670 for(index = 0; index < SERVER_MAXBANS && curpos + 2 < endpos; index++)
671 {
672 // find the end of the address string
673 for(maskpos = curpos + 2; maskpos < endpos && *maskpos != ' '; maskpos++);
674
675 if(maskpos + 1 >= endpos)
676 break;
677
678 *maskpos = '\0';
679 maskpos++;
680
681 // find the end of the subnet specifier
682 for(newlinepos = maskpos; newlinepos < endpos && *newlinepos != '\n'; newlinepos++);
683
684 if(newlinepos >= endpos)
685 break;
686
687 *newlinepos = '\0';
688
689 if(NET_StringToAdr(curpos + 2, &serverBans[index].ip, NA_UNSPEC))
690 {
691 serverBans[index].isexception = (curpos[0] != '0');
692 serverBans[index].subnet = atoi(maskpos);
693
694 if(serverBans[index].ip.type == NA_IP &&
695 (serverBans[index].subnet < 1 || serverBans[index].subnet > 32))
696 {
697 serverBans[index].subnet = 32;
698 }
699 else if(serverBans[index].ip.type == NA_IP6 &&
700 (serverBans[index].subnet < 1 || serverBans[index].subnet > 128))
701 {
702 serverBans[index].subnet = 128;
703 }
704 }
705
706 curpos = newlinepos + 1;
707 }
708
709 serverBansCount = index;
710
711 Z_Free(textbuf);
712 }
713}

715/*
716==================
717SV_WriteBans
718
719Save bans to file.
720==================
721*/
722static void SV_WriteBans(void)
723{
724 int index;
725 fileHandle_t writeto;
726 char filepath[MAX_QPATH];
727
728 if(!sv_banFile->string || !*sv_banFile->string)
729 return;
730
731 Com_sprintf(filepath, sizeof(filepath), "%s/%s", FS_GetCurrentGameDir(), sv_banFile->string);
732
733 if((writeto = FS_SV_FOpenFileWrite(filepath)))
734 {
735 char writebuf[128];
736 serverBan_t *curban;
737
738 for(index = 0; index < serverBansCount; index++)
739 {
740 curban = &serverBans[index];
741
742 Com_sprintf(writebuf, sizeof(writebuf), "%d %s %d\n",
743 curban->isexception, NET_AdrToString(curban->ip), curban->subnet);
744 FS_Write(writebuf, strlen(writebuf), writeto);
745 }
746
747 FS_FCloseFile(writeto);
748 }
749}

751/*
752==================
753SV_DelBanEntryFromList
754
755Remove a ban or an exception from the list.
756==================
757*/
758
759static qboolean SV_DelBanEntryFromList(int index)
760{
761 if(index == serverBansCount - 1)
762 serverBansCount--;
763 else if(index < ARRAY_LEN(serverBans) - 1)
764 {
765 memmove(serverBans + index, serverBans + index + 1, (serverBansCount - index - 1) * sizeof(*serverBans));
766 serverBansCount--;
767 }
768 else
769 return qtrue;
770
771 return qfalse;
772}

774/*
775==================
776SV_ParseCIDRNotation
777
778Parse a CIDR notation type string and return a netadr_t and suffix by reference
779==================
780*/
781
782static qboolean SV_ParseCIDRNotation(netadr_t *dest, int *mask, char *adrstr)
783{
784 char *suffix;
785
786 suffix = strchr(adrstr, '/');
787 if(suffix)
788 {
789 *suffix = '\0';
790 suffix++;
791 }
792
793 if(!NET_StringToAdr(adrstr, dest, NA_UNSPEC))
794 return qtrue;
795
796 if(suffix)
797 {
798 *mask = atoi(suffix);
799
800 if(dest->type == NA_IP)
801 {
802 if(*mask < 1 || *mask > 32)
803 *mask = 32;
804 }
805 else
806 {
807 if(*mask < 1 || *mask > 128)
808 *mask = 128;
809 }
810 }
811 else if(dest->type == NA_IP)
812 *mask = 32;
813 else
814 *mask = 128;
815
816 return qfalse;
817}

821SV_AddBanToList
822
823Ban a user from being able to play on this server based on his ip address.
824==================
825*/
826
827static void SV_AddBanToList(qboolean isexception)
828{
829 char *banstring;
830 char addy2[NET_ADDRSTRMAXLEN];
831 netadr_t ip;
832 int index, argc, mask;
833 serverBan_t *curban;
834
835 // make sure server is running
836 if ( !com_sv_running->integer ) {
837 Com_Printf( "Server is not running.\n" );
838 return;
839 }
840
841 argc = Cmd_Argc();
842
843 if(argc < 2 || argc > 3)
844 {
845 Com_Printf ("Usage: %s (ip[/subnet] | clientnum [subnet])\n", Cmd_Argv(0));
846 return;
847 }
848
849 if(serverBansCount >= ARRAY_LEN(serverBans))
850 {
851 Com_Printf ("Error: Maximum number of bans/exceptions exceeded.\n");
852 return;
853 }
854
855 banstring = Cmd_Argv(1);
856
857 if(strchr(banstring, '.') || strchr(banstring, ':'))
858 {
859 // This is an ip address, not a client num.
860
861 if(SV_ParseCIDRNotation(&ip, &mask, banstring))
862 {
863 Com_Printf("Error: Invalid address %s\n", banstring);
864 return;
865 }
866 }
867 else
868 {
869 client_t *cl;
870
871 // client num.
872
873 cl = SV_GetPlayerByNum();
874
875 if(!cl)
876 {
877 Com_Printf("Error: Playernum %s does not exist.\n", Cmd_Argv(1));
878 return;
879 }
880
881 ip = cl->netchan.remoteAddress;
882
883 if(argc == 3)
884 {
885 mask = atoi(Cmd_Argv(2));
886
887 if(ip.type == NA_IP)
888 {
889 if(mask < 1 || mask > 32)
890 mask = 32;
891 }
892 else
893 {
894 if(mask < 1 || mask > 128)
895 mask = 128;
896 }
897 }
898 else
899 mask = (ip.type == NA_IP6) ? 128 : 32;
900 }
901
902 if(ip.type != NA_IP && ip.type != NA_IP6)
903 {
904 Com_Printf("Error: Can ban players connected via the internet only.\n");
905 return;
906 }
907
908 // first check whether a conflicting ban exists that would supersede the new one.
909 for(index = 0; index < serverBansCount; index++)
910 {
911 curban = &serverBans[index];
912
913 if(curban->subnet <= mask)
914 {
915 if((curban->isexception || !isexception) && NET_CompareBaseAdrMask(curban->ip, ip, curban->subnet))
916 {
917 Q_strncpyz(addy2, NET_AdrToString(ip), sizeof(addy2));
918
919 Com_Printf("Error: %s %s/%d supersedes %s %s/%d\n", curban->isexception ? "Exception" : "Ban",
920 NET_AdrToString(curban->ip), curban->subnet,
921 isexception ? "exception" : "ban", addy2, mask);
922 return;
923 }
924 }
925 if(curban->subnet >= mask)
926 {
927 if(!curban->isexception && isexception && NET_CompareBaseAdrMask(curban->ip, ip, mask))
928 {
929 Q_strncpyz(addy2, NET_AdrToString(curban->ip), sizeof(addy2));
930
931 Com_Printf("Error: %s %s/%d supersedes already existing %s %s/%d\n", isexception ? "Exception" : "Ban",
932 NET_AdrToString(ip), mask,
933 curban->isexception ? "exception" : "ban", addy2, curban->subnet);
934 return;
935 }
936 }
937 }
938
939 // now delete bans that are superseded by the new one
940 index = 0;
941 while(index < serverBansCount)
942 {
943 curban = &serverBans[index];
944
945 if(curban->subnet > mask && (!curban->isexception || isexception) && NET_CompareBaseAdrMask(curban->ip, ip, mask))
946 SV_DelBanEntryFromList(index);
947 else
948 index++;
949 }
950
951 serverBans[serverBansCount].ip = ip;
952 serverBans[serverBansCount].subnet = mask;
953 serverBans[serverBansCount].isexception = isexception;
954
955 serverBansCount++;
956
957 SV_WriteBans();
958
959 Com_Printf("Added %s: %s/%d\n", isexception ? "ban exception" : "ban",
960 NET_AdrToString(ip), mask);
961}

965SV_DelBanFromList
966
967Remove a ban or an exception from the list.
968==================
969*/
970
971static void SV_DelBanFromList(qboolean isexception)
972{
973 int index, count = 0, todel, mask;
974 netadr_t ip;
975 char *banstring;
976
977 // make sure server is running
978 if ( !com_sv_running->integer ) {
979 Com_Printf( "Server is not running.\n" );
980 return;
981 }
982
983 if(Cmd_Argc() != 2)
984 {
985 Com_Printf ("Usage: %s (ip[/subnet] | num)\n", Cmd_Argv(0));
986 return;
987 }
988
989 banstring = Cmd_Argv(1);
990
991 if(strchr(banstring, '.') || strchr(banstring, ':'))
992 {
993 serverBan_t *curban;
994
995 if(SV_ParseCIDRNotation(&ip, &mask, banstring))
996 {
997 Com_Printf("Error: Invalid address %s\n", banstring);
998 return;
999 }
1000
1001 index = 0;
1002
1003 while(index < serverBansCount)
1004 {
1005 curban = &serverBans[index];
1006
1007 if(curban->isexception == isexception &&
1008 curban->subnet >= mask &&
1009 NET_CompareBaseAdrMask(curban->ip, ip, mask))
1010 {
1011 Com_Printf("Deleting %s %s/%d\n",
1012 isexception ? "exception" : "ban",
1013 NET_AdrToString(curban->ip), curban->subnet);
1014
1015 SV_DelBanEntryFromList(index);
1016 }
1017 else
1018 index++;
1019 }
1020 }
1021 else
1022 {
1023 todel = atoi(Cmd_Argv(1));
1024
1025 if(todel < 1 || todel > serverBansCount)
1026 {
1027 Com_Printf("Error: Invalid ban number given\n");
1028 return;
1029 }
1030
1031 for(index = 0; index < serverBansCount; index++)
1032 {
1033 if(serverBans[index].isexception == isexception)
1034 {
1035 count++;
1036
1037 if(count == todel)
1038 {
1039 Com_Printf("Deleting %s %s/%d\n",
1040 isexception ? "exception" : "ban",
1041 NET_AdrToString(serverBans[index].ip), serverBans[index].subnet);
1042
1043 SV_DelBanEntryFromList(index);
1044
1045 break;
1046 }
1047 }
1048 }
1049 }
1050
1051 SV_WriteBans();
1052}

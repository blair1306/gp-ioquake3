115typedef enum {
116 CS_FREE, // can be reused for a new connection
117 CS_ZOMBIE, // client has been disconnected, but don't reuse
118 // connection for a couple seconds

133typedef struct client_s {
134 clientState_t state;
135 char userinfo[MAX_INFO_STRING]; // name, etc
136
137 char reliableCommands[MAX_RELIABLE_COMMANDS][MAX_STRING_CHARS];
138 int reliableSequence; // last added reliable message, not necesarily sent or acknowledged yet
139 int reliableAcknowledge; // last acknowledged reliable message
140 int reliableSent; // last sent reliable message, not necesarily acknowledged yet
141 int messageAcknowledge;
142
143 int gamestateMessageNum; // netchan->outgoingSequence of gamestate
144 int challenge;
145
146 usercmd_t lastUsercmd;
147 int lastMessageNum; // for delta compression
148 int lastClientCommand; // reliable client message sequence
149 char lastClientCommandString[MAX_STRING_CHARS];
150 sharedEntity_t *gentity; // SV_GentityNum(clientnum)
151 char name[MAX_NAME_LENGTH]; // extracted from userinfo, high bits masked
152
153 // downloading
154 char downloadName[MAX_QPATH]; // if not empty string, we are downloading
155 fileHandle_t download; // file being downloaded
156 int downloadSize; // total bytes (can't use EOF because of paks)
157 int downloadCount; // bytes sent
158 int downloadClientBlock; // last block we sent to the client, awaiting ack
159 int downloadCurrentBlock; // current block number
160 int downloadXmitBlock; // last block we xmited
161 unsigned char *downloadBlocks[MAX_DOWNLOAD_WINDOW]; // the buffers for the download blocks
162 int downloadBlockSize[MAX_DOWNLOAD_WINDOW];
163 qboolean downloadEOF; // We have sent the EOF block
164 int downloadSendTime; // time we last got an ack from the client
165
166 int deltaMessage; // frame last client usercmd message
167 int nextReliableTime; // svs.time when another reliable command will be allowed
168 int lastPacketTime; // svs.time when packet was last received
169 int lastConnectTime; // svs.time when connection started
170 int lastSnapshotTime; // svs.time of last sent snapshot
171 qboolean rateDelayed; // true if nextSnapshotTime was set based on rate instead of snapshotMsec
172 int timeoutCount; // must timeout a few frames in a row so debugging doesn't break
173 clientSnapshot_t frames[PACKET_BACKUP]; // updates can be delta'd from here
174 int ping;
175 int rate; // bytes / second
176 int snapshotMsec; // requests a snapshot every snapshotMsec unless rate choked
177 int pureAuthentic;
178 qboolean gotCP; // TTimo - additional flag to distinguish between a bad pure checksum, and no cp command at all
179 netchan_t netchan;
180 // TTimo
181 // queuing outgoing fragmented messages to send them properly, without udp packet bursts
182 // in case large fragmented messages are stacking up
183 // buffer them into this queue, and hand them out to netchan as needed
184 netchan_buffer_t *netchan_start_queue;
185 netchan_buffer_t **netchan_end_queue;

191 voipServerPacket_t *voipPacket[VOIP_QUEUE_LENGTH];
192 int queuedVoipPackets;
193 int queuedVoipIndex;
194#endif
195
196 int oldServerTime;
197 qboolean csUpdated[MAX_CONFIGSTRINGS];
198
199#ifdef LEGACY_PROTOCOL
200 qboolean compat;
201#endif
202} client_t;

207// MAX_CHALLENGES is made large to prevent a denial
208// of service attack that could cycle all of them
209// out before legitimate users connected
210#define MAX_CHALLENGES 2048
211// Allow a certain amount of challenges to have the same IP address
212// to make it a bit harder to DOS one single IP address from connecting
213// while not allowing a single ip to grab all challenge resources
214#define MAX_CHALLENGES_MULTI (MAX_CHALLENGES / 2)

26/*
27===============
28SV_SendConfigstring
29
30Creates and sends the server command necessary to update the CS index for the
31given client
32===============
33*/
34static void SV_SendConfigstring(client_t *client, int index)
35{
36 int maxChunkSize = MAX_STRING_CHARS - 24;
37 int len;
38
39 len = strlen(sv.configstrings[index]);
40
41 if( len >= maxChunkSize ) {
42 int sent = 0;
43 int remaining = len;
44 char *cmd;
45 char buf[MAX_STRING_CHARS];
46
47 while (remaining > 0 ) {
48 if ( sent == 0 ) {
49 cmd = "bcs0";
50 }
51 else if( remaining < maxChunkSize ) {
52 cmd = "bcs2";
53 }
54 else {
55 cmd = "bcs1";
56 }
57 Q_strncpyz( buf, &sv.configstrings[index][sent],
58 maxChunkSize );
59
60 SV_SendServerCommand( client, "%s %i \"%s\"\n", cmd,
61 index, buf );
62
63 sent += (maxChunkSize - 1);
64 remaining -= (maxChunkSize - 1);
65 }
66 } else {
67 // standard cs, just send it
68 SV_SendServerCommand( client, "cs %i \"%s\"\n", index,
69 sv.configstrings[index] );
70 }
71}

73/*
74===============
75SV_UpdateConfigstrings
76
77Called when a client goes from CS_PRIMED to CS_ACTIVE. Updates all
78Configstring indexes that have changed while the client was in CS_PRIMED
79===============
80*/
81void SV_UpdateConfigstrings(client_t *client)
82{
83 int index;
84
85 for( index = 0; index < MAX_CONFIGSTRINGS; index++ ) {
86 // if the CS hasn't changed since we went to CS_PRIMED, ignore
87 if(!client->csUpdated[index])
88 continue;
89
90 // do not always send server info to all clients
91 if ( index == CS_SERVERINFO && client->gentity &&
92 (client->gentity->r.svFlags & SVF_NOSERVERINFO) ) {
93 continue;
94 }
95 SV_SendConfigstring(client, index);
96 client->csUpdated[index] = qfalse;
97 }
98}

100/*
101===============
102SV_SetConfigstring
103
104===============
105*/
106void SV_SetConfigstring (int index, const char *val) {
107 int i;
108 client_t *client;
109
110 if ( index < 0 || index >= MAX_CONFIGSTRINGS ) {
111 Com_Error (ERR_DROP, "SV_SetConfigstring: bad index %i", index);
112 }
113
114 if ( !val ) {
115 val = "";
116 }
117
118 // don't bother broadcasting an update if no change
119 if ( !strcmp( val, sv.configstrings[ index ] ) ) {
120 return;
121 }
122
123 // change the string in sv
124 Z_Free( sv.configstrings[index] );
125 sv.configstrings[index] = CopyString( val );
126
127 // send it to all the clients if we aren't
128 // spawning a new server
129 if ( sv.state == SS_GAME || sv.restarting ) {
130
131 // send the data to all relevent clients
132 for (i = 0, client = svs.clients; i < sv_maxclients->integer ; i++, client++) {
133 if ( client->state < CS_ACTIVE ) {
134 if ( client->state == CS_PRIMED )
135 client->csUpdated[ index ] = qtrue;
136 continue;
137 }
138 // do not always send server info to all clients
139 if ( index == CS_SERVERINFO && client->gentity && (client->gentity->r.svFlags & SVF_NOSERVERINFO) ) {
140 continue;
141 }
142
143 SV_SendConfigstring(client, index);
144 }
145 }
146}

208/*
209================
210SV_CreateBaseline
211
212Entity baselines are used to compress non-delta messages
213to the clients -- only the fields that differ from the
214baseline will be transmitted
215================
216*/
217static void SV_CreateBaseline( void ) {
218 sharedEntity_t *svent;
219 int entnum;
220
221 for ( entnum = 1; entnum < sv.num_entities ; entnum++ ) {
222 svent = SV_GentityNum(entnum);
223 if (!svent->r.linked) {
224 continue;
225 }
226 svent->s.number = entnum;
227
228 //
229 // take current state as baseline
230 //
231 sv.svEntities[entnum].baseline = svent->s;
232 }
233}

256/*
257===============
258SV_Startup
259
260Called when a host starts a map when it wasn't running
261one before. Successive map or map_restart commands will
262NOT cause this to be called, unless the game is exited to
263the menu system first.
264===============
265*/
266static void SV_Startup( void ) {
267 if ( svs.initialized ) {
268 Com_Error( ERR_FATAL, "SV_Startup: svs.initialized" );
269 }
270 SV_BoundMaxClients( 1 );
271
272 svs.clients = Z_Malloc (sizeof(client_t) * sv_maxclients->integer );
273 if ( com_dedicated->integer ) {
274 svs.numSnapshotEntities = sv_maxclients->integer * PACKET_BACKUP * MAX_SNAPSHOT_ENTITIES;
275 } else {
276 // we don't need nearly as many when playing locally
277 svs.numSnapshotEntities = sv_maxclients->integer * 4 * MAX_SNAPSHOT_ENTITIES;
278 }
279 svs.initialized = qtrue;
280
281 // Don't respect sv_killserver unless a server is actually running
282 if ( sv_killserver->integer ) {
283 Cvar_Set( "sv_killserver", "0" );
284 }
285
286 Cvar_Set( "sv_running", "1" );
287
288 // Join the ipv6 multicast group now that a map is running so clients can scan for us on the local network.
289 NET_JoinMulticast6();
290}

293/*
294==================
295SV_ChangeMaxClients
296==================
297*/
298void SV_ChangeMaxClients( void ) {
299 int oldMaxClients;
300 int i;
301 client_t *oldClients;
302 int count;
303
304 // get the highest client number in use
305 count = 0;
306 for ( i = 0 ; i < sv_maxclients->integer ; i++ ) {
307 if ( svs.clients[i].state >= CS_CONNECTED ) {
308 if (i > count)
309 count = i;
310 }
311 }
312 count++;
313
314 oldMaxClients = sv_maxclients->integer;
315 // never go below the highest client number in use
316 SV_BoundMaxClients( count );
317 // if still the same
318 if ( sv_maxclients->integer == oldMaxClients ) {
319 return;
320 }
321
322 oldClients = Hunk_AllocateTempMemory( count * sizeof(client_t) );
323 // copy the clients to hunk memory
324 for ( i = 0 ; i < count ; i++ ) {
325 if ( svs.clients[i].state >= CS_CONNECTED ) {
326 oldClients[i] = svs.clients[i];
327 }
328 else {
329 Com_Memset(&oldClients[i], 0, sizeof(client_t));
330 }
331 }
332
333 // free old clients arrays
334 Z_Free( svs.clients );
335
336 // allocate new clients
337 svs.clients = Z_Malloc ( sv_maxclients->integer * sizeof(client_t) );
338 Com_Memset( svs.clients, 0, sv_maxclients->integer * sizeof(client_t) );
339
340 // copy the clients over
341 for ( i = 0 ; i < count ; i++ ) {
342 if ( oldClients[i].state >= CS_CONNECTED ) {
343 svs.clients[i] = oldClients[i];
344 }
345 }
346
347 // free the old clients on the hunk
348 Hunk_FreeTempMemory( oldClients );
349
350 // allocate new snapshot entities
351 if ( com_dedicated->integer ) {
352 svs.numSnapshotEntities = sv_maxclients->integer * PACKET_BACKUP * MAX_SNAPSHOT_ENTITIES;
353 } else {
354 // we don't need nearly as many when playing locally
355 svs.numSnapshotEntities = sv_maxclients->integer * 4 * MAX_SNAPSHOT_ENTITIES;
356 }
357}

365 int i;

393/*
394================
395SV_SpawnServer
396
397Change the server to a new map, taking all connected
398clients along with it.
399This is NOT called for map_restart
400================
401*/
402void SV_SpawnServer( char *server, qboolean killBots ) {
403 int i;
404 int checksum;
405 qboolean isBot;
406 char systemInfo[16384];
407 const char *p;
408
409 // shut down the existing game if it is running
410 SV_ShutdownGameProgs();
411
412 Com_Printf ("------ Server Initialization ------\n");
413 Com_Printf ("Server: %s\n",server);
414
415 // if not running a dedicated server CL_MapLoading will connect the client to the server
416 // also print some status stuff
417 CL_MapLoading();
418
419 // make sure all the client stuff is unloaded
420 CL_ShutdownAll(qfalse);
421
422 // clear the whole hunk because we're (re)loading the server
423 Hunk_Clear();
424
425 // clear collision map data
426 CM_ClearMap();
427
428 // init client structures and svs.numSnapshotEntities
429 if ( !Cvar_VariableValue("sv_running") ) {
430 SV_Startup();
431 } else {
432 // check for maxclients change
433 if ( sv_maxclients->modified ) {
434 SV_ChangeMaxClients();
435 }
436 }
437
438 // clear pak references
439 FS_ClearPakReferences(0);
440
441 // allocate the snapshot entities on the hunk
442 svs.snapshotEntities = Hunk_Alloc( sizeof(entityState_t)*svs.numSnapshotEntities, h_high );
443 svs.nextSnapshotEntities = 0;
444
445 // toggle the server bit so clients can detect that a
446 // server has changed
447 svs.snapFlagServerBit ^= SNAPFLAG_SERVERCOUNT;
448
449 // set nextmap to the same map, but it may be overriden
450 // by the game startup or another console command
451 Cvar_Set( "nextmap", "map_restart 0");
452// Cvar_Set( "nextmap", va("map %s", server) );
453
454 for (i=0 ; i<sv_maxclients->integer ; i++) {
455 // save when the server started for each client already connected
456 if (svs.clients[i].state >= CS_CONNECTED) {
457 svs.clients[i].oldServerTime = sv.time;
458 }
459 }
460
461 // wipe the entire per-level structure
462 SV_ClearServer();
463 for ( i = 0 ; i < MAX_CONFIGSTRINGS ; i++ ) {
464 sv.configstrings[i] = CopyString("");
465 }
466
467 // make sure we are not paused
468 Cvar_Set("cl_paused", "0");
469
470 // get a new checksum feed and restart the file system
471 sv.checksumFeed = ( ((int) rand() << 16) ^ rand() ) ^ Com_Milliseconds();
472 FS_Restart( sv.checksumFeed );
473
474 CM_LoadMap( va("maps/%s.bsp", server), qfalse, &checksum );
475
476 // set serverinfo visible name
477 Cvar_Set( "mapname", server );
478
479 Cvar_Set( "sv_mapChecksum", va("%i",checksum) );
480
481 // serverid should be different each time
482 sv.serverId = com_frameTime;
483 sv.restartedServerId = sv.serverId; // I suppose the init here is just to be safe
484 sv.checksumFeedServerId = sv.serverId;
485 Cvar_Set( "sv_serverid", va("%i", sv.serverId ) );
486
487 // clear physics interaction links
488 SV_ClearWorld ();
489
490 // media configstring setting should be done during
491 // the loading stage, so connected clients don't have
492 // to load during actual gameplay
493 sv.state = SS_LOADING;
494
495 // load and spawn all other entities
496 SV_InitGameProgs();
497
498 // don't allow a map_restart if game is modified
499 sv_gametype->modified = qfalse;
500
501 // run a few frames to allow everything to settle
502 for (i = 0;i < 3; i++)
503 {
504 VM_Call (gvm, GAME_RUN_FRAME, sv.time);
505 SV_BotFrame (sv.time);
506 sv.time += 100;
507 svs.time += 100;
508 }
509
510 // create a baseline for more efficient communications
511 SV_CreateBaseline ();
512
513 for (i=0 ; i<sv_maxclients->integer ; i++) {
514 // send the new gamestate to all connected clients
515 if (svs.clients[i].state >= CS_CONNECTED) {
516 char *denied;
517
518 if ( svs.clients[i].netchan.remoteAddress.type == NA_BOT ) {
519 if ( killBots ) {
520 SV_DropClient( &svs.clients[i], "" );
521 continue;
522 }
523 isBot = qtrue;
524 }
525 else {
526 isBot = qfalse;
527 }
528
529 // connect the client again
530 denied = VM_ExplicitArgPtr( gvm, VM_Call( gvm, GAME_CLIENT_CONNECT, i, qfalse, isBot ) ); // firstTime = qfalse
531 if ( denied ) {
532 // this generally shouldn't happen, because the client
533 // was connected before the level change
534 SV_DropClient( &svs.clients[i], denied );
535 } else {
536 if( !isBot ) {
537 // when we get the next packet from a connected client,
538 // the new gamestate will be sent
539 svs.clients[i].state = CS_CONNECTED;
540 }
541 else {
542 client_t *client;
543 sharedEntity_t *ent;
544
545 client = &svs.clients[i];
546 client->state = CS_ACTIVE;
547 ent = SV_GentityNum( i );
548 ent->s.number = i;
549 client->gentity = ent;
550
551 client->deltaMessage = -1;
552 client->lastSnapshotTime = 0; // generate a snapshot immediately
553
554 VM_Call( gvm, GAME_CLIENT_BEGIN, i );
555 }
556 }
557 }
558 }
559
560 // run another frame to allow things to look at all the players
561 VM_Call (gvm, GAME_RUN_FRAME, sv.time);
562 SV_BotFrame (sv.time);
563 sv.time += 100;
564 svs.time += 100;
565
566 if ( sv_pure->integer ) {
567 // the server sends these to the clients so they will only
568 // load pk3s also loaded at the server
569 p = FS_LoadedPakChecksums();
570 Cvar_Set( "sv_paks", p );
571 if (strlen(p) == 0) {
572 Com_Printf( "WARNING: sv_pure set but no PK3 files loaded\n" );
573 }
574 p = FS_LoadedPakNames();
575 Cvar_Set( "sv_pakNames", p );
576
577 // if a dedicated pure server we need to touch the cgame because it could be in a
578 // seperate pk3 file and the client will need to load the latest cgame.qvm
579 if ( com_dedicated->integer ) {
580 SV_TouchCGame();
581 }
582 }
583 else {
584 Cvar_Set( "sv_paks", "" );
585 Cvar_Set( "sv_pakNames", "" );
586 }
587 // the server sends these to the clients so they can figure
588 // out which pk3s should be auto-downloaded
589 p = FS_ReferencedPakChecksums();
590 Cvar_Set( "sv_referencedPaks", p );
591 p = FS_ReferencedPakNames();
592 Cvar_Set( "sv_referencedPakNames", p );
593
594 // save systeminfo and serverinfo strings
595 Q_strncpyz( systemInfo, Cvar_InfoString_Big( CVAR_SYSTEMINFO ), sizeof( systemInfo ) );
596 cvar_modifiedFlags &= ~CVAR_SYSTEMINFO;
597 SV_SetConfigstring( CS_SYSTEMINFO, systemInfo );
598
599 SV_SetConfigstring( CS_SERVERINFO, Cvar_InfoString( CVAR_SERVERINFO ) );
600 cvar_modifiedFlags &= ~CVAR_SERVERINFO;
601
602 // any media configstring setting now should issue a warning
603 // and any configstring changes should be reliably transmitted
604 // to all clients
605 sv.state = SS_GAME;
606
607 // send a heartbeat now so the master will get up to date info
608 SV_Heartbeat_f();
609
610 Hunk_SetMark();
611
612#ifndef DEDICATED
613 if ( com_dedicated->integer ) {
614 // restart renderer in order to show console for dedicated servers
615 // launched through the regular binary
616 CL_StartHunkUsers( qtrue );
617 }
618#endif
619
620 Com_Printf ("-----------------------------------\n");
621}

706/*
707==================
708SV_FinalMessage
709
710Used by SV_Shutdown to send a final message to all
711connected clients before the server goes down. The messages are sent immediately,
712not just stuck on the outgoing message list, because the server is going
713to totally exit after returning from this function.
714==================
715*/
716void SV_FinalMessage( char *message ) {
717 int i, j;
718 client_t *cl;
719
720 // send it twice, ignoring rate
721 for ( j = 0 ; j < 2 ; j++ ) {
722 for (i=0, cl = svs.clients ; i < sv_maxclients->integer ; i++, cl++) {
723 if (cl->state >= CS_CONNECTED) {
724 // don't send a disconnect to a local client
725 if ( cl->netchan.remoteAddress.type != NA_LOOPBACK ) {
726 SV_SendServerCommand( cl, "print \"%s\n\"\n", message );
727 SV_SendServerCommand( cl, "disconnect \"%s\"", message );
728 }
729 // force a snapshot to be sent
730 cl->lastSnapshotTime = 0;
731 SV_SendClientSnapshot( cl );
732 }

742Called when each game quits,
743before Sys_Quit or Sys_Error

746void SV_Shutdown( char *finalmsg ) {
747 if ( !com_sv_running || !com_sv_running->integer ) {
748 return;
749 }
750
751 Com_Printf( "----- Server Shutdown (%s) -----\n", finalmsg );
752
753 NET_LeaveMulticast6();
754
755 if ( svs.clients && !com_errorEntered ) {
756 SV_FinalMessage( finalmsg );
757 }
758
759 SV_RemoveOperatorCommands();
760 SV_MasterShutdown();
761 SV_ShutdownGameProgs();
762
763 // free current level
764 SV_ClearServer();
765
766 // free server static data
767 if(svs.clients)
768 {
769 int index;
770
771 for(index = 0; index < sv_maxclients->integer; index++)
772 SV_FreeClient(&svs.clients[index]);
773
774 Z_Free(svs.clients);
775 }
776 Com_Memset( &svs, 0, sizeof( svs ) );
777
778 Cvar_Set( "sv_running", "0" );
779 Cvar_Set("ui_singlePlayerActive", "0");
780
781 Com_Printf( "---------------------------\n" );
782
783 // disconnect any local clients
784 if( sv_killserver->integer != 2 )
785 CL_Disconnect( qfalse );
786}

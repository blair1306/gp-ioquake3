28/*
29=================
30SV_GetChallenge
31
32A "getchallenge" OOB command has been received
33Returns a challenge number that can be used
34in a subsequent connectResponse command.
35We do this to prevent denial of service attacks that
36flood the server with invalid connection IPs. With a
37challenge, they must give a valid IP address.
38
39If we are authorizing, a challenge request will cause a packet
40to be sent to the authorize server.
41
42When an authorizeip is returned, a challenge response will be
43sent to that ip.
44
45ioquake3: we added a possibility for clients to add a challenge
46to their packets, to make it more difficult for malicious servers
47to hi-jack client connections.
48Also, the auth stuff is completely disabled for com_standalone games
49as well as IPv6 connections, since there is no way to use the
50v4-only auth server for these new types of connections.
51=================
52*/
53void SV_GetChallenge(netadr_t from)
54{
55 int i;
56 int oldest;
57 int oldestTime;
58 int oldestClientTime;
59 int clientChallenge;
60 challenge_t *challenge;
61 qboolean wasfound = qfalse;
62 char *gameName;
63 qboolean gameMismatch;
64
65 // ignore if we are in single player
66 if ( Cvar_VariableValue( "g_gametype" ) == GT_SINGLE_PLAYER || Cvar_VariableValue("ui_singlePlayerActive")) {
67 return;
68 }
69
70 // Prevent using getchallenge as an amplifier
71 if ( SVC_RateLimitAddress( from, 10, 1000 ) ) {
72 Com_DPrintf( "SV_GetChallenge: rate limit from %s exceeded, dropping request\n",
73 NET_AdrToString( from ) );
74 return;
75 }
76
77 // Allow getchallenge to be DoSed relatively easily, but prevent
78 // excess outbound bandwidth usage when being flooded inbound
79 if ( SVC_RateLimit( &outboundLeakyBucket, 10, 100 ) ) {
80 Com_DPrintf( "SV_GetChallenge: rate limit exceeded, dropping request\n" );
81 return;
82 }
83
84 gameName = Cmd_Argv(2);
85
86#ifdef LEGACY_PROTOCOL
87 // gamename is optional for legacy protocol
88 if (com_legacyprotocol->integer && !*gameName)
89 gameMismatch = qfalse;
90 else
91#endif
92 gameMismatch = !*gameName || strcmp(gameName, com_gamename->string) != 0;
93
94 // reject client if the gamename string sent by the client doesn't match ours
95 if (gameMismatch)
96 {
97 NET_OutOfBandPrint(NS_SERVER, from, "print\nGame mismatch: This is a %s server\n",
98 com_gamename->string);
99 return;
100 }
101
102 oldest = 0;
103 oldestClientTime = oldestTime = 0x7fffffff;
104
105 // see if we already have a challenge for this ip
106 challenge = &svs.challenges[0];
107 clientChallenge = atoi(Cmd_Argv(1));
108
109 for(i = 0 ; i < MAX_CHALLENGES ; i++, challenge++)
110 {
111 if(!challenge->connected && NET_CompareAdr(from, challenge->adr))
112 {
113 wasfound = qtrue;
114
115 if(challenge->time < oldestClientTime)
116 oldestClientTime = challenge->time;
117 }
118
119 if(wasfound && i >= MAX_CHALLENGES_MULTI)
120 {
121 i = MAX_CHALLENGES;
122 break;
123 }
124
125 if(challenge->time < oldestTime)
126 {
127 oldestTime = challenge->time;
128 oldest = i;
129 }
130 }
131
132 if (i == MAX_CHALLENGES)
133 {
134 // this is the first time this client has asked for a challenge
135 challenge = &svs.challenges[oldest];
136 challenge->clientChallenge = clientChallenge;
137 challenge->adr = from;
138 challenge->firstTime = svs.time;
139 challenge->connected = qfalse;
140 }
141
142 // always generate a new challenge number, so the client cannot circumvent sv_maxping
143 challenge->challenge = ( (rand() << 16) ^ rand() ) ^ svs.time;
144 challenge->wasrefused = qfalse;
145 challenge->time = svs.time;
146
147#ifndef STANDALONE
148 // Drop the authorize stuff if this client is coming in via v6 as the auth server does not support ipv6.
149 // Drop also for addresses coming in on local LAN and for stand-alone games independent from id's assets.
150 if(challenge->adr.type == NA_IP && !com_standalone->integer && !Sys_IsLANAddress(from))
151 {
152 // look up the authorize server's IP
153 if (svs.authorizeAddress.type == NA_BAD)
154 {
155 Com_Printf( "Resolving %s\n", AUTHORIZE_SERVER_NAME );
156
157 if (NET_StringToAdr(AUTHORIZE_SERVER_NAME, &svs.authorizeAddress, NA_IP))
158 {
159 svs.authorizeAddress.port = BigShort( PORT_AUTHORIZE );
160 Com_Printf( "%s resolved to %i.%i.%i.%i:%i\n", AUTHORIZE_SERVER_NAME,
161 svs.authorizeAddress.ip[0], svs.authorizeAddress.ip[1],
162 svs.authorizeAddress.ip[2], svs.authorizeAddress.ip[3],
163 BigShort( svs.authorizeAddress.port ) );
164 }
165 }
166
167 // we couldn't contact the auth server, let them in.
168 if(svs.authorizeAddress.type == NA_BAD)
169 Com_Printf("Couldn't resolve auth server address\n");
170
171 // if they have been challenging for a long time and we
172 // haven't heard anything from the authorize server, go ahead and
173 // let them in, assuming the id server is down
174 else if(svs.time - oldestClientTime > AUTHORIZE_TIMEOUT)
175 Com_DPrintf( "authorize server timed out\n" );
176 else
177 {
178 // otherwise send their ip to the authorize server
179 cvar_t *fs;
180 char game[1024];
181
182 Com_DPrintf( "sending getIpAuthorize for %s\n", NET_AdrToString( from ));
183
184 strcpy(game, BASEGAME);
185 fs = Cvar_Get ("fs_game", "", CVAR_INIT|CVAR_SYSTEMINFO );
186 if (fs && fs->string[0] != 0) {
187 strcpy(game, fs->string);
188 }
189
190 // the 0 is for backwards compatibility with obsolete sv_allowanonymous flags
191 // getIpAuthorize <challenge> <IP> <game> 0 <auth-flag>
192 NET_OutOfBandPrint( NS_SERVER, svs.authorizeAddress,
193 "getIpAuthorize %i %i.%i.%i.%i %s 0 %s", challenge->challenge,
194 from.ip[0], from.ip[1], from.ip[2], from.ip[3], game, sv_strictAuth->string );
195
196 return;
197 }
198 }
199#endif
200
201 challenge->pingTime = svs.time;
202 NET_OutOfBandPrint(NS_SERVER, challenge->adr, "challengeResponse %d %d %d",
203 challenge->challenge, clientChallenge, com_protocol->integer);
204}

207/*
208====================
209SV_AuthorizeIpPacket
210
211A packet has been returned from the authorize server.
212If we have a challenge adr for that ip, send the
213challengeResponse to it
214====================
215*/
216void SV_AuthorizeIpPacket( netadr_t from ) {
217 int challenge;
218 int i;
219 char *s;
220 char *r;
221 challenge_t *challengeptr;
222
223 if ( !NET_CompareBaseAdr( from, svs.authorizeAddress ) ) {
224 Com_Printf( "SV_AuthorizeIpPacket: not from authorize server\n" );
225 return;
226 }
227
228 challenge = atoi( Cmd_Argv( 1 ) );
229
230 for (i = 0 ; i < MAX_CHALLENGES ; i++) {
231 if ( svs.challenges[i].challenge == challenge ) {
232 break;
233 }
234 }
235 if ( i == MAX_CHALLENGES ) {
236 Com_Printf( "SV_AuthorizeIpPacket: challenge not found\n" );
237 return;
238 }
239
240 challengeptr = &svs.challenges[i];
241
242 // send a packet back to the original client
243 challengeptr->pingTime = svs.time;
244 s = Cmd_Argv( 2 );
245 r = Cmd_Argv( 3 ); // reason
246
247 if ( !Q_stricmp( s, "demo" ) ) {
248 // they are a demo client trying to connect to a real server
249 NET_OutOfBandPrint( NS_SERVER, challengeptr->adr, "print\nServer is not a demo server\n" );
250 // clear the challenge record so it won't timeout and let them through
251 Com_Memset( challengeptr, 0, sizeof( *challengeptr ) );
252 return;
253 }
254 if ( !Q_stricmp( s, "accept" ) ) {
255 NET_OutOfBandPrint(NS_SERVER, challengeptr->adr,
256 "challengeResponse %d %d %d", challengeptr->challenge, challengeptr->clientChallenge, com_protocol->integer);
257 return;
258 }
259 if ( !Q_stricmp( s, "unknown" ) ) {
260 if (!r) {
261 NET_OutOfBandPrint( NS_SERVER, challengeptr->adr, "print\nAwaiting CD key authorization\n" );
262 } else {
263 NET_OutOfBandPrint( NS_SERVER, challengeptr->adr, "print\n%s\n", r);
264 }
265 // clear the challenge record so it won't timeout and let them through
266 Com_Memset( challengeptr, 0, sizeof( *challengeptr ) );
267 return;
268 }
269
270 // authorization failed
271 if (!r) {
272 NET_OutOfBandPrint( NS_SERVER, challengeptr->adr, "print\nSomeone is using this CD Key\n" );
273 } else {
274 NET_OutOfBandPrint( NS_SERVER, challengeptr->adr, "print\n%s\n", r );
275 }
276
277 // clear the challenge record so it won't timeout and let them through
278 Com_Memset( challengeptr, 0, sizeof(*challengeptr) );
279}

282/*
283==================
284SV_IsBanned
285
286Check whether a certain address is banned
287==================
288*/
289
290static qboolean SV_IsBanned(netadr_t *from, qboolean isexception)
291{
292 int index;
293 serverBan_t *curban;
294
295 if(!isexception)
296 {
297 // If this is a query for a ban, first check whether the client is excepted
298 if(SV_IsBanned(from, qtrue))
299 return qfalse;
300 }
301
302 for(index = 0; index < serverBansCount; index++)
303 {
304 curban = &serverBans[index];
305
306 if(curban->isexception == isexception)
307 {
308 if(NET_CompareBaseAdrMask(curban->ip, *from, curban->subnet))
309 return qtrue;
310 }
311 }
312
313 return qfalse;
314}

316/*
317==================
318SV_DirectConnect
319
320A "connect" OOB command has been received
321==================
322*/
323
324void SV_DirectConnect( netadr_t from ) {
325 char userinfo[MAX_INFO_STRING];
326 int i;
327 client_t *cl, *newcl;
328 client_t temp;
329 sharedEntity_t *ent;
330 int clientNum;
331 int version;
332 int qport;
333 int challenge;
334 char *password;
335 int startIndex;
336 intptr_t denied;
337 int count;
338 char *ip;
339#ifdef LEGACY_PROTOCOL
340 qboolean compat = qfalse;
341#endif
342
343 Com_DPrintf ("SVC_DirectConnect ()\n");
344
345 // Check whether this client is banned.
346 if(SV_IsBanned(&from, qfalse))
347 {
348 NET_OutOfBandPrint(NS_SERVER, from, "print\nYou are banned from this server.\n");
349 return;
350 }
351
352 Q_strncpyz( userinfo, Cmd_Argv(1), sizeof(userinfo) );
353
354 version = atoi(Info_ValueForKey(userinfo, "protocol"));
355
356#ifdef LEGACY_PROTOCOL
357 if(version > 0 && com_legacyprotocol->integer == version)
358 compat = qtrue;
359 else
360#endif
361 {
362 if(version != com_protocol->integer)
363 {
364 NET_OutOfBandPrint(NS_SERVER, from, "print\nServer uses protocol version %i "
365 "(yours is %i).\n", com_protocol->integer, version);
366 Com_DPrintf(" rejected connect from version %i\n", version);
367 return;
368 }
369 }
370
371 challenge = atoi( Info_ValueForKey( userinfo, "challenge" ) );
372 qport = atoi( Info_ValueForKey( userinfo, "qport" ) );
373
374 // quick reject
375 for (i=0,cl=svs.clients ; i < sv_maxclients->integer ; i++,cl++) {
376 if ( cl->state == CS_FREE ) {
377 continue;
378 }
379 if ( NET_CompareBaseAdr( from, cl->netchan.remoteAddress )
380 && ( cl->netchan.qport == qport
381 || from.port == cl->netchan.remoteAddress.port ) ) {
382 if (( svs.time - cl->lastConnectTime)
383 < (sv_reconnectlimit->integer * 1000)) {
384 Com_DPrintf ("%s:reconnect rejected : too soon\n", NET_AdrToString (from));
385 return;
386 }
387 break;
388 }
389 }
390
391 // don't let "ip" overflow userinfo string
392 if ( NET_IsLocalAddress (from) )
393 ip = "localhost";
394 else
395 ip = (char *)NET_AdrToString( from );
396 if( ( strlen( ip ) + strlen( userinfo ) + 4 ) >= MAX_INFO_STRING ) {
397 NET_OutOfBandPrint( NS_SERVER, from,
398 "print\nUserinfo string length exceeded. "
399 "Try removing setu cvars from your config.\n" );
400 return;
401 }
402 Info_SetValueForKey( userinfo, "ip", ip );
403
404 // see if the challenge is valid (LAN clients don't need to challenge)
405 if (!NET_IsLocalAddress(from))
406 {
407 int ping;
408 challenge_t *challengeptr;
409
410 for (i=0; i<MAX_CHALLENGES; i++)
411 {
412 if (NET_CompareAdr(from, svs.challenges[i].adr))
413 {
414 if(challenge == svs.challenges[i].challenge)
415 break;
416 }
417 }
418
419 if (i == MAX_CHALLENGES)
420 {
421 NET_OutOfBandPrint( NS_SERVER, from, "print\nNo or bad challenge for your address.\n" );
422 return;
423 }
424
425 challengeptr = &svs.challenges[i];
426
427 if(challengeptr->wasrefused)
428 {
429 // Return silently, so that error messages written by the server keep being displayed.
430 return;
431 }
432
433 ping = svs.time - challengeptr->pingTime;
434
435 // never reject a LAN client based on ping
436 if ( !Sys_IsLANAddress( from ) ) {
437 if ( sv_minPing->value && ping < sv_minPing->value ) {
438 NET_OutOfBandPrint( NS_SERVER, from, "print\nServer is for high pings only\n" );
439 Com_DPrintf ("Client %i rejected on a too low ping\n", i);
440 challengeptr->wasrefused = qtrue;
441 return;
442 }
443 if ( sv_maxPing->value && ping > sv_maxPing->value ) {
444 NET_OutOfBandPrint( NS_SERVER, from, "print\nServer is for low pings only\n" );
445 Com_DPrintf ("Client %i rejected on a too high ping\n", i);
446 challengeptr->wasrefused = qtrue;
447 return;
448 }
449 }
450
451 Com_Printf("Client %i connecting with %i challenge ping\n", i, ping);
452 challengeptr->connected = qtrue;
453 }
454
455 newcl = &temp;
456 Com_Memset (newcl, 0, sizeof(client_t));
457
458 // if there is already a slot for this ip, reuse it
459 for (i=0,cl=svs.clients ; i < sv_maxclients->integer ; i++,cl++) {
460 if ( cl->state == CS_FREE ) {
461 continue;
462 }
463 if ( NET_CompareBaseAdr( from, cl->netchan.remoteAddress )
464 && ( cl->netchan.qport == qport
465 || from.port == cl->netchan.remoteAddress.port ) ) {
466 Com_Printf ("%s:reconnect\n", NET_AdrToString (from));
467 newcl = cl;
468
469 // this doesn't work because it nukes the players userinfo
470
471// // disconnect the client from the game first so any flags the
472// // player might have are dropped
473// VM_Call( gvm, GAME_CLIENT_DISCONNECT, newcl - svs.clients );
474 //
475 goto gotnewcl;
476 }
477 }
478
479 // find a client slot
480 // if "sv_privateClients" is set > 0, then that number
481 // of client slots will be reserved for connections that
482 // have "password" set to the value of "sv_privatePassword"
483 // Info requests will report the maxclients as if the private
484 // slots didn't exist, to prevent people from trying to connect
485 // to a full server.
486 // This is to allow us to reserve a couple slots here on our
487 // servers so we can play without having to kick people.
488
489 // check for privateClient password
490 password = Info_ValueForKey( userinfo, "password" );
491 if ( *password && !strcmp( password, sv_privatePassword->string ) ) {
492 startIndex = 0;
493 } else {
494 // skip past the reserved slots
495 startIndex = sv_privateClients->integer;
496 }
497
498 newcl = NULL;
499 for ( i = startIndex; i < sv_maxclients->integer ; i++ ) {
500 cl = &svs.clients[i];
501 if (cl->state == CS_FREE) {
502 newcl = cl;
503 break;
504 }
505 }
506
507 if ( !newcl ) {
508 if ( NET_IsLocalAddress( from ) ) {
509 count = 0;
510 for ( i = startIndex; i < sv_maxclients->integer ; i++ ) {
511 cl = &svs.clients[i];
512 if (cl->netchan.remoteAddress.type == NA_BOT) {
513 count++;
514 }
515 }
516 // if they're all bots
517 if (count >= sv_maxclients->integer - startIndex) {
518 SV_DropClient(&svs.clients[sv_maxclients->integer - 1], "only bots on server");
519 newcl = &svs.clients[sv_maxclients->integer - 1];
520 }
521 else {
522 Com_Error( ERR_FATAL, "server is full on local connect" );
523 return;
524 }
525 }
526 else {
527 NET_OutOfBandPrint( NS_SERVER, from, "print\nServer is full.\n" );
528 Com_DPrintf ("Rejected a connection.\n");
529 return;
530 }
531 }
532
533 // we got a newcl, so reset the reliableSequence and reliableAcknowledge
534 cl->reliableAcknowledge = 0;
535 cl->reliableSequence = 0;
536
537gotnewcl:
538 // build a new connection
539 // accept the new client
540 // this is the only place a client_t is ever initialized
541 *newcl = temp;
542 clientNum = newcl - svs.clients;
543 ent = SV_GentityNum( clientNum );
544 newcl->gentity = ent;
545
546 // save the challenge
547 newcl->challenge = challenge;
548
549 // save the address
550#ifdef LEGACY_PROTOCOL
551 newcl->compat = compat;
552 Netchan_Setup(NS_SERVER, &newcl->netchan, from, qport, challenge, compat);
553#else
554 Netchan_Setup(NS_SERVER, &newcl->netchan, from, qport, challenge, qfalse);
555#endif
556 // init the netchan queue
557 newcl->netchan_end_queue = &newcl->netchan_start_queue;
558
559 // save the userinfo
560 Q_strncpyz( newcl->userinfo, userinfo, sizeof(newcl->userinfo) );
561
562 // get the game a chance to reject this connection or modify the userinfo
563 denied = VM_Call( gvm, GAME_CLIENT_CONNECT, clientNum, qtrue, qfalse ); // firstTime = qtrue
564 if ( denied ) {
565 // we can't just use VM_ArgPtr, because that is only valid inside a VM_Call
566 char *str = VM_ExplicitArgPtr( gvm, denied );
567
568 NET_OutOfBandPrint( NS_SERVER, from, "print\n%s\n", str );
569 Com_DPrintf ("Game rejected a connection: %s.\n", str);
570 return;
571 }
572
573 SV_UserinfoChanged( newcl );
574
575 // send the connect packet to the client
576 NET_OutOfBandPrint(NS_SERVER, from, "connectResponse %d", challenge);
577
578 Com_DPrintf( "Going from CS_FREE to CS_CONNECTED for %s\n", newcl->name );
579
580 newcl->state = CS_CONNECTED;
581 newcl->lastSnapshotTime = 0;
582 newcl->lastPacketTime = svs.time;
583 newcl->lastConnectTime = svs.time;
584
585 // when we receive the first packet from the client, we will
586 // notice that it is from a different serverid and that the
587 // gamestate message was not just sent, forcing a retransmit
588 newcl->gamestateMessageNum = -1;
589
590 // if this was the first client on the server, or the last client
591 // the server can hold, send a heartbeat to the master.
592 count = 0;
593 for (i=0,cl=svs.clients ; i < sv_maxclients->integer ; i++,cl++) {
594 if ( svs.clients[i].state >= CS_CONNECTED ) {
595 count++;
596 }
597 }
598 if ( count == 1 || count == sv_maxclients->integer ) {
599 SV_Heartbeat_f();
600 }
601}

629/*
630=====================
631SV_DropClient
632
633Called when the player is totally leaving the server, either willingly
634or unwillingly. This is NOT called if the entire server is quiting
635or crashing -- SV_FinalMessage() will handle that
636=====================
637*/
638void SV_DropClient( client_t *drop, const char *reason ) {
639 int i;
640 challenge_t *challenge;
641 const qboolean isBot = drop->netchan.remoteAddress.type == NA_BOT;
642
643 if ( drop->state == CS_ZOMBIE ) {
644 return; // already dropped
645 }
646
647 if ( !isBot ) {
648 // see if we already have a challenge for this ip
649 challenge = &svs.challenges[0];
650
651 for (i = 0 ; i < MAX_CHALLENGES ; i++, challenge++)
652 {
653 if(NET_CompareAdr(drop->netchan.remoteAddress, challenge->adr))
654 {
655 Com_Memset(challenge, 0, sizeof(*challenge));
656 break;
657 }
658 }
659 }
660
661 // Free all allocated data on the client structure
662 SV_FreeClient(drop);
663
664 // tell everyone why they got dropped
665 SV_SendServerCommand( NULL, "print \"%s" S_COLOR_WHITE " %s\n\"", drop->name, reason );
666
667 // call the prog function for removing a client
668 // this will remove the body, among other things
669 VM_Call( gvm, GAME_CLIENT_DISCONNECT, drop - svs.clients );
670
671 // add the disconnect command
672 SV_SendServerCommand( drop, "disconnect \"%s\"", reason);
673
674 if ( isBot ) {
675 SV_BotFreeClient( drop - svs.clients );
676 }
677
678 // nuke user info
679 SV_SetUserinfo( drop - svs.clients, "" );
680
681 if ( isBot ) {
682 // bots shouldn't go zombie, as there's no real net connection.
683 drop->state = CS_FREE;
684 } else {
685 Com_DPrintf( "Going to CS_ZOMBIE for %s\n", drop->name );
686 drop->state = CS_ZOMBIE; // become free in a few seconds
687 }
688
689 // if this was the last client on the server, send a heartbeat
690 // to the master so it is known the server is empty
691 // send a heartbeat now so the master will get up to date info
692 // if there is already a slot for this ip, reuse it
693 for (i=0 ; i < sv_maxclients->integer ; i++ ) {
694 if ( svs.clients[i].state >= CS_CONNECTED ) {
695 break;
696 }
697 }
698 if ( i == sv_maxclients->integer ) {
699 SV_Heartbeat_f();
700 }
701}

703/*
704================
705SV_SendClientGameState
706
707Sends the first message from the server to a connected client.
708This will be sent on the initial connection and upon each new map load.
709
710It will be resent if the client acknowledges a later message but has
711the wrong gamestate.
712================
713*/
714static void SV_SendClientGameState( client_t *client ) {
715 int start;
716 entityState_t *base, nullstate;
717 msg_t msg;
718 byte msgBuffer[MAX_MSGLEN];
719
720 Com_DPrintf ("SV_SendClientGameState() for %s\n", client->name);
721 Com_DPrintf( "Going from CS_CONNECTED to CS_PRIMED for %s\n", client->name );
722 client->state = CS_PRIMED;
723 client->pureAuthentic = 0;
724 client->gotCP = qfalse;
725
726 // when we receive the first packet from the client, we will
727 // notice that it is from a different serverid and that the
728 // gamestate message was not just sent, forcing a retransmit
729 client->gamestateMessageNum = client->netchan.outgoingSequence;
730
731 MSG_Init( &msg, msgBuffer, sizeof( msgBuffer ) );
732
733 // NOTE, MRE: all server->client messages now acknowledge
734 // let the client know which reliable clientCommands we have received
735 MSG_WriteLong( &msg, client->lastClientCommand );
736
737 // send any server commands waiting to be sent first.
738 // we have to do this cause we send the client->reliableSequence
739 // with a gamestate and it sets the clc.serverCommandSequence at
740 // the client side
741 SV_UpdateServerCommandsToClient( client, &msg );
742
743 // send the gamestate
744 MSG_WriteByte( &msg, svc_gamestate );
745 MSG_WriteLong( &msg, client->reliableSequence );
746
747 // write the configstrings
748 for ( start = 0 ; start < MAX_CONFIGSTRINGS ; start++ ) {
749 if (sv.configstrings[start][0]) {
750 MSG_WriteByte( &msg, svc_configstring );
751 MSG_WriteShort( &msg, start );
752 MSG_WriteBigString( &msg, sv.configstrings[start] );
753 }
754 }
755
756 // write the baselines
757 Com_Memset( &nullstate, 0, sizeof( nullstate ) );
758 for ( start = 0 ; start < MAX_GENTITIES; start++ ) {
759 base = &sv.svEntities[start].baseline;
760 if ( !base->number ) {
761 continue;
762 }
763 MSG_WriteByte( &msg, svc_baseline );
764 MSG_WriteDeltaEntity( &msg, &nullstate, base, qtrue );
765 }
766
767 MSG_WriteByte( &msg, svc_EOF );
768
769 MSG_WriteLong( &msg, client - svs.clients);
770
771 // write the checksum feed
772 MSG_WriteLong( &msg, sv.checksumFeed);
773
774 // deliver this to the client
775 SV_SendMessageToClient( &msg, client );
776}

925/*
926==================
927SV_WriteDownloadToClient
928
929Check to see if the client wants a file, open it if needed and start pumping the client
930Fill up msg with data, return number of download blocks added
931==================
932*/
933int SV_WriteDownloadToClient(client_t *cl, msg_t *msg)
934{
935 int curindex;
936 int unreferenced = 1;
937 char errorMessage[1024];
938 char pakbuf[MAX_QPATH], *pakptr;
939 int numRefPaks;
940
941 if (!*cl->downloadName)
942 return 0; // Nothing being downloaded
943
944 if(!cl->download)
945 {
946 qboolean idPack = qfalse;
947 #ifndef STANDALONE
948 qboolean missionPack = qfalse;
949 #endif
950
951 // Chop off filename extension.
952 Com_sprintf(pakbuf, sizeof(pakbuf), "%s", cl->downloadName);
953 pakptr = strrchr(pakbuf, '.');
954
955 if(pakptr)
956 {
957 *pakptr = '\0';
958
959 // Check for pk3 filename extension
960 if(!Q_stricmp(pakptr + 1, "pk3"))
961 {
962 const char *referencedPaks = FS_ReferencedPakNames();
963
964 // Check whether the file appears in the list of referenced
965 // paks to prevent downloading of arbitrary files.
966 Cmd_TokenizeStringIgnoreQuotes(referencedPaks);
967 numRefPaks = Cmd_Argc();
968
969 for(curindex = 0; curindex < numRefPaks; curindex++)
970 {
971 if(!FS_FilenameCompare(Cmd_Argv(curindex), pakbuf))
972 {
973 unreferenced = 0;
974
975 // now that we know the file is referenced,
976 // check whether it's legal to download it.
977#ifndef STANDALONE
978 missionPack = FS_idPak(pakbuf, BASETA, NUM_TA_PAKS);
979 idPack = missionPack;
980#endif
981 idPack = idPack || FS_idPak(pakbuf, BASEGAME, NUM_ID_PAKS);
982
983 break;
984 }
985 }
986 }
987 }
988
989 cl->download = 0;
990
991 // We open the file here
992 if ( !(sv_allowDownload->integer & DLF_ENABLE) ||
993 (sv_allowDownload->integer & DLF_NO_UDP) ||
994 idPack || unreferenced ||
995 ( cl->downloadSize = FS_SV_FOpenFileRead( cl->downloadName, &cl->download ) ) < 0 ) {
996 // cannot auto-download file
997 if(unreferenced)
998 {
999 Com_Printf("clientDownload: %d : \"%s\" is not referenced and cannot be downloaded.\n", (int) (cl - svs.clients), cl->downloadName);
1000 Com_sprintf(errorMessage, sizeof(errorMessage), "File \"%s\" is not referenced and cannot be downloaded.", cl->downloadName);
1001 }
1002 else if (idPack) {
1003 Com_Printf("clientDownload: %d : \"%s\" cannot download id pk3 files\n", (int) (cl - svs.clients), cl->downloadName);
1004#ifndef STANDALONE
1005 if(missionPack)
1006 {
1007 Com_sprintf(errorMessage, sizeof(errorMessage), "Cannot autodownload Team Arena file \"%s\"\n"
1008 "The Team Arena mission pack can be found in your local game store.", cl->downloadName);
1009 }
1010 else
1011#endif
1012 {
1013 Com_sprintf(errorMessage, sizeof(errorMessage), "Cannot autodownload id pk3 file \"%s\"", cl->downloadName);
1014 }
1015 }
1016 else if ( !(sv_allowDownload->integer & DLF_ENABLE) ||
1017 (sv_allowDownload->integer & DLF_NO_UDP) ) {
1018
1019 Com_Printf("clientDownload: %d : \"%s\" download disabled\n", (int) (cl - svs.clients), cl->downloadName);
1020 if (sv_pure->integer) {
1021 Com_sprintf(errorMessage, sizeof(errorMessage), "Could not download \"%s\" because autodownloading is disabled on the server.\n\n"
1022 "You will need to get this file elsewhere before you "
1023 "can connect to this pure server.\n", cl->downloadName);
1024 } else {
1025 Com_sprintf(errorMessage, sizeof(errorMessage), "Could not download \"%s\" because autodownloading is disabled on the server.\n\n"
1026 "The server you are connecting to is not a pure server, "
1027 "set autodownload to No in your settings and you might be "
1028 "able to join the game anyway.\n", cl->downloadName);
1029 }
1030 } else {
1031 // NOTE TTimo this is NOT supposed to happen unless bug in our filesystem scheme?
1032 // if the pk3 is referenced, it must have been found somewhere in the filesystem
1033 Com_Printf("clientDownload: %d : \"%s\" file not found on server\n", (int) (cl - svs.clients), cl->downloadName);
1034 Com_sprintf(errorMessage, sizeof(errorMessage), "File \"%s\" not found on server for autodownloading.\n", cl->downloadName);
1035 }
1036 MSG_WriteByte( msg, svc_download );
1037 MSG_WriteShort( msg, 0 ); // client is expecting block zero
1038 MSG_WriteLong( msg, -1 ); // illegal file size
1039 MSG_WriteString( msg, errorMessage );
1040
1041 *cl->downloadName = 0;
1042
1043 if(cl->download)
1044 FS_FCloseFile(cl->download);
1045
1046 return 1;
1047 }
1048
1049 Com_Printf( "clientDownload: %d : beginning \"%s\"\n", (int) (cl - svs.clients), cl->downloadName );
1050
1051 // Init
1052 cl->downloadCurrentBlock = cl->downloadClientBlock = cl->downloadXmitBlock = 0;
1053 cl->downloadCount = 0;
1054 cl->downloadEOF = qfalse;
1055 }
1056
1057 // Perform any reads that we need to
1058 while (cl->downloadCurrentBlock - cl->downloadClientBlock < MAX_DOWNLOAD_WINDOW &&
1059 cl->downloadSize != cl->downloadCount) {
1060
1061 curindex = (cl->downloadCurrentBlock % MAX_DOWNLOAD_WINDOW);
1062
1063 if (!cl->downloadBlocks[curindex])
1064 cl->downloadBlocks[curindex] = Z_Malloc(MAX_DOWNLOAD_BLKSIZE);
1065
1066 cl->downloadBlockSize[curindex] = FS_Read( cl->downloadBlocks[curindex], MAX_DOWNLOAD_BLKSIZE, cl->download );
1067
1068 if (cl->downloadBlockSize[curindex] < 0) {
1069 // EOF right now
1070 cl->downloadCount = cl->downloadSize;
1071 break;
1072 }
1073
1074 cl->downloadCount += cl->downloadBlockSize[curindex];
1075
1076 // Load in next block
1077 cl->downloadCurrentBlock++;
1078 }
1079
1080 // Check to see if we have eof condition and add the EOF block
1081 if (cl->downloadCount == cl->downloadSize &&
1082 !cl->downloadEOF &&
1083 cl->downloadCurrentBlock - cl->downloadClientBlock < MAX_DOWNLOAD_WINDOW) {
1084
1085 cl->downloadBlockSize[cl->downloadCurrentBlock % MAX_DOWNLOAD_WINDOW] = 0;
1086 cl->downloadCurrentBlock++;
1087
1088 cl->downloadEOF = qtrue; // We have added the EOF block
1089 }
1090
1091 if (cl->downloadClientBlock == cl->downloadCurrentBlock)
1092 return 0; // Nothing to transmit
1093
1094 // Write out the next section of the file, if we have already reached our window,
1095 // automatically start retransmitting
1096 if (cl->downloadXmitBlock == cl->downloadCurrentBlock)
1097 {
1098 // We have transmitted the complete window, should we start resending?
1099 if (svs.time - cl->downloadSendTime > 1000)
1100 cl->downloadXmitBlock = cl->downloadClientBlock;
1101 else
1102 return 0;
1103 }
1104
1105 // Send current block
1106 curindex = (cl->downloadXmitBlock % MAX_DOWNLOAD_WINDOW);
1107
1108 MSG_WriteByte( msg, svc_download );
1109 MSG_WriteShort( msg, cl->downloadXmitBlock );
1110
1111 // block zero is special, contains file size
1112 if ( cl->downloadXmitBlock == 0 )
1113 MSG_WriteLong( msg, cl->downloadSize );
1114
1115 MSG_WriteShort( msg, cl->downloadBlockSize[curindex] );
1116
1117 // Write the block
1118 if(cl->downloadBlockSize[curindex])
1119 MSG_WriteData(msg, cl->downloadBlocks[curindex], cl->downloadBlockSize[curindex]);
1120
1121 Com_DPrintf( "clientDownload: %d : writing block %d\n", (int) (cl - svs.clients), cl->downloadXmitBlock );
1122
1123 // Move on to the next block
1124 // It will get sent with next snap shot. The rate will keep us in line.
1125 cl->downloadXmitBlock++;
1126 cl->downloadSendTime = svs.time;
1127
1128 return 1;
1129}

1214/*
1215=================
1216SV_VerifyPaks_f
1217
1218If we are pure, disconnect the client if they do no meet the following conditions:
1219
12201. the first two checksums match our view of cgame and ui
12212. there are no any additional checksums that we do not have
1222
1223This routine would be a bit simpler with a goto but i abstained
1224
1225=================
1226*/
1227static void SV_VerifyPaks_f( client_t *cl ) {
1228 int nChkSum1, nChkSum2, nClientPaks, nServerPaks, i, j, nCurArg;
1229 int nClientChkSum[1024];
1230 int nServerChkSum[1024];
1231 const char *pPaks, *pArg;
1232 qboolean bGood = qtrue;
1233
1234 // if we are pure, we "expect" the client to load certain things from
1235 // certain pk3 files, namely we want the client to have loaded the
1236 // ui and cgame that we think should be loaded based on the pure setting
1237 //
1238 if ( sv_pure->integer != 0 ) {
1239
1240 nChkSum1 = nChkSum2 = 0;
1241 // we run the game, so determine which cgame and ui the client "should" be running
1242 bGood = (FS_FileIsInPAK("vm/cgame.qvm", &nChkSum1) == 1);
1243 if (bGood)
1244 bGood = (FS_FileIsInPAK("vm/ui.qvm", &nChkSum2) == 1);
1245
1246 nClientPaks = Cmd_Argc();
1247
1248 // start at arg 2 ( skip serverId cl_paks )
1249 nCurArg = 1;
1250
1251 pArg = Cmd_Argv(nCurArg++);
1252 if(!pArg) {
1253 bGood = qfalse;
1254 }
1255 else
1256 {
1257 // https://zerowing.idsoftware.com/bugzilla/show_bug.cgi?id=475
1258 // we may get incoming cp sequences from a previous checksumFeed, which we need to ignore
1259 // since serverId is a frame count, it always goes up
1260 if (atoi(pArg) < sv.checksumFeedServerId)
1261 {
1262 Com_DPrintf("ignoring outdated cp command from client %s\n", cl->name);
1263 return;
1264 }
1265 }
1266
1267 // we basically use this while loop to avoid using 'goto' :)
1268 while (bGood) {
1269
1270 // must be at least 6: "cl_paks cgame ui @ firstref ... numChecksums"
1271 // numChecksums is encoded
1272 if (nClientPaks < 6) {
1273 bGood = qfalse;
1274 break;
1275 }
1276 // verify first to be the cgame checksum
1277 pArg = Cmd_Argv(nCurArg++);
1278 if (!pArg || *pArg == '@' || atoi(pArg) != nChkSum1 ) {
1279 bGood = qfalse;
1280 break;
1281 }
1282 // verify the second to be the ui checksum
1283 pArg = Cmd_Argv(nCurArg++);
1284 if (!pArg || *pArg == '@' || atoi(pArg) != nChkSum2 ) {
1285 bGood = qfalse;
1286 break;
1287 }
1288 // should be sitting at the delimeter now
1289 pArg = Cmd_Argv(nCurArg++);
1290 if (*pArg != '@') {
1291 bGood = qfalse;
1292 break;
1293 }
1294 // store checksums since tokenization is not re-entrant
1295 for (i = 0; nCurArg < nClientPaks; i++) {
1296 nClientChkSum[i] = atoi(Cmd_Argv(nCurArg++));
1297 }
1298
1299 // store number to compare against (minus one cause the last is the number of checksums)
1300 nClientPaks = i - 1;
1301
1302 // make sure none of the client check sums are the same
1303 // so the client can't send 5 the same checksums
1304 for (i = 0; i < nClientPaks; i++) {
1305 for (j = 0; j < nClientPaks; j++) {
1306 if (i == j)
1307 continue;
1308 if (nClientChkSum[i] == nClientChkSum[j]) {
1309 bGood = qfalse;
1310 break;
1311 }
1312 }
1313 if (bGood == qfalse)
1314 break;
1315 }
1316 if (bGood == qfalse)
1317 break;
1318
1319 // get the pure checksums of the pk3 files loaded by the server
1320 pPaks = FS_LoadedPakPureChecksums();
1321 Cmd_TokenizeString( pPaks );
1322 nServerPaks = Cmd_Argc();
1323 if (nServerPaks > 1024)
1324 nServerPaks = 1024;
1325
1326 for (i = 0; i < nServerPaks; i++) {
1327 nServerChkSum[i] = atoi(Cmd_Argv(i));
1328 }
1329
1330 // check if the client has provided any pure checksums of pk3 files not loaded by the server
1331 for (i = 0; i < nClientPaks; i++) {
1332 for (j = 0; j < nServerPaks; j++) {
1333 if (nClientChkSum[i] == nServerChkSum[j]) {
1334 break;
1335 }
1336 }
1337 if (j >= nServerPaks) {
1338 bGood = qfalse;
1339 break;
1340 }
1341 }
1342 if ( bGood == qfalse ) {
1343 break;
1344 }
1345
1346 // check if the number of checksums was correct
1347 nChkSum1 = sv.checksumFeed;
1348 for (i = 0; i < nClientPaks; i++) {
1349 nChkSum1 ^= nClientChkSum[i];
1350 }
1351 nChkSum1 ^= nClientPaks;
1352 if (nChkSum1 != nClientChkSum[nClientPaks]) {
1353 bGood = qfalse;
1354 break;
1355 }
1356
1357 // break out
1358 break;
1359 }
1360
1361 cl->gotCP = qtrue;
1362
1363 if (bGood) {
1364 cl->pureAuthentic = 1;
1365 }
1366 else {
1367 cl->pureAuthentic = 0;
1368 cl->lastSnapshotTime = 0;
1369 cl->state = CS_ACTIVE;
1370 SV_SendClientSnapshot( cl );
1371 SV_DropClient( cl, "Unpure client detected. Invalid .PK3 files referenced!" );
1372 }
1373 }
1374}

1591===============
1592SV_ClientCommand
1593===============
1594*/
1595static qboolean SV_ClientCommand( client_t *cl, msg_t *msg ) {
1596 int seq;
1597 const char *s;
1598 qboolean clientOk = qtrue;
1599
1600 seq = MSG_ReadLong( msg );
1601 s = MSG_ReadString( msg );
1602
1603 // see if we have already executed it
1604 if ( cl->lastClientCommand >= seq ) {
1605 return qtrue;
1606 }
1607
1608 Com_DPrintf( "clientCommand: %s : %i : %s\n", cl->name, seq, s );
1609
1610 // drop the connection if we have somehow lost commands
1611 if ( seq > cl->lastClientCommand + 1 ) {
1612 Com_Printf( "Client %s lost %i clientCommands\n", cl->name,
1613 seq - cl->lastClientCommand + 1 );
1614 SV_DropClient( cl, "Lost reliable commands" );
1615 return qfalse;
1616 }
1617
1618 // malicious users may try using too many string commands
1619 // to lag other players. If we decide that we want to stall
1620 // the command, we will stop processing the rest of the packet,
1621 // including the usercmd. This causes flooders to lag themselves
1622 // but not other people
1623 // We don't do this when the client hasn't been active yet since it's
1624 // normal to spam a lot of commands when downloading
1625 if ( !com_cl_running->integer &&
1626 cl->state >= CS_ACTIVE &&
1627 sv_floodProtect->integer &&
1628 svs.time < cl->nextReliableTime ) {
1629 // ignore any other text messages from this client but let them keep playing
1630 // TTimo - moved the ignored verbose to the actual processing in SV_ExecuteClientCommand, only printing if the core doesn't intercept
1631 clientOk = qfalse;
1632 }
1633
1634 // don't allow another command for one second
1635 cl->nextReliableTime = svs.time + 1000;
1636
1637 SV_ExecuteClientCommand( cl, s, clientOk );
1638
1639 cl->lastClientCommand = seq;
1640 Com_sprintf(cl->lastClientCommandString, sizeof(cl->lastClientCommandString), "%s", s);
1641
1642 return qtrue; // continue procesing
1643}

1666/*
1667==================
1668SV_UserMove
1669
1670The message usually contains all the movement commands
1671that were in the last three packets, so that the information
1672in dropped packets can be recovered.
1673
1674On very fast clients, there may be multiple usercmd packed into
1675each of the backup packets.
1676==================
1677*/
1678static void SV_UserMove( client_t *cl, msg_t *msg, qboolean delta ) {
1679 int i, key;
1680 int cmdCount;
1681 usercmd_t nullcmd;
1682 usercmd_t cmds[MAX_PACKET_USERCMDS];
1683 usercmd_t *cmd, *oldcmd;
1684
1685 if ( delta ) {
1686 cl->deltaMessage = cl->messageAcknowledge;
1687 } else {
1688 cl->deltaMessage = -1;
1689 }
1690
1691 cmdCount = MSG_ReadByte( msg );
1692
1693 if ( cmdCount < 1 ) {
1694 Com_Printf( "cmdCount < 1\n" );
1695 return;
1696 }
1697
1698 if ( cmdCount > MAX_PACKET_USERCMDS ) {
1699 Com_Printf( "cmdCount > MAX_PACKET_USERCMDS\n" );
1700 return;
1701 }
1702
1703 // use the checksum feed in the key
1704 key = sv.checksumFeed;
1705 // also use the message acknowledge
1706 key ^= cl->messageAcknowledge;
1707 // also use the last acknowledged server command in the key
1708 key ^= MSG_HashKey(cl->reliableCommands[ cl->reliableAcknowledge & (MAX_RELIABLE_COMMANDS-1) ], 32);
1709
1710 Com_Memset( &nullcmd, 0, sizeof(nullcmd) );
1711 oldcmd = &nullcmd;
1712 for ( i = 0 ; i < cmdCount ; i++ ) {
1713 cmd = &cmds[i];
1714 MSG_ReadDeltaUsercmdKey( msg, key, oldcmd, cmd );
1715 oldcmd = cmd;
1716 }
1717
1718 // save time for ping calculation
1719 cl->frames[ cl->messageAcknowledge & PACKET_MASK ].messageAcked = svs.time;
1720
1721 // TTimo
1722 // catch the no-cp-yet situation before SV_ClientEnterWorld
1723 // if CS_ACTIVE, then it's time to trigger a new gamestate emission
1724 // if not, then we are getting remaining parasite usermove commands, which we should ignore
1725 if (sv_pure->integer != 0 && cl->pureAuthentic == 0 && !cl->gotCP) {
1726 if (cl->state == CS_ACTIVE)
1727 {
1728 // we didn't get a cp yet, don't assume anything and just send the gamestate all over again
1729 Com_DPrintf( "%s: didn't get cp command, resending gamestate\n", cl->name);
1730 SV_SendClientGameState( cl );
1731 }
1732 return;
1733 }
1734
1735 // if this is the first usercmd we have received
1736 // this gamestate, put the client into the world
1737 if ( cl->state == CS_PRIMED ) {
1738 SV_ClientEnterWorld( cl, &cmds[0] );
1739 // the moves can be processed normaly
1740 }
1741
1742 // a bad cp command was sent, drop the client
1743 if (sv_pure->integer != 0 && cl->pureAuthentic == 0) {
1744 SV_DropClient( cl, "Cannot validate pure client!");
1745 return;
1746 }
1747
1748 if ( cl->state != CS_ACTIVE ) {
1749 cl->deltaMessage = -1;
1750 return;
1751 }
1752
1753 // usually, the first couple commands will be duplicates
1754 // of ones we have previously received, but the servertimes
1755 // in the commands will cause them to be immediately discarded
1756 for ( i = 0 ; i < cmdCount ; i++ ) {
1757 // if this is a cmd from before a map_restart ignore it
1758 if ( cmds[i].serverTime > cmds[cmdCount-1].serverTime ) {
1759 continue;
1760 }
1761 // extremely lagged or cmd from before a map_restart
1762 //if ( cmds[i].serverTime > svs.time + 3000 ) {
1763 // continue;
1764 //}
1765 // don't execute if this is an old cmd which is already executed
1766 // these old cmds are included when cl_packetdup > 0
1767 if ( cmds[i].serverTime <= cl->lastUsercmd.serverTime ) {
1768 continue;
1769 }
1770 SV_ClientThink (cl, &cmds[ i ]);
1771 }
1772}

1797void SV_UserVoip(client_t *cl, msg_t *msg, qboolean ignoreData)
1798{
1799 int sender, generation, sequence, frames, packetsize;
1800 uint8_t recips[(MAX_CLIENTS + 7) / 8];
1801 int flags;
1802 byte encoded[sizeof(cl->voipPacket[0]->data)];
1803 client_t *client = NULL;
1804 voipServerPacket_t *packet = NULL;
1805 int i;
1806
1807 sender = cl - svs.clients;
1808 generation = MSG_ReadByte(msg);
1809 sequence = MSG_ReadLong(msg);
1810 frames = MSG_ReadByte(msg);
1811 MSG_ReadData(msg, recips, sizeof(recips));
1812 flags = MSG_ReadByte(msg);
1813 packetsize = MSG_ReadShort(msg);
1814
1815 if (msg->readcount > msg->cursize)
1816 return; // short/invalid packet, bail.
1817
1818 if (packetsize > sizeof (encoded)) { // overlarge packet?
1819 int bytesleft = packetsize;
1820 while (bytesleft) {
1821 int br = bytesleft;
1822 if (br > sizeof (encoded))
1823 br = sizeof (encoded);
1824 MSG_ReadData(msg, encoded, br);
1825 bytesleft -= br;
1826 }
1827 return; // overlarge packet, bail.
1828 }
1829
1830 MSG_ReadData(msg, encoded, packetsize);
1831
1832 if (ignoreData || SV_ShouldIgnoreVoipSender(cl))
1833 return; // Blacklisted, disabled, etc.
1834
1835 // !!! FIXME: see if we read past end of msg...
1836
1837 // !!! FIXME: reject if not opus data.
1838 // !!! FIXME: decide if this is bogus data?
1839
1840 // decide who needs this VoIP packet sent to them...
1841 for (i = 0, client = svs.clients; i < sv_maxclients->integer ; i++, client++) {
1842 if (client->state != CS_ACTIVE)
1843 continue; // not in the game yet, don't send to this guy.
1844 else if (i == sender)
1845 continue; // don't send voice packet back to original author.
1846 else if (!client->hasVoip)
1847 continue; // no VoIP support, or unsupported protocol
1848 else if (client->muteAllVoip)
1849 continue; // client is ignoring everyone.
1850 else if (client->ignoreVoipFromClient[sender])
1851 continue; // client is ignoring this talker.
1852 else if (*cl->downloadName) // !!! FIXME: possible to DoS?
1853 continue; // no VoIP allowed if downloading, to save bandwidth.
1854
1855 if(Com_IsVoipTarget(recips, sizeof(recips), i))
1856 flags |= VOIP_DIRECT;
1857 else
1858 flags &= ~VOIP_DIRECT;
1859
1860 if (!(flags & (VOIP_SPATIAL | VOIP_DIRECT)))
1861 continue; // not addressed to this player.
1862
1863 // Transmit this packet to the client.
1864 if (client->queuedVoipPackets >= ARRAY_LEN(client->voipPacket)) {
1865 Com_Printf("Too many VoIP packets queued for client #%d\n", i);
1866 continue; // no room for another packet right now.
1867 }
1868
1869 packet = Z_Malloc(sizeof(*packet));
1870 packet->sender = sender;
1871 packet->frames = frames;
1872 packet->len = packetsize;
1873 packet->generation = generation;
1874 packet->sequence = sequence;
1875 packet->flags = flags;
1876 memcpy(packet->data, encoded, packetsize);
1877
1878 client->voipPacket[(client->queuedVoipIndex + client->queuedVoipPackets) % ARRAY_LEN(client->voipPacket)] = packet;
1879 client->queuedVoipPackets++;
1880 }
1881}

1896SV_ExecuteClientMessage

1898Parse a client packet

1901void SV_ExecuteClientMessage( client_t *cl, msg_t *msg ) {
1902 int c;
1903 int serverId;
1904
1905 MSG_Bitstream(msg);
1906
1907 serverId = MSG_ReadLong( msg );
1908 cl->messageAcknowledge = MSG_ReadLong( msg );
1909
1910 if (cl->messageAcknowledge < 0) {
1911 // usually only hackers create messages like this
1912 // it is more annoying for them to let them hanging
1913#ifndef NDEBUG
1914 SV_DropClient( cl, "DEBUG: illegible client message" );
1915#endif
1916 return;
1917 }
1918
1919 cl->reliableAcknowledge = MSG_ReadLong( msg );
1920
1921 // NOTE: when the client message is fux0red the acknowledgement numbers
1922 // can be out of range, this could cause the server to send thousands of server
1923 // commands which the server thinks are not yet acknowledged in SV_UpdateServerCommandsToClient
1924 if (cl->reliableAcknowledge < cl->reliableSequence - MAX_RELIABLE_COMMANDS) {
1925 // usually only hackers create messages like this
1926 // it is more annoying for them to let them hanging
1927#ifndef NDEBUG
1928 SV_DropClient( cl, "DEBUG: illegible client message" );
1929#endif
1930 cl->reliableAcknowledge = cl->reliableSequence;
1931 return;
1932 }
1933 // if this is a usercmd from a previous gamestate,
1934 // ignore it or retransmit the current gamestate
1935 //
1936 // if the client was downloading, let it stay at whatever serverId and
1937 // gamestate it was at. This allows it to keep downloading even when
1938 // the gamestate changes. After the download is finished, we'll
1939 // notice and send it a new game state
1940 //
1941 // https://zerowing.idsoftware.com/bugzilla/show_bug.cgi?id=536
1942 // don't drop as long as previous command was a nextdl, after a dl is done, downloadName is set back to ""
1943 // but we still need to read the next message to move to next download or send gamestate
1944 // I don't like this hack though, it must have been working fine at some point, suspecting the fix is somewhere else
1945 if ( serverId != sv.serverId && !*cl->downloadName && !strstr(cl->lastClientCommandString, "nextdl") ) {
1946 if ( serverId >= sv.restartedServerId && serverId < sv.serverId ) { // TTimo - use a comparison here to catch multiple map_restart
1947 // they just haven't caught the map_restart yet
1948 Com_DPrintf("%s : ignoring pre map_restart / outdated client message\n", cl->name);
1949 return;
1950 }
1951 // if we can tell that the client has dropped the last
1952 // gamestate we sent them, resend it
1953 if ( cl->messageAcknowledge > cl->gamestateMessageNum ) {
1954 Com_DPrintf( "%s : dropped gamestate, resending\n", cl->name );
1955 SV_SendClientGameState( cl );
1956 }
1957 return;
1958 }
1959
1960 // this client has acknowledged the new gamestate so it's
1961 // safe to start sending it the real time again
1962 if( cl->oldServerTime && serverId == sv.serverId ){
1963 Com_DPrintf( "%s acknowledged gamestate\n", cl->name );
1964 cl->oldServerTime = 0;
1965 }
1966
1967 // read optional clientCommand strings
1968 do {
1969 c = MSG_ReadByte( msg );
1970
1971 if ( c == clc_EOF ) {
1972 break;
1973 }
1974
1975 if ( c != clc_clientCommand ) {
1976 break;
1977 }
1978 if ( !SV_ClientCommand( cl, msg ) ) {
1979 return; // we couldn't execute it because of the flood protection
1980 }
1981 if (cl->state == CS_ZOMBIE) {
1982 return; // disconnect command
1983 }
1984 } while ( 1 );
1985
1986 // skip legacy speex voip data
1987 if ( c == clc_voipSpeex ) {
1988#ifdef USE_VOIP
1989 SV_UserVoip( cl, msg, qtrue );
1990 c = MSG_ReadByte( msg );
1991#endif
1992 }
1993
1994 // read optional voip data
1995 if ( c == clc_voipOpus ) {
1996#ifdef USE_VOIP
1997 SV_UserVoip( cl, msg, qfalse );
1998 c = MSG_ReadByte( msg );
1999#endif
2000 }
2001
2002 // read the usercmd_t
2003 if ( c == clc_move ) {
2004 SV_UserMove( cl, msg, qtrue );
2005 } else if ( c == clc_moveNoDelta ) {
2006 SV_UserMove( cl, msg, qfalse );
2007 } else if ( c != clc_EOF ) {
2008 Com_Printf( "WARNING: bad command byte for client %i\n", (int) (cl - svs.clients) );
2009 }

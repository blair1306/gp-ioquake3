77/*
78===============
79SV_ExpandNewlines
80
81Converts newlines to "\n" so a line prints nicer
82===============
83*/
84static char *SV_ExpandNewlines( char *in ) {
85 static char string[1024];
86 int l;
87
88 l = 0;
89 while ( *in && l < sizeof(string) - 3 ) {
90 if ( *in == '\n' ) {
91 string[l++] = '\\';
92 string[l++] = 'n';
93 } else {
94 string[l++] = *in;
95 }
96 in++;
97 }
98 string[l] = 0;
99
100 return string;
101}

135/*
136======================
137SV_AddServerCommand
138
139The given command will be transmitted to the client, and is guaranteed to
140not have future snapshot_t executed before it is executed
141======================
142*/
143void SV_AddServerCommand( client_t *client, const char *cmd ) {
144 int index, i;
145
146 // this is very ugly but it's also a waste to for instance send multiple config string updates
147 // for the same config string index in one snapshot
148// if ( SV_ReplacePendingServerCommands( client, cmd ) ) {
149// return;
150// }
151
152 // do not send commands until the gamestate has been sent
153 if( client->state < CS_PRIMED )
154 return;
155
156 client->reliableSequence++;
157 // if we would be losing an old command that hasn't been acknowledged,
158 // we must drop the connection
159 // we check == instead of >= so a broadcast print added by SV_DropClient()
160 // doesn't cause a recursive drop client
161 if ( client->reliableSequence - client->reliableAcknowledge == MAX_RELIABLE_COMMANDS + 1 ) {
162 Com_Printf( "===== pending server commands =====\n" );
163 for ( i = client->reliableAcknowledge + 1 ; i <= client->reliableSequence ; i++ ) {
164 Com_Printf( "cmd %5d: %s\n", i, client->reliableCommands[ i & (MAX_RELIABLE_COMMANDS-1) ] );
165 }
166 Com_Printf( "cmd %5d: %s\n", i, cmd );
167 SV_DropClient( client, "Server command overflow" );
168 return;
169 }
170 index = client->reliableSequence & ( MAX_RELIABLE_COMMANDS - 1 );
171 Q_strncpyz( client->reliableCommands[ index ], cmd, sizeof( client->reliableCommands[ index ] ) );
172}

175/*
176=================
177SV_SendServerCommand
178
179Sends a reliable command string to be interpreted by
180the client game module: "cp", "print", "chat", etc
181A NULL client will broadcast to all clients
182=================
183*/
184void QDECL SV_SendServerCommand(client_t *cl, const char *fmt, ...) {
185 va_list argptr;
186 byte message[MAX_MSGLEN];
187 client_t *client;
188 int j;
189
190 va_start (argptr,fmt);
191 Q_vsnprintf ((char *)message, sizeof(message), fmt,argptr);
192 va_end (argptr);
193
194 // Fix to http://aluigi.altervista.org/adv/q3msgboom-adv.txt
195 // The actual cause of the bug is probably further downstream
196 // and should maybe be addressed later, but this certainly
197 // fixes the problem for now
198 if ( strlen ((char *)message) > 1022 ) {
199 return;
200 }
201
202 if ( cl != NULL ) {
203 SV_AddServerCommand( cl, (char *)message );
204 return;
205 }
206
207 // hack to echo broadcast prints to console
208 if ( com_dedicated->integer && !strncmp( (char *)message, "print", 5) ) {
209 Com_Printf ("broadcast: %s\n", SV_ExpandNewlines((char *)message) );
210 }
211
212 // send the data to all relevent clients
213 for (j = 0, client = svs.clients; j < sv_maxclients->integer ; j++, client++) {
214 SV_AddServerCommand( client, (char *)message );
215 }
216}

227/*
228================
229SV_MasterHeartbeat
230
231Send a message to the masters every few minutes to
232let it know we are alive, and log information.
233We will also have a heartbeat sent when a server
234changes from empty to non-empty, and full to non-full,
235but not on every player enter or exit.
236================
237*/
238#define HEARTBEAT_MSEC 300*1000
239#define MASTERDNS_MSEC 24*60*60*1000
240void SV_MasterHeartbeat(const char *message)
241{
242 static netadr_t adr[MAX_MASTER_SERVERS][2]; // [2] for v4 and v6 address for the same address string.
243 int i;
244 int res;
245 int netenabled;
246
247 netenabled = Cvar_VariableIntegerValue("net_enabled");
248
249 // "dedicated 1" is for lan play, "dedicated 2" is for inet public play
250 if (!com_dedicated || com_dedicated->integer != 2 || !(netenabled & (NET_ENABLEV4 | NET_ENABLEV6)))
251 return; // only dedicated servers send heartbeats
252
253 // if not time yet, don't send anything
254 if ( svs.time < svs.nextHeartbeatTime )
255 return;
256
257 if ( !Q_stricmp( com_gamename->string, LEGACY_MASTER_GAMENAME ) )
258 message = LEGACY_HEARTBEAT_FOR_MASTER;
259
260 svs.nextHeartbeatTime = svs.time + HEARTBEAT_MSEC;
261
262 // send to group masters
263 for (i = 0; i < MAX_MASTER_SERVERS; i++)
264 {
265 if(!sv_master[i]->string[0])
266 continue;
267
268 // see if we haven't already resolved the name or if it's been over 24 hours
269 // resolving usually causes hitches on win95, so only do it when needed
270 if (sv_master[i]->modified || svs.time > svs.masterResolveTime[i])
271 {
272 sv_master[i]->modified = qfalse;
273 svs.masterResolveTime[i] = svs.time + MASTERDNS_MSEC;
274
275 if(netenabled & NET_ENABLEV4)
276 {
277 Com_Printf("Resolving %s (IPv4)\n", sv_master[i]->string);
278 res = NET_StringToAdr(sv_master[i]->string, &adr[i][0], NA_IP);
279
280 if(res == 2)
281 {
282 // if no port was specified, use the default master port
283 adr[i][0].port = BigShort(PORT_MASTER);
284 }
285
286 if(res)
287 Com_Printf( "%s resolved to %s\n", sv_master[i]->string, NET_AdrToStringwPort(adr[i][0]));
288 else
289 Com_Printf( "%s has no IPv4 address.\n", sv_master[i]->string);
290 }
291
292 if(netenabled & NET_ENABLEV6)
293 {
294 Com_Printf("Resolving %s (IPv6)\n", sv_master[i]->string);
295 res = NET_StringToAdr(sv_master[i]->string, &adr[i][1], NA_IP6);
296
297 if(res == 2)
298 {
299 // if no port was specified, use the default master port
300 adr[i][1].port = BigShort(PORT_MASTER);
301 }
302
303 if(res)
304 Com_Printf( "%s resolved to %s\n", sv_master[i]->string, NET_AdrToStringwPort(adr[i][1]));
305 else
306 Com_Printf( "%s has no IPv6 address.\n", sv_master[i]->string);
307 }
308 }
309
310 if(adr[i][0].type == NA_BAD && adr[i][1].type == NA_BAD)
311 {
312 continue;
313 }
314
315
316 Com_Printf ("Sending heartbeat to %s\n", sv_master[i]->string );
317
318 // this command should be changed if the server info / status format
319 // ever incompatably changes
320
321 if(adr[i][0].type != NA_BAD)
322 NET_OutOfBandPrint( NS_SERVER, adr[i][0], "heartbeat %s\n", message);
323 if(adr[i][1].type != NA_BAD)
324 NET_OutOfBandPrint( NS_SERVER, adr[i][1], "heartbeat %s\n", message);
325 }
326}

357// This is deliberately quite large to make it more of an effort to DoS
358#define MAX_BUCKETS 16384
359#define MAX_HASHES 1024
360
361static leakyBucket_t buckets[ MAX_BUCKETS ];
362static leakyBucket_t *bucketHashes[ MAX_HASHES ];
363leakyBucket_t outboundLeakyBucket;
364
365/*
366================
367SVC_HashForAddress
368================
369*/
370static long SVC_HashForAddress( netadr_t address ) {
371 byte *ip = NULL;
372 size_t size = 0;
373 int i;
374 long hash = 0;
375
376 switch ( address.type ) {
377 case NA_IP: ip = address.ip; size = 4; break;
378 case NA_IP6: ip = address.ip6; size = 16; break;
379 default: break;
380 }
381
382 for ( i = 0; i < size; i++ ) {
383 hash += (long)( ip[ i ] ) * ( i + 119 );
384 }
385
386 hash = ( hash ^ ( hash >> 10 ) ^ ( hash >> 20 ) );
387 hash &= ( MAX_HASHES - 1 );
388
389 return hash;
390}
391
392/*
393================
394SVC_BucketForAddress
395
396Find or allocate a bucket for an address
397================
398*/
399static leakyBucket_t *SVC_BucketForAddress( netadr_t address, int burst, int period ) {
400 leakyBucket_t *bucket = NULL;
401 int i;
402 long hash = SVC_HashForAddress( address );
403 int now = Sys_Milliseconds();
404
405 for ( bucket = bucketHashes[ hash ]; bucket; bucket = bucket->next ) {
406 switch ( bucket->type ) {
407 case NA_IP:
408 if ( memcmp( bucket->ipv._4, address.ip, 4 ) == 0 ) {
409 return bucket;
410 }
411 break;
412
413 case NA_IP6:
414 if ( memcmp( bucket->ipv._6, address.ip6, 16 ) == 0 ) {
415 return bucket;
416 }
417 break;
418
419 default:
420 break;
421 }
422 }
423
424 for ( i = 0; i < MAX_BUCKETS; i++ ) {
425 int interval;
426
427 bucket = &buckets[ i ];
428 interval = now - bucket->lastTime;
429
430 // Reclaim expired buckets
431 if ( bucket->lastTime > 0 && ( interval > ( burst * period ) ||
432 interval < 0 ) ) {
433 if ( bucket->prev != NULL ) {
434 bucket->prev->next = bucket->next;
435 } else {
436 bucketHashes[ bucket->hash ] = bucket->next;
437 }
438
439 if ( bucket->next != NULL ) {
440 bucket->next->prev = bucket->prev;
441 }
442
443 Com_Memset( bucket, 0, sizeof( leakyBucket_t ) );
444 }
445
446 if ( bucket->type == NA_BAD ) {
447 bucket->type = address.type;
448 switch ( address.type ) {
449 case NA_IP: Com_Memcpy( bucket->ipv._4, address.ip, 4 ); break;
450 case NA_IP6: Com_Memcpy( bucket->ipv._6, address.ip6, 16 ); break;
451 default: break;
452 }
453
454 bucket->lastTime = now;
455 bucket->burst = 0;
456 bucket->hash = hash;
457
458 // Add to the head of the relevant hash chain
459 bucket->next = bucketHashes[ hash ];
460 if ( bucketHashes[ hash ] != NULL ) {
461 bucketHashes[ hash ]->prev = bucket;
462 }
463
464 bucket->prev = NULL;
465 bucketHashes[ hash ] = bucket;
466
467 return bucket;
468 }
469 }
470
471 // Couldn't allocate a bucket for this address
472 return NULL;
473}

475/*
476================
477SVC_RateLimit
478================
479*/
480qboolean SVC_RateLimit( leakyBucket_t *bucket, int burst, int period ) {
481 if ( bucket != NULL ) {
482 int now = Sys_Milliseconds();
483 int interval = now - bucket->lastTime;
484 int expired = interval / period;
485 int expiredRemainder = interval % period;
486
487 if ( expired > bucket->burst || interval < 0 ) {
488 bucket->burst = 0;
489 bucket->lastTime = now;
490 } else {
491 bucket->burst -= expired;
492 bucket->lastTime = now - expiredRemainder;
493 }
494
495 if ( bucket->burst < burst ) {
496 bucket->burst++;
497
498 return qfalse;
499 }
500 }
501
502 return qtrue;
503}

518/*
519================
520SVC_Status
521
522Responds with all the info that qplug or qspy can see about the server
523and all connected players. Used for getting detailed information after
524the simple info query.
525================
526*/
527static void SVC_Status( netadr_t from ) {
528 char player[1024];
529 char status[MAX_MSGLEN];
530 int i;
531 client_t *cl;
532 playerState_t *ps;
533 int statusLength;
534 int playerLength;
535 char infostring[MAX_INFO_STRING];
536
537 // ignore if we are in single player
538 if ( Cvar_VariableValue( "g_gametype" ) == GT_SINGLE_PLAYER || Cvar_VariableValue("ui_singlePlayerActive")) {
539 return;
540 }
541
542 // Prevent using getstatus as an amplifier
543 if ( SVC_RateLimitAddress( from, 10, 1000 ) ) {
544 Com_DPrintf( "SVC_Status: rate limit from %s exceeded, dropping request\n",
545 NET_AdrToString( from ) );
546 return;
547 }
548
549 // Allow getstatus to be DoSed relatively easily, but prevent
550 // excess outbound bandwidth usage when being flooded inbound
551 if ( SVC_RateLimit( &outboundLeakyBucket, 10, 100 ) ) {
552 Com_DPrintf( "SVC_Status: rate limit exceeded, dropping request\n" );
553 return;
554 }
555
556 // A maximum challenge length of 128 should be more than plenty.
557 if(strlen(Cmd_Argv(1)) > 128)
558 return;
559
560 strcpy( infostring, Cvar_InfoString( CVAR_SERVERINFO ) );
561
562 // echo back the parameter to status. so master servers can use it as a challenge
563 // to prevent timed spoofed reply packets that add ghost servers
564 Info_SetValueForKey( infostring, "challenge", Cmd_Argv(1) );
565
566 status[0] = 0;
567 statusLength = 0;
568
569 for (i=0 ; i < sv_maxclients->integer ; i++) {
570 cl = &svs.clients[i];
571 if ( cl->state >= CS_CONNECTED ) {
572 ps = SV_GameClientNum( i );
573 Com_sprintf (player, sizeof(player), "%i %i \"%s\"\n",
574 ps->persistant[PERS_SCORE], cl->ping, cl->name);
575 playerLength = strlen(player);
576 if (statusLength + playerLength >= sizeof(status) ) {
577 break; // can't hold any more
578 }
579 strcpy (status + statusLength, player);
580 statusLength += playerLength;
581 }
582 }
583
584 NET_OutOfBandPrint( NS_SERVER, from, "statusResponse\n%s\n%s", infostring, status );
585}

619 /*

641 // echo back the parameter to status. so servers can use it as a challenge

659 va("%i", sv_maxclients->integer - sv_privateClients->integer ) );
660 Info_SetValueForKey( infostring, "gametype", va("%i", sv_gametype->integer ) );
661 Info_SetValueForKey( infostring, "pure", va("%i", sv_pure->integer ) );

664#ifdef USE_VOIP

667 }
668#endif
669
670 if( sv_minPing->integer ) {
671 Info_SetValueForKey( infostring, "minPing", va("%i", sv_minPing->integer) );
672 }
673 if( sv_maxPing->integer ) {
674 Info_SetValueForKey( infostring, "maxPing", va("%i", sv_maxPing->integer) );
675 }
676 gamedir = Cvar_VariableString( "fs_game" );
677 if( *gamedir ) {
678 Info_SetValueForKey( infostring, "game", gamedir );
679 }
680
681 NET_OutOfBandPrint( NS_SERVER, from, "infoResponse\n%s", infostring );
682}

694/*
695===============
696SVC_RemoteCommand
697
698An rcon packet arrived from the network.
699Shift down the remaining args
700Redirect all printfs
701===============
702*/
703static void SVC_RemoteCommand( netadr_t from, msg_t *msg ) {
704 qboolean valid;
705 char remaining[1024];
706 // TTimo - scaled down to accumulate, but not overflow anything network wise, print wise etc.
707 // (OOB messages are the bottleneck here)
708#define SV_OUTPUTBUF_LENGTH (1024 - 16)
709 char sv_outputbuf[SV_OUTPUTBUF_LENGTH];
710 char *cmd_aux;
711
712 // Prevent using rcon as an amplifier and make dictionary attacks impractical
713 if ( SVC_RateLimitAddress( from, 10, 1000 ) ) {
714 Com_DPrintf( "SVC_RemoteCommand: rate limit from %s exceeded, dropping request\n",
715 NET_AdrToString( from ) );
716 return;
717 }
718
719 if ( !strlen( sv_rconPassword->string ) ||
720 strcmp (Cmd_Argv(1), sv_rconPassword->string) ) {
721 static leakyBucket_t bucket;
722
723 // Make DoS via rcon impractical
724 if ( SVC_RateLimit( &bucket, 10, 1000 ) ) {
725 Com_DPrintf( "SVC_RemoteCommand: rate limit exceeded, dropping request\n" );
726 return;
727 }
728
729 valid = qfalse;
730 Com_Printf ("Bad rcon from %s: %s\n", NET_AdrToString (from), Cmd_ArgsFrom(2) );
731 } else {
732 valid = qtrue;
733 Com_Printf ("Rcon from %s: %s\n", NET_AdrToString (from), Cmd_ArgsFrom(2) );
734 }
735
736 // start redirecting all print outputs to the packet
737 svs.redirectAddress = from;
738 Com_BeginRedirect (sv_outputbuf, SV_OUTPUTBUF_LENGTH, SV_FlushRedirect);
739
740 if ( !strlen( sv_rconPassword->string ) ) {
741 Com_Printf ("No rconpassword set on the server.\n");
742 } else if ( !valid ) {
743 Com_Printf ("Bad rconpassword.\n");
744 } else {
745 remaining[0] = 0;
746
747 // https://zerowing.idsoftware.com/bugzilla/show_bug.cgi?id=543
748 // get the command directly, "rcon <pass> <command>" to avoid quoting issues
749 // extract the command by walking
750 // since the cmd formatting can fuckup (amount of spaces), using a dumb step by step parsing
751 cmd_aux = Cmd_Cmd();
752 cmd_aux+=4;
753 while(cmd_aux[0]==' ')
754 cmd_aux++;
755 while(cmd_aux[0] && cmd_aux[0]!=' ') // password
756 cmd_aux++;
757 while(cmd_aux[0]==' ')
758 cmd_aux++;
759
760 Q_strcat( remaining, sizeof(remaining), cmd_aux);
761
762 Cmd_ExecuteString (remaining);
763
764 }
765
766 Com_EndRedirect ();
767}

769/*
770=================
771SV_ConnectionlessPacket
772
773A connectionless packet has four leading 0xff
774characters to distinguish it from a game channel.
775Clients that are in the game can still send
776connectionless packets.
777=================
778*/
779static void SV_ConnectionlessPacket( netadr_t from, msg_t *msg ) {
780 char *s;
781 char *c;
782
783 MSG_BeginReadingOOB( msg );
784 MSG_ReadLong( msg ); // skip the -1 marker
785
786 if (!Q_strncmp("connect", (char *) &msg->data[4], 7)) {
787 Huff_Decompress(msg, 12);
788 }
789
790 s = MSG_ReadStringLine( msg );
791 Cmd_TokenizeString( s );
792
793 c = Cmd_Argv(0);
794 Com_DPrintf ("SV packet %s : %s\n", NET_AdrToString(from), c);
795
796 if (!Q_stricmp(c, "getstatus")) {
797 SVC_Status( from );
798 } else if (!Q_stricmp(c, "getinfo")) {
799 SVC_Info( from );
800 } else if (!Q_stricmp(c, "getchallenge")) {
801 SV_GetChallenge(from);
802 } else if (!Q_stricmp(c, "connect")) {
803 SV_DirectConnect( from );
804#ifndef STANDALONE
805 } else if (!Q_stricmp(c, "ipAuthorize")) {
806 SV_AuthorizeIpPacket( from );
807#endif
808 } else if (!Q_stricmp(c, "rcon")) {
809 SVC_RemoteCommand( from, msg );
810 } else if (!Q_stricmp(c, "disconnect")) {
811 // if a client starts up a local server, we may see some spurious
812 // server disconnect messages when their new server sees our final
813 // sequenced messages to the old client
814 } else {
815 Com_DPrintf ("bad connectionless packet from %s:\n%s\n",
816 NET_AdrToString (from), s);
817 }
818}

822/*
823=================
824SV_PacketEvent
825=================
826*/
827void SV_PacketEvent( netadr_t from, msg_t *msg ) {
828 int i;
829 client_t *cl;
830 int qport;
831
832 // check for connectionless packet (0xffffffff) first
833 if ( msg->cursize >= 4 && *(int *)msg->data == -1) {
834 SV_ConnectionlessPacket( from, msg );
835 return;
836 }
837
838 // read the qport out of the message so we can fix up
839 // stupid address translating routers
840 MSG_BeginReadingOOB( msg );
841 MSG_ReadLong( msg ); // sequence number
842 qport = MSG_ReadShort( msg ) & 0xffff;
843
844 // find which client the message is from
845 for (i=0, cl=svs.clients ; i < sv_maxclients->integer ; i++,cl++) {
846 if (cl->state == CS_FREE) {
847 continue;
848 }
849 if ( !NET_CompareBaseAdr( from, cl->netchan.remoteAddress ) ) {
850 continue;
851 }
852 // it is possible to have multiple clients from a single IP
853 // address, so they are differentiated by the qport variable
854 if (cl->netchan.qport != qport) {
855 continue;
856 }
857
858 // the IP port can't be used to differentiate them, because
859 // some address translating routers periodically change UDP
860 // port assignments
861 if (cl->netchan.remoteAddress.port != from.port) {
862 Com_Printf( "SV_PacketEvent: fixing up a translated port\n" );
863 cl->netchan.remoteAddress.port = from.port;
864 }
865
866 // make sure it is a valid, in sequence packet
867 if (SV_Netchan_Process(cl, msg)) {
868 // zombie clients still need to do the Netchan_Process
869 // to make sure they don't need to retransmit the final
870 // reliable message, but they don't do any other processing
871 if (cl->state != CS_ZOMBIE) {
872 cl->lastPacketTime = svs.time; // don't timeout
873 SV_ExecuteClientMessage( cl, msg );
874 }
875 }
876 return;
877 }
878}

881/*
882===================
883SV_CalcPings
884
885Updates the cl->ping variables
886===================
887*/
888static void SV_CalcPings( void ) {
889 int i, j;
890 client_t *cl;
891 int total, count;
892 int delta;
893 playerState_t *ps;
894
895 for (i=0 ; i < sv_maxclients->integer ; i++) {
896 cl = &svs.clients[i];
897 if ( cl->state != CS_ACTIVE ) {
898 cl->ping = 999;
899 continue;
900 }
901 if ( !cl->gentity ) {
902 cl->ping = 999;
903 continue;
904 }
905 if ( cl->gentity->r.svFlags & SVF_BOT ) {
906 cl->ping = 0;
907 continue;
908 }
909
910 total = 0;
911 count = 0;
912 for ( j = 0 ; j < PACKET_BACKUP ; j++ ) {
913 if ( cl->frames[j].messageAcked <= 0 ) {
914 continue;
915 }
916 delta = cl->frames[j].messageAcked - cl->frames[j].messageSent;
917 count++;
918 total += delta;
919 }
920 if (!count) {
921 cl->ping = 999;
922 } else {
923 cl->ping = total/count;
924 if ( cl->ping > 999 ) {
925 cl->ping = 999;
926 }
927 }
928
929 // let the game dll know about the ping
930 ps = SV_GameClientNum( i );
931 ps->ping = cl->ping;
932 }
933}

935/*
936==================
937SV_CheckTimeouts
938
939If a packet has not been received from a client for timeout->integer
940seconds, drop the conneciton. Server time is used instead of
941realtime to avoid dropping the local client while debugging.
942
943When a client is normally dropped, the client_t goes into a zombie state
944for a few seconds to make sure any final reliable message gets resent
945if necessary
946==================
947*/
948static void SV_CheckTimeouts( void ) {
949 int i;
950 client_t *cl;
951 int droppoint;
952 int zombiepoint;
953
954 droppoint = svs.time - 1000 * sv_timeout->integer;
955 zombiepoint = svs.time - 1000 * sv_zombietime->integer;
956
957 for (i=0,cl=svs.clients ; i < sv_maxclients->integer ; i++,cl++) {
958 // message times may be wrong across a changelevel
959 if (cl->lastPacketTime > svs.time) {
960 cl->lastPacketTime = svs.time;
961 }
962
963 if (cl->state == CS_ZOMBIE
964 && cl->lastPacketTime < zombiepoint) {
965 // using the client id cause the cl->name is empty at this point
966 Com_DPrintf( "Going from CS_ZOMBIE to CS_FREE for client %d\n", i );
967 cl->state = CS_FREE; // can now be reused
968 continue;
969 }
970 if ( cl->state >= CS_CONNECTED && cl->lastPacketTime < droppoint) {
971 // wait several frames so a debugger session doesn't
972 // cause a timeout
973 if ( ++cl->timeoutCount > 5 ) {
974 SV_DropClient (cl, "timed out");
975 cl->state = CS_FREE; // don't bother with zombie state
976 }
977 } else {
978 cl->timeoutCount = 0;
979 }
980 }
981}

985==================
986SV_CheckPaused
987==================

989static qboolean SV_CheckPaused( void ) {
990 int count;
991 client_t *cl;
992 int i;
993
994 if ( !cl_paused->integer ) {
995 return qfalse;
996 }
997
998 // only pause if there is just a single client connected
999 count = 0;
1000 for (i=0,cl=svs.clients ; i < sv_maxclients->integer ; i++,cl++) {
1001 if ( cl->state >= CS_CONNECTED && cl->netchan.remoteAddress.type != NA_BOT ) {
1002 count++;
1003 }
1004 }
1005
1006 if ( count > 1 ) {
1007 // don't pause
1008 if (sv_paused->integer)
1009 Cvar_Set("sv_paused", "0");
1010 return qfalse;
1011 }
1012
1013 if (!sv_paused->integer)
1014 Cvar_Set("sv_paused", "1");
1015 return qtrue;

1041/*
1042==================
1043SV_Frame
1044
1045Player movement occurs as a result of packet events, which
1046happen before SV_Frame is called
1047==================
1048*/
1049void SV_Frame( int msec ) {
1050 int frameMsec;
1051 int startTime;
1052
1053 // the menu kills the server with this cvar
1054 if ( sv_killserver->integer ) {
1055 SV_Shutdown ("Server was killed");
1056 Cvar_Set( "sv_killserver", "0" );
1057 return;
1058 }
1059
1060 if (!com_sv_running->integer)
1061 {
1062 // Running as a server, but no map loaded
1063#ifdef DEDICATED
1064 // Block until something interesting happens
1065 Sys_Sleep(-1);
1066#endif
1067
1068 return;
1069 }
1070
1071 // allow pause if only the local client is connected
1072 if ( SV_CheckPaused() ) {
1073 return;
1074 }
1075
1076 // if it isn't time for the next frame, do nothing
1077 if ( sv_fps->integer < 1 ) {
1078 Cvar_Set( "sv_fps", "10" );
1079 }
1080
1081 frameMsec = 1000 / sv_fps->integer * com_timescale->value;
1082 // don't let it scale below 1ms
1083 if(frameMsec < 1)
1084 {
1085 Cvar_Set("timescale", va("%f", sv_fps->integer / 1000.0f));
1086 frameMsec = 1;
1087 }
1088
1089 sv.timeResidual += msec;
1090
1091 if (!com_dedicated->integer) SV_BotFrame (sv.time + sv.timeResidual);
1092
1093 // if time is about to hit the 32nd bit, kick all clients
1094 // and clear sv.time, rather
1095 // than checking for negative time wraparound everywhere.
1096 // 2giga-milliseconds = 23 days, so it won't be too often
1097 if ( svs.time > 0x70000000 ) {
1098 SV_Shutdown( "Restarting server due to time wrapping" );
1099 Cbuf_AddText( va( "map %s\n", Cvar_VariableString( "mapname" ) ) );
1100 return;
1101 }
1102 // this can happen considerably earlier when lots of clients play and the map doesn't change
1103 if ( svs.nextSnapshotEntities >= 0x7FFFFFFE - svs.numSnapshotEntities ) {
1104 SV_Shutdown( "Restarting server due to numSnapshotEntities wrapping" );
1105 Cbuf_AddText( va( "map %s\n", Cvar_VariableString( "mapname" ) ) );
1106 return;
1107 }
1108
1109 if( sv.restartTime && sv.time >= sv.restartTime ) {
1110 sv.restartTime = 0;
1111 Cbuf_AddText( "map_restart 0\n" );
1112 return;
1113 }
1114
1115 // update infostrings if anything has been changed
1116 if ( cvar_modifiedFlags & CVAR_SERVERINFO ) {
1117 SV_SetConfigstring( CS_SERVERINFO, Cvar_InfoString( CVAR_SERVERINFO ) );
1118 cvar_modifiedFlags &= ~CVAR_SERVERINFO;
1119 }
1120 if ( cvar_modifiedFlags & CVAR_SYSTEMINFO ) {
1121 SV_SetConfigstring( CS_SYSTEMINFO, Cvar_InfoString_Big( CVAR_SYSTEMINFO ) );
1122 cvar_modifiedFlags &= ~CVAR_SYSTEMINFO;
1123 }
1124
1125 if ( com_speeds->integer ) {
1126 startTime = Sys_Milliseconds ();
1127 } else {
1128 startTime = 0; // quite a compiler warning
1129 }
1130
1131 // update ping based on the all received frames
1132 SV_CalcPings();
1133
1134 if (com_dedicated->integer) SV_BotFrame (sv.time);
1135
1136 // run the game simulation in chunks
1137 while ( sv.timeResidual >= frameMsec ) {
1138 sv.timeResidual -= frameMsec;
1139 svs.time += frameMsec;
1140 sv.time += frameMsec;
1141
1142 // let everything in the world think and move
1143 VM_Call (gvm, GAME_RUN_FRAME, sv.time);
1144 }
1145
1146 if ( com_speeds->integer ) {
1147 time_game = Sys_Milliseconds () - startTime;
1148 }
1149
1150 // check timeouts
1151 SV_CheckTimeouts();
1152
1153 // send messages back to the clients
1154 SV_SendClientMessages();
1155
1156 // send a heartbeat to the master if needed
1157 SV_MasterHeartbeat(HEARTBEAT_FOR_MASTER);
1158}

1160/*
1161====================
1162SV_RateMsec
1163
1164Return the number of msec until another message can be sent to
1165a client based on its rate settings
1166====================
1167*/
1168
1169#define UDPIP_HEADER_SIZE 28
1170#define UDPIP6_HEADER_SIZE 48
1171
1172int SV_RateMsec(client_t *client)
1173{
1174 int rate, rateMsec;
1175 int messageSize;
1176
1177 messageSize = client->netchan.lastSentSize;
1178 rate = client->rate;
1179
1180 if(sv_maxRate->integer)
1181 {
1182 if(sv_maxRate->integer < 1000)
1183 Cvar_Set( "sv_MaxRate", "1000" );
1184 if(sv_maxRate->integer < rate)
1185 rate = sv_maxRate->integer;
1186 }
1187
1188 if(sv_minRate->integer)
1189 {
1190 if(sv_minRate->integer < 1000)
1191 Cvar_Set("sv_minRate", "1000");
1192 if(sv_minRate->integer > rate)
1193 rate = sv_minRate->integer;
1194 }
1195
1196 if(client->netchan.remoteAddress.type == NA_IP6)
1197 messageSize += UDPIP6_HEADER_SIZE;
1198 else
1199 messageSize += UDPIP_HEADER_SIZE;
1200
1201 rateMsec = messageSize * 1000 / ((int) (rate * com_timescale->value));
1202 rate = Sys_Milliseconds() - client->netchan.lastSentTime;
1203
1204 if(rate > rateMsec)
1205 return 0;
1206 else
1207 return rateMsec - rate;
1208}

1210/*
1211====================
1212SV_SendQueuedPackets
1213
1214Send download messages and queued packets in the time that we're idle, i.e.
1215not computing a server frame or sending client snapshots.
1216Return the time in msec until we expect to be called next
1217====================
1218*/
1219
1220int SV_SendQueuedPackets()
1221{
1222 int numBlocks;
1223 int dlStart, deltaT, delayT;
1224 static int dlNextRound = 0;
1225 int timeVal = INT_MAX;
1226
1227 // Send out fragmented packets now that we're idle
1228 delayT = SV_SendQueuedMessages();
1229 if(delayT >= 0)
1230 timeVal = delayT;
1231
1232 if(sv_dlRate->integer)
1233 {
1234 // Rate limiting. This is very imprecise for high
1235 // download rates due to millisecond timedelta resolution
1236 dlStart = Sys_Milliseconds();
1237 deltaT = dlNextRound - dlStart;
1238
1239 if(deltaT > 0)
1240 {
1241 if(deltaT < timeVal)
1242 timeVal = deltaT + 1;
1243 }
1244 else
1245 {
1246 numBlocks = SV_SendDownloadMessages();
1247
1248 if(numBlocks)
1249 {
1250 // There are active downloads
1251 deltaT = Sys_Milliseconds() - dlStart;
1252
1253 delayT = 1000 * numBlocks * MAX_DOWNLOAD_BLKSIZE;
1254 delayT /= sv_dlRate->integer * 1024;
1255
1256 if(delayT <= deltaT + 1)
1257 {
1258 // Sending the last round of download messages
1259 // took too long for given rate, don't wait for
1260 // next round, but always enforce a 1ms delay
1261 // between DL message rounds so we don't hog
1262 // all of the bandwidth. This will result in an
1263 // effective maximum rate of 1MB/s per user, but the
1264 // low download window size limits this anyways.
1265 if(timeVal > 2)
1266 timeVal = 2;
1267
1268 dlNextRound = dlStart + deltaT + 1;
1269 }
1270 else
1271 {
1272 dlNextRound = dlStart + delayT;
1273 delayT -= deltaT;
1274
1275 if(delayT < timeVal)
1276 timeVal = delayT;
1277 }
1278 }
1279 }
1280 }
1281 else
1282 {
1283 if(SV_SendDownloadMessages())
1284 timeVal = 0;
1285 }
1286
1287 return timeVal;
1288}

30// these functions must be used instead of pointer arithmetic, because
31// the game allocates gentities with private information after the server shared part
32int SV_NumForGentity( sharedEntity_t *ent ) {
33 int num;
34
35 num = ( (byte *)ent - (byte *)sv.gentities ) / sv.gentitySize;
36
37 return num;
38}
39
40sharedEntity_t *SV_GentityNum( int num ) {
41 sharedEntity_t *ent;
42
43 ent = (sharedEntity_t *)((byte *)sv.gentities + sv.gentitySize*(num));
44
45 return ent;
46}
47
48playerState_t *SV_GameClientNum( int num ) {
49 playerState_t *ps;
50
51 ps = (playerState_t *)((byte *)sv.gameClients + sv.gameClientSize*(num));
52
53 return ps;
54}

56svEntity_t *SV_SvEntityForGentity( sharedEntity_t *gEnt ) {
57 if ( !gEnt || gEnt->s.number < 0 || gEnt->s.number >= MAX_GENTITIES ) {
58 Com_Error( ERR_DROP, "SV_SvEntityForGentity: bad gEnt" );
59 }
60 return &sv.svEntities[ gEnt->s.number ];
61}

63sharedEntity_t *SV_GEntityForSvEntity( svEntity_t *svEnt ) {
64 int num;
65
66 num = svEnt - sv.svEntities;
67 return SV_GentityNum( num );
68}

104/*
105=================
106SV_SetBrushModel
107
108sets mins and maxs for inline bmodels
109=================
110*/
111void SV_SetBrushModel( sharedEntity_t *ent, const char *name ) {
112 clipHandle_t h;
113 vec3_t mins, maxs;
114
115 if (!name) {
116 Com_Error( ERR_DROP, "SV_SetBrushModel: NULL" );
117 }
118
119 if (name[0] != '*') {
120 Com_Error( ERR_DROP, "SV_SetBrushModel: %s isn't a brush model", name );
121 }
122
123
124 ent->s.modelindex = atoi( name + 1 );
125
126 h = CM_InlineModel( ent->s.modelindex );
127 CM_ModelBounds( h, mins, maxs );
128 VectorCopy (mins, ent->r.mins);
129 VectorCopy (maxs, ent->r.maxs);
130 ent->r.bmodel = qtrue;
131
132 ent->r.contents = -1; // we don't know exactly what is in the brushes
133
134 SV_LinkEntity( ent ); // FIXME: remove
135}

139/*
140=================
141SV_inPVS
142
143Also checks portalareas so that doors block sight
144=================
145*/
146qboolean SV_inPVS (const vec3_t p1, const vec3_t p2)
147{
148 int leafnum;
149 int cluster;
150 int area1, area2;
151 byte *mask;
152
153 leafnum = CM_PointLeafnum (p1);
154 cluster = CM_LeafCluster (leafnum);
155 area1 = CM_LeafArea (leafnum);
156 mask = CM_ClusterPVS (cluster);
157
158 leafnum = CM_PointLeafnum (p2);
159 cluster = CM_LeafCluster (leafnum);
160 area2 = CM_LeafArea (leafnum);
161 if ( mask && (!(mask[cluster>>3] & (1<<(cluster&7)) ) ) )
162 return qfalse;
163 if (!CM_AreasConnected (area1, area2))
164 return qfalse; // a door blocks sight
165 return qtrue;
166}

176qboolean SV_inPVSIgnorePortals( const vec3_t p1, const vec3_t p2)

178 int leafnum;
179 int cluster;
180 byte *mask;
181
182 leafnum = CM_PointLeafnum (p1);
183 cluster = CM_LeafCluster (leafnum);
184 mask = CM_ClusterPVS (cluster);

187 cluster = CM_LeafCluster (leafnum);
188
189 if ( mask && (!(mask[cluster>>3] & (1<<(cluster&7)) ) ) )
190 return qfalse;
191
192 return qtrue;
193}

212/*
213==================
214SV_EntityContact
215==================
216*/
217qboolean SV_EntityContact( vec3_t mins, vec3_t maxs, const sharedEntity_t *gEnt, int capsule ) {
218 const float *origin, *angles;
219 clipHandle_t ch;
220 trace_t trace;
221
222 // check for exact collision
223 origin = gEnt->r.currentOrigin;
224 angles = gEnt->r.currentAngles;
225
226 ch = SV_ClipHandleForEntity( gEnt );
227 CM_TransformedBoxTrace ( &trace, vec3_origin, vec3_origin, mins, maxs,
228 ch, -1, origin, angles, capsule );
229
230 return trace.startsolid;
231}

868/*
869==================
870SV_InitGameVM
871
872Called for both a full init and a restart
873==================
874*/
875static void SV_InitGameVM( qboolean restart ) {
876 int i;
877
878 // start the entity parsing at the beginning
879 sv.entityParsePoint = CM_EntityString();
880
881 // clear all gentity pointers that might still be set from
882 // a previous level
883 // https://zerowing.idsoftware.com/bugzilla/show_bug.cgi?id=522
884 // now done before GAME_INIT call
885 for ( i = 0 ; i < sv_maxclients->integer ; i++ ) {
886 svs.clients[i].gentity = NULL;
887 }
888
889 // use the current msec count for a random seed
890 // init for this gamestate
891 VM_Call (gvm, GAME_INIT, sv.time, Com_Milliseconds(), restart);
892}

915 SV_InitGameVM( qtrue );
916}

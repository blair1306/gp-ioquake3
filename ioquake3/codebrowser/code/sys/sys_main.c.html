130/*
131==================
132Sys_GetClipboardData
133==================
134*/
135char *Sys_GetClipboardData(void)
136{
137#ifdef DEDICATED
138 return NULL;
139#else
140 char *data = NULL;
141 char *cliptext;
142
143 if ( ( cliptext = SDL_GetClipboardText() ) != NULL ) {
144 if ( cliptext[0] != '\0' ) {
145 size_t bufsize = strlen( cliptext ) + 1;
146
147 data = Z_Malloc( bufsize );
148 Q_strncpyz( data, cliptext, bufsize );
149
150 // find first listed char and set to '\0'
151 strtok( data, "\n\r\b" );
152 }
153 SDL_free( cliptext );
154 }
155 return data;
156#endif
157}

170static char *Sys_PIDFileName( const char *gamedir )
171{
172 const char *homePath = Cvar_VariableString( "fs_homepath" );
173
174 if( *homePath != '\0' )
175 return va( "%s/%s/%s", homePath, gamedir, PID_FILENAME );

193/*
194=================
195Sys_WritePIDFile
196
197Return qtrue if there is an existing stale PID file
198=================
199*/
200static qboolean Sys_WritePIDFile( const char *gamedir )
201{
202 char *pidFile = Sys_PIDFileName( gamedir );
203 FILE *f;
204 qboolean stale = qfalse;
205
206 if( pidFile == NULL )
207 return qfalse;
208
209 // First, check if the pid file is already there
210 if( ( f = fopen( pidFile, "r" ) ) != NULL )
211 {
212 char pidBuffer[ 64 ] = { 0 };
213 int pid;
214
215 pid = fread( pidBuffer, sizeof( char ), sizeof( pidBuffer ) - 1, f );
216 fclose( f );
217
218 if(pid > 0)
219 {
220 pid = atoi( pidBuffer );
221 if( !Sys_PIDIsRunning( pid ) )
222 stale = qtrue;
223 }
224 else
225 stale = qtrue;
226 }
227
228 if( FS_CreatePath( pidFile ) ) {
229 return 0;
230 }
231
232 if( ( f = fopen( pidFile, "w" ) ) != NULL )
233 {
234 fprintf( f, "%d", Sys_PID( ) );
235 fclose( f );
236 }
237 else
238 Com_Printf( S_COLOR_YELLOW "Couldn't write %s.\n", pidFile );
239
240 return stale;
241}

276{

284 {
285 // Normal exit

288

291 Sys_PlatformExit( );
292

294}

311cpuFeatures_t Sys_GetProcessorFeatures( void )
312{
313 cpuFeatures_t features = 0;
314
315#ifndef DEDICATED
316 if( SDL_HasRDTSC( ) ) features |= CF_RDTSC;
317 if( SDL_Has3DNow( ) ) features |= CF_3DNOW;
318 if( SDL_HasMMX( ) ) features |= CF_MMX;
319 if( SDL_HasSSE( ) ) features |= CF_SSE;
320 if( SDL_HasSSE2( ) ) features |= CF_SSE2;
321 if( SDL_HasAltiVec( ) ) features |= CF_ALTIVEC;
322#endif
323
324 return features;
325}

339/*
340=================
341Sys_AnsiColorPrint
342
343Transform Q3 colour codes to ANSI escape sequences
344=================
345*/
346void Sys_AnsiColorPrint( const char *msg )
347{
348 static char buffer[ MAXPRINTMSG ];
349 int length = 0;
350 static int q3ToAnsi[ 8 ] =
351 {
352 30, // COLOR_BLACK
353 31, // COLOR_RED
354 32, // COLOR_GREEN
355 33, // COLOR_YELLOW
356 34, // COLOR_BLUE
357 36, // COLOR_CYAN
358 35, // COLOR_MAGENTA
359 0 // COLOR_WHITE
360 };
361
362 while( *msg )
363 {
364 if( Q_IsColorString( msg ) || *msg == '\n' )
365 {
366 // First empty the buffer
367 if( length > 0 )
368 {
369 buffer[ length ] = '\0';
370 fputs( buffer, stderr );
371 length = 0;
372 }
373
374 if( *msg == '\n' )
375 {
376 // Issue a reset and then the newline
377 fputs( "\033[0m\n", stderr );
378 msg++;
379 }
380 else
381 {
382 // Print the color code
383 Com_sprintf( buffer, sizeof( buffer ), "\033[%dm",
384 q3ToAnsi[ ColorIndex( *( msg + 1 ) ) ] );
385 fputs( buffer, stderr );
386 msg += 2;
387 }
388 }
389 else
390 {
391 if( length >= MAXPRINTMSG - 1 )
392 break;
393
394 buffer[ length ] = *msg;
395 length++;
396 msg++;
397 }
398 }
399
400 // Empty anything still left in the buffer
401 if( length > 0 )
402 {
403 buffer[ length ] = '\0';
404 fputs( buffer, stderr );
405 }
406}

490/*
491=================
492Sys_LoadDll
493
494First try to load library name from system library path,
495from executable path, then fs_basepath.
496=================
497*/
498
499void *Sys_LoadDll(const char *name, qboolean useSystemLib)
500{
501 void *dllhandle = NULL;
502
503 if(!Sys_DllExtension(name))
504 {
505 Com_Printf("Refusing to attempt to load library \"%s\": Extension not allowed.\n", name);
506 return NULL;
507 }
508
509 if(useSystemLib)
510 {
511 Com_Printf("Trying to load \"%s\"...\n", name);
512 dllhandle = Sys_LoadLibrary(name);
513 }
514
515 if(!dllhandle)
516 {
517 const char *topDir;
518 char libPath[MAX_OSPATH];
519 int len;
520
521 topDir = Sys_BinaryPath();
522
523 if(!*topDir)
524 topDir = ".";
525
526 len = Com_sprintf(libPath, sizeof(libPath), "%s%c%s", topDir, PATH_SEP, name);
527 if(len < sizeof(libPath))
528 {
529 Com_Printf("Trying to load \"%s\" from \"%s\"...\n", name, topDir);
530 dllhandle = Sys_LoadLibrary(libPath);
531 }
532 else
533 {
534 Com_Printf("Skipping trying to load \"%s\" from \"%s\", file name is too long.\n", name, topDir);
535 }
536
537 if(!dllhandle)
538 {
539 const char *basePath = Cvar_VariableString("fs_basepath");
540
541 if(!basePath || !*basePath)
542 basePath = ".";
543
544 if(FS_FilenameCompare(topDir, basePath))
545 {
546 len = Com_sprintf(libPath, sizeof(libPath), "%s%c%s", basePath, PATH_SEP, name);
547 if(len < sizeof(libPath))
548 {
549 Com_Printf("Trying to load \"%s\" from \"%s\"...\n", name, basePath);
550 dllhandle = Sys_LoadLibrary(libPath);
551 }
552 else
553 {
554 Com_Printf("Skipping trying to load \"%s\" from \"%s\", file name is too long.\n", name, basePath);
555 }
556 }
557
558 if(!dllhandle)
559 Com_Printf("Loading \"%s\" failed\n", name);
560 }
561 }
562
563 return dllhandle;
564}

566/*
567=================
568Sys_LoadGameDll
569
570Used to load a development dll instead of a virtual machine
571=================
572*/
573void *Sys_LoadGameDll(const char *name,
574 intptr_t (QDECL **entryPoint)(int, ...),
575 intptr_t (*systemcalls)(intptr_t, ...))
576{
577 void *libHandle;
578 void (*dllEntry)(intptr_t (*syscallptr)(intptr_t, ...));
579
580 assert(name);
581
582 if(!Sys_DllExtension(name))
583 {
584 Com_Printf("Refusing to attempt to load library \"%s\": Extension not allowed.\n", name);
585 return NULL;
586 }
587
588 Com_Printf( "Loading DLL file: %s\n", name);
589 libHandle = Sys_LoadLibrary(name);
590
591 if(!libHandle)
592 {
593 Com_Printf("Sys_LoadGameDll(%s) failed:\n\"%s\"\n", name, Sys_LibraryError());
594 return NULL;
595 }
596
597 dllEntry = Sys_LoadFunction( libHandle, "dllEntry" );
598 *entryPoint = Sys_LoadFunction( libHandle, "vmMain" );
599
600 if ( !*entryPoint || !dllEntry )
601 {
602 Com_Printf ( "Sys_LoadGameDll(%s) failed to find vmMain function:\n\"%s\" !\n", name, Sys_LibraryError( ) );
603 Sys_UnloadLibrary(libHandle);
604
605 return NULL;
606 }
607
608 Com_Printf ( "Sys_LoadGameDll(%s) found vmMain function at %p\n", name, *entryPoint );
609 dllEntry( systemcalls );
610
611 return libHandle;
612}

645/*
646=================
647Sys_SigHandler
648=================
649*/
650void Sys_SigHandler( int signal )
651{
652 static qboolean signalcaught = qfalse;
653
654 if( signalcaught )
655 {
656 fprintf( stderr, "DOUBLE SIGNAL FAULT: Received signal %d, exiting...\n",
657 signal );
658 }
659 else
660 {
661 signalcaught = qtrue;
662 VM_Forced_Unload_Start();
663#ifndef DEDICATED
664 CL_Shutdown(va("Received signal %d", signal), qtrue, qtrue);
665#endif
666 SV_Shutdown(va("Received signal %d", signal) );
667 VM_Forced_Unload_Done();
668 }
669
670 if( signal == SIGTERM || signal == SIGINT )
671 Sys_Exit( 1 );
672 else
673 Sys_Exit( 2 );
674}

676/*
677=================
678main
679=================
680*/
681int main( int argc, char **argv )
682{
683 int i;
684 char commandLine[ MAX_STRING_CHARS ] = { 0 };
685
686 extern void Sys_LaunchAutoupdater(int argc, char **argv);
687 Sys_LaunchAutoupdater(argc, argv);
688
689#ifndef DEDICATED
690 // SDL version check
691
692 // Compile time
693# if !SDL_VERSION_ATLEAST(MINSDL_MAJOR,MINSDL_MINOR,MINSDL_PATCH)
694# error A more recent version of SDL is required
695# endif
696
697 // Run time
698 SDL_version ver;
699 SDL_GetVersion( &ver );
700
701#define MINSDL_VERSION \
702 XSTRING(MINSDL_MAJOR) "." \
703 XSTRING(MINSDL_MINOR) "." \
704 XSTRING(MINSDL_PATCH)
705
706 if( SDL_VERSIONNUM( ver.major, ver.minor, ver.patch ) <
707 SDL_VERSIONNUM( MINSDL_MAJOR, MINSDL_MINOR, MINSDL_PATCH ) )
708 {
709 Sys_Dialog( DT_ERROR, va( "SDL version " MINSDL_VERSION " or greater is required, "
710 "but only version %d.%d.%d was found. You may be able to obtain a more recent copy "
711 "from http://www.libsdl.org/.", ver.major, ver.minor, ver.patch ), "SDL Library Too Old" );
712
713 Sys_Exit( 1 );
714 }
715#endif
716
717 Sys_PlatformInit( );
718
719 // Set the initial time base
720 Sys_Milliseconds( );
721
722#ifdef __APPLE__
723 // This is passed if we are launched by double-clicking
724 if ( argc >= 2 && Q_strncmp ( argv[1], "-psn", 4 ) == 0 )
725 argc = 1;
726#endif
727
728 Sys_ParseArgs( argc, argv );
729 Sys_SetBinaryPath( Sys_Dirname( argv[ 0 ] ) );
730 Sys_SetDefaultInstallPath( DEFAULT_BASEDIR );
731
732 // Concatenate the command line for passing to Com_Init
733 for( i = 1; i < argc; i++ )
734 {
735 const qboolean containsSpaces = strchr(argv[i], ' ') != NULL;
736 if (containsSpaces)
737 Q_strcat( commandLine, sizeof( commandLine ), "\"" );
738
739 Q_strcat( commandLine, sizeof( commandLine ), argv[ i ] );
740
741 if (containsSpaces)
742 Q_strcat( commandLine, sizeof( commandLine ), "\"" );
743
744 Q_strcat( commandLine, sizeof( commandLine ), " " );
745 }
746
747 CON_Init( );
748 Com_Init( commandLine );
749 NET_Init( );
750
751 signal( SIGILL, Sys_SigHandler );
752 signal( SIGFPE, Sys_SigHandler );
753 signal( SIGSEGV, Sys_SigHandler );
754 signal( SIGTERM, Sys_SigHandler );
755 signal( SIGINT, Sys_SigHandler );
756
757 while( 1 )
758 {
759 Com_Frame( );
760 }
761
762 return 0;
763}

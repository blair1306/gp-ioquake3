37/*
38=============================================================
39tty console routines
40
41NOTE: if the user is editing a line when something gets printed to the early
42console then it won't look good so we provide CON_Hide and CON_Show to be
43called before and after a stdout or stderr output
44=============================================================
45*/
46
47extern qboolean stdinIsATTY;
48static qboolean stdin_active;
49// general flag to tell about tty console mode
50static qboolean ttycon_on = qfalse;
51static int ttycon_hide = 0;
52static int ttycon_show_overdue = 0;
53
54// some key codes that the terminal may be using, initialised on start up
55static int TTY_erase;
56static int TTY_eof;
57
58static struct termios TTY_tc;
59
60static field_t TTY_con;
61
62// This is somewhat of aduplicate of the graphical console history
63// but it's safer more modular to have our own here
64#define CON_HISTORY 32
65static field_t ttyEditLines[ CON_HISTORY ];
66static int hist_current = -1, hist_count = 0;
67
68#ifndef DEDICATED
69// Don't use "]" as it would be the same as in-game console,
70// this makes it clear where input came from.
71#define TTY_CONSOLE_PROMPT "tty]"
72#else
73#define TTY_CONSOLE_PROMPT "]"
74#endif
75
76/*
77==================
78CON_FlushIn
79
80Flush stdin, I suspect some terminals are sending a LOT of shit
81FIXME relevant?
82==================
83*/
84static void CON_FlushIn( void )
85{
86 char key;
87 while (read(STDIN_FILENO, &key, 1)!=-1);
88}
89
90/*
91==================
92CON_Back
93
94Output a backspace
95
96NOTE: it seems on some terminals just sending '\b' is not enough so instead we
97send "\b \b"
98(FIXME there may be a way to find out if '\b' alone would work though)
99==================
100*/
101static void CON_Back( void )
102{
103 char key;
104 size_t UNUSED_VAR size;
105
106 key = '\b';
107 size = write(STDOUT_FILENO, &key, 1);
108 key = ' ';
109 size = write(STDOUT_FILENO, &key, 1);
110 key = '\b';
111 size = write(STDOUT_FILENO, &key, 1);
112}
113
114/*
115==================
116CON_Hide
117
118Clear the display of the line currently edited
119bring cursor back to beginning of line
120==================
121*/
122static void CON_Hide( void )
123{
124 if( ttycon_on )
125 {
126 int i;
127 if (ttycon_hide)
128 {
129 ttycon_hide++;
130 return;
131 }
132 if (TTY_con.cursor>0)
133 {
134 for (i=0; i<TTY_con.cursor; i++)
135 {
136 CON_Back();
137 }
138 }
139 // Delete prompt
140 for (i = strlen(TTY_CONSOLE_PROMPT); i > 0; i--) {
141 CON_Back();
142 }
143 ttycon_hide++;
144 }
145}
146
147/*
148==================
149CON_Show
150
151Show the current line
152FIXME need to position the cursor if needed?
153==================
154*/
155static void CON_Show( void )
156{
157 if( ttycon_on )
158 {
159 int i;
160
161 assert(ttycon_hide>0);
162 ttycon_hide--;
163 if (ttycon_hide == 0)
164 {
165 size_t UNUSED_VAR size;
166 size = write(STDOUT_FILENO, TTY_CONSOLE_PROMPT, strlen(TTY_CONSOLE_PROMPT));
167 if (TTY_con.cursor)
168 {
169 for (i=0; i<TTY_con.cursor; i++)
170 {
171 size = write(STDOUT_FILENO, TTY_con.buffer+i, 1);
172 }
173 }
174 }
175 }
176}

178/*
179==================
180CON_Shutdown
181
182Never exit without calling this, or your terminal will be left in a pretty bad state
183==================
184*/
185void CON_Shutdown( void )
186{
187 if (ttycon_on)
188 {
189 CON_Hide();
190 tcsetattr (STDIN_FILENO, TCSADRAIN, &TTY_tc);
191 }
192
193 // Restore blocking to stdin reads
194 fcntl(STDIN_FILENO, F_SETFL, fcntl(STDIN_FILENO, F_GETFL, 0) & ~O_NONBLOCK);
195}

197/*
198==================
199Hist_Add
200==================
201*/
202void Hist_Add(field_t *field)
203{
204 int i;
205
206 // Don't save blank lines in history.
207 if (!field->cursor)
208 return;
209
210 assert(hist_count <= CON_HISTORY);
211 assert(hist_count >= 0);
212 assert(hist_current >= -1);
213 assert(hist_current <= hist_count);
214 // make some room
215 for (i=CON_HISTORY-1; i>0; i--)
216 {
217 ttyEditLines[i] = ttyEditLines[i-1];
218 }
219 ttyEditLines[0] = *field;
220 if (hist_count<CON_HISTORY)
221 {
222 hist_count++;
223 }
224 hist_current = -1; // re-init
225}
226
227/*
228==================
229Hist_Prev
230==================
231*/
232field_t *Hist_Prev( void )
233{
234 int hist_prev;
235 assert(hist_count <= CON_HISTORY);
236 assert(hist_count >= 0);
237 assert(hist_current >= -1);
238 assert(hist_current <= hist_count);
239 hist_prev = hist_current + 1;
240 if (hist_prev >= hist_count)
241 {
242 return NULL;
243 }
244 hist_current++;
245 return &(ttyEditLines[hist_current]);
246}
247
248/*
249==================
250Hist_Next
251==================
252*/
253field_t *Hist_Next( void )
254{
255 assert(hist_count <= CON_HISTORY);
256 assert(hist_count >= 0);
257 assert(hist_current >= -1);
258 assert(hist_current <= hist_count);
259 if (hist_current >= 0)
260 {
261 hist_current--;
262 }
263 if (hist_current == -1)
264 {
265 return NULL;
266 }
267 return &(ttyEditLines[hist_current]);
268}

271==================
272CON_SigCont
273Reinitialize console input after receiving SIGCONT, as on Linux the terminal seems to lose all
274set attributes if user did CTRL+Z and then does fg again.
275==================
276*/
277
278void CON_SigCont(int signum)
279{
280 CON_Init();
281}

283/*
284==================
285CON_Init
286
287Initialize the console input (tty mode if possible)
288==================
289*/
290void CON_Init( void )
291{
292 struct termios tc;
293
294 // If the process is backgrounded (running non interactively)
295 // then SIGTTIN or SIGTOU is emitted, if not caught, turns into a SIGSTP
296 signal(SIGTTIN, SIG_IGN);
297 signal(SIGTTOU, SIG_IGN);
298
299 // If SIGCONT is received, reinitialize console
300 signal(SIGCONT, CON_SigCont);
301
302 // Make stdin reads non-blocking
303 fcntl(STDIN_FILENO, F_SETFL, fcntl(STDIN_FILENO, F_GETFL, 0) | O_NONBLOCK );
304
305 if (!stdinIsATTY)
306 {
307 Com_Printf("tty console mode disabled\n");
308 ttycon_on = qfalse;
309 stdin_active = qtrue;
310 return;
311 }
312
313 Field_Clear(&TTY_con);
314 tcgetattr (STDIN_FILENO, &TTY_tc);
315 TTY_erase = TTY_tc.c_cc[VERASE];
316 TTY_eof = TTY_tc.c_cc[VEOF];
317 tc = TTY_tc;
318
319 /*
320 ECHO: don't echo input characters
321 ICANON: enable canonical mode. This enables the special
322 characters EOF, EOL, EOL2, ERASE, KILL, REPRINT,
323 STATUS, and WERASE, and buffers by lines.
324 ISIG: when any of the characters INTR, QUIT, SUSP, or
325 DSUSP are received, generate the corresponding signal
326 */
327 tc.c_lflag &= ~(ECHO | ICANON);
328
329 /*
330 ISTRIP strip off bit 8
331 INPCK enable input parity checking
332 */
333 tc.c_iflag &= ~(ISTRIP | INPCK);
334 tc.c_cc[VMIN] = 1;
335 tc.c_cc[VTIME] = 0;
336 tcsetattr (STDIN_FILENO, TCSADRAIN, &tc);
337 ttycon_on = qtrue;
338 ttycon_hide = 1; // Mark as hidden, so prompt is shown in CON_Show
339 CON_Show();
340}

342/*
343==================
344CON_Input
345==================
346*/
347char *CON_Input( void )
348{
349 // we use this when sending back commands
350 static char text[MAX_EDIT_LINE];
351 int avail;
352 char key;
353 field_t *history;
354 size_t UNUSED_VAR size;
355
356 if(ttycon_on)
357 {
358 avail = read(STDIN_FILENO, &key, 1);
359 if (avail != -1)
360 {
361 // we have something
362 // backspace?
363 // NOTE TTimo testing a lot of values .. seems it's the only way to get it to work everywhere
364 if ((key == TTY_erase) || (key == 127) || (key == 8))
365 {
366 if (TTY_con.cursor > 0)
367 {
368 TTY_con.cursor--;
369 TTY_con.buffer[TTY_con.cursor] = '\0';
370 CON_Back();
371 }
372 return NULL;
373 }
374 // check if this is a control char
375 if ((key) && (key) < ' ')
376 {
377 if (key == '\n')
378 {
379#ifndef DEDICATED
380 // if not in the game explicitly prepend a slash if needed
381 if (clc.state != CA_ACTIVE && con_autochat->integer && TTY_con.cursor &&
382 TTY_con.buffer[0] != '/' && TTY_con.buffer[0] != '\\')
383 {
384 memmove(TTY_con.buffer + 1, TTY_con.buffer, sizeof(TTY_con.buffer) - 1);
385 TTY_con.buffer[0] = '\\';
386 TTY_con.cursor++;
387 }
388
389 if (TTY_con.buffer[0] == '/' || TTY_con.buffer[0] == '\\') {
390 Q_strncpyz(text, TTY_con.buffer + 1, sizeof(text));
391 } else if (TTY_con.cursor) {
392 if (con_autochat->integer) {
393 Com_sprintf(text, sizeof(text), "cmd say %s", TTY_con.buffer);
394 } else {
395 Q_strncpyz(text, TTY_con.buffer, sizeof(text));
396 }
397 } else {
398 text[0] = '\0';
399 }
400
401 // push it in history
402 Hist_Add(&TTY_con);
403 CON_Hide();
404 Com_Printf("%s%s\n", TTY_CONSOLE_PROMPT, TTY_con.buffer);
405 Field_Clear(&TTY_con);
406 CON_Show();
407#else
408 // push it in history
409 Hist_Add(&TTY_con);
410 Q_strncpyz(text, TTY_con.buffer, sizeof(text));
411 Field_Clear(&TTY_con);
412 key = '\n';
413 size = write(STDOUT_FILENO, &key, 1);
414 size = write(STDOUT_FILENO, TTY_CONSOLE_PROMPT, strlen(TTY_CONSOLE_PROMPT));
415#endif
416 return text;
417 }
418 if (key == '\t')
419 {
420 CON_Hide();
421 Field_AutoComplete( &TTY_con );
422 CON_Show();
423 return NULL;
424 }
425 avail = read(STDIN_FILENO, &key, 1);
426 if (avail != -1)
427 {
428 // VT 100 keys
429 if (key == '[' || key == 'O')
430 {
431 avail = read(STDIN_FILENO, &key, 1);
432 if (avail != -1)
433 {
434 switch (key)
435 {
436 case 'A':
437 history = Hist_Prev();
438 if (history)
439 {
440 CON_Hide();
441 TTY_con = *history;
442 CON_Show();
443 }
444 CON_FlushIn();
445 return NULL;
446 break;
447 case 'B':
448 history = Hist_Next();
449 CON_Hide();
450 if (history)
451 {
452 TTY_con = *history;
453 } else
454 {
455 Field_Clear(&TTY_con);
456 }
457 CON_Show();
458 CON_FlushIn();
459 return NULL;
460 break;
461 case 'C':
462 return NULL;
463 case 'D':
464 return NULL;
465 }
466 }
467 }
468 }
469 Com_DPrintf("droping ISCTL sequence: %d, TTY_erase: %d\n", key, TTY_erase);
470 CON_FlushIn();
471 return NULL;
472 }
473 if (TTY_con.cursor >= sizeof(text) - 1)
474 return NULL;
475 // push regular character
476 TTY_con.buffer[TTY_con.cursor] = key;
477 TTY_con.cursor++; // next char will always be '\0'
478 // print the current line (this is differential)
479 size = write(STDOUT_FILENO, &key, 1);
480 }
481
482 return NULL;
483 }
484 else if (stdin_active)
485 {
486 int len;
487 fd_set fdset;
488 struct timeval timeout;
489
490 FD_ZERO(&fdset);
491 FD_SET(STDIN_FILENO, &fdset); // stdin
492 timeout.tv_sec = 0;
493 timeout.tv_usec = 0;
494 if(select (STDIN_FILENO + 1, &fdset, NULL, NULL, &timeout) == -1 || !FD_ISSET(STDIN_FILENO, &fdset))
495 return NULL;
496
497 len = read(STDIN_FILENO, text, sizeof(text));
498 if (len == 0)
499 { // eof!
500 stdin_active = qfalse;
501 return NULL;
502 }
503
504 if (len < 1)
505 return NULL;
506 text[len-1] = 0; // rip off the /n and terminate
507
508 return text;
509 }
510 return NULL;
511}

513/*
514==================
515CON_Print
516==================
517*/
518void CON_Print( const char *msg )
519{
520 if (!msg[0])
521 return;
522
523 CON_Hide( );
524
525 if( com_ansiColor && com_ansiColor->integer )
526 Sys_AnsiColorPrint( msg );
527 else
528 fputs( msg, stderr );
529
530 if (!ttycon_on) {
531 // CON_Hide didn't do anything.
532 return;
533 }
534
535 // Only print prompt when msg ends with a newline, otherwise the console
536 // might get garbled when output does not fit on one line.
537 if (msg[strlen(msg) - 1] == '\n') {
538 CON_Show();
539
540 // Run CON_Show the number of times it was deferred.
541 while (ttycon_show_overdue > 0) {
542 CON_Show();
543 ttycon_show_overdue--;
544 }
545 }
546 else
547 {
548 // Defer calling CON_Show
549 ttycon_show_overdue++;
550 }
551}

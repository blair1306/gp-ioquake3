35/*******************************************/
36/* LPC analysis filter */
37/* NB! State is kept internally and the */
38/* filter always starts with zero state */
39/* first d output samples are set to zero */
40/*******************************************/
41
42void silk_LPC_analysis_filter(
43 opus_int16 *out, /* O Output signal */
44 const opus_int16 *in, /* I Input signal */
45 const opus_int16 *B, /* I MA prediction coefficients, Q12 [order] */
46 const opus_int32 len, /* I Signal length */
47 const opus_int32 d, /* I Filter order */
48 int arch /* I Run-time architecture */
49)
50{
51 opus_int j;
52#ifdef FIXED_POINT
53 opus_int16 mem[SILK_MAX_ORDER_LPC];
54 opus_int16 num[SILK_MAX_ORDER_LPC];
55#else
56 int ix;
57 opus_int32 out32_Q12, out32;
58 const opus_int16 *in_ptr;
59#endif
60
61 silk_assert( d >= 6 );
62 silk_assert( (d & 1) == 0 );
63 silk_assert( d <= len );
64
65#ifdef FIXED_POINT
66 silk_assert( d <= SILK_MAX_ORDER_LPC );
67 for ( j = 0; j < d; j++ ) {
68 num[ j ] = -B[ j ];
69 }
70 for (j=0;j<d;j++) {
71 mem[ j ] = in[ d - j - 1 ];
72 }
73 celt_fir( in + d, num, out + d, len - d, d, mem, arch );
74 for ( j = 0; j < d; j++ ) {
75 out[ j ] = 0;
76 }
77#else
78 (void)arch;
79 for( ix = d; ix < len; ix++ ) {
80 in_ptr = &in[ ix - 1 ];
81
82 out32_Q12 = silk_SMULBB( in_ptr[ 0 ], B[ 0 ] );
83 /* Allowing wrap around so that two wraps can cancel each other. The rare
84 cases where the result wraps around can only be triggered by invalid streams*/
85 out32_Q12 = silk_SMLABB_ovflw( out32_Q12, in_ptr[ -1 ], B[ 1 ] );
86 out32_Q12 = silk_SMLABB_ovflw( out32_Q12, in_ptr[ -2 ], B[ 2 ] );
87 out32_Q12 = silk_SMLABB_ovflw( out32_Q12, in_ptr[ -3 ], B[ 3 ] );
88 out32_Q12 = silk_SMLABB_ovflw( out32_Q12, in_ptr[ -4 ], B[ 4 ] );
89 out32_Q12 = silk_SMLABB_ovflw( out32_Q12, in_ptr[ -5 ], B[ 5 ] );
90 for( j = 6; j < d; j += 2 ) {
91 out32_Q12 = silk_SMLABB_ovflw( out32_Q12, in_ptr[ -j ], B[ j ] );
92 out32_Q12 = silk_SMLABB_ovflw( out32_Q12, in_ptr[ -j - 1 ], B[ j + 1 ] );
93 }
94
95 /* Subtract prediction */
96 out32_Q12 = silk_SUB32_ovflw( silk_LSHIFT( (opus_int32)in_ptr[ 1 ], 12 ), out32_Q12 );
97
98 /* Scale to Q0 */
99 out32 = silk_RSHIFT_ROUND( out32_Q12, 12 );
100
101 /* Saturate output */
102 out[ ix ] = (opus_int16)silk_SAT16( out32 );
103 }
104
105 /* Set first d output samples to zero */
106 silk_memset( out, 0, d * sizeof( opus_int16 ) );
107#endif
108}

35void silk_quant_LTP_gains(
36 opus_int16 B_Q14[ MAX_NB_SUBFR * LTP_ORDER ], /* I/O (un)quantized LTP gains */
37 opus_int8 cbk_index[ MAX_NB_SUBFR ], /* O Codebook Index */
38 opus_int8 *periodicity_index, /* O Periodicity Index */
39 opus_int32 *sum_log_gain_Q7, /* I/O Cumulative max prediction gain */
40 const opus_int32 W_Q18[ MAX_NB_SUBFR*LTP_ORDER*LTP_ORDER ], /* I Error Weights in Q18 */
41 opus_int mu_Q9, /* I Mu value (R/D tradeoff) */
42 opus_int lowComplexity, /* I Flag for low complexity */
43 const opus_int nb_subfr, /* I number of subframes */
44 int arch /* I Run-time architecture */
45)
46{
47 opus_int j, k, cbk_size;
48 opus_int8 temp_idx[ MAX_NB_SUBFR ];
49 const opus_uint8 *cl_ptr_Q5;
50 const opus_int8 *cbk_ptr_Q7;
51 const opus_uint8 *cbk_gain_ptr_Q7;
52 const opus_int16 *b_Q14_ptr;
53 const opus_int32 *W_Q18_ptr;
54 opus_int32 rate_dist_Q14_subfr, rate_dist_Q14, min_rate_dist_Q14;
55 opus_int32 sum_log_gain_tmp_Q7, best_sum_log_gain_Q7, max_gain_Q7, gain_Q7;
56
57 /***************************************************/
58 /* iterate over different codebooks with different */
59 /* rates/distortions, and choose best */
60 /***************************************************/
61 min_rate_dist_Q14 = silk_int32_MAX;
62 best_sum_log_gain_Q7 = 0;
63 for( k = 0; k < 3; k++ ) {
64 /* Safety margin for pitch gain control, to take into account factors
65 such as state rescaling/rewhitening. */
66 opus_int32 gain_safety = SILK_FIX_CONST( 0.4, 7 );
67
68 cl_ptr_Q5 = silk_LTP_gain_BITS_Q5_ptrs[ k ];
69 cbk_ptr_Q7 = silk_LTP_vq_ptrs_Q7[ k ];
70 cbk_gain_ptr_Q7 = silk_LTP_vq_gain_ptrs_Q7[ k ];
71 cbk_size = silk_LTP_vq_sizes[ k ];
72
73 /* Set up pointer to first subframe */
74 W_Q18_ptr = W_Q18;
75 b_Q14_ptr = B_Q14;
76
77 rate_dist_Q14 = 0;
78 sum_log_gain_tmp_Q7 = *sum_log_gain_Q7;
79 for( j = 0; j < nb_subfr; j++ ) {
80 max_gain_Q7 = silk_log2lin( ( SILK_FIX_CONST( MAX_SUM_LOG_GAIN_DB / 6.0, 7 ) - sum_log_gain_tmp_Q7 )
81 + SILK_FIX_CONST( 7, 7 ) ) - gain_safety;
82
83 silk_VQ_WMat_EC(
84 &temp_idx[ j ], /* O index of best codebook vector */
85 &rate_dist_Q14_subfr, /* O best weighted quantization error + mu * rate */
86 &gain_Q7, /* O sum of absolute LTP coefficients */
87 b_Q14_ptr, /* I input vector to be quantized */
88 W_Q18_ptr, /* I weighting matrix */
89 cbk_ptr_Q7, /* I codebook */
90 cbk_gain_ptr_Q7, /* I codebook effective gains */
91 cl_ptr_Q5, /* I code length for each codebook vector */
92 mu_Q9, /* I tradeoff between weighted error and rate */
93 max_gain_Q7, /* I maximum sum of absolute LTP coefficients */
94 cbk_size, /* I number of vectors in codebook */
95 arch /* I Run-time architecture */
96 );
97
98 rate_dist_Q14 = silk_ADD_POS_SAT32( rate_dist_Q14, rate_dist_Q14_subfr );
99 sum_log_gain_tmp_Q7 = silk_max(0, sum_log_gain_tmp_Q7
100 + silk_lin2log( gain_safety + gain_Q7 ) - SILK_FIX_CONST( 7, 7 ));
101
102 b_Q14_ptr += LTP_ORDER;
103 W_Q18_ptr += LTP_ORDER * LTP_ORDER;
104 }
105
106 /* Avoid never finding a codebook */
107 rate_dist_Q14 = silk_min( silk_int32_MAX - 1, rate_dist_Q14 );
108
109 if( rate_dist_Q14 < min_rate_dist_Q14 ) {
110 min_rate_dist_Q14 = rate_dist_Q14;
111 *periodicity_index = (opus_int8)k;
112 silk_memcpy( cbk_index, temp_idx, nb_subfr * sizeof( opus_int8 ) );
113 best_sum_log_gain_Q7 = sum_log_gain_tmp_Q7;
114 }
115
116 /* Break early in low-complexity mode if rate distortion is below threshold */
117 if( lowComplexity && ( rate_dist_Q14 < silk_LTP_gain_middle_avg_RD_Q14 ) ) {
118 break;
119 }
120 }
121
122 cbk_ptr_Q7 = silk_LTP_vq_ptrs_Q7[ *periodicity_index ];
123 for( j = 0; j < nb_subfr; j++ ) {
124 for( k = 0; k < LTP_ORDER; k++ ) {
125 B_Q14[ j * LTP_ORDER + k ] = silk_LSHIFT( cbk_ptr_Q7[ cbk_index[ j ] * LTP_ORDER + k ], 7 );
126 }
127 }
128 *sum_log_gain_Q7 = best_sum_log_gain_Q7;
129}

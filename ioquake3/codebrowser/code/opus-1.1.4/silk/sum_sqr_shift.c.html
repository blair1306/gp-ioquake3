34/* Compute number of bits to right shift the sum of squares of a vector */
35/* of int16s to make it fit in an int32 */
36void silk_sum_sqr_shift(
37 opus_int32 *energy, /* O Energy of x, after shifting to the right */
38 opus_int *shift, /* O Number of bits right shift applied to energy */
39 const opus_int16 *x, /* I Input vector */
40 opus_int len /* I Length of input vector */
41)
42{
43 opus_int i, shft;
44 opus_int32 nrg_tmp, nrg;
45
46 nrg = 0;
47 shft = 0;
48 len--;
49 for( i = 0; i < len; i += 2 ) {
50 nrg = silk_SMLABB_ovflw( nrg, x[ i ], x[ i ] );
51 nrg = silk_SMLABB_ovflw( nrg, x[ i + 1 ], x[ i + 1 ] );
52 if( nrg < 0 ) {
53 /* Scale down */
54 nrg = (opus_int32)silk_RSHIFT_uint( (opus_uint32)nrg, 2 );
55 shft = 2;
56 i+=2;
57 break;
58 }
59 }
60 for( ; i < len; i += 2 ) {
61 nrg_tmp = silk_SMULBB( x[ i ], x[ i ] );
62 nrg_tmp = silk_SMLABB_ovflw( nrg_tmp, x[ i + 1 ], x[ i + 1 ] );
63 nrg = (opus_int32)silk_ADD_RSHIFT_uint( nrg, (opus_uint32)nrg_tmp, shft );
64 if( nrg < 0 ) {
65 /* Scale down */
66 nrg = (opus_int32)silk_RSHIFT_uint( (opus_uint32)nrg, 2 );
67 shft += 2;
68 }
69 }
70 if( i == len ) {
71 /* One sample left to process */
72 nrg_tmp = silk_SMULBB( x[ i ], x[ i ] );
73 nrg = (opus_int32)silk_ADD_RSHIFT_uint( nrg, nrg_tmp, shft );
74 }
75
76 /* Make sure to have at least one extra leading zero (two leading zeros in total) */
77 if( nrg & 0xC0000000 ) {
78 nrg = silk_RSHIFT_uint( (opus_uint32)nrg, 2 );
79 shft += 2;
80 }
81
82 /* Output arguments */
83 *shift = shft;
84 *energy = nrg;
85}

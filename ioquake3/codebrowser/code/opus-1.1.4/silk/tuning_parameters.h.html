31#ifdef __cplusplus

33{
34#endif
35
36/* Decay time for bitreservoir */
37#define BITRESERVOIR_DECAY_TIME_MS 500
38
39/*******************/
40/* Pitch estimator */
41/*******************/
42
43/* Level of noise floor for whitening filter LPC analysis in pitch analysis */
44#define FIND_PITCH_WHITE_NOISE_FRACTION 1e-3f
45
46/* Bandwidth expansion for whitening filter in pitch analysis */
47#define FIND_PITCH_BANDWIDTH_EXPANSION 0.99f
48
49/*********************/
50/* Linear prediction */
51/*********************/
52
53/* LPC analysis regularization */
54#define FIND_LPC_COND_FAC 1e-5f
55
56/* LTP analysis defines */
57#define FIND_LTP_COND_FAC 1e-5f
58#define LTP_DAMPING 0.05f
59#define LTP_SMOOTHING 0.1f
60
61/* LTP quantization settings */
62#define MU_LTP_QUANT_NB 0.03f
63#define MU_LTP_QUANT_MB 0.025f
64#define MU_LTP_QUANT_WB 0.02f
65
66/* Max cumulative LTP gain */
67#define MAX_SUM_LOG_GAIN_DB 250.0f
68
69/***********************/
70/* High pass filtering */
71/***********************/
72
73/* Smoothing parameters for low end of pitch frequency range estimation */
74#define VARIABLE_HP_SMTH_COEF1 0.1f
75#define VARIABLE_HP_SMTH_COEF2 0.015f
76#define VARIABLE_HP_MAX_DELTA_FREQ 0.4f
77
78/* Min and max cut-off frequency values (-3 dB points) */
79#define VARIABLE_HP_MIN_CUTOFF_HZ 60
80#define VARIABLE_HP_MAX_CUTOFF_HZ 100
81
82/***********/
83/* Various */
84/***********/
85
86/* VAD threshold */
87#define SPEECH_ACTIVITY_DTX_THRES 0.05f
88
89/* Speech Activity LBRR enable threshold */
90#define LBRR_SPEECH_ACTIVITY_THRES 0.3f
91
92/*************************/
93/* Perceptual parameters */
94/*************************/
95
96/* reduction in coding SNR during low speech activity */
97#define BG_SNR_DECR_dB 2.0f
98
99/* factor for reducing quantization noise during voiced speech */
100#define HARM_SNR_INCR_dB 2.0f
101
102/* factor for reducing quantization noise for unvoiced sparse signals */
103#define SPARSE_SNR_INCR_dB 2.0f
104
105/* threshold for sparseness measure above which to use lower quantization offset during unvoiced */
106#define SPARSENESS_THRESHOLD_QNT_OFFSET 0.75f
107
108/* warping control */
109#define WARPING_MULTIPLIER 0.015f
110
111/* fraction added to first autocorrelation value */
112#define SHAPE_WHITE_NOISE_FRACTION 5e-5f
113
114/* noise shaping filter chirp factor */
115#define BANDWIDTH_EXPANSION 0.95f
116
117/* difference between chirp factors for analysis and synthesis noise shaping filters at low bitrates */
118#define LOW_RATE_BANDWIDTH_EXPANSION_DELTA 0.01f
119
120/* extra harmonic boosting (signal shaping) at low bitrates */
121#define LOW_RATE_HARMONIC_BOOST 0.1f
122
123/* extra harmonic boosting (signal shaping) for noisy input signals */
124#define LOW_INPUT_QUALITY_HARMONIC_BOOST 0.1f
125
126/* harmonic noise shaping */
127#define HARMONIC_SHAPING 0.3f
128
129/* extra harmonic noise shaping for high bitrates or noisy input */
130#define HIGH_RATE_OR_LOW_QUALITY_HARMONIC_SHAPING 0.2f
131
132/* parameter for shaping noise towards higher frequencies */
133#define HP_NOISE_COEF 0.25f
134
135/* parameter for shaping noise even more towards higher frequencies during voiced speech */
136#define HARM_HP_NOISE_COEF 0.35f
137
138/* parameter for applying a high-pass tilt to the input signal */
139#define INPUT_TILT 0.05f
140
141/* parameter for extra high-pass tilt to the input signal at high rates */
142#define HIGH_RATE_INPUT_TILT 0.1f
143
144/* parameter for reducing noise at the very low frequencies */
145#define LOW_FREQ_SHAPING 4.0f
146
147/* less reduction of noise at the very low frequencies for signals with low SNR at low frequencies */
148#define LOW_QUALITY_LOW_FREQ_SHAPING_DECR 0.5f
149
150/* subframe smoothing coefficient for HarmBoost, HarmShapeGain, Tilt (lower -> more smoothing) */
151#define SUBFR_SMTH_COEF 0.4f
152
153/* parameters defining the R/D tradeoff in the residual quantizer */
154#define LAMBDA_OFFSET 1.2f
155#define LAMBDA_SPEECH_ACT -0.2f
156#define LAMBDA_DELAYED_DECISIONS -0.05f
157#define LAMBDA_INPUT_QUALITY -0.1f
158#define LAMBDA_CODING_QUALITY -0.2f
159#define LAMBDA_QUANT_OFFSET 0.8f
160
161/* Compensation in bitrate calculations for 10 ms modes */
162#define REDUCE_BITRATE_10_MS_BPS 2200
163
164/* Maximum time before allowing a bandwidth transition */
165#define MAX_BANDWIDTH_SWITCH_DELAY_MS 5000

168}

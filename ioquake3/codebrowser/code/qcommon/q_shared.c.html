37/*
38============
39COM_SkipPath
40============
41*/
42char *COM_SkipPath (char *pathname)
43{
44 char *last;
45
46 last = pathname;
47 while (*pathname)
48 {
49 if (*pathname=='/')
50 last = pathname+1;
51 pathname++;
52 }
53 return last;
54}

56/*
57============
58COM_GetExtension
59============
60*/
61const char *COM_GetExtension( const char *name )
62{
63 const char *dot = strrchr(name, '.'), *slash;
64 if (dot && (!(slash = strrchr(name, '/')) || slash < dot))
65 return dot + 1;
66 else
67 return "";
68}

71/*
72============
73COM_StripExtension
74============
75*/
76void COM_StripExtension( const char *in, char *out, int destsize )
77{
78 const char *dot = strrchr(in, '.'), *slash;
79
80 if (dot && (!(slash = strrchr(in, '/')) || slash < dot))
81 destsize = (destsize < dot-in+1 ? destsize : dot-in+1);
82
83 if ( in == out && destsize > 1 )
84 out[destsize-1] = '\0';
85 else
86 Q_strncpyz(out, in, destsize);
87}

89/*
90============
91COM_CompareExtension
92
93string compare the end of the strings and return qtrue if strings match
94============
95*/
96qboolean COM_CompareExtension(const char *in, const char *ext)
97{
98 int inlen, extlen;
99
100 inlen = strlen(in);
101 extlen = strlen(ext);
102
103 if(extlen <= inlen)
104 {
105 in += inlen - extlen;
106
107 if(!Q_stricmp(in, ext))
108 return qtrue;
109 }
110
111 return qfalse;
112}

178short ShortSwap (short l)
179{
180 byte b1,b2;
181
182 b1 = l&255;
183 b2 = (l>>8)&255;
184
185 return (b1<<8) + b2;
186}
187
188short ShortNoSwap (short l)
189{
190 return l;
191}
192
193int LongSwap (int l)
194{
195 byte b1,b2,b3,b4;
196
197 b1 = l&255;
198 b2 = (l>>8)&255;
199 b3 = (l>>16)&255;
200 b4 = (l>>24)&255;
201
202 return ((int)b1<<24) + ((int)b2<<16) + ((int)b3<<8) + b4;
203}

205int LongNoSwap (int l)
206{
207 return l;
208}

210qint64 Long64Swap (qint64 ll)

212 qint64 result;

214 result.b0 = ll.b7;

216 result.b2 = ll.b5;
217 result.b3 = ll.b4;
218 result.b4 = ll.b3;
219 result.b5 = ll.b2;
220 result.b6 = ll.b1;
221 result.b7 = ll.b0;

341/*
342==============
343COM_Parse
344
345Parse a token out of a string
346Will never return NULL, just empty strings
347
348If "allowLineBreaks" is qtrue then an empty
349string will be returned if the next token is
350a newline.
351==============
352*/
353static char *SkipWhitespace( char *data, qboolean *hasNewLines ) {
354 int c;
355
356 while( (c = *data) <= ' ') {
357 if( !c ) {
358 return NULL;
359 }
360 if( c == '\n' ) {
361 com_lines++;
362 *hasNewLines = qtrue;
363 }
364 data++;
365 }
366
367 return data;
368}
369
370int COM_Compress( char *data_p ) {
371 char *in, *out;
372 int c;
373 qboolean newline = qfalse, whitespace = qfalse;
374
375 in = out = data_p;
376 if (in) {
377 while ((c = *in) != 0) {
378 // skip double slash comments
379 if ( c == '/' && in[1] == '/' ) {
380 while (*in && *in != '\n') {
381 in++;
382 }
383 // skip /* */ comments
384 } else if ( c == '/' && in[1] == '*' ) {
385 while ( *in && ( *in != '*' || in[1] != '/' ) )
386 in++;
387 if ( *in )
388 in += 2;
389 // record when we hit a newline
390 } else if ( c == '\n' || c == '\r' ) {
391 newline = qtrue;
392 in++;
393 // record when we hit whitespace
394 } else if ( c == ' ' || c == '\t') {
395 whitespace = qtrue;
396 in++;
397 // an actual token
398 } else {
399 // if we have a pending newline, emit it (and it counts as whitespace)
400 if (newline) {
401 *out++ = '\n';
402 newline = qfalse;
403 whitespace = qfalse;
404 } if (whitespace) {
405 *out++ = ' ';
406 whitespace = qfalse;
407 }
408
409 // copy quoted strings unmolested
410 if (c == '"') {
411 *out++ = c;
412 in++;
413 while (1) {
414 c = *in;
415 if (c && c != '"') {
416 *out++ = c;
417 in++;
418 } else {
419 break;
420 }
421 }
422 if (c == '"') {
423 *out++ = c;
424 in++;
425 }
426 } else {
427 *out = c;
428 out++;
429 in++;
430 }
431 }
432 }
433
434 *out = 0;
435 }
436 return out - data_p;
437}

439char *COM_ParseExt( char **data_p, qboolean allowLineBreaks )
440{
441 int c = 0, len;
442 qboolean hasNewLines = qfalse;
443 char *data;
444
445 data = *data_p;
446 len = 0;
447 com_token[0] = 0;
448 com_tokenline = 0;
449
450 // make sure incoming data is valid
451 if ( !data )
452 {
453 *data_p = NULL;
454 return com_token;
455 }
456
457 while ( 1 )
458 {
459 // skip whitespace
460 data = SkipWhitespace( data, &hasNewLines );
461 if ( !data )
462 {
463 *data_p = NULL;
464 return com_token;
465 }
466 if ( hasNewLines && !allowLineBreaks )
467 {
468 *data_p = data;
469 return com_token;
470 }
471
472 c = *data;
473
474 // skip double slash comments
475 if ( c == '/' && data[1] == '/' )
476 {
477 data += 2;
478 while (*data && *data != '\n') {
479 data++;
480 }
481 }
482 // skip /* */ comments
483 else if ( c=='/' && data[1] == '*' )
484 {
485 data += 2;
486 while ( *data && ( *data != '*' || data[1] != '/' ) )
487 {
488 if ( *data == '\n' )
489 {
490 com_lines++;
491 }
492 data++;
493 }
494 if ( *data )
495 {
496 data += 2;
497 }
498 }
499 else
500 {
501 break;
502 }
503 }
504
505 // token starts on this line
506 com_tokenline = com_lines;
507
508 // handle quoted strings
509 if (c == '\"')
510 {
511 data++;
512 while (1)
513 {
514 c = *data++;
515 if (c=='\"' || !c)
516 {
517 com_token[len] = 0;
518 *data_p = ( char * ) data;
519 return com_token;
520 }
521 if ( c == '\n' )
522 {
523 com_lines++;
524 }
525 if (len < MAX_TOKEN_CHARS - 1)
526 {
527 com_token[len] = c;
528 len++;
529 }
530 }
531 }
532
533 // parse a regular word
534 do
535 {
536 if (len < MAX_TOKEN_CHARS - 1)
537 {
538 com_token[len] = c;
539 len++;
540 }
541 data++;
542 c = *data;
543 } while (c>32);
544
545 com_token[len] = 0;
546
547 *data_p = ( char * ) data;
548 return com_token;
549}

566/*
567=================
568SkipBracedSection
569
570The next token should be an open brace or set depth to 1 if already parsed it.
571Skips until a matching close brace is found.
572Internal brace depths are properly skipped.
573=================
574*/
575qboolean SkipBracedSection (char **program, int depth) {
576 char *token;
577
578 do {
579 token = COM_ParseExt( program, qtrue );
580 if( token[1] == 0 ) {
581 if( token[0] == '{' ) {
582 depth++;
583 }
584 else if( token[0] == '}' ) {
585 depth--;
586 }
587 }
588 } while( depth && *program );
589
590 return ( depth == 0 );
591}

594=================
595SkipRestOfLine
596=================
597*/
598void SkipRestOfLine ( char **data ) {
599 char *p;
600 int c;
601
602 p = *data;
603
604 if ( !*p )
605 return;
606
607 while ( (c = *p++) != 0 ) {
608 if ( c == '\n' ) {
609 com_lines++;
610 break;
611 }
612 }
613
614 *data = p;
615}

618void Parse1DMatrix (char **buf_p, int x, float *m) {
619 char *token;
620 int i;
621
622 COM_MatchToken( buf_p, "(" );
623
624 for (i = 0 ; i < x ; i++) {
625 token = COM_Parse(buf_p);
626 m[i] = atof(token);
627 }
628
629 COM_MatchToken( buf_p, ")" );
630}

632void Parse2DMatrix (char **buf_p, int y, int x, float *m) {
633 int i;
634
635 COM_MatchToken( buf_p, "(" );
636
637 for (i = 0 ; i < y ; i++) {
638 Parse1DMatrix (buf_p, x, m + i * x);
639 }
640
641 COM_MatchToken( buf_p, ")" );
642}

656/*
657===================
658Com_HexStrToInt
659===================
660*/
661int Com_HexStrToInt( const char *str )
662{
663 if ( !str || !str[ 0 ] )
664 return -1;
665
666 // check for hex code
667 if( str[ 0 ] == '0' && str[ 1 ] == 'x' )
668 {
669 int i, n = 0;
670
671 for( i = 2; i < strlen( str ); i++ )
672 {
673 char digit;
674
675 n *= 16;
676
677 digit = tolower( str[ i ] );
678
679 if( digit >= '0' && digit <= '9' )
680 digit -= '0';
681 else if( digit >= 'a' && digit <= 'f' )
682 digit = digit - 'a' + 10;
683 else
684 return -1;
685
686 n += digit;
687 }
688
689 return n;
690 }
691
692 return -1;
693}

749#ifdef _MSC_VER
750/*
751=============
752Q_vsnprintf
753
754Special wrapper function for Microsoft's broken _vsnprintf() function.
755MinGW comes with its own snprintf() which is not broken.
756=============
757*/
758
759int Q_vsnprintf(char *str, size_t size, const char *format, va_list ap)
760{
761 int retval;
762
763 retval = _vsnprintf(str, size, format, ap);
764
765 if(retval < 0 || retval == size)
766 {
767 // Microsoft doesn't adhere to the C99 standard of vsnprintf,
768 // which states that the return value must be the number of
769 // bytes written if the output string had sufficient length.
770 //
771 // Obviously we cannot determine that value from Microsoft's
772 // implementation, so we have no choice but to return size.
773
774 str[size - 1] = '\0';
775 return size;
776 }
777
778 return retval;
779}

783=============
784Q_strncpyz
785
786Safe strncpy that ensures a trailing zero
787=============
788*/
789void Q_strncpyz( char *dest, const char *src, int destsize ) {
790 if ( !dest ) {
791 Com_Error( ERR_FATAL, "Q_strncpyz: NULL dest" );
792 }
793 if ( !src ) {
794 Com_Error( ERR_FATAL, "Q_strncpyz: NULL src" );
795 }
796 if ( destsize < 1 ) {
797 Com_Error(ERR_FATAL,"Q_strncpyz: destsize < 1" );
798 }
799
800 strncpy( dest, src, destsize-1 );
801 dest[destsize-1] = 0;
802}

804int Q_stricmpn (const char *s1, const char *s2, int n) {
805 int c1, c2;
806
807 if ( s1 == NULL ) {
808 if ( s2 == NULL )
809 return 0;
810 else
811 return -1;
812 }
813 else if ( s2==NULL )
814 return 1;
815
816
817
818 do {
819 c1 = *s1++;
820 c2 = *s2++;
821
822 if (!n--) {
823 return 0; // strings are equal until end point
824 }
825
826 if (c1 != c2) {
827 if (c1 >= 'a' && c1 <= 'z') {
828 c1 -= ('a' - 'A');
829 }
830 if (c2 >= 'a' && c2 <= 'z') {
831 c2 -= ('a' - 'A');
832 }
833 if (c1 != c2) {
834 return c1 < c2 ? -1 : 1;
835 }
836 }
837 } while (c1);
838
839 return 0; // strings are equal
840}

842int Q_strncmp (const char *s1, const char *s2, int n) {
843 int c1, c2;
844
845 do {
846 c1 = *s1++;
847 c2 = *s2++;
848
849 if (!n--) {
850 return 0; // strings are equal until end point
851 }
852
853 if (c1 != c2) {
854 return c1 < c2 ? -1 : 1;
855 }
856 } while (c1);
857
858 return 0; // strings are equal
859}

900/*
901* Find the first occurrence of find in s.
902*/
903const char *Q_stristr( const char *s, const char *find)
904{
905 char c, sc;
906 size_t len;
907
908 if ((c = *find++) != 0)
909 {
910 if (c >= 'a' && c <= 'z')
911 {
912 c -= ('a' - 'A');
913 }
914 len = strlen(find);
915 do
916 {
917 do
918 {
919 if ((sc = *s++) == 0)
920 return NULL;
921 if (sc >= 'a' && sc <= 'z')
922 {
923 sc -= ('a' - 'A');
924 }
925 } while (sc != c);
926 } while (Q_stricmpn(s, find, len) != 0);
927 s--;
928 }
929 return s;
930}

1058/*
1059===============
1060Info_ValueForKey
1061
1062Searches the string for the given
1063key and returns the associated value, or an empty string.
1064FIXME: overflow check?
1065===============
1066*/
1067char *Info_ValueForKey( const char *s, const char *key ) {
1068 char pkey[BIG_INFO_KEY];
1069 static char value[2][BIG_INFO_VALUE]; // use two buffers so compares
1070 // work without stomping on each other
1071 static int valueindex = 0;
1072 char *o;
1073
1074 if ( !s || !key ) {
1075 return "";
1076 }
1077
1078 if ( strlen( s ) >= BIG_INFO_STRING ) {
1079 Com_Error( ERR_DROP, "Info_ValueForKey: oversize infostring" );
1080 }
1081
1082 valueindex ^= 1;
1083 if (*s == '\\')
1084 s++;
1085 while (1)
1086 {
1087 o = pkey;
1088 while (*s != '\\')
1089 {
1090 if (!*s)
1091 return "";
1092 *o++ = *s++;
1093 }
1094 *o = 0;
1095 s++;
1096
1097 o = value[valueindex];
1098
1099 while (*s != '\\' && *s)
1100 {
1101 *o++ = *s++;
1102 }
1103 *o = 0;
1104
1105 if (!Q_stricmp (key, pkey) )
1106 return value[valueindex];
1107
1108 if (!*s)
1109 break;
1110 s++;
1111 }
1112
1113 return "";
1114}

1323 return;
1324 }

1327 strcpy (s, newi);

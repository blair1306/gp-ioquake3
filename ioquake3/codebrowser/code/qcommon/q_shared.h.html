504static ID_INLINE float Q_rsqrt( float number ) {
505 float x = 0.5f * number;
506 float y;
507#ifdef __GNUC__
508 asm("frsqrte %0,%1" : "=f" (y) : "f" (number));
509#else
510 y = __frsqrte( number );
511#endif
512 return y * (1.5f - (x * y * y));
513 }

626// fast vector normalize routine that does not check to make sure
627// that length != 0, nor does it return length, uses rsqrt approximation
628static ID_INLINE void VectorNormalizeFast( vec3_t v )
629{
630 float ilength;
631
632 ilength = Q_rsqrt( DotProduct( v, v ) );
633
634 v[0] *= ilength;
635 v[1] *= ilength;
636 v[2] *= ilength;
637}

325//endianness
326void CopyShortSwap (void *dest, void *src);
327void CopyLongSwap (void *dest, void *src);
328short ShortSwap (short l);
329int LongSwap (int l);
330float FloatSwap (const float *f);
331
332#if defined( Q3_BIG_ENDIAN ) && defined( Q3_LITTLE_ENDIAN )
333#error "Endianness defined as both big and little"
334#elif defined( Q3_BIG_ENDIAN )
335
336#define CopyLittleShort(dest, src) CopyShortSwap(dest, src)
337#define CopyLittleLong(dest, src) CopyLongSwap(dest, src)
338#define LittleShort(x) ShortSwap(x)
339#define LittleLong(x) LongSwap(x)
340#define LittleFloat(x) FloatSwap(&x)
341#define BigShort
342#define BigLong
343#define BigFloat
344
345#elif defined( Q3_LITTLE_ENDIAN )
346
347#define CopyLittleShort(dest, src) Com_Memcpy(dest, src, 2)
348#define CopyLittleLong(dest, src) Com_Memcpy(dest, src, 4)
349#define LittleShort
350#define LittleLong
351#define LittleFloat
352#define BigShort(x) ShortSwap(x)
353#define BigLong(x) LongSwap(x)
354#define BigFloat(x) FloatSwap(&x)
355
356#elif defined( Q3_VM )
357
358#define LittleShort
359#define LittleLong
360#define LittleFloat
361#define BigShort
362#define BigLong
363#define BigFloat
364
365#else
366#error "Endianness not defined"
367#endif

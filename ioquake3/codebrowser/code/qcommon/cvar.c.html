35#define FILE_HASH_SIZE 256
36static cvar_t *hashTable[FILE_HASH_SIZE];

38/*
39================
40return a hash value for the filename
41================
42*/
43static long generateHashValue( const char *fname ) {
44 int i;
45 long hash;
46 char letter;
47
48 hash = 0;
49 i = 0;
50 while (fname[i] != '\0') {
51 letter = tolower(fname[i]);
52 hash+=(long)(letter)*(i+119);
53 i++;
54 }
55 hash &= (FILE_HASH_SIZE-1);
56 return hash;
57}

59/*
60============
61Cvar_ValidateString
62============
63*/
64static qboolean Cvar_ValidateString( const char *s ) {
65 if ( !s ) {
66 return qfalse;
67 }
68 if ( strchr( s, '\\' ) ) {
69 return qfalse;
70 }
71 if ( strchr( s, '\"' ) ) {
72 return qfalse;
73 }
74 if ( strchr( s, ';' ) ) {
75 return qfalse;
76 }
77 return qtrue;
78}

80/*
81============
82Cvar_FindVar
83============
84*/
85static cvar_t *Cvar_FindVar( const char *var_name ) {
86 cvar_t *var;
87 long hash;
88
89 hash = generateHashValue(var_name);
90
91 for (var=hashTable[hash] ; var ; var=var->hashNext) {
92 if (!Q_stricmp(var_name, var->name)) {
93 return var;
94 }
95 }
96
97 return NULL;
98}

198/*
199============
200Cvar_Validate
201============
202*/
203static const char *Cvar_Validate( cvar_t *var,
204 const char *value, qboolean warn )
205{
206 static char s[ MAX_CVAR_VALUE_STRING ];
207 float valuef;
208 qboolean changed = qfalse;
209
210 if( !var->validate )
211 return value;
212
213 if( !value )
214 return value;
215
216 if( Q_isanumber( value ) )
217 {
218 valuef = atof( value );
219
220 if( var->integral )
221 {
222 if( !Q_isintegral( valuef ) )
223 {
224 if( warn )
225 Com_Printf( "WARNING: cvar '%s' must be integral", var->name );
226
227 valuef = (int)valuef;
228 changed = qtrue;
229 }
230 }
231 }
232 else
233 {
234 if( warn )
235 Com_Printf( "WARNING: cvar '%s' must be numeric", var->name );
236
237 valuef = atof( var->resetString );
238 changed = qtrue;
239 }
240
241 if( valuef < var->min )
242 {
243 if( warn )
244 {
245 if( changed )
246 Com_Printf( " and is" );
247 else
248 Com_Printf( "WARNING: cvar '%s'", var->name );
249
250 if( Q_isintegral( var->min ) )
251 Com_Printf( " out of range (min %d)", (int)var->min );
252 else
253 Com_Printf( " out of range (min %f)", var->min );
254 }
255
256 valuef = var->min;
257 changed = qtrue;
258 }
259 else if( valuef > var->max )
260 {
261 if( warn )
262 {
263 if( changed )
264 Com_Printf( " and is" );
265 else
266 Com_Printf( "WARNING: cvar '%s'", var->name );
267
268 if( Q_isintegral( var->max ) )
269 Com_Printf( " out of range (max %d)", (int)var->max );
270 else
271 Com_Printf( " out of range (max %f)", var->max );
272 }
273
274 valuef = var->max;
275 changed = qtrue;
276 }
277
278 if( changed )
279 {
280 if( Q_isintegral( valuef ) )
281 {
282 Com_sprintf( s, sizeof( s ), "%d", (int)valuef );
283
284 if( warn )
285 Com_Printf( ", setting to %d\n", (int)valuef );
286 }
287 else
288 {
289 Com_sprintf( s, sizeof( s ), "%f", valuef );
290
291 if( warn )
292 Com_Printf( ", setting to %f\n", valuef );
293 }
294
295 return s;
296 }
297 else
298 return value;
299}

302/*
303============
304Cvar_Get
305
306If the variable already exists, the value will not be set unless CVAR_ROM
307The flags will be or'ed in if the variable exists.
308============
309*/
310cvar_t *Cvar_Get( const char *var_name, const char *var_value, int flags ) {
311 cvar_t *var;
312 long hash;
313 int index;
314
315 if ( !var_name || ! var_value ) {
316 Com_Error( ERR_FATAL, "Cvar_Get: NULL parameter" );
317 }
318
319 if ( !Cvar_ValidateString( var_name ) ) {
320 Com_Printf("invalid cvar name string: %s\n", var_name );
321 var_name = "BADNAME";
322 }
323
324#if 0 // FIXME: values with backslash happen
325 if ( !Cvar_ValidateString( var_value ) ) {
326 Com_Printf("invalid cvar value string: %s\n", var_value );
327 var_value = "BADVALUE";
328 }
329#endif
330
331 var = Cvar_FindVar (var_name);
332
333 if(var)
334 {
335 var_value = Cvar_Validate(var, var_value, qfalse);
336
337 // Make sure the game code cannot mark engine-added variables as gamecode vars
338 if(var->flags & CVAR_VM_CREATED)
339 {
340 if(!(flags & CVAR_VM_CREATED))
341 var->flags &= ~CVAR_VM_CREATED;
342 }
343 else if (!(var->flags & CVAR_USER_CREATED))
344 {
345 if(flags & CVAR_VM_CREATED)
346 flags &= ~CVAR_VM_CREATED;
347 }
348
349 // if the C code is now specifying a variable that the user already
350 // set a value for, take the new value as the reset value
351 if(var->flags & CVAR_USER_CREATED)
352 {
353 var->flags &= ~CVAR_USER_CREATED;
354 Z_Free( var->resetString );
355 var->resetString = CopyString( var_value );
356
357 if(flags & CVAR_ROM)
358 {
359 // this variable was set by the user,
360 // so force it to value given by the engine.
361
362 if(var->latchedString)
363 Z_Free(var->latchedString);
364
365 var->latchedString = CopyString(var_value);
366 }
367 }
368
369 // Make sure servers cannot mark engine-added variables as SERVER_CREATED
370 if(var->flags & CVAR_SERVER_CREATED)
371 {
372 if(!(flags & CVAR_SERVER_CREATED))
373 var->flags &= ~CVAR_SERVER_CREATED;
374 }
375 else
376 {
377 if(flags & CVAR_SERVER_CREATED)
378 flags &= ~CVAR_SERVER_CREATED;
379 }
380
381 var->flags |= flags;
382
383 // only allow one non-empty reset string without a warning
384 if ( !var->resetString[0] ) {
385 // we don't have a reset string yet
386 Z_Free( var->resetString );
387 var->resetString = CopyString( var_value );
388 } else if ( var_value[0] && strcmp( var->resetString, var_value ) ) {
389 Com_DPrintf( "Warning: cvar \"%s\" given initial values: \"%s\" and \"%s\"\n",
390 var_name, var->resetString, var_value );
391 }
392 // if we have a latched string, take that value now
393 if ( var->latchedString ) {
394 char *s;
395
396 s = var->latchedString;
397 var->latchedString = NULL; // otherwise cvar_set2 would free it
398 Cvar_Set2( var_name, s, qtrue );
399 Z_Free( s );
400 }
401
402 // ZOID--needs to be set so that cvars the game sets as
403 // SERVERINFO get sent to clients
404 cvar_modifiedFlags |= flags;
405
406 return var;
407 }
408
409 //
410 // allocate a new cvar
411 //
412
413 // find a free cvar
414 for(index = 0; index < MAX_CVARS; index++)
415 {
416 if(!cvar_indexes[index].name)
417 break;
418 }
419
420 if(index >= MAX_CVARS)
421 {
422 if(!com_errorEntered)
423 Com_Error(ERR_FATAL, "Error: Too many cvars, cannot create a new one!");
424
425 return NULL;
426 }
427
428 var = &cvar_indexes[index];
429
430 if(index >= cvar_numIndexes)
431 cvar_numIndexes = index + 1;
432
433 var->name = CopyString (var_name);
434 var->string = CopyString (var_value);
435 var->modified = qtrue;
436 var->modificationCount = 1;
437 var->value = atof (var->string);
438 var->integer = atoi(var->string);
439 var->resetString = CopyString( var_value );
440 var->validate = qfalse;
441 var->description = NULL;
442
443 // link the variable in
444 var->next = cvar_vars;
445 if(cvar_vars)
446 cvar_vars->prev = var;
447
448 var->prev = NULL;
449 cvar_vars = var;
450
451 var->flags = flags;
452 // note what types of cvars have been modified (userinfo, archive, serverinfo, systeminfo)
453 cvar_modifiedFlags |= var->flags;
454
455 hash = generateHashValue(var_name);
456 var->hashIndex = hash;
457
458 var->hashNext = hashTable[hash];
459 if(hashTable[hash])
460 hashTable[hash]->hashPrev = var;
461
462 var->hashPrev = NULL;
463 hashTable[hash] = var;
464
465 return var;
466}

468/*
469============
470Cvar_Print
471
472Prints the value, default, and latched string of the given variable
473============
474*/
475void Cvar_Print( cvar_t *v ) {
476 Com_Printf ("\"%s\" is:\"%s" S_COLOR_WHITE "\"",
477 v->name, v->string );
478
479 if ( !( v->flags & CVAR_ROM ) ) {
480 if ( !Q_stricmp( v->string, v->resetString ) ) {
481 Com_Printf (", the default" );
482 } else {
483 Com_Printf (" default:\"%s" S_COLOR_WHITE "\"",
484 v->resetString );
485 }
486 }
487
488 Com_Printf ("\n");
489
490 if ( v->latchedString ) {
491 Com_Printf( "latched: \"%s\"\n", v->latchedString );
492 }
493
494 if ( v->description ) {
495 Com_Printf( "%s\n", v->description );
496 }
497}

499/*
500============
501Cvar_Set2
502============
503*/
504cvar_t *Cvar_Set2( const char *var_name, const char *value, qboolean force ) {
505 cvar_t *var;
506
507// Com_DPrintf( "Cvar_Set2: %s %s\n", var_name, value );
508
509 if ( !Cvar_ValidateString( var_name ) ) {
510 Com_Printf("invalid cvar name string: %s\n", var_name );
511 var_name = "BADNAME";
512 }
513
514#if 0 // FIXME
515 if ( value && !Cvar_ValidateString( value ) ) {
516 Com_Printf("invalid cvar value string: %s\n", value );
517 var_value = "BADVALUE";
518 }
519#endif
520
521 var = Cvar_FindVar (var_name);
522 if (!var) {
523 if ( !value ) {
524 return NULL;
525 }
526 // create it
527 if ( !force ) {
528 return Cvar_Get( var_name, value, CVAR_USER_CREATED );
529 } else {
530 return Cvar_Get (var_name, value, 0);
531 }
532 }
533
534 if (!value ) {
535 value = var->resetString;
536 }
537
538 value = Cvar_Validate(var, value, qtrue);
539
540 if((var->flags & CVAR_LATCH) && var->latchedString)
541 {
542 if(!strcmp(value, var->string))
543 {
544 Z_Free(var->latchedString);
545 var->latchedString = NULL;
546 return var;
547 }
548
549 if(!strcmp(value, var->latchedString))
550 return var;
551 }
552 else if(!strcmp(value, var->string))
553 return var;
554
555 // note what types of cvars have been modified (userinfo, archive, serverinfo, systeminfo)
556 cvar_modifiedFlags |= var->flags;
557
558 if (!force)
559 {
560 if (var->flags & CVAR_ROM)
561 {
562 Com_Printf ("%s is read only.\n", var_name);
563 return var;
564 }
565
566 if (var->flags & CVAR_INIT)
567 {
568 Com_Printf ("%s is write protected.\n", var_name);
569 return var;
570 }
571
572 if (var->flags & CVAR_LATCH)
573 {
574 if (var->latchedString)
575 {
576 if (strcmp(value, var->latchedString) == 0)
577 return var;
578 Z_Free (var->latchedString);
579 }
580 else
581 {
582 if (strcmp(value, var->string) == 0)
583 return var;
584 }
585
586 Com_Printf ("%s will be changed upon restarting.\n", var_name);
587 var->latchedString = CopyString(value);
588 var->modified = qtrue;
589 var->modificationCount++;
590 return var;
591 }
592
593 if ( (var->flags & CVAR_CHEAT) && !cvar_cheats->integer )
594 {
595 Com_Printf ("%s is cheat protected.\n", var_name);
596 return var;
597 }
598
599 }
600 else
601 {
602 if (var->latchedString)
603 {
604 Z_Free (var->latchedString);
605 var->latchedString = NULL;
606 }
607 }
608
609 if (!strcmp(value, var->string))
610 return var; // not changed
611
612 var->modified = qtrue;
613 var->modificationCount++;
614
615 Z_Free (var->string); // free the old value string
616
617 var->string = CopyString(value);
618 var->value = atof (var->string);
619 var->integer = atoi (var->string);
620
621 return var;
622}

715/*
716============
717Cvar_SetCheatState
718
719Any testing variables will be reset to the safe values
720============
721*/
722void Cvar_SetCheatState(void)
723{
724 cvar_t *var;
725
726 // set all default vars to the safe value
727 for(var = cvar_vars; var ; var = var->next)
728 {
729 if(var->flags & CVAR_CHEAT)
730 {
731 // the CVAR_LATCHED|CVAR_CHEAT vars might escape the reset here
732 // because of a different var->latchedString
733 if (var->latchedString)
734 {
735 Z_Free(var->latchedString);
736 var->latchedString = NULL;
737 }
738 if (strcmp(var->resetString,var->string))
739 Cvar_Set(var->name, var->resetString);
740 }
741 }
742}

745============
746Cvar_Command
747
748Handles variable inspection and changing from the console
749============
750*/
751qboolean Cvar_Command( void ) {
752 cvar_t *v;
753
754 // check variables
755 v = Cvar_FindVar (Cmd_Argv(0));
756 if (!v) {
757 return qfalse;
758 }
759
760 // perform a variable print or set
761 if ( Cmd_Argc() == 1 ) {
762 Cvar_Print( v );
763 return qtrue;
764 }
765
766 // set the value if forcing isn't required
767 Cvar_Set2 (v->name, Cmd_Args(), qfalse);
768 return qtrue;
769}

801/*
802============
803Cvar_Toggle_f
804
805Toggles a cvar for easy single key binding, optionally through a list of
806given values
807============
808*/
809void Cvar_Toggle_f( void ) {
810 int i, c = Cmd_Argc();
811 char *curval;
812
813 if(c < 2) {
814 Com_Printf("usage: toggle <variable> [value1, value2, ...]\n");
815 return;
816 }
817
818 if(c == 2) {
819 Cvar_Set2(Cmd_Argv(1), va("%d",
820 !Cvar_VariableValue(Cmd_Argv(1))),
821 qfalse);
822 return;
823 }
824
825 if(c == 3) {
826 Com_Printf("toggle: nothing to toggle to\n");
827 return;
828 }
829
830 curval = Cvar_VariableString(Cmd_Argv(1));
831
832 // don't bother checking the last arg for a match since the desired
833 // behaviour is the same as no match (set to the first argument)
834 for(i = 2; i + 1 < c; i++) {
835 if(strcmp(curval, Cmd_Argv(i)) == 0) {
836 Cvar_Set2(Cmd_Argv(1), Cmd_Argv(i + 1), qfalse);
837 return;
838 }
839 }
840
841 // fallback
842 Cvar_Set2(Cmd_Argv(1), Cmd_Argv(2), qfalse);
843}

845/*
846============
847Cvar_Set_f
848
849Allows setting and defining of arbitrary cvars from console, even if they
850weren't declared in C code.
851============
852*/
853void Cvar_Set_f( void ) {
854 int c;
855 char *cmd;
856 cvar_t *v;
857
858 c = Cmd_Argc();
859 cmd = Cmd_Argv(0);
860
861 if ( c < 2 ) {
862 Com_Printf ("usage: %s <variable> <value>\n", cmd);
863 return;
864 }
865 if ( c == 2 ) {
866 Cvar_Print_f();
867 return;
868 }
869
870 v = Cvar_Set2 (Cmd_Argv(1), Cmd_ArgsFrom(2), qfalse);
871 if( !v ) {
872 return;
873 }
874 switch( cmd[3] ) {
875 case 'a':
876 if( !( v->flags & CVAR_ARCHIVE ) ) {
877 v->flags |= CVAR_ARCHIVE;
878 cvar_modifiedFlags |= CVAR_ARCHIVE;
879 }
880 break;
881 case 'u':
882 if( !( v->flags & CVAR_USERINFO ) ) {
883 v->flags |= CVAR_USERINFO;
884 cvar_modifiedFlags |= CVAR_USERINFO;
885 }
886 break;
887 case 's':
888 if( !( v->flags & CVAR_SERVERINFO ) ) {
889 v->flags |= CVAR_SERVERINFO;
890 cvar_modifiedFlags |= CVAR_SERVERINFO;
891 }
892 break;
893 }
894}

909/*
910============
911Cvar_WriteVariables
912
913Appends lines containing "set variable value" for all variables
914with the archive flag set to qtrue.
915============
916*/
917void Cvar_WriteVariables(fileHandle_t f)
918{
919 cvar_t *var;
920 char buffer[1024];
921
922 for (var = cvar_vars; var; var = var->next)
923 {

936 } else {

939 "\"%s\" too long to write to file\n", var->name );

942 Com_sprintf (buffer, sizeof(buffer), "seta %s \"%s\"\n", var->name, var->string);

1110Cvar_Unset
1111
1112Unsets a cvar
1113============
1114*/
1115
1116cvar_t *Cvar_Unset(cvar_t *cv)
1117{
1118 cvar_t *next = cv->next;
1119
1120 // note what types of cvars have been modified (userinfo, archive, serverinfo, systeminfo)
1121 cvar_modifiedFlags |= cv->flags;
1122
1123 if(cv->name)
1124 Z_Free(cv->name);
1125 if(cv->string)
1126 Z_Free(cv->string);
1127 if(cv->latchedString)
1128 Z_Free(cv->latchedString);
1129 if(cv->resetString)
1130 Z_Free(cv->resetString);
1131 if(cv->description)
1132 Z_Free(cv->description);
1133
1134 if(cv->prev)
1135 cv->prev->next = cv->next;
1136 else
1137 cvar_vars = cv->next;
1138 if(cv->next)
1139 cv->next->prev = cv->prev;
1140
1141 if(cv->hashPrev)
1142 cv->hashPrev->hashNext = cv->hashNext;
1143 else
1144 hashTable[cv->hashIndex] = cv->hashNext;
1145 if(cv->hashNext)
1146 cv->hashNext->hashPrev = cv->hashPrev;
1147
1148 Com_Memset(cv, '\0', sizeof(*cv));
1149
1150 return next;
1151}

1184/*
1185============
1186Cvar_Restart
1187
1188Resets all cvars to their hardcoded values and removes userdefined variables
1189and variables added via the VMs if requested.
1190============
1191*/
1192
1193void Cvar_Restart(qboolean unsetVM)
1194{
1195 cvar_t *curvar;
1196
1197 curvar = cvar_vars;
1198
1199 while(curvar)
1200 {
1201 if((curvar->flags & CVAR_USER_CREATED) ||
1202 (unsetVM && (curvar->flags & CVAR_VM_CREATED)))
1203 {
1204 // throw out any variables the user/vm created
1205 curvar = Cvar_Unset(curvar);
1206 continue;
1207 }
1208
1209 if(!(curvar->flags & (CVAR_ROM | CVAR_INIT | CVAR_NORESTART)))
1210 {
1211 // Just reset the rest to their default values.
1212 Cvar_Set2(curvar->name, curvar->resetString, qfalse);
1213 }
1214
1215 curvar = curvar->next;
1216 }
1217}

1319/*
1320=====================
1321Cvar_Register
1322
1323basically a slightly modified Cvar_Get for the interpreted modules
1324=====================
1325*/
1326void Cvar_Register(vmCvar_t *vmCvar, const char *varName, const char *defaultValue, int flags)
1327{
1328 cvar_t *cv;
1329
1330 // There is code in Cvar_Get to prevent CVAR_ROM cvars being changed by the
1331 // user. In other words CVAR_ARCHIVE and CVAR_ROM are mutually exclusive
1332 // flags. Unfortunately some historical game code (including single player
1333 // baseq3) sets both flags. We unset CVAR_ROM for such cvars.
1334 if ((flags & (CVAR_ARCHIVE | CVAR_ROM)) == (CVAR_ARCHIVE | CVAR_ROM)) {
1335 Com_DPrintf( S_COLOR_YELLOW "WARNING: Unsetting CVAR_ROM cvar '%s', "
1336 "since it is also CVAR_ARCHIVE\n", varName );
1337 flags &= ~CVAR_ROM;
1338 }
1339
1340 cv = Cvar_Get(varName, defaultValue, flags | CVAR_VM_CREATED);
1341
1342 if (!vmCvar)
1343 return;
1344
1345 vmCvar->handle = cv - cvar_indexes;
1346 vmCvar->modificationCount = -1;
1347 Cvar_Update( vmCvar );
1348}

1351/*
1352=====================
1353Cvar_Update
1354
1355updates an interpreted modules' version of a cvar
1356=====================
1357*/
1358void Cvar_Update( vmCvar_t *vmCvar ) {
1359 cvar_t *cv = NULL;
1360 assert(vmCvar);
1361
1362 if ( (unsigned)vmCvar->handle >= cvar_numIndexes ) {
1363 Com_Error( ERR_DROP, "Cvar_Update: handle out of range" );
1364 }
1365
1366 cv = cvar_indexes + vmCvar->handle;
1367
1368 if ( cv->modificationCount == vmCvar->modificationCount ) {
1369 return;
1370 }
1371 if ( !cv->string ) {
1372 return; // variable might have been cleared by a cvar_restart
1373 }
1374 vmCvar->modificationCount = cv->modificationCount;
1375 if ( strlen(cv->string)+1 > MAX_CVAR_VALUE_STRING )
1376 Com_Error( ERR_DROP, "Cvar_Update: src %s length %u exceeds MAX_CVAR_VALUE_STRING",
1377 cv->string,
1378 (unsigned int) strlen(cv->string));
1379 Q_strncpyz( vmCvar->string, cv->string, MAX_CVAR_VALUE_STRING );
1380
1381 vmCvar->value = cv->value;
1382 vmCvar->integer = cv->integer;
1383}

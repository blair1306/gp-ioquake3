186/*
187 * Here are the definitions for Ravensoft's model format of md4. Raven stores their
188 * playermodels in .mdr files, in some games, which are pretty much like the md4
189 * format implemented by ID soft. It seems like ID's original md4 stuff is not used at all.
190 * MDR is being used in EliteForce, JediKnight2 and Soldiers of Fortune2 (I think).
191 * So this comes in handy for anyone who wants to make it possible to load player
192 * models from these games.
193 * This format has bone tags, which is similar to the thing you have in md3 I suppose.
194 * Raven has released their version of md3view under GPL enabling me to add support
195 * to this codebase. Thanks to Steven Howes aka Skinner for helping with example
196 * source code.
197 *
198 * - Thilo Schulz (arny@ats.s.bawue.de)
199 */
200
201#define MDR_IDENT (('5'<<24)+('M'<<16)+('D'<<8)+'R')
202#define MDR_VERSION 2
203#define MDR_MAX_BONES 128
204
205typedef struct {
206 int boneIndex; // these are indexes into the boneReferences,
207 float boneWeight; // not the global per-frame bone list
208 vec3_t offset;
209} mdrWeight_t;
210
211typedef struct {
212 vec3_t normal;
213 vec2_t texCoords;
214 int numWeights;
215 mdrWeight_t weights[1]; // variable sized
216} mdrVertex_t;
217
218typedef struct {
219 int indexes[3];
220} mdrTriangle_t;
221
222typedef struct {
223 int ident;
224
225 char name[MAX_QPATH]; // polyset name
226 char shader[MAX_QPATH];
227 int shaderIndex; // for in-game use
228
229 int ofsHeader; // this will be a negative number
230
231 int numVerts;
232 int ofsVerts;
233
234 int numTriangles;
235 int ofsTriangles;
236
237 // Bone references are a set of ints representing all the bones
238 // present in any vertex weights for this surface. This is
239 // needed because a model may have surfaces that need to be
240 // drawn at different sort times, and we don't want to have
241 // to re-interpolate all the bones for each surface.
242 int numBoneReferences;
243 int ofsBoneReferences;
244
245 int ofsEnd; // next surface follows
246} mdrSurface_t;
247
248typedef struct {
249 float matrix[3][4];
250} mdrBone_t;

252typedef struct {
253 vec3_t bounds[2]; // bounds of all surfaces of all LOD's for this frame
254 vec3_t localOrigin; // midpoint of bounds, used for sphere cull
255 float radius; // dist from localOrigin to corner
256 char name[16];
257 mdrBone_t bones[1]; // [numBones]
258} mdrFrame_t;
259

261 unsigned char Comp[24]; // MC_COMP_BYTES is in MatComp.h, but don't want to couple
262} mdrCompBone_t;
263
264typedef struct {
265 vec3_t bounds[2]; // bounds of all surfaces of all LOD's for this frame
266 vec3_t localOrigin; // midpoint of bounds, used for sphere cull
267 float radius; // dist from localOrigin to corner
268 mdrCompBone_t bones[1]; // [numBones]
269} mdrCompFrame_t;

271typedef struct {
272 int numSurfaces;

274 int ofsEnd; // next lod follows

282typedef struct {
283 int ident;
284 int version;
285
286 char name[MAX_QPATH]; // model name
287
288 // frames and bones are shared by all levels of detail
289 int numFrames;
290 int numBones;

294 int numLODs;
295 int ofsLODs;
296
297 int numTags;

299
300 int ofsEnd; // end of file
301} mdrHeader_t;

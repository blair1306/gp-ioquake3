186// this isn't a real cheap function to call!
187int DirToByte( vec3_t dir ) {
188 int i, best;
189 float d, bestd;
190
191 if ( !dir ) {
192 return 0;
193 }
194
195 bestd = 0;
196 best = 0;
197 for (i=0 ; i<NUMVERTEXNORMALS ; i++)
198 {
199 d = DotProduct (dir, bytedirs[i]);
200 if (d > bestd)
201 {
202 bestd = d;
203 best = i;
204 }
205 }
206
207 return best;
208}

240float NormalizeColor( const vec3_t in, vec3_t out ) {
241 float max;
242
243 max = in[0];
244 if ( in[1] > max ) {
245 max = in[1];
246 }
247 if ( in[2] > max ) {
248 max = in[2];
249 }
250
251 if ( !max ) {
252 VectorClear( out );
253 } else {
254 out[0] = in[0] / max;
255 out[1] = in[1] / max;
256 out[2] = in[2] / max;
257 }
258 return max;
259}

262/*
263=====================
264PlaneFromPoints
265
266Returns false if the triangle is degenrate.
267The normal will point out of the clock for clockwise ordered points
268=====================
269*/
270qboolean PlaneFromPoints( vec4_t plane, const vec3_t a, const vec3_t b, const vec3_t c ) {
271 vec3_t d1, d2;
272
273 VectorSubtract( b, a, d1 );
274 VectorSubtract( c, a, d2 );
275 CrossProduct( d2, d1, plane );
276 if ( VectorNormalize( plane ) == 0 ) {
277 return qfalse;
278 }
279
280 plane[3] = DotProduct( a, plane );
281 return qtrue;
282}

284/*
285===============
286RotatePointAroundVector
287
288This is not implemented very well...
289===============
290*/
291void RotatePointAroundVector( vec3_t dst, const vec3_t dir, const vec3_t point,
292 float degrees ) {
293 float m[3][3];
294 float im[3][3];
295 float zrot[3][3];
296 float tmpmat[3][3];
297 float rot[3][3];
298 int i;
299 vec3_t vr, vup, vf;
300 float rad;
301
302 vf[0] = dir[0];
303 vf[1] = dir[1];
304 vf[2] = dir[2];
305
306 PerpendicularVector( vr, dir );
307 CrossProduct( vr, vf, vup );
308
309 m[0][0] = vr[0];
310 m[1][0] = vr[1];
311 m[2][0] = vr[2];
312
313 m[0][1] = vup[0];
314 m[1][1] = vup[1];
315 m[2][1] = vup[2];
316
317 m[0][2] = vf[0];
318 m[1][2] = vf[1];
319 m[2][2] = vf[2];
320
321 memcpy( im, m, sizeof( im ) );
322
323 im[0][1] = m[1][0];
324 im[0][2] = m[2][0];
325 im[1][0] = m[0][1];
326 im[1][2] = m[2][1];
327 im[2][0] = m[0][2];
328 im[2][1] = m[1][2];
329
330 memset( zrot, 0, sizeof( zrot ) );
331 zrot[0][0] = zrot[1][1] = zrot[2][2] = 1.0F;
332
333 rad = DEG2RAD( degrees );
334 zrot[0][0] = cos( rad );
335 zrot[0][1] = sin( rad );
336 zrot[1][0] = -sin( rad );
337 zrot[1][1] = cos( rad );
338
339 MatrixMultiply( m, zrot, tmpmat );
340 MatrixMultiply( tmpmat, im, rot );
341
342 for ( i = 0; i < 3; i++ ) {
343 dst[i] = rot[i][0] * point[0] + rot[i][1] * point[1] + rot[i][2] * point[2];
344 }
345}

347/*
348===============
349RotateAroundDirection
350===============
351*/
352void RotateAroundDirection( vec3_t axis[3], float yaw ) {
353
354 // create an arbitrary axis[1]
355 PerpendicularVector( axis[1], axis[0] );
356
357 // rotate it around axis[0] by yaw
358 if ( yaw ) {
359 vec3_t temp;
360
361 VectorCopy( axis[1], temp );
362 RotatePointAroundVector( axis[1], axis[0], temp, yaw );
363 }
364
365 // cross to get axis[2]
366 CrossProduct( axis[0], axis[1], axis[2] );
367}

371void vectoangles( const vec3_t value1, vec3_t angles ) {
372 float forward;
373 float yaw, pitch;
374
375 if ( value1[1] == 0 && value1[0] == 0 ) {
376 yaw = 0;
377 if ( value1[2] > 0 ) {
378 pitch = 90;
379 }
380 else {
381 pitch = 270;
382 }
383 }
384 else {
385 if ( value1[0] ) {
386 yaw = ( atan2 ( value1[1], value1[0] ) * 180 / M_PI );
387 }
388 else if ( value1[1] > 0 ) {
389 yaw = 90;
390 }
391 else {
392 yaw = 270;
393 }
394 if ( yaw < 0 ) {
395 yaw += 360;
396 }
397
398 forward = sqrt ( value1[0]*value1[0] + value1[1]*value1[1] );
399 pitch = ( atan2(value1[2], forward) * 180 / M_PI );
400 if ( pitch < 0 ) {
401 pitch += 360;
402 }
403 }
404
405 angles[PITCH] = -pitch;
406 angles[YAW] = yaw;
407 angles[ROLL] = 0;
408}

442void ProjectPointOnPlane( vec3_t dst, const vec3_t p, const vec3_t normal )
443{
444 float d;
445 vec3_t n;
446 float inv_denom;
447
448 inv_denom = DotProduct( normal, normal );
449#ifndef Q3_VM
450 assert( Q_fabs(inv_denom) != 0.0f ); // zero vectors get here
451#endif
452 inv_denom = 1.0f / inv_denom;
453
454 d = DotProduct( normal, p ) * inv_denom;
455
456 n[0] = normal[0] * inv_denom;
457 n[1] = normal[1] * inv_denom;
458 n[2] = normal[2] * inv_denom;
459
460 dst[0] = p[0] - d * n[0];
461 dst[1] = p[1] - d * n[1];
462 dst[2] = p[2] - d * n[2];
463}

465/*
466================
467MakeNormalVectors
468
469Given a normalized forward vector, create two
470other perpendicular vectors
471================
472*/
473void MakeNormalVectors( const vec3_t forward, vec3_t right, vec3_t up) {
474 float d;
475
476 // this rotate and negate guarantees a vector
477 // not colinear with the original
478 right[1] = -forward[0];
479 right[2] = forward[1];
480 right[0] = forward[2];
481
482 d = DotProduct (right, forward);
483 VectorMA (right, -d, forward, right);
484 VectorNormalize (right);
485 CrossProduct (right, forward, up);
486}

499/*
500** float q_rsqrt( float number )
501*/
502float Q_rsqrt( float number )
503{
504 floatint_t t;
505 float x2, y;
506 const float threehalfs = 1.5F;
507
508 x2 = number * 0.5F;
509 t.f = number;
510 t.i = 0x5f3759df - ( t.i >> 1 ); // what the fuck?
511 y = t.f;
512 y = y * ( threehalfs - ( x2 * y * y ) ); // 1st iteration
513// y = y * ( threehalfs - ( x2 * y * y ) ); // 2nd iteration, this can be removed
514
515 return y;
516}

645/*
646==================
647BoxOnPlaneSide
648
649Returns 1, 2, or 1 + 2
650==================
651*/
652int BoxOnPlaneSide(vec3_t emins, vec3_t emaxs, struct cplane_s *p)
653{
654 float dist[2];
655 int sides, b, i;
656
657 // fast axial cases
658 if (p->type < 3)
659 {
660 if (p->dist <= emins[p->type])
661 return 1;
662 if (p->dist >= emaxs[p->type])
663 return 2;
664 return 3;
665 }
666
667 // general case
668 dist[0] = dist[1] = 0;
669 if (p->signbits < 8) // >= 8: default case is original code (dist[0]=dist[1]=0)
670 {
671 for (i=0 ; i<3 ; i++)
672 {
673 b = (p->signbits >> i) & 1;
674 dist[ b] += p->normal[i]*emaxs[i];
675 dist[!b] += p->normal[i]*emins[i];
676 }
677 }
678
679 sides = 0;
680 if (dist[0] >= p->dist)
681 sides = 1;
682 if (dist[1] < p->dist)
683 sides |= 2;
684
685 return sides;
686}

785vec_t VectorNormalize( vec3_t v ) {
786 // NOTE: TTimo - Apple G4 altivec source uses double?
787 float length, ilength;
788
789 length = v[0]*v[0] + v[1]*v[1] + v[2]*v[2];
790
791 if ( length ) {
792 /* writing it this way allows gcc to recognize that rsqrt can be used */
793 ilength = 1/(float)sqrt (length);
794 /* sqrt(length) = length * (1 / sqrt(length)) */
795 length *= ilength;
796 v[0] *= ilength;
797 v[1] *= ilength;
798 v[2] *= ilength;
799 }
800
801 return length;
802}

927void AngleVectors( const vec3_t angles, vec3_t forward, vec3_t right, vec3_t up) {
928 float angle;
929 static float sr, sp, sy, cr, cp, cy;
930 // static to help MS compiler fp bugs
931
932 angle = angles[YAW] * (M_PI*2 / 360);
933 sy = sin(angle);
934 cy = cos(angle);
935 angle = angles[PITCH] * (M_PI*2 / 360);
936 sp = sin(angle);
937 cp = cos(angle);
938 angle = angles[ROLL] * (M_PI*2 / 360);
939 sr = sin(angle);
940 cr = cos(angle);
941
942 if (forward)
943 {
944 forward[0] = cp*cy;
945 forward[1] = cp*sy;
946 forward[2] = -sp;
947 }
948 if (right)
949 {
950 right[0] = (-1*sr*sp*cy+-1*cr*-sy);
951 right[1] = (-1*sr*sp*sy+-1*cr*cy);
952 right[2] = -1*sr*cp;
953 }
954 if (up)
955 {
956 up[0] = (cr*sp*cy+-sr*-sy);
957 up[1] = (cr*sp*sy+-sr*cy);
958 up[2] = cr*cp;
959 }
960}

962/*
963** assumes "src" is normalized
964*/
965void PerpendicularVector( vec3_t dst, const vec3_t src )
966{
967 int pos;
968 int i;
969 float minelem = 1.0F;
970 vec3_t tempvec;
971
972 /*
973 ** find the smallest magnitude axially aligned vector
974 */
975 for ( pos = 0, i = 0; i < 3; i++ )
976 {
977 if ( fabs( src[i] ) < minelem )
978 {
979 pos = i;
980 minelem = fabs( src[i] );
981 }
982 }
983 tempvec[0] = tempvec[1] = tempvec[2] = 0.0F;
984 tempvec[pos] = 1.0F;
985
986 /*
987 ** project the point onto the plane defined by src
988 */
989 ProjectPointOnPlane( dst, tempvec, src );
990
991 /*
992 ** normalize the result
993 */
994 VectorNormalize( dst );
995}

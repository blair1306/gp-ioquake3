132static void (*rd_flush)( char *buffer );
133
134void Com_BeginRedirect (char *buffer, int buffersize, void (*flush)( char *) )
135{
136 if (!buffer || !buffersize || !flush)
137 return;
138 rd_buffer = buffer;
139 rd_buffersize = buffersize;
140 rd_flush = flush;
141
142 *rd_buffer = 0;
143}
144
145void Com_EndRedirect (void)
146{
147 if ( rd_flush ) {
148 rd_flush(rd_buffer);
149 }
150
151 rd_buffer = NULL;
152 rd_buffersize = 0;
153 rd_flush = NULL;
154}

156/*
157=============
158Com_Printf
159
160Both client and server can use this, and it will output
161to the apropriate place.
162
163A raw string should NEVER be passed as fmt, because of "%f" type crashers.
164=============
165*/
166void QDECL Com_Printf( const char *fmt, ... ) {
167 va_list argptr;
168 char msg[MAXPRINTMSG];
169 static qboolean opening_qconsole = qfalse;
170
171
172 va_start (argptr,fmt);
173 Q_vsnprintf (msg, sizeof(msg), fmt, argptr);
174 va_end (argptr);
175
176 if ( rd_buffer ) {
177 if ((strlen (msg) + strlen(rd_buffer)) > (rd_buffersize - 1)) {
178 rd_flush(rd_buffer);
179 *rd_buffer = 0;
180 }
181 Q_strcat(rd_buffer, rd_buffersize, msg);
182 // TTimo nooo .. that would defeat the purpose
183 //rd_flush(rd_buffer);
184 //*rd_buffer = 0;
185 return;
186 }
187
188#ifndef DEDICATED
189 CL_ConsolePrint( msg );
190#endif
191
192 // echo to dedicated console and early console
193 Sys_Print( msg );
194
195 // logfile
196 if ( com_logfile && com_logfile->integer ) {
197 // TTimo: only open the qconsole.log if the filesystem is in an initialized state
198 // also, avoid recursing in the qconsole.log opening (i.e. if fs_debug is on)
199 if ( !logfile && FS_Initialized() && !opening_qconsole) {
200 struct tm *newtime;
201 time_t aclock;
202
203 opening_qconsole = qtrue;
204
205 time( &aclock );
206 newtime = localtime( &aclock );
207
208 logfile = FS_FOpenFileWrite( "qconsole.log" );
209
210 if(logfile)
211 {
212 Com_Printf( "logfile opened on %s\n", asctime( newtime ) );
213
214 if ( com_logfile->integer > 1 )
215 {
216 // force it to not buffer so we get valid
217 // data even if we are crashing
218 FS_ForceFlush(logfile);
219 }
220 }
221 else
222 {
223 Com_Printf("Opening qconsole.log failed!\n");
224 Cvar_SetValue("logfile", 0);
225 }
226
227 opening_qconsole = qfalse;
228 }
229 if ( logfile && FS_Initialized()) {
230 FS_Write(msg, strlen(msg), logfile);
231 }
232 }
233}

258/*
259=============
260Com_Error
261
262Both client and server can use this, and it will
263do the appropriate thing.
264=============
265*/
266void QDECL Com_Error( int code, const char *fmt, ... ) {
267 va_list argptr;
268 static int lastErrorTime;
269 static int errorCount;
270 int currentTime;
271 qboolean restartClient;
272
273 if(com_errorEntered)
274 Sys_Error("recursive error after: %s", com_errorMessage);
275
276 com_errorEntered = qtrue;
277
278 Cvar_Set("com_errorCode", va("%i", code));
279
280 // when we are running automated scripts, make sure we
281 // know if anything failed
282 if ( com_buildScript && com_buildScript->integer ) {
283 code = ERR_FATAL;
284 }
285
286 // if we are getting a solid stream of ERR_DROP, do an ERR_FATAL
287 currentTime = Sys_Milliseconds();
288 if ( currentTime - lastErrorTime < 100 ) {
289 if ( ++errorCount > 3 ) {
290 code = ERR_FATAL;
291 }
292 } else {
293 errorCount = 0;
294 }
295 lastErrorTime = currentTime;
296
297 va_start (argptr,fmt);
298 Q_vsnprintf (com_errorMessage, sizeof(com_errorMessage),fmt,argptr);
299 va_end (argptr);
300
301 if (code != ERR_DISCONNECT && code != ERR_NEED_CD)
302 Cvar_Set("com_errorMessage", com_errorMessage);
303
304 restartClient = com_gameClientRestarting && !( com_cl_running && com_cl_running->integer );
305
306 com_gameRestarting = qfalse;
307 com_gameClientRestarting = qfalse;
308
309 if (code == ERR_DISCONNECT || code == ERR_SERVERDISCONNECT) {
310 VM_Forced_Unload_Start();
311 SV_Shutdown( "Server disconnected" );
312 if ( restartClient ) {
313 CL_Init();
314 }
315 CL_Disconnect( qtrue );
316 CL_FlushMemory( );
317 VM_Forced_Unload_Done();
318 // make sure we can get at our local stuff
319 FS_PureServerSetLoadedPaks("", "");
320 com_errorEntered = qfalse;
321 longjmp (abortframe, -1);
322 } else if (code == ERR_DROP) {
323 Com_Printf ("********************\nERROR: %s\n********************\n", com_errorMessage);
324 VM_Forced_Unload_Start();
325 SV_Shutdown (va("Server crashed: %s", com_errorMessage));
326 if ( restartClient ) {
327 CL_Init();
328 }
329 CL_Disconnect( qtrue );
330 CL_FlushMemory( );
331 VM_Forced_Unload_Done();
332 FS_PureServerSetLoadedPaks("", "");
333 com_errorEntered = qfalse;
334 longjmp (abortframe, -1);
335 } else if ( code == ERR_NEED_CD ) {
336 VM_Forced_Unload_Start();
337 SV_Shutdown( "Server didn't have CD" );
338 if ( restartClient ) {
339 CL_Init();
340 }
341 if ( com_cl_running && com_cl_running->integer ) {
342 CL_Disconnect( qtrue );
343 CL_FlushMemory( );
344 VM_Forced_Unload_Done();
345 CL_CDDialog();
346 } else {
347 Com_Printf("Server didn't have CD\n" );
348 VM_Forced_Unload_Done();
349 }
350
351 FS_PureServerSetLoadedPaks("", "");
352
353 com_errorEntered = qfalse;
354 longjmp (abortframe, -1);
355 } else {
356 VM_Forced_Unload_Start();
357 CL_Shutdown(va("Client fatal crashed: %s", com_errorMessage), qtrue, qtrue);
358 SV_Shutdown(va("Server fatal crashed: %s", com_errorMessage));
359 VM_Forced_Unload_Done();
360 }
361
362 Com_Shutdown ();
363
364 Sys_Error ("%s", com_errorMessage);
365}

368/*
369=============
370Com_Quit_f
371
372Both client and server can use this, and it will
373do the apropriate things.
374=============
375*/
376void Com_Quit_f( void ) {
377 // don't try to shutdown if we are in a recursive error
378 char *p = Cmd_Args( );
379 if ( !com_errorEntered ) {
380 // Some VMs might execute "quit" command directly,
381 // which would trigger an unload of active VM error.
382 // Sys_Quit will kill this process anyways, so
383 // a corrupt call stack makes no difference
384 VM_Forced_Unload_Start();
385 SV_Shutdown(p[0] ? p : "Server quit");
386 CL_Shutdown(p[0] ? p : "Client quit", qtrue, qtrue);
387 VM_Forced_Unload_Done();
388 Com_Shutdown ();
389 FS_Shutdown(qtrue);
390 }
391 Sys_Quit ();
392}

396/*
397============================================================================
398
399COMMAND LINE FUNCTIONS
400
401+ characters seperate the commandLine string into multiple console
402command lines.
403
404All of these are valid:
405
406quake3 +set test blah +map test
407quake3 set test blah+map test
408quake3 set test blah + map test
409
410============================================================================
411*/
412
413#define MAX_CONSOLE_LINES 32
414int com_numConsoleLines;
415char *com_consoleLines[MAX_CONSOLE_LINES];
416
417/*
418==================
419Com_ParseCommandLine
420
421Break it up into multiple console lines
422==================
423*/
424void Com_ParseCommandLine( char *commandLine ) {
425 int inq = 0;
426 com_consoleLines[0] = commandLine;
427 com_numConsoleLines = 1;
428
429 while ( *commandLine ) {
430 if (*commandLine == '"') {
431 inq = !inq;
432 }
433 // look for a + separating character
434 // if commandLine came from a file, we might have real line seperators
435 if ( (*commandLine == '+' && !inq) || *commandLine == '\n' || *commandLine == '\r' ) {
436 if ( com_numConsoleLines == MAX_CONSOLE_LINES ) {
437 return;
438 }
439 com_consoleLines[com_numConsoleLines] = commandLine + 1;
440 com_numConsoleLines++;
441 *commandLine = 0;
442 }
443 commandLine++;
444 }
445}

473Com_StartupVariable
474
475Searches for command line parameters that are set commands.
476If match is not NULL, only that cvar will be looked for.
477That is necessary because cddir and basedir need to be set

483 int i;
484 char *s;
485
486 for (i=0 ; i < com_numConsoleLines ; i++) {

488 if ( strcmp( Cmd_Argv(0), "set" ) ) {

490 }

492 s = Cmd_Argv(1);

494 if(!match || !strcmp(s, match))
495 {
496 if(Cvar_Flags(s) == CVAR_NONEXISTENT)
497 Cvar_Get(s, Cmd_ArgsFrom(2), CVAR_USER_CREATED);
498 else
499 Cvar_Set2(s, Cmd_ArgsFrom(2), qfalse);
500 }
501 }

585/*
586============
587Com_StringContains
588============
589*/
590char *Com_StringContains(char *str1, char *str2, int casesensitive) {
591 int len, i, j;
592
593 len = strlen(str1) - strlen(str2);
594 for (i = 0; i <= len; i++, str1++) {
595 for (j = 0; str2[j]; j++) {
596 if (casesensitive) {
597 if (str1[j] != str2[j]) {
598 break;
599 }
600 }
601 else {
602 if (toupper(str1[j]) != toupper(str2[j])) {
603 break;
604 }
605 }
606 }
607 if (!str2[j]) {
608 return str1;
609 }
610 }
611 return NULL;
612}

614/*
615============
616Com_Filter
617============
618*/
619int Com_Filter(char *filter, char *name, int casesensitive)
620{
621 char buf[MAX_TOKEN_CHARS];
622 char *ptr;
623 int i, found;
624
625 while(*filter) {
626 if (*filter == '*') {
627 filter++;
628 for (i = 0; *filter; i++) {
629 if (*filter == '*' || *filter == '?') break;
630 buf[i] = *filter;
631 filter++;
632 }
633 buf[i] = '\0';
634 if (strlen(buf)) {
635 ptr = Com_StringContains(name, buf, casesensitive);
636 if (!ptr) return qfalse;
637 name = ptr + strlen(buf);
638 }
639 }
640 else if (*filter == '?') {
641 filter++;
642 name++;
643 }
644 else if (*filter == '[' && *(filter+1) == '[') {
645 filter++;
646 }
647 else if (*filter == '[') {
648 filter++;
649 found = qfalse;
650 while(*filter && !found) {
651 if (*filter == ']' && *(filter+1) != ']') break;
652 if (*(filter+1) == '-' && *(filter+2) && (*(filter+2) != ']' || *(filter+3) == ']')) {
653 if (casesensitive) {
654 if (*name >= *filter && *name <= *(filter+2)) found = qtrue;
655 }
656 else {
657 if (toupper(*name) >= toupper(*filter) &&
658 toupper(*name) <= toupper(*(filter+2))) found = qtrue;
659 }
660 filter += 3;
661 }
662 else {
663 if (casesensitive) {
664 if (*filter == *name) found = qtrue;
665 }
666 else {
667 if (toupper(*filter) == toupper(*name)) found = qtrue;
668 }
669 filter++;
670 }
671 }
672 if (!found) return qfalse;
673 while(*filter) {
674 if (*filter == ']' && *(filter+1) != ']') break;
675 filter++;
676 }
677 filter++;
678 name++;
679 }
680 else {
681 if (casesensitive) {
682 if (*filter != *name) return qfalse;
683 }
684 else {
685 if (toupper(*filter) != toupper(*name)) return qfalse;
686 }
687 filter++;
688 name++;
689 }
690 }
691 return qtrue;
692}

694/*
695============
696Com_FilterPath
697============
698*/
699int Com_FilterPath(char *filter, char *name, int casesensitive)
700{
701 int i;
702 char new_filter[MAX_QPATH];
703 char new_name[MAX_QPATH];
704
705 for (i = 0; i < MAX_QPATH-1 && filter[i]; i++) {
706 if ( filter[i] == '\\' || filter[i] == ':' ) {
707 new_filter[i] = '/';
708 }
709 else {
710 new_filter[i] = filter[i];
711 }
712 }
713 new_filter[i] = '\0';
714 for (i = 0; i < MAX_QPATH-1 && name[i]; i++) {
715 if ( name[i] == '\\' || name[i] == ':' ) {
716 new_name[i] = '/';
717 }
718 else {
719 new_name[i] = name[i];
720 }
721 }
722 new_name[i] = '\0';
723 return Com_Filter(new_filter, new_name, casesensitive);
724}

754/*
755==============================================================================
756
757 ZONE MEMORY ALLOCATION
758
759There is never any space between memblocks, and there will never be two
760contiguous free memblocks.
761
762The rover can be left pointing at a non-empty block
763
764The zone calls are pretty much only used for small strings and structures,
765all big things are allocated on the hunk.
766==============================================================================
767*/
768
769#define ZONEID 0x1d4a11
770#define MINFRAGMENT 64
771
772typedef struct zonedebug_s {
773 char *label;
774 char *file;
775 int line;
776 int allocSize;
777} zonedebug_t;
778
779typedef struct memblock_s {
780 int size; // including the header and possibly tiny fragments
781 int tag; // a tag of 0 is a free block
782 struct memblock_s *next, *prev;
783 int id; // should be ZONEID
784#ifdef ZONE_DEBUG
785 zonedebug_t d;
786#endif
787} memblock_t;
788
789typedef struct {
790 int size; // total bytes malloced, including header
791 int used; // total bytes used
792 memblock_t blocklist; // start / end cap for linked list
793 memblock_t *rover;
794} memzone_t;
795
796// main zone for all "dynamic" memory allocation
797memzone_t *mainzone;
798// we also have a small zone for small allocations that would only
799// fragment the main zone (think of cvar and cmd strings)
800memzone_t *smallzone;
801
802void Z_CheckHeap( void );
803
804/*
805========================
806Z_ClearZone
807========================
808*/
809void Z_ClearZone( memzone_t *zone, int size ) {
810 memblock_t *block;
811
812 // set the entire zone to one free block
813
814 zone->blocklist.next = zone->blocklist.prev = block =
815 (memblock_t *)( (byte *)zone + sizeof(memzone_t) );
816 zone->blocklist.tag = 1; // in use block
817 zone->blocklist.id = 0;
818 zone->blocklist.size = 0;
819 zone->rover = block;
820 zone->size = size;
821 zone->used = 0;
822
823 block->prev = block->next = &zone->blocklist;
824 block->tag = 0; // free block
825 block->id = ZONEID;
826 block->size = size - sizeof(memzone_t);
827}

849Z_Free
850========================
851*/
852void Z_Free( void *ptr ) {
853 memblock_t *block, *other;
854 memzone_t *zone;
855
856 if (!ptr) {
857 Com_Error( ERR_DROP, "Z_Free: NULL pointer" );
858 }
859
860 block = (memblock_t *) ( (byte *)ptr - sizeof(memblock_t));
861 if (block->id != ZONEID) {
862 Com_Error( ERR_FATAL, "Z_Free: freed a pointer without ZONEID" );
863 }
864 if (block->tag == 0) {
865 Com_Error( ERR_FATAL, "Z_Free: freed a freed pointer" );
866 }
867 // if static memory
868 if (block->tag == TAG_STATIC) {
869 return;
870 }
871
872 // check the memory trash tester
873 if ( *(int *)((byte *)block + block->size - 4 ) != ZONEID ) {
874 Com_Error( ERR_FATAL, "Z_Free: memory block wrote past end" );
875 }
876
877 if (block->tag == TAG_SMALL) {
878 zone = smallzone;
879 }
880 else {
881 zone = mainzone;
882 }
883
884 zone->used -= block->size;
885 // set the block to something that should cause problems
886 // if it is referenced...
887 Com_Memset( ptr, 0xaa, block->size - sizeof( *block ) );
888
889 block->tag = 0; // mark as free
890
891 other = block->prev;
892 if (!other->tag) {
893 // merge with previous free block
894 other->size += block->size;
895 other->next = block->next;
896 other->next->prev = other;
897 if (block == zone->rover) {
898 zone->rover = other;
899 }
900 block = other;
901 }
902
903 zone->rover = block;
904
905 other = block->next;
906 if ( !other->tag ) {
907 // merge the next free block onto the end
908 block->size += other->size;
909 block->next = other->next;
910 block->next->prev = block;
911 }
912}

947#ifdef ZONE_DEBUG
948void *Z_TagMallocDebug( int size, int tag, char *label, char *file, int line ) {
949 int allocSize;
950#else
951void *Z_TagMalloc( int size, int tag ) {
952#endif
953 int extra;
954 memblock_t *start, *rover, *new, *base;
955 memzone_t *zone;
956
957 if (!tag) {
958 Com_Error( ERR_FATAL, "Z_TagMalloc: tried to use a 0 tag" );
959 }
960
961 if ( tag == TAG_SMALL ) {
962 zone = smallzone;
963 }
964 else {
965 zone = mainzone;
966 }
967
968#ifdef ZONE_DEBUG
969 allocSize = size;
970#endif
971 //
972 // scan through the block list looking for the first free block
973 // of sufficient size
974 //
975 size += sizeof(memblock_t); // account for size of block header
976 size += 4; // space for memory trash tester
977 size = PAD(size, sizeof(intptr_t)); // align to 32/64 bit boundary
978
979 base = rover = zone->rover;
980 start = base->prev;
981
982 do {
983 if (rover == start) {
984 // scaned all the way around the list
985#ifdef ZONE_DEBUG
986 Z_LogHeap();
987
988 Com_Error(ERR_FATAL, "Z_Malloc: failed on allocation of %i bytes from the %s zone: %s, line: %d (%s)",

990#else
991 Com_Error(ERR_FATAL, "Z_Malloc: failed on allocation of %i bytes from the %s zone",
992 size, zone == smallzone ? "small" : "main");
993#endif
994 return NULL;
995 }
996 if (rover->tag) {
997 base = rover = rover->next;
998 } else {
999 rover = rover->next;
1000 }
1001 } while (base->tag || base->size < size);
1002
1003 //
1004 // found a block big enough
1005 //
1006 extra = base->size - size;
1007 if (extra > MINFRAGMENT) {
1008 // there will be a free fragment after the allocated block
1009 new = (memblock_t *) ((byte *)base + size );
1010 new->size = extra;
1011 new->tag = 0; // free block
1012 new->prev = base;
1013 new->id = ZONEID;
1014 new->next = base->next;
1015 new->next->prev = new;
1016 base->next = new;
1017 base->size = size;
1018 }
1019
1020 base->tag = tag; // no longer a free block
1021
1022 zone->rover = base->next; // next allocation will start looking here
1023 zone->used += base->size; //
1024
1025 base->id = ZONEID;
1026
1027#ifdef ZONE_DEBUG
1028 base->d.label = label;
1029 base->d.file = file;
1030 base->d.line = line;
1031 base->d.allocSize = allocSize;
1032#endif
1033
1034 // marker for memory trash testing
1035 *(int *)((byte *)base + base->size - 4) = ZONEID;
1036
1037 return (void *) ((byte *)base + sizeof(memblock_t));
1038}

1208/*
1209==============================================================================
1210
1211Goals:
1212 reproducable without history effects -- no out of memory errors on weird map to map changes
1213 allow restarting of the client without fragmentation
1214 minimize total pages in use at run time
1215 minimize total pages needed during load time
1216
1217 Single block of memory with stack allocators coming from both ends towards the middle.
1218
1219 One side is designated the temporary memory allocator.
1220
1221 Temporary memory can be allocated and freed in any order.
1222
1223 A highwater mark is kept of the most in use at any time.
1224
1225 When there is no temporary memory allocated, the permanent and temp sides
1226 can be switched, allowing the already touched temp memory to be used for
1227 permanent storage.
1228
1229 Temp memory must never be allocated on two ends at once, or fragmentation
1230 could occur.
1231
1232 If we have any in-use temp memory, additional temp allocations must come from
1233 that side.
1234
1235 If not, we can choose to make either side the new temp side and push future
1236 permanent allocations to the other side. Permanent allocations should be
1237 kept on the side that has the current greatest wasted highwater mark.
1238
1239==============================================================================
1240*/

1781Hunk_AllocateTempMemory
1782
1783This is used by the file loading system.
1784Multiple files can be loaded in temporary memory.
1785When the files-in-use count reaches zero, all temp memory will be deleted
1786=================
1787*/
1788void *Hunk_AllocateTempMemory( int size ) {
1789 void *buf;
1790 hunkHeader_t *hdr;
1791
1792 // return a Z_Malloc'd block if the hunk has not been initialized
1793 // this allows the config and product id files ( journal files too ) to be loaded
1794 // by the file system without redunant routines in the file system utilizing different
1795 // memory systems
1796 if ( s_hunkData == NULL )
1797 {
1798 return Z_Malloc(size);
1799 }
1800
1801 Hunk_SwapBanks();
1802
1803 size = PAD(size, sizeof(intptr_t)) + sizeof( hunkHeader_t );
1804

1811 hunk_temp->temp += size;

1822 buf = (void *)(hdr+1);
1823
1824 hdr->magic = HUNK_MAGIC;
1825 hdr->size = size;

1834Hunk_FreeTempMemory
1835==================
1836*/
1837void Hunk_FreeTempMemory( void *buf ) {

1839
1840 // free with Z_Free if the hunk has not been initialized
1841 // this allows the config and product id files ( journal files too ) to be loaded
1842 // by the file system without redunant routines in the file system utilizing different
1843 // memory systems
1844 if ( s_hunkData == NULL )
1845 {

1865 }
1866 } else {
1867 if ( hdr == (void *)(s_hunkData + s_hunkTotal - hunk_temp->temp ) ) {
1868 hunk_temp->temp -= hdr->size;
1869 } else {
1870 Com_Printf( "Hunk_FreeTempMemory: not the final block\n" );
1871 }
1872 }
1873}

1951/*

1955A time of 0 will get the current time
1956Ptr should either be null, or point to a block of data that can
1957be freed by the game later.
1958================
1959*/
1960void Com_QueueEvent( int time, sysEventType_t type, int value, int value2, int ptrLength, void *ptr )
1961{

1965 if ( type == SE_MOUSE && eventHead != eventTail )

1984 {
1985 Z_Free( ev->evPtr );
1986 }
1987 eventTail++;
1988 }
1989
1990 eventHead++;
1991
1992 if ( time == 0 )

1994 time = Sys_Milliseconds();
1995 }
1996
1997 ev->evTime = time;
1998 ev->evType = type;
1999 ev->evValue = value;
2000 ev->evValue2 = value2;

2874===============

2881
2882 if( !pipefile )
2883 return;

2892 if( buf[ i ] == '\0' )

2903 char tmp = *brk;
2904 *brk = '\0';
2905 Cbuf_ExecuteText( EXEC_APPEND, buf );
2906 *brk = tmp;
2907
2908 accu -= brk - buf;
2909 memmove( buf, brk, accu + 1 );
2910 }
2911 else if( accu >= sizeof( buf ) - 1 ) // full
2912 {
2913 Cbuf_ExecuteText( EXEC_APPEND, buf );

3010*/
3011int Com_ModifyMsec( int msec ) {
3012 int clampTime;
3013
3014 //
3015 // modify time for debugging values
3016 //
3017 if ( com_fixedtime->integer ) {
3018 msec = com_fixedtime->integer;
3019 } else if ( com_timescale->value ) {
3020 msec *= com_timescale->value;
3021 } else if (com_cameraMode->integer) {
3022 msec *= com_timescale->value;
3023 }
3024
3025 // don't let it scale below 1 msec
3026 if ( msec < 1 && com_timescale->value) {
3027 msec = 1;
3028 }
3029
3030 if ( com_dedicated->integer ) {
3031 // dedicated servers don't want to clamp for a much longer
3032 // period, because it would mess up all the client's views
3033 // of time.
3034 if (com_sv_running->integer && msec > 500)
3035 Com_Printf( "Hitch warning: %i msec frame time\n", msec );
3036
3037 clampTime = 5000;
3038 } else
3039 if ( !com_sv_running->integer ) {
3040 // clients of remote servers do not want to clamp time, because
3041 // it would skew their view of the server's time temporarily
3042 clampTime = 5000;
3043 } else {
3044 // for local single player gaming

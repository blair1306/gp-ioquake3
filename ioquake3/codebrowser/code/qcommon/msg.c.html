104// negative bit values include signs
105void MSG_WriteBits( msg_t *msg, int value, int bits ) {
106 int i;
107
108 oldsize += bits;
109
110 if ( msg->overflowed ) {
111 return;
112 }
113
114 if ( bits == 0 || bits < -31 || bits > 32 ) {
115 Com_Error( ERR_DROP, "MSG_WriteBits: bad bits %i", bits );
116 }
117
118 if ( bits < 0 ) {
119 bits = -bits;
120 }
121
122 if ( msg->oob ) {
123 if ( msg->cursize + ( bits >> 3 ) > msg->maxsize ) {
124 msg->overflowed = qtrue;
125 return;
126 }
127
128 if ( bits == 8 ) {
129 msg->data[msg->cursize] = value;
130 msg->cursize += 1;
131 msg->bit += 8;
132 } else if ( bits == 16 ) {
133 short temp = value;
134
135 CopyLittleShort( &msg->data[msg->cursize], &temp );
136 msg->cursize += 2;
137 msg->bit += 16;
138 } else if ( bits==32 ) {
139 CopyLittleLong( &msg->data[msg->cursize], &value );
140 msg->cursize += 4;
141 msg->bit += 32;
142 } else {
143 Com_Error( ERR_DROP, "can't write %d bits", bits );
144 }
145 } else {
146 value &= (0xffffffff >> (32 - bits));
147 if ( bits&7 ) {
148 int nbits;
149 nbits = bits&7;
150 if ( msg->bit + nbits > msg->maxsize << 3 ) {
151 msg->overflowed = qtrue;
152 return;
153 }
154 for( i = 0; i < nbits; i++ ) {
155 Huff_putBit( (value & 1), msg->data, &msg->bit );
156 value = (value >> 1);
157 }
158 bits = bits - nbits;
159 }
160 if ( bits ) {
161 for( i = 0; i < bits; i += 8 ) {
162 Huff_offsetTransmit( &msgHuff.compressor, (value & 0xff), msg->data, &msg->bit, msg->maxsize << 3 );
163 value = (value >> 8);
164
165 if ( msg->bit > msg->maxsize << 3 ) {
166 msg->overflowed = qtrue;
167 return;
168 }
169 }
170 }
171 msg->cursize = (msg->bit >> 3) + 1;
172 }
173}
174
175int MSG_ReadBits( msg_t *msg, int bits ) {
176 int value;
177 int get;
178 qboolean sgn;
179 int i, nbits;
180// FILE* fp;
181
182 if ( msg->readcount > msg->cursize ) {
183 return 0;
184 }
185
186 value = 0;
187
188 if ( bits < 0 ) {
189 bits = -bits;
190 sgn = qtrue;
191 } else {
192 sgn = qfalse;
193 }
194
195 if (msg->oob) {
196 if (msg->readcount + (bits>>3) > msg->cursize) {
197 msg->readcount = msg->cursize + 1;
198 return 0;
199 }
200
201 if(bits==8)
202 {
203 value = msg->data[msg->readcount];
204 msg->readcount += 1;
205 msg->bit += 8;
206 }
207 else if(bits==16)
208 {
209 short temp;
210
211 CopyLittleShort(&temp, &msg->data[msg->readcount]);
212 value = temp;
213 msg->readcount += 2;
214 msg->bit += 16;
215 }
216 else if(bits==32)
217 {
218 CopyLittleLong(&value, &msg->data[msg->readcount]);
219 msg->readcount += 4;
220 msg->bit += 32;
221 }
222 else
223 Com_Error(ERR_DROP, "can't read %d bits", bits);
224 } else {
225 nbits = 0;
226 if (bits&7) {
227 nbits = bits&7;
228 if (msg->bit + nbits > msg->cursize << 3) {
229 msg->readcount = msg->cursize + 1;
230 return 0;
231 }
232 for(i=0;i<nbits;i++) {
233 value |= (Huff_getBit(msg->data, &msg->bit)<<i);
234 }
235 bits = bits - nbits;
236 }
237 if (bits) {
238// fp = fopen("c:\\netchan.bin", "a");
239 for(i=0;i<bits;i+=8) {
240 Huff_offsetReceive (msgHuff.decompressor.tree, &get, msg->data, &msg->bit, msg->cursize<<3);
241// fwrite(&get, 1, 1, fp);
242 value |= (get<<(i+nbits));
243
244 if (msg->bit > msg->cursize<<3) {
245 msg->readcount = msg->cursize + 1;
246 return 0;
247 }
248 }
249// fclose(fp);
250 }
251 msg->readcount = (msg->bit>>3)+1;
252 }
253 if ( sgn && bits > 0 && bits < 32 ) {
254 if ( value & ( 1 << ( bits - 1 ) ) ) {
255 value |= -1 ^ ( ( 1 << bits ) - 1 );
256 }
257 }
258
259 return value;
260}

314void MSG_WriteString( msg_t *sb, const char *s ) {
315 if ( !s ) {
316 MSG_WriteData (sb, "", 1);
317 } else {
318 int l,i;
319 char string[MAX_STRING_CHARS];
320
321 l = strlen( s );
322 if ( l >= MAX_STRING_CHARS ) {
323 Com_Printf( "MSG_WriteString: MAX_STRING_CHARS" );
324 MSG_WriteData (sb, "", 1);
325 return;
326 }
327 Q_strncpyz( string, s, sizeof( string ) );
328
329 // get rid of 0x80+ and '%' chars, because old clients don't like them
330 for ( i = 0 ; i < l ; i++ ) {
331 if ( ((byte *)string)[i] > 127 || string[i] == '%' ) {
332 string[i] = '.';
333 }
334 }
335
336 MSG_WriteData (sb, string, l+1);
337 }
338}

447char *MSG_ReadString( msg_t *msg ) {
448 static char string[MAX_STRING_CHARS];
449 int l,c;
450
451 l = 0;
452 do {
453 c = MSG_ReadByte(msg); // use ReadByte so -1 is out of bounds
454 if ( c == -1 || c == 0 ) {
455 break;
456 }
457 // translate all fmt spec to avoid crash bugs
458 if ( c == '%' ) {
459 c = '.';
460 }
461 // don't allow higher ascii values
462 if ( c > 127 ) {
463 c = '.';
464 }
465
466 string[l] = c;
467 l++;
468 } while (l < sizeof(string)-1);
469
470 string[l] = 0;
471
472 return string;
473}

543// a string hasher which gives the same hash value even if the
544// string is later modified via the legacy MSG read/write code
545int MSG_HashKey(const char *string, int maxlen) {
546 int hash, i;
547
548 hash = 0;
549 for (i = 0; i < maxlen && string[i] != '\0'; i++) {
550 if (string[i] & 0x80 || string[i] == '%')
551 hash += '.' * (119 + i);
552 else
553 hash += string[i] * (119 + i);
554 }
555 hash = (hash ^ (hash >> 10) ^ (hash >> 20));
556 return hash;
557}

616int kbitmask[32] = {
617 0x00000001, 0x00000003, 0x00000007, 0x0000000F,
618 0x0000001F, 0x0000003F, 0x0000007F, 0x000000FF,
619 0x000001FF, 0x000003FF, 0x000007FF, 0x00000FFF,
620 0x00001FFF, 0x00003FFF, 0x00007FFF, 0x0000FFFF,
621 0x0001FFFF, 0x0003FFFF, 0x0007FFFF, 0x000FFFFF,
622 0x001FFFFf, 0x003FFFFF, 0x007FFFFF, 0x00FFFFFF,
623 0x01FFFFFF, 0x03FFFFFF, 0x07FFFFFF, 0x0FFFFFFF,
624 0x1FFFFFFF, 0x3FFFFFFF, 0x7FFFFFFF, 0xFFFFFFFF,
625};
626
627void MSG_WriteDeltaKey( msg_t *msg, int key, int oldV, int newV, int bits ) {
628 if ( oldV == newV ) {
629 MSG_WriteBits( msg, 0, 1 );
630 return;
631 }
632 MSG_WriteBits( msg, 1, 1 );
633 MSG_WriteBits( msg, newV ^ key, bits );
634}
635
636int MSG_ReadDeltaKey( msg_t *msg, int key, int oldV, int bits ) {
637 if ( MSG_ReadBits( msg, 1 ) ) {
638 return MSG_ReadBits( msg, bits ) ^ (key & kbitmask[ bits - 1 ]);
639 }
640 return oldV;
641}
642
643void MSG_WriteDeltaKeyFloat( msg_t *msg, int key, float oldV, float newV ) {
644 floatint_t fi;
645 if ( oldV == newV ) {
646 MSG_WriteBits( msg, 0, 1 );
647 return;
648 }
649 fi.f = newV;
650 MSG_WriteBits( msg, 1, 1 );
651 MSG_WriteBits( msg, fi.i ^ key, 32 );
652}

654float MSG_ReadDeltaKeyFloat( msg_t *msg, int key, float oldV ) {

656 floatint_t fi;

658 fi.i = MSG_ReadBits( msg, 32 ) ^ key;
659 return fi.f;
660 }
661 return oldV;

673/*
674=====================
675MSG_WriteDeltaUsercmdKey
676=====================
677*/
678void MSG_WriteDeltaUsercmdKey( msg_t *msg, int key, usercmd_t *from, usercmd_t *to ) {
679 if ( to->serverTime - from->serverTime < 256 ) {
680 MSG_WriteBits( msg, 1, 1 );
681 MSG_WriteBits( msg, to->serverTime - from->serverTime, 8 );
682 } else {
683 MSG_WriteBits( msg, 0, 1 );
684 MSG_WriteBits( msg, to->serverTime, 32 );
685 }
686 if (from->angles[0] == to->angles[0] &&
687 from->angles[1] == to->angles[1] &&
688 from->angles[2] == to->angles[2] &&
689 from->forwardmove == to->forwardmove &&
690 from->rightmove == to->rightmove &&
691 from->upmove == to->upmove &&
692 from->buttons == to->buttons &&
693 from->weapon == to->weapon) {
694 MSG_WriteBits( msg, 0, 1 ); // no change
695 oldsize += 7;
696 return;
697 }
698 key ^= to->serverTime;
699 MSG_WriteBits( msg, 1, 1 );
700 MSG_WriteDeltaKey( msg, key, from->angles[0], to->angles[0], 16 );
701 MSG_WriteDeltaKey( msg, key, from->angles[1], to->angles[1], 16 );
702 MSG_WriteDeltaKey( msg, key, from->angles[2], to->angles[2], 16 );
703 MSG_WriteDeltaKey( msg, key, from->forwardmove, to->forwardmove, 8 );
704 MSG_WriteDeltaKey( msg, key, from->rightmove, to->rightmove, 8 );
705 MSG_WriteDeltaKey( msg, key, from->upmove, to->upmove, 8 );
706 MSG_WriteDeltaKey( msg, key, from->buttons, to->buttons, 16 );
707 MSG_WriteDeltaKey( msg, key, from->weapon, to->weapon, 8 );
708}

711/*
712=====================
713MSG_ReadDeltaUsercmdKey
714=====================
715*/
716void MSG_ReadDeltaUsercmdKey( msg_t *msg, int key, usercmd_t *from, usercmd_t *to ) {
717 if ( MSG_ReadBits( msg, 1 ) ) {
718 to->serverTime = from->serverTime + MSG_ReadBits( msg, 8 );
719 } else {
720 to->serverTime = MSG_ReadBits( msg, 32 );
721 }
722 if ( MSG_ReadBits( msg, 1 ) ) {
723 key ^= to->serverTime;
724 to->angles[0] = MSG_ReadDeltaKey( msg, key, from->angles[0], 16);
725 to->angles[1] = MSG_ReadDeltaKey( msg, key, from->angles[1], 16);
726 to->angles[2] = MSG_ReadDeltaKey( msg, key, from->angles[2], 16);
727 to->forwardmove = MSG_ReadDeltaKey( msg, key, from->forwardmove, 8);
728 if( to->forwardmove == -128 )
729 to->forwardmove = -127;
730 to->rightmove = MSG_ReadDeltaKey( msg, key, from->rightmove, 8);
731 if( to->rightmove == -128 )
732 to->rightmove = -127;
733 to->upmove = MSG_ReadDeltaKey( msg, key, from->upmove, 8);
734 if( to->upmove == -128 )
735 to->upmove = -127;
736 to->buttons = MSG_ReadDeltaKey( msg, key, from->buttons, 16);
737 to->weapon = MSG_ReadDeltaKey( msg, key, from->weapon, 8);
738 } else {
739 to->angles[0] = from->angles[0];
740 to->angles[1] = from->angles[1];
741 to->angles[2] = from->angles[2];
742 to->forwardmove = from->forwardmove;
743 to->rightmove = from->rightmove;
744 to->upmove = from->upmove;
745 to->buttons = from->buttons;
746 to->weapon = from->weapon;
747 }

844/*
845==================
846MSG_WriteDeltaEntity
847
848Writes part of a packetentities message, including the entity number.
849Can delta from either a baseline or a previous packet_entity
850If to is NULL, a remove entity update will be sent
851If force is not set, then nothing at all will be generated if the entity is
852identical, under the assumption that the in-order delta code will catch it.
853==================
854*/
855void MSG_WriteDeltaEntity( msg_t *msg, struct entityState_s *from, struct entityState_s *to,
856 qboolean force ) {
857 int i, lc;
858 int numFields;
859 netField_t *field;
860 int trunc;
861 float fullFloat;
862 int *fromF, *toF;
863
864 numFields = ARRAY_LEN( entityStateFields );
865
866 // all fields should be 32 bits to avoid any compiler packing issues
867 // the "number" field is not part of the field list
868 // if this assert fails, someone added a field to the entityState_t
869 // struct without updating the message fields
870 assert( numFields + 1 == sizeof( *from )/4 );
871
872 // a NULL to is a delta remove message
873 if ( to == NULL ) {
874 if ( from == NULL ) {
875 return;
876 }
877 MSG_WriteBits( msg, from->number, GENTITYNUM_BITS );
878 MSG_WriteBits( msg, 1, 1 );
879 return;
880 }
881
882 if ( to->number < 0 || to->number >= MAX_GENTITIES ) {
883 Com_Error (ERR_FATAL, "MSG_WriteDeltaEntity: Bad entity number: %i", to->number );
884 }
885
886 lc = 0;
887 // build the change vector as bytes so it is endien independent
888 for ( i = 0, field = entityStateFields ; i < numFields ; i++, field++ ) {
889 fromF = (int *)( (byte *)from + field->offset );
890 toF = (int *)( (byte *)to + field->offset );
891 if ( *fromF != *toF ) {
892 lc = i+1;
893 }
894 }
895
896 if ( lc == 0 ) {
897 // nothing at all changed
898 if ( !force ) {
899 return; // nothing at all
900 }
901 // write two bits for no change
902 MSG_WriteBits( msg, to->number, GENTITYNUM_BITS );
903 MSG_WriteBits( msg, 0, 1 ); // not removed
904 MSG_WriteBits( msg, 0, 1 ); // no delta
905 return;
906 }
907
908 MSG_WriteBits( msg, to->number, GENTITYNUM_BITS );
909 MSG_WriteBits( msg, 0, 1 ); // not removed
910 MSG_WriteBits( msg, 1, 1 ); // we have a delta
911
912 MSG_WriteByte( msg, lc ); // # of changes
913
914 oldsize += numFields;
915
916 for ( i = 0, field = entityStateFields ; i < lc ; i++, field++ ) {
917 fromF = (int *)( (byte *)from + field->offset );
918 toF = (int *)( (byte *)to + field->offset );
919
920 if ( *fromF == *toF ) {
921 MSG_WriteBits( msg, 0, 1 ); // no change
922 continue;
923 }
924
925 MSG_WriteBits( msg, 1, 1 ); // changed
926
927 if ( field->bits == 0 ) {
928 // float
929 fullFloat = *(float *)toF;
930 trunc = (int)fullFloat;
931
932 if (fullFloat == 0.0f) {
933 MSG_WriteBits( msg, 0, 1 );
934 oldsize += FLOAT_INT_BITS;
935 } else {
936 MSG_WriteBits( msg, 1, 1 );
937 if ( trunc == fullFloat && trunc + FLOAT_INT_BIAS >= 0 &&
938 trunc + FLOAT_INT_BIAS < ( 1 << FLOAT_INT_BITS ) ) {
939 // send as small integer
940 MSG_WriteBits( msg, 0, 1 );
941 MSG_WriteBits( msg, trunc + FLOAT_INT_BIAS, FLOAT_INT_BITS );
942 } else {
943 // send as full floating point value
944 MSG_WriteBits( msg, 1, 1 );
945 MSG_WriteBits( msg, *toF, 32 );
946 }
947 }
948 } else {
949 if (*toF == 0) {
950 MSG_WriteBits( msg, 0, 1 );
951 } else {
952 MSG_WriteBits( msg, 1, 1 );
953 // integer
954 MSG_WriteBits( msg, *toF, field->bits );
955 }
956 }
957 }
958}

960/*
961==================
962MSG_ReadDeltaEntity
963
964The entity number has already been read from the message, which
965is how the from state is identified.
966
967If the delta removes the entity, entityState_t->number will be set to MAX_GENTITIES-1
968
969Can go from either a baseline or a previous packet_entity
970==================
971*/
972void MSG_ReadDeltaEntity( msg_t *msg, entityState_t *from, entityState_t *to,
973 int number) {
974 int i, lc;
975 int numFields;
976 netField_t *field;
977 int *fromF, *toF;
978 int print;
979 int trunc;
980 int startBit, endBit;
981
982 if ( number < 0 || number >= MAX_GENTITIES) {
983 Com_Error( ERR_DROP, "Bad delta entity number: %i", number );
984 }
985
986 if ( msg->bit == 0 ) {
987 startBit = msg->readcount * 8 - GENTITYNUM_BITS;
988 } else {
989 startBit = ( msg->readcount - 1 ) * 8 + msg->bit - GENTITYNUM_BITS;
990 }
991
992 // check for a remove
993 if ( MSG_ReadBits( msg, 1 ) == 1 ) {
994 Com_Memset( to, 0, sizeof( *to ) );
995 to->number = MAX_GENTITIES - 1;
996 if ( cl_shownet && ( cl_shownet->integer >= 2 || cl_shownet->integer == -1 ) ) {
997 Com_Printf( "%3i: #%-3i remove\n", msg->readcount, number );
998 }
999 return;
1000 }
1001
1002 // check for no delta
1003 if ( MSG_ReadBits( msg, 1 ) == 0 ) {
1004 *to = *from;
1005 to->number = number;
1006 return;
1007 }
1008
1009 numFields = ARRAY_LEN( entityStateFields );
1010 lc = MSG_ReadByte(msg);
1011
1012 if ( lc > numFields || lc < 0 ) {
1013 Com_Error( ERR_DROP, "invalid entityState field count" );
1014 }
1015
1016 // shownet 2/3 will interleave with other printed info, -1 will
1017 // just print the delta records`
1018 if ( cl_shownet && ( cl_shownet->integer >= 2 || cl_shownet->integer == -1 ) ) {
1019 print = 1;
1020 Com_Printf( "%3i: #%-3i ", msg->readcount, to->number );
1021 } else {
1022 print = 0;
1023 }
1024
1025 to->number = number;
1026
1027 for ( i = 0, field = entityStateFields ; i < lc ; i++, field++ ) {
1028 fromF = (int *)( (byte *)from + field->offset );
1029 toF = (int *)( (byte *)to + field->offset );
1030
1031 if ( ! MSG_ReadBits( msg, 1 ) ) {
1032 // no change
1033 *toF = *fromF;
1034 } else {
1035 if ( field->bits == 0 ) {
1036 // float
1037 if ( MSG_ReadBits( msg, 1 ) == 0 ) {
1038 *(float *)toF = 0.0f;
1039 } else {
1040 if ( MSG_ReadBits( msg, 1 ) == 0 ) {
1041 // integral float
1042 trunc = MSG_ReadBits( msg, FLOAT_INT_BITS );
1043 // bias to allow equal parts positive and negative
1044 trunc -= FLOAT_INT_BIAS;
1045 *(float *)toF = trunc;
1046 if ( print ) {
1047 Com_Printf( "%s:%i ", field->name, trunc );
1048 }
1049 } else {
1050 // full floating point value
1051 *toF = MSG_ReadBits( msg, 32 );
1052 if ( print ) {
1053 Com_Printf( "%s:%f ", field->name, *(float *)toF );
1054 }
1055 }
1056 }
1057 } else {
1058 if ( MSG_ReadBits( msg, 1 ) == 0 ) {
1059 *toF = 0;
1060 } else {
1061 // integer
1062 *toF = MSG_ReadBits( msg, field->bits );
1063 if ( print ) {
1064 Com_Printf( "%s:%i ", field->name, *toF );
1065 }
1066 }
1067 }
1068// pcount[i]++;
1069 }
1070 }
1071 for ( i = lc, field = &entityStateFields[lc] ; i < numFields ; i++, field++ ) {
1072 fromF = (int *)( (byte *)from + field->offset );
1073 toF = (int *)( (byte *)to + field->offset );
1074 // no change
1075 *toF = *fromF;
1076 }
1077
1078 if ( print ) {
1079 if ( msg->bit == 0 ) {
1080 endBit = msg->readcount * 8 - GENTITYNUM_BITS;
1081 } else {
1082 endBit = ( msg->readcount - 1 ) * 8 + msg->bit - GENTITYNUM_BITS;
1083 }
1084 Com_Printf( " (%i bits)\n", endBit - startBit );
1085 }
1086}

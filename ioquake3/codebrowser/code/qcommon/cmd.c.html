43/*
44============
45Cmd_Wait_f
46
47Causes execution of the remainder of the command buffer to be delayed until
48next frame. This allows commands like:
49bind g "cmd use rocket ; +attack ; wait ; -attack ; cmd use blaster"
50============
51*/
52void Cmd_Wait_f( void ) {
53 if ( Cmd_Argc() == 2 ) {
54 cmd_wait = atoi( Cmd_Argv( 1 ) );
55 if ( cmd_wait < 0 )
56 cmd_wait = 1; // ignore the argument
57 } else {
58 cmd_wait = 1;
59 }

83/*
84============
85Cbuf_AddText
86
87Adds command text at the end of the buffer, does NOT add a final \n
88============
89*/
90void Cbuf_AddText( const char *text ) {
91 int l;
92
93 l = strlen (text);
94
95 if (cmd_text.cursize + l >= cmd_text.maxsize)
96 {
97 Com_Printf ("Cbuf_AddText: overflow\n");
98 return;
99 }
100 Com_Memcpy(&cmd_text.data[cmd_text.cursize], text, l);
101 cmd_text.cursize += l;
102}

105/*
106============
107Cbuf_InsertText
108
109Adds command text immediately after the current command
110Adds a \n to the text
111============
112*/
113void Cbuf_InsertText( const char *text ) {
114 int len;
115 int i;
116
117 len = strlen( text ) + 1;
118 if ( len + cmd_text.cursize > cmd_text.maxsize ) {
119 Com_Printf( "Cbuf_InsertText overflowed\n" );
120 return;
121 }
122
123 // move the existing command text
124 for ( i = cmd_text.cursize - 1 ; i >= 0 ; i-- ) {
125 cmd_text.data[ i + len ] = cmd_text.data[ i ];
126 }
127
128 // copy the new text in
129 Com_Memcpy( cmd_text.data, text, len - 1 );
130
131 // add a \n
132 cmd_text.data[ len - 1 ] = '\n';
133
134 cmd_text.cursize += len;
135}

138/*
139============
140Cbuf_ExecuteText
141============
142*/
143void Cbuf_ExecuteText (int exec_when, const char *text)
144{
145 switch (exec_when)
146 {
147 case EXEC_NOW:
148 if (text && strlen(text) > 0) {
149 Com_DPrintf(S_COLOR_YELLOW "EXEC_NOW %s\n", text);
150 Cmd_ExecuteString (text);
151 } else {
152 Cbuf_Execute();
153 Com_DPrintf(S_COLOR_YELLOW "EXEC_NOW %s\n", cmd_text.data);
154 }
155 break;
156 case EXEC_INSERT:
157 Cbuf_InsertText (text);
158 break;
159 case EXEC_APPEND:
160 Cbuf_AddText (text);
161 break;
162 default:
163 Com_Error (ERR_FATAL, "Cbuf_ExecuteText: bad exec_when");
164 }
165}

167/*
168============
169Cbuf_Execute
170============
171*/
172void Cbuf_Execute (void)
173{
174 int i;
175 char *text;
176 char line[MAX_CMD_LINE];
177 int quotes;
178
179 // This will keep // style comments all on one line by not breaking on
180 // a semicolon. It will keep /* ... */ style comments all on one line by not
181 // breaking it for semicolon or newline.
182 qboolean in_star_comment = qfalse;
183 qboolean in_slash_comment = qfalse;
184 while (cmd_text.cursize)
185 {
186 if ( cmd_wait > 0 ) {
187 // skip out while text still remains in buffer, leaving it
188 // for next frame
189 cmd_wait--;
190 break;
191 }
192
193 // find a \n or ; line break or comment: // or /* */
194 text = (char *)cmd_text.data;
195
196 quotes = 0;
197 for (i=0 ; i< cmd_text.cursize ; i++)
198 {
199 if (text[i] == '"')
200 quotes++;
201
202 if ( !(quotes&1)) {
203 if (i < cmd_text.cursize - 1) {
204 if (! in_star_comment && text[i] == '/' && text[i+1] == '/')
205 in_slash_comment = qtrue;
206 else if (! in_slash_comment && text[i] == '/' && text[i+1] == '*')
207 in_star_comment = qtrue;
208 else if (in_star_comment && text[i] == '*' && text[i+1] == '/') {
209 in_star_comment = qfalse;
210 // If we are in a star comment, then the part after it is valid
211 // Note: This will cause it to NUL out the terminating '/'
212 // but ExecuteString doesn't require it anyway.
213 i++;
214 break;
215 }
216 }
217 if (! in_slash_comment && ! in_star_comment && text[i] == ';')
218 break;
219 }
220 if (! in_star_comment && (text[i] == '\n' || text[i] == '\r')) {
221 in_slash_comment = qfalse;
222 break;
223 }
224 }
225
226 if( i >= (MAX_CMD_LINE - 1)) {
227 i = MAX_CMD_LINE - 1;
228 }
229
230 Com_Memcpy (line, text, i);
231 line[i] = 0;
232
233// delete the text from the command buffer and move remaining commands down
234// this is necessary because commands (exec) can insert data at the
235// beginning of the text buffer
236
237 if (i == cmd_text.cursize)
238 cmd_text.cursize = 0;
239 else
240 {
241 i++;
242 cmd_text.cursize -= i;
243 memmove (text, text+i, cmd_text.cursize);
244 }
245
246// execute the command line
247
248 Cmd_ExecuteString (line);
249 }
250}

262/*
263===============
264Cmd_Exec_f
265===============
266*/
267void Cmd_Exec_f( void ) {
268 qboolean quiet;
269 union {
270 char *c;
271 void *v;
272 } f;
273 char filename[MAX_QPATH];
274
275 quiet = !Q_stricmp(Cmd_Argv(0), "execq");
276
277 if (Cmd_Argc () != 2) {
278 Com_Printf ("exec%s <filename> : execute a script file%s\n",
279 quiet ? "q" : "", quiet ? " without notification" : "");
280 return;
281 }
282
283 Q_strncpyz( filename, Cmd_Argv(1), sizeof( filename ) );
284 COM_DefaultExtension( filename, sizeof( filename ), ".cfg" );
285 FS_ReadFile( filename, &f.v);
286 if (!f.c) {
287 Com_Printf ("couldn't exec %s\n", filename);
288 return;
289 }
290 if (!quiet)
291 Com_Printf ("execing %s\n", filename);
292
293 Cbuf_InsertText (f.c);
294
295 FS_FreeFile (f.v);
296}

462/*
463 Replace command separators with space to prevent interpretation
464 This is a hack to protect buggy qvms
465 https://bugzilla.icculus.org/show_bug.cgi?id=3593
466 https://bugzilla.icculus.org/show_bug.cgi?id=4769
467*/
468
469void Cmd_Args_Sanitize(void)
470{
471 int i;
472
473 for(i = 1; i < cmd_argc; i++)
474 {
475 char *c = cmd_argv[i];
476
477 if(strlen(c) > MAX_CVAR_VALUE_STRING - 1)
478 c[MAX_CVAR_VALUE_STRING - 1] = '\0';
479
480 while ((c = strpbrk(c, "\n\r;"))) {
481 *c = ' ';
482 ++c;
483 }
484 }
485}

487/*
488============
489Cmd_TokenizeString
490
491Parses the given string into command line tokens.
492The text is copied to a seperate buffer and 0 characters
493are inserted in the apropriate place, The argv array
494will point into this temporary buffer.
495============
496*/
497// NOTE TTimo define that to track tokenization issues
498//#define TKN_DBG
499static void Cmd_TokenizeString2( const char *text_in, qboolean ignoreQuotes ) {
500 const char *text;
501 char *textOut;
502
503#ifdef TKN_DBG
504 // FIXME TTimo blunt hook to try to find the tokenization of userinfo
505 Com_DPrintf("Cmd_TokenizeString: %s\n", text_in);
506#endif
507
508 // clear previous args
509 cmd_argc = 0;
510
511 if ( !text_in ) {
512 return;
513 }
514
515 Q_strncpyz( cmd_cmd, text_in, sizeof(cmd_cmd) );
516
517 text = text_in;
518 textOut = cmd_tokenized;
519
520 while ( 1 ) {
521 if ( cmd_argc == MAX_STRING_TOKENS ) {
522 return; // this is usually something malicious
523 }
524
525 while ( 1 ) {
526 // skip whitespace
527 while ( *text && *text <= ' ' ) {
528 text++;
529 }
530 if ( !*text ) {
531 return; // all tokens parsed
532 }
533
534 // skip // comments
535 if ( text[0] == '/' && text[1] == '/' ) {
536 return; // all tokens parsed
537 }
538
539 // skip /* */ comments
540 if ( text[0] == '/' && text[1] =='*' ) {
541 while ( *text && ( text[0] != '*' || text[1] != '/' ) ) {
542 text++;
543 }
544 if ( !*text ) {
545 return; // all tokens parsed
546 }
547 text += 2;
548 } else {
549 break; // we are ready to parse a token
550 }
551 }
552
553 // handle quoted strings
554 // NOTE TTimo this doesn't handle \" escaping
555 if ( !ignoreQuotes && *text == '"' ) {
556 cmd_argv[cmd_argc] = textOut;
557 cmd_argc++;
558 text++;
559 while ( *text && *text != '"' ) {
560 *textOut++ = *text++;
561 }
562 *textOut++ = 0;
563 if ( !*text ) {
564 return; // all tokens parsed
565 }
566 text++;
567 continue;
568 }
569
570 // regular token
571 cmd_argv[cmd_argc] = textOut;
572 cmd_argc++;
573
574 // skip until whitespace, quote, or command
575 while ( *text > ' ' ) {
576 if ( !ignoreQuotes && text[0] == '"' ) {
577 break;
578 }
579
580 if ( text[0] == '/' && text[1] == '/' ) {
581 break;
582 }
583
584 // skip /* */ comments
585 if ( text[0] == '/' && text[1] =='*' ) {
586 break;
587 }
588
589 *textOut++ = *text++;
590 }
591
592 *textOut++ = 0;
593
594 if ( !*text ) {
595 return; // all tokens parsed
596 }
597 }
598
599}

624cmd_function_t *Cmd_FindCommand( const char *cmd_name )
625{
626 cmd_function_t *cmd;

633/*
634============
635Cmd_AddCommand
636============
637*/
638void Cmd_AddCommand( const char *cmd_name, xcommand_t function ) {
639 cmd_function_t *cmd;
640
641 // fail if the command already exists
642 if( Cmd_FindCommand( cmd_name ) )
643 {
644 // allow completion-only commands to be silently doubled
645 if( function != NULL )
646 Com_Printf( "Cmd_AddCommand: %s already defined\n", cmd_name );
647 return;
648 }
649
650 // use a small malloc to avoid zone fragmentation
651 cmd = S_Malloc (sizeof(cmd_function_t));
652 cmd->name = CopyString( cmd_name );
653 cmd->function = function;
654 cmd->complete = NULL;
655 cmd->next = cmd_functions;
656 cmd_functions = cmd;
657}

675/*
676============
677Cmd_RemoveCommand
678============
679*/
680void Cmd_RemoveCommand( const char *cmd_name ) {
681 cmd_function_t *cmd, **back;
682
683 back = &cmd_functions;
684 while( 1 ) {
685 cmd = *back;
686 if ( !cmd ) {
687 // command wasn't active
688 return;
689 }
690 if ( !strcmp( cmd_name, cmd->name ) ) {
691 *back = cmd->next;
692 if (cmd->name) {
693 Z_Free(cmd->name);
694 }
695 Z_Free (cmd);
696 return;
697 }
698 back = &cmd->next;
699 }
700}

757/*
758============
759Cmd_ExecuteString
760
761A complete command line has been parsed, so try to execute it
762============
763*/
764void Cmd_ExecuteString( const char *text ) {
765 cmd_function_t *cmd, **prev;
766
767 // execute the command line
768 Cmd_TokenizeString( text );
769 if ( !Cmd_Argc() ) {
770 return; // no tokens
771 }
772
773 // check registered command functions
774 for ( prev = &cmd_functions ; *prev ; prev = &cmd->next ) {
775 cmd = *prev;
776 if ( !Q_stricmp( cmd_argv[0],cmd->name ) ) {
777 // rearrange the links so that the command will be
778 // near the head of the list next time it is used
779 *prev = cmd->next;
780 cmd->next = cmd_functions;
781 cmd_functions = cmd;
782
783 // perform the action
784 if ( !cmd->function ) {
785 // let the cgame or game handle it
786 break;
787 } else {
788 cmd->function ();
789 }
790 return;
791 }
792 }
793
794 // check cvars
795 if ( Cvar_Command() ) {
796 return;
797 }
798
799 // check client game commands
800 if ( com_cl_running && com_cl_running->integer && CL_GameCommand() ) {
801 return;
802 }
803
804 // check server game commands
805 if ( com_sv_running && com_sv_running->integer && SV_GameCommand() ) {
806 return;
807 }
808
809 // check ui commands
810 if ( com_cl_running && com_cl_running->integer && UI_GameCommand() ) {
811 return;
812 }
813
814 // send it as a server command if we are connected
815 // this will usually result in a chat message
816 CL_ForwardCommandToServer ( text );
817}

819/*
820============
821Cmd_List_f
822============
823*/
824void Cmd_List_f (void)
825{
826 cmd_function_t *cmd;
827 int i;
828 char *match;
829
830 if ( Cmd_Argc() > 1 ) {
831 match = Cmd_Argv( 1 );
832 } else {
833 match = NULL;
834 }
835
836 i = 0;
837 for (cmd=cmd_functions ; cmd ; cmd=cmd->next) {
838 if (match && !Com_Filter(match, cmd->name, qfalse)) continue;
839
840 Com_Printf ("%s\n", cmd->name);
841 i++;
842 }
843 Com_Printf ("%i commands\n", i);
844}

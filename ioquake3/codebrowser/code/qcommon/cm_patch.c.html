102/*
103=================
104CM_SignbitsForNormal
105=================
106*/
107static int CM_SignbitsForNormal( vec3_t normal ) {
108 int bits, j;
109
110 bits = 0;
111 for (j=0 ; j<3 ; j++) {
112 if ( normal[j] < 0 ) {
113 bits |= 1<<j;
114 }
115 }
116 return bits;
117}

119/*
120=====================
121CM_PlaneFromPoints
122
123Returns false if the triangle is degenrate.
124The normal will point out of the clock for clockwise ordered points
125=====================
126*/
127static qboolean CM_PlaneFromPoints( vec4_t plane, vec3_t a, vec3_t b, vec3_t c ) {
128 vec3_t d1, d2;
129
130 VectorSubtract( b, a, d1 );
131 VectorSubtract( c, a, d2 );
132 CrossProduct( d2, d1, plane );
133 if ( VectorNormalize( plane ) == 0 ) {
134 return qfalse;
135 }
136
137 plane[3] = DotProduct( a, plane );
138 return qtrue;
139}

150/*
151=================
152CM_NeedsSubdivision
153
154Returns true if the given quadratic curve is not flat enough for our
155collision detection purposes
156=================
157*/
158static qboolean CM_NeedsSubdivision( vec3_t a, vec3_t b, vec3_t c ) {
159 vec3_t cmid;
160 vec3_t lmid;
161 vec3_t delta;
162 float dist;
163 int i;
164
165 // calculate the linear midpoint
166 for ( i = 0 ; i < 3 ; i++ ) {
167 lmid[i] = 0.5*(a[i] + c[i]);
168 }
169
170 // calculate the exact curve midpoint
171 for ( i = 0 ; i < 3 ; i++ ) {
172 cmid[i] = 0.5 * ( 0.5*(a[i] + b[i]) + 0.5*(b[i] + c[i]) );
173 }
174
175 // see if the curve is far enough away from the linear mid
176 VectorSubtract( cmid, lmid, delta );
177 dist = VectorLength( delta );
178
179 return dist >= SUBDIVIDE_DISTANCE;
180}

182/*
183===============
184CM_Subdivide
185
186a, b, and c are control points.
187the subdivided sequence will be: a, out1, out2, out3, c
188===============
189*/
190static void CM_Subdivide( vec3_t a, vec3_t b, vec3_t c, vec3_t out1, vec3_t out2, vec3_t out3 ) {
191 int i;
192
193 for ( i = 0 ; i < 3 ; i++ ) {
194 out1[i] = 0.5 * (a[i] + b[i]);
195 out3[i] = 0.5 * (b[i] + c[i]);
196 out2[i] = 0.5 * (out1[i] + out3[i]);
197 }
198}

200/*
201=================
202CM_TransposeGrid
203
204Swaps the rows and columns in place
205=================
206*/
207static void CM_TransposeGrid( cGrid_t *grid ) {
208 int i, j, l;
209 vec3_t temp;
210 qboolean tempWrap;
211
212 if ( grid->width > grid->height ) {
213 for ( i = 0 ; i < grid->height ; i++ ) {
214 for ( j = i + 1 ; j < grid->width ; j++ ) {
215 if ( j < grid->height ) {
216 // swap the value
217 VectorCopy( grid->points[i][j], temp );
218 VectorCopy( grid->points[j][i], grid->points[i][j] );
219 VectorCopy( temp, grid->points[j][i] );
220 } else {
221 // just copy
222 VectorCopy( grid->points[j][i], grid->points[i][j] );
223 }
224 }
225 }
226 } else {
227 for ( i = 0 ; i < grid->width ; i++ ) {
228 for ( j = i + 1 ; j < grid->height ; j++ ) {
229 if ( j < grid->width ) {
230 // swap the value
231 VectorCopy( grid->points[j][i], temp );
232 VectorCopy( grid->points[i][j], grid->points[j][i] );
233 VectorCopy( temp, grid->points[i][j] );
234 } else {
235 // just copy
236 VectorCopy( grid->points[i][j], grid->points[j][i] );
237 }
238 }
239 }
240 }
241
242 l = grid->width;
243 grid->width = grid->height;
244 grid->height = l;
245
246 tempWrap = grid->wrapWidth;
247 grid->wrapWidth = grid->wrapHeight;
248 grid->wrapHeight = tempWrap;
249}

280/*
281=================
282CM_SubdivideGridColumns
283
284Adds columns as necessary to the grid until
285all the aproximating points are within SUBDIVIDE_DISTANCE
286from the true curve
287=================
288*/
289static void CM_SubdivideGridColumns( cGrid_t *grid ) {
290 int i, j, k;
291
292 for ( i = 0 ; i < grid->width - 2 ; ) {
293 // grid->points[i][x] is an interpolating control point
294 // grid->points[i+1][x] is an aproximating control point
295 // grid->points[i+2][x] is an interpolating control point
296
297 //
298 // first see if we can collapse the aproximating collumn away
299 //
300 for ( j = 0 ; j < grid->height ; j++ ) {
301 if ( CM_NeedsSubdivision( grid->points[i][j], grid->points[i+1][j], grid->points[i+2][j] ) ) {
302 break;
303 }
304 }
305 if ( j == grid->height ) {
306 // all of the points were close enough to the linear midpoints
307 // that we can collapse the entire column away
308 for ( j = 0 ; j < grid->height ; j++ ) {
309 // remove the column
310 for ( k = i + 2 ; k < grid->width ; k++ ) {
311 VectorCopy( grid->points[k][j], grid->points[k-1][j] );
312 }
313 }
314
315 grid->width--;
316
317 // go to the next curve segment
318 i++;
319 continue;
320 }
321
322 //
323 // we need to subdivide the curve
324 //
325 for ( j = 0 ; j < grid->height ; j++ ) {
326 vec3_t prev, mid, next;
327
328 // save the control points now
329 VectorCopy( grid->points[i][j], prev );
330 VectorCopy( grid->points[i+1][j], mid );
331 VectorCopy( grid->points[i+2][j], next );
332
333 // make room for two additional columns in the grid
334 // columns i+1 will be replaced, column i+2 will become i+4
335 // i+1, i+2, and i+3 will be generated
336 for ( k = grid->width - 1 ; k > i + 1 ; k-- ) {
337 VectorCopy( grid->points[k][j], grid->points[k+2][j] );
338 }
339
340 // generate the subdivided points
341 CM_Subdivide( prev, mid, next, grid->points[i+1][j], grid->points[i+2][j], grid->points[i+3][j] );
342 }
343
344 grid->width += 2;
345
346 // the new aproximating point at i+1 may need to be removed
347 // or subdivided farther, so don't advance i
348 }
349}

351/*
352======================
353CM_ComparePoints
354======================
355*/
356#define POINT_EPSILON 0.1
357static qboolean CM_ComparePoints( float *a, float *b ) {
358 float d;
359
360 d = a[0] - b[0];
361 if ( d < -POINT_EPSILON || d > POINT_EPSILON ) {
362 return qfalse;
363 }
364 d = a[1] - b[1];
365 if ( d < -POINT_EPSILON || d > POINT_EPSILON ) {
366 return qfalse;
367 }
368 d = a[2] - b[2];
369 if ( d < -POINT_EPSILON || d > POINT_EPSILON ) {
370 return qfalse;
371 }
372 return qtrue;
373}

375/*
376=================
377CM_RemoveDegenerateColumns
378
379If there are any identical columns, remove them
380=================
381*/
382static void CM_RemoveDegenerateColumns( cGrid_t *grid ) {
383 int i, j, k;
384
385 for ( i = 0 ; i < grid->width - 1 ; i++ ) {
386 for ( j = 0 ; j < grid->height ; j++ ) {
387 if ( !CM_ComparePoints( grid->points[i][j], grid->points[i+1][j] ) ) {
388 break;
389 }
390 }
391
392 if ( j != grid->height ) {
393 continue; // not degenerate
394 }
395
396 for ( j = 0 ; j < grid->height ; j++ ) {
397 // remove the column
398 for ( k = i + 2 ; k < grid->width ; k++ ) {
399 VectorCopy( grid->points[k][j], grid->points[k-1][j] );
400 }
401 }
402 grid->width--;
403
404 // check against the next column
405 i--;
406 }
407}

426/*
427==================
428CM_PlaneEqual
429==================
430*/
431int CM_PlaneEqual(patchPlane_t *p, float plane[4], int *flipped) {
432 float invplane[4];
433
434 if (
435 fabs(p->plane[0] - plane[0]) < NORMAL_EPSILON
436 && fabs(p->plane[1] - plane[1]) < NORMAL_EPSILON
437 && fabs(p->plane[2] - plane[2]) < NORMAL_EPSILON
438 && fabs(p->plane[3] - plane[3]) < DIST_EPSILON )
439 {
440 *flipped = qfalse;
441 return qtrue;
442 }
443
444 VectorNegate(plane, invplane);
445 invplane[3] = -plane[3];
446
447 if (
448 fabs(p->plane[0] - invplane[0]) < NORMAL_EPSILON
449 && fabs(p->plane[1] - invplane[1]) < NORMAL_EPSILON
450 && fabs(p->plane[2] - invplane[2]) < NORMAL_EPSILON
451 && fabs(p->plane[3] - invplane[3]) < DIST_EPSILON )
452 {
453 *flipped = qtrue;
454 return qtrue;
455 }
456
457 return qfalse;
458}

460/*
461==================
462CM_SnapVector
463==================
464*/
465void CM_SnapVector(vec3_t normal) {
466 int i;
467
468 for (i=0 ; i<3 ; i++)
469 {
470 if ( fabs(normal[i] - 1) < NORMAL_EPSILON )
471 {
472 VectorClear (normal);
473 normal[i] = 1;
474 break;
475 }
476 if ( fabs(normal[i] - -1) < NORMAL_EPSILON )
477 {
478 VectorClear (normal);
479 normal[i] = -1;
480 break;
481 }
482 }

485/*
486==================
487CM_FindPlane2
488==================
489*/
490int CM_FindPlane2(float plane[4], int *flipped) {
491 int i;
492
493 // see if the points are close enough to an existing plane
494 for ( i = 0 ; i < numPlanes ; i++ ) {
495 if (CM_PlaneEqual(&planes[i], plane, flipped)) return i;
496 }
497
498 // add a new plane
499 if ( numPlanes == MAX_PATCH_PLANES ) {
500 Com_Error( ERR_DROP, "MAX_PATCH_PLANES" );
501 }
502
503 Vector4Copy( plane, planes[numPlanes].plane );
504 planes[numPlanes].signbits = CM_SignbitsForNormal( plane );
505
506 numPlanes++;
507
508 *flipped = qfalse;
509
510 return numPlanes-1;
511}

513/*
514==================
515CM_FindPlane
516==================
517*/
518static int CM_FindPlane( float *p1, float *p2, float *p3 ) {
519 float plane[4];
520 int i;
521 float d;
522
523 if ( !CM_PlaneFromPoints( plane, p1, p2, p3 ) ) {
524 return -1;
525 }
526
527 // see if the points are close enough to an existing plane
528 for ( i = 0 ; i < numPlanes ; i++ ) {
529 if ( DotProduct( plane, planes[i].plane ) < 0 ) {
530 continue; // allow backwards planes?
531 }
532
533 d = DotProduct( p1, planes[i].plane ) - planes[i].plane[3];
534 if ( d < -PLANE_TRI_EPSILON || d > PLANE_TRI_EPSILON ) {
535 continue;
536 }
537
538 d = DotProduct( p2, planes[i].plane ) - planes[i].plane[3];
539 if ( d < -PLANE_TRI_EPSILON || d > PLANE_TRI_EPSILON ) {
540 continue;
541 }
542
543 d = DotProduct( p3, planes[i].plane ) - planes[i].plane[3];
544 if ( d < -PLANE_TRI_EPSILON || d > PLANE_TRI_EPSILON ) {
545 continue;
546 }
547
548 // found it
549 return i;
550 }
551
552 // add a new plane
553 if ( numPlanes == MAX_PATCH_PLANES ) {
554 Com_Error( ERR_DROP, "MAX_PATCH_PLANES" );
555 }
556
557 Vector4Copy( plane, planes[numPlanes].plane );
558 planes[numPlanes].signbits = CM_SignbitsForNormal( plane );
559
560 numPlanes++;
561
562 return numPlanes-1;
563}

565/*
566==================
567CM_PointOnPlaneSide
568==================
569*/
570static int CM_PointOnPlaneSide( float *p, int planeNum ) {
571 float *plane;
572 float d;
573
574 if ( planeNum == -1 ) {
575 return SIDE_ON;
576 }
577 plane = planes[ planeNum ].plane;
578
579 d = DotProduct( p, plane ) - plane[3];
580
581 if ( d > PLANE_TRI_EPSILON ) {
582 return SIDE_FRONT;
583 }
584
585 if ( d < -PLANE_TRI_EPSILON ) {
586 return SIDE_BACK;
587 }
588
589 return SIDE_ON;
590}

621 vec3_t up;
622 int p;
623

634
635 case 2: // bottom border
636 p1 = grid->points[i][j+1];
637 p2 = grid->points[i+1][j+1];

639 if ( p == -1 ) {
640 return -1;
641 }
642 VectorMA( p1, 4, planes[ p ].plane, up );
643 return CM_FindPlane( p2, p1, up );
644
645 case 3: // left border
646 p1 = grid->points[i][j];
647 p2 = grid->points[i][j+1];
648 p = CM_GridPlane( gridPlanes, i, j, 1 );
649 if ( p == -1 ) {
650 return -1;
651 }

654
655 case 1: // right border

692===================
693CM_SetBorderInward
694===================
695*/
696static void CM_SetBorderInward( facet_t *facet, cGrid_t *grid, int gridPlanes[MAX_GRID_SIZE][MAX_GRID_SIZE][2],
697 int i, int j, int which ) {
698 int k, l;
699 float *points[4];
700 int numPoints;
701
702 switch ( which ) {
703 case -1:
704 points[0] = grid->points[i][j];
705 points[1] = grid->points[i+1][j];
706 points[2] = grid->points[i+1][j+1];
707 points[3] = grid->points[i][j+1];
708 numPoints = 4;
709 break;
710 case 0:
711 points[0] = grid->points[i][j];
712 points[1] = grid->points[i+1][j];
713 points[2] = grid->points[i+1][j+1];
714 numPoints = 3;
715 break;
716 case 1:
717 points[0] = grid->points[i+1][j+1];
718 points[1] = grid->points[i][j+1];
719 points[2] = grid->points[i][j];
720 numPoints = 3;
721 break;
722 default:
723 Com_Error( ERR_FATAL, "CM_SetBorderInward: bad parameter" );
724 numPoints = 0;
725 break;
726 }
727
728 for ( k = 0 ; k < facet->numBorders ; k++ ) {
729 int front, back;
730
731 front = 0;
732 back = 0;
733
734 for ( l = 0 ; l < numPoints ; l++ ) {
735 int side;
736
737 side = CM_PointOnPlaneSide( points[l], facet->borderPlanes[k] );
738 if ( side == SIDE_FRONT ) {
739 front++;
740 } if ( side == SIDE_BACK ) {
741 back++;
742 }
743 }
744
745 if ( front && !back ) {
746 facet->borderInward[k] = qtrue;
747 } else if ( back && !front ) {
748 facet->borderInward[k] = qfalse;
749 } else if ( !front && !back ) {
750 // flat side border
751 facet->borderPlanes[k] = -1;
752 } else {
753 // bisecting side border
754 Com_DPrintf( "WARNING: CM_SetBorderInward: mixed plane sides\n" );
755 facet->borderInward[k] = qfalse;
756 if ( !debugBlock ) {
757 debugBlock = qtrue;
758 VectorCopy( grid->points[i][j], debugBlockPoints[0] );
759 VectorCopy( grid->points[i+1][j], debugBlockPoints[1] );
760 VectorCopy( grid->points[i+1][j+1], debugBlockPoints[2] );
761 VectorCopy( grid->points[i][j+1], debugBlockPoints[3] );
762 }
763 }
764 }

767/*
768==================
769CM_ValidateFacet
770
771If the facet isn't bounded by its borders, we screwed up.
772==================
773*/
774static qboolean CM_ValidateFacet( facet_t *facet ) {
775 float plane[4];
776 int j;
777 winding_t *w;
778 vec3_t bounds[2];
779
780 if ( facet->surfacePlane == -1 ) {
781 return qfalse;
782 }
783
784 Vector4Copy( planes[ facet->surfacePlane ].plane, plane );
785 w = BaseWindingForPlane( plane, plane[3] );
786 for ( j = 0 ; j < facet->numBorders && w ; j++ ) {
787 if ( facet->borderPlanes[j] == -1 ) {
788 FreeWinding( w );
789 return qfalse;
790 }
791 Vector4Copy( planes[ facet->borderPlanes[j] ].plane, plane );
792 if ( !facet->borderInward[j] ) {
793 VectorSubtract( vec3_origin, plane, plane );
794 plane[3] = -plane[3];
795 }
796 ChopWindingInPlace( &w, plane, plane[3], 0.1f );
797 }
798
799 if ( !w ) {
800 return qfalse; // winding was completely chopped away
801 }
802
803 // see if the facet is unreasonably large
804 WindingBounds( w, bounds[0], bounds[1] );
805 FreeWinding( w );
806
807 for ( j = 0 ; j < 3 ; j++ ) {
808 if ( bounds[1][j] - bounds[0][j] > MAX_MAP_BOUNDS ) {
809 return qfalse; // we must be missing a plane
810 }
811 if ( bounds[0][j] >= MAX_MAP_BOUNDS ) {
812 return qfalse;
813 }
814 if ( bounds[1][j] <= -MAX_MAP_BOUNDS ) {
815 return qfalse;
816 }
817 }
818 return qtrue; // winding is fine
819}

821/*
822==================
823CM_AddFacetBevels
824==================
825*/
826void CM_AddFacetBevels( facet_t *facet ) {
827
828 int i, j, k, l;
829 int axis, dir, flipped;
830 float plane[4], d, newplane[4];
831 winding_t *w, *w2;
832 vec3_t mins, maxs, vec, vec2;
833
834 Vector4Copy( planes[ facet->surfacePlane ].plane, plane );
835
836 w = BaseWindingForPlane( plane, plane[3] );
837 for ( j = 0 ; j < facet->numBorders && w ; j++ ) {
838 if (facet->borderPlanes[j] == facet->surfacePlane) continue;
839 Vector4Copy( planes[ facet->borderPlanes[j] ].plane, plane );
840
841 if ( !facet->borderInward[j] ) {
842 VectorSubtract( vec3_origin, plane, plane );
843 plane[3] = -plane[3];
844 }
845
846 ChopWindingInPlace( &w, plane, plane[3], 0.1f );
847 }
848 if ( !w ) {
849 return;
850 }
851
852 WindingBounds(w, mins, maxs);
853
854 // add the axial planes
855 for ( axis = 0 ; axis < 3 ; axis++ )
856 {
857 for ( dir = -1 ; dir <= 1 ; dir += 2 )
858 {
859 VectorClear(plane);
860 plane[axis] = dir;
861 if (dir == 1) {
862 plane[3] = maxs[axis];
863 }
864 else {
865 plane[3] = -mins[axis];
866 }
867 //if it's the surface plane
868 if (CM_PlaneEqual(&planes[facet->surfacePlane], plane, &flipped)) {
869 continue;
870 }
871 // see if the plane is already present
872 for ( i = 0 ; i < facet->numBorders ; i++ ) {
873 if (CM_PlaneEqual(&planes[facet->borderPlanes[i]], plane, &flipped))
874 break;
875 }
876
877 if ( i == facet->numBorders ) {
878 if ( facet->numBorders >= 4 + 6 + 16 ) {
879 Com_Printf( "ERROR: too many bevels\n" );
880 continue;
881 }
882 facet->borderPlanes[facet->numBorders] = CM_FindPlane2(plane, &flipped);
883 facet->borderNoAdjust[facet->numBorders] = 0;
884 facet->borderInward[facet->numBorders] = flipped;
885 facet->numBorders++;
886 }
887 }
888 }
889 //
890 // add the edge bevels
891 //
892 // test the non-axial plane edges
893 for ( j = 0 ; j < w->numpoints ; j++ )
894 {
895 k = (j+1)%w->numpoints;
896 VectorSubtract (w->p[j], w->p[k], vec);
897 //if it's a degenerate edge
898 if (VectorNormalize (vec) < 0.5)
899 continue;
900 CM_SnapVector(vec);
901 for ( k = 0; k < 3 ; k++ )
902 if ( vec[k] == -1 || vec[k] == 1 )
903 break; // axial
904 if ( k < 3 )
905 continue; // only test non-axial edges
906
907 // try the six possible slanted axials from this edge
908 for ( axis = 0 ; axis < 3 ; axis++ )
909 {
910 for ( dir = -1 ; dir <= 1 ; dir += 2 )
911 {
912 // construct a plane
913 VectorClear (vec2);
914 vec2[axis] = dir;
915 CrossProduct (vec, vec2, plane);
916 if (VectorNormalize (plane) < 0.5)
917 continue;
918 plane[3] = DotProduct (w->p[j], plane);
919
920 // if all the points of the facet winding are
921 // behind this plane, it is a proper edge bevel
922 for ( l = 0 ; l < w->numpoints ; l++ )
923 {
924 d = DotProduct (w->p[l], plane) - plane[3];
925 if (d > 0.1)
926 break; // point in front
927 }
928 if ( l < w->numpoints )
929 continue;
930
931 //if it's the surface plane
932 if (CM_PlaneEqual(&planes[facet->surfacePlane], plane, &flipped)) {
933 continue;
934 }
935 // see if the plane is already present
936 for ( i = 0 ; i < facet->numBorders ; i++ ) {
937 if (CM_PlaneEqual(&planes[facet->borderPlanes[i]], plane, &flipped)) {
938 break;
939 }
940 }
941
942 if ( i == facet->numBorders ) {
943 if ( facet->numBorders >= 4 + 6 + 16 ) {
944 Com_Printf( "ERROR: too many bevels\n" );
945 continue;
946 }
947 facet->borderPlanes[facet->numBorders] = CM_FindPlane2(plane, &flipped);
948
949 for ( k = 0 ; k < facet->numBorders ; k++ ) {
950 if (facet->borderPlanes[facet->numBorders] ==
951 facet->borderPlanes[k]) Com_Printf("WARNING: bevel plane already used\n");
952 }
953
954 facet->borderNoAdjust[facet->numBorders] = 0;
955 facet->borderInward[facet->numBorders] = flipped;
956 //
957 w2 = CopyWinding(w);
958 Vector4Copy(planes[facet->borderPlanes[facet->numBorders]].plane, newplane);
959 if (!facet->borderInward[facet->numBorders])
960 {
961 VectorNegate(newplane, newplane);
962 newplane[3] = -newplane[3];
963 } //end if
964 ChopWindingInPlace( &w2, newplane, newplane[3], 0.1f );
965 if (!w2) {
966 Com_DPrintf("WARNING: CM_AddFacetBevels... invalid bevel\n");
967 continue;
968 }
969 else {
970 FreeWinding(w2);
971 }
972 //
973 facet->numBorders++;
974 //already got a bevel
975// break;
976 }
977 }
978 }
979 }
980 FreeWinding( w );
981
982#ifndef BSPC
983 //add opposite plane
984 if ( facet->numBorders >= 4 + 6 + 16 ) {
985 Com_Printf( "ERROR: too many bevels\n" );
986 return;
987 }
988 facet->borderPlanes[facet->numBorders] = facet->surfacePlane;
989 facet->borderNoAdjust[facet->numBorders] = 0;
990 facet->borderInward[facet->numBorders] = qtrue;
991 facet->numBorders++;
992#endif //BSPC
993
994}

1166/*
1167===================
1168CM_GeneratePatchCollide
1169
1170Creates an internal structure that will be used to perform
1171collision detection with a patch mesh.
1172
1173Points is packed as concatenated rows.
1174===================
1175*/
1176struct patchCollide_s *CM_GeneratePatchCollide( int width, int height, vec3_t *points ) {
1177 patchCollide_t *pf;
1178 cGrid_t grid;
1179 int i, j;
1180
1181 if ( width <= 2 || height <= 2 || !points ) {
1182 Com_Error( ERR_DROP, "CM_GeneratePatchFacets: bad parameters: (%i, %i, %p)",
1183 width, height, (void *)points );
1184 }
1185
1186 if ( !(width & 1) || !(height & 1) ) {
1187 Com_Error( ERR_DROP, "CM_GeneratePatchFacets: even sizes are invalid for quadratic meshes" );
1188 }
1189
1190 if ( width > MAX_GRID_SIZE || height > MAX_GRID_SIZE ) {
1191 Com_Error( ERR_DROP, "CM_GeneratePatchFacets: source is > MAX_GRID_SIZE" );
1192 }
1193
1194 // build a grid
1195 grid.width = width;
1196 grid.height = height;
1197 grid.wrapWidth = qfalse;
1198 grid.wrapHeight = qfalse;
1199 for ( i = 0 ; i < width ; i++ ) {
1200 for ( j = 0 ; j < height ; j++ ) {
1201 VectorCopy( points[j*width + i], grid.points[i][j] );
1202 }
1203 }
1204
1205 // subdivide the grid
1206 CM_SetGridWrapWidth( &grid );
1207 CM_SubdivideGridColumns( &grid );
1208 CM_RemoveDegenerateColumns( &grid );
1209
1210 CM_TransposeGrid( &grid );
1211
1212 CM_SetGridWrapWidth( &grid );
1213 CM_SubdivideGridColumns( &grid );
1214 CM_RemoveDegenerateColumns( &grid );
1215
1216 // we now have a grid of points exactly on the curve
1217 // the aproximate surface defined by these points will be
1218 // collided against
1219 pf = Hunk_Alloc( sizeof( *pf ), h_high );
1220 ClearBounds( pf->bounds[0], pf->bounds[1] );
1221 for ( i = 0 ; i < grid.width ; i++ ) {
1222 for ( j = 0 ; j < grid.height ; j++ ) {
1223 AddPointToBounds( grid.points[i][j], pf->bounds[0], pf->bounds[1] );
1224 }
1225 }
1226
1227 c_totalPatchBlocks += ( grid.width - 1 ) * ( grid.height - 1 );
1228
1229 // generate a bsp tree for the surface
1230 CM_PatchCollideFromGrid( &grid, pf );
1231
1232 // expand by one unit for epsilon purposes
1233 pf->bounds[0][0] -= 1;
1234 pf->bounds[0][1] -= 1;
1235 pf->bounds[0][2] -= 1;
1236
1237 pf->bounds[1][0] += 1;
1238 pf->bounds[1][1] += 1;
1239 pf->bounds[1][2] += 1;
1240
1241 return pf;
1242}

1252/*
1253====================
1254CM_TracePointThroughPatchCollide
1255
1256 special case for point traces because the patch collide "brushes" have no volume
1257====================
1258*/
1259void CM_TracePointThroughPatchCollide( traceWork_t *tw, const struct patchCollide_s *pc ) {
1260 qboolean frontFacing[MAX_PATCH_PLANES];
1261 float intersection[MAX_PATCH_PLANES];
1262 float intersect;
1263 const patchPlane_t *planes;
1264 const facet_t *facet;
1265 int i, j, k;
1266 float offset;
1267 float d1, d2;
1268#ifndef BSPC
1269 static cvar_t *cv;
1270#endif //BSPC
1271
1272#ifndef BSPC
1273 if ( !cm_playerCurveClip->integer || !tw->isPoint ) {
1274 return;
1275 }
1276#endif
1277
1278 // determine the trace's relationship to all planes
1279 planes = pc->planes;
1280 for ( i = 0 ; i < pc->numPlanes ; i++, planes++ ) {
1281 offset = DotProduct( tw->offsets[ planes->signbits ], planes->plane );
1282 d1 = DotProduct( tw->start, planes->plane ) - planes->plane[3] + offset;
1283 d2 = DotProduct( tw->end, planes->plane ) - planes->plane[3] + offset;
1284 if ( d1 <= 0 ) {
1285 frontFacing[i] = qfalse;
1286 } else {
1287 frontFacing[i] = qtrue;
1288 }
1289 if ( d1 == d2 ) {
1290 intersection[i] = 99999;
1291 } else {
1292 intersection[i] = d1 / ( d1 - d2 );
1293 if ( intersection[i] <= 0 ) {
1294 intersection[i] = 99999;
1295 }
1296 }
1297 }
1298
1299
1300 // see if any of the surface planes are intersected
1301 facet = pc->facets;
1302 for ( i = 0 ; i < pc->numFacets ; i++, facet++ ) {
1303 if ( !frontFacing[facet->surfacePlane] ) {
1304 continue;
1305 }
1306 intersect = intersection[facet->surfacePlane];
1307 if ( intersect < 0 ) {
1308 continue; // surface is behind the starting point
1309 }
1310 if ( intersect > tw->trace.fraction ) {
1311 continue; // already hit something closer
1312 }
1313 for ( j = 0 ; j < facet->numBorders ; j++ ) {
1314 k = facet->borderPlanes[j];
1315 if ( frontFacing[k] ^ facet->borderInward[j] ) {
1316 if ( intersection[k] > intersect ) {
1317 break;
1318 }
1319 } else {
1320 if ( intersection[k] < intersect ) {
1321 break;
1322 }
1323 }
1324 }
1325 if ( j == facet->numBorders ) {
1326 // we hit this facet
1327#ifndef BSPC
1328 if (!cv) {
1329 cv = Cvar_Get( "r_debugSurfaceUpdate", "1", 0 );
1330 }
1331 if (cv->integer) {
1332 debugPatchCollide = pc;
1333 debugFacet = facet;
1334 }
1335#endif //BSPC
1336 planes = &pc->planes[facet->surfacePlane];
1337
1338 // calculate intersection with a slight pushoff
1339 offset = DotProduct( tw->offsets[ planes->signbits ], planes->plane );
1340 d1 = DotProduct( tw->start, planes->plane ) - planes->plane[3] + offset;
1341 d2 = DotProduct( tw->end, planes->plane ) - planes->plane[3] + offset;
1342 tw->trace.fraction = ( d1 - SURFACE_CLIP_EPSILON ) / ( d1 - d2 );
1343
1344 if ( tw->trace.fraction < 0 ) {
1345 tw->trace.fraction = 0;
1346 }
1347
1348 VectorCopy( planes->plane, tw->trace.plane.normal );
1349 tw->trace.plane.dist = planes->plane[3];
1350 }
1351 }
1352}

1354/*
1355====================
1356CM_CheckFacetPlane
1357====================
1358*/
1359int CM_CheckFacetPlane(float *plane, vec3_t start, vec3_t end, float *enterFrac, float *leaveFrac, int *hit) {
1360 float d1, d2, f;
1361
1362 *hit = qfalse;
1363
1364 d1 = DotProduct( start, plane ) - plane[3];
1365 d2 = DotProduct( end, plane ) - plane[3];
1366
1367 // if completely in front of face, no intersection with the entire facet
1368 if (d1 > 0 && ( d2 >= SURFACE_CLIP_EPSILON || d2 >= d1 ) ) {
1369 return qfalse;
1370 }
1371
1372 // if it doesn't cross the plane, the plane isn't relevent
1373 if (d1 <= 0 && d2 <= 0 ) {
1374 return qtrue;
1375 }
1376
1377 // crosses face
1378 if (d1 > d2) { // enter
1379 f = (d1-SURFACE_CLIP_EPSILON) / (d1-d2);
1380 if ( f < 0 ) {
1381 f = 0;
1382 }
1383 //always favor previous plane hits and thus also the surface plane hit
1384 if (f > *enterFrac) {
1385 *enterFrac = f;
1386 *hit = qtrue;
1387 }
1388 } else { // leave
1389 f = (d1+SURFACE_CLIP_EPSILON) / (d1-d2);
1390 if ( f > 1 ) {
1391 f = 1;
1392 }
1393 if (f < *leaveFrac) {
1394 *leaveFrac = f;
1395 }
1396 }
1397 return qtrue;
1398}

1400/*
1401====================
1402CM_TraceThroughPatchCollide
1403====================
1404*/
1405void CM_TraceThroughPatchCollide( traceWork_t *tw, const struct patchCollide_s *pc ) {
1406 int i, j, hit, hitnum;
1407 float offset, enterFrac, leaveFrac, t;
1408 patchPlane_t *planes;
1409 facet_t *facet;
1410 float plane[4] = {0, 0, 0, 0}, bestplane[4] = {0, 0, 0, 0};
1411 vec3_t startp, endp;
1412#ifndef BSPC
1413 static cvar_t *cv;
1414#endif //BSPC
1415
1416 if ( !CM_BoundsIntersect( tw->bounds[0], tw->bounds[1],
1417 pc->bounds[0], pc->bounds[1] ) ) {
1418 return;
1419 }
1420
1421 if (tw->isPoint) {
1422 CM_TracePointThroughPatchCollide( tw, pc );
1423 return;
1424 }
1425
1426 facet = pc->facets;
1427 for ( i = 0 ; i < pc->numFacets ; i++, facet++ ) {
1428 enterFrac = -1.0;
1429 leaveFrac = 1.0;
1430 hitnum = -1;
1431 //
1432 planes = &pc->planes[ facet->surfacePlane ];
1433 VectorCopy(planes->plane, plane);
1434 plane[3] = planes->plane[3];
1435 if ( tw->sphere.use ) {
1436 // adjust the plane distance apropriately for radius
1437 plane[3] += tw->sphere.radius;
1438
1439 // find the closest point on the capsule to the plane
1440 t = DotProduct( plane, tw->sphere.offset );
1441 if ( t > 0.0f ) {
1442 VectorSubtract( tw->start, tw->sphere.offset, startp );
1443 VectorSubtract( tw->end, tw->sphere.offset, endp );
1444 }
1445 else {
1446 VectorAdd( tw->start, tw->sphere.offset, startp );
1447 VectorAdd( tw->end, tw->sphere.offset, endp );
1448 }
1449 }
1450 else {
1451 offset = DotProduct( tw->offsets[ planes->signbits ], plane);

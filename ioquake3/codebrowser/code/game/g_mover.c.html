45/*
46============
47G_TestEntityPosition
48
49============
50*/
51gentity_t *G_TestEntityPosition( gentity_t *ent ) {
52 trace_t tr;
53 int mask;
54
55 if ( ent->clipmask ) {
56 mask = ent->clipmask;
57 } else {
58 mask = MASK_SOLID;
59 }
60 if ( ent->client ) {
61 trap_Trace( &tr, ent->client->ps.origin, ent->r.mins, ent->r.maxs, ent->client->ps.origin, ent->s.number, mask );
62 } else {
63 trap_Trace( &tr, ent->s.pos.trBase, ent->r.mins, ent->r.maxs, ent->s.pos.trBase, ent->s.number, mask );
64 }
65
66 if (tr.startsolid)
67 return &g_entities[ tr.entityNum ];
68
69 return NULL;
70}

72/*
73================
74G_CreateRotationMatrix
75================
76*/
77void G_CreateRotationMatrix(vec3_t angles, vec3_t matrix[3]) {
78 AngleVectors(angles, matrix[0], matrix[1], matrix[2]);
79 VectorInverse(matrix[1]);
80}
81
82/*
83================
84G_TransposeMatrix
85================
86*/
87void G_TransposeMatrix(vec3_t matrix[3], vec3_t transpose[3]) {
88 int i, j;
89 for (i = 0; i < 3; i++) {
90 for (j = 0; j < 3; j++) {
91 transpose[i][j] = matrix[j][i];
92 }
93 }
94}

96/*
97================
98G_RotatePoint
99================
100*/
101void G_RotatePoint(vec3_t point, vec3_t matrix[3]) {
102 vec3_t tvec;
103
104 VectorCopy(point, tvec);
105 point[0] = DotProduct(matrix[0], tvec);
106 point[1] = DotProduct(matrix[1], tvec);
107 point[2] = DotProduct(matrix[2], tvec);
108}

110/*
111==================
112G_TryPushingEntity
113
114Returns qfalse if the move is blocked
115==================
116*/
117qboolean G_TryPushingEntity( gentity_t *check, gentity_t *pusher, vec3_t move, vec3_t amove ) {
118 vec3_t matrix[3], transpose[3];
119 vec3_t org, org2, move2;
120 gentity_t *block;
121
122 // EF_MOVER_STOP will just stop when contacting another entity
123 // instead of pushing it, but entities can still ride on top of it
124 if ( ( pusher->s.eFlags & EF_MOVER_STOP ) &&
125 check->s.groundEntityNum != pusher->s.number ) {
126 return qfalse;
127 }
128
129 // save off the old position
130 if (pushed_p > &pushed[MAX_GENTITIES]) {
131 G_Error( "pushed_p > &pushed[MAX_GENTITIES]" );
132 }
133 pushed_p->ent = check;
134 VectorCopy (check->s.pos.trBase, pushed_p->origin);
135 VectorCopy (check->s.apos.trBase, pushed_p->angles);
136 if ( check->client ) {
137 pushed_p->deltayaw = check->client->ps.delta_angles[YAW];
138 VectorCopy (check->client->ps.origin, pushed_p->origin);
139 }
140 pushed_p++;
141
142 // try moving the contacted entity
143 // figure movement due to the pusher's amove
144 G_CreateRotationMatrix( amove, transpose );
145 G_TransposeMatrix( transpose, matrix );
146 if ( check->client ) {
147 VectorSubtract (check->client->ps.origin, pusher->r.currentOrigin, org);
148 }
149 else {
150 VectorSubtract (check->s.pos.trBase, pusher->r.currentOrigin, org);
151 }
152 VectorCopy( org, org2 );
153 G_RotatePoint( org2, matrix );
154 VectorSubtract (org2, org, move2);
155 // add movement
156 VectorAdd (check->s.pos.trBase, move, check->s.pos.trBase);
157 VectorAdd (check->s.pos.trBase, move2, check->s.pos.trBase);
158 if ( check->client ) {
159 VectorAdd (check->client->ps.origin, move, check->client->ps.origin);
160 VectorAdd (check->client->ps.origin, move2, check->client->ps.origin);
161 // make sure the client's view rotates when on a rotating mover
162 check->client->ps.delta_angles[YAW] += ANGLE2SHORT(amove[YAW]);
163 }
164
165 // may have pushed them off an edge
166 if ( check->s.groundEntityNum != pusher->s.number ) {
167 check->s.groundEntityNum = ENTITYNUM_NONE;
168 }
169
170 block = G_TestEntityPosition( check );
171 if (!block) {
172 // pushed ok
173 if ( check->client ) {
174 VectorCopy( check->client->ps.origin, check->r.currentOrigin );
175 } else {
176 VectorCopy( check->s.pos.trBase, check->r.currentOrigin );
177 }
178 trap_LinkEntity (check);
179 return qtrue;
180 }
181
182 // if it is ok to leave in the old position, do it
183 // this is only relevent for riding entities, not pushed
184 // Sliding trapdoors can cause this.
185 VectorCopy( (pushed_p-1)->origin, check->s.pos.trBase);
186 if ( check->client ) {
187 VectorCopy( (pushed_p-1)->origin, check->client->ps.origin);
188 }
189 VectorCopy( (pushed_p-1)->angles, check->s.apos.trBase );
190 block = G_TestEntityPosition (check);
191 if ( !block ) {
192 check->s.groundEntityNum = ENTITYNUM_NONE;
193 pushed_p--;
194 return qtrue;
195 }
196
197 // blocked
198 return qfalse;
199}

201/*
202==================
203G_CheckProxMinePosition
204==================
205*/
206qboolean G_CheckProxMinePosition( gentity_t *check ) {
207 vec3_t start, end;
208 trace_t tr;
209
210 VectorMA(check->s.pos.trBase, 0.125, check->movedir, start);
211 VectorMA(check->s.pos.trBase, 2, check->movedir, end);
212 trap_Trace( &tr, start, NULL, NULL, end, check->s.number, MASK_SOLID );
213
214 if (tr.startsolid || tr.fraction < 1)
215 return qfalse;
216
217 return qtrue;
218}

220/*
221==================
222G_TryPushingProxMine
223==================
224*/
225qboolean G_TryPushingProxMine( gentity_t *check, gentity_t *pusher, vec3_t move, vec3_t amove ) {
226 vec3_t forward, right, up;
227 vec3_t org, org2, move2;
228 int ret;
229
230 // we need this for pushing things later
231 VectorSubtract (vec3_origin, amove, org);
232 AngleVectors (org, forward, right, up);
233
234 // try moving the contacted entity
235 VectorAdd (check->s.pos.trBase, move, check->s.pos.trBase);
236
237 // figure movement due to the pusher's amove
238 VectorSubtract (check->s.pos.trBase, pusher->r.currentOrigin, org);
239 org2[0] = DotProduct (org, forward);
240 org2[1] = -DotProduct (org, right);
241 org2[2] = DotProduct (org, up);
242 VectorSubtract (org2, org, move2);
243 VectorAdd (check->s.pos.trBase, move2, check->s.pos.trBase);
244
245 ret = G_CheckProxMinePosition( check );
246 if (ret) {
247 VectorCopy( check->s.pos.trBase, check->r.currentOrigin );
248 trap_LinkEntity (check);
249 }
250 return ret;
251}

255/*
256============
257G_MoverPush
258
259Objects need to be moved back on a failed push,
260otherwise riders would continue to slide.
261If qfalse is returned, *obstacle will be the blocking entity
262============
263*/
264qboolean G_MoverPush( gentity_t *pusher, vec3_t move, vec3_t amove, gentity_t **obstacle ) {
265 int i, e;
266 gentity_t *check;
267 vec3_t mins, maxs;
268 pushed_t *p;
269 int entityList[MAX_GENTITIES];
270 int listedEntities;
271 vec3_t totalMins, totalMaxs;
272
273 *obstacle = NULL;
274
275
276 // mins/maxs are the bounds at the destination
277 // totalMins / totalMaxs are the bounds for the entire move
278 if ( pusher->r.currentAngles[0] || pusher->r.currentAngles[1] || pusher->r.currentAngles[2]
279 || amove[0] || amove[1] || amove[2] ) {
280 float radius;
281
282 radius = RadiusFromBounds( pusher->r.mins, pusher->r.maxs );
283 for ( i = 0 ; i < 3 ; i++ ) {
284 mins[i] = pusher->r.currentOrigin[i] + move[i] - radius;
285 maxs[i] = pusher->r.currentOrigin[i] + move[i] + radius;
286 totalMins[i] = mins[i] - move[i];
287 totalMaxs[i] = maxs[i] - move[i];
288 }
289 } else {
290 for (i=0 ; i<3 ; i++) {
291 mins[i] = pusher->r.absmin[i] + move[i];
292 maxs[i] = pusher->r.absmax[i] + move[i];
293 }
294
295 VectorCopy( pusher->r.absmin, totalMins );
296 VectorCopy( pusher->r.absmax, totalMaxs );
297 for (i=0 ; i<3 ; i++) {
298 if ( move[i] > 0 ) {
299 totalMaxs[i] += move[i];
300 } else {
301 totalMins[i] += move[i];
302 }
303 }
304 }
305
306 // unlink the pusher so we don't get it in the entityList
307 trap_UnlinkEntity( pusher );
308
309 listedEntities = trap_EntitiesInBox( totalMins, totalMaxs, entityList, MAX_GENTITIES );
310
311 // move the pusher to its final position
312 VectorAdd( pusher->r.currentOrigin, move, pusher->r.currentOrigin );
313 VectorAdd( pusher->r.currentAngles, amove, pusher->r.currentAngles );
314 trap_LinkEntity( pusher );
315
316 // see if any solid entities are inside the final position
317 for ( e = 0 ; e < listedEntities ; e++ ) {
318 check = &g_entities[ entityList[ e ] ];
319
320#ifdef MISSIONPACK
321 if ( check->s.eType == ET_MISSILE ) {
322 // if it is a prox mine
323 if ( !strcmp(check->classname, "prox mine") ) {
324 // if this prox mine is attached to this mover try to move it with the pusher
325 if ( check->enemy == pusher ) {
326 if (!G_TryPushingProxMine( check, pusher, move, amove )) {
327 //explode
328 check->s.loopSound = 0;
329 G_AddEvent( check, EV_PROXIMITY_MINE_TRIGGER, 0 );
330 G_ExplodeMissile(check);
331 if (check->activator) {
332 G_FreeEntity(check->activator);
333 check->activator = NULL;
334 }
335 //G_Printf("prox mine explodes\n");
336 }
337 }
338 else {
339 //check if the prox mine is crushed by the mover
340 if (!G_CheckProxMinePosition( check )) {
341 //explode
342 check->s.loopSound = 0;
343 G_AddEvent( check, EV_PROXIMITY_MINE_TRIGGER, 0 );
344 G_ExplodeMissile(check);
345 if (check->activator) {
346 G_FreeEntity(check->activator);
347 check->activator = NULL;
348 }
349 //G_Printf("prox mine explodes\n");
350 }
351 }
352 continue;
353 }
354 }
355#endif
356 // only push items and players
357 if ( check->s.eType != ET_ITEM && check->s.eType != ET_PLAYER && !check->physicsObject ) {
358 continue;
359 }
360
361 // if the entity is standing on the pusher, it will definitely be moved
362 if ( check->s.groundEntityNum != pusher->s.number ) {
363 // see if the ent needs to be tested
364 if ( check->r.absmin[0] >= maxs[0]
365 || check->r.absmin[1] >= maxs[1]
366 || check->r.absmin[2] >= maxs[2]
367 || check->r.absmax[0] <= mins[0]
368 || check->r.absmax[1] <= mins[1]
369 || check->r.absmax[2] <= mins[2] ) {
370 continue;
371 }
372 // see if the ent's bbox is inside the pusher's final position
373 // this does allow a fast moving object to pass through a thin entity...
374 if (!G_TestEntityPosition (check)) {
375 continue;
376 }
377 }
378
379 // the entity needs to be pushed
380 if ( G_TryPushingEntity( check, pusher, move, amove ) ) {
381 continue;
382 }
383
384 // the move was blocked an entity
385
386 // bobbing entities are instant-kill and never get blocked
387 if ( pusher->s.pos.trType == TR_SINE || pusher->s.apos.trType == TR_SINE ) {
388 G_Damage( check, pusher, pusher, NULL, NULL, 99999, 0, MOD_CRUSH );
389 continue;
390 }
391
392
393 // save off the obstacle so we can call the block function (crush, etc)
394 *obstacle = check;
395
396 // move back any entities we already moved
397 // go backwards, so if the same entity was pushed
398 // twice, it goes back to the original position
399 for ( p=pushed_p-1 ; p>=pushed ; p-- ) {
400 VectorCopy (p->origin, p->ent->s.pos.trBase);
401 VectorCopy (p->angles, p->ent->s.apos.trBase);
402 if ( p->ent->client ) {
403 p->ent->client->ps.delta_angles[YAW] = p->deltayaw;
404 VectorCopy (p->origin, p->ent->client->ps.origin);
405 }
406 trap_LinkEntity (p->ent);
407 }
408 return qfalse;
409 }
410
411 return qtrue;
412}

415/*
416=================
417G_MoverTeam
418=================
419*/
420void G_MoverTeam( gentity_t *ent ) {
421 vec3_t move, amove;
422 gentity_t *part, *obstacle;
423 vec3_t origin, angles;
424
425 obstacle = NULL;
426
427 // make sure all team slaves can move before commiting
428 // any moves or calling any think functions
429 // if the move is blocked, all moved objects will be backed out
430 pushed_p = pushed;
431 for (part = ent ; part ; part=part->teamchain) {
432 // get current position
433 BG_EvaluateTrajectory( &part->s.pos, level.time, origin );
434 BG_EvaluateTrajectory( &part->s.apos, level.time, angles );
435 VectorSubtract( origin, part->r.currentOrigin, move );
436 VectorSubtract( angles, part->r.currentAngles, amove );
437 if ( !G_MoverPush( part, move, amove, &obstacle ) ) {
438 break; // move was blocked
439 }
440 }
441
442 if (part) {
443 // go back to the previous position
444 for ( part = ent ; part ; part = part->teamchain ) {
445 part->s.pos.trTime += level.time - level.previousTime;
446 part->s.apos.trTime += level.time - level.previousTime;
447 BG_EvaluateTrajectory( &part->s.pos, level.time, part->r.currentOrigin );
448 BG_EvaluateTrajectory( &part->s.apos, level.time, part->r.currentAngles );
449 trap_LinkEntity( part );
450 }
451
452 // if the pusher has a "blocked" function, call it
453 if (ent->blocked) {
454 ent->blocked( ent, obstacle );
455 }
456 return;
457 }
458
459 // the move succeeded
460 for ( part = ent ; part ; part = part->teamchain ) {
461 // call the reached function if time is at or past end point
462 if ( part->s.pos.trType == TR_LINEAR_STOP ) {
463 if ( level.time >= part->s.pos.trTime + part->s.pos.trDuration ) {
464 if ( part->reached ) {
465 part->reached( part );
466 }
467 }
468 }
469 }
470}

472/*
473================
474G_RunMover
475
476================
477*/
478void G_RunMover( gentity_t *ent ) {
479 // if not a team captain, don't do anything, because
480 // the captain will handle everything
481 if ( ent->flags & FL_TEAMSLAVE ) {
482 return;
483 }
484
485 // if stationary at one of the positions, don't move anything
486 if ( ent->s.pos.trType != TR_STATIONARY || ent->s.apos.trType != TR_STATIONARY ) {
487 G_MoverTeam( ent );
488 }
489
490 // check think function
491 G_RunThink( ent );
492}

497GENERAL MOVERS
498
499Doors, plats, and buttons are all binary (two position) movers
500Pos1 is "at rest", pos2 is "activated"
501============================================================================

504/*
505===============
506SetMoverState
507===============
508*/
509void SetMoverState( gentity_t *ent, moverState_t moverState, int time ) {
510 vec3_t delta;
511 float f;
512
513 ent->moverState = moverState;
514
515 ent->s.pos.trTime = time;
516 switch( moverState ) {
517 case MOVER_POS1:
518 VectorCopy( ent->pos1, ent->s.pos.trBase );
519 ent->s.pos.trType = TR_STATIONARY;
520 break;
521 case MOVER_POS2:
522 VectorCopy( ent->pos2, ent->s.pos.trBase );
523 ent->s.pos.trType = TR_STATIONARY;
524 break;
525 case MOVER_1TO2:
526 VectorCopy( ent->pos1, ent->s.pos.trBase );
527 VectorSubtract( ent->pos2, ent->pos1, delta );
528 f = 1000.0 / ent->s.pos.trDuration;
529 VectorScale( delta, f, ent->s.pos.trDelta );
530 ent->s.pos.trType = TR_LINEAR_STOP;
531 break;
532 case MOVER_2TO1:
533 VectorCopy( ent->pos2, ent->s.pos.trBase );
534 VectorSubtract( ent->pos1, ent->pos2, delta );
535 f = 1000.0 / ent->s.pos.trDuration;
536 VectorScale( delta, f, ent->s.pos.trDelta );
537 ent->s.pos.trType = TR_LINEAR_STOP;
538 break;
539 }
540 BG_EvaluateTrajectory( &ent->s.pos, level.time, ent->r.currentOrigin );
541 trap_LinkEntity( ent );
542}

544/*
545================
546MatchTeam
547
548All entities in a mover team will move from pos1 to pos2
549in the same amount of time
550================
551*/
552void MatchTeam( gentity_t *teamLeader, int moverState, int time ) {
553 gentity_t *slave;
554
555 for ( slave = teamLeader ; slave ; slave = slave->teamchain ) {
556 SetMoverState( slave, moverState, time );
557 }
558}

580/*
581================
582Reached_BinaryMover
583================
584*/
585void Reached_BinaryMover( gentity_t *ent ) {
586
587 // stop the looping sound
588 ent->s.loopSound = ent->soundLoop;
589
590 if ( ent->moverState == MOVER_1TO2 ) {
591 // reached pos2
592 SetMoverState( ent, MOVER_POS2, level.time );
593
594 // play sound
595 if ( ent->soundPos2 ) {
596 G_AddEvent( ent, EV_GENERAL_SOUND, ent->soundPos2 );
597 }
598
599 // return to pos1 after a delay
600 ent->think = ReturnToPos1;
601 ent->nextthink = level.time + ent->wait;
602
603 // fire targets
604 if ( !ent->activator ) {
605 ent->activator = ent;
606 }
607 G_UseTargets( ent, ent->activator );
608 } else if ( ent->moverState == MOVER_2TO1 ) {
609 // reached pos1
610 SetMoverState( ent, MOVER_POS1, level.time );
611
612 // play sound
613 if ( ent->soundPos1 ) {
614 G_AddEvent( ent, EV_GENERAL_SOUND, ent->soundPos1 );
615 }
616
617 // close areaportals
618 if ( ent->teammaster == ent || !ent->teammaster ) {
619 trap_AdjustAreaPortalState( ent, qfalse );
620 }
621 } else {
622 G_Error( "Reached_BinaryMover: bad moverState" );
623 }
624}

627/*
628================
629Use_BinaryMover
630================
631*/
632void Use_BinaryMover( gentity_t *ent, gentity_t *other, gentity_t *activator ) {
633 int total;
634 int partial;
635
636 // only the master should be used
637 if ( ent->flags & FL_TEAMSLAVE ) {
638 Use_BinaryMover( ent->teammaster, other, activator );
639 return;
640 }
641
642 ent->activator = activator;
643
644 if ( ent->moverState == MOVER_POS1 ) {
645 // start moving 50 msec later, becase if this was player
646 // triggered, level.time hasn't been advanced yet
647 MatchTeam( ent, MOVER_1TO2, level.time + 50 );
648
649 // starting sound
650 if ( ent->sound1to2 ) {
651 G_AddEvent( ent, EV_GENERAL_SOUND, ent->sound1to2 );
652 }
653
654 // looping sound
655 ent->s.loopSound = ent->soundLoop;
656
657 // open areaportal
658 if ( ent->teammaster == ent || !ent->teammaster ) {
659 trap_AdjustAreaPortalState( ent, qtrue );
660 }
661 return;
662 }
663
664 // if all the way up, just delay before coming down
665 if ( ent->moverState == MOVER_POS2 ) {
666 ent->nextthink = level.time + ent->wait;
667 return;
668 }
669
670 // only partway down before reversing
671 if ( ent->moverState == MOVER_2TO1 ) {
672 total = ent->s.pos.trDuration;
673 partial = level.time - ent->s.pos.trTime;
674 if ( partial > total ) {
675 partial = total;
676 }
677
678 MatchTeam( ent, MOVER_1TO2, level.time - ( total - partial ) );
679
680 if ( ent->sound1to2 ) {
681 G_AddEvent( ent, EV_GENERAL_SOUND, ent->sound1to2 );
682 }
683 return;
684 }
685
686 // only partway up before reversing
687 if ( ent->moverState == MOVER_1TO2 ) {
688 total = ent->s.pos.trDuration;
689 partial = level.time - ent->s.pos.trTime;
690 if ( partial > total ) {
691 partial = total;
692 }
693
694 MatchTeam( ent, MOVER_2TO1, level.time - ( total - partial ) );
695
696 if ( ent->sound2to1 ) {
697 G_AddEvent( ent, EV_GENERAL_SOUND, ent->sound2to1 );
698 }
699 return;
700 }
701}

705/*
706================
707InitMover
708
709"pos1", "pos2", and "speed" should be set before calling,
710so the movement delta can be calculated
711================
712*/
713void InitMover( gentity_t *ent ) {
714 vec3_t move;
715 float distance;
716 float light;
717 vec3_t color;
718 qboolean lightSet, colorSet;
719 char *sound;
720
721 // if the "model2" key is set, use a seperate model
722 // for drawing, but clip against the brushes
723 if ( ent->model2 ) {
724 ent->s.modelindex2 = G_ModelIndex( ent->model2 );
725 }
726
727 // if the "loopsound" key is set, use a constant looping sound when moving
728 if ( G_SpawnString( "noise", "100", &sound ) ) {
729 ent->s.loopSound = G_SoundIndex( sound );
730 }
731
732 // if the "color" or "light" keys are set, setup constantLight
733 lightSet = G_SpawnFloat( "light", "100", &light );
734 colorSet = G_SpawnVector( "color", "1 1 1", color );
735 if ( lightSet || colorSet ) {
736 int r, g, b, i;
737
738 r = color[0] * 255;
739 if ( r > 255 ) {
740 r = 255;
741 }
742 g = color[1] * 255;
743 if ( g > 255 ) {
744 g = 255;
745 }
746 b = color[2] * 255;
747 if ( b > 255 ) {
748 b = 255;
749 }
750 i = light / 4;
751 if ( i > 255 ) {
752 i = 255;
753 }
754 ent->s.constantLight = r | ( g << 8 ) | ( b << 16 ) | ( i << 24 );
755 }
756
757
758 ent->use = Use_BinaryMover;

826Touch_DoorTriggerSpectator
827================
828*/
829static void Touch_DoorTriggerSpectator( gentity_t *ent, gentity_t *other, trace_t *trace ) {
830 int axis;
831 float doorMin, doorMax;
832 vec3_t origin;
833
834 axis = ent->count;
835 // the constants below relate to constants in Think_SpawnNewDoorTrigger()
836 doorMin = ent->r.absmin[axis] + 100;
837 doorMax = ent->r.absmax[axis] - 100;
838
839 VectorCopy(other->client->ps.origin, origin);
840
841 if (origin[axis] < doorMin || origin[axis] > doorMax) return;
842
843 if (fabs(origin[axis] - doorMax) < fabs(origin[axis] - doorMin)) {
844 origin[axis] = doorMin - 10;
845 } else {
846 origin[axis] = doorMax + 10;
847 }
848
849 TeleportPlayer(other, origin, tv(10000000.0, 0, 0));
850}

871/*
872======================
873Think_SpawnNewDoorTrigger
874
875All of the parts of a door have been spawned, so create
876a trigger that encloses all of them
877======================
878*/
879void Think_SpawnNewDoorTrigger( gentity_t *ent ) {
880 gentity_t *other;
881 vec3_t mins, maxs;
882 int i, best;
883
884 if (!ent) {
885 return;
886 }
887
888 // set all of the slaves as shootable
889 for ( other = ent ; other ; other = other->teamchain ) {
890 other->takedamage = qtrue;
891 }
892
893 // find the bounds of everything on the team
894 VectorCopy (ent->r.absmin, mins);
895 VectorCopy (ent->r.absmax, maxs);
896
897 for (other = ent->teamchain ; other ; other=other->teamchain) {
898 AddPointToBounds (other->r.absmin, mins, maxs);
899 AddPointToBounds (other->r.absmax, mins, maxs);
900 }
901
902 // find the thinnest axis, which will be the one we expand
903 best = 0;
904 for ( i = 1 ; i < 3 ; i++ ) {
905 if ( maxs[i] - mins[i] < maxs[best] - mins[best] ) {
906 best = i;
907 }
908 }
909 maxs[best] += 120;
910 mins[best] -= 120;
911
912 // create a trigger with this size
913 other = G_Spawn ();
914 other->classname = "door_trigger";
915 VectorCopy (mins, other->r.mins);
916 VectorCopy (maxs, other->r.maxs);
917 other->parent = ent;
918 other->r.contents = CONTENTS_TRIGGER;
919 other->touch = Touch_DoorTrigger;
920 // remember the thinnest axis
921 other->count = best;
922 trap_LinkEntity (other);
923
924 MatchTeam( ent, ent->moverState, level.time );
925}

1271/*
1272===============
1273Reached_Train
1274===============
1275*/
1276void Reached_Train( gentity_t *ent ) {
1277 gentity_t *next;
1278 float speed;
1279 vec3_t move;
1280 float length;
1281
1282 // copy the apropriate values
1283 next = ent->nextTrain;
1284 if ( !next || !next->nextTrain ) {
1285 return; // just stop
1286 }
1287
1288 // fire all other targets
1289 G_UseTargets( next, NULL );
1290
1291 // set the new trajectory
1292 ent->nextTrain = next->nextTrain;
1293 VectorCopy( next->s.origin, ent->pos1 );
1294 VectorCopy( next->nextTrain->s.origin, ent->pos2 );
1295
1296 // if the path_corner has a speed, use that
1297 if ( next->speed ) {
1298 speed = next->speed;
1299 } else {
1300 // otherwise use the train's speed
1301 speed = ent->speed;
1302 }
1303 if ( speed < 1 ) {
1304 speed = 1;
1305 }
1306
1307 // calculate duration
1308 VectorSubtract( ent->pos2, ent->pos1, move );
1309 length = VectorLength( move );
1310
1311 ent->s.pos.trDuration = length * 1000 / speed;
1312
1313 // Tequila comment: Be sure to send to clients after any fast move case
1314 ent->r.svFlags &= ~SVF_NOCLIENT;
1315
1316 // Tequila comment: Fast move case
1317 if(ent->s.pos.trDuration<1) {
1318 // Tequila comment: As trDuration is used later in a division, we need to avoid that case now
1319 // With null trDuration,
1320 // the calculated rocks bounding box becomes infinite and the engine think for a short time
1321 // any entity is riding that mover but not the world entity... In rare case, I found it
1322 // can also stuck every map entities after func_door are used.
1323 // The desired effect with very very big speed is to have instant move, so any not null duration
1324 // lower than a frame duration should be sufficient.
1325 // Afaik, the negative case don't have to be supported.
1326 ent->s.pos.trDuration=1;
1327
1328 // Tequila comment: Don't send entity to clients so it becomes really invisible
1329 ent->r.svFlags |= SVF_NOCLIENT;
1330 }
1331
1332 // looping sound
1333 ent->s.loopSound = next->soundLoop;
1334
1335 // start it going
1336 SetMoverState( ent, MOVER_1TO2, level.time );
1337
1338 // if there is a "wait" value on the target, don't start moving yet
1339 if ( next->wait ) {
1340 ent->nextthink = level.time + next->wait * 1000;
1341 ent->think = Think_BeginMoving;
1342 ent->s.pos.trType = TR_STATIONARY;
1343 }
1344}

1350
1351Link all the corners together
1352===============
1353*/
1354void Think_SetupTrainTargets( gentity_t *ent ) {

1356
1357 ent->nextTrain = G_Find( NULL, FOFS(targetname), ent->target );

1359 G_Printf( "func_train at %s with an unfound target\n",
1360 vtos(ent->r.absmin) );
1361 return;
1362 }
1363
1364 start = NULL;
1365 for ( path = ent->nextTrain ; path != start ; path = next ) {
1366 if ( !start ) {
1367 start = path;

1371 G_Printf( "Train corner at %s without a target\n",
1372 vtos(path->s.origin) );

1377 // there may also be other targets that get fired when the corner
1378 // is reached

1380 do {
1381 next = G_Find( next, FOFS(targetname), path->target );
1382 if ( !next ) {
1383 G_Printf( "Train corner at %s without a target path_corner\n",
1384 vtos(path->s.origin) );
1385 return;
1386 }
1387 } while ( strcmp( next->classname, "path_corner" ) );
1388
1389 path->nextTrain = next;
1390 }
1391
1392 // start the train moving from the first corner
1393 Reached_Train( ent );
1394}

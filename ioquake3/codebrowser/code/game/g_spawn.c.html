26qboolean G_SpawnString( const char *key, const char *defaultString, char **out ) {
27 int i;
28
29 if ( !level.spawning ) {
30 *out = (char *)defaultString;
31// G_Error( "G_SpawnString() called while not spawning" );
32 }
33
34 for ( i = 0 ; i < level.numSpawnVars ; i++ ) {
35 if ( !Q_stricmp( key, level.spawnVars[i][0] ) ) {
36 *out = level.spawnVars[i][1];
37 return qtrue;
38 }
39 }
40
41 *out = (char *)defaultString;
42 return qfalse;
43}

260/*
261===============
262G_CallSpawn
263
264Finds the spawn function for the entity and calls it,
265returning qfalse if not found
266===============
267*/
268qboolean G_CallSpawn( gentity_t *ent ) {
269 spawn_t *s;
270 gitem_t *item;
271
272 if ( !ent->classname ) {
273 G_Printf ("G_CallSpawn: NULL classname\n");
274 return qfalse;
275 }
276
277 // check item spawn functions
278 for ( item=bg_itemlist+1 ; item->classname ; item++ ) {
279 if ( !strcmp(item->classname, ent->classname) ) {
280 G_SpawnItem( ent, item );
281 return qtrue;
282 }
283 }
284
285 // check normal spawn functions
286 for ( s=spawns ; s->name ; s++ ) {
287 if ( !strcmp(s->name, ent->classname) ) {
288 // found it
289 s->spawn(ent);
290 return qtrue;
291 }
292 }
293 G_Printf ("%s doesn't have a spawn function\n", ent->classname);
294 return qfalse;
295}

297/*
298=============
299G_NewString
300
301Builds a copy of the string, translating \n to real linefeeds
302so message texts can be multi-line
303=============
304*/
305char *G_NewString( const char *string ) {
306 char *newb, *new_p;
307 int i,l;
308
309 l = strlen(string) + 1;
310
311 newb = G_Alloc( l );
312
313 new_p = newb;
314
315 // turn \n into a real linefeed
316 for ( i=0 ; i< l ; i++ ) {
317 if (string[i] == '\\' && i < l-1) {
318 i++;
319 if (string[i] == 'n') {
320 *new_p++ = '\n';
321 } else {
322 *new_p++ = '\\';
323 }
324 } else {
325 *new_p++ = string[i];
326 }
327 }
328
329 return newb;
330}

335/*
336===============
337G_ParseField
338
339Takes a key/value pair and sets the binary values
340in a gentity
341===============
342*/
343void G_ParseField( const char *key, const char *value, gentity_t *ent ) {
344 field_t *f;
345 byte *b;
346 float v;
347 vec3_t vec;
348
349 for ( f=fields ; f->name ; f++ ) {
350 if ( !Q_stricmp(f->name, key) ) {
351 // found it
352 b = (byte *)ent;
353
354 switch( f->type ) {
355 case F_STRING:
356 *(char **)(b+f->ofs) = G_NewString (value);
357 break;
358 case F_VECTOR:
359 sscanf (value, "%f %f %f", &vec[0], &vec[1], &vec[2]);
360 ((float *)(b+f->ofs))[0] = vec[0];
361 ((float *)(b+f->ofs))[1] = vec[1];
362 ((float *)(b+f->ofs))[2] = vec[2];
363 break;
364 case F_INT:
365 *(int *)(b+f->ofs) = atoi(value);
366 break;
367 case F_FLOAT:
368 *(float *)(b+f->ofs) = atof(value);
369 break;
370 case F_ANGLEHACK:
371 v = atof(value);
372 ((float *)(b+f->ofs))[0] = 0;
373 ((float *)(b+f->ofs))[1] = v;
374 ((float *)(b+f->ofs))[2] = 0;
375 break;
376 }
377 return;
378 }
379 }
380}

383 if(ent->s.eType == ET_MOVER) \
384 { \
385 trap_LinkEntity(ent); \
386 trap_AdjustAreaPortalState(ent, qtrue); \

389/*
390===================
391G_SpawnGEntityFromSpawnVars
392
393Spawn an entity and fill in all of the level fields from
394level.spawnVars[], then call the class specfic spawn function
395===================
396*/
397void G_SpawnGEntityFromSpawnVars( void ) {
398 int i;
399 gentity_t *ent;
400 char *s, *value, *gametypeName;
401 static char *gametypeNames[] = {"ffa", "tournament", "single", "team", "ctf", "oneflag", "obelisk", "harvester"};
402
403 // get the next free entity
404 ent = G_Spawn();
405
406 for ( i = 0 ; i < level.numSpawnVars ; i++ ) {
407 G_ParseField( level.spawnVars[i][0], level.spawnVars[i][1], ent );
408 }
409
410 // check for "notsingle" flag
411 if ( g_gametype.integer == GT_SINGLE_PLAYER ) {
412 G_SpawnInt( "notsingle", "0", &i );
413 if ( i ) {
414 ADJUST_AREAPORTAL();
415 G_FreeEntity( ent );
416 return;
417 }
418 }
419 // check for "notteam" flag (GT_FFA, GT_TOURNAMENT, GT_SINGLE_PLAYER)
420 if ( g_gametype.integer >= GT_TEAM ) {
421 G_SpawnInt( "notteam", "0", &i );
422 if ( i ) {
423 ADJUST_AREAPORTAL();
424 G_FreeEntity( ent );
425 return;
426 }
427 } else {
428 G_SpawnInt( "notfree", "0", &i );
429 if ( i ) {
430 ADJUST_AREAPORTAL();
431 G_FreeEntity( ent );
432 return;
433 }
434 }
435
436#ifdef MISSIONPACK
437 G_SpawnInt( "notta", "0", &i );
438 if ( i ) {
439 ADJUST_AREAPORTAL();
440 G_FreeEntity( ent );
441 return;
442 }
443#else
444 G_SpawnInt( "notq3a", "0", &i );
445 if ( i ) {
446 ADJUST_AREAPORTAL();
447 G_FreeEntity( ent );
448 return;
449 }
450#endif
451
452 if( G_SpawnString( "gametype", NULL, &value ) ) {
453 if( g_gametype.integer >= GT_FFA && g_gametype.integer < GT_MAX_GAME_TYPE ) {
454 gametypeName = gametypeNames[g_gametype.integer];
455
456 s = strstr( value, gametypeName );
457 if( !s ) {
458 ADJUST_AREAPORTAL();
459 G_FreeEntity( ent );
460 return;
461 }
462 }
463 }
464
465 // move editor origin to pos
466 VectorCopy( ent->s.origin, ent->s.pos.trBase );
467 VectorCopy( ent->s.origin, ent->r.currentOrigin );
468
469 // if we didn't get a classname, don't bother spawning anything
470 if ( !G_CallSpawn( ent ) ) {
471 G_FreeEntity( ent );
472 }
473}

477/*
478====================
479G_AddSpawnVarToken
480====================
481*/
482char *G_AddSpawnVarToken( const char *string ) {
483 int l;
484 char *dest;
485
486 l = strlen( string );
487 if ( level.numSpawnVarChars + l + 1 > MAX_SPAWN_VARS_CHARS ) {
488 G_Error( "G_AddSpawnVarToken: MAX_SPAWN_VARS_CHARS" );
489 }
490
491 dest = level.spawnVarChars + level.numSpawnVarChars;
492 memcpy( dest, string, l+1 );
493
494 level.numSpawnVarChars += l + 1;
495
496 return dest;
497}

499/*
500====================
501G_ParseSpawnVars
502
503Parses a brace bounded set of key / value pairs out of the
504level's entity strings into level.spawnVars[]
505
506This does not actually spawn an entity.
507====================
508*/
509qboolean G_ParseSpawnVars( void ) {
510 char keyname[MAX_TOKEN_CHARS];
511 char com_token[MAX_TOKEN_CHARS];
512
513 level.numSpawnVars = 0;
514 level.numSpawnVarChars = 0;
515
516 // parse the opening brace
517 if ( !trap_GetEntityToken( com_token, sizeof( com_token ) ) ) {
518 // end of spawn string
519 return qfalse;
520 }
521 if ( com_token[0] != '{' ) {
522 G_Error( "G_ParseSpawnVars: found %s when expecting {",com_token );
523 }
524
525 // go through all the key / value pairs
526 while ( 1 ) {
527 // parse key
528 if ( !trap_GetEntityToken( keyname, sizeof( keyname ) ) ) {
529 G_Error( "G_ParseSpawnVars: EOF without closing brace" );
530 }
531
532 if ( keyname[0] == '}' ) {
533 break;
534 }
535
536 // parse value
537 if ( !trap_GetEntityToken( com_token, sizeof( com_token ) ) ) {
538 G_Error( "G_ParseSpawnVars: EOF without closing brace" );
539 }
540
541 if ( com_token[0] == '}' ) {
542 G_Error( "G_ParseSpawnVars: closing brace without data" );
543 }
544 if ( level.numSpawnVars == MAX_SPAWN_VARS ) {
545 G_Error( "G_ParseSpawnVars: MAX_SPAWN_VARS" );
546 }
547 level.spawnVars[ level.numSpawnVars ][0] = G_AddSpawnVarToken( keyname );
548 level.spawnVars[ level.numSpawnVars ][1] = G_AddSpawnVarToken( com_token );
549 level.numSpawnVars++;
550 }
551
552 return qtrue;
553}

557/*QUAKED worldspawn (0 0 0) ?
558
559Every map should have exactly one worldspawn.
560"music" music wav file
561"gravity" 800 is default gravity
562"message" Text to print during connection process
563*/
564void SP_worldspawn( void ) {
565 char *s;
566
567 G_SpawnString( "classname", "", &s );
568 if ( Q_stricmp( s, "worldspawn" ) ) {
569 G_Error( "SP_worldspawn: The first entity isn't 'worldspawn'" );
570 }
571
572 // make some data visible to connecting client
573 trap_SetConfigstring( CS_GAME_VERSION, GAME_VERSION );
574
575 trap_SetConfigstring( CS_LEVEL_START_TIME, va("%i", level.startTime ) );
576
577 G_SpawnString( "music", "", &s );
578 trap_SetConfigstring( CS_MUSIC, s );
579
580 G_SpawnString( "message", "", &s );
581 trap_SetConfigstring( CS_MESSAGE, s ); // map specific message
582
583 trap_SetConfigstring( CS_MOTD, g_motd.string ); // message of the day
584
585 G_SpawnString( "gravity", "800", &s );
586 trap_Cvar_Set( "g_gravity", s );
587
588 G_SpawnString( "enableDust", "0", &s );
589 trap_Cvar_Set( "g_enableDust", s );
590
591 G_SpawnString( "enableBreath", "0", &s );
592 trap_Cvar_Set( "g_enableBreath", s );
593
594 g_entities[ENTITYNUM_WORLD].s.number = ENTITYNUM_WORLD;
595 g_entities[ENTITYNUM_WORLD].r.ownerNum = ENTITYNUM_NONE;
596 g_entities[ENTITYNUM_WORLD].classname = "worldspawn";
597
598 g_entities[ENTITYNUM_NONE].s.number = ENTITYNUM_NONE;
599 g_entities[ENTITYNUM_NONE].r.ownerNum = ENTITYNUM_NONE;
600 g_entities[ENTITYNUM_NONE].classname = "nothing";
601
602 // see if we want a warmup time
603 trap_SetConfigstring( CS_WARMUP, "" );
604 if ( g_restarted.integer ) {
605 trap_Cvar_Set( "g_restarted", "0" );
606 level.warmupTime = 0;
607 } else if ( g_doWarmup.integer ) { // Turn it on
608 level.warmupTime = -1;
609 trap_SetConfigstring( CS_WARMUP, va("%i", level.warmupTime) );
610 G_LogPrintf( "Warmup:\n" );
611 }
612
613}

616/*
617==============
618G_SpawnEntitiesFromString
619
620Parses textual entity definitions out of an entstring and spawns gentities.
621==============
622*/
623void G_SpawnEntitiesFromString( void ) {
624 // allow calls to G_Spawn*()
625 level.spawning = qtrue;
626 level.numSpawnVars = 0;
627
628 // the worldspawn is not an actual entity, but it still
629 // has a "spawn" function to perform any global setup
630 // needed by a level (setting configstrings or cvars, etc)
631 if ( !G_ParseSpawnVars() ) {
632 G_Error( "SpawnEntities: no entities" );
633 }
634 SP_worldspawn();
635
636 // parse ents
637 while( G_ParseSpawnVars() ) {
638 G_SpawnGEntityFromSpawnVars();
639 }
640
641 level.spawning = qfalse; // any future calls to G_Spawn*() will be errors
642}

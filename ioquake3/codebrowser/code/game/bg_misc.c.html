1031/*
1032================
1033BG_CanItemBeGrabbed
1034
1035Returns false if the item should not be picked up.
1036This needs to be the same for client side prediction and server use.
1037================
1038*/
1039qboolean BG_CanItemBeGrabbed( int gametype, const entityState_t *ent, const playerState_t *ps ) {
1040 gitem_t *item;
1041#ifdef MISSIONPACK
1042 int upperBound;
1043#endif
1044
1045 if ( ent->modelindex < 1 || ent->modelindex >= bg_numItems ) {
1046 Com_Error( ERR_DROP, "BG_CanItemBeGrabbed: index out of range" );
1047 }
1048
1049 item = &bg_itemlist[ent->modelindex];
1050
1051 switch( item->giType ) {
1052 case IT_WEAPON:
1053 return qtrue; // weapons are always picked up
1054
1055 case IT_AMMO:
1056 if ( ps->ammo[ item->giTag ] >= 200 ) {
1057 return qfalse; // can't hold any more
1058 }
1059 return qtrue;
1060
1061 case IT_ARMOR:
1062#ifdef MISSIONPACK
1063 if( bg_itemlist[ps->stats[STAT_PERSISTANT_POWERUP]].giTag == PW_SCOUT ) {
1064 return qfalse;
1065 }
1066
1067 // we also clamp armor to the maxhealth for handicapping
1068 if( bg_itemlist[ps->stats[STAT_PERSISTANT_POWERUP]].giTag == PW_GUARD ) {
1069 upperBound = ps->stats[STAT_MAX_HEALTH];
1070 }
1071 else {
1072 upperBound = ps->stats[STAT_MAX_HEALTH] * 2;
1073 }
1074
1075 if ( ps->stats[STAT_ARMOR] >= upperBound ) {
1076 return qfalse;
1077 }
1078#else
1079 if ( ps->stats[STAT_ARMOR] >= ps->stats[STAT_MAX_HEALTH] * 2 ) {
1080 return qfalse;
1081 }
1082#endif
1083 return qtrue;
1084
1085 case IT_HEALTH:
1086 // small and mega healths will go over the max, otherwise
1087 // don't pick up if already at max
1088#ifdef MISSIONPACK
1089 if( bg_itemlist[ps->stats[STAT_PERSISTANT_POWERUP]].giTag == PW_GUARD ) {
1090 }
1091 else
1092#endif
1093 if ( item->quantity == 5 || item->quantity == 100 ) {
1094 if ( ps->stats[STAT_HEALTH] >= ps->stats[STAT_MAX_HEALTH] * 2 ) {
1095 return qfalse;
1096 }
1097 return qtrue;
1098 }
1099
1100 if ( ps->stats[STAT_HEALTH] >= ps->stats[STAT_MAX_HEALTH] ) {
1101 return qfalse;
1102 }
1103 return qtrue;
1104
1105 case IT_POWERUP:
1106 return qtrue; // powerups are always picked up
1107
1108#ifdef MISSIONPACK
1109 case IT_PERSISTANT_POWERUP:
1110 // can only hold one item at a time
1111 if ( ps->stats[STAT_PERSISTANT_POWERUP] ) {
1112 return qfalse;
1113 }
1114
1115 // check team only
1116 if( ( ent->generic1 & 2 ) && ( ps->persistant[PERS_TEAM] != TEAM_RED ) ) {
1117 return qfalse;
1118 }
1119 if( ( ent->generic1 & 4 ) && ( ps->persistant[PERS_TEAM] != TEAM_BLUE ) ) {
1120 return qfalse;
1121 }
1122
1123 return qtrue;
1124#endif
1125
1126 case IT_TEAM: // team items, such as flags
1127#ifdef MISSIONPACK
1128 if( gametype == GT_1FCTF ) {
1129 // neutral flag can always be picked up
1130 if( item->giTag == PW_NEUTRALFLAG ) {
1131 return qtrue;
1132 }
1133 if (ps->persistant[PERS_TEAM] == TEAM_RED) {
1134 if (item->giTag == PW_BLUEFLAG && ps->powerups[PW_NEUTRALFLAG] ) {
1135 return qtrue;
1136 }
1137 } else if (ps->persistant[PERS_TEAM] == TEAM_BLUE) {
1138 if (item->giTag == PW_REDFLAG && ps->powerups[PW_NEUTRALFLAG] ) {
1139 return qtrue;
1140 }
1141 }
1142 }
1143#endif
1144 if( gametype == GT_CTF ) {
1145 // ent->modelindex2 is non-zero on items if they are dropped
1146 // we need to know this because we can pick up our dropped flag (and return it)
1147 // but we can't pick up our flag at base
1148 if (ps->persistant[PERS_TEAM] == TEAM_RED) {
1149 if (item->giTag == PW_BLUEFLAG ||
1150 (item->giTag == PW_REDFLAG && ent->modelindex2) ||
1151 (item->giTag == PW_REDFLAG && ps->powerups[PW_BLUEFLAG]) )
1152 return qtrue;
1153 } else if (ps->persistant[PERS_TEAM] == TEAM_BLUE) {
1154 if (item->giTag == PW_REDFLAG ||
1155 (item->giTag == PW_BLUEFLAG && ent->modelindex2) ||
1156 (item->giTag == PW_BLUEFLAG && ps->powerups[PW_REDFLAG]) )
1157 return qtrue;
1158 }
1159 }
1160
1161#ifdef MISSIONPACK
1162 if( gametype == GT_HARVESTER ) {
1163 return qtrue;
1164 }
1165#endif
1166 return qfalse;
1167
1168 case IT_HOLDABLE:
1169 // can only hold one item at a time
1170 if ( ps->stats[STAT_HOLDABLE_ITEM] ) {
1171 return qfalse;
1172 }
1173 return qtrue;
1174
1175 case IT_BAD:
1176 Com_Error( ERR_DROP, "BG_CanItemBeGrabbed: IT_BAD" );
1177 default:
1178#ifndef Q3_VM
1179#ifndef NDEBUG
1180 Com_Printf("BG_CanItemBeGrabbed: unknown enum %d\n", item->giType );
1181#endif
1182#endif
1183 break;
1184 }
1185
1186 return qfalse;
1187}

1191/*
1192================
1193BG_EvaluateTrajectory
1194
1195================
1196*/
1197void BG_EvaluateTrajectory( const trajectory_t *tr, int atTime, vec3_t result ) {
1198 float deltaTime;
1199 float phase;
1200
1201 switch( tr->trType ) {
1202 case TR_STATIONARY:
1203 case TR_INTERPOLATE:
1204 VectorCopy( tr->trBase, result );
1205 break;
1206 case TR_LINEAR:
1207 deltaTime = ( atTime - tr->trTime ) * 0.001; // milliseconds to seconds
1208 VectorMA( tr->trBase, deltaTime, tr->trDelta, result );
1209 break;
1210 case TR_SINE:
1211 deltaTime = ( atTime - tr->trTime ) / (float) tr->trDuration;
1212 phase = sin( deltaTime * M_PI * 2 );
1213 VectorMA( tr->trBase, phase, tr->trDelta, result );
1214 break;
1215 case TR_LINEAR_STOP:
1216 if ( atTime > tr->trTime + tr->trDuration ) {
1217 atTime = tr->trTime + tr->trDuration;
1218 }
1219 deltaTime = ( atTime - tr->trTime ) * 0.001; // milliseconds to seconds
1220 if ( deltaTime < 0 ) {
1221 deltaTime = 0;
1222 }
1223 VectorMA( tr->trBase, deltaTime, tr->trDelta, result );
1224 break;
1225 case TR_GRAVITY:
1226 deltaTime = ( atTime - tr->trTime ) * 0.001; // milliseconds to seconds
1227 VectorMA( tr->trBase, deltaTime, tr->trDelta, result );
1228 result[2] -= 0.5 * DEFAULT_GRAVITY * deltaTime * deltaTime; // FIXME: local gravity...
1229 break;
1230 default:
1231 Com_Error( ERR_DROP, "BG_EvaluateTrajectory: unknown trType: %i", tr->trType );
1232 break;
1233 }
1234}
1235
1236/*
1237================
1238BG_EvaluateTrajectoryDelta
1239
1240For determining velocity at a given time
1241================
1242*/
1243void BG_EvaluateTrajectoryDelta( const trajectory_t *tr, int atTime, vec3_t result ) {
1244 float deltaTime;
1245 float phase;
1246
1247 switch( tr->trType ) {
1248 case TR_STATIONARY:
1249 case TR_INTERPOLATE:
1250 VectorClear( result );
1251 break;
1252 case TR_LINEAR:
1253 VectorCopy( tr->trDelta, result );
1254 break;
1255 case TR_SINE:
1256 deltaTime = ( atTime - tr->trTime ) / (float) tr->trDuration;
1257 phase = cos( deltaTime * M_PI * 2 ); // derivative of sin = cos
1258 phase *= 0.5;
1259 VectorScale( tr->trDelta, phase, result );
1260 break;
1261 case TR_LINEAR_STOP:
1262 if ( atTime > tr->trTime + tr->trDuration ) {
1263 VectorClear( result );
1264 return;
1265 }
1266 VectorCopy( tr->trDelta, result );
1267 break;
1268 case TR_GRAVITY:
1269 deltaTime = ( atTime - tr->trTime ) * 0.001; // milliseconds to seconds
1270 VectorCopy( tr->trDelta, result );
1271 result[2] -= DEFAULT_GRAVITY * deltaTime; // FIXME: local gravity...
1272 break;
1273 default:
1274 Com_Error( ERR_DROP, "BG_EvaluateTrajectoryDelta: unknown trType: %i", tr->trType );
1275 break;
1276 }
1277}

1418BG_TouchJumpPad
1419========================
1420*/
1421void BG_TouchJumpPad( playerState_t *ps, entityState_t *jumppad ) {
1422 vec3_t angles;
1423 float p;
1424 int effectNum;
1425
1426 // spectators don't use jump pads
1427 if ( ps->pm_type != PM_NORMAL ) {
1428 return;
1429 }
1430
1431 // flying characters don't hit bounce pads
1432 if ( ps->powerups[PW_FLIGHT] ) {
1433 return;
1434 }
1435
1436 // if we didn't hit this same jumppad the previous frame
1437 // then don't play the event sound again if we are in a fat trigger
1438 if ( ps->jumppad_ent != jumppad->number ) {
1439
1440 vectoangles( jumppad->origin2, angles);
1441 p = fabs( AngleNormalize180( angles[PITCH] ) );
1442 if( p < 45 ) {
1443 effectNum = 0;

1445 effectNum = 1;
1446 }
1447 BG_AddPredictableEventToPlayerstate( EV_JUMP_PAD, effectNum, ps );
1448 }
1449 // remember hitting this jumppad this frame
1450 ps->jumppad_ent = jumppad->number;
1451 ps->jumppad_frame = ps->pmove_framecount;
1452 // give the player the velocity from the jumppad
1453 VectorCopy( jumppad->origin2, ps->velocity );
1454}

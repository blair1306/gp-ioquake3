196vmMain
197
198This is the only way control passes into the module.
199This must be the very first function compiled into the .q3vm file

202Q_EXPORT intptr_t vmMain( int command, int arg0, int arg1, int arg2, int arg3, int arg4, int arg5, int arg6, int arg7, int arg8, int arg9, int arg10, int arg11 ) {
203 switch ( command ) {
204 case GAME_INIT:
205 G_InitGame( arg0, arg1, arg2 );
206 return 0;
207 case GAME_SHUTDOWN:
208 G_ShutdownGame( arg0 );
209 return 0;
210 case GAME_CLIENT_CONNECT:
211 return (intptr_t)ClientConnect( arg0, arg1, arg2 );
212 case GAME_CLIENT_THINK:
213 ClientThink( arg0 );
214 return 0;
215 case GAME_CLIENT_USERINFO_CHANGED:
216 ClientUserinfoChanged( arg0 );
217 return 0;
218 case GAME_CLIENT_DISCONNECT:
219 ClientDisconnect( arg0 );
220 return 0;
221 case GAME_CLIENT_BEGIN:
222 ClientBegin( arg0 );
223 return 0;
224 case GAME_CLIENT_COMMAND:
225 ClientCommand( arg0 );
226 return 0;
227 case GAME_RUN_FRAME:
228 G_RunFrame( arg0 );
229 return 0;
230 case GAME_CONSOLE_COMMAND:
231 return ConsoleCommand();
232 case BOTAI_START_FRAME:
233 return BotAIStartFrame( arg0 );
234 }
235
236 return -1;
237}

262/*
263================
264G_FindTeams
265
266Chain together all entities with a matching team field.
267Entity teams are used for item groups and multi-entity mover groups.
268
269All but the first will have the FL_TEAMSLAVE flag set and teammaster field set
270All but the last will have the teamchain field set to the next one
271================
272*/
273void G_FindTeams( void ) {
274 gentity_t *e, *e2;
275 int i, j;
276 int c, c2;
277
278 c = 0;
279 c2 = 0;
280 for ( i=MAX_CLIENTS, e=g_entities+i ; i < level.num_entities ; i++,e++ ) {
281 if (!e->inuse)
282 continue;
283 if (!e->team)
284 continue;
285 if (e->flags & FL_TEAMSLAVE)
286 continue;
287 e->teammaster = e;
288 c++;
289 c2++;
290 for (j=i+1, e2=e+1 ; j < level.num_entities ; j++,e2++)
291 {
292 if (!e2->inuse)
293 continue;
294 if (!e2->team)
295 continue;
296 if (e2->flags & FL_TEAMSLAVE)
297 continue;
298 if (!strcmp(e->team, e2->team))
299 {
300 c2++;
301 e2->teamchain = e->teamchain;
302 e->teamchain = e2;
303 e2->teammaster = e;
304 e2->flags |= FL_TEAMSLAVE;
305
306 // make sure that targets only point at the master
307 if ( e2->targetname ) {
308 e->targetname = e2->targetname;
309 e2->targetname = NULL;
310 }
311 }
312 }
313 }
314
315 G_Printf ("%i teams with %i entities\n", c, c2);
316}

333/*
334=================
335G_RegisterCvars
336=================
337*/
338void G_RegisterCvars( void ) {
339 int i;
340 cvarTable_t *cv;
341 qboolean remapped = qfalse;
342
343 for ( i = 0, cv = gameCvarTable ; i < gameCvarTableSize ; i++, cv++ ) {
344 trap_Cvar_Register( cv->vmCvar, cv->cvarName,
345 cv->defaultString, cv->cvarFlags );
346 if ( cv->vmCvar )
347 cv->modificationCount = cv->vmCvar->modificationCount;
348
349 if (cv->teamShader) {
350 remapped = qtrue;
351 }
352 }
353
354 if (remapped) {
355 G_RemapTeamShaders();
356 }
357
358 // check some things
359 if ( g_gametype.integer < 0 || g_gametype.integer >= GT_MAX_GAME_TYPE ) {
360 G_Printf( "g_gametype %i is out of range, defaulting to 0\n", g_gametype.integer );
361 trap_Cvar_Set( "g_gametype", "0" );
362 trap_Cvar_Update( &g_gametype );
363 }
364
365 level.warmupModificationCount = g_warmup.modificationCount;
366}

368/*
369=================
370G_UpdateCvars
371=================
372*/
373void G_UpdateCvars( void ) {
374 int i;
375 cvarTable_t *cv;
376 qboolean remapped = qfalse;
377
378 for ( i = 0, cv = gameCvarTable ; i < gameCvarTableSize ; i++, cv++ ) {
379 if ( cv->vmCvar ) {
380 trap_Cvar_Update( cv->vmCvar );
381
382 if ( cv->modificationCount != cv->vmCvar->modificationCount ) {
383 cv->modificationCount = cv->vmCvar->modificationCount;
384
385 if ( cv->trackChange ) {
386 trap_SendServerCommand( -1, va("print \"Server: %s changed to %s\n\"",
387 cv->cvarName, cv->vmCvar->string ) );
388 }
389
390 if (cv->teamShader) {
391 remapped = qtrue;
392 }
393 }
394 }
395 }
396
397 if (remapped) {
398 G_RemapTeamShaders();
399 }
400}

408void G_InitGame( int levelTime, int randomSeed, int restart ) {
409 int i;
410
411 G_Printf ("------- Game Initialization -------\n");
412 G_Printf ("gamename: %s\n", GAMEVERSION);
413 G_Printf ("gamedate: %s\n", PRODUCT_DATE);
414
415 srand( randomSeed );
416
417 G_RegisterCvars();
418
419 G_ProcessIPBans();
420
421 G_InitMemory();
422
423 // set some level globals
424 memset( &level, 0, sizeof( level ) );
425 level.time = levelTime;
426 level.startTime = levelTime;
427
428 level.snd_fry = G_SoundIndex("sound/player/fry.wav"); // FIXME standing in lava / slime
429
430 if ( g_gametype.integer != GT_SINGLE_PLAYER && g_logfile.string[0] ) {
431 if ( g_logfileSync.integer ) {
432 trap_FS_FOpenFile( g_logfile.string, &level.logFile, FS_APPEND_SYNC );
433 } else {
434 trap_FS_FOpenFile( g_logfile.string, &level.logFile, FS_APPEND );
435 }
436 if ( !level.logFile ) {
437 G_Printf( "WARNING: Couldn't open logfile: %s\n", g_logfile.string );
438 } else {
439 char serverinfo[MAX_INFO_STRING];
440
441 trap_GetServerinfo( serverinfo, sizeof( serverinfo ) );
442
443 G_LogPrintf("------------------------------------------------------------\n" );
444 G_LogPrintf("InitGame: %s\n", serverinfo );
445 }
446 } else {
447 G_Printf( "Not logging to disk.\n" );
448 }
449
450 G_InitWorldSession();
451
452 // initialize all entities for this game
453 memset( g_entities, 0, MAX_GENTITIES * sizeof(g_entities[0]) );
454 level.gentities = g_entities;
455
456 // initialize all clients for this game
457 level.maxclients = g_maxclients.integer;
458 memset( g_clients, 0, MAX_CLIENTS * sizeof(g_clients[0]) );
459 level.clients = g_clients;
460
461 // set client fields on player ents
462 for ( i=0 ; i<level.maxclients ; i++ ) {
463 g_entities[i].client = level.clients + i;
464 }
465
466 // always leave room for the max number of clients,
467 // even if they aren't all used, so numbers inside that
468 // range are NEVER anything but clients
469 level.num_entities = MAX_CLIENTS;
470
471 for ( i=0 ; i<MAX_CLIENTS ; i++ ) {
472 g_entities[i].classname = "clientslot";
473 }
474
475 // let the server system know where the entites are
476 trap_LocateGameData( level.gentities, level.num_entities, sizeof( gentity_t ),
477 &level.clients[0].ps, sizeof( level.clients[0] ) );
478
479 // reserve some spots for dead player bodies
480 InitBodyQue();
481
482 ClearRegisteredItems();
483
484 // parse the key/value pairs and spawn gentities
485 G_SpawnEntitiesFromString();
486
487 // general initialization
488 G_FindTeams();
489
490 // make sure we have flags for CTF, etc
491 if( g_gametype.integer >= GT_TEAM ) {
492 G_CheckTeamItems();
493 }
494
495 SaveRegisteredItems();
496
497 G_Printf ("-----------------------------------\n");
498
499 if( g_gametype.integer == GT_SINGLE_PLAYER || trap_Cvar_VariableIntegerValue( "com_buildScript" ) ) {
500 G_ModelIndex( SP_PODIUM_MODEL );
501 }
502
503 if ( trap_Cvar_VariableIntegerValue( "bot_enable" ) ) {
504 BotAISetup( restart );
505 BotAILoadMap( restart );
506 G_InitBots( restart );
507 }
508
509 G_RemapTeamShaders();
510
511 trap_SetConfigstring( CS_INTERMISSION, "" );
512}

573/*
574=============
575AddTournamentPlayer
576
577If there are less than two tournament players, put a
578spectator in the game and restart
579=============
580*/
581void AddTournamentPlayer( void ) {
582 int i;
583 gclient_t *client;
584 gclient_t *nextInLine;
585
586 if ( level.numPlayingClients >= 2 ) {
587 return;
588 }
589
590 // never change during intermission
591 if ( level.intermissiontime ) {
592 return;
593 }
594
595 nextInLine = NULL;
596
597 for ( i = 0 ; i < level.maxclients ; i++ ) {
598 client = &level.clients[i];
599 if ( client->pers.connected != CON_CONNECTED ) {
600 continue;
601 }
602 if ( client->sess.sessionTeam != TEAM_SPECTATOR ) {
603 continue;
604 }
605 // never select the dedicated follow or scoreboard clients
606 if ( client->sess.spectatorState == SPECTATOR_SCOREBOARD ||
607 client->sess.spectatorClient < 0 ) {
608 continue;
609 }
610
611 if(!nextInLine || client->sess.spectatorNum > nextInLine->sess.spectatorNum)
612 nextInLine = client;
613 }
614
615 if ( !nextInLine ) {
616 return;
617 }
618
619 level.warmupTime = -1;
620
621 // set them to free-for-all team
622 SetTeam( &g_entities[ nextInLine - level.clients ], "f" );
623}

628
629Add client to end of tournament queue
630=======================
631*/
632
633void AddTournamentQueue(gclient_t *client)
634{
635 int index;
636 gclient_t *curclient;
637
638 for(index = 0; index < level.maxclients; index++)
639 {
640 curclient = &level.clients[index];
641
642 if(curclient->pers.connected != CON_DISCONNECTED)
643 {
644 if(curclient == client)
645 curclient->sess.spectatorNum = 0;
646 else if(curclient->sess.sessionTeam == TEAM_SPECTATOR)
647 curclient->sess.spectatorNum++;
648 }
649 }
650}

720/*
721=============
722SortRanks
723
724=============
725*/
726int QDECL SortRanks( const void *a, const void *b ) {
727 gclient_t *ca, *cb;
728
729 ca = &level.clients[*(int *)a];
730 cb = &level.clients[*(int *)b];
731
732 // sort special clients last
733 if ( ca->sess.spectatorState == SPECTATOR_SCOREBOARD || ca->sess.spectatorClient < 0 ) {
734 return 1;
735 }
736 if ( cb->sess.spectatorState == SPECTATOR_SCOREBOARD || cb->sess.spectatorClient < 0 ) {
737 return -1;
738 }
739
740 // then connecting clients
741 if ( ca->pers.connected == CON_CONNECTING ) {
742 return 1;
743 }
744 if ( cb->pers.connected == CON_CONNECTING ) {
745 return -1;
746 }
747
748
749 // then spectators
750 if ( ca->sess.sessionTeam == TEAM_SPECTATOR && cb->sess.sessionTeam == TEAM_SPECTATOR ) {
751 if ( ca->sess.spectatorNum > cb->sess.spectatorNum ) {
752 return -1;
753 }
754 if ( ca->sess.spectatorNum < cb->sess.spectatorNum ) {
755 return 1;
756 }
757 return 0;
758 }
759 if ( ca->sess.sessionTeam == TEAM_SPECTATOR ) {
760 return 1;
761 }
762 if ( cb->sess.sessionTeam == TEAM_SPECTATOR ) {
763 return -1;
764 }
765
766 // then sort by score
767 if ( ca->ps.persistant[PERS_SCORE]
768 > cb->ps.persistant[PERS_SCORE] ) {
769 return -1;
770 }
771 if ( ca->ps.persistant[PERS_SCORE]
772 < cb->ps.persistant[PERS_SCORE] ) {
773 return 1;
774 }
775 return 0;
776}

778/*
779============
780CalculateRanks
781
782Recalculates the score ranks of all players
783This will be called on every client connect, begin, disconnect, death,
784and team change.
785============
786*/
787void CalculateRanks( void ) {
788 int i;
789 int rank;
790 int score;
791 int newScore;
792 gclient_t *cl;
793
794 level.follow1 = -1;
795 level.follow2 = -1;
796 level.numConnectedClients = 0;
797 level.numNonSpectatorClients = 0;
798 level.numPlayingClients = 0;
799 level.numVotingClients = 0; // don't count bots
800
801 for (i = 0; i < ARRAY_LEN(level.numteamVotingClients); i++)
802 level.numteamVotingClients[i] = 0;
803
804 for ( i = 0 ; i < level.maxclients ; i++ ) {
805 if ( level.clients[i].pers.connected != CON_DISCONNECTED ) {
806 level.sortedClients[level.numConnectedClients] = i;
807 level.numConnectedClients++;
808
809 if ( level.clients[i].sess.sessionTeam != TEAM_SPECTATOR ) {
810 level.numNonSpectatorClients++;
811
812 // decide if this should be auto-followed
813 if ( level.clients[i].pers.connected == CON_CONNECTED ) {
814 level.numPlayingClients++;
815 if ( !(g_entities[i].r.svFlags & SVF_BOT) ) {
816 level.numVotingClients++;
817 if ( level.clients[i].sess.sessionTeam == TEAM_RED )
818 level.numteamVotingClients[0]++;
819 else if ( level.clients[i].sess.sessionTeam == TEAM_BLUE )
820 level.numteamVotingClients[1]++;
821 }
822 if ( level.follow1 == -1 ) {
823 level.follow1 = i;
824 } else if ( level.follow2 == -1 ) {
825 level.follow2 = i;
826 }
827 }
828 }
829 }
830 }
831
832 qsort( level.sortedClients, level.numConnectedClients,
833 sizeof(level.sortedClients[0]), SortRanks );
834
835 // set the rank value for all clients that are connected and not spectators
836 if ( g_gametype.integer >= GT_TEAM ) {
837 // in team games, rank is just the order of the teams, 0=red, 1=blue, 2=tied
838 for ( i = 0; i < level.numConnectedClients; i++ ) {
839 cl = &level.clients[ level.sortedClients[i] ];
840 if ( level.teamScores[TEAM_RED] == level.teamScores[TEAM_BLUE] ) {
841 cl->ps.persistant[PERS_RANK] = 2;
842 } else if ( level.teamScores[TEAM_RED] > level.teamScores[TEAM_BLUE] ) {
843 cl->ps.persistant[PERS_RANK] = 0;
844 } else {
845 cl->ps.persistant[PERS_RANK] = 1;
846 }
847 }
848 } else {
849 rank = -1;
850 score = 0;
851 for ( i = 0; i < level.numPlayingClients; i++ ) {
852 cl = &level.clients[ level.sortedClients[i] ];
853 newScore = cl->ps.persistant[PERS_SCORE];
854 if ( i == 0 || newScore != score ) {
855 rank = i;
856 // assume we aren't tied until the next client is checked
857 level.clients[ level.sortedClients[i] ].ps.persistant[PERS_RANK] = rank;
858 } else {
859 // we are tied with the previous client
860 level.clients[ level.sortedClients[i-1] ].ps.persistant[PERS_RANK] = rank | RANK_TIED_FLAG;
861 level.clients[ level.sortedClients[i] ].ps.persistant[PERS_RANK] = rank | RANK_TIED_FLAG;
862 }
863 score = newScore;
864 if ( g_gametype.integer == GT_SINGLE_PLAYER && level.numPlayingClients == 1 ) {
865 level.clients[ level.sortedClients[i] ].ps.persistant[PERS_RANK] = rank | RANK_TIED_FLAG;
866 }
867 }
868 }
869
870 // set the CS_SCORES1/2 configstrings, which will be visible to everyone
871 if ( g_gametype.integer >= GT_TEAM ) {
872 trap_SetConfigstring( CS_SCORES1, va("%i", level.teamScores[TEAM_RED] ) );
873 trap_SetConfigstring( CS_SCORES2, va("%i", level.teamScores[TEAM_BLUE] ) );
874 } else {
875 if ( level.numConnectedClients == 0 ) {
876 trap_SetConfigstring( CS_SCORES1, va("%i", SCORE_NOT_PRESENT) );
877 trap_SetConfigstring( CS_SCORES2, va("%i", SCORE_NOT_PRESENT) );
878 } else if ( level.numConnectedClients == 1 ) {
879 trap_SetConfigstring( CS_SCORES1, va("%i", level.clients[ level.sortedClients[0] ].ps.persistant[PERS_SCORE] ) );
880 trap_SetConfigstring( CS_SCORES2, va("%i", SCORE_NOT_PRESENT) );
881 } else {
882 trap_SetConfigstring( CS_SCORES1, va("%i", level.clients[ level.sortedClients[0] ].ps.persistant[PERS_SCORE] ) );
883 trap_SetConfigstring( CS_SCORES2, va("%i", level.clients[ level.sortedClients[1] ].ps.persistant[PERS_SCORE] ) );
884 }
885 }
886
887 // see if it is time to end the level
888 CheckExitRules();
889
890 // if we are at the intermission, send the new info to everyone
891 if ( level.intermissiontime ) {
892 SendScoreboardMessageToAllClients();
893 }
894}

1216/*
1217=================
1218CheckIntermissionExit
1219
1220The level will stay at the intermission for a minimum of 5 seconds
1221If all players wish to continue, the level will then exit.
1222If one or more players have not acknowledged the continue, the game will
1223wait 10 seconds before going on.
1224=================
1225*/
1226void CheckIntermissionExit( void ) {
1227 int ready, notReady, playerCount;
1228 int i;
1229 gclient_t *cl;
1230 int readyMask;
1231
1232 if ( g_gametype.integer == GT_SINGLE_PLAYER ) {
1233 return;
1234 }
1235
1236 // see which players are ready
1237 ready = 0;
1238 notReady = 0;
1239 readyMask = 0;
1240 playerCount = 0;
1241 for (i=0 ; i< g_maxclients.integer ; i++) {
1242 cl = level.clients + i;
1243 if ( cl->pers.connected != CON_CONNECTED ) {
1244 continue;
1245 }
1246 if ( g_entities[i].r.svFlags & SVF_BOT ) {
1247 continue;
1248 }
1249
1250 playerCount++;
1251 if ( cl->readyToExit ) {
1252 ready++;
1253 if ( i < 16 ) {
1254 readyMask |= 1 << i;
1255 }
1256 } else {
1257 notReady++;
1258 }
1259 }
1260
1261 // copy the readyMask to each player's stats so
1262 // it can be displayed on the scoreboard
1263 for (i=0 ; i< g_maxclients.integer ; i++) {
1264 cl = level.clients + i;
1265 if ( cl->pers.connected != CON_CONNECTED ) {
1266 continue;
1267 }
1268 cl->ps.stats[STAT_CLIENTS_READY] = readyMask;
1269 }
1270
1271 // never exit in less than five seconds
1272 if ( level.time < level.intermissiontime + 5000 ) {
1273 return;
1274 }
1275
1276 // only test ready status when there are real players present
1277 if ( playerCount > 0 ) {
1278 // if nobody wants to go, clear timer
1279 if ( !ready ) {
1280 level.readyToExit = qfalse;
1281 return;
1282 }
1283
1284 // if everyone wants to go, go now
1285 if ( !notReady ) {
1286 ExitLevel();
1287 return;
1288 }
1289 }
1290
1291 // the first person to ready starts the ten second timeout
1292 if ( !level.readyToExit ) {
1293 level.readyToExit = qtrue;
1294 level.exitTime = level.time;
1295 }
1296
1297 // if we have waited ten seconds since at least one player
1298 // wanted to exit, go ahead
1299 if ( level.time < level.exitTime + 10000 ) {
1300 return;
1301 }
1302
1303 ExitLevel();
1304}

1769 gentity_t *ent;
1770

1772 if ( level.restarted ) {
1773 return;
1774 }
1775
1776 level.framenum++;
1777 level.previousTime = level.time;
1778 level.time = levelTime;
1779
1780 // get any cvar changes
1781 G_UpdateCvars();
1782
1783 //
1784 // go through all allocated objects
1785 //
1786 ent = &g_entities[0];
1787 for (i=0 ; i<level.num_entities ; i++, ent++) {
1788 if ( !ent->inuse ) {
1789 continue;
1790 }
1791
1792 // clear events that are too old
1793 if ( level.time - ent->eventTime > EVENT_VALID_MSEC ) {
1794 if ( ent->s.event ) {
1795 ent->s.event = 0; // &= EV_EVENT_BITS;
1796 if ( ent->client ) {
1797 ent->client->ps.externalEvent = 0;
1798 // predicted events should never be set to zero
1799 //ent->client->ps.events[0] = 0;
1800 //ent->client->ps.events[1] = 0;
1801 }
1802 }
1803 if ( ent->freeAfterEvent ) {
1804 // tempEntities or dropped items completely go away after their event
1805 G_FreeEntity( ent );
1806 continue;
1807 } else if ( ent->unlinkAfterEvent ) {
1808 // items that will respawn will hide themselves after their pickup event
1809 ent->unlinkAfterEvent = qfalse;
1810 trap_UnlinkEntity( ent );
1811 }
1812 }
1813
1814 // temporary entities don't think
1815 if ( ent->freeAfterEvent ) {
1816 continue;
1817 }
1818
1819 if ( !ent->r.linked && ent->neverFree ) {
1820 continue;
1821 }
1822

1824 G_RunMissile( ent );
1825 continue;
1826 }
1827
1828 if ( ent->s.eType == ET_ITEM || ent->physicsObject ) {
1829 G_RunItem( ent );
1830 continue;
1831 }
1832
1833 if ( ent->s.eType == ET_MOVER ) {
1834 G_RunMover( ent );
1835 continue;
1836 }

1839 G_RunClient( ent );
1840 continue;
1841 }
1842
1843 G_RunThink( ent );
1844 }

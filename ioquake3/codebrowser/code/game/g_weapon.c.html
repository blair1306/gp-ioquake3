34/*
35================
36G_BounceProjectile
37================
38*/
39void G_BounceProjectile( vec3_t start, vec3_t impact, vec3_t dir, vec3_t endout ) {
40 vec3_t v, newv;
41 float dot;
42
43 VectorSubtract( impact, start, v );
44 dot = DotProduct( v, dir );
45 VectorMA( v, -2*dot, dir, newv );
46
47 VectorNormalize(newv);
48 VectorMA(impact, 8192, newv, endout);
49}

64/*
65===============
66CheckGauntletAttack
67===============
68*/
69qboolean CheckGauntletAttack( gentity_t *ent ) {
70 trace_t tr;
71 vec3_t end;
72 gentity_t *tent;
73 gentity_t *traceEnt;
74 int damage;
75
76 // set aiming directions
77 AngleVectors (ent->client->ps.viewangles, forward, right, up);
78
79 CalcMuzzlePoint ( ent, forward, right, up, muzzle );
80
81 VectorMA (muzzle, 32, forward, end);
82
83 trap_Trace (&tr, muzzle, NULL, NULL, end, ent->s.number, MASK_SHOT);
84 if ( tr.surfaceFlags & SURF_NOIMPACT ) {
85 return qfalse;
86 }
87
88 if ( ent->client->noclip ) {
89 return qfalse;
90 }
91
92 traceEnt = &g_entities[ tr.entityNum ];
93
94 // send blood impact
95 if ( traceEnt->takedamage && traceEnt->client ) {
96 tent = G_TempEntity( tr.endpos, EV_MISSILE_HIT );
97 tent->s.otherEntityNum = traceEnt->s.number;
98 tent->s.eventParm = DirToByte( tr.plane.normal );
99 tent->s.weapon = ent->s.weapon;
100 }
101
102 if ( !traceEnt->takedamage) {
103 return qfalse;
104 }
105
106 if (ent->client->ps.powerups[PW_QUAD] ) {
107 G_AddEvent( ent, EV_POWERUP_QUAD, 0 );
108 s_quadFactor = g_quadfactor.value;
109 } else {
110 s_quadFactor = 1;
111 }
112#ifdef MISSIONPACK
113 if( ent->client->persistantPowerup && ent->client->persistantPowerup->item && ent->client->persistantPowerup->item->giTag == PW_DOUBLER ) {
114 s_quadFactor *= 2;
115 }
116#endif
117
118 damage = 50 * s_quadFactor;
119 G_Damage( traceEnt, ent, ent, forward, tr.endpos,
120 damage, 0, MOD_GAUNTLET );
121
122 return qtrue;
123}

134/*
135======================
136SnapVectorTowards
137
138Round a vector to integers for more efficient network
139transmission, but make sure that it rounds towards a given point
140rather than blindly truncating. This prevents it from truncating
141into a wall.
142======================
143*/
144void SnapVectorTowards( vec3_t v, vec3_t to ) {
145 int i;
146
147 for ( i = 0 ; i < 3 ; i++ ) {
148 if ( to[i] <= v[i] ) {
149 v[i] = floor(v[i]);
150 } else {
151 v[i] = ceil(v[i]);
152 }
153 }
154}
155
156#ifdef MISSIONPACK
157#define CHAINGUN_SPREAD 600
158#define CHAINGUN_DAMAGE 7
159#endif
160#define MACHINEGUN_SPREAD 200
161#define MACHINEGUN_DAMAGE 7
162#define MACHINEGUN_TEAM_DAMAGE 5 // wimpier MG in teamplay
163
164void Bullet_Fire (gentity_t *ent, float spread, int damage, int mod ) {
165 trace_t tr;
166 vec3_t end;
167#ifdef MISSIONPACK
168 vec3_t impactpoint, bouncedir;
169#endif
170 float r;
171 float u;
172 gentity_t *tent;
173 gentity_t *traceEnt;
174 int i, passent;
175
176 damage *= s_quadFactor;
177
178 r = random() * M_PI * 2.0f;
179 u = sin(r) * crandom() * spread * 16;
180 r = cos(r) * crandom() * spread * 16;
181 VectorMA (muzzle, 8192*16, forward, end);
182 VectorMA (end, r, right, end);
183 VectorMA (end, u, up, end);
184
185 passent = ent->s.number;
186 for (i = 0; i < 10; i++) {
187
188 trap_Trace (&tr, muzzle, NULL, NULL, end, passent, MASK_SHOT);
189 if ( tr.surfaceFlags & SURF_NOIMPACT ) {
190 return;
191 }
192
193 traceEnt = &g_entities[ tr.entityNum ];
194
195 // snap the endpos to integers, but nudged towards the line
196 SnapVectorTowards( tr.endpos, muzzle );
197
198 // send bullet impact
199 if ( traceEnt->takedamage && traceEnt->client ) {
200 tent = G_TempEntity( tr.endpos, EV_BULLET_HIT_FLESH );
201 tent->s.eventParm = traceEnt->s.number;
202 if( LogAccuracyHit( traceEnt, ent ) ) {
203 ent->client->accuracy_hits++;
204 }
205 } else {
206 tent = G_TempEntity( tr.endpos, EV_BULLET_HIT_WALL );
207 tent->s.eventParm = DirToByte( tr.plane.normal );
208 }
209 tent->s.otherEntityNum = ent->s.number;
210
211 if ( traceEnt->takedamage) {
212#ifdef MISSIONPACK
213 if ( traceEnt->client && traceEnt->client->invulnerabilityTime > level.time ) {
214 if (G_InvulnerabilityEffect( traceEnt, forward, tr.endpos, impactpoint, bouncedir )) {
215 G_BounceProjectile( muzzle, impactpoint, bouncedir, end );
216 VectorCopy( impactpoint, muzzle );
217 // the player can hit him/herself with the bounced rail
218 passent = ENTITYNUM_NONE;
219 }
220 else {
221 VectorCopy( tr.endpos, muzzle );
222 passent = traceEnt->s.number;
223 }
224 continue;
225 }
226 else {
227#endif
228 G_Damage( traceEnt, ent, ent, forward, tr.endpos,
229 damage, 0, mod);
230#ifdef MISSIONPACK
231 }
232#endif
233 }
234 break;
235 }
236}

266// DEFAULT_SHOTGUN_SPREAD and DEFAULT_SHOTGUN_COUNT are in bg_public.h, because
267// client predicts same spreads
268#define DEFAULT_SHOTGUN_DAMAGE 10
269
270qboolean ShotgunPellet( vec3_t start, vec3_t end, gentity_t *ent ) {
271 trace_t tr;
272 int damage, i, passent;
273 gentity_t *traceEnt;
274#ifdef MISSIONPACK
275 vec3_t impactpoint, bouncedir;
276#endif
277 vec3_t tr_start, tr_end;
278
279 passent = ent->s.number;
280 VectorCopy( start, tr_start );
281 VectorCopy( end, tr_end );
282 for (i = 0; i < 10; i++) {
283 trap_Trace (&tr, tr_start, NULL, NULL, tr_end, passent, MASK_SHOT);
284 traceEnt = &g_entities[ tr.entityNum ];
285
286 // send bullet impact
287 if ( tr.surfaceFlags & SURF_NOIMPACT ) {
288 return qfalse;
289 }
290
291 if ( traceEnt->takedamage) {
292 damage = DEFAULT_SHOTGUN_DAMAGE * s_quadFactor;
293#ifdef MISSIONPACK
294 if ( traceEnt->client && traceEnt->client->invulnerabilityTime > level.time ) {
295 if (G_InvulnerabilityEffect( traceEnt, forward, tr.endpos, impactpoint, bouncedir )) {
296 G_BounceProjectile( tr_start, impactpoint, bouncedir, tr_end );
297 VectorCopy( impactpoint, tr_start );
298 // the player can hit him/herself with the bounced rail
299 passent = ENTITYNUM_NONE;
300 }
301 else {
302 VectorCopy( tr.endpos, tr_start );
303 passent = traceEnt->s.number;
304 }
305 continue;
306 }
307 else {
308 G_Damage( traceEnt, ent, ent, forward, tr.endpos,
309 damage, 0, MOD_SHOTGUN);
310 if( LogAccuracyHit( traceEnt, ent ) ) {
311 return qtrue;
312 }
313 }
314#else
315 G_Damage( traceEnt, ent, ent, forward, tr.endpos, damage, 0, MOD_SHOTGUN);
316 if( LogAccuracyHit( traceEnt, ent ) ) {
317 return qtrue;
318 }
319#endif
320 }
321 return qfalse;
322 }
323 return qfalse;
324}

326// this should match CG_ShotgunPattern
327void ShotgunPattern( vec3_t origin, vec3_t origin2, int seed, gentity_t *ent ) {
328 int i;
329 float r, u;
330 vec3_t end;
331 vec3_t forward, right, up;
332 qboolean hitClient = qfalse;
333
334 // derive the right and up vectors from the forward vector, because
335 // the client won't have any other information
336 VectorNormalize2( origin2, forward );
337 PerpendicularVector( right, forward );
338 CrossProduct( forward, right, up );
339
340 // generate the "random" spread pattern
341 for ( i = 0 ; i < DEFAULT_SHOTGUN_COUNT ; i++ ) {
342 r = Q_crandom( &seed ) * DEFAULT_SHOTGUN_SPREAD * 16;
343 u = Q_crandom( &seed ) * DEFAULT_SHOTGUN_SPREAD * 16;
344 VectorMA( origin, 8192 * 16, forward, end);
345 VectorMA (end, r, right, end);
346 VectorMA (end, u, up, end);
347 if( ShotgunPellet( origin, end, ent ) && !hitClient ) {
348 hitClient = qtrue;
349 ent->client->accuracy_hits++;
350 }
351 }

437/*
438=================
439weapon_railgun_fire
440=================
441*/
442#define MAX_RAIL_HITS 4
443void weapon_railgun_fire (gentity_t *ent) {
444 vec3_t end;
445#ifdef MISSIONPACK
446 vec3_t impactpoint, bouncedir;
447#endif
448 trace_t trace;
449 gentity_t *tent;
450 gentity_t *traceEnt;
451 int damage;
452 int i;
453 int hits;
454 int unlinked;
455 int passent;
456 gentity_t *unlinkedEntities[MAX_RAIL_HITS];
457
458 damage = 100 * s_quadFactor;
459
460 VectorMA (muzzle, 8192, forward, end);
461
462 // trace only against the solids, so the railgun will go through people
463 unlinked = 0;
464 hits = 0;
465 passent = ent->s.number;
466 do {
467 trap_Trace (&trace, muzzle, NULL, NULL, end, passent, MASK_SHOT );
468 if ( trace.entityNum >= ENTITYNUM_MAX_NORMAL ) {
469 break;
470 }
471 traceEnt = &g_entities[ trace.entityNum ];
472 if ( traceEnt->takedamage ) {
473#ifdef MISSIONPACK
474 if ( traceEnt->client && traceEnt->client->invulnerabilityTime > level.time ) {
475 if ( G_InvulnerabilityEffect( traceEnt, forward, trace.endpos, impactpoint, bouncedir ) ) {
476 G_BounceProjectile( muzzle, impactpoint, bouncedir, end );
477 // snap the endpos to integers to save net bandwidth, but nudged towards the line
478 SnapVectorTowards( trace.endpos, muzzle );
479 // send railgun beam effect
480 tent = G_TempEntity( trace.endpos, EV_RAILTRAIL );
481 // set player number for custom colors on the railtrail
482 tent->s.clientNum = ent->s.clientNum;
483 VectorCopy( muzzle, tent->s.origin2 );
484 // move origin a bit to come closer to the drawn gun muzzle
485 VectorMA( tent->s.origin2, 4, right, tent->s.origin2 );
486 VectorMA( tent->s.origin2, -1, up, tent->s.origin2 );
487 tent->s.eventParm = 255; // don't make the explosion at the end
488 //
489 VectorCopy( impactpoint, muzzle );
490 // the player can hit him/herself with the bounced rail
491 passent = ENTITYNUM_NONE;
492 }
493 }
494 else {
495 if( LogAccuracyHit( traceEnt, ent ) ) {
496 hits++;
497 }
498 G_Damage (traceEnt, ent, ent, forward, trace.endpos, damage, 0, MOD_RAILGUN);
499 }
500#else
501 if( LogAccuracyHit( traceEnt, ent ) ) {
502 hits++;
503 }
504 G_Damage (traceEnt, ent, ent, forward, trace.endpos, damage, 0, MOD_RAILGUN);
505#endif
506 }
507 if ( trace.contents & CONTENTS_SOLID ) {
508 break; // we hit something solid enough to stop the beam
509 }
510 // unlink this entity, so the next trace will go past it
511 trap_UnlinkEntity( traceEnt );
512 unlinkedEntities[unlinked] = traceEnt;
513 unlinked++;
514 } while ( unlinked < MAX_RAIL_HITS );
515
516 // link back in any entities we unlinked
517 for ( i = 0 ; i < unlinked ; i++ ) {
518 trap_LinkEntity( unlinkedEntities[i] );
519 }
520
521 // the final trace endpos will be the terminal point of the rail trail
522
523 // snap the endpos to integers to save net bandwidth, but nudged towards the line
524 SnapVectorTowards( trace.endpos, muzzle );
525
526 // send railgun beam effect
527 tent = G_TempEntity( trace.endpos, EV_RAILTRAIL );
528
529 // set player number for custom colors on the railtrail
530 tent->s.clientNum = ent->s.clientNum;
531
532 VectorCopy( muzzle, tent->s.origin2 );
533 // move origin a bit to come closer to the drawn gun muzzle
534 VectorMA( tent->s.origin2, 4, right, tent->s.origin2 );
535 VectorMA( tent->s.origin2, -1, up, tent->s.origin2 );
536
537 // no explosion at end if SURF_NOIMPACT, but still make the trail
538 if ( trace.surfaceFlags & SURF_NOIMPACT ) {
539 tent->s.eventParm = 255; // don't make the explosion at the end
540 } else {
541 tent->s.eventParm = DirToByte( trace.plane.normal );
542 }
543 tent->s.clientNum = ent->s.clientNum;
544
545 // give the shooter a reward sound if they have made two railgun hits in a row
546 if ( hits == 0 ) {
547 // complete miss
548 ent->client->accurateCount = 0;
549 } else {
550 // check for "impressive" reward sound
551 ent->client->accurateCount += hits;
552 if ( ent->client->accurateCount >= 2 ) {
553 ent->client->accurateCount -= 2;
554 ent->client->ps.persistant[PERS_IMPRESSIVE_COUNT]++;
555 // add the sprite over the player's head
556 ent->client->ps.eFlags &= ~(EF_AWARD_IMPRESSIVE | EF_AWARD_EXCELLENT | EF_AWARD_GAUNTLET | EF_AWARD_ASSIST | EF_AWARD_DEFEND | EF_AWARD_CAP );
557 ent->client->ps.eFlags |= EF_AWARD_IMPRESSIVE;
558 ent->client->rewardTime = level.time + REWARD_SPRITE_TIME;
559 }
560 ent->client->accuracy_hits++;
561 }
562
563}

589void Weapon_HookThink (gentity_t *ent)
590{
591 if (ent->enemy) {
592 vec3_t v, oldorigin;
593
594 VectorCopy(ent->r.currentOrigin, oldorigin);
595 v[0] = ent->enemy->r.currentOrigin[0] + (ent->enemy->r.mins[0] + ent->enemy->r.maxs[0]) * 0.5;
596 v[1] = ent->enemy->r.currentOrigin[1] + (ent->enemy->r.mins[1] + ent->enemy->r.maxs[1]) * 0.5;
597 v[2] = ent->enemy->r.currentOrigin[2] + (ent->enemy->r.mins[2] + ent->enemy->r.maxs[2]) * 0.5;
598 SnapVectorTowards( v, oldorigin ); // save net bandwidth
599
600 G_SetOrigin( ent, v );
601 }
602
603 VectorCopy( ent->r.currentOrigin, ent->parent->client->ps.grapplePoint);
604}

614void Weapon_LightningFire( gentity_t *ent ) {
615 trace_t tr;
616 vec3_t end;
617#ifdef MISSIONPACK
618 vec3_t impactpoint, bouncedir;
619#endif
620 gentity_t *traceEnt, *tent;
621 int damage, i, passent;
622
623 damage = 8 * s_quadFactor;
624
625 passent = ent->s.number;
626 for (i = 0; i < 10; i++) {
627 VectorMA( muzzle, LIGHTNING_RANGE, forward, end );
628
629 trap_Trace( &tr, muzzle, NULL, NULL, end, passent, MASK_SHOT );
630
631#ifdef MISSIONPACK
632 // if not the first trace (the lightning bounced of an invulnerability sphere)
633 if (i) {
634 // add bounced off lightning bolt temp entity
635 // the first lightning bolt is a cgame only visual
636 //
637 tent = G_TempEntity( muzzle, EV_LIGHTNINGBOLT );
638 VectorCopy( tr.endpos, end );
639 SnapVector( end );
640 VectorCopy( end, tent->s.origin2 );
641 }
642#endif
643 if ( tr.entityNum == ENTITYNUM_NONE ) {
644 return;
645 }
646
647 traceEnt = &g_entities[ tr.entityNum ];
648
649 if ( traceEnt->takedamage) {
650#ifdef MISSIONPACK
651 if ( traceEnt->client && traceEnt->client->invulnerabilityTime > level.time ) {
652 if (G_InvulnerabilityEffect( traceEnt, forward, tr.endpos, impactpoint, bouncedir )) {
653 G_BounceProjectile( muzzle, impactpoint, bouncedir, end );
654 VectorCopy( impactpoint, muzzle );
655 VectorSubtract( end, impactpoint, forward );
656 VectorNormalize(forward);
657 // the player can hit him/herself with the bounced lightning
658 passent = ENTITYNUM_NONE;
659 }
660 else {
661 VectorCopy( tr.endpos, muzzle );
662 passent = traceEnt->s.number;
663 }
664 continue;
665 }
666 else {
667 G_Damage( traceEnt, ent, ent, forward, tr.endpos,
668 damage, 0, MOD_LIGHTNING);
669 }
670#else
671 G_Damage( traceEnt, ent, ent, forward, tr.endpos,
672 damage, 0, MOD_LIGHTNING);
673#endif
674 }
675
676 if ( traceEnt->takedamage && traceEnt->client ) {
677 tent = G_TempEntity( tr.endpos, EV_MISSILE_HIT );
678 tent->s.otherEntityNum = traceEnt->s.number;
679 tent->s.eventParm = DirToByte( tr.plane.normal );
680 tent->s.weapon = ent->s.weapon;
681 if( LogAccuracyHit( traceEnt, ent ) ) {
682 ent->client->accuracy_hits++;
683 }
684 } else if ( !( tr.surfaceFlags & SURF_NOIMPACT ) ) {
685 tent = G_TempEntity( tr.endpos, EV_MISSILE_MISS );
686 tent->s.eventParm = DirToByte( tr.plane.normal );
687 }
688
689 break;
690 }
691}

743/*
744===============
745LogAccuracyHit
746===============
747*/
748qboolean LogAccuracyHit( gentity_t *target, gentity_t *attacker ) {
749 if( !target->takedamage ) {
750 return qfalse;
751 }
752
753 if ( target == attacker ) {
754 return qfalse;
755 }
756
757 if( !target->client ) {
758 return qfalse;
759 }
760
761 if( !attacker->client ) {
762 return qfalse;
763 }
764
765 if( target->client->ps.stats[STAT_HEALTH] <= 0 ) {
766 return qfalse;
767 }
768
769 if ( OnSameTeam( target, attacker ) ) {
770 return qfalse;
771 }
772
773 return qtrue;
774}

1033KamikazeDamage

1036static void KamikazeDamage( gentity_t *self ) {
1037 int i;
1038 float t;
1039 gentity_t *ent;
1040 vec3_t newangles;
1041
1042 self->count += 100;
1043
1044 if (self->count >= KAMI_SHOCKWAVE_STARTTIME) {
1045 // shockwave push back
1046 t = self->count - KAMI_SHOCKWAVE_STARTTIME;
1047 KamikazeShockWave(self->s.pos.trBase, self->activator, 25, 400, (int) (float) t * KAMI_SHOCKWAVE_MAXRADIUS / (KAMI_SHOCKWAVE_ENDTIME - KAMI_SHOCKWAVE_STARTTIME) );
1048 }
1049 //
1050 if (self->count >= KAMI_EXPLODE_STARTTIME) {
1051 // do our damage
1052 t = self->count - KAMI_EXPLODE_STARTTIME;
1053 KamikazeRadiusDamage( self->s.pos.trBase, self->activator, 400, (int) (float) t * KAMI_BOOMSPHERE_MAXRADIUS / (KAMI_IMPLODE_STARTTIME - KAMI_EXPLODE_STARTTIME) );
1054 }
1055
1056 // either cycle or kill self
1057 if( self->count >= KAMI_SHOCKWAVE_ENDTIME ) {
1058 G_FreeEntity( self );
1059 return;
1060 }
1061 self->nextthink = level.time + 100;
1062
1063 // add earth quake effect
1064 newangles[0] = crandom() * 2;
1065 newangles[1] = crandom() * 2;
1066 newangles[2] = 0;
1067 for (i = 0; i < MAX_CLIENTS; i++)
1068 {
1069 ent = &g_entities[i];
1070 if (!ent->inuse)
1071 continue;
1072 if (!ent->client)
1073 continue;
1074
1075 if (ent->client->ps.groundEntityNum != ENTITYNUM_NONE) {
1076 ent->client->ps.velocity[0] += crandom() * 120;
1077 ent->client->ps.velocity[1] += crandom() * 120;
1078 ent->client->ps.velocity[2] = 30 + random() * 25;
1079 }
1080
1081 ent->client->ps.delta_angles[0] += ANGLE2SHORT(newangles[0] - self->movedir[0]);
1082 ent->client->ps.delta_angles[1] += ANGLE2SHORT(newangles[1] - self->movedir[1]);
1083 ent->client->ps.delta_angles[2] += ANGLE2SHORT(newangles[2] - self->movedir[2]);
1084 }
1085 VectorCopy(newangles, self->movedir);
1086}

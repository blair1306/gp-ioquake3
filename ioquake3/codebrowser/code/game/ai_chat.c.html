68int BotNumActivePlayers(void) {
69 int i, num;
70 char buf[MAX_INFO_STRING];
71
72 num = 0;
73 for (i = 0; i < level.maxclients; i++) {
74 trap_GetConfigstring(CS_PLAYERS+i, buf, sizeof(buf));
75 //if no config string or no name
76 if (!strlen(buf) || !strlen(Info_ValueForKey(buf, "n"))) continue;
77 //skip spectators
78 if (atoi(Info_ValueForKey(buf, "t")) == TEAM_SPECTATOR) continue;
79 //
80 num++;
81 }
82 return num;
83}

189/*
190==================
191BotRandomOpponentName
192==================
193*/
194char *BotRandomOpponentName(bot_state_t *bs) {
195 int i, count;
196 char buf[MAX_INFO_STRING];
197 int opponents[MAX_CLIENTS], numopponents;
198 static char name[32];
199
200 numopponents = 0;
201 opponents[0] = 0;
202 for (i = 0; i < level.maxclients; i++) {
203 if (i == bs->client) continue;
204 //
205 trap_GetConfigstring(CS_PLAYERS+i, buf, sizeof(buf));
206 //if no config string or no name
207 if (!strlen(buf) || !strlen(Info_ValueForKey(buf, "n"))) continue;
208 //skip spectators
209 if (atoi(Info_ValueForKey(buf, "t")) == TEAM_SPECTATOR) continue;
210 //skip team mates
211 if (BotSameTeam(bs, i)) continue;
212 //
213 opponents[numopponents] = i;
214 numopponents++;
215 }
216 count = random() * numopponents;
217 for (i = 0; i < numopponents; i++) {
218 count--;
219 if (count <= 0) {
220 EasyClientName(opponents[i], name, sizeof(name));
221 return name;
222 }
223 }
224 EasyClientName(opponents[0], name, sizeof(name));
225 return name;
226}

311/*
312==================
313BotVisibleEnemies
314==================
315*/
316int BotVisibleEnemies(bot_state_t *bs) {
317 float vis;
318 int i;
319 aas_entityinfo_t entinfo;
320
321 for (i = 0; i < MAX_CLIENTS; i++) {
322
323 if (i == bs->client) continue;
324 //
325 BotEntityInfo(i, &entinfo);
326 //
327 if (!entinfo.valid) continue;
328 //if the enemy isn't dead and the enemy isn't the bot self
329 if (EntityIsDead(&entinfo) || entinfo.number == bs->entitynum) continue;
330 //if the enemy is invisible and not shooting
331 if (EntityIsInvisible(&entinfo) && !EntityIsShooting(&entinfo)) {
332 continue;
333 }
334 //if on the same team
335 if (BotSameTeam(bs, i)) continue;
336 //check if the enemy is visible
337 vis = BotEntityVisible(bs->entitynum, bs->eye, bs->viewangles, 360, i);
338 if (vis > 0) return qtrue;
339 }
340 return qfalse;
341}

343/*
344==================
345BotValidChatPosition
346==================
347*/
348int BotValidChatPosition(bot_state_t *bs) {
349 vec3_t point, start, end, mins, maxs;
350 bsp_trace_t trace;
351
352 //if the bot is dead all positions are valid
353 if (BotIsDead(bs)) return qtrue;
354 //never start chatting with a powerup
355 if (bs->inventory[INVENTORY_QUAD] ||
356 bs->inventory[INVENTORY_ENVIRONMENTSUIT] ||
357 bs->inventory[INVENTORY_HASTE] ||
358 bs->inventory[INVENTORY_INVISIBILITY] ||
359 bs->inventory[INVENTORY_REGEN] ||
360 bs->inventory[INVENTORY_FLIGHT]) return qfalse;
361 //must be on the ground
362 //if (bs->cur_ps.groundEntityNum != ENTITYNUM_NONE) return qfalse;
363 //do not chat if in lava or slime
364 VectorCopy(bs->origin, point);
365 point[2] -= 24;
366 if (trap_PointContents(point,bs->entitynum) & (CONTENTS_LAVA|CONTENTS_SLIME)) return qfalse;
367 //do not chat if under water
368 VectorCopy(bs->origin, point);
369 point[2] += 32;
370 if (trap_PointContents(point,bs->entitynum) & MASK_WATER) return qfalse;
371 //must be standing on the world entity
372 VectorCopy(bs->origin, start);
373 VectorCopy(bs->origin, end);
374 start[2] += 1;
375 end[2] -= 10;
376 trap_AAS_PresenceTypeBoundingBox(PRESENCE_CROUCH, mins, maxs);
377 BotAI_Trace(&trace, start, mins, maxs, end, bs->client, MASK_SOLID);
378 if (trace.ent != ENTITYNUM_WORLD) return qfalse;
379 //the bot is in a position where it can chat
380 return qtrue;
381}

383/*
384==================
385BotChat_EnterGame
386==================
387*/
388int BotChat_EnterGame(bot_state_t *bs) {
389 char name[32];
390 float rnd;
391
392 if (bot_nochat.integer) return qfalse;
393 if (bs->lastchat_time > FloatTime() - TIME_BETWEENCHATTING) return qfalse;
394 //don't chat in teamplay
395 if (TeamPlayIsOn()) return qfalse;
396 // don't chat in tournament mode
397 if (gametype == GT_TOURNAMENT) return qfalse;
398 rnd = trap_Characteristic_BFloat(bs->character, CHARACTERISTIC_CHAT_ENTEREXITGAME, 0, 1);
399 if (!bot_fastchat.integer) {
400 if (random() > rnd) return qfalse;
401 }
402 if (BotNumActivePlayers() <= 1) return qfalse;
403 if (!BotValidChatPosition(bs)) return qfalse;
404 BotAI_BotInitialChat(bs, "game_enter",
405 EasyClientName(bs->client, name, 32), // 0
406 BotRandomOpponentName(bs), // 1
407 "[invalid var]", // 2
408 "[invalid var]", // 3
409 BotMapTitle(), // 4
410 NULL);
411 bs->lastchat_time = FloatTime();
412 bs->chatto = CHAT_ALL;
413 return qtrue;
414}

546==================
547BotChat_Death
548==================
549*/
550int BotChat_Death(bot_state_t *bs) {
551 char name[32];
552 float rnd;
553
554 if (bot_nochat.integer) return qfalse;
555 if (bs->lastchat_time > FloatTime() - TIME_BETWEENCHATTING) return qfalse;
556 rnd = trap_Characteristic_BFloat(bs->character, CHARACTERISTIC_CHAT_DEATH, 0, 1);
557 // don't chat in tournament mode
558 if (gametype == GT_TOURNAMENT) return qfalse;
559 //if fast chatting is off
560 if (!bot_fastchat.integer) {
561 if (random() > rnd) return qfalse;
562 }
563 if (BotNumActivePlayers() <= 1) return qfalse;
564 //
565 if (bs->lastkilledby >= 0 && bs->lastkilledby < MAX_CLIENTS)
566 EasyClientName(bs->lastkilledby, name, 32);
567 else
568 strcpy(name, "[world]");
569 //
570 if (TeamPlayIsOn() && BotSameTeam(bs, bs->lastkilledby)) {
571 if (bs->lastkilledby == bs->client) return qfalse;
572 BotAI_BotInitialChat(bs, "death_teammate", name, NULL);
573 bs->chatto = CHAT_TEAM;
574 }
575 else
576 {
577 //teamplay
578 if (TeamPlayIsOn()) {
579#ifdef MISSIONPACK
580 trap_EA_Command(bs->client, "vtaunt");
581#endif
582 return qtrue;
583 }
584 //
585 if (bs->botdeathtype == MOD_WATER)
586 BotAI_BotInitialChat(bs, "death_drown", BotRandomOpponentName(bs), NULL);
587 else if (bs->botdeathtype == MOD_SLIME)
588 BotAI_BotInitialChat(bs, "death_slime", BotRandomOpponentName(bs), NULL);
589 else if (bs->botdeathtype == MOD_LAVA)
590 BotAI_BotInitialChat(bs, "death_lava", BotRandomOpponentName(bs), NULL);
591 else if (bs->botdeathtype == MOD_FALLING)
592 BotAI_BotInitialChat(bs, "death_cratered", BotRandomOpponentName(bs), NULL);
593 else if (bs->botsuicide || //all other suicides by own weapon
594 bs->botdeathtype == MOD_CRUSH ||
595 bs->botdeathtype == MOD_SUICIDE ||
596 bs->botdeathtype == MOD_TARGET_LASER ||
597 bs->botdeathtype == MOD_TRIGGER_HURT ||
598 bs->botdeathtype == MOD_UNKNOWN)
599 BotAI_BotInitialChat(bs, "death_suicide", BotRandomOpponentName(bs), NULL);
600 else if (bs->botdeathtype == MOD_TELEFRAG)
601 BotAI_BotInitialChat(bs, "death_telefrag", name, NULL);
602#ifdef MISSIONPACK
603 else if (bs->botdeathtype == MOD_KAMIKAZE && trap_BotNumInitialChats(bs->cs, "death_kamikaze"))
604 BotAI_BotInitialChat(bs, "death_kamikaze", name, NULL);
605#endif
606 else {
607 if ((bs->botdeathtype == MOD_GAUNTLET ||
608 bs->botdeathtype == MOD_RAILGUN ||
609 bs->botdeathtype == MOD_BFG ||
610 bs->botdeathtype == MOD_BFG_SPLASH) && random() < 0.5) {
611
612 if (bs->botdeathtype == MOD_GAUNTLET)
613 BotAI_BotInitialChat(bs, "death_gauntlet",
614 name, // 0
615 BotWeaponNameForMeansOfDeath(bs->botdeathtype), // 1
616 NULL);
617 else if (bs->botdeathtype == MOD_RAILGUN)
618 BotAI_BotInitialChat(bs, "death_rail",
619 name, // 0
620 BotWeaponNameForMeansOfDeath(bs->botdeathtype), // 1
621 NULL);
622 else
623 BotAI_BotInitialChat(bs, "death_bfg",
624 name, // 0
625 BotWeaponNameForMeansOfDeath(bs->botdeathtype), // 1
626 NULL);
627 }
628 //choose between insult and praise
629 else if (random() < trap_Characteristic_BFloat(bs->character, CHARACTERISTIC_CHAT_INSULT, 0, 1)) {
630 BotAI_BotInitialChat(bs, "death_insult",
631 name, // 0
632 BotWeaponNameForMeansOfDeath(bs->botdeathtype), // 1
633 NULL);
634 }
635 else {
636 BotAI_BotInitialChat(bs, "death_praise",
637 name, // 0
638 BotWeaponNameForMeansOfDeath(bs->botdeathtype), // 1
639 NULL);
640 }
641 }
642 bs->chatto = CHAT_ALL;
643 }
644 bs->lastchat_time = FloatTime();
645 return qtrue;
646}

653int BotChat_Kill(bot_state_t *bs) {
654 char name[32];
655 float rnd;
656
657 if (bot_nochat.integer) return qfalse;
658 if (bs->lastchat_time > FloatTime() - TIME_BETWEENCHATTING) return qfalse;
659 rnd = trap_Characteristic_BFloat(bs->character, CHARACTERISTIC_CHAT_KILL, 0, 1);
660 // don't chat in tournament mode
661 if (gametype == GT_TOURNAMENT) return qfalse;
662 //if fast chat is off
663 if (!bot_fastchat.integer) {
664 if (random() > rnd) return qfalse;
665 }
666 if (bs->lastkilledplayer == bs->client) return qfalse;
667 if (BotNumActivePlayers() <= 1) return qfalse;
668 if (!BotValidChatPosition(bs)) return qfalse;
669 //
670 if (BotVisibleEnemies(bs)) return qfalse;
671 //
672 EasyClientName(bs->lastkilledplayer, name, 32);
673 //
674 bs->chatto = CHAT_ALL;
675 if (TeamPlayIsOn() && BotSameTeam(bs, bs->lastkilledplayer)) {
676 BotAI_BotInitialChat(bs, "kill_teammate", name, NULL);
677 bs->chatto = CHAT_TEAM;
678 }
679 else
680 {
681 //don't chat in teamplay
682 if (TeamPlayIsOn()) {
683#ifdef MISSIONPACK
684 trap_EA_Command(bs->client, "vtaunt");
685#endif
686 return qfalse; // don't wait
687 }
688 //
689 if (bs->enemydeathtype == MOD_GAUNTLET) {
690 BotAI_BotInitialChat(bs, "kill_gauntlet", name, NULL);
691 }
692 else if (bs->enemydeathtype == MOD_RAILGUN) {
693 BotAI_BotInitialChat(bs, "kill_rail", name, NULL);
694 }
695 else if (bs->enemydeathtype == MOD_TELEFRAG) {
696 BotAI_BotInitialChat(bs, "kill_telefrag", name, NULL);
697 }
698#ifdef MISSIONPACK
699 else if (bs->botdeathtype == MOD_KAMIKAZE && trap_BotNumInitialChats(bs->cs, "kill_kamikaze"))
700 BotAI_BotInitialChat(bs, "kill_kamikaze", name, NULL);
701#endif
702 //choose between insult and praise
703 else if (random() < trap_Characteristic_BFloat(bs->character, CHARACTERISTIC_CHAT_INSULT, 0, 1)) {
704 BotAI_BotInitialChat(bs, "kill_insult", name, NULL);

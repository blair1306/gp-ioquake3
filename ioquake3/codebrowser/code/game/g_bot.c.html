63/*
64===============
65G_ParseInfos
66===============
67*/
68int G_ParseInfos( char *buf, int max, char *infos[] ) {
69 char *token;
70 int count;
71 char key[MAX_TOKEN_CHARS];
72 char info[MAX_INFO_STRING];
73
74 count = 0;
75
76 while ( 1 ) {
77 token = COM_Parse( &buf );
78 if ( !token[0] ) {
79 break;
80 }
81 if ( strcmp( token, "{" ) ) {
82 Com_Printf( "Missing { in info file\n" );
83 break;
84 }
85
86 if ( count == max ) {
87 Com_Printf( "Max infos exceeded\n" );
88 break;
89 }
90
91 info[0] = '\0';
92 while ( 1 ) {
93 token = COM_ParseExt( &buf, qtrue );
94 if ( !token[0] ) {
95 Com_Printf( "Unexpected end of info file\n" );
96 break;
97 }
98 if ( !strcmp( token, "}" ) ) {
99 break;
100 }
101 Q_strncpyz( key, token, sizeof( key ) );
102
103 token = COM_ParseExt( &buf, qfalse );
104 if ( !token[0] ) {
105 strcpy( token, "<NULL>" );
106 }
107 Info_SetValueForKey( info, key, token );
108 }
109 //NOTE: extra space for arena number
110 infos[count] = G_Alloc(strlen(info) + strlen("\\num\\") + strlen(va("%d", MAX_ARENAS)) + 1);
111 if (infos[count]) {
112 strcpy(infos[count], info);
113 count++;
114 }
115 }
116 return count;
117}

119/*
120===============
121G_LoadArenasFromFile
122===============
123*/
124static void G_LoadArenasFromFile( char *filename ) {
125 int len;
126 fileHandle_t f;
127 char buf[MAX_ARENAS_TEXT];
128
129 len = trap_FS_FOpenFile( filename, &f, FS_READ );
130 if ( !f ) {
131 trap_Print( va( S_COLOR_RED "file not found: %s\n", filename ) );
132 return;
133 }
134 if ( len >= MAX_ARENAS_TEXT ) {
135 trap_FS_FCloseFile( f );
136 trap_Print( va( S_COLOR_RED "file too large: %s is %i, max allowed is %i\n", filename, len, MAX_ARENAS_TEXT ) );
137 return;
138 }
139
140 trap_FS_Read( buf, len, f );
141 buf[len] = 0;
142 trap_FS_FCloseFile( f );
143
144 g_numArenas += G_ParseInfos( buf, MAX_ARENAS - g_numArenas, &g_arenaInfos[g_numArenas] );
145}

147/*
148===============
149G_LoadArenas
150===============
151*/
152static void G_LoadArenas( void ) {
153 int numdirs;
154 vmCvar_t arenasFile;
155 char filename[128];
156 char dirlist[1024];
157 char* dirptr;
158 int i, n;
159 int dirlen;
160
161 g_numArenas = 0;
162
163 trap_Cvar_Register( &arenasFile, "g_arenasFile", "", CVAR_INIT|CVAR_ROM );
164 if( *arenasFile.string ) {
165 G_LoadArenasFromFile(arenasFile.string);
166 }
167 else {
168 G_LoadArenasFromFile("scripts/arenas.txt");
169 }
170
171 // get all arenas from .arena files
172 numdirs = trap_FS_GetFileList("scripts", ".arena", dirlist, 1024 );
173 dirptr = dirlist;
174 for (i = 0; i < numdirs; i++, dirptr += dirlen+1) {
175 dirlen = strlen(dirptr);
176 strcpy(filename, "scripts/");
177 strcat(filename, dirptr);
178 G_LoadArenasFromFile(filename);
179 }
180 trap_Print( va( "%i arenas parsed\n", g_numArenas ) );
181
182 for( n = 0; n < g_numArenas; n++ ) {
183 Info_SetValueForKey( g_arenaInfos[n], "num", va( "%i", n ) );
184 }
185}

231/*
232===============
233G_CountBotPlayersByName
234
235Check connected and connecting (delay join) bots.
236
237Returns number of bots with name on specified team or whole server if team is -1.
238===============
239*/
240int G_CountBotPlayersByName( const char *name, int team ) {
241 int i, num;
242 gclient_t *cl;
243
244 num = 0;
245 for ( i=0 ; i< g_maxclients.integer ; i++ ) {
246 cl = level.clients + i;
247 if ( cl->pers.connected == CON_DISCONNECTED ) {
248 continue;
249 }
250 if ( !(g_entities[i].r.svFlags & SVF_BOT) ) {
251 continue;
252 }
253 if ( team >= 0 && cl->sess.sessionTeam != team ) {
254 continue;
255 }
256 if ( name && Q_stricmp( name, cl->pers.netname ) ) {
257 continue;
258 }
259 num++;
260 }
261 return num;
262}

264/*
265===============
266G_SelectRandomBotInfo
267
268Get random least used bot info on team or whole server if team is -1.
269===============
270*/
271int G_SelectRandomBotInfo( int team ) {
272 int selection[MAX_BOTS];
273 int n, num;
274 int count, bestCount;
275 char *value;
276
277 // don't add duplicate bots to the server if there are less bots than bot types
278 if ( team != -1 && G_CountBotPlayersByName( NULL, -1 ) < g_numBots ) {
279 team = -1;
280 }
281
282 num = 0;
283 bestCount = MAX_CLIENTS;
284 for ( n = 0; n < g_numBots ; n++ ) {
285 value = Info_ValueForKey( g_botInfos[n], "funname" );
286 if ( !value[0] ) {
287 value = Info_ValueForKey( g_botInfos[n], "name" );
288 }
289 //
290 count = G_CountBotPlayersByName( value, team );
291
292 if ( count < bestCount ) {
293 bestCount = count;
294 num = 0;
295 }
296
297 if ( count == bestCount ) {
298 selection[num++] = n;
299
300 if ( num == MAX_BOTS ) {
301 break;
302 }
303 }
304 }
305
306 if ( num > 0 ) {
307 num = random() * ( num - 1 );
308 return selection[num];
309 }
310
311 return -1;
312}

410/*
411===============
412G_CheckMinimumPlayers
413===============
414*/
415void G_CheckMinimumPlayers( void ) {
416 int minplayers;
417 int humanplayers, botplayers;
418 static int checkminimumplayers_time;
419
420 if (level.intermissiontime) return;
421 //only check once each 10 seconds
422 if (checkminimumplayers_time > level.time - 10000) {
423 return;
424 }
425 checkminimumplayers_time = level.time;
426 trap_Cvar_Update(&bot_minplayers);
427 minplayers = bot_minplayers.integer;
428 if (minplayers <= 0) return;
429
430 if (g_gametype.integer >= GT_TEAM) {
431 if (minplayers >= g_maxclients.integer / 2) {
432 minplayers = (g_maxclients.integer / 2) -1;
433 }
434
435 humanplayers = G_CountHumanPlayers( TEAM_RED );
436 botplayers = G_CountBotPlayers( TEAM_RED );
437 //
438 if (humanplayers + botplayers < minplayers) {
439 G_AddRandomBot( TEAM_RED );
440 } else if (humanplayers + botplayers > minplayers && botplayers) {
441 G_RemoveRandomBot( TEAM_RED );
442 }
443 //
444 humanplayers = G_CountHumanPlayers( TEAM_BLUE );
445 botplayers = G_CountBotPlayers( TEAM_BLUE );
446 //
447 if (humanplayers + botplayers < minplayers) {
448 G_AddRandomBot( TEAM_BLUE );
449 } else if (humanplayers + botplayers > minplayers && botplayers) {
450 G_RemoveRandomBot( TEAM_BLUE );
451 }
452 }
453 else if (g_gametype.integer == GT_TOURNAMENT ) {
454 if (minplayers >= g_maxclients.integer) {
455 minplayers = g_maxclients.integer-1;
456 }
457 humanplayers = G_CountHumanPlayers( -1 );
458 botplayers = G_CountBotPlayers( -1 );
459 //
460 if (humanplayers + botplayers < minplayers) {
461 G_AddRandomBot( TEAM_FREE );
462 } else if (humanplayers + botplayers > minplayers && botplayers) {
463 // try to remove spectators first
464 if (!G_RemoveRandomBot( TEAM_SPECTATOR )) {
465 // just remove the bot that is playing
466 G_RemoveRandomBot( -1 );
467 }
468 }
469 }
470 else if (g_gametype.integer == GT_FFA) {
471 if (minplayers >= g_maxclients.integer) {
472 minplayers = g_maxclients.integer-1;
473 }
474 humanplayers = G_CountHumanPlayers( TEAM_FREE );
475 botplayers = G_CountBotPlayers( TEAM_FREE );
476 //
477 if (humanplayers + botplayers < minplayers) {
478 G_AddRandomBot( TEAM_FREE );
479 } else if (humanplayers + botplayers > minplayers && botplayers) {
480 G_RemoveRandomBot( TEAM_FREE );
481 }
482 }
483}

485/*
486===============
487G_CheckBotSpawn
488===============
489*/
490void G_CheckBotSpawn( void ) {
491 int n;
492 char userinfo[MAX_INFO_VALUE];
493
494 G_CheckMinimumPlayers();
495
496 for( n = 0; n < BOT_SPAWN_QUEUE_DEPTH; n++ ) {
497 if( !botSpawnQueue[n].spawnTime ) {
498 continue;
499 }
500 if ( botSpawnQueue[n].spawnTime > level.time ) {
501 continue;
502 }
503 ClientBegin( botSpawnQueue[n].clientNum );
504 botSpawnQueue[n].spawnTime = 0;
505
506 if( g_gametype.integer == GT_SINGLE_PLAYER ) {
507 trap_GetUserinfo( botSpawnQueue[n].clientNum, userinfo, sizeof(userinfo) );
508 PlayerIntroSound( Info_ValueForKey (userinfo, "model") );
509 }
510 }
511}

578/*
579===============
580G_AddBot
581===============
582*/
583static void G_AddBot( const char *name, float skill, const char *team, int delay, char *altname) {
584 int clientNum;
585 int teamNum;
586 int botinfoNum;
587 char *botinfo;
588 char *key;
589 char *s;
590 char *botname;
591 char *model;
592 char *headmodel;
593 char userinfo[MAX_INFO_STRING];
594
595 // have the server allocate a client slot
596 clientNum = trap_BotAllocateClient();
597 if ( clientNum == -1 ) {
598 G_Printf( S_COLOR_RED "Unable to add bot. All player slots are in use.\n" );
599 G_Printf( S_COLOR_RED "Start server with more 'open' slots (or check setting of sv_maxclients cvar).\n" );
600 return;
601 }
602
603 // set default team
604 if( !team || !*team ) {
605 if( g_gametype.integer >= GT_TEAM ) {
606 if( PickTeam(clientNum) == TEAM_RED) {
607 team = "red";
608 }
609 else {
610 team = "blue";
611 }
612 }
613 else {
614 team = "free";
615 }
616 }
617
618 // get the botinfo from bots.txt
619 if ( Q_stricmp( name, "random" ) == 0 ) {
620 if ( Q_stricmp( team, "red" ) == 0 || Q_stricmp( team, "r" ) == 0 ) {
621 teamNum = TEAM_RED;
622 }
623 else if ( Q_stricmp( team, "blue" ) == 0 || Q_stricmp( team, "b" ) == 0 ) {
624 teamNum = TEAM_BLUE;
625 }
626 else if ( !Q_stricmp( team, "spectator" ) || !Q_stricmp( team, "s" ) ) {
627 teamNum = TEAM_SPECTATOR;
628 }
629 else {
630 teamNum = TEAM_FREE;
631 }
632
633 botinfoNum = G_SelectRandomBotInfo( teamNum );
634
635 if ( botinfoNum < 0 ) {
636 G_Printf( S_COLOR_RED "Error: Cannot add random bot, no bot info available.\n" );
637 trap_BotFreeClient( clientNum );
638 return;
639 }
640
641 botinfo = G_GetBotInfoByNumber( botinfoNum );
642 }
643 else {
644 botinfo = G_GetBotInfoByName( name );
645 }
646
647 if ( !botinfo ) {
648 G_Printf( S_COLOR_RED "Error: Bot '%s' not defined\n", name );
649 trap_BotFreeClient( clientNum );
650 return;
651 }
652
653 // create the bot's userinfo
654 userinfo[0] = '\0';
655
656 botname = Info_ValueForKey( botinfo, "funname" );
657 if( !botname[0] ) {
658 botname = Info_ValueForKey( botinfo, "name" );
659 }
660 // check for an alternative name
661 if (altname && altname[0]) {
662 botname = altname;
663 }
664 Info_SetValueForKey( userinfo, "name", botname );
665 Info_SetValueForKey( userinfo, "rate", "25000" );
666 Info_SetValueForKey( userinfo, "snaps", "20" );
667 Info_SetValueForKey( userinfo, "skill", va("%.2f", skill) );
668 Info_SetValueForKey( userinfo, "teampref", team );
669
670 if ( skill >= 1 && skill < 2 ) {
671 Info_SetValueForKey( userinfo, "handicap", "50" );
672 }
673 else if ( skill >= 2 && skill < 3 ) {
674 Info_SetValueForKey( userinfo, "handicap", "70" );
675 }
676 else if ( skill >= 3 && skill < 4 ) {
677 Info_SetValueForKey( userinfo, "handicap", "90" );
678 }
679
680 key = "model";
681 model = Info_ValueForKey( botinfo, key );
682 if ( !*model ) {
683 model = "visor/default";
684 }
685 Info_SetValueForKey( userinfo, key, model );
686 key = "team_model";
687 Info_SetValueForKey( userinfo, key, model );
688
689 key = "headmodel";
690 headmodel = Info_ValueForKey( botinfo, key );
691 if ( !*headmodel ) {
692 headmodel = model;
693 }
694 Info_SetValueForKey( userinfo, key, headmodel );
695 key = "team_headmodel";
696 Info_SetValueForKey( userinfo, key, headmodel );
697
698 key = "gender";
699 s = Info_ValueForKey( botinfo, key );
700 if ( !*s ) {
701 s = "male";
702 }
703 Info_SetValueForKey( userinfo, "sex", s );
704
705 key = "color1";
706 s = Info_ValueForKey( botinfo, key );
707 if ( !*s ) {
708 s = "4";
709 }
710 Info_SetValueForKey( userinfo, key, s );
711
712 key = "color2";
713 s = Info_ValueForKey( botinfo, key );
714 if ( !*s ) {
715 s = "5";
716 }
717 Info_SetValueForKey( userinfo, key, s );
718
719 s = Info_ValueForKey(botinfo, "aifile");
720 if (!*s ) {
721 trap_Print( S_COLOR_RED "Error: bot has no aifile specified\n" );
722 trap_BotFreeClient( clientNum );
723 return;
724 }
725 Info_SetValueForKey( userinfo, "characterfile", s );
726
727 // register the userinfo
728 trap_SetUserinfo( clientNum, userinfo );
729
730 // have it connect to the game as a normal client
731 if ( ClientConnect( clientNum, qtrue, qtrue ) ) {
732 return;
733 }
734
735 if( delay == 0 ) {
736 ClientBegin( clientNum );
737 return;
738 }
739
740 AddBotToSpawnQueue( clientNum, delay );
741}

744/*
745===============
746Svcmd_AddBot_f
747===============
748*/
749void Svcmd_AddBot_f( void ) {
750 float skill;
751 int delay;
752 char name[MAX_TOKEN_CHARS];
753 char altname[MAX_TOKEN_CHARS];
754 char string[MAX_TOKEN_CHARS];
755 char team[MAX_TOKEN_CHARS];
756
757 // are bots enabled?
758 if ( !trap_Cvar_VariableIntegerValue( "bot_enable" ) ) {
759 return;
760 }
761
762 // name
763 trap_Argv( 1, name, sizeof( name ) );
764 if ( !name[0] ) {
765 trap_Print( "Usage: Addbot <botname> [skill 1-5] [team] [msec delay] [altname]\n" );
766 return;
767 }
768
769 // skill
770 trap_Argv( 2, string, sizeof( string ) );
771 if ( !string[0] ) {
772 skill = 4;
773 }
774 else {
775 skill = Com_Clamp( 1, 5, atof( string ) );
776 }
777
778 // team
779 trap_Argv( 3, team, sizeof( team ) );
780
781 // delay
782 trap_Argv( 4, string, sizeof( string ) );
783 if ( !string[0] ) {
784 delay = 0;
785 }
786 else {
787 delay = atoi( string );
788 }
789
790 // alternative name
791 trap_Argv( 5, altname, sizeof( altname ) );
792
793 G_AddBot( name, skill, team, delay, altname );
794
795 // if this was issued during gameplay and we are playing locally,
796 // go ahead and load the bot's media immediately
797 if ( level.time - level.startTime > 1000 &&
798 trap_Cvar_VariableIntegerValue( "cl_running" ) ) {
799 trap_SendServerCommand( -1, "loaddefered\n" ); // FIXME: spelled wrong, but not changing for demo
800 }
801}

803/*
804===============
805Svcmd_BotList_f

807*/
808void Svcmd_BotList_f( void ) {
809 int i;
810 char name[MAX_TOKEN_CHARS];
811 char funname[MAX_TOKEN_CHARS];
812 char model[MAX_TOKEN_CHARS];
813 char aifile[MAX_TOKEN_CHARS];
814
815 trap_Print("^1name model aifile funname\n");
816 for (i = 0; i < g_numBots; i++) {
817 strcpy(name, Info_ValueForKey( g_botInfos[i], "name" ));

838/*
839===============
840G_SpawnBots
841===============
842*/
843static void G_SpawnBots( char *botList, int baseDelay ) {
844 char *bot;
845 char *p;
846 float skill;
847 int delay;
848 char bots[MAX_INFO_VALUE];
849
850 podium1 = NULL;
851 podium2 = NULL;
852 podium3 = NULL;
853
854 skill = trap_Cvar_VariableValue( "g_spSkill" );
855 if( skill < 1 ) {
856 trap_Cvar_Set( "g_spSkill", "1" );
857 skill = 1;
858 }
859 else if ( skill > 5 ) {
860 trap_Cvar_Set( "g_spSkill", "5" );
861 skill = 5;
862 }
863
864 Q_strncpyz( bots, botList, sizeof(bots) );
865 p = &bots[0];
866 delay = baseDelay;
867 while( *p ) {
868 //skip spaces
869 while( *p && *p == ' ' ) {
870 p++;
871 }
872 if( !*p ) {
873 break;
874 }
875
876 // mark start of bot name
877 bot = p;
878
879 // skip until space of null
880 while( *p && *p != ' ' ) {
881 p++;
882 }
883 if( *p ) {
884 *p++ = 0;
885 }
886
887 // we must add the bot this way, calling G_AddBot directly at this stage
888 // does "Bad Things"
889 trap_SendConsoleCommand( EXEC_INSERT, va("addbot %s %f free %i\n", bot, skill, delay) );
890
891 delay += BOT_BEGIN_DELAY_INCREMENT;
892 }
893}

897===============
898G_LoadBotsFromFile
899===============
900*/
901static void G_LoadBotsFromFile( char *filename ) {
902 int len;
903 fileHandle_t f;
904 char buf[MAX_BOTS_TEXT];
905
906 len = trap_FS_FOpenFile( filename, &f, FS_READ );
907 if ( !f ) {
908 trap_Print( va( S_COLOR_RED "file not found: %s\n", filename ) );
909 return;
910 }
911 if ( len >= MAX_BOTS_TEXT ) {
912 trap_Print( va( S_COLOR_RED "file too large: %s is %i, max allowed is %i\n", filename, len, MAX_BOTS_TEXT ) );
913 trap_FS_FCloseFile( f );
914 return;
915 }
916
917 trap_FS_Read( buf, len, f );
918 buf[len] = 0;
919 trap_FS_FCloseFile( f );
920
921 g_numBots += G_ParseInfos( buf, MAX_BOTS - g_numBots, &g_botInfos[g_numBots] );
922}

929static void G_LoadBots( void ) {
930 vmCvar_t botsFile;
931 int numdirs;
932 char filename[128];

959 G_LoadBotsFromFile(filename);

1004void G_InitBots( qboolean restart ) {
1005 int fragLimit;
1006 int timeLimit;
1007 const char *arenainfo;
1008 char *strValue;
1009 int basedelay;
1010 char map[MAX_QPATH];
1011 char serverinfo[MAX_INFO_STRING];
1012
1013 G_LoadBots();
1014 G_LoadArenas();
1015
1016 trap_Cvar_Register( &bot_minplayers, "bot_minplayers", "0", CVAR_SERVERINFO );
1017
1018 if( g_gametype.integer == GT_SINGLE_PLAYER ) {
1019 trap_GetServerinfo( serverinfo, sizeof(serverinfo) );
1020 Q_strncpyz( map, Info_ValueForKey( serverinfo, "mapname" ), sizeof(map) );
1021 arenainfo = G_GetArenaInfoByMap( map );
1022 if ( !arenainfo ) {
1023 return;
1024 }
1025
1026 strValue = Info_ValueForKey( arenainfo, "fraglimit" );
1027 fragLimit = atoi( strValue );
1028 if ( fragLimit ) {
1029 trap_Cvar_Set( "fraglimit", strValue );
1030 }
1031 else {
1032 trap_Cvar_Set( "fraglimit", "0" );
1033 }
1034
1035 strValue = Info_ValueForKey( arenainfo, "timelimit" );
1036 timeLimit = atoi( strValue );
1037 if ( timeLimit ) {
1038 trap_Cvar_Set( "timelimit", strValue );
1039 }
1040 else {
1041 trap_Cvar_Set( "timelimit", "0" );
1042 }
1043
1044 if ( !fragLimit && !timeLimit ) {
1045 trap_Cvar_Set( "fraglimit", "10" );
1046 trap_Cvar_Set( "timelimit", "0" );
1047 }
1048
1049 basedelay = BOT_BEGIN_DELAY_BASE;
1050 strValue = Info_ValueForKey( arenainfo, "special" );
1051 if( Q_stricmp( strValue, "training" ) == 0 ) {
1052 basedelay += 10000;
1053 }
1054
1055 if( !restart ) {
1056 G_SpawnBots( Info_ValueForKey( arenainfo, "bots" ), basedelay );
1057 }
1058 }

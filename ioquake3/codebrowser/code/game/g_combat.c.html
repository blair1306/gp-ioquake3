45/*
46============
47AddScore
48
49Adds score to both the client and his team
50============
51*/
52void AddScore( gentity_t *ent, vec3_t origin, int score ) {
53 if ( !ent->client ) {
54 return;
55 }
56 // no scoring during pre-match warmup
57 if ( level.warmupTime ) {
58 return;
59 }
60 // show score plum
61 ScorePlum(ent, origin, score);
62 //
63 ent->client->ps.persistant[PERS_SCORE] += score;
64 if ( g_gametype.integer == GT_TEAM )
65 level.teamScores[ ent->client->ps.persistant[PERS_TEAM] ] += score;
66 CalculateRanks();
67}

69/*
70=================
71TossClientItems
72
73Toss the weapon and powerups for the killed player
74=================
75*/
76void TossClientItems( gentity_t *self ) {
77 gitem_t *item;
78 int weapon;
79 float angle;
80 int i;
81 gentity_t *drop;
82
83 // drop the weapon if not a gauntlet or machinegun
84 weapon = self->s.weapon;
85
86 // make a special check to see if they are changing to a new
87 // weapon that isn't the mg or gauntlet. Without this, a client
88 // can pick up a weapon, be killed, and not drop the weapon because
89 // their weapon change hasn't completed yet and they are still holding the MG.
90 if ( weapon == WP_MACHINEGUN || weapon == WP_GRAPPLING_HOOK ) {
91 if ( self->client->ps.weaponstate == WEAPON_DROPPING ) {
92 weapon = self->client->pers.cmd.weapon;
93 }
94 if ( !( self->client->ps.stats[STAT_WEAPONS] & ( 1 << weapon ) ) ) {
95 weapon = WP_NONE;
96 }
97 }
98
99 if ( weapon > WP_MACHINEGUN && weapon != WP_GRAPPLING_HOOK &&
100 self->client->ps.ammo[ weapon ] ) {
101 // find the item type for this weapon
102 item = BG_FindItemForWeapon( weapon );
103
104 // spawn the item
105 Drop_Item( self, item, 0 );
106 }
107
108 // drop all the powerups if not in teamplay
109 if ( g_gametype.integer != GT_TEAM ) {
110 angle = 45;
111 for ( i = 1 ; i < PW_NUM_POWERUPS ; i++ ) {
112 if ( self->client->ps.powerups[ i ] > level.time ) {
113 item = BG_FindItemForPowerup( i );
114 if ( !item ) {
115 continue;
116 }
117 drop = Drop_Item( self, item, angle );
118 // decide how many seconds it has left
119 drop->count = ( self->client->ps.powerups[ i ] - level.time ) / 1000;
120 if ( drop->count < 1 ) {
121 drop->count = 1;
122 }
123 angle += 45;
124 }
125 }
126 }
127}

132=================
133TossClientCubes

136extern gentity_t *neutralObelisk;

138void TossClientCubes( gentity_t *self ) {
139 gitem_t *item;
140 gentity_t *drop;
141 vec3_t velocity;
142 vec3_t angles;
143 vec3_t origin;
144
145 self->client->ps.generic1 = 0;
146
147 // this should never happen but we should never
148 // get the server to crash due to skull being spawned in
149 if (!G_EntitiesFree()) {
150 return;
151 }
152
153 if( self->client->sess.sessionTeam == TEAM_RED ) {
154 item = BG_FindItem( "Red Cube" );
155 }
156 else {
157 item = BG_FindItem( "Blue Cube" );
158 }
159
160 angles[YAW] = (float)(level.time % 360);
161 angles[PITCH] = 0; // always forward
162 angles[ROLL] = 0;
163
164 AngleVectors( angles, velocity, NULL, NULL );
165 VectorScale( velocity, 150, velocity );
166 velocity[2] += 200 + crandom() * 50;
167
168 if( neutralObelisk ) {
169 VectorCopy( neutralObelisk->s.pos.trBase, origin );
170 origin[2] += 44;
171 } else {
172 VectorClear( origin ) ;
173 }
174
175 drop = LaunchItem( item, origin, velocity );
176
177 drop->nextthink = level.time + g_cubeTimeout.integer * 1000;
178 drop->think = G_FreeEntity;
179 drop->spawnflags = self->client->sess.sessionTeam;
180}

212/*
213==================
214LookAtKiller
215==================
216*/
217void LookAtKiller( gentity_t *self, gentity_t *inflictor, gentity_t *attacker ) {
218 vec3_t dir;
219
220 if ( attacker && attacker != self ) {
221 VectorSubtract (attacker->s.pos.trBase, self->s.pos.trBase, dir);
222 } else if ( inflictor && inflictor != self ) {
223 VectorSubtract (inflictor->s.pos.trBase, self->s.pos.trBase, dir);
224 } else {
225 self->client->ps.stats[STAT_DEAD_YAW] = self->s.angles[YAW];
226 return;
227 }
228
229 self->client->ps.stats[STAT_DEAD_YAW] = vectoyaw ( dir );
230}

232/*
233==================
234GibEntity
235==================
236*/
237void GibEntity( gentity_t *self, int killer ) {
238 gentity_t *ent;
239 int i;
240
241 //if this entity still has kamikaze
242 if (self->s.eFlags & EF_KAMIKAZE) {
243 // check if there is a kamikaze timer around for this owner
244 for (i = 0; i < level.num_entities; i++) {
245 ent = &g_entities[i];
246 if (!ent->inuse)
247 continue;
248 if (ent->activator != self)
249 continue;
250 if (strcmp(ent->classname, "kamikaze timer"))
251 continue;
252 G_FreeEntity(ent);
253 break;
254 }
255 }
256 G_AddEvent( self, EV_GIB_PLAYER, killer );
257 self->takedamage = qfalse;
258 self->s.eType = ET_INVISIBLE;
259 self->r.contents = 0;
260}

346/*
347==================
348CheckAlmostCapture
349==================
350*/
351void CheckAlmostCapture( gentity_t *self, gentity_t *attacker ) {
352 gentity_t *ent;
353 vec3_t dir;
354 char *classname;
355
356 // if this player was carrying a flag
357 if ( self->client->ps.powerups[PW_REDFLAG] ||
358 self->client->ps.powerups[PW_BLUEFLAG] ||
359 self->client->ps.powerups[PW_NEUTRALFLAG] ) {
360 // get the goal flag this player should have been going for
361 if ( g_gametype.integer == GT_CTF ) {
362 if ( self->client->sess.sessionTeam == TEAM_BLUE ) {
363 classname = "team_CTF_blueflag";
364 }
365 else {
366 classname = "team_CTF_redflag";
367 }
368 }
369 else {
370 if ( self->client->sess.sessionTeam == TEAM_BLUE ) {
371 classname = "team_CTF_redflag";
372 }
373 else {
374 classname = "team_CTF_blueflag";
375 }
376 }
377 ent = NULL;
378 do
379 {
380 ent = G_Find(ent, FOFS(classname), classname);
381 } while (ent && (ent->flags & FL_DROPPED_ITEM));
382 // if we found the destination flag and it's not picked up
383 if (ent && !(ent->r.svFlags & SVF_NOCLIENT) ) {
384 // if the player was *very* close
385 VectorSubtract( self->client->ps.origin, ent->s.origin, dir );
386 if ( VectorLength(dir) < 200 ) {
387 self->client->ps.persistant[PERS_PLAYEREVENTS] ^= PLAYEREVENT_HOLYSHIT;
388 if ( attacker->client ) {
389 attacker->client->ps.persistant[PERS_PLAYEREVENTS] ^= PLAYEREVENT_HOLYSHIT;
390 }
391 }
392 }
393 }

429/*
430==================
431player_die
432==================
433*/
434void player_die( gentity_t *self, gentity_t *inflictor, gentity_t *attacker, int damage, int meansOfDeath ) {
435 gentity_t *ent;
436 int anim;
437 int contents;
438 int killer;
439 int i;
440 char *killerName, *obit;
441
442 if ( self->client->ps.pm_type == PM_DEAD ) {
443 return;
444 }
445
446 if ( level.intermissiontime ) {
447 return;
448 }
449
450 // check for an almost capture
451 CheckAlmostCapture( self, attacker );
452 // check for a player that almost brought in cubes
453 CheckAlmostScored( self, attacker );
454
455 if (self->client && self->client->hook) {
456 Weapon_HookFree(self->client->hook);
457 }
458#ifdef MISSIONPACK
459 if ((self->client->ps.eFlags & EF_TICKING) && self->activator) {
460 self->client->ps.eFlags &= ~EF_TICKING;
461 self->activator->think = G_FreeEntity;
462 self->activator->nextthink = level.time;
463 }
464#endif
465 self->client->ps.pm_type = PM_DEAD;
466
467 if ( attacker ) {
468 killer = attacker->s.number;
469 if ( attacker->client ) {
470 killerName = attacker->client->pers.netname;
471 } else {
472 killerName = "<non-client>";
473 }
474 } else {
475 killer = ENTITYNUM_WORLD;
476 killerName = "<world>";
477 }
478
479 if ( killer < 0 || killer >= MAX_CLIENTS ) {
480 killer = ENTITYNUM_WORLD;
481 killerName = "<world>";
482 }
483
484 if ( meansOfDeath < 0 || meansOfDeath >= ARRAY_LEN( modNames ) ) {
485 obit = "<bad obituary>";
486 } else {
487 obit = modNames[meansOfDeath];
488 }
489
490 G_LogPrintf("Kill: %i %i %i: %s killed %s by %s\n",
491 killer, self->s.number, meansOfDeath, killerName,
492 self->client->pers.netname, obit );
493
494 // broadcast the death event to everyone
495 ent = G_TempEntity( self->r.currentOrigin, EV_OBITUARY );
496 ent->s.eventParm = meansOfDeath;
497 ent->s.otherEntityNum = self->s.number;
498 ent->s.otherEntityNum2 = killer;
499 ent->r.svFlags = SVF_BROADCAST; // send to everyone
500
501 self->enemy = attacker;
502
503 self->client->ps.persistant[PERS_KILLED]++;
504
505 if (attacker && attacker->client) {
506 attacker->client->lastkilled_client = self->s.number;
507
508 if ( attacker == self || OnSameTeam (self, attacker ) ) {
509 AddScore( attacker, self->r.currentOrigin, -1 );
510 } else {
511 AddScore( attacker, self->r.currentOrigin, 1 );
512
513 if( meansOfDeath == MOD_GAUNTLET ) {
514
515 // play humiliation on player
516 attacker->client->ps.persistant[PERS_GAUNTLET_FRAG_COUNT]++;
517
518 // add the sprite over the player's head
519 attacker->client->ps.eFlags &= ~(EF_AWARD_IMPRESSIVE | EF_AWARD_EXCELLENT | EF_AWARD_GAUNTLET | EF_AWARD_ASSIST | EF_AWARD_DEFEND | EF_AWARD_CAP );
520 attacker->client->ps.eFlags |= EF_AWARD_GAUNTLET;
521 attacker->client->rewardTime = level.time + REWARD_SPRITE_TIME;
522
523 // also play humiliation on target
524 self->client->ps.persistant[PERS_PLAYEREVENTS] ^= PLAYEREVENT_GAUNTLETREWARD;
525 }
526
527 // check for two kills in a short amount of time
528 // if this is close enough to the last kill, give a reward sound
529 if ( level.time - attacker->client->lastKillTime < CARNAGE_REWARD_TIME ) {
530 // play excellent on player
531 attacker->client->ps.persistant[PERS_EXCELLENT_COUNT]++;
532
533 // add the sprite over the player's head
534 attacker->client->ps.eFlags &= ~(EF_AWARD_IMPRESSIVE | EF_AWARD_EXCELLENT | EF_AWARD_GAUNTLET | EF_AWARD_ASSIST | EF_AWARD_DEFEND | EF_AWARD_CAP );
535 attacker->client->ps.eFlags |= EF_AWARD_EXCELLENT;
536 attacker->client->rewardTime = level.time + REWARD_SPRITE_TIME;
537 }
538 attacker->client->lastKillTime = level.time;
539
540 }
541 } else {
542 AddScore( self, self->r.currentOrigin, -1 );
543 }
544
545 // Add team bonuses
546 Team_FragBonuses(self, inflictor, attacker);
547
548 // if I committed suicide, the flag does not fall, it returns.
549 if (meansOfDeath == MOD_SUICIDE) {
550 if ( self->client->ps.powerups[PW_NEUTRALFLAG] ) { // only happens in One Flag CTF
551 Team_ReturnFlag( TEAM_FREE );
552 self->client->ps.powerups[PW_NEUTRALFLAG] = 0;
553 }
554 else if ( self->client->ps.powerups[PW_REDFLAG] ) { // only happens in standard CTF
555 Team_ReturnFlag( TEAM_RED );
556 self->client->ps.powerups[PW_REDFLAG] = 0;
557 }
558 else if ( self->client->ps.powerups[PW_BLUEFLAG] ) { // only happens in standard CTF
559 Team_ReturnFlag( TEAM_BLUE );
560 self->client->ps.powerups[PW_BLUEFLAG] = 0;
561 }
562 }
563
564 TossClientItems( self );
565#ifdef MISSIONPACK
566 TossClientPersistantPowerups( self );
567 if( g_gametype.integer == GT_HARVESTER ) {
568 TossClientCubes( self );
569 }
570#endif
571
572 Cmd_Score_f( self ); // show scores
573 // send updated scores to any clients that are following this one,
574 // or they would get stale scoreboards
575 for ( i = 0 ; i < level.maxclients ; i++ ) {
576 gclient_t *client;
577
578 client = &level.clients[i];
579 if ( client->pers.connected != CON_CONNECTED ) {
580 continue;
581 }
582 if ( client->sess.sessionTeam != TEAM_SPECTATOR ) {
583 continue;
584 }
585 if ( client->sess.spectatorClient == self->s.number ) {
586 Cmd_Score_f( g_entities + i );
587 }
588 }
589
590 self->takedamage = qtrue; // can still be gibbed
591
592 self->s.weapon = WP_NONE;
593 self->s.powerups = 0;
594 self->r.contents = CONTENTS_CORPSE;
595
596 self->s.angles[0] = 0;
597 self->s.angles[2] = 0;
598 LookAtKiller (self, inflictor, attacker);
599
600 VectorCopy( self->s.angles, self->client->ps.viewangles );
601
602 self->s.loopSound = 0;
603
604 self->r.maxs[2] = -8;
605
606 // don't allow respawn until the death anim is done
607 // g_forcerespawn may force spawning at some later time
608 self->client->respawnTime = level.time + 1700;
609
610 // remove powerups
611 memset( self->client->ps.powerups, 0, sizeof(self->client->ps.powerups) );
612
613 // never gib in a nodrop
614 contents = trap_PointContents( self->r.currentOrigin, -1 );
615
616 if ( (self->health <= GIB_HEALTH && !(contents & CONTENTS_NODROP) && g_blood.integer) || meansOfDeath == MOD_SUICIDE) {
617 // gib death
618 GibEntity( self, killer );
619 } else {
620 // normal death
621 static int i;
622
623 switch ( i ) {
624 case 0:
625 anim = BOTH_DEATH1;
626 break;
627 case 1:
628 anim = BOTH_DEATH2;
629 break;
630 case 2:
631 default:
632 anim = BOTH_DEATH3;
633 break;
634 }
635
636 // for the no-blood option, we need to prevent the health
637 // from going to gib level
638 if ( self->health <= GIB_HEALTH ) {
639 self->health = GIB_HEALTH+1;
640 }
641
642 self->client->ps.legsAnim =
643 ( ( self->client->ps.legsAnim & ANIM_TOGGLEBIT ) ^ ANIM_TOGGLEBIT ) | anim;
644 self->client->ps.torsoAnim =
645 ( ( self->client->ps.torsoAnim & ANIM_TOGGLEBIT ) ^ ANIM_TOGGLEBIT ) | anim;
646
647 G_AddEvent( self, EV_DEATH1 + i, killer );
648
649 // the body can still be gibbed
650 self->die = body_die;
651
652 // globally cycle through the different death animations
653 i = ( i + 1 ) % 3;
654
655#ifdef MISSIONPACK
656 if (self->s.eFlags & EF_KAMIKAZE) {
657 Kamikaze_DeathTimer( self );
658 }
659#endif
660 }
661
662 trap_LinkEntity (self);
663
664}

667/*
668================
669CheckArmor
670================
671*/
672int CheckArmor (gentity_t *ent, int damage, int dflags)
673{
674 gclient_t *client;
675 int save;
676 int count;
677
678 if (!damage)
679 return 0;
680
681 client = ent->client;
682
683 if (!client)
684 return 0;
685
686 if (dflags & DAMAGE_NO_ARMOR)
687 return 0;
688
689 // armor
690 count = client->ps.stats[STAT_ARMOR];
691 save = ceil( damage * ARMOR_PROTECTION );
692 if (save >= count)
693 save = count;
694
695 if (!save)
696 return 0;
697
698 client->ps.stats[STAT_ARMOR] -= save;
699
700 return save;
701}

703/*
704================
705RaySphereIntersections
706================
707*/
708int RaySphereIntersections( vec3_t origin, float radius, vec3_t point, vec3_t dir, vec3_t intersections[2] ) {
709 float b, c, d, t;
710
711 // | origin - (point + t * dir) | = radius
712 // a = dir[0]^2 + dir[1]^2 + dir[2]^2;
713 // b = 2 * (dir[0] * (point[0] - origin[0]) + dir[1] * (point[1] - origin[1]) + dir[2] * (point[2] - origin[2]));
714 // c = (point[0] - origin[0])^2 + (point[1] - origin[1])^2 + (point[2] - origin[2])^2 - radius^2;
715
716 // normalize dir so a = 1
717 VectorNormalize(dir);
718 b = 2 * (dir[0] * (point[0] - origin[0]) + dir[1] * (point[1] - origin[1]) + dir[2] * (point[2] - origin[2]));
719 c = (point[0] - origin[0]) * (point[0] - origin[0]) +
720 (point[1] - origin[1]) * (point[1] - origin[1]) +
721 (point[2] - origin[2]) * (point[2] - origin[2]) -
722 radius * radius;
723
724 d = b * b - 4 * c;
725 if (d > 0) {
726 t = (- b + sqrt(d)) / 2;
727 VectorMA(point, t, dir, intersections[0]);
728 t = (- b - sqrt(d)) / 2;
729 VectorMA(point, t, dir, intersections[1]);
730 return 2;
731 }
732 else if (d == 0) {
733 t = (- b ) / 2;
734 VectorMA(point, t, dir, intersections[0]);
735 return 1;
736 }
737 return 0;
738}

740#ifdef MISSIONPACK
741/*
742================
743G_InvulnerabilityEffect
744================
745*/
746int G_InvulnerabilityEffect( gentity_t *targ, vec3_t dir, vec3_t point, vec3_t impactpoint, vec3_t bouncedir ) {
747 gentity_t *impact;
748 vec3_t intersections[2], vec;
749 int n;
750
751 if ( !targ->client ) {
752 return qfalse;
753 }
754 VectorCopy(dir, vec);
755 VectorInverse(vec);
756 // sphere model radius = 42 units
757 n = RaySphereIntersections( targ->client->ps.origin, 42, point, vec, intersections);
758 if (n > 0) {
759 impact = G_TempEntity( targ->client->ps.origin, EV_INVUL_IMPACT );
760 VectorSubtract(intersections[0], targ->client->ps.origin, vec);
761 vectoangles(vec, impact->s.angles);
762 impact->s.angles[0] += 90;
763 if (impact->s.angles[0] > 360)
764 impact->s.angles[0] -= 360;
765 if ( impactpoint ) {
766 VectorCopy( intersections[0], impactpoint );
767 }
768 if ( bouncedir ) {
769 VectorCopy( vec, bouncedir );
770 VectorNormalize( bouncedir );
771 }
772 return qtrue;
773 }
774 else {
775 return qfalse;
776 }

779/*
780============
781G_Damage
782
783targ entity that is being damaged
784inflictor entity that is causing the damage
785attacker entity that caused the inflictor to damage targ
786 example: targ=monster, inflictor=rocket, attacker=player
787
788dir direction of the attack for knockback
789point point at which the damage is being inflicted, used for headshots
790damage amount of damage being inflicted
791knockback force to be applied against targ as a result of the damage
792
793inflictor, attacker, dir, and point can be NULL for environmental effects
794
795dflags these flags are used to control how T_Damage works
796 DAMAGE_RADIUS damage was indirect (from a nearby explosion)
797 DAMAGE_NO_ARMOR armor does not protect from this damage
798 DAMAGE_NO_KNOCKBACK do not affect velocity, just view angles
799 DAMAGE_NO_PROTECTION kills godmode, armor, everything
800============
801*/
802
803void G_Damage( gentity_t *targ, gentity_t *inflictor, gentity_t *attacker,
804 vec3_t dir, vec3_t point, int damage, int dflags, int mod ) {
805 gclient_t *client;
806 int take;
807 int asave;
808 int knockback;
809 int max;
810#ifdef MISSIONPACK
811 vec3_t bouncedir, impactpoint;
812#endif
813
814 if (!targ->takedamage) {
815 return;
816 }
817
818 // the intermission has already been qualified for, so don't
819 // allow any extra scoring
820 if ( level.intermissionQueued ) {
821 return;
822 }
823#ifdef MISSIONPACK
824 if ( targ->client && mod != MOD_JUICED) {
825 if ( targ->client->invulnerabilityTime > level.time) {
826 if ( dir && point ) {
827 G_InvulnerabilityEffect( targ, dir, point, impactpoint, bouncedir );
828 }
829 return;
830 }
831 }
832#endif
833 if ( !inflictor ) {
834 inflictor = &g_entities[ENTITYNUM_WORLD];
835 }
836 if ( !attacker ) {
837 attacker = &g_entities[ENTITYNUM_WORLD];
838 }
839
840 // shootable doors / buttons don't actually have any health
841 if ( targ->s.eType == ET_MOVER ) {
842 if ( targ->use && targ->moverState == MOVER_POS1 ) {
843 targ->use( targ, inflictor, attacker );
844 }
845 return;
846 }
847#ifdef MISSIONPACK
848 if( g_gametype.integer == GT_OBELISK && CheckObeliskAttack( targ, attacker ) ) {
849 return;
850 }
851#endif
852 // reduce damage by the attacker's handicap value
853 // unless they are rocket jumping
854 if ( attacker->client && attacker != targ ) {
855 max = attacker->client->ps.stats[STAT_MAX_HEALTH];
856#ifdef MISSIONPACK
857 if( bg_itemlist[attacker->client->ps.stats[STAT_PERSISTANT_POWERUP]].giTag == PW_GUARD ) {
858 max /= 2;
859 }
860#endif
861 damage = damage * max / 100;
862 }
863
864 client = targ->client;
865
866 if ( client ) {
867 if ( client->noclip ) {
868 return;
869 }
870 }
871
872 if ( !dir ) {
873 dflags |= DAMAGE_NO_KNOCKBACK;
874 } else {
875 VectorNormalize(dir);
876 }
877
878 knockback = damage;
879 if ( knockback > 200 ) {
880 knockback = 200;
881 }
882 if ( targ->flags & FL_NO_KNOCKBACK ) {
883 knockback = 0;
884 }
885 if ( dflags & DAMAGE_NO_KNOCKBACK ) {
886 knockback = 0;
887 }
888
889 // figure momentum add, even if the damage won't be taken
890 if ( knockback && targ->client ) {
891 vec3_t kvel;
892 float mass;
893
894 mass = 200;
895
896 VectorScale (dir, g_knockback.value * (float)knockback / mass, kvel);
897 VectorAdd (targ->client->ps.velocity, kvel, targ->client->ps.velocity);
898
899 // set the timer so that the other client can't cancel
900 // out the movement immediately
901 if ( !targ->client->ps.pm_time ) {
902 int t;
903
904 t = knockback * 2;
905 if ( t < 50 ) {
906 t = 50;
907 }
908 if ( t > 200 ) {
909 t = 200;
910 }
911 targ->client->ps.pm_time = t;
912 targ->client->ps.pm_flags |= PMF_TIME_KNOCKBACK;
913 }
914 }
915
916 // check for completely getting out of the damage
917 if ( !(dflags & DAMAGE_NO_PROTECTION) ) {
918
919 // if TF_NO_FRIENDLY_FIRE is set, don't do damage to the target
920 // if the attacker was on the same team
921#ifdef MISSIONPACK
922 if ( mod != MOD_JUICED && targ != attacker && !(dflags & DAMAGE_NO_TEAM_PROTECTION) && OnSameTeam (targ, attacker) ) {
923#else
924 if ( targ != attacker && OnSameTeam (targ, attacker) ) {
925#endif
926 if ( !g_friendlyFire.integer ) {
927 return;
928 }
929 }
930#ifdef MISSIONPACK
931 if (mod == MOD_PROXIMITY_MINE) {
932 if (inflictor && inflictor->parent && OnSameTeam(targ, inflictor->parent)) {
933 return;
934 }
935 if (targ == attacker) {
936 return;
937 }
938 }
939#endif
940
941 // check for godmode
942 if ( targ->flags & FL_GODMODE ) {
943 return;
944 }
945 }
946
947 // battlesuit protects from all radius damage (but takes knockback)
948 // and protects 50% against all damage
949 if ( client && client->ps.powerups[PW_BATTLESUIT] ) {
950 G_AddEvent( targ, EV_POWERUP_BATTLESUIT, 0 );
951 if ( ( dflags & DAMAGE_RADIUS ) || ( mod == MOD_FALLING ) ) {
952 return;
953 }
954 damage *= 0.5;
955 }
956
957 // add to the attacker's hit counter (if the target isn't a general entity like a prox mine)
958 if ( attacker->client && client
959 && targ != attacker && targ->health > 0
960 && targ->s.eType != ET_MISSILE
961 && targ->s.eType != ET_GENERAL) {
962 if ( OnSameTeam( targ, attacker ) ) {
963 attacker->client->ps.persistant[PERS_HITS]--;
964 } else {
965 attacker->client->ps.persistant[PERS_HITS]++;
966 }
967 attacker->client->ps.persistant[PERS_ATTACKEE_ARMOR] = (targ->health<<8)|(client->ps.stats[STAT_ARMOR]);
968 }
969
970 // always give half damage if hurting self
971 // calculated after knockback, so rocket jumping works
972 if ( targ == attacker) {
973 damage *= 0.5;
974 }
975
976 if ( damage < 1 ) {
977 damage = 1;
978 }
979 take = damage;
980
981 // save some from armor
982 asave = CheckArmor (targ, take, dflags);
983 take -= asave;
984
985 if ( g_debugDamage.integer ) {
986 G_Printf( "%i: client:%i health:%i damage:%i armor:%i\n", level.time, targ->s.number,
987 targ->health, take, asave );
988 }
989
990 // add to the damage inflicted on a player this frame
991 // the total will be turned into screen blends and view angle kicks
992 // at the end of the frame
993 if ( client ) {
994 if ( attacker ) {
995 client->ps.persistant[PERS_ATTACKER] = attacker->s.number;
996 } else {
997 client->ps.persistant[PERS_ATTACKER] = ENTITYNUM_WORLD;
998 }
999 client->damage_armor += asave;
1000 client->damage_blood += take;
1001 client->damage_knockback += knockback;
1002 if ( dir ) {
1003 VectorCopy ( dir, client->damage_from );
1004 client->damage_fromWorld = qfalse;
1005 } else {
1006 VectorCopy ( targ->r.currentOrigin, client->damage_from );
1007 client->damage_fromWorld = qtrue;
1008 }
1009 }
1010
1011 // See if it's the player hurting the emeny flag carrier
1012#ifdef MISSIONPACK
1013 if( g_gametype.integer == GT_CTF || g_gametype.integer == GT_1FCTF ) {
1014#else
1015 if( g_gametype.integer == GT_CTF) {
1016#endif
1017 Team_CheckHurtCarrier(targ, attacker);
1018 }
1019
1020 if (targ->client) {
1021 // set the last client who damaged the target
1022 targ->client->lasthurt_client = attacker->s.number;
1023 targ->client->lasthurt_mod = mod;
1024 }
1025
1026 // do the damage
1027 if (take) {
1028 targ->health = targ->health - take;
1029 if ( targ->client ) {
1030 targ->client->ps.stats[STAT_HEALTH] = targ->health;
1031 }
1032
1033 if ( targ->health <= 0 ) {
1034 if ( client )
1035 targ->flags |= FL_NO_KNOCKBACK;
1036
1037 if (targ->health < -999)
1038 targ->health = -999;
1039
1040 targ->enemy = attacker;
1041 targ->die (targ, inflictor, attacker, take, mod);
1042 return;
1043 } else if ( targ->pain ) {
1044 targ->pain (targ, attacker, take);
1045 }
1046 }
1047
1048}

1053CanDamage
1054
1055Returns qtrue if the inflictor can directly damage the target. Used for
1056explosions and melee attacks.

1059qboolean CanDamage (gentity_t *targ, vec3_t origin) {
1060 vec3_t dest;
1061 trace_t tr;
1062 vec3_t midpoint;
1063 vec3_t offsetmins = {-15, -15, -15};
1064 vec3_t offsetmaxs = {15, 15, 15};
1065
1066 // use the midpoint of the bounds instead of the origin, because
1067 // bmodels may have their origin is 0,0,0
1068 VectorAdd (targ->r.absmin, targ->r.absmax, midpoint);
1069 VectorScale (midpoint, 0.5, midpoint);
1070
1071 VectorCopy(midpoint, dest);
1072 trap_Trace(&tr, origin, vec3_origin, vec3_origin, dest, ENTITYNUM_NONE, MASK_SOLID);
1073
1074 if (tr.fraction == 1.0 || tr.entityNum == targ->s.number)
1075 return qtrue;
1076
1077 // this should probably check in the plane of projection,
1078 // rather than in world coordinate
1079 VectorCopy(midpoint, dest);
1080 dest[0] += offsetmaxs[0];
1081 dest[1] += offsetmaxs[1];
1082 dest[2] += offsetmaxs[2];
1083 trap_Trace(&tr, origin, vec3_origin, vec3_origin, dest, ENTITYNUM_NONE, MASK_SOLID);
1084
1085 if (tr.fraction == 1.0)
1086 return qtrue;
1087
1088 VectorCopy(midpoint, dest);
1089 dest[0] += offsetmaxs[0];
1090 dest[1] += offsetmins[1];
1091 dest[2] += offsetmaxs[2];
1092 trap_Trace(&tr, origin, vec3_origin, vec3_origin, dest, ENTITYNUM_NONE, MASK_SOLID);
1093
1094 if (tr.fraction == 1.0)
1095 return qtrue;
1096
1097 VectorCopy(midpoint, dest);
1098 dest[0] += offsetmins[0];
1099 dest[1] += offsetmaxs[1];
1100 dest[2] += offsetmaxs[2];
1101 trap_Trace(&tr, origin, vec3_origin, vec3_origin, dest, ENTITYNUM_NONE, MASK_SOLID);
1102
1103 if (tr.fraction == 1.0)
1104 return qtrue;
1105
1106 VectorCopy(midpoint, dest);
1107 dest[0] += offsetmins[0];
1108 dest[1] += offsetmins[1];
1109 dest[2] += offsetmaxs[2];
1110 trap_Trace(&tr, origin, vec3_origin, vec3_origin, dest, ENTITYNUM_NONE, MASK_SOLID);
1111
1112 if (tr.fraction == 1.0)
1113 return qtrue;
1114
1115 VectorCopy(midpoint, dest);
1116 dest[0] += offsetmaxs[0];
1117 dest[1] += offsetmaxs[1];
1118 dest[2] += offsetmins[2];
1119 trap_Trace(&tr, origin, vec3_origin, vec3_origin, dest, ENTITYNUM_NONE, MASK_SOLID);
1120
1121 if (tr.fraction == 1.0)
1122 return qtrue;
1123
1124 VectorCopy(midpoint, dest);
1125 dest[0] += offsetmaxs[0];
1126 dest[1] += offsetmins[1];
1127 dest[2] += offsetmins[2];
1128 trap_Trace(&tr, origin, vec3_origin, vec3_origin, dest, ENTITYNUM_NONE, MASK_SOLID);
1129
1130 if (tr.fraction == 1.0)
1131 return qtrue;
1132
1133 VectorCopy(midpoint, dest);
1134 dest[0] += offsetmins[0];
1135 dest[1] += offsetmaxs[1];
1136 dest[2] += offsetmins[2];
1137 trap_Trace(&tr, origin, vec3_origin, vec3_origin, dest, ENTITYNUM_NONE, MASK_SOLID);
1138
1139 if (tr.fraction == 1.0)
1140 return qtrue;
1141
1142 VectorCopy(midpoint, dest);
1143 dest[0] += offsetmins[0];
1144 dest[1] += offsetmins[2];
1145 dest[2] += offsetmins[2];
1146 trap_Trace(&tr, origin, vec3_origin, vec3_origin, dest, ENTITYNUM_NONE, MASK_SOLID);
1147
1148 if (tr.fraction == 1.0)
1149 return qtrue;
1150
1151 return qfalse;

1155/*
1156============
1157G_RadiusDamage
1158============
1159*/
1160qboolean G_RadiusDamage ( vec3_t origin, gentity_t *attacker, float damage, float radius,
1161 gentity_t *ignore, int mod) {
1162 float points, dist;
1163 gentity_t *ent;
1164 int entityList[MAX_GENTITIES];
1165 int numListedEntities;
1166 vec3_t mins, maxs;
1167 vec3_t v;
1168 vec3_t dir;
1169 int i, e;
1170 qboolean hitClient = qfalse;
1171
1172 if ( radius < 1 ) {
1173 radius = 1;
1174 }
1175
1176 for ( i = 0 ; i < 3 ; i++ ) {
1177 mins[i] = origin[i] - radius;
1178 maxs[i] = origin[i] + radius;
1179 }
1180
1181 numListedEntities = trap_EntitiesInBox( mins, maxs, entityList, MAX_GENTITIES );
1182
1183 for ( e = 0 ; e < numListedEntities ; e++ ) {
1184 ent = &g_entities[entityList[ e ]];
1185
1186 if (ent == ignore)
1187 continue;
1188 if (!ent->takedamage)
1189 continue;
1190
1191 // find the distance from the edge of the bounding box
1192 for ( i = 0 ; i < 3 ; i++ ) {
1193 if ( origin[i] < ent->r.absmin[i] ) {
1194 v[i] = ent->r.absmin[i] - origin[i];
1195 } else if ( origin[i] > ent->r.absmax[i] ) {
1196 v[i] = origin[i] - ent->r.absmax[i];
1197 } else {
1198 v[i] = 0;
1199 }
1200 }
1201
1202 dist = VectorLength( v );
1203 if ( dist >= radius ) {
1204 continue;
1205 }
1206
1207 points = damage * ( 1.0 - dist / radius );
1208
1209 if( CanDamage (ent, origin) ) {
1210 if( LogAccuracyHit( ent, attacker ) ) {
1211 hitClient = qtrue;
1212 }
1213 VectorSubtract (ent->r.currentOrigin, origin, dir);
1214 // push the center of mass higher than the origin so players
1215 // get knocked into the air more
1216 dir[2] += 24;
1217 G_Damage (ent, NULL, attacker, dir, origin, (int)points, DAMAGE_RADIUS, mod);
1218 }
1219 }
1220
1221 return hitClient;
1222}

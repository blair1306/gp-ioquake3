51/*
52 * Qsort routine from Bentley & McIlroy's "Engineering a Sort Function".
53 */
54#define swapcode(TYPE, parmi, parmj, n) { \
55 long i = (n) / sizeof (TYPE); \
56 TYPE *pi = (TYPE *) (parmi); \
57 TYPE *pj = (TYPE *) (parmj); \
58 do { \
59 TYPE t = *pi; \
60 *pi++ = *pj; \
61 *pj++ = t; \
62 } while (--i > 0); \
63}

65#define SWAPINIT(a, es) swaptype = ((char *)a - (char *)0) % sizeof(long) || \
66 es % sizeof(long) ? 2 : es == sizeof(long)? 0 : 1;

68static void
69swapfunc(a, b, n, swaptype)
70 char *a, *b;
71 int n, swaptype;
72{
73 if(swaptype <= 1)
74 swapcode(long, a, b, n)
75 else
76 swapcode(char, a, b, n)
77}

79#define swap(a, b) \
80 if (swaptype == 0) { \
81 long t = *(long *)(a); \
82 *(long *)(a) = *(long *)(b); \
83 *(long *)(b) = t; \
84 } else \
85 swapfunc(a, b, es, swaptype)

87#define vecswap(a, b, n) if ((n) > 0) swapfunc(a, b, n, swaptype)

89static char *
90med3(a, b, c, cmp)
91 char *a, *b, *c;
92 cmp_t *cmp;
93{
94 return cmp(a, b) < 0 ?
95 (cmp(b, c) < 0 ? b : (cmp(a, c) < 0 ? c : a ))
96 :(cmp(b, c) > 0 ? b : (cmp(a, c) < 0 ? a : c ));
97}

99void
100qsort(a, n, es, cmp)
101 void *a;
102 size_t n, es;
103 cmp_t *cmp;
104{
105 char *pa, *pb, *pc, *pd, *pl, *pm, *pn;
106 int d, r, swaptype, swap_cnt;
107
108loop: SWAPINIT(a, es);
109 swap_cnt = 0;
110 if (n < 7) {
111 for (pm = (char *)a + es; pm < (char *)a + n * es; pm += es)
112 for (pl = pm; pl > (char *)a && cmp(pl - es, pl) > 0;
113 pl -= es)
114 swap(pl, pl - es);
115 return;
116 }
117 pm = (char *)a + (n / 2) * es;
118 if (n > 7) {
119 pl = a;
120 pn = (char *)a + (n - 1) * es;
121 if (n > 40) {
122 d = (n / 8) * es;
123 pl = med3(pl, pl + d, pl + 2 * d, cmp);
124 pm = med3(pm - d, pm, pm + d, cmp);
125 pn = med3(pn - 2 * d, pn - d, pn, cmp);
126 }
127 pm = med3(pl, pm, pn, cmp);
128 }
129 swap(a, pm);
130 pa = pb = (char *)a + es;
131
132 pc = pd = (char *)a + (n - 1) * es;
133 for (;;) {
134 while (pb <= pc && (r = cmp(pb, a)) <= 0) {
135 if (r == 0) {
136 swap_cnt = 1;
137 swap(pa, pb);
138 pa += es;
139 }
140 pb += es;
141 }
142 while (pb <= pc && (r = cmp(pc, a)) >= 0) {
143 if (r == 0) {
144 swap_cnt = 1;
145 swap(pc, pd);
146 pd -= es;
147 }
148 pc -= es;
149 }
150 if (pb > pc)
151 break;
152 swap(pb, pc);
153 swap_cnt = 1;
154 pb += es;
155 pc -= es;
156 }
157 if (swap_cnt == 0) { /* Switch to insertion sort */
158 for (pm = (char *)a + es; pm < (char *)a + n * es; pm += es)
159 for (pl = pm; pl > (char *)a && cmp(pl - es, pl) > 0;
160 pl -= es)
161 swap(pl, pl - es);
162 return;
163 }
164
165 pn = (char *)a + n * es;
166 r = MIN(pa - (char *)a, pb - pa);
167 vecswap(a, pb - r, r);
168 r = MIN(pd - pc, pn - pd - es);
169 vecswap(pb, pn - r, r);
170 if ((r = pb - pa) > es)
171 qsort(a, r / es, es, cmp);
172 if ((r = pd - pc) > es) {
173 /* Iterate rather than recurse to save stack space */
174 a = pn - r;
175 n = r / es;
176 goto loop;
177 }
178/* qsort(pn - r, r / es, es, cmp);*/
179}

354double sqrt( double x ) {
355 float y;
356 float delta;
357 float maxError;
358
359 if ( x <= 0 ) {
360 return 0;
361 }
362
363 // initial guess
364 y = x / 2;
365
366 // refine
367 maxError = x * 0.001;
368
369 do {
370 delta = ( y * y ) - x;
371 y -= delta / ( 2 * y );
372 } while ( delta > maxError || delta < -maxError );
373
374 return y;
375}

773powN
774
775Raise a double to a integer power
776===============

778static double powN( double base, int exp )
779{
780 if( exp >= 0 )
781 {
782 double result = 1.0;
783
784 // calculate x, x^2, x^4, ... by repeated squaring
785 // and multiply together the ones corresponding to the
786 // binary digits of the exponent
787 // e.g. x^73 = x^(1 + 8 + 64) = x * x^8 * x^64
788 while( exp > 0 )
789 {
790 if( exp % 2 == 1 )
791 result *= base;
792
793 base *= base;
794 exp /= 2;
795 }
796
797 return result;
798 }
799 // if exp is INT_MIN, the next clause will be upset,
800 // because -exp isn't representable
801 else if( exp == INT_MIN )
802 return powN( base, exp + 1 ) / base;
803 // x < 0
804 else
805 return 1.0 / powN( base, -exp );
806}

959/*
960==============
961strtod
962
963Without an errno variable, this is a fair bit less useful than it is in libc
964but it's still a fair bit more capable than atof or _atof
965Handles inf[inity], nan (ignoring case), hexadecimals, and decimals
966Handles decimal exponents like 10e10 and hex exponents like 0x7f8p20
96710e10 == 10000000000 (power of ten)
9680x7f8p20 == 0x7f800000 (decimal power of two)
969The variable pointed to by endptr will hold the location of the first character
970in the nptr string that was not used in the conversion
971==============
972*/
973double strtod( const char *nptr, char **endptr )
974{
975 double res;
976 qboolean neg = qfalse;
977
978 // skip whitespace
979 while( isspace( *nptr ) )
980 nptr++;
981
982 // special string parsing
983 if( Q_stricmpn( nptr, "nan", 3 ) == 0 )
984 {
985 floatint_t nan;
986
987 if( endptr )
988 *endptr = (char *)&nptr[3];
989
990 // nan can be followed by a bracketed number (in hex, octal,
991 // or decimal) which is then put in the mantissa
992 // this can be used to generate signalling or quiet NaNs, for
993 // example (though I doubt it'll ever be used)
994 // note that nan(0) is infinity!
995 if( nptr[3] == '(' )
996 {
997 char *end;
998 int mantissa = strtol( &nptr[4], &end, 0 );
999 if( *end == ')' )
1000 {
1001 nan.ui = 0x7f800000 | ( mantissa & 0x7fffff );
1002 if( endptr )
1003 *endptr = &end[1];
1004 return nan.f;
1005 }
1006 }
1007 nan.ui = 0x7fffffff;
1008 return nan.f;
1009 }
1010 if( Q_stricmpn( nptr, "inf", 3 ) == 0 )
1011 {
1012 floatint_t inf;
1013 inf.ui = 0x7f800000;
1014 if( endptr == NULL )
1015 return inf.f;
1016 if( Q_stricmpn( &nptr[3], "inity", 5 ) == 0 )
1017 *endptr = (char *)&nptr[8];
1018 else
1019 *endptr = (char *)&nptr[3];
1020 return inf.f;
1021 }
1022
1023 // normal numeric parsing
1024 // sign
1025 if( *nptr == '-' )
1026 {
1027 nptr++;
1028 neg = qtrue;
1029 }
1030 else if( *nptr == '+' )
1031 nptr++;
1032 // hex
1033 if( Q_stricmpn( nptr, "0x", 2 ) == 0 )
1034 {
1035 // track if we use any digits
1036 const char *s = &nptr[1], *end = s;
1037 nptr += 2;
1038 res = 0;
1039 while( qtrue )
1040 {
1041 if( isdigit( *nptr ) )
1042 res = 16 * res + ( *nptr++ - '0' );
1043 else if( *nptr >= 'A' && *nptr <= 'F' )
1044 res = 16 * res + 10 + *nptr++ - 'A';
1045 else if( *nptr >= 'a' && *nptr <= 'f' )
1046 res = 16 * res + 10 + *nptr++ - 'a';
1047 else
1048 break;
1049 }
1050 // if nptr moved, save it
1051 if( end + 1 < nptr )
1052 end = nptr;
1053 if( *nptr == '.' )
1054 {
1055 float place;
1056 nptr++;
1057 // 1.0 / 16.0 == 0.0625
1058 // I don't expect the float accuracy to hold out for
1059 // very long but since we need to know the length of
1060 // the string anyway we keep on going regardless
1061 for( place = 0.0625;; place /= 16.0 )
1062 {
1063 if( isdigit( *nptr ) )
1064 res += place * ( *nptr++ - '0' );
1065 else if( *nptr >= 'A' && *nptr <= 'F' )
1066 res += place * ( 10 + *nptr++ - 'A' );
1067 else if( *nptr >= 'a' && *nptr <= 'f' )
1068 res += place * ( 10 + *nptr++ - 'a' );
1069 else
1070 break;
1071 }
1072 if( end < nptr )
1073 end = nptr;
1074 }
1075 // parse an optional exponent, representing multiplication
1076 // by a power of two
1077 // exponents are only valid if we encountered at least one
1078 // digit already (and have therefore set end to something)
1079 if( end != s && tolower( *nptr ) == 'p' )
1080 {
1081 int exp;
1082 // apparently (confusingly) the exponent should be
1083 // decimal
1084 exp = strtol( &nptr[1], (char **)&end, 10 );
1085 if( &nptr[1] == end )
1086 {
1087 // no exponent
1088 if( endptr )
1089 *endptr = (char *)nptr;
1090 return res;
1091 }
1092
1093 res *= powN( 2, exp );
1094 }
1095 if( endptr )
1096 *endptr = (char *)end;
1097 return res;
1098 }
1099 // decimal
1100 else
1101 {
1102 // track if we find any digits
1103 const char *end = nptr, *p = nptr;
1104 // this is most of the work
1105 for( res = 0; isdigit( *nptr );
1106 res = 10 * res + *nptr++ - '0' );
1107 // if nptr moved, we read something
1108 if( end < nptr )
1109 end = nptr;
1110 if( *nptr == '.' )
1111 {
1112 // fractional part
1113 float place;
1114 nptr++;
1115 for( place = 0.1; isdigit( *nptr ); place /= 10.0 )
1116 res += ( *nptr++ - '0' ) * place;
1117 // if nptr moved, we read something
1118 if( end + 1 < nptr )
1119 end = nptr;
1120 }
1121 // exponent
1122 // meaningless without having already read digits, so check
1123 // we've set end to something
1124 if( p != end && tolower( *nptr ) == 'e' )
1125 {
1126 int exp;
1127 exp = strtol( &nptr[1], (char **)&end, 10 );
1128 if( &nptr[1] == end )
1129 {
1130 // no exponent
1131 if( endptr )
1132 *endptr = (char *)nptr;
1133 return res;
1134 }
1135
1136 res *= powN( 10, exp );
1137 }
1138 if( endptr )
1139 *endptr = (char *)end;
1140 return res;
1141 }
1142}

1239/*
1240==============
1241strtol
1242
1243Handles any base from 2 to 36. If base is 0 then it guesses
1244decimal, hex, or octal based on the format of the number (leading 0 or 0x)
1245Will not overflow - returns LONG_MIN or LONG_MAX as appropriate
1246*endptr is set to the location of the first character not used
1247==============
1248*/
1249long strtol( const char *nptr, char **endptr, int base )
1250{
1251 long res;
1252 qboolean pos = qtrue;
1253
1254 if( endptr )
1255 *endptr = (char *)nptr;
1256 // bases other than 0, 2, 8, 16 are very rarely used, but they're
1257 // not much extra effort to support
1258 if( base < 0 || base == 1 || base > 36 )
1259 return 0;
1260 // skip leading whitespace
1261 while( isspace( *nptr ) )
1262 nptr++;
1263 // sign
1264 if( *nptr == '-' )
1265 {
1266 nptr++;
1267 pos = qfalse;
1268 }
1269 else if( *nptr == '+' )
1270 nptr++;
1271 // look for base-identifying sequences e.g. 0x for hex, 0 for octal
1272 if( nptr[0] == '0' )
1273 {
1274 nptr++;
1275 // 0 is always a valid digit
1276 if( endptr )
1277 *endptr = (char *)nptr;
1278 if( *nptr == 'x' || *nptr == 'X' )
1279 {
1280 if( base != 0 && base != 16 )
1281 {
1282 // can't be hex, reject x (accept 0)
1283 if( endptr )
1284 *endptr = (char *)nptr;
1285 return 0;
1286 }
1287 nptr++;
1288 base = 16;
1289 }
1290 else if( base == 0 )
1291 base = 8;
1292 }
1293 else if( base == 0 )
1294 base = 10;
1295 res = 0;
1296 while( qtrue )
1297 {
1298 int val;
1299 if( isdigit( *nptr ) )
1300 val = *nptr - '0';
1301 else if( islower( *nptr ) )
1302 val = 10 + *nptr - 'a';
1303 else if( isupper( *nptr ) )
1304 val = 10 + *nptr - 'A';
1305 else
1306 break;
1307 if( val >= base )
1308 break;
1309 // we go negative because LONG_MIN is further from 0 than
1310 // LONG_MAX
1311 if( res < ( LONG_MIN + val ) / base )
1312 res = LONG_MIN; // overflow
1313 else
1314 res = res * base - val;
1315 nptr++;
1316 if( endptr )
1317 *endptr = (char *)nptr;
1318 }
1319 if( pos )
1320 {
1321 // can't represent LONG_MIN positive
1322 if( res == LONG_MIN )
1323 res = LONG_MAX;
1324 else
1325 res = -res;
1326 }
1327 return res;
1328}

1471static int dopr (char *buffer, size_t maxlen, const char *format, va_list args)
1472{
1473 char ch;
1474 LLONG value;
1475 LDOUBLE fvalue;
1476 char *strvalue;
1477 int min;
1478 int max;
1479 int state;
1480 int flags;
1481 int cflags;
1482 int total;
1483 size_t currlen;
1484
1485 state = DP_S_DEFAULT;
1486 currlen = flags = cflags = min = 0;
1487 max = -1;
1488 ch = *format++;
1489 total = 0;
1490
1491 while (state != DP_S_DONE)
1492 {
1493 if (ch == '\0')
1494 state = DP_S_DONE;
1495
1496 switch(state)
1497 {
1498 case DP_S_DEFAULT:
1499 if (ch == '%')
1500 state = DP_S_FLAGS;
1501 else
1502 total += dopr_outch (buffer, &currlen, maxlen, ch);
1503 ch = *format++;
1504 break;
1505 case DP_S_FLAGS:
1506 switch (ch)
1507 {
1508 case '-':
1509 flags |= DP_F_MINUS;
1510 ch = *format++;
1511 break;
1512 case '+':
1513 flags |= DP_F_PLUS;
1514 ch = *format++;
1515 break;
1516 case ' ':
1517 flags |= DP_F_SPACE;
1518 ch = *format++;
1519 break;
1520 case '#':
1521 flags |= DP_F_NUM;
1522 ch = *format++;
1523 break;
1524 case '0':
1525 flags |= DP_F_ZERO;
1526 ch = *format++;
1527 break;
1528 default:
1529 state = DP_S_MIN;
1530 break;
1531 }
1532 break;
1533 case DP_S_MIN:

1535 {
1536 min = 10*min + char_to_int (ch);
1537 ch = *format++;
1538 }
1539 else if (ch == '*')
1540 {
1541 min = va_arg (args, int);
1542 ch = *format++;
1543 state = DP_S_DOT;
1544 }
1545 else
1546 state = DP_S_DOT;
1547 break;
1548 case DP_S_DOT:
1549 if (ch == '.')
1550 {
1551 state = DP_S_MAX;
1552 ch = *format++;
1553 }
1554 else
1555 state = DP_S_MOD;
1556 break;
1557 case DP_S_MAX:
1558 if ('0' <= ch && ch <= '9')
1559 {
1560 if (max < 0)
1561 max = 0;
1562 max = 10*max + char_to_int (ch);
1563 ch = *format++;
1564 }
1565 else if (ch == '*')
1566 {
1567 max = va_arg (args, int);
1568 ch = *format++;
1569 state = DP_S_MOD;
1570 }
1571 else
1572 state = DP_S_MOD;
1573 break;
1574 case DP_S_MOD:
1575 switch (ch)
1576 {
1577 case 'h':
1578 cflags = DP_C_SHORT;
1579 ch = *format++;
1580 break;
1581 case 'l':
1582 cflags = DP_C_LONG;
1583 ch = *format++;
1584 break;
1585 case 'L':
1586 cflags = DP_C_LDOUBLE;
1587 ch = *format++;
1588 break;
1589 default:
1590 break;
1591 }
1592 if (cflags != DP_C_LONG)
1593 state = DP_S_CONV;
1594 else
1595 state = DP_S_MOD_L;
1596 break;
1597 case DP_S_MOD_L:
1598 switch (ch)
1599 {
1600 case 'l':
1601 cflags = DP_C_LLONG;
1602 ch = *format++;
1603 break;
1604 default:
1605 break;
1606 }
1607 state = DP_S_CONV;
1608 break;
1609 case DP_S_CONV:
1610 switch (ch)
1611 {
1612 case 'd':
1613 case 'i':
1614 if (cflags == DP_C_SHORT)
1615 value = (short int)va_arg (args, int);
1616 else if (cflags == DP_C_LONG)
1617 value = va_arg (args, long int);
1618 else if (cflags == DP_C_LLONG)
1619 value = va_arg (args, LLONG);
1620 else
1621 value = va_arg (args, int);
1622 total += fmtint (buffer, &currlen, maxlen, value, 10, min, max, flags);
1623 break;
1624 case 'o':

1627// value = (unsigned short int) va_arg (args, unsigned short int); // Thilo: This does not work because the rcc compiler cannot do that cast correctly.

1629 else if (cflags == DP_C_LONG)
1630 value = va_arg (args, unsigned long int);
1631 else if (cflags == DP_C_LLONG)
1632 value = va_arg (args, unsigned LLONG);
1633 else
1634 value = va_arg (args, unsigned int);
1635 total += fmtint (buffer, &currlen, maxlen, value, 8, min, max, flags);
1636 break;
1637 case 'u':
1638 flags |= DP_F_UNSIGNED;
1639 if (cflags == DP_C_SHORT)
1640 value = va_arg (args, unsigned int) & ( (1 << sizeof(unsigned short int) * 8) - 1);
1641 else if (cflags == DP_C_LONG)
1642 value = va_arg (args, unsigned long int);
1643 else if (cflags == DP_C_LLONG)
1644 value = va_arg (args, unsigned LLONG);
1645 else
1646 value = va_arg (args, unsigned int);
1647 total += fmtint (buffer, &currlen, maxlen, value, 10, min, max, flags);
1648 break;
1649 case 'X':
1650 flags |= DP_F_UP;
1651 case 'x':
1652 flags |= DP_F_UNSIGNED;
1653 if (cflags == DP_C_SHORT)
1654 value = va_arg (args, unsigned int) & ( (1 << sizeof(unsigned short int) * 8) - 1);
1655 else if (cflags == DP_C_LONG)
1656 value = va_arg (args, unsigned long int);
1657 else if (cflags == DP_C_LLONG)
1658 value = va_arg (args, unsigned LLONG);
1659 else
1660 value = va_arg (args, unsigned int);
1661 total += fmtint (buffer, &currlen, maxlen, value, 16, min, max, flags);
1662 break;
1663 case 'f':
1664 if (cflags == DP_C_LDOUBLE)
1665 fvalue = va_arg (args, LDOUBLE);
1666 else
1667 fvalue = va_arg (args, double);
1668 /* um, floating point? */
1669 total += fmtfp (buffer, &currlen, maxlen, fvalue, min, max, flags);
1670 break;
1671 case 'E':
1672 flags |= DP_F_UP;
1673 case 'e':
1674 if (cflags == DP_C_LDOUBLE)
1675 fvalue = va_arg (args, LDOUBLE);
1676 else
1677 fvalue = va_arg (args, double);
1678 /* um, floating point? */
1679 total += fmtfp (buffer, &currlen, maxlen, fvalue, min, max, flags);
1680 break;
1681 case 'G':
1682 flags |= DP_F_UP;
1683 case 'g':
1684 if (cflags == DP_C_LDOUBLE)
1685 fvalue = va_arg (args, LDOUBLE);
1686 else
1687 fvalue = va_arg (args, double);
1688 /* um, floating point? */
1689 total += fmtfp (buffer, &currlen, maxlen, fvalue, min, max, flags);
1690 break;
1691 case 'c':
1692 total += dopr_outch (buffer, &currlen, maxlen, va_arg (args, int));
1693 break;
1694 case 's':
1695 strvalue = va_arg (args, char *);
1696 total += fmtstr (buffer, &currlen, maxlen, strvalue, flags, min, max);
1697 break;
1698 case 'p':
1699 strvalue = va_arg (args, void *);
1700 total += fmtint (buffer, &currlen, maxlen, (long) strvalue, 16, min,
1701 max, flags);
1702 break;
1703 case 'n':
1704 if (cflags == DP_C_SHORT)
1705 {
1706 short int *num;
1707 num = va_arg (args, short int *);
1708 *num = currlen;
1709 }
1710 else if (cflags == DP_C_LONG)
1711 {
1712 long int *num;
1713 num = va_arg (args, long int *);
1714 *num = currlen;
1715 }
1716 else if (cflags == DP_C_LLONG)
1717 {
1718 LLONG *num;
1719 num = va_arg (args, LLONG *);
1720 *num = currlen;
1721 }
1722 else
1723 {
1724 int *num;
1725 num = va_arg (args, int *);
1726 *num = currlen;
1727 }
1728 break;
1729 case '%':
1730 total += dopr_outch (buffer, &currlen, maxlen, ch);
1731 break;
1732 case 'w':
1733 /* not supported yet, treat as next char */
1734 ch = *format++;
1735 break;
1736 default:
1737 /* Unknown, skip */
1738 break;
1739 }
1740 ch = *format++;
1741 state = DP_S_DEFAULT;
1742 flags = cflags = min = 0;
1743 max = -1;
1744 break;
1745 case DP_S_DONE:
1746 break;
1747 default:
1748 /* hmm? */
1749 break; /* some picky compilers need this */
1750 }
1751 }
1752 if (maxlen > 0)
1753 buffer[currlen] = '\0';
1754 return total;
1755}

1757static int fmtstr (char *buffer, size_t *currlen, size_t maxlen,
1758 char *value, int flags, int min, int max)
1759{
1760 int padlen, strln; /* amount to pad */
1761 int cnt = 0;
1762 int total = 0;
1763
1764 if (value == 0)
1765 {
1766 value = "<NULL>";
1767 }
1768
1769 for (strln = 0; value[strln]; ++strln); /* strlen */
1770 if (max >= 0 && max < strln)
1771 strln = max;
1772 padlen = min - strln;
1773 if (padlen < 0)
1774 padlen = 0;
1775 if (flags & DP_F_MINUS)
1776 padlen = -padlen; /* Left Justify */
1777
1778 while (padlen > 0)
1779 {
1780 total += dopr_outch (buffer, currlen, maxlen, ' ');
1781 --padlen;
1782 }
1783 while (*value && ((max < 0) || (cnt < max)))
1784 {
1785 total += dopr_outch (buffer, currlen, maxlen, *value++);
1786 ++cnt;
1787 }
1788 while (padlen < 0)
1789 {
1790 total += dopr_outch (buffer, currlen, maxlen, ' ');
1791 ++padlen;
1792 }
1793 return total;
1794}

1798static int fmtint (char *buffer, size_t *currlen, size_t maxlen,
1799 LLONG value, int base, int min, int max, int flags)
1800{
1801 int signvalue = 0;
1802 unsigned LLONG uvalue;
1803 char convert[24];
1804 int place = 0;
1805 int spadlen = 0; /* amount to space pad */
1806 int zpadlen = 0; /* amount to zero pad */
1807 const char *digits;
1808 int total = 0;
1809
1810 if (max < 0)
1811 max = 0;
1812
1813 uvalue = value;
1814
1815 if(!(flags & DP_F_UNSIGNED))
1816 {
1817 if( value < 0 ) {
1818 signvalue = '-';
1819 uvalue = -value;
1820 }
1821 else
1822 if (flags & DP_F_PLUS) /* Do a sign (+/i) */
1823 signvalue = '+';
1824 else
1825 if (flags & DP_F_SPACE)
1826 signvalue = ' ';
1827 }
1828
1829 if (flags & DP_F_UP)
1830 /* Should characters be upper case? */
1831 digits = "0123456789ABCDEF";
1832 else
1833 digits = "0123456789abcdef";
1834
1835 do {
1836 convert[place++] = digits[uvalue % (unsigned)base];
1837 uvalue = (uvalue / (unsigned)base );
1838 } while(uvalue && (place < sizeof (convert)));
1839 if (place == sizeof (convert)) place--;
1840 convert[place] = 0;
1841
1842 zpadlen = max - place;
1843 spadlen = min - MAX (max, place) - (signvalue ? 1 : 0);
1844 if (zpadlen < 0) zpadlen = 0;
1845 if (spadlen < 0) spadlen = 0;
1846 if (flags & DP_F_ZERO)
1847 {
1848 zpadlen = MAX(zpadlen, spadlen);
1849 spadlen = 0;
1850 }
1851 if (flags & DP_F_MINUS)
1852 spadlen = -spadlen; /* Left Justifty */
1853
1854#ifdef DEBUG_SNPRINTF
1855 dprint (1, (debugfile, "zpad: %d, spad: %d, min: %d, max: %d, place: %d\n",
1856 zpadlen, spadlen, min, max, place));
1857#endif
1858
1859 /* Spaces */
1860 while (spadlen > 0)
1861 {
1862 total += dopr_outch (buffer, currlen, maxlen, ' ');
1863 --spadlen;
1864 }
1865
1866 /* Sign */
1867 if (signvalue)
1868 total += dopr_outch (buffer, currlen, maxlen, signvalue);
1869
1870 /* Zeros */
1871 if (zpadlen > 0)
1872 {
1873 while (zpadlen > 0)
1874 {
1875 total += dopr_outch (buffer, currlen, maxlen, '0');
1876 --zpadlen;
1877 }
1878 }
1879
1880 /* Digits */
1881 while (place > 0)
1882 total += dopr_outch (buffer, currlen, maxlen, convert[--place]);
1883
1884 /* Left Justified spaces */
1885 while (spadlen < 0) {
1886 total += dopr_outch (buffer, currlen, maxlen, ' ');
1887 ++spadlen;
1888 }
1889
1890 return total;
1891}

1915static int fmtfp (char *buffer, size_t *currlen, size_t maxlen,
1916 LDOUBLE fvalue, int min, int max, int flags)
1917{
1918 int signvalue = 0;
1919 LDOUBLE ufvalue;
1920 char iconvert[20];
1921 char fconvert[20];
1922 int iplace = 0;
1923 int fplace = 0;
1924 int padlen = 0; /* amount to pad */
1925 int zpadlen = 0;
1926 int caps = 0;
1927 int total = 0;
1928 long intpart;
1929 long fracpart;
1930
1931 /*
1932 * AIX manpage says the default is 0, but Solaris says the default
1933 * is 6, and sprintf on AIX defaults to 6
1934 */
1935 if (max < 0)
1936 max = 6;
1937
1938 ufvalue = abs_val (fvalue);
1939
1940 if (fvalue < 0)
1941 signvalue = '-';
1942 else
1943 if (flags & DP_F_PLUS) /* Do a sign (+/i) */
1944 signvalue = '+';
1945 else
1946 if (flags & DP_F_SPACE)
1947 signvalue = ' ';
1948
1949#if 0
1950 if (flags & DP_F_UP) caps = 1; /* Should characters be upper case? */
1951#endif
1952
1953 intpart = ufvalue;
1954
1955 /*
1956 * Sorry, we only support 9 digits past the decimal because of our
1957 * conversion method
1958 */
1959 if (max > 9)
1960 max = 9;
1961
1962 /* We "cheat" by converting the fractional part to integer by
1963 * multiplying by a factor of 10
1964 */
1965 fracpart = round ((powN (10, max)) * (ufvalue - intpart));
1966
1967 if (fracpart >= powN (10, max))
1968 {
1969 intpart++;
1970 fracpart -= powN (10, max);
1971 }
1972
1973#ifdef DEBUG_SNPRINTF
1974 dprint (1, (debugfile, "fmtfp: %f =? %d.%d\n", fvalue, intpart, fracpart));
1975#endif
1976
1977 /* Convert integer part */
1978 do {
1979 iconvert[iplace++] =
1980 (caps? "0123456789ABCDEF":"0123456789abcdef")[intpart % 10];
1981 intpart = (intpart / 10);
1982 } while(intpart && (iplace < 20));
1983 if (iplace == 20) iplace--;
1984 iconvert[iplace] = 0;
1985
1986 /* Convert fractional part */
1987 do {
1988 fconvert[fplace++] =
1989 (caps? "0123456789ABCDEF":"0123456789abcdef")[fracpart % 10];
1990 fracpart = (fracpart / 10);
1991 } while(fracpart && (fplace < 20));
1992 if (fplace == 20) fplace--;
1993 fconvert[fplace] = 0;
1994
1995 /* -1 for decimal point, another -1 if we are printing a sign */
1996 padlen = min - iplace - max - 1 - ((signvalue) ? 1 : 0);
1997 zpadlen = max - fplace;
1998 if (zpadlen < 0)
1999 zpadlen = 0;
2000 if (padlen < 0)
2001 padlen = 0;
2002 if (flags & DP_F_MINUS)
2003 padlen = -padlen; /* Left Justifty */
2004
2005 if ((flags & DP_F_ZERO) && (padlen > 0))
2006 {
2007 if (signvalue)
2008 {
2009 total += dopr_outch (buffer, currlen, maxlen, signvalue);
2010 --padlen;
2011 signvalue = 0;
2012 }
2013 while (padlen > 0)
2014 {
2015 total += dopr_outch (buffer, currlen, maxlen, '0');
2016 --padlen;
2017 }
2018 }
2019 while (padlen > 0)
2020 {
2021 total += dopr_outch (buffer, currlen, maxlen, ' ');
2022 --padlen;
2023 }
2024 if (signvalue)
2025 total += dopr_outch (buffer, currlen, maxlen, signvalue);
2026
2027 while (iplace > 0)
2028 total += dopr_outch (buffer, currlen, maxlen, iconvert[--iplace]);
2029
2030 /*
2031 * Decimal point. This should probably use locale to find the correct
2032 * char to print out.
2033 */
2034 if (max > 0)
2035 {
2036 total += dopr_outch (buffer, currlen, maxlen, '.');
2037
2038 while (zpadlen-- > 0)
2039 total += dopr_outch (buffer, currlen, maxlen, '0');
2040
2041 while (fplace > 0)
2042 total += dopr_outch (buffer, currlen, maxlen, fconvert[--fplace]);
2043 }
2044
2045 while (padlen < 0)
2046 {
2047 total += dopr_outch (buffer, currlen, maxlen, ' ');
2048 ++padlen;
2049 }
2050
2051 return total;
2052}

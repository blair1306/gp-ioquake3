27/*
28===============
29G_DamageFeedback
30
31Called just before a snapshot is sent to the given player.
32Totals up all damage and generates both the player_state_t
33damage values to that client for pain blends and kicks, and
34global pain sound events for all clients.
35===============
36*/
37void P_DamageFeedback( gentity_t *player ) {
38 gclient_t *client;
39 float count;
40 vec3_t angles;
41
42 client = player->client;
43 if ( client->ps.pm_type == PM_DEAD ) {
44 return;
45 }
46
47 // total points of damage shot at the player this frame
48 count = client->damage_blood + client->damage_armor;
49 if ( count == 0 ) {
50 return; // didn't take any damage
51 }
52
53 if ( count > 255 ) {
54 count = 255;
55 }
56
57 // send the information to the client
58
59 // world damage (falling, slime, etc) uses a special code
60 // to make the blend blob centered instead of positional
61 if ( client->damage_fromWorld ) {
62 client->ps.damagePitch = 255;
63 client->ps.damageYaw = 255;
64
65 client->damage_fromWorld = qfalse;
66 } else {
67 vectoangles( client->damage_from, angles );
68 client->ps.damagePitch = angles[PITCH]/360.0 * 256;
69 client->ps.damageYaw = angles[YAW]/360.0 * 256;
70 }
71
72 // play an apropriate pain sound
73 if ( (level.time > player->pain_debounce_time) && !(player->flags & FL_GODMODE) ) {
74 player->pain_debounce_time = level.time + 700;
75 G_AddEvent( player, EV_PAIN, player->health );
76 client->ps.damageEvent++;
77 }
78
79
80 client->ps.damageCount = count;
81
82 //
83 // clear totals
84 //
85 client->damage_blood = 0;
86 client->damage_armor = 0;
87 client->damage_knockback = 0;
88}

92/*
93=============
94P_WorldEffects
95
96Check for lava / slime contents and drowning
97=============
98*/
99void P_WorldEffects( gentity_t *ent ) {
100 qboolean envirosuit;
101 int waterlevel;
102
103 if ( ent->client->noclip ) {
104 ent->client->airOutTime = level.time + 12000; // don't need air
105 return;
106 }
107
108 waterlevel = ent->waterlevel;
109
110 envirosuit = ent->client->ps.powerups[PW_BATTLESUIT] > level.time;
111
112 //
113 // check for drowning
114 //
115 if ( waterlevel == 3 ) {
116 // envirosuit give air
117 if ( envirosuit ) {
118 ent->client->airOutTime = level.time + 10000;
119 }
120
121 // if out of air, start drowning
122 if ( ent->client->airOutTime < level.time) {
123 // drown!
124 ent->client->airOutTime += 1000;
125 if ( ent->health > 0 ) {
126 // take more damage the longer underwater
127 ent->damage += 2;
128 if (ent->damage > 15)
129 ent->damage = 15;
130
131 // don't play a normal pain sound
132 ent->pain_debounce_time = level.time + 200;
133
134 G_Damage (ent, NULL, NULL, NULL, NULL,
135 ent->damage, DAMAGE_NO_ARMOR, MOD_WATER);
136 }
137 }
138 } else {
139 ent->client->airOutTime = level.time + 12000;
140 ent->damage = 2;
141 }
142
143 //
144 // check for sizzle damage (move to pmove?)
145 //
146 if (waterlevel &&
147 (ent->watertype&(CONTENTS_LAVA|CONTENTS_SLIME)) ) {
148 if (ent->health > 0
149 && ent->pain_debounce_time <= level.time ) {
150
151 if ( envirosuit ) {
152 G_AddEvent( ent, EV_POWERUP_BATTLESUIT, 0 );
153 } else {
154 if (ent->watertype & CONTENTS_LAVA) {
155 G_Damage (ent, NULL, NULL, NULL, NULL,
156 30*waterlevel, 0, MOD_LAVA);
157 }
158
159 if (ent->watertype & CONTENTS_SLIME) {
160 G_Damage (ent, NULL, NULL, NULL, NULL,
161 10*waterlevel, 0, MOD_SLIME);
162 }
163 }
164 }
165 }
166}

193/*
194==============
195ClientImpacts
196==============
197*/
198void ClientImpacts( gentity_t *ent, pmove_t *pm ) {
199 int i, j;
200 trace_t trace;
201 gentity_t *other;
202
203 memset( &trace, 0, sizeof( trace ) );
204 for (i=0 ; i<pm->numtouch ; i++) {
205 for (j=0 ; j<i ; j++) {
206 if (pm->touchents[j] == pm->touchents[i] ) {
207 break;
208 }
209 }
210 if (j != i) {
211 continue; // duplicated
212 }
213 other = &g_entities[ pm->touchents[i] ];
214
215 if ( ( ent->r.svFlags & SVF_BOT ) && ( ent->touch ) ) {
216 ent->touch( ent, other, &trace );
217 }
218
219 if ( !other->touch ) {
220 continue;
221 }
222
223 other->touch( other, ent, &trace );
224 }
225

228/*
229============
230G_TouchTriggers
231
232Find all trigger entities that ent's current position touches.
233Spectators will only interact with teleporters.
234============
235*/
236void G_TouchTriggers( gentity_t *ent ) {
237 int i, num;
238 int touch[MAX_GENTITIES];
239 gentity_t *hit;
240 trace_t trace;
241 vec3_t mins, maxs;
242 static vec3_t range = { 40, 40, 52 };
243
244 if ( !ent->client ) {
245 return;
246 }
247
248 // dead clients don't activate triggers!
249 if ( ent->client->ps.stats[STAT_HEALTH] <= 0 ) {
250 return;
251 }
252
253 VectorSubtract( ent->client->ps.origin, range, mins );
254 VectorAdd( ent->client->ps.origin, range, maxs );
255
256 num = trap_EntitiesInBox( mins, maxs, touch, MAX_GENTITIES );
257
258 // can't use ent->absmin, because that has a one unit pad
259 VectorAdd( ent->client->ps.origin, ent->r.mins, mins );
260 VectorAdd( ent->client->ps.origin, ent->r.maxs, maxs );
261
262 for ( i=0 ; i<num ; i++ ) {
263 hit = &g_entities[touch[i]];
264
265 if ( !hit->touch && !ent->touch ) {
266 continue;
267 }
268 if ( !( hit->r.contents & CONTENTS_TRIGGER ) ) {
269 continue;
270 }
271
272 // ignore most entities if a spectator
273 if ( ent->client->sess.sessionTeam == TEAM_SPECTATOR ) {
274 if ( hit->s.eType != ET_TELEPORT_TRIGGER &&
275 // this is ugly but adding a new ET_? type will
276 // most likely cause network incompatibilities
277 hit->touch != Touch_DoorTrigger) {
278 continue;
279 }
280 }
281
282 // use seperate code for determining if an item is picked up
283 // so you don't have to actually contact its bounding box
284 if ( hit->s.eType == ET_ITEM ) {
285 if ( !BG_PlayerTouchesItem( &ent->client->ps, &hit->s, level.time ) ) {
286 continue;
287 }
288 } else {
289 if ( !trap_EntityContact( mins, maxs, hit ) ) {
290 continue;
291 }
292 }
293
294 memset( &trace, 0, sizeof(trace) );
295
296 if ( hit->touch ) {
297 hit->touch (hit, ent, &trace);
298 }
299
300 if ( ( ent->r.svFlags & SVF_BOT ) && ( ent->touch ) ) {
301 ent->touch( ent, hit, &trace );
302 }
303 }
304
305 // if we didn't touch a jump pad this pmove frame
306 if ( ent->client->ps.jumppad_frame != ent->client->ps.pmove_framecount ) {
307 ent->client->ps.jumppad_frame = 0;
308 ent->client->ps.jumppad_ent = 0;
309 }
310}

312/*
313=================
314SpectatorThink
315=================
316*/
317void SpectatorThink( gentity_t *ent, usercmd_t *ucmd ) {
318 pmove_t pm;
319 gclient_t *client;
320
321 client = ent->client;
322
323 if ( client->sess.spectatorState != SPECTATOR_FOLLOW ) {
324 if ( client->noclip ) {
325 client->ps.pm_type = PM_NOCLIP;
326 } else {
327 client->ps.pm_type = PM_SPECTATOR;
328 }
329
330 client->ps.speed = 400; // faster than normal
331
332 // set up for pmove
333 memset (&pm, 0, sizeof(pm));
334 pm.ps = &client->ps;
335 pm.cmd = *ucmd;
336 pm.tracemask = MASK_PLAYERSOLID & ~CONTENTS_BODY; // spectators can fly through bodies
337 pm.trace = trap_Trace;
338 pm.pointcontents = trap_PointContents;
339
340 // perform a pmove
341 Pmove (&pm);
342 // save results of pmove
343 VectorCopy( client->ps.origin, ent->s.origin );
344
345 G_TouchTriggers( ent );
346 trap_UnlinkEntity( ent );
347 }
348
349 client->oldbuttons = client->buttons;
350 client->buttons = ucmd->buttons;
351
352 // attack button cycles through spectators
353 if ( ( client->buttons & BUTTON_ATTACK ) && ! ( client->oldbuttons & BUTTON_ATTACK ) ) {
354 Cmd_FollowCycle_f( ent, 1 );
355 }
356}

360/*
361=================
362ClientInactivityTimer
363
364Returns qfalse if the client is dropped
365=================
366*/
367qboolean ClientInactivityTimer( gclient_t *client ) {
368 if ( ! g_inactivity.integer ) {
369 // give everyone some time, so if the operator sets g_inactivity during
370 // gameplay, everyone isn't kicked
371 client->inactivityTime = level.time + 60 * 1000;
372 client->inactivityWarning = qfalse;
373 } else if ( client->pers.cmd.forwardmove ||
374 client->pers.cmd.rightmove ||
375 client->pers.cmd.upmove ||
376 (client->pers.cmd.buttons & BUTTON_ATTACK) ) {
377 client->inactivityTime = level.time + g_inactivity.integer * 1000;
378 client->inactivityWarning = qfalse;
379 } else if ( !client->pers.localClient ) {
380 if ( level.time > client->inactivityTime ) {
381 trap_DropClient( client - level.clients, "Dropped due to inactivity" );
382 return qfalse;
383 }
384 if ( level.time > client->inactivityTime - 10000 && !client->inactivityWarning ) {
385 client->inactivityWarning = qtrue;
386 trap_SendServerCommand( client - level.clients, "cp \"Ten seconds until inactivity drop!\n\"" );
387 }
388 }
389 return qtrue;
390}

392/*
393==================
394ClientTimerActions
395
396Actions that happen once a second
397==================
398*/
399void ClientTimerActions( gentity_t *ent, int msec ) {
400 gclient_t *client;
401#ifdef MISSIONPACK
402 int maxHealth;
403#endif
404
405 client = ent->client;
406 client->timeResidual += msec;
407
408 while ( client->timeResidual >= 1000 ) {
409 client->timeResidual -= 1000;
410
411 // regenerate
412#ifdef MISSIONPACK
413 if( bg_itemlist[client->ps.stats[STAT_PERSISTANT_POWERUP]].giTag == PW_GUARD ) {
414 maxHealth = client->ps.stats[STAT_MAX_HEALTH] / 2;
415 }
416 else if ( client->ps.powerups[PW_REGEN] ) {
417 maxHealth = client->ps.stats[STAT_MAX_HEALTH];
418 }
419 else {
420 maxHealth = 0;
421 }
422 if( maxHealth ) {
423 if ( ent->health < maxHealth ) {
424 ent->health += 15;
425 if ( ent->health > maxHealth * 1.1 ) {
426 ent->health = maxHealth * 1.1;
427 }
428 G_AddEvent( ent, EV_POWERUP_REGEN, 0 );
429 } else if ( ent->health < maxHealth * 2) {
430 ent->health += 5;
431 if ( ent->health > maxHealth * 2 ) {
432 ent->health = maxHealth * 2;
433 }
434 G_AddEvent( ent, EV_POWERUP_REGEN, 0 );
435 }
436#else
437 if ( client->ps.powerups[PW_REGEN] ) {
438 if ( ent->health < client->ps.stats[STAT_MAX_HEALTH]) {
439 ent->health += 15;
440 if ( ent->health > client->ps.stats[STAT_MAX_HEALTH] * 1.1 ) {
441 ent->health = client->ps.stats[STAT_MAX_HEALTH] * 1.1;
442 }
443 G_AddEvent( ent, EV_POWERUP_REGEN, 0 );
444 } else if ( ent->health < client->ps.stats[STAT_MAX_HEALTH] * 2) {
445 ent->health += 5;
446 if ( ent->health > client->ps.stats[STAT_MAX_HEALTH] * 2 ) {
447 ent->health = client->ps.stats[STAT_MAX_HEALTH] * 2;
448 }
449 G_AddEvent( ent, EV_POWERUP_REGEN, 0 );
450 }
451#endif
452 } else {
453 // count down health when over max
454 if ( ent->health > client->ps.stats[STAT_MAX_HEALTH] ) {
455 ent->health--;
456 }
457 }
458
459 // count down armor when over max
460 if ( client->ps.stats[STAT_ARMOR] > client->ps.stats[STAT_MAX_HEALTH] ) {
461 client->ps.stats[STAT_ARMOR]--;
462 }
463 }
464#ifdef MISSIONPACK
465 if( bg_itemlist[client->ps.stats[STAT_PERSISTANT_POWERUP]].giTag == PW_AMMOREGEN ) {
466 int w, max, inc, t, i;
467 int weapList[]={WP_MACHINEGUN,WP_SHOTGUN,WP_GRENADE_LAUNCHER,WP_ROCKET_LAUNCHER,WP_LIGHTNING,WP_RAILGUN,WP_PLASMAGUN,WP_BFG,WP_NAILGUN,WP_PROX_LAUNCHER,WP_CHAINGUN};
468 int weapCount = ARRAY_LEN( weapList );
469 //
470 for (i = 0; i < weapCount; i++) {
471 w = weapList[i];
472
473 switch(w) {
474 case WP_MACHINEGUN: max = 50; inc = 4; t = 1000; break;
475 case WP_SHOTGUN: max = 10; inc = 1; t = 1500; break;
476 case WP_GRENADE_LAUNCHER: max = 10; inc = 1; t = 2000; break;
477 case WP_ROCKET_LAUNCHER: max = 10; inc = 1; t = 1750; break;
478 case WP_LIGHTNING: max = 50; inc = 5; t = 1500; break;
479 case WP_RAILGUN: max = 10; inc = 1; t = 1750; break;
480 case WP_PLASMAGUN: max = 50; inc = 5; t = 1500; break;
481 case WP_BFG: max = 10; inc = 1; t = 4000; break;
482 case WP_NAILGUN: max = 10; inc = 1; t = 1250; break;
483 case WP_PROX_LAUNCHER: max = 5; inc = 1; t = 2000; break;
484 case WP_CHAINGUN: max = 100; inc = 5; t = 1000; break;
485 default: max = 0; inc = 0; t = 1000; break;
486 }
487 client->ammoTimes[w] += msec;
488 if ( client->ps.ammo[w] >= max ) {
489 client->ammoTimes[w] = 0;
490 }
491 if ( client->ammoTimes[w] >= t ) {
492 while ( client->ammoTimes[w] >= t )
493 client->ammoTimes[w] -= t;
494 client->ps.ammo[w] += inc;
495 if ( client->ps.ammo[w] > max ) {
496 client->ps.ammo[w] = max;
497 }
498 }
499 }
500 }
501#endif
502}

525/*
526================
527ClientEvents
528
529Events will be passed on to the clients for presentation,
530but any server game effects are handled here
531================
532*/
533void ClientEvents( gentity_t *ent, int oldEventSequence ) {
534 int i, j;
535 int event;
536 gclient_t *client;
537 int damage;
538 vec3_t origin, angles;
539// qboolean fired;
540 gitem_t *item;
541 gentity_t *drop;
542
543 client = ent->client;
544
545 if ( oldEventSequence < client->ps.eventSequence - MAX_PS_EVENTS ) {
546 oldEventSequence = client->ps.eventSequence - MAX_PS_EVENTS;
547 }
548 for ( i = oldEventSequence ; i < client->ps.eventSequence ; i++ ) {
549 event = client->ps.events[ i & (MAX_PS_EVENTS-1) ];
550
551 switch ( event ) {
552 case EV_FALL_MEDIUM:
553 case EV_FALL_FAR:
554 if ( ent->s.eType != ET_PLAYER ) {
555 break; // not in the player model
556 }
557 if ( g_dmflags.integer & DF_NO_FALLING ) {
558 break;
559 }
560 if ( event == EV_FALL_FAR ) {
561 damage = 10;
562 } else {
563 damage = 5;
564 }
565 ent->pain_debounce_time = level.time + 200; // no normal pain sound
566 G_Damage (ent, NULL, NULL, NULL, NULL, damage, 0, MOD_FALLING);
567 break;
568
569 case EV_FIRE_WEAPON:
570 FireWeapon( ent );
571 break;
572
573 case EV_USE_ITEM1: // teleporter
574 // drop flags in CTF
575 item = NULL;
576 j = 0;
577
578 if ( ent->client->ps.powerups[ PW_REDFLAG ] ) {
579 item = BG_FindItemForPowerup( PW_REDFLAG );
580 j = PW_REDFLAG;
581 } else if ( ent->client->ps.powerups[ PW_BLUEFLAG ] ) {
582 item = BG_FindItemForPowerup( PW_BLUEFLAG );
583 j = PW_BLUEFLAG;
584 } else if ( ent->client->ps.powerups[ PW_NEUTRALFLAG ] ) {
585 item = BG_FindItemForPowerup( PW_NEUTRALFLAG );
586 j = PW_NEUTRALFLAG;
587 }
588
589 if ( item ) {
590 drop = Drop_Item( ent, item, 0 );
591 // decide how many seconds it has left
592 drop->count = ( ent->client->ps.powerups[ j ] - level.time ) / 1000;
593 if ( drop->count < 1 ) {
594 drop->count = 1;
595 }
596
597 ent->client->ps.powerups[ j ] = 0;
598 }
599
600#ifdef MISSIONPACK
601 if ( g_gametype.integer == GT_HARVESTER ) {
602 if ( ent->client->ps.generic1 > 0 ) {
603 if ( ent->client->sess.sessionTeam == TEAM_RED ) {
604 item = BG_FindItem( "Blue Cube" );
605 } else {
606 item = BG_FindItem( "Red Cube" );
607 }
608 if ( item ) {
609 for ( j = 0; j < ent->client->ps.generic1; j++ ) {
610 drop = Drop_Item( ent, item, 0 );
611 if ( ent->client->sess.sessionTeam == TEAM_RED ) {
612 drop->spawnflags = TEAM_BLUE;
613 } else {
614 drop->spawnflags = TEAM_RED;
615 }
616 }
617 }
618 ent->client->ps.generic1 = 0;
619 }
620 }
621#endif
622 SelectSpawnPoint( ent->client->ps.origin, origin, angles, qfalse );
623 TeleportPlayer( ent, origin, angles );
624 break;
625
626 case EV_USE_ITEM2: // medkit
627 ent->health = ent->client->ps.stats[STAT_MAX_HEALTH] + 25;
628
629 break;
630
631#ifdef MISSIONPACK
632 case EV_USE_ITEM3: // kamikaze
633 // make sure the invulnerability is off
634 ent->client->invulnerabilityTime = 0;
635 // start the kamikze
636 G_StartKamikaze( ent );
637 break;
638
639 case EV_USE_ITEM4: // portal
640 if( ent->client->portalID ) {
641 DropPortalSource( ent );
642 }
643 else {
644 DropPortalDestination( ent );
645 }
646 break;
647 case EV_USE_ITEM5: // invulnerability
648 ent->client->invulnerabilityTime = level.time + 10000;
649 break;
650#endif
651
652 default:
653 break;
654 }
655 }
656
657}

704/*
705==============
706SendPendingPredictableEvents
707==============
708*/
709void SendPendingPredictableEvents( playerState_t *ps ) {
710 gentity_t *t;
711 int event, seq;
712 int extEvent, number;
713
714 // if there are still events pending
715 if ( ps->entityEventSequence < ps->eventSequence ) {
716 // create a temporary entity for this event which is sent to everyone
717 // except the client who generated the event
718 seq = ps->entityEventSequence & (MAX_PS_EVENTS-1);
719 event = ps->events[ seq ] | ( ( ps->entityEventSequence & 3 ) << 8 );
720 // set external event to zero before calling BG_PlayerStateToEntityState
721 extEvent = ps->externalEvent;
722 ps->externalEvent = 0;
723 // create temporary entity for event
724 t = G_TempEntity( ps->origin, event );
725 number = t->s.number;
726 BG_PlayerStateToEntityState( ps, &t->s, qtrue );
727 t->s.number = number;
728 t->s.eType = ET_EVENTS + event;
729 t->s.eFlags |= EF_PLAYER_EVENT;
730 t->s.otherEntityNum = ps->clientNum;
731 // send to everyone except the client who generated the event
732 t->r.svFlags |= SVF_NOTSINGLECLIENT;
733 t->r.singleClient = ps->clientNum;
734 // set back external event
735 ps->externalEvent = extEvent;
736 }
737}

739/*
740==============
741ClientThink
742
743This will be called once for each client frame, which will
744usually be a couple times for each server frame on fast clients.
745
746If "g_synchronousClients 1" is set, this will be called exactly
747once for each server frame, which makes for smooth demo recording.
748==============
749*/
750void ClientThink_real( gentity_t *ent ) {
751 gclient_t *client;
752 pmove_t pm;
753 int oldEventSequence;
754 int msec;
755 usercmd_t *ucmd;
756
757 client = ent->client;
758
759 // don't think if the client is not yet connected (and thus not yet spawned in)
760 if (client->pers.connected != CON_CONNECTED) {
761 return;
762 }
763 // mark the time, so the connection sprite can be removed
764 ucmd = &ent->client->pers.cmd;
765
766 // sanity check the command time to prevent speedup cheating
767 if ( ucmd->serverTime > level.time + 200 ) {
768 ucmd->serverTime = level.time + 200;
769// G_Printf("serverTime <<<<<\n" );
770 }
771 if ( ucmd->serverTime < level.time - 1000 ) {
772 ucmd->serverTime = level.time - 1000;
773// G_Printf("serverTime >>>>>\n" );
774 }
775
776 msec = ucmd->serverTime - client->ps.commandTime;
777 // following others may result in bad times, but we still want
778 // to check for follow toggles
779 if ( msec < 1 && client->sess.spectatorState != SPECTATOR_FOLLOW ) {
780 return;
781 }
782 if ( msec > 200 ) {
783 msec = 200;
784 }
785
786 if ( pmove_msec.integer < 8 ) {
787 trap_Cvar_Set("pmove_msec", "8");
788 trap_Cvar_Update(&pmove_msec);
789 }
790 else if (pmove_msec.integer > 33) {
791 trap_Cvar_Set("pmove_msec", "33");
792 trap_Cvar_Update(&pmove_msec);
793 }
794
795 if ( pmove_fixed.integer || client->pers.pmoveFixed ) {
796 ucmd->serverTime = ((ucmd->serverTime + pmove_msec.integer-1) / pmove_msec.integer) * pmove_msec.integer;
797 //if (ucmd->serverTime - client->ps.commandTime <= 0)
798 // return;
799 }
800
801 //
802 // check for exiting intermission
803 //
804 if ( level.intermissiontime ) {
805 ClientIntermissionThink( client );
806 return;
807 }
808
809 // spectators don't do much
810 if ( client->sess.sessionTeam == TEAM_SPECTATOR ) {
811 if ( client->sess.spectatorState == SPECTATOR_SCOREBOARD ) {
812 return;
813 }
814 SpectatorThink( ent, ucmd );
815 return;
816 }
817
818 // check for inactivity timer, but never drop the local client of a non-dedicated server
819 if ( !ClientInactivityTimer( client ) ) {
820 return;
821 }
822
823 // clear the rewards if time
824 if ( level.time > client->rewardTime ) {
825 client->ps.eFlags &= ~(EF_AWARD_IMPRESSIVE | EF_AWARD_EXCELLENT | EF_AWARD_GAUNTLET | EF_AWARD_ASSIST | EF_AWARD_DEFEND | EF_AWARD_CAP );
826 }
827
828 if ( client->noclip ) {
829 client->ps.pm_type = PM_NOCLIP;
830 } else if ( client->ps.stats[STAT_HEALTH] <= 0 ) {
831 client->ps.pm_type = PM_DEAD;
832 } else {
833 client->ps.pm_type = PM_NORMAL;
834 }
835
836 client->ps.gravity = g_gravity.value;
837
838 // set speed
839 client->ps.speed = g_speed.value;
840
841#ifdef MISSIONPACK
842 if( bg_itemlist[client->ps.stats[STAT_PERSISTANT_POWERUP]].giTag == PW_SCOUT ) {
843 client->ps.speed *= 1.5;
844 }
845 else
846#endif
847 if ( client->ps.powerups[PW_HASTE] ) {
848 client->ps.speed *= 1.3;
849 }
850
851 // Let go of the hook if we aren't firing
852 if ( client->ps.weapon == WP_GRAPPLING_HOOK &&
853 client->hook && !( ucmd->buttons & BUTTON_ATTACK ) ) {
854 Weapon_HookFree(client->hook);
855 }
856
857 // set up for pmove
858 oldEventSequence = client->ps.eventSequence;
859
860 memset (&pm, 0, sizeof(pm));
861
862 // check for the hit-scan gauntlet, don't let the action
863 // go through as an attack unless it actually hits something
864 if ( client->ps.weapon == WP_GAUNTLET && !( ucmd->buttons & BUTTON_TALK ) &&
865 ( ucmd->buttons & BUTTON_ATTACK ) && client->ps.weaponTime <= 0 ) {
866 pm.gauntletHit = CheckGauntletAttack( ent );
867 }
868
869 if ( ent->flags & FL_FORCE_GESTURE ) {
870 ent->flags &= ~FL_FORCE_GESTURE;
871 ent->client->pers.cmd.buttons |= BUTTON_GESTURE;
872 }
873
874#ifdef MISSIONPACK
875 // check for invulnerability expansion before doing the Pmove
876 if (client->ps.powerups[PW_INVULNERABILITY] ) {
877 if ( !(client->ps.pm_flags & PMF_INVULEXPAND) ) {
878 vec3_t mins = { -42, -42, -42 };
879 vec3_t maxs = { 42, 42, 42 };
880 vec3_t oldmins, oldmaxs;
881
882 VectorCopy (ent->r.mins, oldmins);
883 VectorCopy (ent->r.maxs, oldmaxs);
884 // expand
885 VectorCopy (mins, ent->r.mins);
886 VectorCopy (maxs, ent->r.maxs);
887 trap_LinkEntity(ent);
888 // check if this would get anyone stuck in this player
889 if ( !StuckInOtherClient(ent) ) {
890 // set flag so the expanded size will be set in PM_CheckDuck
891 client->ps.pm_flags |= PMF_INVULEXPAND;
892 }
893 // set back
894 VectorCopy (oldmins, ent->r.mins);
895 VectorCopy (oldmaxs, ent->r.maxs);
896 trap_LinkEntity(ent);
897 }
898 }
899#endif
900
901 pm.ps = &client->ps;
902 pm.cmd = *ucmd;
903 if ( pm.ps->pm_type == PM_DEAD ) {
904 pm.tracemask = MASK_PLAYERSOLID & ~CONTENTS_BODY;
905 }
906 else if ( ent->r.svFlags & SVF_BOT ) {
907 pm.tracemask = MASK_PLAYERSOLID | CONTENTS_BOTCLIP;
908 }
909 else {
910 pm.tracemask = MASK_PLAYERSOLID;
911 }
912 pm.trace = trap_Trace;
913 pm.pointcontents = trap_PointContents;
914 pm.debugLevel = g_debugMove.integer;
915 pm.noFootsteps = ( g_dmflags.integer & DF_NO_FOOTSTEPS ) > 0;
916
917 pm.pmove_fixed = pmove_fixed.integer | client->pers.pmoveFixed;
918 pm.pmove_msec = pmove_msec.integer;
919
920 VectorCopy( client->ps.origin, client->oldOrigin );
921
922#ifdef MISSIONPACK
923 if (level.intermissionQueued != 0 && g_singlePlayer.integer) {
924 if ( level.time - level.intermissionQueued >= 1000 ) {
925 pm.cmd.buttons = 0;
926 pm.cmd.forwardmove = 0;
927 pm.cmd.rightmove = 0;
928 pm.cmd.upmove = 0;
929 if ( level.time - level.intermissionQueued >= 2000 && level.time - level.intermissionQueued <= 2500 ) {
930 trap_SendConsoleCommand( EXEC_APPEND, "centerview\n");
931 }
932 ent->client->ps.pm_type = PM_SPINTERMISSION;
933 }
934 }
935 Pmove (&pm);
936#else
937 Pmove (&pm);
938#endif
939
940 // save results of pmove
941 if ( ent->client->ps.eventSequence != oldEventSequence ) {
942 ent->eventTime = level.time;
943 }
944 if (g_smoothClients.integer) {
945 BG_PlayerStateToEntityStateExtraPolate( &ent->client->ps, &ent->s, ent->client->ps.commandTime, qtrue );
946 }
947 else {
948 BG_PlayerStateToEntityState( &ent->client->ps, &ent->s, qtrue );
949 }
950 SendPendingPredictableEvents( &ent->client->ps );
951
952 if ( !( ent->client->ps.eFlags & EF_FIRING ) ) {
953 client->fireHeld = qfalse; // for grapple
954 }
955
956 // use the snapped origin for linking so it matches client predicted versions
957 VectorCopy( ent->s.pos.trBase, ent->r.currentOrigin );
958
959 VectorCopy (pm.mins, ent->r.mins);
960 VectorCopy (pm.maxs, ent->r.maxs);
961
962 ent->waterlevel = pm.waterlevel;
963 ent->watertype = pm.watertype;
964
965 // execute client events
966 ClientEvents( ent, oldEventSequence );
967
968 // link entity now, after any personal teleporters have been used
969 trap_LinkEntity (ent);
970 if ( !ent->client->noclip ) {
971 G_TouchTriggers( ent );
972 }
973
974 // NOTE: now copy the exact origin over otherwise clients can be snapped into solid
975 VectorCopy( ent->client->ps.origin, ent->r.currentOrigin );
976
977 //test for solid areas in the AAS file
978 BotTestAAS(ent->r.currentOrigin);
979
980 // touch other objects
981 ClientImpacts( ent, &pm );
982
983 // save results of triggers and client events
984 if (ent->client->ps.eventSequence != oldEventSequence) {
985 ent->eventTime = level.time;
986 }
987
988 // swap and latch button actions
989 client->oldbuttons = client->buttons;
990 client->buttons = ucmd->buttons;
991 client->latched_buttons |= client->buttons & ~client->oldbuttons;
992
993 // check for respawning
994 if ( client->ps.stats[STAT_HEALTH] <= 0 ) {
995 // wait for the attack button to be pressed
996 if ( level.time > client->respawnTime ) {
997 // forcerespawn is to prevent users from waiting out powerups
998 if ( g_forcerespawn.integer > 0 &&
999 ( level.time - client->respawnTime ) > g_forcerespawn.integer * 1000 ) {
1000 ClientRespawn( ent );
1001 return;
1002 }
1003
1004 // pressing attack or use is the normal respawn method
1005 if ( ucmd->buttons & ( BUTTON_ATTACK | BUTTON_USE_HOLDABLE ) ) {
1006 ClientRespawn( ent );
1007 }
1008 }
1009 return;
1010 }
1011
1012 // perform once-a-second actions
1013 ClientTimerActions( ent, msec );
1014}

1054void SpectatorClientEndFrame( gentity_t *ent ) {
1055 gclient_t *cl;
1056
1057 // if we are doing a chase cam or a remote view, grab the latest info
1058 if ( ent->client->sess.spectatorState == SPECTATOR_FOLLOW ) {
1059 int clientNum, flags;
1060
1061 clientNum = ent->client->sess.spectatorClient;
1062
1063 // team follow1 and team follow2 go to whatever clients are playing
1064 if ( clientNum == -1 ) {
1065 clientNum = level.follow1;
1066 } else if ( clientNum == -2 ) {
1067 clientNum = level.follow2;
1068 }
1069 if ( clientNum >= 0 ) {
1070 cl = &level.clients[ clientNum ];
1071 if ( cl->pers.connected == CON_CONNECTED && cl->sess.sessionTeam != TEAM_SPECTATOR ) {
1072 flags = (cl->ps.eFlags & ~(EF_VOTED | EF_TEAMVOTED)) | (ent->client->ps.eFlags & (EF_VOTED | EF_TEAMVOTED));
1073 ent->client->ps = cl->ps;
1074 ent->client->ps.pm_flags |= PMF_FOLLOW;
1075 ent->client->ps.eFlags = flags;
1076 return;
1077 } else {
1078 // drop them to free spectators unless they are dedicated camera followers
1079 if ( ent->client->sess.spectatorClient >= 0 ) {
1080 ent->client->sess.spectatorState = SPECTATOR_FREE;
1081 ClientBegin( ent->client - level.clients );
1082 }
1083 }
1084 }
1085 }
1086
1087 if ( ent->client->sess.spectatorState == SPECTATOR_SCOREBOARD ) {
1088 ent->client->ps.pm_flags |= PMF_SCOREBOARD;
1089 } else {
1090 ent->client->ps.pm_flags &= ~PMF_SCOREBOARD;
1091 }
1092}

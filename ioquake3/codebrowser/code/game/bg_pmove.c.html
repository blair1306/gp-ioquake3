66void PM_AddTouchEnt( int entityNum ) {
67 int i;
68
69 if ( entityNum == ENTITYNUM_WORLD ) {
70 return;
71 }
72 if ( pm->numtouch == MAXTOUCH ) {
73 return;
74 }
75
76 // see if it is already added
77 for ( i = 0 ; i < pm->numtouch ; i++ ) {
78 if ( pm->touchents[ i ] == entityNum ) {
79 return;
80 }
81 }
82
83 // add it
84 pm->touchents[pm->numtouch] = entityNum;
85 pm->numtouch++;
86}

137/*
138==================
139PM_ClipVelocity
140
141Slide off of the impacting surface
142==================
143*/
144void PM_ClipVelocity( vec3_t in, vec3_t normal, vec3_t out, float overbounce ) {
145 float backoff;
146 float change;
147 int i;
148
149 backoff = DotProduct (in, normal);
150
151 if ( backoff < 0 ) {
152 backoff *= overbounce;
153 } else {
154 backoff /= overbounce;
155 }
156
157 for ( i=0 ; i<3 ; i++ ) {
158 change = normal[i]*backoff;
159 out[i] = in[i] - change;
160 }
161}

164/*
165==================
166PM_Friction
167
168Handles both ground friction and water friction
169==================
170*/
171static void PM_Friction( void ) {
172 vec3_t vec;
173 float *vel;
174 float speed, newspeed, control;
175 float drop;
176
177 vel = pm->ps->velocity;
178
179 VectorCopy( vel, vec );
180 if ( pml.walking ) {
181 vec[2] = 0; // ignore slope movement
182 }
183
184 speed = VectorLength(vec);
185 if (speed < 1) {
186 vel[0] = 0;
187 vel[1] = 0; // allow sinking underwater
188 // FIXME: still have z friction underwater?
189 return;
190 }
191
192 drop = 0;
193
194 // apply ground friction
195 if ( pm->waterlevel <= 1 ) {
196 if ( pml.walking && !(pml.groundTrace.surfaceFlags & SURF_SLICK) ) {
197 // if getting knocked back, no friction
198 if ( ! (pm->ps->pm_flags & PMF_TIME_KNOCKBACK) ) {
199 control = speed < pm_stopspeed ? pm_stopspeed : speed;
200 drop += control*pm_friction*pml.frametime;
201 }
202 }
203 }
204
205 // apply water friction even if just wading
206 if ( pm->waterlevel ) {
207 drop += speed*pm_waterfriction*pm->waterlevel*pml.frametime;
208 }
209
210 // apply flying friction
211 if ( pm->ps->powerups[PW_FLIGHT]) {
212 drop += speed*pm_flightfriction*pml.frametime;
213 }
214
215 if ( pm->ps->pm_type == PM_SPECTATOR) {
216 drop += speed*pm_spectatorfriction*pml.frametime;
217 }
218
219 // scale the velocity
220 newspeed = speed - drop;
221 if (newspeed < 0) {
222 newspeed = 0;
223 }
224 newspeed /= speed;
225
226 vel[0] = vel[0] * newspeed;
227 vel[1] = vel[1] * newspeed;
228 vel[2] = vel[2] * newspeed;
229}

232/*
233==============
234PM_Accelerate
235
236Handles user intended acceleration
237==============
238*/
239static void PM_Accelerate( vec3_t wishdir, float wishspeed, float accel ) {
240#if 1
241 // q2 style
242 int i;
243 float addspeed, accelspeed, currentspeed;
244
245 currentspeed = DotProduct (pm->ps->velocity, wishdir);
246 addspeed = wishspeed - currentspeed;
247 if (addspeed <= 0) {
248 return;
249 }
250 accelspeed = accel*pml.frametime*wishspeed;
251 if (accelspeed > addspeed) {
252 accelspeed = addspeed;
253 }
254
255 for (i=0 ; i<3 ; i++) {
256 pm->ps->velocity[i] += accelspeed*wishdir[i];
257 }
258#else
259 // proper way (avoids strafe jump maxspeed bug), but feels bad
260 vec3_t wishVelocity;
261 vec3_t pushDir;
262 float pushLen;
263 float canPush;
264
265 VectorScale( wishdir, wishspeed, wishVelocity );
266 VectorSubtract( wishVelocity, pm->ps->velocity, pushDir );
267 pushLen = VectorNormalize( pushDir );
268
269 canPush = accel*pml.frametime*wishspeed;
270 if (canPush > pushLen) {
271 canPush = pushLen;
272 }
273
274 VectorMA( pm->ps->velocity, canPush, pushDir, pm->ps->velocity );
275#endif
276}

280/*
281============
282PM_CmdScale
283
284Returns the scale factor to apply to cmd movements
285This allows the clients to use axial -127 to 127 values for all directions
286without getting a sqrt(2) distortion in speed.
287============
288*/
289static float PM_CmdScale( usercmd_t *cmd ) {
290 int max;
291 float total;
292 float scale;
293
294 max = abs( cmd->forwardmove );
295 if ( abs( cmd->rightmove ) > max ) {
296 max = abs( cmd->rightmove );
297 }
298 if ( abs( cmd->upmove ) > max ) {
299 max = abs( cmd->upmove );
300 }
301 if ( !max ) {
302 return 0;
303 }
304
305 total = sqrt( cmd->forwardmove * cmd->forwardmove
306 + cmd->rightmove * cmd->rightmove + cmd->upmove * cmd->upmove );
307 scale = (float)pm->ps->speed * max / ( 127.0 * total );
308
309 return scale;
310}

313/*
314================
315PM_SetMovementDir
316
317Determine the rotation of the legs relative
318to the facing dir
319================
320*/
321static void PM_SetMovementDir( void ) {
322 if ( pm->cmd.forwardmove || pm->cmd.rightmove ) {
323 if ( pm->cmd.rightmove == 0 && pm->cmd.forwardmove > 0 ) {
324 pm->ps->movementDir = 0;
325 } else if ( pm->cmd.rightmove < 0 && pm->cmd.forwardmove > 0 ) {
326 pm->ps->movementDir = 1;
327 } else if ( pm->cmd.rightmove < 0 && pm->cmd.forwardmove == 0 ) {
328 pm->ps->movementDir = 2;
329 } else if ( pm->cmd.rightmove < 0 && pm->cmd.forwardmove < 0 ) {
330 pm->ps->movementDir = 3;
331 } else if ( pm->cmd.rightmove == 0 && pm->cmd.forwardmove < 0 ) {
332 pm->ps->movementDir = 4;
333 } else if ( pm->cmd.rightmove > 0 && pm->cmd.forwardmove < 0 ) {
334 pm->ps->movementDir = 5;
335 } else if ( pm->cmd.rightmove > 0 && pm->cmd.forwardmove == 0 ) {
336 pm->ps->movementDir = 6;
337 } else if ( pm->cmd.rightmove > 0 && pm->cmd.forwardmove > 0 ) {
338 pm->ps->movementDir = 7;
339 }
340 } else {
341 // if they aren't actively going directly sideways,
342 // change the animation to the diagonal so they
343 // don't stop too crooked
344 if ( pm->ps->movementDir == 2 ) {
345 pm->ps->movementDir = 1;
346 } else if ( pm->ps->movementDir == 6 ) {
347 pm->ps->movementDir = 7;
348 }
349 }
350}

353/*
354=============
355PM_CheckJump
356=============
357*/
358static qboolean PM_CheckJump( void ) {
359 if ( pm->ps->pm_flags & PMF_RESPAWNED ) {
360 return qfalse; // don't allow jump until all buttons are up
361 }
362
363 if ( pm->cmd.upmove < 10 ) {
364 // not holding jump
365 return qfalse;
366 }
367
368 // must wait for jump to be released
369 if ( pm->ps->pm_flags & PMF_JUMP_HELD ) {
370 // clear upmove so cmdscale doesn't lower running speed
371 pm->cmd.upmove = 0;
372 return qfalse;
373 }
374
375 pml.groundPlane = qfalse; // jumping away
376 pml.walking = qfalse;
377 pm->ps->pm_flags |= PMF_JUMP_HELD;
378
379 pm->ps->groundEntityNum = ENTITYNUM_NONE;
380 pm->ps->velocity[2] = JUMP_VELOCITY;
381 PM_AddEvent( EV_JUMP );
382
383 if ( pm->cmd.forwardmove >= 0 ) {
384 PM_ForceLegsAnim( LEGS_JUMP );
385 pm->ps->pm_flags &= ~PMF_BACKWARDS_JUMP;
386 } else {
387 PM_ForceLegsAnim( LEGS_JUMPB );
388 pm->ps->pm_flags |= PMF_BACKWARDS_JUMP;
389 }
390
391 return qtrue;
392}

394/*
395=============
396PM_CheckWaterJump
397=============
398*/
399static qboolean PM_CheckWaterJump( void ) {
400 vec3_t spot;
401 int cont;
402 vec3_t flatforward;
403
404 if (pm->ps->pm_time) {
405 return qfalse;
406 }
407
408 // check for water jump
409 if ( pm->waterlevel != 2 ) {
410 return qfalse;
411 }
412
413 flatforward[0] = pml.forward[0];
414 flatforward[1] = pml.forward[1];
415 flatforward[2] = 0;
416 VectorNormalize (flatforward);
417
418 VectorMA (pm->ps->origin, 30, flatforward, spot);
419 spot[2] += 4;
420 cont = pm->pointcontents (spot, pm->ps->clientNum );
421 if ( !(cont & CONTENTS_SOLID) ) {
422 return qfalse;
423 }
424
425 spot[2] += 16;
426 cont = pm->pointcontents (spot, pm->ps->clientNum );
427 if ( cont & (CONTENTS_SOLID|CONTENTS_PLAYERCLIP|CONTENTS_BODY) ) {
428 return qfalse;
429 }
430
431 // jump out of water
432 VectorScale (pml.forward, 200, pm->ps->velocity);
433 pm->ps->velocity[2] = 350;
434
435 pm->ps->pm_flags |= PMF_TIME_WATERJUMP;
436 pm->ps->pm_time = 2000;
437
438 return qtrue;
439}

464/*
465===================
466PM_WaterMove
467
468===================
469*/
470static void PM_WaterMove( void ) {
471 int i;
472 vec3_t wishvel;
473 float wishspeed;
474 vec3_t wishdir;
475 float scale;
476 float vel;
477
478 if ( PM_CheckWaterJump() ) {
479 PM_WaterJumpMove();
480 return;
481 }
482#if 0
483 // jump = head for surface
484 if ( pm->cmd.upmove >= 10 ) {
485 if (pm->ps->velocity[2] > -300) {
486 if ( pm->watertype & CONTENTS_WATER ) {
487 pm->ps->velocity[2] = 100;
488 } else if ( pm->watertype & CONTENTS_SLIME ) {
489 pm->ps->velocity[2] = 80;
490 } else {
491 pm->ps->velocity[2] = 50;
492 }
493 }
494 }
495#endif
496 PM_Friction ();
497
498 scale = PM_CmdScale( &pm->cmd );
499 //
500 // user intentions
501 //
502 if ( !scale ) {
503 wishvel[0] = 0;
504 wishvel[1] = 0;
505 wishvel[2] = -60; // sink towards bottom
506 } else {
507 for (i=0 ; i<3 ; i++)
508 wishvel[i] = scale * pml.forward[i]*pm->cmd.forwardmove + scale * pml.right[i]*pm->cmd.rightmove;
509
510 wishvel[2] += scale * pm->cmd.upmove;
511 }
512
513 VectorCopy (wishvel, wishdir);
514 wishspeed = VectorNormalize(wishdir);
515
516 if ( wishspeed > pm->ps->speed * pm_swimScale ) {
517 wishspeed = pm->ps->speed * pm_swimScale;
518 }
519
520 PM_Accelerate (wishdir, wishspeed, pm_wateraccelerate);
521
522 // make sure we can go up slopes easily under water
523 if ( pml.groundPlane && DotProduct( pm->ps->velocity, pml.groundTrace.plane.normal ) < 0 ) {
524 vel = VectorLength(pm->ps->velocity);
525 // slide along the ground plane
526 PM_ClipVelocity (pm->ps->velocity, pml.groundTrace.plane.normal,
527 pm->ps->velocity, OVERCLIP );
528
529 VectorNormalize(pm->ps->velocity);
530 VectorScale(pm->ps->velocity, vel, pm->ps->velocity);
531 }
532
533 PM_SlideMove( qfalse );
534}

594/*
595===================
596PM_AirMove
597
598===================
599*/
600static void PM_AirMove( void ) {
601 int i;
602 vec3_t wishvel;
603 float fmove, smove;
604 vec3_t wishdir;
605 float wishspeed;
606 float scale;
607 usercmd_t cmd;
608
609 PM_Friction();
610
611 fmove = pm->cmd.forwardmove;
612 smove = pm->cmd.rightmove;
613
614 cmd = pm->cmd;
615 scale = PM_CmdScale( &cmd );
616
617 // set the movementDir so clients can rotate the legs for strafing
618 PM_SetMovementDir();
619
620 // project moves down to flat plane
621 pml.forward[2] = 0;
622 pml.right[2] = 0;
623 VectorNormalize (pml.forward);
624 VectorNormalize (pml.right);
625
626 for ( i = 0 ; i < 2 ; i++ ) {
627 wishvel[i] = pml.forward[i]*fmove + pml.right[i]*smove;
628 }
629 wishvel[2] = 0;
630
631 VectorCopy (wishvel, wishdir);
632 wishspeed = VectorNormalize(wishdir);
633 wishspeed *= scale;
634
635 // not on ground, so little effect on velocity
636 PM_Accelerate (wishdir, wishspeed, pm_airaccelerate);
637
638 // we may have a ground plane that is very steep, even
639 // though we don't have a groundentity
640 // slide along the steep plane
641 if ( pml.groundPlane ) {
642 PM_ClipVelocity (pm->ps->velocity, pml.groundTrace.plane.normal,
643 pm->ps->velocity, OVERCLIP );
644 }
645
646#if 0
647 //ZOID: If we are on the grapple, try stair-stepping
648 //this allows a player to use the grapple to pull himself
649 //over a ledge
650 if (pm->ps->pm_flags & PMF_GRAPPLE_PULL)
651 PM_StepSlideMove ( qtrue );
652 else
653 PM_SlideMove ( qtrue );
654#endif
655
656 PM_StepSlideMove ( qtrue );
657}

685/*
686===================
687PM_WalkMove
688
689===================
690*/
691static void PM_WalkMove( void ) {
692 int i;
693 vec3_t wishvel;
694 float fmove, smove;
695 vec3_t wishdir;
696 float wishspeed;
697 float scale;
698 usercmd_t cmd;
699 float accelerate;
700 float vel;
701
702 if ( pm->waterlevel > 2 && DotProduct( pml.forward, pml.groundTrace.plane.normal ) > 0 ) {
703 // begin swimming
704 PM_WaterMove();
705 return;
706 }
707
708
709 if ( PM_CheckJump () ) {
710 // jumped away
711 if ( pm->waterlevel > 1 ) {
712 PM_WaterMove();
713 } else {
714 PM_AirMove();
715 }
716 return;
717 }
718
719 PM_Friction ();
720
721 fmove = pm->cmd.forwardmove;
722 smove = pm->cmd.rightmove;
723
724 cmd = pm->cmd;
725 scale = PM_CmdScale( &cmd );
726
727 // set the movementDir so clients can rotate the legs for strafing
728 PM_SetMovementDir();
729
730 // project moves down to flat plane
731 pml.forward[2] = 0;
732 pml.right[2] = 0;
733
734 // project the forward and right directions onto the ground plane
735 PM_ClipVelocity (pml.forward, pml.groundTrace.plane.normal, pml.forward, OVERCLIP );
736 PM_ClipVelocity (pml.right, pml.groundTrace.plane.normal, pml.right, OVERCLIP );
737 //
738 VectorNormalize (pml.forward);
739 VectorNormalize (pml.right);
740
741 for ( i = 0 ; i < 3 ; i++ ) {
742 wishvel[i] = pml.forward[i]*fmove + pml.right[i]*smove;
743 }
744 // when going up or down slopes the wish velocity should Not be zero
745// wishvel[2] = 0;
746
747 VectorCopy (wishvel, wishdir);
748 wishspeed = VectorNormalize(wishdir);
749 wishspeed *= scale;
750
751 // clamp the speed lower if ducking
752 if ( pm->ps->pm_flags & PMF_DUCKED ) {
753 if ( wishspeed > pm->ps->speed * pm_duckScale ) {
754 wishspeed = pm->ps->speed * pm_duckScale;
755 }
756 }
757
758 // clamp the speed lower if wading or walking on the bottom
759 if ( pm->waterlevel ) {
760 float waterScale;
761
762 waterScale = pm->waterlevel / 3.0;
763 waterScale = 1.0 - ( 1.0 - pm_swimScale ) * waterScale;
764 if ( wishspeed > pm->ps->speed * waterScale ) {
765 wishspeed = pm->ps->speed * waterScale;
766 }
767 }
768
769 // when a player gets hit, they temporarily lose
770 // full control, which allows them to be moved a bit
771 if ( ( pml.groundTrace.surfaceFlags & SURF_SLICK ) || pm->ps->pm_flags & PMF_TIME_KNOCKBACK ) {
772 accelerate = pm_airaccelerate;
773 } else {
774 accelerate = pm_accelerate;
775 }
776
777 PM_Accelerate (wishdir, wishspeed, accelerate);
778
779 //Com_Printf("velocity = %1.1f %1.1f %1.1f\n", pm->ps->velocity[0], pm->ps->velocity[1], pm->ps->velocity[2]);
780 //Com_Printf("velocity1 = %1.1f\n", VectorLength(pm->ps->velocity));
781
782 if ( ( pml.groundTrace.surfaceFlags & SURF_SLICK ) || pm->ps->pm_flags & PMF_TIME_KNOCKBACK ) {
783 pm->ps->velocity[2] -= pm->ps->gravity * pml.frametime;
784 } else {
785 // don't reset the z velocity for slopes
786// pm->ps->velocity[2] = 0;
787 }
788
789 vel = VectorLength(pm->ps->velocity);
790
791 // slide along the ground plane
792 PM_ClipVelocity (pm->ps->velocity, pml.groundTrace.plane.normal,
793 pm->ps->velocity, OVERCLIP );
794
795 // don't decrease velocity when going up or down a slope
796 VectorNormalize(pm->ps->velocity);
797 VectorScale(pm->ps->velocity, vel, pm->ps->velocity);
798
799 // don't do anything if standing still
800 if (!pm->ps->velocity[0] && !pm->ps->velocity[1]) {
801 return;
802 }
803
804 PM_StepSlideMove( qfalse );
805
806 //Com_Printf("velocity2 = %1.1f\n", VectorLength(pm->ps->velocity));
807
808}

840*/
841static void PM_NoclipMove( void ) {
842 float speed, drop, friction, control, newspeed;
843 int i;
844 vec3_t wishvel;
845 float fmove, smove;
846 vec3_t wishdir;
847 float wishspeed;
848 float scale;
849
850 pm->ps->viewheight = DEFAULT_VIEWHEIGHT;
851
852 // friction
853
854 speed = VectorLength (pm->ps->velocity);
855 if (speed < 1)
856 {
857 VectorCopy (vec3_origin, pm->ps->velocity);
858 }
859 else
860 {
861 drop = 0;
862
863 friction = pm_friction*1.5; // extra friction
864 control = speed < pm_stopspeed ? pm_stopspeed : speed;
865 drop += control*friction*pml.frametime;
866
867 // scale the velocity
868 newspeed = speed - drop;
869 if (newspeed < 0)
870 newspeed = 0;
871 newspeed /= speed;
872
873 VectorScale (pm->ps->velocity, newspeed, pm->ps->velocity);
874 }
875
876 // accelerate
877 scale = PM_CmdScale( &pm->cmd );
878
879 fmove = pm->cmd.forwardmove;
880 smove = pm->cmd.rightmove;
881
882 for (i=0 ; i<3 ; i++)
883 wishvel[i] = pml.forward[i]*fmove + pml.right[i]*smove;
884 wishvel[2] += pm->cmd.upmove;
885
886 VectorCopy (wishvel, wishdir);
887 wishspeed = VectorNormalize(wishdir);
888 wishspeed *= scale;
889
890 PM_Accelerate( wishdir, wishspeed, pm_accelerate );
891
892 // move
893 VectorMA (pm->ps->origin, pml.frametime, pm->ps->velocity, pm->ps->origin);
894}

916/*
917=================
918PM_CrashLand
919
920Check for hard landings that generate sound events
921=================
922*/
923static void PM_CrashLand( void ) {
924 float delta;
925 float dist;
926 float vel, acc;
927 float t;
928 float a, b, c, den;
929
930 // decide which landing animation to use
931 if ( pm->ps->pm_flags & PMF_BACKWARDS_JUMP ) {
932 PM_ForceLegsAnim( LEGS_LANDB );
933 } else {
934 PM_ForceLegsAnim( LEGS_LAND );
935 }
936
937 pm->ps->legsTimer = TIMER_LAND;
938
939 // calculate the exact velocity on landing
940 dist = pm->ps->origin[2] - pml.previous_origin[2];
941 vel = pml.previous_velocity[2];
942 acc = -pm->ps->gravity;
943
944 a = acc / 2;
945 b = vel;
946 c = -dist;
947
948 den = b * b - 4 * a * c;
949 if ( den < 0 ) {
950 return;
951 }
952 t = (-b - sqrt( den ) ) / ( 2 * a );
953
954 delta = vel + t * acc;
955 delta = delta*delta * 0.0001;
956
957 // ducking while falling doubles damage
958 if ( pm->ps->pm_flags & PMF_DUCKED ) {
959 delta *= 2;
960 }
961
962 // never take falling damage if completely underwater
963 if ( pm->waterlevel == 3 ) {
964 return;
965 }
966
967 // reduce falling damage if there is standing water
968 if ( pm->waterlevel == 2 ) {
969 delta *= 0.25;
970 }
971 if ( pm->waterlevel == 1 ) {
972 delta *= 0.5;
973 }
974
975 if ( delta < 1 ) {
976 return;
977 }
978
979 // create a local entity event to play the sound
980
981 // SURF_NODAMAGE is used for bounce pads where you don't ever
982 // want to take damage or play a crunch sound
983 if ( !(pml.groundTrace.surfaceFlags & SURF_NODAMAGE) ) {
984 if ( delta > 60 ) {
985 PM_AddEvent( EV_FALL_FAR );
986 } else if ( delta > 40 ) {
987 // this is a pain grunt, so don't play it if dead
988 if ( pm->ps->stats[STAT_HEALTH] > 0 ) {
989 PM_AddEvent( EV_FALL_MEDIUM );
990 }
991 } else if ( delta > 7 ) {
992 PM_AddEvent( EV_FALL_SHORT );
993 } else {
994 PM_AddEvent( PM_FootstepForSurface() );
995 }
996 }
997
998 // start footstep cycle over
999 pm->ps->bobCycle = 0;
1000}

1796void PM_UpdateViewAngles( playerState_t *ps, const usercmd_t *cmd ) {
1797 short temp;
1798 int i;
1799
1800 if ( ps->pm_type == PM_INTERMISSION || ps->pm_type == PM_SPINTERMISSION) {
1801 return; // no view changes at all
1802 }
1803
1804 if ( ps->pm_type != PM_SPECTATOR && ps->stats[STAT_HEALTH] <= 0 ) {
1805 return; // no view changes at all
1806 }
1807
1808 // circularly clamp the angles with deltas
1809 for (i=0 ; i<3 ; i++) {
1810 temp = cmd->angles[i] + ps->delta_angles[i];
1811 if ( i == PITCH ) {
1812 // don't let the player look up or down more than 90 degrees
1813 if ( temp > 16000 ) {
1814 ps->delta_angles[i] = 16000 - cmd->angles[i];
1815 temp = 16000;
1816 } else if ( temp < -16000 ) {
1817 ps->delta_angles[i] = -16000 - cmd->angles[i];
1818 temp = -16000;
1819 }
1820 }
1821 ps->viewangles[i] = SHORT2ANGLE(temp);
1822 }
1823
1824}

2041 // dependent behavior

2043 int msec;

2046

2051 }

2054 msec = 66;
2055 }

97void QDECL BotAI_Print(int type, char *fmt, ...) {
98 char str[2048];
99 va_list ap;
100
101 va_start(ap, fmt);
102 Q_vsnprintf(str, sizeof(str), fmt, ap);
103 va_end(ap);
104
105 switch(type) {
106 case PRT_MESSAGE: {
107 G_Printf("%s", str);
108 break;
109 }
110 case PRT_WARNING: {
111 G_Printf( S_COLOR_YELLOW "Warning: %s", str );
112 break;
113 }
114 case PRT_ERROR: {
115 G_Printf( S_COLOR_RED "Error: %s", str );
116 break;
117 }
118 case PRT_FATAL: {
119 G_Printf( S_COLOR_RED "Fatal: %s", str );
120 break;
121 }
122 case PRT_EXIT: {
123 G_Error( S_COLOR_RED "Exit: %s", str );
124 break;
125 }
126 default: {
127 G_Printf( "unknown print type\n" );
128 break;
129 }
130 }
131}

563==============
564BotInterbreedBots
565==============
566*/
567void BotInterbreedBots(void) {
568 float ranks[MAX_CLIENTS];
569 int parent1, parent2, child;
570 int i;
571
572 // get rankings for all the bots
573 for (i = 0; i < MAX_CLIENTS; i++) {
574 if ( botstates[i] && botstates[i]->inuse ) {
575 ranks[i] = botstates[i]->num_kills * 2 - botstates[i]->num_deaths;
576 }
577 else {
578 ranks[i] = -1;
579 }
580 }
581
582 if (trap_GeneticParentsAndChildSelection(MAX_CLIENTS, ranks, &parent1, &parent2, &child)) {
583 trap_BotInterbreedGoalFuzzyLogic(botstates[parent1]->gs, botstates[parent2]->gs, botstates[child]->gs);
584 trap_BotMutateGoalFuzzyLogic(botstates[child]->gs, 1);
585 }
586 // reset the kills and deaths
587 for (i = 0; i < MAX_CLIENTS; i++) {
588 if (botstates[i] && botstates[i]->inuse) {
589 botstates[i]->num_kills = 0;
590 botstates[i]->num_deaths = 0;
591 }
592 }
593}

718*/
719float AngleDifference(float ang1, float ang2) {
720 float diff;
721
722 diff = ang1 - ang2;
723 if (ang1 > ang2) {
724 if (diff > 180.0) diff -= 360.0;
725 }
726 else {
727 if (diff < -180.0) diff += 360.0;
728 }
729 return diff;
730}

732/*
733==============
734BotChangeViewAngle
735==============
736*/
737float BotChangeViewAngle(float angle, float ideal_angle, float speed) {
738 float move;
739
740 angle = AngleMod(angle);
741 ideal_angle = AngleMod(ideal_angle);
742 if (angle == ideal_angle) return angle;
743 move = ideal_angle - angle;
744 if (ideal_angle > angle) {
745 if (move > 180.0) move -= 360.0;
746 }
747 else {
748 if (move < -180.0) move += 360.0;
749 }
750 if (move > 0) {
751 if (move > speed) move = speed;
752 }
753 else {
754 if (move < -speed) move = -speed;
755 }
756 return AngleMod(angle + move);
757}

842 if (bi->actionflags & ACTION_WALK) ucmd->buttons |= BUTTON_WALKING;

845 if (bi->actionflags & ACTION_GETFLAG) ucmd->buttons |= BUTTON_GETFLAG;
846 if (bi->actionflags & ACTION_GUARDBASE) ucmd->buttons |= BUTTON_GUARDBASE;
847 if (bi->actionflags & ACTION_PATROL) ucmd->buttons |= BUTTON_PATROL;
848 if (bi->actionflags & ACTION_FOLLOWME) ucmd->buttons |= BUTTON_FOLLOWME;

850 ucmd->weapon = bi->weapon;
851 //set the view angles
852 //NOTE: the ucmd->angles are the angles WITHOUT the delta angles
853 ucmd->angles[PITCH] = ANGLE2SHORT(bi->viewangles[PITCH]);
854 ucmd->angles[YAW] = ANGLE2SHORT(bi->viewangles[YAW]);
855 ucmd->angles[ROLL] = ANGLE2SHORT(bi->viewangles[ROLL]);
856 //subtract the delta angles
857 for (j = 0; j < 3; j++) {
858 temp = ucmd->angles[j] - delta_angles[j];
859 /*NOTE: disabled because temp should be mod first
860 if ( j == PITCH ) {
861 // don't let the player look up or down more than 90 degrees
862 if ( temp > 16000 ) temp = 16000;
863 else if ( temp < -16000 ) temp = -16000;
864 }
865 */
866 ucmd->angles[j] = temp;
867 }

869 //get the horizontal forward and right vector
870 //get the pitch in the range [-180, 180]
871 if (bi->dir[2]) angles[PITCH] = bi->viewangles[PITCH];

878 //set the view independent movement

880 r = DotProduct(right, bi->dir);
881 u = fabs(forward[2]) * bi->dir[2];

883
884 if (fabs(r) > m) {
885 m = fabs(r);
886 }
887
888 if (fabs(u) > m) {
889 m = fabs(u);
890 }
891
892 if (m > 0) {
893 f *= bi->speed / m;
894 r *= bi->speed / m;
895 u *= bi->speed / m;
896 }
897
898 ucmd->forwardmove = f;
899 ucmd->rightmove = r;
900 ucmd->upmove = u;
901
902 if (bi->actionflags & ACTION_MOVEFORWARD) ucmd->forwardmove = 127;
903 if (bi->actionflags & ACTION_MOVEBACK) ucmd->forwardmove = -127;
904 if (bi->actionflags & ACTION_MOVELEFT) ucmd->rightmove = -127;
905 if (bi->actionflags & ACTION_MOVERIGHT) ucmd->rightmove = 127;
906 //jump/moveup
907 if (bi->actionflags & ACTION_JUMP) ucmd->upmove = 127;
908 //crouch/movedown
909 if (bi->actionflags & ACTION_CROUCH) ucmd->upmove = -127;
910}

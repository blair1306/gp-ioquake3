194/*
195==================
196BotGetTime
197==================
198*/
199float BotGetTime(bot_match_t *match) {
200 bot_match_t timematch;
201 char timestring[MAX_MESSAGE_SIZE];
202 float t;
203
204 //if the matched string has a time
205 if (match->subtype & ST_TIME) {
206 //get the time string
207 trap_BotMatchVariable(match, TIME, timestring, MAX_MESSAGE_SIZE);
208 //match it to find out if the time is in seconds or minutes
209 if (trap_BotFindMatch(timestring, &timematch, MTCONTEXT_TIME)) {
210 if (timematch.type == MSG_FOREVER) {
211 t = 99999999.0f;
212 }
213 else if (timematch.type == MSG_FORAWHILE) {
214 t = 10 * 60; // 10 minutes
215 }
216 else if (timematch.type == MSG_FORALONGTIME) {
217 t = 30 * 60; // 30 minutes
218 }
219 else {
220 trap_BotMatchVariable(&timematch, TIME, timestring, MAX_MESSAGE_SIZE);
221 if (timematch.type == MSG_MINUTES) t = atof(timestring) * 60;
222 else if (timematch.type == MSG_SECONDS) t = atof(timestring);
223 else t = 0;
224 }
225 //if there's a valid time
226 if (t > 0) return FloatTime() + t;
227 }
228 }
229 return 0;
230}

389 //if the message is addressed to someone
390 if (match->subtype & ST_ADDRESSED) {

392 //the name of this bot

394 //

396 if (addresseematch.type == MSG_EVERYONE) {
397 return qtrue;
398 }

400 trap_BotMatchVariable(&addresseematch, TEAMMATE, name, sizeof(name));
401 if (strlen(name)) {
402 if (stristr(botname, name)) return qtrue;
403 if (stristr(bs->subteam, name)) return qtrue;
404 }
405 trap_BotMatchVariable(&addresseematch, MORE, addressedto, MAX_MESSAGE_SIZE);
406 }
407 else {
408 trap_BotMatchVariable(&addresseematch, TEAMMATE, name, MAX_MESSAGE_SIZE);
409 if (strlen(name)) {
410 if (stristr(botname, name)) return qtrue;
411 if (stristr(bs->subteam, name)) return qtrue;
412 }
413 break;
414 }
415 }
416 //Com_sprintf(buf, sizeof(buf), "not addressed to me but %s", addressedto);
417 //trap_EA_Say(bs->client, buf);
418 return qfalse;
419 }
420 else {
421 bot_match_t tellmatch;
422
423 tellmatch.type = 0;
424 //if this message wasn't directed solely to this bot
425 if (!trap_BotFindMatch(match->string, &tellmatch, MTCONTEXT_REPLYCHAT) ||
426 tellmatch.type != MSG_CHATTELL) {
427 //make sure not everyone reacts to this message
428 if (random() > (float ) 1.0 / (NumPlayersOnSameTeam(bs)-1)) return qfalse;
429 }
430 }
431 return qtrue;

435==================
436BotGPSToPosition
437==================
438*/
439int BotGPSToPosition(char *buf, vec3_t position) {
440 int i, j = 0;
441 int num, sign;
442
443 for (i = 0; i < 3; i++) {
444 num = 0;
445 while(buf[j] == ' ') j++;
446 if (buf[j] == '-') {
447 j++;
448 sign = -1;
449 }
450 else {
451 sign = 1;
452 }
453 while (buf[j]) {
454 if (buf[j] >= '0' && buf[j] <= '9') {
455 num = num * 10 + buf[j] - '0';
456 j++;
457 }
458 else {
459 j++;
460 break;
461 }
462 }
463 BotAI_Print(PRT_MESSAGE, "%d\n", sign * num);
464 position[i] = (float) sign * num;
465 }
466 return qtrue;
467}

474void BotMatch_HelpAccompany(bot_state_t *bs, bot_match_t *match) {

478 char itemname[MAX_MESSAGE_SIZE];

483 //if not addressed to this bot
484 if (!BotAddressedToBot(bs, match)) return;
485 //get the team mate name
486 trap_BotMatchVariable(match, TEAMMATE, teammate, sizeof(teammate));
487 //get the client to help
488 if (trap_BotFindMatch(teammate, &teammatematch, MTCONTEXT_TEAMMATE) &&
489 //if someone asks for him or herself
490 teammatematch.type == MSG_ME) {
491 //get the netname
492 trap_BotMatchVariable(match, NETNAME, netname, sizeof(netname));
493 client = ClientFromName(netname);
494 other = qfalse;
495 }
496 else {
497 //asked for someone else
498 client = FindClientByName(teammate);
499 //if this is the bot self
500 if (client == bs->client) {
501 other = qfalse;
502 }
503 else if (!BotSameTeam(bs, client)) {
504 //FIXME: say "I don't help the enemy"
505 return;
506 }
507 else {
508 other = qtrue;
509 }
510 }
511 //if the bot doesn't know who to help (FindClientByName returned -1)
512 if (client < 0) {
513 if (other) BotAI_BotInitialChat(bs, "whois", teammate, NULL);
514 else BotAI_BotInitialChat(bs, "whois", netname, NULL);
515 client = ClientFromName(netname);
516 trap_BotEnterChat(bs->cs, client, CHAT_TELL);
517 return;
518 }
519 //don't help or accompany yourself
520 if (client == bs->client) {
521 return;
522 }
523 //
524 bs->teamgoal.entitynum = -1;
525 BotEntityInfo(client, &entinfo);
526 //if info is valid (in PVS)
527 if (entinfo.valid) {
528 areanum = BotPointAreaNum(entinfo.origin);
529 if (areanum) {// && trap_AAS_AreaReachability(areanum)) {
530 bs->teamgoal.entitynum = client;
531 bs->teamgoal.areanum = areanum;
532 VectorCopy(entinfo.origin, bs->teamgoal.origin);
533 VectorSet(bs->teamgoal.mins, -8, -8, -8);
534 VectorSet(bs->teamgoal.maxs, 8, 8, 8);
535 }
536 }
537 //if no teamgoal yet
538 if (bs->teamgoal.entitynum < 0) {
539 //if near an item
540 if (match->subtype & ST_NEARITEM) {
541 //get the match variable
542 trap_BotMatchVariable(match, ITEM, itemname, sizeof(itemname));
543 //
544 if (!BotGetMessageTeamGoal(bs, itemname, &bs->teamgoal)) {
545 //BotAI_BotInitialChat(bs, "cannotfind", itemname, NULL);
546 //trap_BotEnterChat(bs->cs, bs->client, CHAT_TEAM);
547 return;
548 }
549 }
550 }
551 //
552 if (bs->teamgoal.entitynum < 0) {
553 if (other) BotAI_BotInitialChat(bs, "whereis", teammate, NULL);
554 else BotAI_BotInitialChat(bs, "whereareyou", netname, NULL);
555 client = ClientFromName(netname);
556 trap_BotEnterChat(bs->cs, client, CHAT_TEAM);
557 return;
558 }
559 //the team mate
560 bs->teammate = client;
561 //
562 trap_BotMatchVariable(match, NETNAME, netname, sizeof(netname));
563 //
564 client = ClientFromName(netname);
565 //the team mate who ordered
566 bs->decisionmaker = client;
567 bs->ordered = qtrue;
568 bs->order_time = FloatTime();
569 //last time the team mate was assumed visible
570 bs->teammatevisible_time = FloatTime();
571 //set the time to send a message to the team mates
572 bs->teammessage_time = FloatTime() + 2 * random();
573 //get the team goal time
574 bs->teamgoal_time = BotGetTime(match);
575 //set the ltg type
576 if (match->type == MSG_HELP) {
577 bs->ltgtype = LTG_TEAMHELP;
578 if (!bs->teamgoal_time) bs->teamgoal_time = FloatTime() + TEAM_HELP_TIME;
579 }
580 else {
581 bs->ltgtype = LTG_TEAMACCOMPANY;
582 if (!bs->teamgoal_time) bs->teamgoal_time = FloatTime() + TEAM_ACCOMPANY_TIME;
583 bs->formation_dist = 3.5 * 32; //3.5 meter
584 bs->arrive_time = 0;
585 //
586 BotSetTeamStatus(bs);
587 // remember last ordered task
588 BotRememberLastOrderedTask(bs);
589 }
590#ifdef DEBUG
591 BotPrintTeamGoal(bs);
592#endif //DEBUG
593}

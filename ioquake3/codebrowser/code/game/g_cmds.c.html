29/*
30==================
31DeathmatchScoreboardMessage
32
33==================
34*/
35void DeathmatchScoreboardMessage( gentity_t *ent ) {
36 char entry[1024];
37 char string[1000];
38 int stringlength;
39 int i, j;
40 gclient_t *cl;
41 int numSorted, scoreFlags, accuracy, perfect;
42
43 // don't send scores to bots, they don't parse it
44 if ( ent->r.svFlags & SVF_BOT ) {
45 return;
46 }
47
48 // send the latest information on all clients
49 string[0] = 0;
50 stringlength = 0;
51 scoreFlags = 0;
52
53 numSorted = level.numConnectedClients;
54
55 for (i=0 ; i < numSorted ; i++) {
56 int ping;
57
58 cl = &level.clients[level.sortedClients[i]];
59
60 if ( cl->pers.connected == CON_CONNECTING ) {
61 ping = -1;
62 } else {
63 ping = cl->ps.ping < 999 ? cl->ps.ping : 999;
64 }
65
66 if( cl->accuracy_shots ) {
67 accuracy = cl->accuracy_hits * 100 / cl->accuracy_shots;
68 }
69 else {
70 accuracy = 0;
71 }
72 perfect = ( cl->ps.persistant[PERS_RANK] == 0 && cl->ps.persistant[PERS_KILLED] == 0 ) ? 1 : 0;
73
74 Com_sprintf (entry, sizeof(entry),
75 " %i %i %i %i %i %i %i %i %i %i %i %i %i %i", level.sortedClients[i],
76 cl->ps.persistant[PERS_SCORE], ping, (level.time - cl->pers.enterTime)/60000,
77 scoreFlags, g_entities[level.sortedClients[i]].s.powerups, accuracy,
78 cl->ps.persistant[PERS_IMPRESSIVE_COUNT],
79 cl->ps.persistant[PERS_EXCELLENT_COUNT],
80 cl->ps.persistant[PERS_GAUNTLET_FRAG_COUNT],
81 cl->ps.persistant[PERS_DEFEND_COUNT],
82 cl->ps.persistant[PERS_ASSIST_COUNT],
83 perfect,
84 cl->ps.persistant[PERS_CAPTURES]);
85 j = strlen(entry);
86 if (stringlength + j >= sizeof(string))
87 break;
88 strcpy (string + stringlength, entry);
89 stringlength += j;
90 }
91
92 trap_SendServerCommand( ent-g_entities, va("scores %i %i %i%s", i,
93 level.teamScores[TEAM_RED], level.teamScores[TEAM_BLUE],
94 string ) );
95}

129/*
130==================
131ConcatArgs
132==================
133*/
134char *ConcatArgs( int start ) {
135 int i, c, tlen;
136 static char line[MAX_STRING_CHARS];
137 int len;
138 char arg[MAX_STRING_CHARS];
139
140 len = 0;
141 c = trap_Argc();
142 for ( i = start ; i < c ; i++ ) {
143 trap_Argv( i, arg, sizeof( arg ) );
144 tlen = strlen( arg );
145 if ( len + tlen >= MAX_STRING_CHARS - 1 ) {
146 break;
147 }
148 memcpy( line + len, arg, tlen );
149 len += tlen;
150 if ( i != c - 1 ) {
151 line[len] = ' ';
152 len++;
153 }
154 }
155
156 line[len] = 0;
157
158 return line;
159}

162/*
163==================
164StringIsInteger
165==================
166*/
167qboolean StringIsInteger( const char * s ) {
168 int i;
169 int len;
170 qboolean foundDigit;
171
172 len = strlen( s );
173 foundDigit = qfalse;
174
175 for ( i=0 ; i < len ; i++ ) {
176 if ( !isdigit( s[i] ) ) {
177 return qfalse;
178 }
179
180 foundDigit = qtrue;
181 }
182
183 return foundDigit;
184}

187/*
188==================
189ClientNumberFromString
190
191Returns a player number for either a number or name string
192Returns -1 if invalid
193==================
194*/
195int ClientNumberFromString( gentity_t *to, char *s, qboolean checkNums, qboolean checkNames ) {
196 gclient_t *cl;
197 int idnum;
198 char cleanName[MAX_STRING_CHARS];
199
200 if ( checkNums ) {
201 // numeric values could be slot numbers
202 if ( StringIsInteger( s ) ) {
203 idnum = atoi( s );
204 if ( idnum >= 0 && idnum < level.maxclients ) {
205 cl = &level.clients[idnum];
206 if ( cl->pers.connected == CON_CONNECTED ) {
207 return idnum;
208 }
209 }
210 }
211 }
212
213 if ( checkNames ) {
214 // check for a name match
215 for ( idnum=0,cl=level.clients ; idnum < level.maxclients ; idnum++,cl++ ) {
216 if ( cl->pers.connected != CON_CONNECTED ) {
217 continue;
218 }
219 Q_strncpyz(cleanName, cl->pers.netname, sizeof(cleanName));
220 Q_CleanStr(cleanName);
221 if ( !Q_stricmp( cleanName, s ) ) {
222 return idnum;
223 }
224 }
225 }
226
227 trap_SendServerCommand( to-g_entities, va("print \"User %s is not on the server\n\"", s));
228 return -1;
229}

231/*
232==================
233Cmd_Give_f
234
235Give items to a client
236==================
237*/
238void Cmd_Give_f (gentity_t *ent)
239{
240 char *name;
241 gitem_t *it;
242 int i;
243 qboolean give_all;
244 gentity_t *it_ent;
245 trace_t trace;
246
247 if ( !CheatsOk( ent ) ) {
248 return;
249 }
250
251 name = ConcatArgs( 1 );
252
253 if (Q_stricmp(name, "all") == 0)
254 give_all = qtrue;
255 else
256 give_all = qfalse;
257
258 if (give_all || Q_stricmp( name, "health") == 0)
259 {
260 ent->health = ent->client->ps.stats[STAT_MAX_HEALTH];
261 if (!give_all)
262 return;
263 }
264
265 if (give_all || Q_stricmp(name, "weapons") == 0)
266 {
267 ent->client->ps.stats[STAT_WEAPONS] = (1 << WP_NUM_WEAPONS) - 1 -
268 ( 1 << WP_GRAPPLING_HOOK ) - ( 1 << WP_NONE );
269 if (!give_all)
270 return;
271 }
272
273 if (give_all || Q_stricmp(name, "ammo") == 0)
274 {
275 for ( i = 0 ; i < MAX_WEAPONS ; i++ ) {
276 ent->client->ps.ammo[i] = 999;
277 }
278 if (!give_all)
279 return;
280 }
281
282 if (give_all || Q_stricmp(name, "armor") == 0)
283 {
284 ent->client->ps.stats[STAT_ARMOR] = 200;
285
286 if (!give_all)
287 return;
288 }
289
290 if (Q_stricmp(name, "excellent") == 0) {
291 ent->client->ps.persistant[PERS_EXCELLENT_COUNT]++;
292 return;
293 }
294 if (Q_stricmp(name, "impressive") == 0) {
295 ent->client->ps.persistant[PERS_IMPRESSIVE_COUNT]++;
296 return;
297 }
298 if (Q_stricmp(name, "gauntletaward") == 0) {
299 ent->client->ps.persistant[PERS_GAUNTLET_FRAG_COUNT]++;
300 return;
301 }
302 if (Q_stricmp(name, "defend") == 0) {
303 ent->client->ps.persistant[PERS_DEFEND_COUNT]++;
304 return;
305 }
306 if (Q_stricmp(name, "assist") == 0) {
307 ent->client->ps.persistant[PERS_ASSIST_COUNT]++;
308 return;
309 }
310
311 // spawn a specific item right on the player
312 if ( !give_all ) {
313 it = BG_FindItem (name);
314 if (!it) {
315 return;
316 }
317
318 it_ent = G_Spawn();
319 VectorCopy( ent->r.currentOrigin, it_ent->s.origin );
320 it_ent->classname = it->classname;
321 G_SpawnItem (it_ent, it);
322 FinishSpawningItem(it_ent );
323 memset( &trace, 0, sizeof( trace ) );
324 Touch_Item (it_ent, ent, &trace);
325 if (it_ent->inuse) {
326 G_FreeEntity( it_ent );
327 }
328 }

510/*
511=================
512SetTeam
513=================
514*/
515void SetTeam( gentity_t *ent, const char *s ) {
516 int team, oldTeam;
517 gclient_t *client;
518 int clientNum;
519 spectatorState_t specState;
520 int specClient;
521 int teamLeader;
522
523 //
524 // see what change is requested
525 //
526 client = ent->client;
527
528 clientNum = client - level.clients;
529 specClient = 0;
530 specState = SPECTATOR_NOT;
531 if ( !Q_stricmp( s, "scoreboard" ) || !Q_stricmp( s, "score" ) ) {
532 team = TEAM_SPECTATOR;
533 specState = SPECTATOR_SCOREBOARD;
534 } else if ( !Q_stricmp( s, "follow1" ) ) {
535 team = TEAM_SPECTATOR;
536 specState = SPECTATOR_FOLLOW;
537 specClient = -1;
538 } else if ( !Q_stricmp( s, "follow2" ) ) {
539 team = TEAM_SPECTATOR;
540 specState = SPECTATOR_FOLLOW;
541 specClient = -2;
542 } else if ( !Q_stricmp( s, "spectator" ) || !Q_stricmp( s, "s" ) ) {
543 team = TEAM_SPECTATOR;
544 specState = SPECTATOR_FREE;
545 } else if ( g_gametype.integer >= GT_TEAM ) {
546 // if running a team game, assign player to one of the teams
547 specState = SPECTATOR_NOT;
548 if ( !Q_stricmp( s, "red" ) || !Q_stricmp( s, "r" ) ) {
549 team = TEAM_RED;
550 } else if ( !Q_stricmp( s, "blue" ) || !Q_stricmp( s, "b" ) ) {
551 team = TEAM_BLUE;
552 } else {
553 // pick the team with the least number of players
554 team = PickTeam( clientNum );
555 }
556
557 if ( g_teamForceBalance.integer && !client->pers.localClient && !( ent->r.svFlags & SVF_BOT ) ) {
558 int counts[TEAM_NUM_TEAMS];
559
560 counts[TEAM_BLUE] = TeamCount( clientNum, TEAM_BLUE );
561 counts[TEAM_RED] = TeamCount( clientNum, TEAM_RED );
562
563 // We allow a spread of two
564 if ( team == TEAM_RED && counts[TEAM_RED] - counts[TEAM_BLUE] > 1 ) {
565 trap_SendServerCommand( clientNum,
566 "cp \"Red team has too many players.\n\"" );
567 return; // ignore the request
568 }
569 if ( team == TEAM_BLUE && counts[TEAM_BLUE] - counts[TEAM_RED] > 1 ) {
570 trap_SendServerCommand( clientNum,
571 "cp \"Blue team has too many players.\n\"" );
572 return; // ignore the request
573 }
574
575 // It's ok, the team we are switching to has less or same number of players
576 }
577
578 } else {
579 // force them to spectators if there aren't any spots free
580 team = TEAM_FREE;
581 }
582
583 // override decision if limiting the players
584 if ( (g_gametype.integer == GT_TOURNAMENT)
585 && level.numNonSpectatorClients >= 2 ) {
586 team = TEAM_SPECTATOR;
587 } else if ( g_maxGameClients.integer > 0 &&
588 level.numNonSpectatorClients >= g_maxGameClients.integer ) {
589 team = TEAM_SPECTATOR;
590 }
591
592 //
593 // decide if we will allow the change
594 //
595 oldTeam = client->sess.sessionTeam;
596 if ( team == oldTeam && team != TEAM_SPECTATOR ) {
597 return;
598 }
599
600 //
601 // execute the team change
602 //
603
604 // if the player was dead leave the body, but only if they're actually in game
605 if ( client->ps.stats[STAT_HEALTH] <= 0 && client->pers.connected == CON_CONNECTED ) {
606 CopyToBodyQue(ent);
607 }
608
609 // he starts at 'base'
610 client->pers.teamState.state = TEAM_BEGIN;
611 if ( oldTeam != TEAM_SPECTATOR ) {
612 // Kill him (makes sure he loses flags, etc)
613 ent->flags &= ~FL_GODMODE;
614 ent->client->ps.stats[STAT_HEALTH] = ent->health = 0;
615 player_die (ent, ent, ent, 100000, MOD_SUICIDE);
616
617 }
618
619 // they go to the end of the line for tournements
620 if(team == TEAM_SPECTATOR && oldTeam != team)
621 AddTournamentQueue(client);
622
623 client->sess.sessionTeam = team;
624 client->sess.spectatorState = specState;
625 client->sess.spectatorClient = specClient;
626
627 client->sess.teamLeader = qfalse;
628 if ( team == TEAM_RED || team == TEAM_BLUE ) {
629 teamLeader = TeamLeader( team );
630 // if there is no team leader or the team leader is a bot and this client is not a bot
631 if ( teamLeader == -1 || ( !(g_entities[clientNum].r.svFlags & SVF_BOT) && (g_entities[teamLeader].r.svFlags & SVF_BOT) ) ) {
632 SetLeader( team, clientNum );
633 }
634 }
635 // make sure there is a team leader on the team the player came from
636 if ( oldTeam == TEAM_RED || oldTeam == TEAM_BLUE ) {
637 CheckTeamLeader( oldTeam );
638 }
639
640 BroadcastTeamChange( client, oldTeam );
641
642 // get and distribute relevent paramters
643 ClientUserinfoChanged( clientNum );
644
645 // client hasn't spawned yet, they sent an early team command, teampref userinfo, or g_teamAutoJoin is enabled
646 if ( client->pers.connected != CON_CONNECTED ) {
647 return;
648 }
649
650 ClientBegin( clientNum );
651}

677/*
678=================
679Cmd_Team_f
680=================
681*/
682void Cmd_Team_f( gentity_t *ent ) {
683 int oldTeam;
684 char s[MAX_TOKEN_CHARS];
685
686 if ( trap_Argc() != 2 ) {
687 oldTeam = ent->client->sess.sessionTeam;
688 switch ( oldTeam ) {
689 case TEAM_BLUE:
690 trap_SendServerCommand( ent-g_entities, "print \"Blue team\n\"" );
691 break;
692 case TEAM_RED:
693 trap_SendServerCommand( ent-g_entities, "print \"Red team\n\"" );
694 break;
695 case TEAM_FREE:
696 trap_SendServerCommand( ent-g_entities, "print \"Free team\n\"" );
697 break;
698 case TEAM_SPECTATOR:
699 trap_SendServerCommand( ent-g_entities, "print \"Spectator team\n\"" );
700 break;
701 }
702 return;
703 }
704
705 if ( ent->client->switchTeamTime > level.time ) {
706 trap_SendServerCommand( ent-g_entities, "print \"May not switch teams more than once per 5 seconds.\n\"" );
707 return;
708 }
709
710 // if they are playing a tournement game, count as a loss
711 if ( (g_gametype.integer == GT_TOURNAMENT )
712 && ent->client->sess.sessionTeam == TEAM_FREE ) {
713 ent->client->sess.losses++;
714 }
715
716 trap_Argv( 1, s, sizeof( s ) );
717
718 SetTeam( ent, s );
719
720 ent->client->switchTeamTime = level.time + 5000;
721}

724/*
725=================
726Cmd_Follow_f
727=================
728*/
729void Cmd_Follow_f( gentity_t *ent ) {
730 int i;
731 char arg[MAX_TOKEN_CHARS];
732
733 if ( trap_Argc() != 2 ) {
734 if ( ent->client->sess.spectatorState == SPECTATOR_FOLLOW ) {
735 StopFollowing( ent );
736 }
737 return;
738 }
739
740 trap_Argv( 1, arg, sizeof( arg ) );
741 i = ClientNumberFromString( ent, arg, qtrue, qtrue );
742 if ( i == -1 ) {
743 return;
744 }
745
746 // can't follow self
747 if ( &level.clients[ i ] == ent->client ) {
748 return;
749 }
750
751 // can't follow another spectator
752 if ( level.clients[ i ].sess.sessionTeam == TEAM_SPECTATOR ) {
753 return;
754 }
755
756 // if they are playing a tournement game, count as a loss
757 if ( (g_gametype.integer == GT_TOURNAMENT )
758 && ent->client->sess.sessionTeam == TEAM_FREE ) {
759 ent->client->sess.losses++;
760 }
761
762 // first set them to spectator
763 if ( ent->client->sess.sessionTeam != TEAM_SPECTATOR ) {
764 SetTeam( ent, "spectator" );
765 }
766
767 ent->client->sess.spectatorState = SPECTATOR_FOLLOW;
768 ent->client->sess.spectatorClient = i;
769}

771/*
772=================
773Cmd_FollowCycle_f
774=================
775*/
776void Cmd_FollowCycle_f( gentity_t *ent, int dir ) {
777 int clientnum;
778 int original;
779
780 // if they are playing a tournement game, count as a loss
781 if ( (g_gametype.integer == GT_TOURNAMENT )
782 && ent->client->sess.sessionTeam == TEAM_FREE ) {
783 ent->client->sess.losses++;
784 }
785 // first set them to spectator
786 if ( ent->client->sess.spectatorState == SPECTATOR_NOT ) {
787 SetTeam( ent, "spectator" );
788 }
789
790 if ( dir != 1 && dir != -1 ) {
791 G_Error( "Cmd_FollowCycle_f: bad dir %i", dir );
792 }
793
794 // if dedicated follow client, just switch between the two auto clients
795 if (ent->client->sess.spectatorClient < 0) {
796 if (ent->client->sess.spectatorClient == -1) {
797 ent->client->sess.spectatorClient = -2;

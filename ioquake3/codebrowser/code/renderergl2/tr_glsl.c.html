161static void GLSL_PrintLog(GLuint programOrShader, glslPrintLog_t type, qboolean developerOnly)
162{
163 char *msg;
164 static char msgPart[1024];
165 int maxLength = 0;
166 int i;
167 int printLevel = developerOnly ? PRINT_DEVELOPER : PRINT_ALL;
168
169 switch (type)
170 {
171 case GLSL_PRINTLOG_PROGRAM_INFO:
172 ri.Printf(printLevel, "Program info log:\n");
173 qglGetProgramiv(programOrShader, GL_INFO_LOG_LENGTH, &maxLength);
174 break;
175
176 case GLSL_PRINTLOG_SHADER_INFO:
177 ri.Printf(printLevel, "Shader info log:\n");
178 qglGetShaderiv(programOrShader, GL_INFO_LOG_LENGTH, &maxLength);
179 break;
180
181 case GLSL_PRINTLOG_SHADER_SOURCE:
182 ri.Printf(printLevel, "Shader source:\n");
183 qglGetShaderiv(programOrShader, GL_SHADER_SOURCE_LENGTH, &maxLength);
184 break;
185 }
186
187 if (maxLength <= 0)
188 {
189 ri.Printf(printLevel, "None.\n");
190 return;
191 }
192
193 if (maxLength < 1023)
194 msg = msgPart;
195 else
196 msg = ri.Malloc(maxLength);
197
198 switch (type)
199 {
200 case GLSL_PRINTLOG_PROGRAM_INFO:
201 qglGetProgramInfoLog(programOrShader, maxLength, &maxLength, msg);
202 break;
203
204 case GLSL_PRINTLOG_SHADER_INFO:
205 qglGetShaderInfoLog(programOrShader, maxLength, &maxLength, msg);
206 break;
207
208 case GLSL_PRINTLOG_SHADER_SOURCE:
209 qglGetShaderSource(programOrShader, maxLength, &maxLength, msg);
210 break;
211 }
212
213 if (maxLength < 1023)
214 {
215 msgPart[maxLength + 1] = '\0';
216
217 ri.Printf(printLevel, "%s\n", msgPart);
218 }
219 else
220 {
221 for(i = 0; i < maxLength; i += 1023)
222 {
223 Q_strncpyz(msgPart, msg + i, sizeof(msgPart));
224
225 ri.Printf(printLevel, "%s", msgPart);
226 }
227
228 ri.Printf(printLevel, "\n");
229
230 ri.Free(msg);
231 }
232
233}

235static void GLSL_GetShaderHeader( GLenum shaderType, const GLchar *extra, char *dest, int size )
236{
237 float fbufWidthScale, fbufHeightScale;
238
239 dest[0] = '\0';
240
241 // HACK: abuse the GLSL preprocessor to turn GLSL 1.20 shaders into 1.30 ones
242 if(glRefConfig.glslMajorVersion > 1 || (glRefConfig.glslMajorVersion == 1 && glRefConfig.glslMinorVersion >= 30))
243 {
244 if (glRefConfig.glslMajorVersion > 1 || (glRefConfig.glslMajorVersion == 1 && glRefConfig.glslMinorVersion >= 50))
245 Q_strcat(dest, size, "#version 150\n");
246 else
247 Q_strcat(dest, size, "#version 130\n");
248
249 if(shaderType == GL_VERTEX_SHADER)
250 {
251 Q_strcat(dest, size, "#define attribute in\n");
252 Q_strcat(dest, size, "#define varying out\n");
253 }
254 else
255 {
256 Q_strcat(dest, size, "#define varying in\n");
257
258 Q_strcat(dest, size, "out vec4 out_Color;\n");
259 Q_strcat(dest, size, "#define gl_FragColor out_Color\n");
260 Q_strcat(dest, size, "#define texture2D texture\n");
261 Q_strcat(dest, size, "#define textureCubeLod textureLod\n");
262 Q_strcat(dest, size, "#define shadow2D texture\n");
263 }
264 }
265 else
266 {
267 Q_strcat(dest, size, "#version 120\n");
268 Q_strcat(dest, size, "#define shadow2D(a,b) shadow2D(a,b).r \n");
269 }
270
271 // HACK: add some macros to avoid extra uniforms and save speed and code maintenance
272 //Q_strcat(dest, size,
273 // va("#ifndef r_SpecularExponent\n#define r_SpecularExponent %f\n#endif\n", r_specularExponent->value));
274 //Q_strcat(dest, size,
275 // va("#ifndef r_SpecularScale\n#define r_SpecularScale %f\n#endif\n", r_specularScale->value));
276 //Q_strcat(dest, size,
277 // va("#ifndef r_NormalScale\n#define r_NormalScale %f\n#endif\n", r_normalScale->value));
278
279
280 Q_strcat(dest, size, "#ifndef M_PI\n#define M_PI 3.14159265358979323846\n#endif\n");
281
282 //Q_strcat(dest, size, va("#ifndef MAX_SHADOWMAPS\n#define MAX_SHADOWMAPS %i\n#endif\n", MAX_SHADOWMAPS));
283
284 Q_strcat(dest, size,
285 va("#ifndef deformGen_t\n"
286 "#define deformGen_t\n"
287 "#define DGEN_WAVE_SIN %i\n"
288 "#define DGEN_WAVE_SQUARE %i\n"
289 "#define DGEN_WAVE_TRIANGLE %i\n"
290 "#define DGEN_WAVE_SAWTOOTH %i\n"
291 "#define DGEN_WAVE_INVERSE_SAWTOOTH %i\n"
292 "#define DGEN_BULGE %i\n"
293 "#define DGEN_MOVE %i\n"
294 "#endif\n",
295 DGEN_WAVE_SIN,
296 DGEN_WAVE_SQUARE,
297 DGEN_WAVE_TRIANGLE,
298 DGEN_WAVE_SAWTOOTH,
299 DGEN_WAVE_INVERSE_SAWTOOTH,
300 DGEN_BULGE,
301 DGEN_MOVE));
302
303 Q_strcat(dest, size,
304 va("#ifndef tcGen_t\n"
305 "#define tcGen_t\n"
306 "#define TCGEN_LIGHTMAP %i\n"
307 "#define TCGEN_TEXTURE %i\n"
308 "#define TCGEN_ENVIRONMENT_MAPPED %i\n"
309 "#define TCGEN_FOG %i\n"
310 "#define TCGEN_VECTOR %i\n"
311 "#endif\n",
312 TCGEN_LIGHTMAP,
313 TCGEN_TEXTURE,
314 TCGEN_ENVIRONMENT_MAPPED,
315 TCGEN_FOG,
316 TCGEN_VECTOR));
317
318 Q_strcat(dest, size,
319 va("#ifndef colorGen_t\n"
320 "#define colorGen_t\n"
321 "#define CGEN_LIGHTING_DIFFUSE %i\n"
322 "#endif\n",
323 CGEN_LIGHTING_DIFFUSE));
324
325 Q_strcat(dest, size,
326 va("#ifndef alphaGen_t\n"
327 "#define alphaGen_t\n"
328 "#define AGEN_LIGHTING_SPECULAR %i\n"
329 "#define AGEN_PORTAL %i\n"
330 "#endif\n",
331 AGEN_LIGHTING_SPECULAR,
332 AGEN_PORTAL));
333
334 Q_strcat(dest, size,
335 va("#ifndef texenv_t\n"
336 "#define texenv_t\n"
337 "#define TEXENV_MODULATE %i\n"
338 "#define TEXENV_ADD %i\n"
339 "#define TEXENV_REPLACE %i\n"
340 "#endif\n",
341 GL_MODULATE,
342 GL_ADD,
343 GL_REPLACE));
344
345 fbufWidthScale = 1.0f / ((float)glConfig.vidWidth);
346 fbufHeightScale = 1.0f / ((float)glConfig.vidHeight);
347 Q_strcat(dest, size,
348 va("#ifndef r_FBufScale\n#define r_FBufScale vec2(%f, %f)\n#endif\n", fbufWidthScale, fbufHeightScale));
349
350 if (r_pbr->integer)
351 Q_strcat(dest, size, "#define USE_PBR\n");
352
353 if (r_cubeMapping->integer)
354 {
355 int cubeMipSize = r_cubemapSize->integer;
356 int numRoughnessMips = 0;
357
358 while (cubeMipSize)
359 {
360 cubeMipSize >>= 1;
361 numRoughnessMips++;
362 }
363 numRoughnessMips = MAX(1, numRoughnessMips - 2);
364 Q_strcat(dest, size, va("#define ROUGHNESS_MIPS float(%d)\n", numRoughnessMips));
365 }
366
367 if (extra)
368 {
369 Q_strcat(dest, size, extra);
370 }
371
372 // OK we added a lot of stuff but if we do something bad in the GLSL shaders then we want the proper line
373 // so we have to reset the line counting
374 Q_strcat(dest, size, "#line 0\n");
375}

377static int GLSL_CompileGPUShader(GLuint program, GLuint *prevShader, const GLchar *buffer, int size, GLenum shaderType)
378{
379 GLint compiled;
380 GLuint shader;
381
382 shader = qglCreateShader(shaderType);
383
384 qglShaderSource(shader, 1, (const GLchar **)&buffer, &size);
385
386 // compile shader
387 qglCompileShader(shader);
388
389 // check if shader compiled
390 qglGetShaderiv(shader, GL_COMPILE_STATUS, &compiled);
391 if(!compiled)
392 {
393 GLSL_PrintLog(shader, GLSL_PRINTLOG_SHADER_SOURCE, qfalse);
394 GLSL_PrintLog(shader, GLSL_PRINTLOG_SHADER_INFO, qfalse);
395 ri.Error(ERR_DROP, "Couldn't compile shader");
396 return 0;
397 }
398
399 if (*prevShader)
400 {
401 qglDetachShader(program, *prevShader);
402 qglDeleteShader(*prevShader);
403 }
404
405 // attach shader to program
406 qglAttachShader(program, shader);
407
408 *prevShader = shader;
409
410 return 1;
411}

413static int GLSL_LoadGPUShaderText(const char *name, const char *fallback,
414 GLenum shaderType, char *dest, int destSize)
415{
416 char filename[MAX_QPATH];
417 GLchar *buffer = NULL;
418 const GLchar *shaderText = NULL;
419 int size;
420 int result;
421
422 if(shaderType == GL_VERTEX_SHADER)
423 {
424 Com_sprintf(filename, sizeof(filename), "glsl/%s_vp.glsl", name);
425 }
426 else
427 {
428 Com_sprintf(filename, sizeof(filename), "glsl/%s_fp.glsl", name);
429 }
430
431 if ( r_externalGLSL->integer ) {
432 size = ri.FS_ReadFile(filename, (void **)&buffer);
433 } else {
434 size = 0;
435 buffer = NULL;
436 }
437
438 if(!buffer)
439 {
440 if (fallback)
441 {
442 ri.Printf(PRINT_DEVELOPER, "...loading built-in '%s'\n", filename);
443 shaderText = fallback;
444 size = strlen(shaderText);
445 }
446 else
447 {
448 ri.Printf(PRINT_DEVELOPER, "couldn't load '%s'\n", filename);
449 return 0;
450 }
451 }
452 else
453 {
454 ri.Printf(PRINT_DEVELOPER, "...loading '%s'\n", filename);
455 shaderText = buffer;
456 }
457
458 if (size > destSize)
459 {
460 result = 0;
461 }
462 else
463 {
464 Q_strncpyz(dest, shaderText, size + 1);
465 result = 1;
466 }
467
468 if (buffer)
469 {
470 ri.FS_FreeFile(buffer);
471 }
472
473 return result;
474}

476static void GLSL_LinkProgram(GLuint program)
477{
478 GLint linked;
479
480 qglLinkProgram(program);
481
482 qglGetProgramiv(program, GL_LINK_STATUS, &linked);
483 if(!linked)
484 {
485 GLSL_PrintLog(program, GLSL_PRINTLOG_PROGRAM_INFO, qfalse);
486 ri.Error(ERR_DROP, "shaders failed to link");
487 }
488}

490static void GLSL_ShowProgramUniforms(GLuint program)
491{
492 int i, count, size;
493 GLenum type;
494 char uniformName[1000];
495
496 // query the number of active uniforms
497 qglGetProgramiv(program, GL_ACTIVE_UNIFORMS, &count);
498
499 // Loop over each of the active uniforms, and set their value
500 for(i = 0; i < count; i++)
501 {
502 qglGetActiveUniform(program, i, sizeof(uniformName), NULL, &size, &type, uniformName);
503
504 ri.Printf(PRINT_DEVELOPER, "active uniform: '%s'\n", uniformName);
505 }
506}

508static int GLSL_InitGPUShader2(shaderProgram_t * program, const char *name, int attribs, const char *vpCode, const char *fpCode)
509{
510 ri.Printf(PRINT_DEVELOPER, "------- GPU shader -------\n");
511
512 if(strlen(name) >= MAX_QPATH)
513 {
514 ri.Error(ERR_DROP, "GLSL_InitGPUShader2: \"%s\" is too long", name);
515 }
516
517 Q_strncpyz(program->name, name, sizeof(program->name));
518
519 program->program = qglCreateProgram();
520 program->attribs = attribs;
521
522 if (!(GLSL_CompileGPUShader(program->program, &program->vertexShader, vpCode, strlen(vpCode), GL_VERTEX_SHADER)))
523 {
524 ri.Printf(PRINT_ALL, "GLSL_InitGPUShader2: Unable to load \"%s\" as GL_VERTEX_SHADER\n", name);
525 qglDeleteProgram(program->program);
526 return 0;
527 }
528
529 if(fpCode)
530 {
531 if(!(GLSL_CompileGPUShader(program->program, &program->fragmentShader, fpCode, strlen(fpCode), GL_FRAGMENT_SHADER)))
532 {
533 ri.Printf(PRINT_ALL, "GLSL_InitGPUShader2: Unable to load \"%s\" as GL_FRAGMENT_SHADER\n", name);
534 qglDeleteProgram(program->program);
535 return 0;
536 }
537 }
538
539 if(attribs & ATTR_POSITION)
540 qglBindAttribLocation(program->program, ATTR_INDEX_POSITION, "attr_Position");
541
542 if(attribs & ATTR_TEXCOORD)
543 qglBindAttribLocation(program->program, ATTR_INDEX_TEXCOORD, "attr_TexCoord0");
544
545 if(attribs & ATTR_LIGHTCOORD)
546 qglBindAttribLocation(program->program, ATTR_INDEX_LIGHTCOORD, "attr_TexCoord1");
547
548// if(attribs & ATTR_TEXCOORD2)
549// qglBindAttribLocation(program->program, ATTR_INDEX_TEXCOORD2, "attr_TexCoord2");
550
551// if(attribs & ATTR_TEXCOORD3)
552// qglBindAttribLocation(program->program, ATTR_INDEX_TEXCOORD3, "attr_TexCoord3");
553
554 if(attribs & ATTR_TANGENT)
555 qglBindAttribLocation(program->program, ATTR_INDEX_TANGENT, "attr_Tangent");
556
557 if(attribs & ATTR_NORMAL)
558 qglBindAttribLocation(program->program, ATTR_INDEX_NORMAL, "attr_Normal");
559
560 if(attribs & ATTR_COLOR)
561 qglBindAttribLocation(program->program, ATTR_INDEX_COLOR, "attr_Color");
562
563 if(attribs & ATTR_PAINTCOLOR)
564 qglBindAttribLocation(program->program, ATTR_INDEX_PAINTCOLOR, "attr_PaintColor");
565
566 if(attribs & ATTR_LIGHTDIRECTION)
567 qglBindAttribLocation(program->program, ATTR_INDEX_LIGHTDIRECTION, "attr_LightDirection");
568
569 if(attribs & ATTR_POSITION2)
570 qglBindAttribLocation(program->program, ATTR_INDEX_POSITION2, "attr_Position2");
571
572 if(attribs & ATTR_NORMAL2)
573 qglBindAttribLocation(program->program, ATTR_INDEX_NORMAL2, "attr_Normal2");
574
575 if(attribs & ATTR_TANGENT2)
576 qglBindAttribLocation(program->program, ATTR_INDEX_TANGENT2, "attr_Tangent2");
577
578 GLSL_LinkProgram(program->program);
579
580 return 1;
581}

583static int GLSL_InitGPUShader(shaderProgram_t * program, const char *name,
584 int attribs, qboolean fragmentShader, const GLchar *extra, qboolean addHeader,
585 const char *fallback_vp, const char *fallback_fp)
586{
587 char vpCode[32000];
588 char fpCode[32000];
589 char *postHeader;
590 int size;
591 int result;
592
593 size = sizeof(vpCode);
594 if (addHeader)
595 {
596 GLSL_GetShaderHeader(GL_VERTEX_SHADER, extra, vpCode, size);
597 postHeader = &vpCode[strlen(vpCode)];
598 size -= strlen(vpCode);
599 }
600 else
601 {
602 postHeader = &vpCode[0];
603 }
604
605 if (!GLSL_LoadGPUShaderText(name, fallback_vp, GL_VERTEX_SHADER, postHeader, size))
606 {
607 return 0;
608 }
609
610 if (fragmentShader)
611 {
612 size = sizeof(fpCode);
613 if (addHeader)
614 {
615 GLSL_GetShaderHeader(GL_FRAGMENT_SHADER, extra, fpCode, size);
616 postHeader = &fpCode[strlen(fpCode)];
617 size -= strlen(fpCode);
618 }
619 else
620 {
621 postHeader = &fpCode[0];
622 }
623
624 if (!GLSL_LoadGPUShaderText(name, fallback_fp, GL_FRAGMENT_SHADER, postHeader, size))
625 {
626 return 0;
627 }
628 }
629
630 result = GLSL_InitGPUShader2(program, name, attribs, vpCode, fragmentShader ? fpCode : NULL);
631
632 return result;
633}

635void GLSL_InitUniforms(shaderProgram_t *program)
636{
637 int i, size;
638
639 GLint *uniforms = program->uniforms;
640
641 size = 0;
642 for (i = 0; i < UNIFORM_COUNT; i++)
643 {
644 uniforms[i] = qglGetUniformLocation(program->program, uniformsInfo[i].name);
645
646 if (uniforms[i] == -1)
647 continue;
648
649 program->uniformBufferOffsets[i] = size;
650
651 switch(uniformsInfo[i].type)
652 {
653 case GLSL_INT:
654 size += sizeof(GLint);
655 break;
656 case GLSL_FLOAT:
657 size += sizeof(GLfloat);
658 break;
659 case GLSL_FLOAT5:
660 size += sizeof(vec_t) * 5;
661 break;
662 case GLSL_VEC2:
663 size += sizeof(vec_t) * 2;
664 break;
665 case GLSL_VEC3:
666 size += sizeof(vec_t) * 3;
667 break;
668 case GLSL_VEC4:
669 size += sizeof(vec_t) * 4;
670 break;
671 case GLSL_MAT16:
672 size += sizeof(vec_t) * 16;
673 break;
674 default:
675 break;
676 }
677 }
678
679 program->uniformBuffer = ri.Malloc(size);
680}

688void GLSL_SetUniformInt(shaderProgram_t *program, int uniformNum, GLint value)
689{
690 GLint *uniforms = program->uniforms;
691 GLint *compare = (GLint *)(program->uniformBuffer + program->uniformBufferOffsets[uniformNum]);
692
693 if (uniforms[uniformNum] == -1)
694 return;
695
696 if (uniformsInfo[uniformNum].type != GLSL_INT)
697 {

1408shaderProgram_t *GLSL_GetGenericShaderProgram(int stage)
1409{
1410 shaderStage_t *pStage = tess.xstages[stage];
1411 int shaderAttribs = 0;
1412
1413 if (tess.fogNum && pStage->adjustColorsForFog)
1414 {
1415 shaderAttribs |= GENERICDEF_USE_FOG;
1416 }
1417
1418 switch (pStage->rgbGen)
1419 {
1420 case CGEN_LIGHTING_DIFFUSE:
1421 shaderAttribs |= GENERICDEF_USE_RGBAGEN;
1422 break;
1423 default:
1424 break;
1425 }
1426
1427 switch (pStage->alphaGen)
1428 {
1429 case AGEN_LIGHTING_SPECULAR:
1430 case AGEN_PORTAL:
1431 shaderAttribs |= GENERICDEF_USE_RGBAGEN;
1432 break;
1433 default:
1434 break;
1435 }
1436
1437 if (pStage->bundle[0].tcGen != TCGEN_TEXTURE)
1438 {
1439 shaderAttribs |= GENERICDEF_USE_TCGEN_AND_TCMOD;
1440 }
1441
1442 if (tess.shader->numDeforms && !ShaderRequiresCPUDeforms(tess.shader))
1443 {
1444 shaderAttribs |= GENERICDEF_USE_DEFORM_VERTEXES;
1445 }
1446
1447 if (glState.vertexAnimation)
1448 {
1449 shaderAttribs |= GENERICDEF_USE_VERTEX_ANIMATION;
1450 }
1451
1452 if (pStage->bundle[0].numTexMods)
1453 {
1454 shaderAttribs |= GENERICDEF_USE_TCGEN_AND_TCMOD;
1455 }
1456
1457 return &tr.genericShader[shaderAttribs];
1458}

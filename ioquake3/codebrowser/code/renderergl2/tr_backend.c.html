40/*
41** GL_BindToTMU
42*/
43void GL_BindToTMU( image_t *image, int tmu )
44{
45 GLuint texture = (tmu == TB_COLORMAP) ? tr.defaultImage->texnum : 0;
46 GLenum target = GL_TEXTURE_2D;
47
48 if (image)
49 {
50 if (image->flags & IMGFLAG_CUBEMAP)
51 target = GL_TEXTURE_CUBE_MAP;
52
53 image->frameUsed = tr.frameCount;
54 texture = image->texnum;
55 }
56 else
57 {
58 ri.Printf(PRINT_WARNING, "GL_BindToTMU: NULL image\n");
59 }
60
61 GL_BindMultiTexture(GL_TEXTURE0_ARB + tmu, target, texture);
62}

65/*
66** GL_Cull
67*/
68void GL_Cull( int cullType ) {
69 if ( glState.faceCulling == cullType ) {
70 return;
71 }
72
73 if ( cullType == CT_TWO_SIDED )
74 {
75 qglDisable( GL_CULL_FACE );
76 }
77 else
78 {
79 qboolean cullFront = (cullType == CT_FRONT_SIDED);
80
81 if ( glState.faceCulling == CT_TWO_SIDED )
82 qglEnable( GL_CULL_FACE );
83
84 if ( glState.faceCullFront != cullFront )
85 qglCullFace( cullFront ? GL_FRONT : GL_BACK );
86
87 glState.faceCullFront = cullFront;
88 }
89
90 glState.faceCulling = cullType;
91}

93/*
94** GL_State
95**
96** This routine is responsible for setting the most commonly changed state
97** in Q3.
98*/
99void GL_State( unsigned long stateBits )
100{
101 unsigned long diff = stateBits ^ glState.glStateBits;
102
103 if ( !diff )
104 {
105 return;
106 }
107
108 //
109 // check depthFunc bits
110 //
111 if ( diff & GLS_DEPTHFUNC_BITS )
112 {
113 if ( stateBits & GLS_DEPTHFUNC_EQUAL )
114 {
115 qglDepthFunc( GL_EQUAL );
116 }
117 else if ( stateBits & GLS_DEPTHFUNC_GREATER)
118 {
119 qglDepthFunc( GL_GREATER );
120 }
121 else
122 {
123 qglDepthFunc( GL_LEQUAL );
124 }
125 }
126
127 //
128 // check blend bits
129 //
130 if ( diff & ( GLS_SRCBLEND_BITS | GLS_DSTBLEND_BITS ) )
131 {
132 uint32_t oldState = glState.glStateBits & ( GLS_SRCBLEND_BITS | GLS_DSTBLEND_BITS );
133 uint32_t newState = stateBits & ( GLS_SRCBLEND_BITS | GLS_DSTBLEND_BITS );
134 uint32_t storedState = glState.storedGlState & ( GLS_SRCBLEND_BITS | GLS_DSTBLEND_BITS );
135
136 if (oldState == 0)
137 {
138 qglEnable( GL_BLEND );
139 }
140 else if (newState == 0)
141 {
142 qglDisable( GL_BLEND );
143 }
144
145 if (newState != 0 && storedState != newState)
146 {
147 GLenum srcFactor = GL_ONE, dstFactor = GL_ONE;
148
149 glState.storedGlState &= ~( GLS_SRCBLEND_BITS | GLS_DSTBLEND_BITS );
150 glState.storedGlState |= newState;
151
152 switch ( stateBits & GLS_SRCBLEND_BITS )
153 {
154 case GLS_SRCBLEND_ZERO:
155 srcFactor = GL_ZERO;
156 break;
157 case GLS_SRCBLEND_ONE:
158 srcFactor = GL_ONE;
159 break;
160 case GLS_SRCBLEND_DST_COLOR:
161 srcFactor = GL_DST_COLOR;
162 break;
163 case GLS_SRCBLEND_ONE_MINUS_DST_COLOR:
164 srcFactor = GL_ONE_MINUS_DST_COLOR;
165 break;
166 case GLS_SRCBLEND_SRC_ALPHA:
167 srcFactor = GL_SRC_ALPHA;
168 break;
169 case GLS_SRCBLEND_ONE_MINUS_SRC_ALPHA:
170 srcFactor = GL_ONE_MINUS_SRC_ALPHA;
171 break;
172 case GLS_SRCBLEND_DST_ALPHA:
173 srcFactor = GL_DST_ALPHA;
174 break;
175 case GLS_SRCBLEND_ONE_MINUS_DST_ALPHA:
176 srcFactor = GL_ONE_MINUS_DST_ALPHA;
177 break;
178 case GLS_SRCBLEND_ALPHA_SATURATE:
179 srcFactor = GL_SRC_ALPHA_SATURATE;
180 break;
181 default:
182 ri.Error( ERR_DROP, "GL_State: invalid src blend state bits" );
183 break;
184 }
185
186 switch ( stateBits & GLS_DSTBLEND_BITS )
187 {
188 case GLS_DSTBLEND_ZERO:
189 dstFactor = GL_ZERO;
190 break;
191 case GLS_DSTBLEND_ONE:
192 dstFactor = GL_ONE;
193 break;
194 case GLS_DSTBLEND_SRC_COLOR:
195 dstFactor = GL_SRC_COLOR;
196 break;
197 case GLS_DSTBLEND_ONE_MINUS_SRC_COLOR:
198 dstFactor = GL_ONE_MINUS_SRC_COLOR;
199 break;
200 case GLS_DSTBLEND_SRC_ALPHA:
201 dstFactor = GL_SRC_ALPHA;
202 break;
203 case GLS_DSTBLEND_ONE_MINUS_SRC_ALPHA:
204 dstFactor = GL_ONE_MINUS_SRC_ALPHA;
205 break;
206 case GLS_DSTBLEND_DST_ALPHA:
207 dstFactor = GL_DST_ALPHA;
208 break;
209 case GLS_DSTBLEND_ONE_MINUS_DST_ALPHA:
210 dstFactor = GL_ONE_MINUS_DST_ALPHA;
211 break;
212 default:
213 ri.Error( ERR_DROP, "GL_State: invalid dst blend state bits" );
214 break;
215 }
216
217 qglBlendFunc( srcFactor, dstFactor );
218 }
219 }
220
221 //
222 // check depthmask
223 //
224 if ( diff & GLS_DEPTHMASK_TRUE )
225 {
226 if ( stateBits & GLS_DEPTHMASK_TRUE )
227 {
228 qglDepthMask( GL_TRUE );
229 }
230 else
231 {
232 qglDepthMask( GL_FALSE );
233 }
234 }
235
236 //
237 // fill/line mode
238 //
239 if ( diff & GLS_POLYMODE_LINE )
240 {
241 if ( stateBits & GLS_POLYMODE_LINE )
242 {
243 qglPolygonMode( GL_FRONT_AND_BACK, GL_LINE );
244 }
245 else
246 {
247 qglPolygonMode( GL_FRONT_AND_BACK, GL_FILL );
248 }
249 }
250
251 //
252 // depthtest
253 //
254 if ( diff & GLS_DEPTHTEST_DISABLE )
255 {
256 if ( stateBits & GLS_DEPTHTEST_DISABLE )
257 {
258 qglDisable( GL_DEPTH_TEST );
259 }
260 else
261 {
262 qglEnable( GL_DEPTH_TEST );
263 }
264 }
265
266 glState.glStateBits = stateBits;
267}

317/*
318=================
319RB_BeginDrawingView
320
321Any mirrored or portaled views have already been drawn, so prepare
322to actually render the visible surfaces for this view
323=================
324*/
325void RB_BeginDrawingView (void) {
326 int clearBits = 0;
327
328 // sync with gl if needed
329 if ( r_finish->integer == 1 && !glState.finishCalled ) {
330 qglFinish ();
331 glState.finishCalled = qtrue;
332 }
333 if ( r_finish->integer == 0 ) {
334 glState.finishCalled = qtrue;
335 }
336
337 // we will need to change the projection matrix before drawing
338 // 2D images again
339 backEnd.projection2D = qfalse;
340
341 if (glRefConfig.framebufferObject)
342 {
343 FBO_t *fbo = backEnd.viewParms.targetFbo;
344
345 // FIXME: HUGE HACK: render to the screen fbo if we've already postprocessed the frame and aren't drawing more world
346 // drawing more world check is in case of double renders, such as skyportals
347 if (fbo == NULL && !(backEnd.framePostProcessed && (backEnd.refdef.rdflags & RDF_NOWORLDMODEL)))
348 fbo = tr.renderFbo;
349
350 if (tr.renderCubeFbo && fbo == tr.renderCubeFbo)
351 {
352 cubemap_t *cubemap = &tr.cubemaps[backEnd.viewParms.targetFboCubemapIndex];
353 FBO_AttachImage(fbo, cubemap->image, GL_COLOR_ATTACHMENT0_EXT, backEnd.viewParms.targetFboLayer);
354 }
355
356 FBO_Bind(fbo);
357 }
358
359 //
360 // set the modelview matrix for the viewer
361 //
362 SetViewportAndScissor();
363
364 // ensures that depth writes are enabled for the depth clear
365 GL_State( GLS_DEFAULT );
366 // clear relevant buffers
367 clearBits = GL_DEPTH_BUFFER_BIT;
368
369 if ( r_measureOverdraw->integer || r_shadows->integer == 2 )
370 {
371 clearBits |= GL_STENCIL_BUFFER_BIT;
372 }
373 if ( r_fastsky->integer && !( backEnd.refdef.rdflags & RDF_NOWORLDMODEL ) )
374 {
375 clearBits |= GL_COLOR_BUFFER_BIT; // FIXME: only if sky shaders have been used
376 }
377
378 // clear to black for cube maps
379 if (tr.renderCubeFbo && backEnd.viewParms.targetFbo == tr.renderCubeFbo)
380 {
381 clearBits |= GL_COLOR_BUFFER_BIT;
382 }
383
384 qglClear( clearBits );
385
386 if ( ( backEnd.refdef.rdflags & RDF_HYPERSPACE ) )
387 {
388 RB_Hyperspace();
389 return;
390 }
391 else
392 {
393 backEnd.isHyperspace = qfalse;
394 }
395
396 // we will only draw a sun if there was sky rendered in this view
397 backEnd.skyRenderedThisView = qfalse;
398
399 // clip to the plane of the portal
400 if ( backEnd.viewParms.isPortal ) {
401#if 0
402 float plane[4];
403 GLdouble plane2[4];
404
405 plane[0] = backEnd.viewParms.portalPlane.normal[0];
406 plane[1] = backEnd.viewParms.portalPlane.normal[1];
407 plane[2] = backEnd.viewParms.portalPlane.normal[2];
408 plane[3] = backEnd.viewParms.portalPlane.dist;
409
410 plane2[0] = DotProduct (backEnd.viewParms.or.axis[0], plane);
411 plane2[1] = DotProduct (backEnd.viewParms.or.axis[1], plane);
412 plane2[2] = DotProduct (backEnd.viewParms.or.axis[2], plane);
413 plane2[3] = DotProduct (plane, backEnd.viewParms.or.origin) - plane[3];
414#endif
415 GL_SetModelviewMatrix( s_flipMatrix );
416 }
417}

420/*
421==================
422RB_RenderDrawSurfList
423==================
424*/
425void RB_RenderDrawSurfList( drawSurf_t *drawSurfs, int numDrawSurfs ) {
426 shader_t *shader, *oldShader;
427 int fogNum, oldFogNum;
428 int entityNum, oldEntityNum;
429 int dlighted, oldDlighted;
430 int pshadowed, oldPshadowed;
431 int cubemapIndex, oldCubemapIndex;
432 qboolean depthRange, oldDepthRange, isCrosshair, wasCrosshair;
433 int i;
434 drawSurf_t *drawSurf;
435 int oldSort;
436 double originalTime;
437 FBO_t* fbo = NULL;
438 qboolean inQuery = qfalse;
439
440 // save original time for entity shader offsets
441 originalTime = backEnd.refdef.floatTime;
442
443 fbo = glState.currentFBO;
444
445 // draw everything
446 oldEntityNum = -1;
447 backEnd.currentEntity = &tr.worldEntity;
448 oldShader = NULL;
449 oldFogNum = -1;
450 oldDepthRange = qfalse;
451 wasCrosshair = qfalse;
452 oldDlighted = qfalse;
453 oldPshadowed = qfalse;
454 oldCubemapIndex = -1;
455 oldSort = -1;
456
457 backEnd.pc.c_surfaces += numDrawSurfs;
458
459 for (i = 0, drawSurf = drawSurfs ; i < numDrawSurfs ; i++, drawSurf++) {
460 if ( drawSurf->sort == oldSort && drawSurf->cubemapIndex == oldCubemapIndex) {
461 if (backEnd.depthFill && shader && shader->sort != SS_OPAQUE)
462 continue;
463
464 // fast path, same as previous sort
465 rb_surfaceTable[ *drawSurf->surface ]( drawSurf->surface );
466 continue;
467 }
468 oldSort = drawSurf->sort;
469 R_DecomposeSort( drawSurf->sort, &entityNum, &shader, &fogNum, &dlighted, &pshadowed );
470 cubemapIndex = drawSurf->cubemapIndex;
471
472 //
473 // change the tess parameters if needed
474 // a "entityMergable" shader is a shader that can have surfaces from seperate
475 // entities merged into a single batch, like smoke and blood puff sprites
476 if ( shader != NULL && ( shader != oldShader || fogNum != oldFogNum || dlighted != oldDlighted || pshadowed != oldPshadowed || cubemapIndex != oldCubemapIndex
477 || ( entityNum != oldEntityNum && !shader->entityMergable ) ) ) {
478 if (oldShader != NULL) {
479 RB_EndSurface();
480 }
481 RB_BeginSurface( shader, fogNum, cubemapIndex );
482 backEnd.pc.c_surfBatches++;
483 oldShader = shader;
484 oldFogNum = fogNum;
485 oldDlighted = dlighted;
486 oldPshadowed = pshadowed;
487 oldCubemapIndex = cubemapIndex;
488 }
489
490 if (backEnd.depthFill && shader && shader->sort != SS_OPAQUE)
491 continue;
492
493 //
494 // change the modelview matrix if needed
495 //
496 if ( entityNum != oldEntityNum ) {
497 depthRange = isCrosshair = qfalse;
498
499 if ( entityNum != REFENTITYNUM_WORLD ) {
500 backEnd.currentEntity = &backEnd.refdef.entities[entityNum];
501
502 // FIXME: e.shaderTime must be passed as int to avoid fp-precision loss issues
503 backEnd.refdef.floatTime = originalTime - (double)backEnd.currentEntity->e.shaderTime;
504
505 // we have to reset the shaderTime as well otherwise image animations start
506 // from the wrong frame
507 tess.shaderTime = backEnd.refdef.floatTime - tess.shader->timeOffset;
508
509 // set up the transformation matrix
510 R_RotateForEntity( backEnd.currentEntity, &backEnd.viewParms, &backEnd.or );
511
512 // set up the dynamic lighting if needed
513 if ( backEnd.currentEntity->needDlights ) {
514 R_TransformDlights( backEnd.refdef.num_dlights, backEnd.refdef.dlights, &backEnd.or );
515 }
516
517 if(backEnd.currentEntity->e.renderfx & RF_DEPTHHACK)
518 {
519 // hack the depth range to prevent view model from poking into walls
520 depthRange = qtrue;
521
522 if(backEnd.currentEntity->e.renderfx & RF_CROSSHAIR)
523 isCrosshair = qtrue;
524 }
525 } else {
526 backEnd.currentEntity = &tr.worldEntity;
527 backEnd.refdef.floatTime = originalTime;
528 backEnd.or = backEnd.viewParms.world;
529 // we have to reset the shaderTime as well otherwise image animations on
530 // the world (like water) continue with the wrong frame
531 tess.shaderTime = backEnd.refdef.floatTime - tess.shader->timeOffset;
532 R_TransformDlights( backEnd.refdef.num_dlights, backEnd.refdef.dlights, &backEnd.or );
533 }
534
535 GL_SetModelviewMatrix( backEnd.or.modelMatrix );
536
537 //
538 // change depthrange. Also change projection matrix so first person weapon does not look like coming
539 // out of the screen.
540 //
541 if (oldDepthRange != depthRange || wasCrosshair != isCrosshair)
542 {
543 if (depthRange)
544 {
545 if(backEnd.viewParms.stereoFrame != STEREO_CENTER)
546 {
547 if(isCrosshair)
548 {
549 if(oldDepthRange)
550 {
551 // was not a crosshair but now is, change back proj matrix
552 GL_SetProjectionMatrix( backEnd.viewParms.projectionMatrix );
553 }
554 }
555 else
556 {
557 viewParms_t temp = backEnd.viewParms;
558
559 R_SetupProjection(&temp, r_znear->value, 0, qfalse);
560
561 GL_SetProjectionMatrix( temp.projectionMatrix );
562 }
563 }
564
565 if(!oldDepthRange)
566 qglDepthRange (0, 0.3);
567 }
568 else
569 {
570 if(!wasCrosshair && backEnd.viewParms.stereoFrame != STEREO_CENTER)
571 {
572 GL_SetProjectionMatrix( backEnd.viewParms.projectionMatrix );
573 }
574
575 qglDepthRange (0, 1);
576 }
577
578 oldDepthRange = depthRange;
579 wasCrosshair = isCrosshair;
580 }
581
582 oldEntityNum = entityNum;
583 }
584
585 // add the triangles for this surface
586 rb_surfaceTable[ *drawSurf->surface ]( drawSurf->surface );
587 }
588
589 backEnd.refdef.floatTime = originalTime;
590
591 // draw the contents of the last shader batch
592 if (oldShader != NULL) {
593 RB_EndSurface();
594 }
595
596 if (inQuery) {
597 qglEndQuery(GL_SAMPLES_PASSED);
598 }
599
600 if (glRefConfig.framebufferObject)
601 FBO_Bind(fbo);
602
603 // go back to the world modelview matrix
604
605 GL_SetModelviewMatrix( backEnd.viewParms.world.modelMatrix );
606
607 qglDepthRange (0, 1);
608}

619/*
620================
621RB_SetGL2D
622
623================
624*/
625void RB_SetGL2D (void) {
626 mat4_t matrix;
627 int width, height;
628
629 if (backEnd.projection2D && backEnd.last2DFBO == glState.currentFBO)
630 return;
631
632 backEnd.projection2D = qtrue;
633 backEnd.last2DFBO = glState.currentFBO;
634
635 if (glState.currentFBO)
636 {
637 width = glState.currentFBO->width;
638 height = glState.currentFBO->height;
639 }
640 else
641 {
642 width = glConfig.vidWidth;
643 height = glConfig.vidHeight;
644 }
645
646 // set 2D virtual screen size
647 qglViewport( 0, 0, width, height );
648 qglScissor( 0, 0, width, height );
649
650 Mat4Ortho(0, width, height, 0, 0, 1, matrix);
651 GL_SetProjectionMatrix(matrix);
652 Mat4Identity(matrix);
653 GL_SetModelviewMatrix(matrix);
654
655 GL_State( GLS_DEPTHTEST_DISABLE |
656 GLS_SRCBLEND_SRC_ALPHA |
657 GLS_DSTBLEND_ONE_MINUS_SRC_ALPHA );
658
659 GL_Cull( CT_TWO_SIDED );
660 qglDisable( GL_CLIP_PLANE0 );
661
662 // set time for 2D shaders
663 backEnd.refdef.time = ri.Milliseconds();
664 backEnd.refdef.floatTime = backEnd.refdef.time * 0.001;
665}

668/*
669=============
670RE_StretchRaw
671
672FIXME: not exactly backend
673Stretches a raw 32 bit power of 2 bitmap image over the given screen rectangle.
674Used for cinematics.
675=============
676*/
677void RE_StretchRaw (int x, int y, int w, int h, int cols, int rows, const byte *data, int client, qboolean dirty) {
678 int i, j;
679 int start, end;
680 vec4_t quadVerts[4];
681 vec2_t texCoords[4];
682
683 if ( !tr.registered ) {
684 return;
685 }
686 R_IssuePendingRenderCommands();
687
688 if ( tess.numIndexes ) {
689 RB_EndSurface();
690 }
691
692 // we definately want to sync every frame for the cinematics
693 qglFinish();
694
695 start = 0;
696 if ( r_speeds->integer ) {
697 start = ri.Milliseconds();
698 }
699
700 // make sure rows and cols are powers of 2
701 for ( i = 0 ; ( 1 << i ) < cols ; i++ ) {
702 }
703 for ( j = 0 ; ( 1 << j ) < rows ; j++ ) {
704 }
705 if ( ( 1 << i ) != cols || ( 1 << j ) != rows) {
706 ri.Error (ERR_DROP, "Draw_StretchRaw: size not a power of 2: %i by %i", cols, rows);
707 }
708
709 RE_UploadCinematic (w, h, cols, rows, data, client, dirty);
710 GL_BindToTMU(tr.scratchImage[client], TB_COLORMAP);
711
712 if ( r_speeds->integer ) {
713 end = ri.Milliseconds();
714 ri.Printf( PRINT_ALL, "qglTexSubImage2D %i, %i: %i msec\n", cols, rows, end - start );
715 }
716
717 // FIXME: HUGE hack
718 if (glRefConfig.framebufferObject)
719 {
720 FBO_Bind(backEnd.framePostProcessed ? NULL : tr.renderFbo);
721 }
722
723 RB_SetGL2D();
724
725 VectorSet4(quadVerts[0], x, y, 0.0f, 1.0f);
726 VectorSet4(quadVerts[1], x + w, y, 0.0f, 1.0f);
727 VectorSet4(quadVerts[2], x + w, y + h, 0.0f, 1.0f);
728 VectorSet4(quadVerts[3], x, y + h, 0.0f, 1.0f);
729
730 VectorSet2(texCoords[0], 0.5f / cols, 0.5f / rows);
731 VectorSet2(texCoords[1], (cols - 0.5f) / cols, 0.5f / rows);
732 VectorSet2(texCoords[2], (cols - 0.5f) / cols, (rows - 0.5f) / rows);
733 VectorSet2(texCoords[3], 0.5f / cols, (rows - 0.5f) / rows);
734
735 GLSL_BindProgram(&tr.textureColorShader);
736
737 GLSL_SetUniformMat4(&tr.textureColorShader, UNIFORM_MODELVIEWPROJECTIONMATRIX, glState.modelviewProjection);
738 GLSL_SetUniformVec4(&tr.textureColorShader, UNIFORM_COLOR, colorWhite);
739
740 RB_InstantQuad2(quadVerts, texCoords);
741}

743void RE_UploadCinematic (int w, int h, int cols, int rows, const byte *data, int client, qboolean dirty) {
744 GLuint texture;
745
746 if (!tr.scratchImage[client])
747 {
748 ri.Printf(PRINT_WARNING, "RE_UploadCinematic: scratch images not initialized\n");
749 return;
750 }
751
752 texture = tr.scratchImage[client]->texnum;
753
754 // if the scratchImage isn't in the format we want, specify it as a new texture
755 if ( cols != tr.scratchImage[client]->width || rows != tr.scratchImage[client]->height ) {
756 tr.scratchImage[client]->width = tr.scratchImage[client]->uploadWidth = cols;
757 tr.scratchImage[client]->height = tr.scratchImage[client]->uploadHeight = rows;

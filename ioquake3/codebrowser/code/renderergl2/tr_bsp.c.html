48static void HSVtoRGB( float h, float s, float v, float rgb[3] )
49{
50 int i;
51 float f;
52 float p, q, t;
53
54 h *= 5;
55
56 i = floor( h );
57 f = h - i;
58
59 p = v * ( 1 - s );
60 q = v * ( 1 - s * f );
61 t = v * ( 1 - s * ( 1 - f ) );
62
63 switch ( i )
64 {
65 case 0:
66 rgb[0] = v;
67 rgb[1] = t;
68 rgb[2] = p;
69 break;
70 case 1:
71 rgb[0] = q;
72 rgb[1] = v;
73 rgb[2] = p;
74 break;
75 case 2:
76 rgb[0] = p;
77 rgb[1] = v;
78 rgb[2] = t;
79 break;
80 case 3:
81 rgb[0] = p;
82 rgb[1] = q;
83 rgb[2] = v;
84 break;
85 case 4:
86 rgb[0] = t;
87 rgb[1] = p;
88 rgb[2] = v;
89 break;
90 case 5:
91 rgb[0] = v;
92 rgb[1] = p;
93 rgb[2] = q;
94 break;
95 }
96}

98/*
99===============
100R_ColorShiftLightingBytes
101
102===============
103*/
104static void R_ColorShiftLightingBytes( byte in[4], byte out[4] ) {
105 int shift, r, g, b;
106
107 // shift the color data based on overbright range
108 shift = r_mapOverBrightBits->integer - tr.overbrightBits;
109
110 // shift the data based on overbright range
111 r = in[0] << shift;
112 g = in[1] << shift;
113 b = in[2] << shift;
114
115 // normalize by color instead of saturating to white
116 if ( ( r | g | b ) > 255 ) {
117 int max;
118
119 max = r > g ? r : g;
120 max = max > b ? max : b;
121 r = r * 255 / max;
122 g = g * 255 / max;
123 b = b * 255 / max;
124 }
125
126 out[0] = r;
127 out[1] = g;
128 out[2] = b;
129 out[3] = in[3];
130}

133/*
134===============
135R_ColorShiftLightingFloats
136
137===============
138*/
139static void R_ColorShiftLightingFloats(float in[4], float out[4])
140{
141 float r, g, b;
142 float scale = (1 << (r_mapOverBrightBits->integer - tr.overbrightBits)) / 255.0f;
143
144 r = in[0] * scale;
145 g = in[1] * scale;
146 b = in[2] * scale;
147
148 // normalize by color instead of saturating to white
149 if ( r > 1 || g > 1 || b > 1 ) {
150 float max;
151
152 max = r > g ? r : g;
153 max = max > b ? max : b;
154 r = r / max;
155 g = g / max;
156 b = b / max;
157 }
158
159 out[0] = r;
160 out[1] = g;
161 out[2] = b;
162 out[3] = in[3];
163}

165// Modified from http://graphicrants.blogspot.jp/2009/04/rgbm-color-encoding.html
166void ColorToRGBM(const vec3_t color, unsigned char rgbm[4])
167{
168 vec3_t sample;
169 float maxComponent;
170
171 VectorCopy(color, sample);
172
173 maxComponent = MAX(sample[0], sample[1]);
174 maxComponent = MAX(maxComponent, sample[2]);
175 maxComponent = CLAMP(maxComponent, 1.0f/255.0f, 1.0f);
176
177 rgbm[3] = (unsigned char) ceil(maxComponent * 255.0f);
178 maxComponent = 255.0f / rgbm[3];
179
180 VectorScale(sample, maxComponent, sample);
181
182 rgbm[0] = (unsigned char) (sample[0] * 255);
183 rgbm[1] = (unsigned char) (sample[1] * 255);
184 rgbm[2] = (unsigned char) (sample[2] * 255);
185}

187void ColorToRGB16(const vec3_t color, uint16_t rgb16[3])
188{
189 rgb16[0] = color[0] * 65535.0f + 0.5f;
190 rgb16[1] = color[1] * 65535.0f + 0.5f;
191 rgb16[2] = color[2] * 65535.0f + 0.5f;
192}

196===============
197R_LoadLightmaps

201#define DEFAULT_LIGHTMAP_SIZE 128
202static void R_LoadLightmaps( lump_t *l, lump_t *surfs ) {
203 imgFlags_t imgFlags = IMGFLAG_NOLIGHTSCALE | IMGFLAG_NO_COMPRESSION | IMGFLAG_CLAMPTOEDGE;
204 byte *buf, *buf_p;
205 dsurface_t *surf;
206 int len;
207 byte *image;
208 int i, j, numLightmaps, textureInternalFormat = 0;
209 int numLightmapsPerPage = 16;
210 float maxIntensity = 0;
211 double sumIntensity = 0;
212
213 len = l->filelen;
214 if ( !len ) {
215 return;
216 }
217 buf = fileBase + l->fileofs;
218
219 // we are about to upload textures
220 R_IssuePendingRenderCommands();
221
222 tr.lightmapSize = DEFAULT_LIGHTMAP_SIZE;
223 numLightmaps = len / (tr.lightmapSize * tr.lightmapSize * 3);
224
225 // check for deluxe mapping
226 if (numLightmaps <= 1)
227 {
228 tr.worldDeluxeMapping = qfalse;
229 }
230 else
231 {
232 tr.worldDeluxeMapping = qtrue;
233 for( i = 0, surf = (dsurface_t *)(fileBase + surfs->fileofs);
234 i < surfs->filelen / sizeof(dsurface_t); i++, surf++ ) {
235 int lightmapNum = LittleLong( surf->lightmapNum );
236
237 if ( lightmapNum >= 0 && (lightmapNum & 1) != 0 ) {
238 tr.worldDeluxeMapping = qfalse;
239 break;
240 }
241 }
242 }
243
244 image = ri.Malloc(tr.lightmapSize * tr.lightmapSize * 4 * 2);
245
246 if (tr.worldDeluxeMapping)
247 numLightmaps >>= 1;
248
249 // Use fat lightmaps of an appropriate size.
250 if (r_mergeLightmaps->integer)
251 {
252 int maxLightmapsPerAxis = glConfig.maxTextureSize / tr.lightmapSize;
253 int lightmapCols = 4, lightmapRows = 4;
254
255 // Increase width at first, then height.
256 while (lightmapCols * lightmapRows < numLightmaps && lightmapCols != maxLightmapsPerAxis)
257 lightmapCols <<= 1;
258
259 while (lightmapCols * lightmapRows < numLightmaps && lightmapRows != maxLightmapsPerAxis)
260 lightmapRows <<= 1;
261
262 tr.fatLightmapCols = lightmapCols;
263 tr.fatLightmapRows = lightmapRows;
264 numLightmapsPerPage = lightmapCols * lightmapRows;
265
266 tr.numLightmaps = (numLightmaps + (numLightmapsPerPage - 1)) / numLightmapsPerPage;
267 }
268 else
269 {
270 tr.numLightmaps = numLightmaps;
271 }
272
273 tr.lightmaps = ri.Hunk_Alloc( tr.numLightmaps * sizeof(image_t *), h_low );
274
275 if (tr.worldDeluxeMapping)
276 tr.deluxemaps = ri.Hunk_Alloc( tr.numLightmaps * sizeof(image_t *), h_low );
277
278 textureInternalFormat = GL_RGBA8;
279 if (r_hdr->integer)
280 {
281 // Check for the first hdr lightmap, if it exists, use GL_RGBA16 for textures.
282 char filename[MAX_QPATH];
283
284 Com_sprintf(filename, sizeof(filename), "maps/%s/lm_0000.hdr", s_worldData.baseName);
285 if (ri.FS_FileExists(filename))
286 textureInternalFormat = GL_RGBA16;
287 }
288
289 if (r_mergeLightmaps->integer)
290 {
291 int width = tr.fatLightmapCols * tr.lightmapSize;
292 int height = tr.fatLightmapRows * tr.lightmapSize;
293
294 for (i = 0; i < tr.numLightmaps; i++)
295 {
296 tr.lightmaps[i] = R_CreateImage(va("_fatlightmap%d", i), NULL, width, height, IMGTYPE_COLORALPHA, imgFlags, textureInternalFormat);
297
298 if (tr.worldDeluxeMapping)
299 tr.deluxemaps[i] = R_CreateImage(va("_fatdeluxemap%d", i), NULL, width, height, IMGTYPE_DELUXE, imgFlags, 0);
300 }
301 }
302
303 for(i = 0; i < numLightmaps; i++)
304 {
305 int xoff = 0, yoff = 0;
306 int lightmapnum = i;
307 // expand the 24 bit on-disk to 32 bit
308
309 if (r_mergeLightmaps->integer)
310 {
311 int lightmaponpage = i % numLightmapsPerPage;
312 xoff = (lightmaponpage % tr.fatLightmapCols) * tr.lightmapSize;
313 yoff = (lightmaponpage / tr.fatLightmapCols) * tr.lightmapSize;
314
315 lightmapnum /= numLightmapsPerPage;
316 }
317
318 // if (tr.worldLightmapping)
319 {
320 char filename[MAX_QPATH];
321 byte *hdrLightmap = NULL;
322 int size = 0;
323
324 // look for hdr lightmaps
325 if (textureInternalFormat == GL_RGBA16)
326 {
327 Com_sprintf( filename, sizeof( filename ), "maps/%s/lm_%04d.hdr", s_worldData.baseName, i * (tr.worldDeluxeMapping ? 2 : 1) );
328 //ri.Printf(PRINT_ALL, "looking for %s\n", filename);
329
330 size = ri.FS_ReadFile(filename, (void **)&hdrLightmap);
331 }
332
333 if (hdrLightmap)
334 {
335 byte *p = hdrLightmap, *end = hdrLightmap + size;
336 //ri.Printf(PRINT_ALL, "found!\n");
337
338 /* FIXME: don't just skip over this header and actually parse it */
339 while (p < end && !(*p == '\n' && *(p+1) == '\n'))
340 p++;
341
342 p += 2;
343
344 while (p < end && !(*p == '\n'))
345 p++;
346
347 p++;
348
349 if (p >= end)
350 ri.Error(ERR_DROP, "Bad header for %s!", filename);
351
352 buf_p = p;
353
354#if 0 // HDRFILE_RGBE
355 if ((int)(end - hdrLightmap) != tr.lightmapSize * tr.lightmapSize * 4)
356 ri.Error(ERR_DROP, "Bad size for %s (%i)!", filename, size);
357#else // HDRFILE_FLOAT
358 if ((int)(end - hdrLightmap) != tr.lightmapSize * tr.lightmapSize * 12)
359 ri.Error(ERR_DROP, "Bad size for %s (%i)!", filename, size);
360#endif
361 }
362 else
363 {
364 int imgOffset = tr.worldDeluxeMapping ? i * 2 : i;
365 buf_p = buf + imgOffset * tr.lightmapSize * tr.lightmapSize * 3;
366 }
367
368 for ( j = 0 ; j < tr.lightmapSize * tr.lightmapSize; j++ )
369 {
370 if (hdrLightmap)
371 {
372 vec4_t color;
373
374#if 0 // HDRFILE_RGBE
375 float exponent = exp2(buf_p[j*4+3] - 128);
376
377 color[0] = buf_p[j*4+0] * exponent;
378 color[1] = buf_p[j*4+1] * exponent;
379 color[2] = buf_p[j*4+2] * exponent;
380#else // HDRFILE_FLOAT
381 memcpy(color, &buf_p[j*12], 12);
382
383 color[0] = LittleFloat(color[0]);
384 color[1] = LittleFloat(color[1]);
385 color[2] = LittleFloat(color[2]);
386#endif
387 color[3] = 1.0f;
388
389 R_ColorShiftLightingFloats(color, color);
390
391 ColorToRGB16(color, (uint16_t *)(&image[j * 8]));
392 ((uint16_t *)(&image[j * 8]))[3] = 65535;
393 }
394 else if (textureInternalFormat == GL_RGBA16)
395 {
396 vec4_t color;
397
398 //hack: convert LDR lightmap to HDR one
399 color[0] = MAX(buf_p[j*3+0], 0.499f);
400 color[1] = MAX(buf_p[j*3+1], 0.499f);
401 color[2] = MAX(buf_p[j*3+2], 0.499f);
402
403 // if under an arbitrary value (say 12) grey it out
404 // this prevents weird splotches in dimly lit areas
405 if (color[0] + color[1] + color[2] < 12.0f)
406 {
407 float avg = (color[0] + color[1] + color[2]) * 0.3333f;
408 color[0] = avg;
409 color[1] = avg;
410 color[2] = avg;
411 }
412 color[3] = 1.0f;
413
414 R_ColorShiftLightingFloats(color, color);
415
416 ColorToRGB16(color, (uint16_t *)(&image[j * 8]));
417 ((uint16_t *)(&image[j * 8]))[3] = 65535;
418 }
419 else
420 {
421 if ( r_lightmap->integer == 2 )
422 { // color code by intensity as development tool (FIXME: check range)
423 float r = buf_p[j*3+0];
424 float g = buf_p[j*3+1];
425 float b = buf_p[j*3+2];
426 float intensity;
427 float out[3] = {0.0, 0.0, 0.0};
428
429 intensity = 0.33f * r + 0.685f * g + 0.063f * b;
430
431 if ( intensity > 255 )
432 intensity = 1.0f;
433 else
434 intensity /= 255.0f;
435
436 if ( intensity > maxIntensity )
437 maxIntensity = intensity;
438
439 HSVtoRGB( intensity, 1.00, 0.50, out );
440
441 image[j*4+0] = out[0] * 255;
442 image[j*4+1] = out[1] * 255;
443 image[j*4+2] = out[2] * 255;
444 image[j*4+3] = 255;
445
446 sumIntensity += intensity;
447 }
448 else
449 {
450 R_ColorShiftLightingBytes( &buf_p[j*3], &image[j*4] );
451 image[j*4+3] = 255;
452 }
453 }
454 }
455
456 if (r_mergeLightmaps->integer)
457 R_UpdateSubImage(tr.lightmaps[lightmapnum], image, xoff, yoff, tr.lightmapSize, tr.lightmapSize, textureInternalFormat);
458 else
459 tr.lightmaps[i] = R_CreateImage(va("*lightmap%d", i), image, tr.lightmapSize, tr.lightmapSize, IMGTYPE_COLORALPHA, imgFlags, textureInternalFormat );
460
461 if (hdrLightmap)
462 ri.FS_FreeFile(hdrLightmap);
463 }
464
465 if (tr.worldDeluxeMapping)
466 {
467 buf_p = buf + (i * 2 + 1) * tr.lightmapSize * tr.lightmapSize * 3;
468
469 for ( j = 0 ; j < tr.lightmapSize * tr.lightmapSize; j++ ) {
470 image[j*4+0] = buf_p[j*3+0];
471 image[j*4+1] = buf_p[j*3+1];
472 image[j*4+2] = buf_p[j*3+2];
473
474 // make 0,0,0 into 127,127,127
475 if ((image[j*4+0] == 0) && (image[j*4+1] == 0) && (image[j*4+2] == 0))
476 {
477 image[j*4+0] =
478 image[j*4+1] =
479 image[j*4+2] = 127;
480 }
481
482 image[j*4+3] = 255;
483 }
484
485 if (r_mergeLightmaps->integer)
486 R_UpdateSubImage(tr.deluxemaps[lightmapnum], image, xoff, yoff, tr.lightmapSize, tr.lightmapSize, GL_RGBA8 );
487 else
488 tr.deluxemaps[i] = R_CreateImage(va("*deluxemap%d", i), image, tr.lightmapSize, tr.lightmapSize, IMGTYPE_DELUXE, imgFlags, 0 );
489 }
490 }
491
492 if ( r_lightmap->integer == 2 ) {
493 ri.Printf( PRINT_ALL, "Brightest lightmap value: %d\n", ( int ) ( maxIntensity * 255 ) );
494 }
495
496 ri.Free(image);
497}

500static float FatPackU(float input, int lightmapnum)
501{
502 if (lightmapnum < 0)
503 return input;
504
505 if (tr.worldDeluxeMapping)
506 lightmapnum >>= 1;
507
508 if (tr.fatLightmapCols > 0)
509 {
510 lightmapnum %= (tr.fatLightmapCols * tr.fatLightmapRows);
511 return (input + (lightmapnum % tr.fatLightmapCols)) / (float)(tr.fatLightmapCols);
512 }
513
514 return input;
515}

517static float FatPackV(float input, int lightmapnum)
518{
519 if (lightmapnum < 0)
520 return input;
521
522 if (tr.worldDeluxeMapping)
523 lightmapnum >>= 1;
524
525 if (tr.fatLightmapCols > 0)
526 {
527 lightmapnum %= (tr.fatLightmapCols * tr.fatLightmapRows);
528 return (input + (lightmapnum / tr.fatLightmapCols)) / (float)(tr.fatLightmapRows);
529 }
530
531 return input;
532}

601static shader_t *ShaderForShaderNum( int shaderNum, int lightmapNum ) {
602 shader_t *shader;
603 dshader_t *dsh;
604
605 int _shaderNum = LittleLong( shaderNum );
606 if ( _shaderNum < 0 || _shaderNum >= s_worldData.numShaders ) {
607 ri.Error( ERR_DROP, "ShaderForShaderNum: bad num %i", _shaderNum );
608 }
609 dsh = &s_worldData.shaders[ _shaderNum ];
610
611 if ( r_vertexLight->integer || glConfig.hardwareType == GLHW_PERMEDIA2 ) {
612 lightmapNum = LIGHTMAP_BY_VERTEX;
613 }
614
615 if ( r_fullbright->integer ) {
616 lightmapNum = LIGHTMAP_WHITEIMAGE;
617 }
618
619 shader = R_FindShader( dsh->shader, lightmapNum, qtrue );
620
621 // if the shader had errors, just use default shader
622 if ( shader->defaultShader ) {
623 return tr.defaultShader;
624 }
625
626 return shader;
627}

629void LoadDrawVertToSrfVert(srfVert_t *s, drawVert_t *d, int realLightmapNum, float hdrVertColors[3], vec3_t *bounds)
630{
631 vec4_t v;
632
633 s->xyz[0] = LittleFloat(d->xyz[0]);
634 s->xyz[1] = LittleFloat(d->xyz[1]);
635 s->xyz[2] = LittleFloat(d->xyz[2]);
636
637 if (bounds)
638 AddPointToBounds(s->xyz, bounds[0], bounds[1]);
639
640 s->st[0] = LittleFloat(d->st[0]);
641 s->st[1] = LittleFloat(d->st[1]);
642
643 if (realLightmapNum >= 0)
644 {
645 s->lightmap[0] = FatPackU(LittleFloat(d->lightmap[0]), realLightmapNum);
646 s->lightmap[1] = FatPackV(LittleFloat(d->lightmap[1]), realLightmapNum);
647 }
648 else
649 {
650 s->lightmap[0] = LittleFloat(d->lightmap[0]);
651 s->lightmap[1] = LittleFloat(d->lightmap[1]);
652 }
653
654 v[0] = LittleFloat(d->normal[0]);
655 v[1] = LittleFloat(d->normal[1]);
656 v[2] = LittleFloat(d->normal[2]);
657
658 R_VaoPackNormal(s->normal, v);
659
660 if (hdrVertColors)
661 {
662 v[0] = hdrVertColors[0];
663 v[1] = hdrVertColors[1];
664 v[2] = hdrVertColors[2];
665 }
666 else
667 {
668 //hack: convert LDR vertex colors to HDR
669 if (r_hdr->integer)
670 {
671 v[0] = MAX(d->color[0], 0.499f);
672 v[1] = MAX(d->color[1], 0.499f);
673 v[2] = MAX(d->color[2], 0.499f);
674 }
675 else
676 {
677 v[0] = d->color[0];
678 v[1] = d->color[1];
679 v[2] = d->color[2];
680 }
681
682 }
683 v[3] = d->color[3] / 255.0f;
684
685 R_ColorShiftLightingFloats(v, v);
686 R_VaoPackColor(s->color, v);
687}

690/*
691===============
692ParseFace
693===============
694*/
695static void ParseFace( dsurface_t *ds, drawVert_t *verts, float *hdrVertColors, msurface_t *surf, int *indexes ) {
696 int i, j;
697 srfBspSurface_t *cv;
698 glIndex_t *tri;
699 int numVerts, numIndexes, badTriangles;
700 int realLightmapNum;
701
702 realLightmapNum = LittleLong( ds->lightmapNum );
703
704 // get fog volume
705 surf->fogIndex = LittleLong( ds->fogNum ) + 1;
706
707 // get shader value
708 surf->shader = ShaderForShaderNum( ds->shaderNum, FatLightmap(realLightmapNum) );
709 if ( r_singleShader->integer && !surf->shader->isSky ) {
710 surf->shader = tr.defaultShader;
711 }
712
713 numVerts = LittleLong(ds->numVerts);
714 if (numVerts > MAX_FACE_POINTS) {
715 ri.Printf( PRINT_WARNING, "WARNING: MAX_FACE_POINTS exceeded: %i\n", numVerts);
716 numVerts = MAX_FACE_POINTS;
717 surf->shader = tr.defaultShader;
718 }
719
720 numIndexes = LittleLong(ds->numIndexes);
721
722 //cv = ri.Hunk_Alloc(sizeof(*cv), h_low);
723 cv = (void *)surf->data;
724 cv->surfaceType = SF_FACE;
725
726 cv->numIndexes = numIndexes;
727 cv->indexes = ri.Hunk_Alloc(numIndexes * sizeof(cv->indexes[0]), h_low);
728
729 cv->numVerts = numVerts;
730 cv->verts = ri.Hunk_Alloc(numVerts * sizeof(cv->verts[0]), h_low);
731
732 // copy vertexes
733 surf->cullinfo.type = CULLINFO_PLANE | CULLINFO_BOX;
734 ClearBounds(surf->cullinfo.bounds[0], surf->cullinfo.bounds[1]);
735 verts += LittleLong(ds->firstVert);
736 for(i = 0; i < numVerts; i++)
737 LoadDrawVertToSrfVert(&cv->verts[i], &verts[i], realLightmapNum, hdrVertColors ? hdrVertColors + (ds->firstVert + i) * 3 : NULL, surf->cullinfo.bounds);
738
739 // copy triangles
740 badTriangles = 0;
741 indexes += LittleLong(ds->firstIndex);
742 for(i = 0, tri = cv->indexes; i < numIndexes; i += 3, tri += 3)
743 {

799static void ParseMesh ( dsurface_t *ds, drawVert_t *verts, float *hdrVertColors, msurface_t *surf ) {
800 srfBspSurface_t *grid = (srfBspSurface_t *)surf->data;
801 int i;
802 int width, height, numPoints;
803 srfVert_t points[MAX_PATCH_SIZE*MAX_PATCH_SIZE];
804 vec3_t bounds[2];
805 vec3_t tmpVec;
806 static surfaceType_t skipData = SF_SKIP;
807 int realLightmapNum;
808
809 realLightmapNum = LittleLong( ds->lightmapNum );
810
811 // get fog volume
812 surf->fogIndex = LittleLong( ds->fogNum ) + 1;
813
814 // get shader value
815 surf->shader = ShaderForShaderNum( ds->shaderNum, FatLightmap(realLightmapNum) );
816 if ( r_singleShader->integer && !surf->shader->isSky ) {
817 surf->shader = tr.defaultShader;
818 }
819
820 // we may have a nodraw surface, because they might still need to
821 // be around for movement clipping
822 if ( s_worldData.shaders[ LittleLong( ds->shaderNum ) ].surfaceFlags & SURF_NODRAW ) {
823 surf->data = &skipData;
824 return;
825 }
826
827 width = LittleLong( ds->patchWidth );
828 height = LittleLong( ds->patchHeight );
829
830 if(width < 0 || width > MAX_PATCH_SIZE || height < 0 || height > MAX_PATCH_SIZE)
831 ri.Error(ERR_DROP, "ParseMesh: bad size");
832
833 verts += LittleLong( ds->firstVert );
834 numPoints = width * height;
835 for(i = 0; i < numPoints; i++)
836 LoadDrawVertToSrfVert(&points[i], &verts[i], realLightmapNum, hdrVertColors ? hdrVertColors + (ds->firstVert + i) * 3 : NULL, NULL);
837
838 // pre-tesseleate
839 R_SubdividePatchToGrid( grid, width, height, points );
840
841 // copy the level of detail origin, which is the center
842 // of the group of all curves that must subdivide the same
843 // to avoid cracking
844 for ( i = 0 ; i < 3 ; i++ ) {
845 bounds[0][i] = LittleFloat( ds->lightmapVecs[0][i] );
846 bounds[1][i] = LittleFloat( ds->lightmapVecs[1][i] );
847 }
848 VectorAdd( bounds[0], bounds[1], bounds[1] );
849 VectorScale( bounds[1], 0.5f, grid->lodOrigin );
850 VectorSubtract( bounds[0], grid->lodOrigin, tmpVec );
851 grid->lodRadius = VectorLength( tmpVec );
852
853 surf->cullinfo.type = CULLINFO_BOX | CULLINFO_SPHERE;
854 VectorCopy(grid->cullBounds[0], surf->cullinfo.bounds[0]);
855 VectorCopy(grid->cullBounds[1], surf->cullinfo.bounds[1]);
856 VectorCopy(grid->cullOrigin, surf->cullinfo.localOrigin);
857 surf->cullinfo.radius = grid->cullRadius;
858}

1570/*
1571===============
1572R_TryStitchPatch
1573
1574This function will try to stitch patches in the same LoD group together for the highest LoD.
1575
1576Only single missing vertice cracks will be fixed.
1577
1578Vertices will be joined at the patch side a crack is first found, at the other side
1579of the patch (on the same row or column) the vertices will not be joined and cracks
1580might still appear at that side.
1581===============
1582*/
1583int R_TryStitchingPatch( int grid1num ) {
1584 int j, numstitches;
1585 srfBspSurface_t *grid1, *grid2;
1586
1587 numstitches = 0;
1588 grid1 = (srfBspSurface_t *) s_worldData.surfaces[grid1num].data;
1589 for ( j = 0; j < s_worldData.numsurfaces; j++ ) {
1590 //
1591 grid2 = (srfBspSurface_t *) s_worldData.surfaces[j].data;
1592 // if this surface is not a grid
1593 if ( grid2->surfaceType != SF_GRID ) continue;
1594 // grids in the same LOD group should have the exact same lod radius
1595 if ( grid1->lodRadius != grid2->lodRadius ) continue;
1596 // grids in the same LOD group should have the exact same lod origin
1597 if ( grid1->lodOrigin[0] != grid2->lodOrigin[0] ) continue;
1598 if ( grid1->lodOrigin[1] != grid2->lodOrigin[1] ) continue;
1599 if ( grid1->lodOrigin[2] != grid2->lodOrigin[2] ) continue;
1600 //
1601 while (R_StitchPatches(grid1num, j))
1602 {
1603 numstitches++;
1604 }
1605 }
1606 return numstitches;
1607}

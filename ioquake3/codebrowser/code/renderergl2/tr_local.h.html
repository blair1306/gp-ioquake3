1771static ID_INLINE qboolean ShaderRequiresCPUDeforms(const shader_t * shader)
1772{
1773 if(shader->numDeforms)
1774 {
1775 const deformStage_t *ds = &shader->deforms[0];
1776
1777 if (shader->numDeforms > 1)
1778 return qtrue;
1779
1780 switch (ds->deformation)
1781 {
1782 case DEFORM_WAVE:
1783 case DEFORM_BULGE:
1784 // need CPU deforms at high level-times to avoid floating point percision loss
1785 return ( backEnd.refdef.floatTime != (float)backEnd.refdef.floatTime );
1786
1787 default:
1788 return qtrue;
1789 }
1790 }
1791
1792 return qfalse;
1793}

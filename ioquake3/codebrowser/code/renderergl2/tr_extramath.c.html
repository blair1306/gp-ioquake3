29void Mat4Zero( mat4_t out )
30{
31 out[ 0] = 0.0f; out[ 4] = 0.0f; out[ 8] = 0.0f; out[12] = 0.0f;
32 out[ 1] = 0.0f; out[ 5] = 0.0f; out[ 9] = 0.0f; out[13] = 0.0f;
33 out[ 2] = 0.0f; out[ 6] = 0.0f; out[10] = 0.0f; out[14] = 0.0f;
34 out[ 3] = 0.0f; out[ 7] = 0.0f; out[11] = 0.0f; out[15] = 0.0f;
35}

37void Mat4Identity( mat4_t out )
38{
39 out[ 0] = 1.0f; out[ 4] = 0.0f; out[ 8] = 0.0f; out[12] = 0.0f;
40 out[ 1] = 0.0f; out[ 5] = 1.0f; out[ 9] = 0.0f; out[13] = 0.0f;
41 out[ 2] = 0.0f; out[ 6] = 0.0f; out[10] = 1.0f; out[14] = 0.0f;
42 out[ 3] = 0.0f; out[ 7] = 0.0f; out[11] = 0.0f; out[15] = 1.0f;
43}

45void Mat4Copy( const mat4_t in, mat4_t out )
46{
47 out[ 0] = in[ 0]; out[ 4] = in[ 4]; out[ 8] = in[ 8]; out[12] = in[12];
48 out[ 1] = in[ 1]; out[ 5] = in[ 5]; out[ 9] = in[ 9]; out[13] = in[13];
49 out[ 2] = in[ 2]; out[ 6] = in[ 6]; out[10] = in[10]; out[14] = in[14];
50 out[ 3] = in[ 3]; out[ 7] = in[ 7]; out[11] = in[11]; out[15] = in[15];
51}

53void Mat4Multiply( const mat4_t in1, const mat4_t in2, mat4_t out )
54{
55 out[ 0] = in1[ 0] * in2[ 0] + in1[ 4] * in2[ 1] + in1[ 8] * in2[ 2] + in1[12] * in2[ 3];
56 out[ 1] = in1[ 1] * in2[ 0] + in1[ 5] * in2[ 1] + in1[ 9] * in2[ 2] + in1[13] * in2[ 3];
57 out[ 2] = in1[ 2] * in2[ 0] + in1[ 6] * in2[ 1] + in1[10] * in2[ 2] + in1[14] * in2[ 3];
58 out[ 3] = in1[ 3] * in2[ 0] + in1[ 7] * in2[ 1] + in1[11] * in2[ 2] + in1[15] * in2[ 3];
59
60 out[ 4] = in1[ 0] * in2[ 4] + in1[ 4] * in2[ 5] + in1[ 8] * in2[ 6] + in1[12] * in2[ 7];
61 out[ 5] = in1[ 1] * in2[ 4] + in1[ 5] * in2[ 5] + in1[ 9] * in2[ 6] + in1[13] * in2[ 7];
62 out[ 6] = in1[ 2] * in2[ 4] + in1[ 6] * in2[ 5] + in1[10] * in2[ 6] + in1[14] * in2[ 7];
63 out[ 7] = in1[ 3] * in2[ 4] + in1[ 7] * in2[ 5] + in1[11] * in2[ 6] + in1[15] * in2[ 7];
64
65 out[ 8] = in1[ 0] * in2[ 8] + in1[ 4] * in2[ 9] + in1[ 8] * in2[10] + in1[12] * in2[11];
66 out[ 9] = in1[ 1] * in2[ 8] + in1[ 5] * in2[ 9] + in1[ 9] * in2[10] + in1[13] * in2[11];
67 out[10] = in1[ 2] * in2[ 8] + in1[ 6] * in2[ 9] + in1[10] * in2[10] + in1[14] * in2[11];
68 out[11] = in1[ 3] * in2[ 8] + in1[ 7] * in2[ 9] + in1[11] * in2[10] + in1[15] * in2[11];
69
70 out[12] = in1[ 0] * in2[12] + in1[ 4] * in2[13] + in1[ 8] * in2[14] + in1[12] * in2[15];
71 out[13] = in1[ 1] * in2[12] + in1[ 5] * in2[13] + in1[ 9] * in2[14] + in1[13] * in2[15];
72 out[14] = in1[ 2] * in2[12] + in1[ 6] * in2[13] + in1[10] * in2[14] + in1[14] * in2[15];
73 out[15] = in1[ 3] * in2[12] + in1[ 7] * in2[13] + in1[11] * in2[14] + in1[15] * in2[15];
74}

76void Mat4Transform( const mat4_t in1, const vec4_t in2, vec4_t out )
77{
78 out[ 0] = in1[ 0] * in2[ 0] + in1[ 4] * in2[ 1] + in1[ 8] * in2[ 2] + in1[12] * in2[ 3];
79 out[ 1] = in1[ 1] * in2[ 0] + in1[ 5] * in2[ 1] + in1[ 9] * in2[ 2] + in1[13] * in2[ 3];
80 out[ 2] = in1[ 2] * in2[ 0] + in1[ 6] * in2[ 1] + in1[10] * in2[ 2] + in1[14] * in2[ 3];
81 out[ 3] = in1[ 3] * in2[ 0] + in1[ 7] * in2[ 1] + in1[11] * in2[ 2] + in1[15] * in2[ 3];
82}

84qboolean Mat4Compare( const mat4_t a, const mat4_t b )
85{
86 return !(a[ 0] != b[ 0] || a[ 4] != b[ 4] || a[ 8] != b[ 8] || a[12] != b[12] ||
87 a[ 1] != b[ 1] || a[ 5] != b[ 5] || a[ 9] != b[ 9] || a[13] != b[13] ||
88 a[ 2] != b[ 2] || a[ 6] != b[ 6] || a[10] != b[10] || a[14] != b[14] ||
89 a[ 3] != b[ 3] || a[ 7] != b[ 7] || a[11] != b[11] || a[15] != b[15]);
90}

100void Mat4Translation( vec3_t vec, mat4_t out )
101{
102 out[ 0] = 1.0f; out[ 4] = 0.0f; out[ 8] = 0.0f; out[12] = vec[0];
103 out[ 1] = 0.0f; out[ 5] = 1.0f; out[ 9] = 0.0f; out[13] = vec[1];
104 out[ 2] = 0.0f; out[ 6] = 0.0f; out[10] = 1.0f; out[14] = vec[2];

108void Mat4Ortho( float left, float right, float bottom, float top, float znear, float zfar, mat4_t out )
109{
110 out[ 0] = 2.0f / (right - left); out[ 4] = 0.0f; out[ 8] = 0.0f; out[12] = -(right + left) / (right - left);
111 out[ 1] = 0.0f; out[ 5] = 2.0f / (top - bottom); out[ 9] = 0.0f; out[13] = -(top + bottom) / (top - bottom);
112 out[ 2] = 0.0f; out[ 6] = 0.0f; out[10] = 2.0f / (zfar - znear); out[14] = -(zfar + znear) / (zfar - znear);
113 out[ 3] = 0.0f; out[ 7] = 0.0f; out[11] = 0.0f; out[15] = 1.0f;
114}

116void Mat4View(vec3_t axes[3], vec3_t origin, mat4_t out)
117{
118 out[0] = axes[0][0];
119 out[1] = axes[1][0];
120 out[2] = axes[2][0];
121 out[3] = 0;
122
123 out[4] = axes[0][1];
124 out[5] = axes[1][1];
125 out[6] = axes[2][1];
126 out[7] = 0;
127
128 out[8] = axes[0][2];
129 out[9] = axes[1][2];
130 out[10] = axes[2][2];
131 out[11] = 0;
132
133 out[12] = -DotProduct(origin, axes[0]);
134 out[13] = -DotProduct(origin, axes[1]);
135 out[14] = -DotProduct(origin, axes[2]);
136 out[15] = 1;
137}

139void Mat4SimpleInverse( const mat4_t in, mat4_t out)
140{
141 vec3_t v;
142 float invSqrLen;
143
144 VectorCopy(in + 0, v);
145 invSqrLen = 1.0f / DotProduct(v, v); VectorScale(v, invSqrLen, v);
146 out[ 0] = v[0]; out[ 4] = v[1]; out[ 8] = v[2]; out[12] = -DotProduct(v, &in[12]);
147
148 VectorCopy(in + 4, v);
149 invSqrLen = 1.0f / DotProduct(v, v); VectorScale(v, invSqrLen, v);
150 out[ 1] = v[0]; out[ 5] = v[1]; out[ 9] = v[2]; out[13] = -DotProduct(v, &in[12]);
151
152 VectorCopy(in + 8, v);
153 invSqrLen = 1.0f / DotProduct(v, v); VectorScale(v, invSqrLen, v);
154 out[ 2] = v[0]; out[ 6] = v[1]; out[10] = v[2]; out[14] = -DotProduct(v, &in[12]);
155
156 out[ 3] = 0.0f; out[ 7] = 0.0f; out[11] = 0.0f; out[15] = 1.0f;
157}

159void VectorLerp( vec3_t a, vec3_t b, float lerp, vec3_t c)
160{
161 c[0] = a[0] * (1.0f - lerp) + b[0] * lerp;
162 c[1] = a[1] * (1.0f - lerp) + b[1] * lerp;
163 c[2] = a[2] * (1.0f - lerp) + b[2] * lerp;
164}

166qboolean SpheresIntersect(vec3_t origin1, float radius1, vec3_t origin2, float radius2)
167{
168 float radiusSum = radius1 + radius2;
169 vec3_t diff;
170
171 VectorSubtract(origin1, origin2, diff);
172
173 if (DotProduct(diff, diff) <= radiusSum * radiusSum)
174 {
175 return qtrue;
176 }
177
178 return qfalse;
179}

181void BoundingSphereOfSpheres(vec3_t origin1, float radius1, vec3_t origin2, float radius2, vec3_t origin3, float *radius3)
182{
183 vec3_t diff;
184
185 VectorScale(origin1, 0.5f, origin3);
186 VectorMA(origin3, 0.5f, origin2, origin3);
187
188 VectorSubtract(origin1, origin2, diff);
189 *radius3 = VectorLength(diff) * 0.5f + MAX(radius1, radius2);
190}

202union f32_u {
203 float f;
204 uint32_t ui;
205 struct {
206 unsigned int fraction:23;
207 unsigned int exponent:8;
208 unsigned int sign:1;
209 } pack;
210};
211
212union f16_u {
213 uint16_t ui;
214 struct {
215 unsigned int fraction:10;
216 unsigned int exponent:5;
217 unsigned int sign:1;
218 } pack;
219};
220
221uint16_t FloatToHalf(float in)
222{
223 union f32_u f32;
224 union f16_u f16;
225
226 f32.f = in;
227
228 f16.pack.exponent = CLAMP((int)(f32.pack.exponent) - 112, 0, 31);
229 f16.pack.fraction = f32.pack.fraction >> 13;
230 f16.pack.sign = f32.pack.sign;
231
232 return f16.ui;
233}
234
235float HalfToFloat(uint16_t in)
236{
237 union f32_u f32;
238 union f16_u f16;
239
240 f16.ui = in;
241
242 f32.pack.exponent = (int)(f16.pack.exponent) + 112;
243 f32.pack.fraction = f16.pack.fraction << 13;
244 f32.pack.sign = f16.pack.sign;
245
246 return f32.f;
247}

40/*
41================
42return a hash value for the filename
43================
44*/
45#ifdef __GNUCC__
46 #warning TODO: check if long is ok here
47#endif
48static long generateHashValue( const char *fname, const int size ) {
49 int i;
50 long hash;
51 char letter;
52
53 hash = 0;
54 i = 0;
55 while (fname[i] != '\0') {
56 letter = tolower(fname[i]);
57 if (letter =='.') break; // don't include extension
58 if (letter =='\\') letter = '/'; // damn path names
59 if (letter == PATH_SEP) letter = '/'; // damn path names
60 hash+=(long)(letter)*(i+119);
61 i++;
62 }
63 hash = (hash ^ (hash >> 10) ^ (hash >> 20));
64 hash &= (size-1);
65 return hash;
66}

68void R_RemapShader(const char *shaderName, const char *newShaderName, const char *timeOffset) {
69 char strippedName[MAX_QPATH];
70 int hash;
71 shader_t *sh, *sh2;
72 qhandle_t h;
73
74 sh = R_FindShaderByName( shaderName );
75 if (sh == NULL || sh == tr.defaultShader) {
76 h = RE_RegisterShaderLightMap(shaderName, 0);
77 sh = R_GetShaderByHandle(h);
78 }
79 if (sh == NULL || sh == tr.defaultShader) {
80 ri.Printf( PRINT_WARNING, "WARNING: R_RemapShader: shader %s not found\n", shaderName );
81 return;
82 }
83
84 sh2 = R_FindShaderByName( newShaderName );
85 if (sh2 == NULL || sh2 == tr.defaultShader) {
86 h = RE_RegisterShaderLightMap(newShaderName, 0);
87 sh2 = R_GetShaderByHandle(h);
88 }
89
90 if (sh2 == NULL || sh2 == tr.defaultShader) {
91 ri.Printf( PRINT_WARNING, "WARNING: R_RemapShader: new shader %s not found\n", newShaderName );
92 return;
93 }
94
95 // remap all the shaders with the given name
96 // even tho they might have different lightmaps
97 COM_StripExtension(shaderName, strippedName, sizeof(strippedName));
98 hash = generateHashValue(strippedName, FILE_HASH_SIZE);
99 for (sh = hashTable[hash]; sh; sh = sh->next) {
100 if (Q_stricmp(sh->name, strippedName) == 0) {
101 if (sh != sh2) {
102 sh->remappedShader = sh2;
103 } else {
104 sh->remappedShader = NULL;
105 }
106 }
107 }
108 if (timeOffset) {
109 sh2->timeOffset = atof(timeOffset);
110 }
111}

113/*
114===============
115ParseVector
116===============
117*/
118static qboolean ParseVector( char **text, int count, float *v ) {
119 char *token;
120 int i;
121
122 // FIXME: spaces are currently required after parens, should change parseext...
123 token = COM_ParseExt( text, qfalse );
124 if ( strcmp( token, "(" ) ) {
125 ri.Printf( PRINT_WARNING, "WARNING: missing parenthesis in shader '%s'\n", shader.name );
126 return qfalse;
127 }
128
129 for ( i = 0 ; i < count ; i++ ) {
130 token = COM_ParseExt( text, qfalse );
131 if ( !token[0] ) {
132 ri.Printf( PRINT_WARNING, "WARNING: missing vector element in shader '%s'\n", shader.name );
133 return qfalse;
134 }
135 v[i] = atof( token );
136 }
137
138 token = COM_ParseExt( text, qfalse );
139 if ( strcmp( token, ")" ) ) {
140 ri.Printf( PRINT_WARNING, "WARNING: missing parenthesis in shader '%s'\n", shader.name );
141 return qfalse;
142 }
143
144 return qtrue;
145}

596/*
597===================
598ParseStage
599===================
600*/
601static qboolean ParseStage( shaderStage_t *stage, char **text )
602{
603 char *token;
604 int depthMaskBits = GLS_DEPTHMASK_TRUE, blendSrcBits = 0, blendDstBits = 0, atestBits = 0, depthFuncBits = 0;
605 qboolean depthMaskExplicit = qfalse;
606
607 stage->active = qtrue;
608
609 while ( 1 )
610 {
611 token = COM_ParseExt( text, qtrue );
612 if ( !token[0] )
613 {
614 ri.Printf( PRINT_WARNING, "WARNING: no matching '}' found\n" );
615 return qfalse;
616 }
617
618 if ( token[0] == '}' )
619 {
620 break;
621 }
622 //
623 // map <name>
624 //
625 else if ( !Q_stricmp( token, "map" ) )
626 {
627 token = COM_ParseExt( text, qfalse );
628 if ( !token[0] )
629 {
630 ri.Printf( PRINT_WARNING, "WARNING: missing parameter for 'map' keyword in shader '%s'\n", shader.name );
631 return qfalse;
632 }
633
634 if ( !Q_stricmp( token, "$whiteimage" ) )
635 {
636 stage->bundle[0].image[0] = tr.whiteImage;
637 continue;
638 }
639 else if ( !Q_stricmp( token, "$lightmap" ) )
640 {
641 stage->bundle[0].isLightmap = qtrue;
642 if ( shader.lightmapIndex < 0 || !tr.lightmaps ) {
643 stage->bundle[0].image[0] = tr.whiteImage;
644 } else {
645 stage->bundle[0].image[0] = tr.lightmaps[shader.lightmapIndex];
646 }
647 continue;
648 }
649 else if ( !Q_stricmp( token, "$deluxemap" ) )
650 {
651 if (!tr.worldDeluxeMapping)
652 {
653 ri.Printf( PRINT_WARNING, "WARNING: shader '%s' wants a deluxe map in a map compiled without them\n", shader.name );
654 return qfalse;
655 }
656
657 stage->bundle[0].isLightmap = qtrue;
658 if ( shader.lightmapIndex < 0 ) {
659 stage->bundle[0].image[0] = tr.whiteImage;
660 } else {
661 stage->bundle[0].image[0] = tr.deluxemaps[shader.lightmapIndex];
662 }
663 continue;
664 }
665 else
666 {
667 imgType_t type = IMGTYPE_COLORALPHA;
668 imgFlags_t flags = IMGFLAG_NONE;
669
670 if (!shader.noMipMaps)
671 flags |= IMGFLAG_MIPMAP;
672
673 if (!shader.noPicMip)
674 flags |= IMGFLAG_PICMIP;
675
676 if (stage->type == ST_NORMALMAP || stage->type == ST_NORMALPARALLAXMAP)
677 {
678 type = IMGTYPE_NORMAL;
679 flags |= IMGFLAG_NOLIGHTSCALE;
680
681 if (stage->type == ST_NORMALPARALLAXMAP)
682 type = IMGTYPE_NORMALHEIGHT;
683 }
684 else
685 {
686 if (r_genNormalMaps->integer)
687 flags |= IMGFLAG_GENNORMALMAP;
688 }
689
690 stage->bundle[0].image[0] = R_FindImageFile( token, type, flags );
691
692 if ( !stage->bundle[0].image[0] )
693 {
694 ri.Printf( PRINT_WARNING, "WARNING: R_FindImageFile could not find '%s' in shader '%s'\n", token, shader.name );
695 return qfalse;
696 }
697 }
698 }
699 //
700 // clampmap <name>
701 //
702 else if ( !Q_stricmp( token, "clampmap" ) )
703 {
704 imgType_t type = IMGTYPE_COLORALPHA;
705 imgFlags_t flags = IMGFLAG_CLAMPTOEDGE;
706
707 token = COM_ParseExt( text, qfalse );
708 if ( !token[0] )
709 {
710 ri.Printf( PRINT_WARNING, "WARNING: missing parameter for 'clampmap' keyword in shader '%s'\n", shader.name );
711 return qfalse;
712 }
713
714 if (!shader.noMipMaps)
715 flags |= IMGFLAG_MIPMAP;
716
717 if (!shader.noPicMip)
718 flags |= IMGFLAG_PICMIP;
719
720 if (stage->type == ST_NORMALMAP || stage->type == ST_NORMALPARALLAXMAP)
721 {
722 type = IMGTYPE_NORMAL;
723 flags |= IMGFLAG_NOLIGHTSCALE;
724
725 if (stage->type == ST_NORMALPARALLAXMAP)
726 type = IMGTYPE_NORMALHEIGHT;
727 }
728 else
729 {
730 if (r_genNormalMaps->integer)
731 flags |= IMGFLAG_GENNORMALMAP;
732 }
733
734
735 stage->bundle[0].image[0] = R_FindImageFile( token, type, flags );
736 if ( !stage->bundle[0].image[0] )
737 {
738 ri.Printf( PRINT_WARNING, "WARNING: R_FindImageFile could not find '%s' in shader '%s'\n", token, shader.name );
739 return qfalse;
740 }
741 }
742 //
743 // animMap <frequency> <image1> .... <imageN>
744 //
745 else if ( !Q_stricmp( token, "animMap" ) )
746 {
747 int totalImages = 0;
748
749 token = COM_ParseExt( text, qfalse );
750 if ( !token[0] )
751 {
752 ri.Printf( PRINT_WARNING, "WARNING: missing parameter for 'animMap' keyword in shader '%s'\n", shader.name );
753 return qfalse;
754 }
755 stage->bundle[0].imageAnimationSpeed = atof( token );
756
757 // parse up to MAX_IMAGE_ANIMATIONS animations
758 while ( 1 ) {

1408deformVertexes wave <spread> <waveform> <base> <amplitude> <phase> <frequency>
1409deformVertexes normal <frequency> <amplitude>
1410deformVertexes move <vector> <waveform> <base> <amplitude> <phase> <frequency>
1411deformVertexes bulge <bulgeWidth> <bulgeHeight> <bulgeSpeed>
1412deformVertexes projectionShadow
1413deformVertexes autoSprite
1414deformVertexes autoSprite2
1415deformVertexes text[0-7]
1416===============
1417*/
1418static void ParseDeform( char **text ) {
1419 char *token;
1420 deformStage_t *ds;
1421
1422 token = COM_ParseExt( text, qfalse );
1423 if ( token[0] == 0 )
1424 {
1425 ri.Printf( PRINT_WARNING, "WARNING: missing deform parm in shader '%s'\n", shader.name );
1426 return;
1427 }
1428

1430 ri.Printf( PRINT_WARNING, "WARNING: MAX_SHADER_DEFORMS in '%s'\n", shader.name );
1431 return;
1432 }
1433
1434 ds = &shader.deforms[ shader.numDeforms ];
1435 shader.numDeforms++;
1436
1437 if ( !Q_stricmp( token, "projectionShadow" ) ) {
1438 ds->deformation = DEFORM_PROJECTION_SHADOW;
1439 return;
1440 }
1441
1442 if ( !Q_stricmp( token, "autosprite" ) ) {
1443 ds->deformation = DEFORM_AUTOSPRITE;
1444 return;
1445 }
1446
1447 if ( !Q_stricmp( token, "autosprite2" ) ) {
1448 ds->deformation = DEFORM_AUTOSPRITE2;
1449 return;
1450 }
1451
1452 if ( !Q_stricmpn( token, "text", 4 ) ) {
1453 int n;
1454
1455 n = token[4] - '0';
1456 if ( n < 0 || n > 7 ) {
1457 n = 0;
1458 }
1459 ds->deformation = DEFORM_TEXT0 + n;
1460 return;
1461 }
1462
1463 if ( !Q_stricmp( token, "bulge" ) ) {
1464 token = COM_ParseExt( text, qfalse );

1742/*
1743=================
1744ParseShader
1745
1746The current text pointer is at the explicit text definition of the
1747shader. Parse it into the global shader variable. Later functions
1748will optimize it.
1749=================
1750*/
1751static qboolean ParseShader( char **text )
1752{
1753 char *token;
1754 int s;
1755
1756 s = 0;
1757
1758 token = COM_ParseExt( text, qtrue );
1759 if ( token[0] != '{' )
1760 {
1761 ri.Printf( PRINT_WARNING, "WARNING: expecting '{', found '%s' instead in shader '%s'\n", token, shader.name );
1762 return qfalse;
1763 }
1764
1765 while ( 1 )
1766 {
1767 token = COM_ParseExt( text, qtrue );
1768 if ( !token[0] )
1769 {
1770 ri.Printf( PRINT_WARNING, "WARNING: no concluding '}' in shader %s\n", shader.name );
1771 return qfalse;
1772 }
1773
1774 // end of shader definition
1775 if ( token[0] == '}' )
1776 {
1777 break;
1778 }
1779 // stage definition
1780 else if ( token[0] == '{' )
1781 {
1782 if ( s >= MAX_SHADER_STAGES ) {
1783 ri.Printf( PRINT_WARNING, "WARNING: too many stages in shader %s (max is %i)\n", shader.name, MAX_SHADER_STAGES );
1784 return qfalse;
1785 }
1786
1787 if ( !ParseStage( &stages[s], text ) )
1788 {
1789 return qfalse;
1790 }
1791 stages[s].active = qtrue;
1792 s++;
1793
1794 continue;
1795 }
1796 // skip stuff that only the QuakeEdRadient needs
1797 else if ( !Q_stricmpn( token, "qer", 3 ) ) {
1798 SkipRestOfLine( text );
1799 continue;
1800 }
1801 // sun parms
1802 else if ( !Q_stricmp( token, "q3map_sun" ) || !Q_stricmp( token, "q3map_sunExt" ) || !Q_stricmp( token, "q3gl2_sun" ) ) {
1803 float a, b;
1804 qboolean isGL2Sun = qfalse;
1805
1806 if (!Q_stricmp( token, "q3gl2_sun" ) && r_sunShadows->integer )
1807 {
1808 isGL2Sun = qtrue;
1809 tr.sunShadows = qtrue;
1810 }
1811
1812 token = COM_ParseExt( text, qfalse );
1813 tr.sunLight[0] = atof( token );
1814 token = COM_ParseExt( text, qfalse );
1815 tr.sunLight[1] = atof( token );
1816 token = COM_ParseExt( text, qfalse );
1817 tr.sunLight[2] = atof( token );
1818
1819 VectorNormalize( tr.sunLight );
1820
1821 token = COM_ParseExt( text, qfalse );
1822 a = atof( token );
1823 VectorScale( tr.sunLight, a, tr.sunLight);
1824
1825 token = COM_ParseExt( text, qfalse );
1826 a = atof( token );
1827 a = a / 180 * M_PI;
1828
1829 token = COM_ParseExt( text, qfalse );
1830 b = atof( token );
1831 b = b / 180 * M_PI;
1832
1833 tr.sunDirection[0] = cos( a ) * cos( b );
1834 tr.sunDirection[1] = sin( a ) * cos( b );
1835 tr.sunDirection[2] = sin( b );
1836
1837 if (isGL2Sun)
1838 {
1839 token = COM_ParseExt( text, qfalse );
1840 tr.sunShadowScale = atof(token);
1841
1842 // parse twice, since older shaders may include mapLightScale before sunShadowScale
1843 token = COM_ParseExt( text, qfalse );
1844 if (token[0])
1845 tr.sunShadowScale = atof(token);
1846 }
1847
1848 SkipRestOfLine( text );
1849 continue;
1850 }
1851 // tonemap parms
1852 else if ( !Q_stricmp( token, "q3gl2_tonemap" ) ) {
1853 token = COM_ParseExt( text, qfalse );
1854 tr.toneMinAvgMaxLevel[0] = atof( token );
1855 token = COM_ParseExt( text, qfalse );
1856 tr.toneMinAvgMaxLevel[1] = atof( token );
1857 token = COM_ParseExt( text, qfalse );
1858 tr.toneMinAvgMaxLevel[2] = atof( token );
1859
1860 token = COM_ParseExt( text, qfalse );
1861 tr.autoExposureMinMax[0] = atof( token );
1862 token = COM_ParseExt( text, qfalse );
1863 tr.autoExposureMinMax[1] = atof( token );
1864
1865 SkipRestOfLine( text );
1866 continue;
1867 }
1868 else if ( !Q_stricmp( token, "deformVertexes" ) ) {
1869 ParseDeform( text );
1870 continue;
1871 }
1872 else if ( !Q_stricmp( token, "tesssize" ) ) {
1873 SkipRestOfLine( text );
1874 continue;
1875 }
1876 else if ( !Q_stricmp( token, "clampTime" ) ) {
1877 token = COM_ParseExt( text, qfalse );
1878 if (token[0]) {
1879 shader.clampTime = atof(token);
1880 }
1881 }
1882 // skip stuff that only the q3map needs
1883 else if ( !Q_stricmpn( token, "q3map", 5 ) ) {
1884 SkipRestOfLine( text );
1885 continue;
1886 }
1887 // skip stuff that only q3map or the server needs
1888 else if ( !Q_stricmp( token, "surfaceParm" ) ) {
1889 ParseSurfaceParm( text );
1890 continue;
1891 }
1892 // no mip maps
1893 else if ( !Q_stricmp( token, "nomipmaps" ) )
1894 {
1895 shader.noMipMaps = qtrue;
1896 shader.noPicMip = qtrue;
1897 continue;
1898 }
1899 // no picmip adjustment
1900 else if ( !Q_stricmp( token, "nopicmip" ) )
1901 {
1902 shader.noPicMip = qtrue;
1903 continue;
1904 }
1905 // polygonOffset
1906 else if ( !Q_stricmp( token, "polygonOffset" ) )
1907 {
1908 shader.polygonOffset = qtrue;
1909 continue;
1910 }
1911 // entityMergable, allowing sprite surfaces from multiple entities
1912 // to be merged into one batch. This is a savings for smoke
1913 // puffs and blood, but can't be used for anything where the
1914 // shader calcs (not the surface function) reference the entity color or scroll
1915 else if ( !Q_stricmp( token, "entityMergable" ) )
1916 {
1917 shader.entityMergable = qtrue;
1918 continue;
1919 }
1920 // fogParms
1921 else if ( !Q_stricmp( token, "fogParms" ) )
1922 {
1923 if ( !ParseVector( text, 3, shader.fogParms.color ) ) {
1924 return qfalse;
1925 }
1926
1927 if ( r_greyscale->integer )
1928 {
1929 float luminance;
1930
1931 luminance = LUMA( shader.fogParms.color[0], shader.fogParms.color[1], shader.fogParms.color[2] );
1932 VectorSet( shader.fogParms.color, luminance, luminance, luminance );
1933 }
1934 else if ( r_greyscale->value )
1935 {
1936 float luminance;
1937
1938 luminance = LUMA( shader.fogParms.color[0], shader.fogParms.color[1], shader.fogParms.color[2] );
1939 shader.fogParms.color[0] = LERP( shader.fogParms.color[0], luminance, r_greyscale->value );
1940 shader.fogParms.color[1] = LERP( shader.fogParms.color[1], luminance, r_greyscale->value );
1941 shader.fogParms.color[2] = LERP( shader.fogParms.color[2], luminance, r_greyscale->value );
1942 }
1943
1944 token = COM_ParseExt( text, qfalse );
1945 if ( !token[0] )
1946 {
1947 ri.Printf( PRINT_WARNING, "WARNING: missing parm for 'fogParms' keyword in shader '%s'\n", shader.name );
1948 continue;
1949 }
1950 shader.fogParms.depthForOpaque = atof( token );
1951
1952 // skip any old gradient directions
1953 SkipRestOfLine( text );
1954 continue;
1955 }
1956 // portal
1957 else if ( !Q_stricmp(token, "portal") )
1958 {
1959 shader.sort = SS_PORTAL;
1960 shader.isPortal = qtrue;
1961 continue;
1962 }
1963 // skyparms <cloudheight> <outerbox> <innerbox>
1964 else if ( !Q_stricmp( token, "skyparms" ) )
1965 {
1966 ParseSkyParms( text );
1967 continue;
1968 }
1969 // light <value> determines flaring in q3map, not needed here
1970 else if ( !Q_stricmp(token, "light") )
1971 {
1972 COM_ParseExt( text, qfalse );
1973 continue;
1974 }
1975 // cull <face>
1976 else if ( !Q_stricmp( token, "cull") )
1977 {
1978 token = COM_ParseExt( text, qfalse );
1979 if ( token[0] == 0 )
1980 {
1981 ri.Printf( PRINT_WARNING, "WARNING: missing cull parms in shader '%s'\n", shader.name );
1982 continue;
1983 }
1984
1985 if ( !Q_stricmp( token, "none" ) || !Q_stricmp( token, "twosided" ) || !Q_stricmp( token, "disable" ) )
1986 {
1987 shader.cullType = CT_TWO_SIDED;
1988 }
1989 else if ( !Q_stricmp( token, "back" ) || !Q_stricmp( token, "backside" ) || !Q_stricmp( token, "backsided" ) )
1990 {
1991 shader.cullType = CT_BACK_SIDED;
1992 }
1993 else
1994 {
1995 ri.Printf( PRINT_WARNING, "WARNING: invalid cull parm '%s' in shader '%s'\n", token, shader.name );
1996 }
1997 continue;
1998 }
1999 // sort
2000 else if ( !Q_stricmp( token, "sort" ) )
2001 {
2002 ParseSort( text );
2003 continue;
2004 }
2005 else
2006 {
2007 ri.Printf( PRINT_WARNING, "WARNING: unknown general shader parameter '%s' in '%s'\n", token, shader.name );
2008 return qfalse;
2009 }
2010 }
2011
2012 //
2013 // ignore shaders that don't have any stages, unless it is a sky or fog
2014 //
2015 if ( s == 0 && !shader.isSky && !(shader.contentFlags & CONTENTS_FOG ) ) {
2016 return qfalse;
2017 }
2018
2019 shader.explicitlyDefined = qtrue;
2020
2021 return qtrue;
2022}

2054/*
2055===================
2056ComputeVertexAttribs
2057
2058Check which vertex attributes we only need, so we
2059don't need to submit/copy all of them.
2060===================
2061*/
2062static void ComputeVertexAttribs(void)
2063{
2064 int i, stage;
2065
2066 // dlights always need ATTR_NORMAL
2067 shader.vertexAttribs = ATTR_POSITION | ATTR_NORMAL;
2068
2069 // portals always need normals, for SurfIsOffscreen()
2070 if (shader.isPortal)
2071 {
2072 shader.vertexAttribs |= ATTR_NORMAL;
2073 }
2074
2075 if (shader.defaultShader)
2076 {
2077 shader.vertexAttribs |= ATTR_TEXCOORD;
2078 return;
2079 }
2080
2081 if(shader.numDeforms)
2082 {
2083 for ( i = 0; i < shader.numDeforms; i++)
2084 {
2085 deformStage_t *ds = &shader.deforms[i];
2086
2087 switch (ds->deformation)
2088 {
2089 case DEFORM_BULGE:
2090 shader.vertexAttribs |= ATTR_NORMAL | ATTR_TEXCOORD;
2091 break;
2092
2093 case DEFORM_AUTOSPRITE:
2094 shader.vertexAttribs |= ATTR_NORMAL | ATTR_COLOR;
2095 break;
2096
2097 case DEFORM_WAVE:
2098 case DEFORM_NORMALS:
2099 case DEFORM_TEXT0:
2100 case DEFORM_TEXT1:
2101 case DEFORM_TEXT2:
2102 case DEFORM_TEXT3:
2103 case DEFORM_TEXT4:
2104 case DEFORM_TEXT5:
2105 case DEFORM_TEXT6:
2106 case DEFORM_TEXT7:
2107 shader.vertexAttribs |= ATTR_NORMAL;
2108 break;
2109
2110 default:
2111 case DEFORM_NONE:
2112 case DEFORM_MOVE:
2113 case DEFORM_PROJECTION_SHADOW:
2114 case DEFORM_AUTOSPRITE2:
2115 break;
2116 }
2117 }
2118 }
2119
2120 for ( stage = 0; stage < MAX_SHADER_STAGES; stage++ )
2121 {
2122 shaderStage_t *pStage = &stages[stage];
2123
2124 if ( !pStage->active )
2125 {
2126 break;
2127 }
2128
2129 if (pStage->glslShaderGroup == tr.lightallShader)
2130 {
2131 shader.vertexAttribs |= ATTR_NORMAL;
2132
2133 if ((pStage->glslShaderIndex & LIGHTDEF_LIGHTTYPE_MASK) && !(r_normalMapping->integer == 0 && r_specularMapping->integer == 0))
2134 {
2135 shader.vertexAttribs |= ATTR_TANGENT;
2136 }
2137
2138 switch (pStage->glslShaderIndex & LIGHTDEF_LIGHTTYPE_MASK)
2139 {
2140 case LIGHTDEF_USE_LIGHTMAP:
2141 case LIGHTDEF_USE_LIGHT_VERTEX:
2142 shader.vertexAttribs |= ATTR_LIGHTDIRECTION;
2143 break;
2144 default:
2145 break;
2146 }
2147 }
2148
2149 for (i = 0; i < NUM_TEXTURE_BUNDLES; i++)
2150 {

2818 break;

2846 } else {

2848 }

2853 stages[0] = stages[1];

2858 stages[0].rgbGen = CGEN_IDENTITY_LIGHTING;

3638/*
3639====================
3640ScanAndLoadShaderFiles
3641
3642Finds and loads all .shader files, combining them into
3643a single large text block that can be scanned for shader names
3644=====================
3645*/
3646#define MAX_SHADER_FILES 4096
3647static void ScanAndLoadShaderFiles( void )
3648{
3649 char **shaderFiles;
3650 char *buffers[MAX_SHADER_FILES] = {NULL};
3651 char *p;
3652 int numShaderFiles;
3653 int i;
3654 char *oldp, *token, *hashMem, *textEnd;
3655 int shaderTextHashTableSizes[MAX_SHADERTEXT_HASH], hash, size;
3656 char shaderName[MAX_QPATH];
3657 int shaderLine;
3658
3659 long sum = 0, summand;
3660 // scan for shader files
3661 shaderFiles = ri.FS_ListFiles( "scripts", ".shader", &numShaderFiles );
3662
3663 if ( !shaderFiles || !numShaderFiles )
3664 {
3665 ri.Printf( PRINT_WARNING, "WARNING: no shader files found\n" );
3666 return;
3667 }
3668
3669 if ( numShaderFiles > MAX_SHADER_FILES ) {
3670 numShaderFiles = MAX_SHADER_FILES;
3671 }
3672
3673 // load and parse shader files
3674 for ( i = 0; i < numShaderFiles; i++ )
3675 {
3676 char filename[MAX_QPATH];
3677
3678 // look for a .mtr file first
3679 {
3680 char *ext;
3681 Com_sprintf( filename, sizeof( filename ), "scripts/%s", shaderFiles[i] );
3682 if ( (ext = strrchr(filename, '.')) )
3683 {
3684 strcpy(ext, ".mtr");
3685 }
3686
3687 if ( ri.FS_ReadFile( filename, NULL ) <= 0 )
3688 {
3689 Com_sprintf( filename, sizeof( filename ), "scripts/%s", shaderFiles[i] );
3690 }
3691 }
3692
3693 ri.Printf( PRINT_DEVELOPER, "...loading '%s'\n", filename );
3694 summand = ri.FS_ReadFile( filename, (void **)&buffers[i] );
3695
3696 if ( !buffers[i] )
3697 ri.Error( ERR_DROP, "Couldn't load %s", filename );
3698
3699 // Do a simple check on the shader structure in that file to make sure one bad shader file cannot fuck up all other shaders.
3700 p = buffers[i];
3701 COM_BeginParseSession(filename);
3702 while(1)
3703 {
3704 token = COM_ParseExt(&p, qtrue);
3705
3706 if(!*token)
3707 break;
3708
3709 Q_strncpyz(shaderName, token, sizeof(shaderName));
3710 shaderLine = COM_GetCurrentParseLine();
3711
3712 token = COM_ParseExt(&p, qtrue);
3713 if(token[0] != '{' || token[1] != '\0')
3714 {
3715 ri.Printf(PRINT_WARNING, "WARNING: Ignoring shader file %s. Shader \"%s\" on line %d missing opening brace",
3716 filename, shaderName, shaderLine);
3717 if (token[0])
3718 {
3719 ri.Printf(PRINT_WARNING, " (found \"%s\" on line %d)", token, COM_GetCurrentParseLine());
3720 }
3721 ri.Printf(PRINT_WARNING, ".\n");
3722 ri.FS_FreeFile(buffers[i]);
3723 buffers[i] = NULL;
3724 break;
3725 }
3726
3727 if(!SkipBracedSection(&p, 1))
3728 {
3729 ri.Printf(PRINT_WARNING, "WARNING: Ignoring shader file %s. Shader \"%s\" on line %d missing closing brace.\n",
3730 filename, shaderName, shaderLine);
3731 ri.FS_FreeFile(buffers[i]);
3732 buffers[i] = NULL;
3733 break;
3734 }
3735 }
3736
3737
3738 if (buffers[i])
3739 sum += summand;
3740 }
3741
3742 // build single large buffer
3743 s_shaderText = ri.Hunk_Alloc( sum + numShaderFiles*2, h_low );
3744 s_shaderText[ 0 ] = '\0';
3745 textEnd = s_shaderText;
3746
3747 // free in reverse order, so the temp files are all dumped
3748 for ( i = numShaderFiles - 1; i >= 0 ; i-- )
3749 {
3750 if ( !buffers[i] )
3751 continue;
3752
3753 strcat( textEnd, buffers[i] );
3754 strcat( textEnd, "\n" );
3755 textEnd += strlen( textEnd );
3756 ri.FS_FreeFile( buffers[i] );
3757 }
3758
3759 COM_Compress( s_shaderText );
3760
3761 // free up memory
3762 ri.FS_FreeFileList( shaderFiles );
3763
3764 Com_Memset(shaderTextHashTableSizes, 0, sizeof(shaderTextHashTableSizes));
3765 size = 0;
3766
3767 p = s_shaderText;
3768 // look for shader names
3769 while ( 1 ) {
3770 token = COM_ParseExt( &p, qtrue );
3771 if ( token[0] == 0 ) {
3772 break;
3773 }
3774
3775 hash = generateHashValue(token, MAX_SHADERTEXT_HASH);
3776 shaderTextHashTableSizes[hash]++;
3777 size++;
3778 SkipBracedSection(&p, 0);
3779 }
3780
3781 size += MAX_SHADERTEXT_HASH;
3782
3783 hashMem = ri.Hunk_Alloc( size * sizeof(char *), h_low );
3784
3785 for (i = 0; i < MAX_SHADERTEXT_HASH; i++) {
3786 shaderTextHashTable[i] = (char **) hashMem;
3787 hashMem = ((char *) hashMem) + ((shaderTextHashTableSizes[i] + 1) * sizeof(char *));
3788 }
3789
3790 Com_Memset(shaderTextHashTableSizes, 0, sizeof(shaderTextHashTableSizes));
3791
3792 p = s_shaderText;
3793 // look for shader names
3794 while ( 1 ) {
3795 oldp = p;
3796 token = COM_ParseExt( &p, qtrue );
3797 if ( token[0] == 0 ) {
3798 break;
3799 }
3800
3801 hash = generateHashValue(token, MAX_SHADERTEXT_HASH);
3802 shaderTextHashTable[hash][shaderTextHashTableSizes[hash]++] = oldp;
3803
3804 SkipBracedSection(&p, 0);
3805 }
3806
3807 return;
3808
3809}

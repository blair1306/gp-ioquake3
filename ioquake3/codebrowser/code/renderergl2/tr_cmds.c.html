24/*
25=====================
26R_PerformanceCounters
27=====================
28*/
29void R_PerformanceCounters( void ) {
30 if ( !r_speeds->integer ) {
31 // clear the counters even if we aren't printing
32 Com_Memset( &tr.pc, 0, sizeof( tr.pc ) );
33 Com_Memset( &backEnd.pc, 0, sizeof( backEnd.pc ) );
34 return;
35 }
36
37 if (r_speeds->integer == 1) {
38 ri.Printf (PRINT_ALL, "%i/%i/%i shaders/batches/surfs %i leafs %i verts %i/%i tris %.2f mtex %.2f dc\n",
39 backEnd.pc.c_shaders, backEnd.pc.c_surfBatches, backEnd.pc.c_surfaces, tr.pc.c_leafs, backEnd.pc.c_vertexes,
40 backEnd.pc.c_indexes/3, backEnd.pc.c_totalIndexes/3,
41 R_SumOfUsedImages()/(1000000.0f), backEnd.pc.c_overDraw / (float)(glConfig.vidWidth * glConfig.vidHeight) );
42 } else if (r_speeds->integer == 2) {
43 ri.Printf (PRINT_ALL, "(patch) %i sin %i sclip %i sout %i bin %i bclip %i bout\n",
44 tr.pc.c_sphere_cull_patch_in, tr.pc.c_sphere_cull_patch_clip, tr.pc.c_sphere_cull_patch_out,
45 tr.pc.c_box_cull_patch_in, tr.pc.c_box_cull_patch_clip, tr.pc.c_box_cull_patch_out );
46 ri.Printf (PRINT_ALL, "(md3) %i sin %i sclip %i sout %i bin %i bclip %i bout\n",
47 tr.pc.c_sphere_cull_md3_in, tr.pc.c_sphere_cull_md3_clip, tr.pc.c_sphere_cull_md3_out,
48 tr.pc.c_box_cull_md3_in, tr.pc.c_box_cull_md3_clip, tr.pc.c_box_cull_md3_out );
49 } else if (r_speeds->integer == 3) {
50 ri.Printf (PRINT_ALL, "viewcluster: %i\n", tr.viewCluster );
51 } else if (r_speeds->integer == 4) {
52 if ( backEnd.pc.c_dlightVertexes ) {
53 ri.Printf (PRINT_ALL, "dlight srf:%i culled:%i verts:%i tris:%i\n",
54 tr.pc.c_dlightSurfaces, tr.pc.c_dlightSurfacesCulled,
55 backEnd.pc.c_dlightVertexes, backEnd.pc.c_dlightIndexes / 3 );
56 }
57 }
58 else if (r_speeds->integer == 5 )
59 {
60 ri.Printf( PRINT_ALL, "zFar: %.0f\n", tr.viewParms.zFar );
61 }
62 else if (r_speeds->integer == 6 )
63 {
64 ri.Printf( PRINT_ALL, "flare adds:%i tests:%i renders:%i\n",
65 backEnd.pc.c_flareAdds, backEnd.pc.c_flareTests, backEnd.pc.c_flareRenders );
66 }
67 else if (r_speeds->integer == 7 )
68 {
69 ri.Printf( PRINT_ALL, "VAO draws: static %i dynamic %i\n",
70 backEnd.pc.c_staticVaoDraws, backEnd.pc.c_dynamicVaoDraws);
71 ri.Printf( PRINT_ALL, "GLSL binds: %i draws: gen %i light %i fog %i dlight %i\n",
72 backEnd.pc.c_glslShaderBinds, backEnd.pc.c_genericDraws, backEnd.pc.c_lightallDraws, backEnd.pc.c_fogDraws, backEnd.pc.c_dlightDraws);
73 }
74
75 Com_Memset( &tr.pc, 0, sizeof( tr.pc ) );
76 Com_Memset( &backEnd.pc, 0, sizeof( backEnd.pc ) );
77}

80/*
81====================
82R_IssueRenderCommands
83====================
84*/
85void R_IssueRenderCommands( qboolean runPerformanceCounters ) {
86 renderCommandList_t *cmdList;
87
88 cmdList = &backEndData->commands;
89 assert(cmdList);
90 // add an end-of-list command
91 *(int *)(cmdList->cmds + cmdList->used) = RC_END_OF_LIST;
92
93 // clear it out, in case this is a sync and not a buffer flip
94 cmdList->used = 0;
95
96 if ( runPerformanceCounters ) {
97 R_PerformanceCounters();
98 }
99
100 // actually start the commands going
101 if ( !r_skipBackEnd->integer ) {
102 // let it start on the new batch
103 RB_ExecuteRenderCommands( cmdList->cmds );
104 }
105}

122/*
123============
124R_GetCommandBufferReserved
125
126make sure there is enough command space
127============
128*/
129void *R_GetCommandBufferReserved( int bytes, int reservedBytes ) {
130 renderCommandList_t *cmdList;
131
132 cmdList = &backEndData->commands;
133 bytes = PAD(bytes, sizeof(void *));
134
135 // always leave room for the end of list command
136 if ( cmdList->used + bytes + sizeof( int ) + reservedBytes > MAX_RENDER_COMMANDS ) {
137 if ( bytes > MAX_RENDER_COMMANDS - sizeof( int ) ) {
138 ri.Error( ERR_FATAL, "R_GetCommandBuffer: bad size %i", bytes );
139 }
140 // if we run out of room, just start dropping commands
141 return NULL;
142 }
143
144 cmdList->used += bytes;
145
146 return cmdList->cmds + cmdList->used - bytes;
147}
148
149/*
150=============
151R_GetCommandBuffer
152
153returns NULL if there is not enough space for important commands
154=============
155*/
156void *R_GetCommandBuffer( int bytes ) {
157 return R_GetCommandBufferReserved( bytes, PAD( sizeof( swapBuffersCommand_t ), sizeof(void *) ) );
158}

223/*
224=============
225RE_SetColor
226
227Passing NULL will set the color to white
228=============
229*/
230void RE_SetColor( const float *rgba ) {
231 setColorCommand_t *cmd;
232
233 if ( !tr.registered ) {
234 return;
235 }
236 cmd = R_GetCommandBuffer( sizeof( *cmd ) );
237 if ( !cmd ) {
238 return;
239 }
240 cmd->commandId = RC_SET_COLOR;
241 if ( !rgba ) {
242 static float colorWhite[4] = { 1, 1, 1, 1 };
243
244 rgba = colorWhite;
245 }
246
247 cmd->color[0] = rgba[0];
248 cmd->color[1] = rgba[1];
249 cmd->color[2] = rgba[2];
250 cmd->color[3] = rgba[3];
251}

282#define MODE_RED_CYAN 1
283#define MODE_RED_BLUE 2
284#define MODE_RED_GREEN 3
285#define MODE_GREEN_MAGENTA 4
286#define MODE_MAX MODE_GREEN_MAGENTA
287
288void R_SetColorMode(GLboolean *rgba, stereoFrame_t stereoFrame, int colormode)
289{
290 rgba[0] = rgba[1] = rgba[2] = rgba[3] = GL_TRUE;
291
292 if(colormode > MODE_MAX)
293 {
294 if(stereoFrame == STEREO_LEFT)
295 stereoFrame = STEREO_RIGHT;
296 else if(stereoFrame == STEREO_RIGHT)
297 stereoFrame = STEREO_LEFT;
298
299 colormode -= MODE_MAX;
300 }
301
302 if(colormode == MODE_GREEN_MAGENTA)
303 {
304 if(stereoFrame == STEREO_LEFT)
305 rgba[0] = rgba[2] = GL_FALSE;
306 else if(stereoFrame == STEREO_RIGHT)
307 rgba[1] = GL_FALSE;
308 }
309 else
310 {
311 if(stereoFrame == STEREO_LEFT)
312 rgba[1] = rgba[2] = GL_FALSE;
313 else if(stereoFrame == STEREO_RIGHT)
314 {
315 rgba[0] = GL_FALSE;
316
317 if(colormode == MODE_RED_BLUE)
318 rgba[1] = GL_FALSE;
319 else if(colormode == MODE_RED_GREEN)
320 rgba[2] = GL_FALSE;
321 }
322 }
323}

326/*
327====================
328RE_BeginFrame
329
330If running in stereo, RE_BeginFrame will be called twice
331for each RE_EndFrame
332====================
333*/
334void RE_BeginFrame( stereoFrame_t stereoFrame ) {
335 drawBufferCommand_t *cmd = NULL;
336 colorMaskCommand_t *colcmd = NULL;
337
338 if ( !tr.registered ) {
339 return;
340 }
341 glState.finishCalled = qfalse;
342
343 tr.frameCount++;
344 tr.frameSceneNum = 0;
345
346 //
347 // do overdraw measurement
348 //
349 if ( r_measureOverdraw->integer )
350 {
351 if ( glConfig.stencilBits < 4 )
352 {
353 ri.Printf( PRINT_ALL, "Warning: not enough stencil bits to measure overdraw: %d\n", glConfig.stencilBits );
354 ri.Cvar_Set( "r_measureOverdraw", "0" );
355 r_measureOverdraw->modified = qfalse;
356 }
357 else if ( r_shadows->integer == 2 )
358 {
359 ri.Printf( PRINT_ALL, "Warning: stencil shadows and overdraw measurement are mutually exclusive\n" );
360 ri.Cvar_Set( "r_measureOverdraw", "0" );
361 r_measureOverdraw->modified = qfalse;
362 }
363 else
364 {
365 R_IssuePendingRenderCommands();
366 qglEnable( GL_STENCIL_TEST );
367 qglStencilMask( ~0U );
368 qglClearStencil( 0U );
369 qglStencilFunc( GL_ALWAYS, 0U, ~0U );
370 qglStencilOp( GL_KEEP, GL_INCR, GL_INCR );
371 }
372 r_measureOverdraw->modified = qfalse;
373 }
374 else
375 {
376 // this is only reached if it was on and is now off
377 if ( r_measureOverdraw->modified ) {
378 R_IssuePendingRenderCommands();
379 qglDisable( GL_STENCIL_TEST );
380 }
381 r_measureOverdraw->modified = qfalse;
382 }
383
384 //
385 // texturemode stuff
386 //
387 if ( r_textureMode->modified ) {
388 R_IssuePendingRenderCommands();
389 GL_TextureMode( r_textureMode->string );
390 r_textureMode->modified = qfalse;
391 }
392
393 //
394 // gamma stuff
395 //
396 if ( r_gamma->modified ) {
397 r_gamma->modified = qfalse;
398
399 R_IssuePendingRenderCommands();
400 R_SetColorMappings();
401 }
402
403 // check for errors
404 if ( !r_ignoreGLErrors->integer )
405 {
406 int err;
407
408 R_IssuePendingRenderCommands();
409 if ((err = qglGetError()) != GL_NO_ERROR)
410 ri.Error(ERR_FATAL, "RE_BeginFrame() - glGetError() failed (0x%x)!", err);
411 }
412
413 if (glConfig.stereoEnabled) {
414 if( !(cmd = R_GetCommandBuffer(sizeof(*cmd))) )
415 return;
416
417 cmd->commandId = RC_DRAW_BUFFER;
418
419 if ( stereoFrame == STEREO_LEFT ) {
420 cmd->buffer = (int)GL_BACK_LEFT;
421 } else if ( stereoFrame == STEREO_RIGHT ) {
422 cmd->buffer = (int)GL_BACK_RIGHT;
423 } else {
424 ri.Error( ERR_FATAL, "RE_BeginFrame: Stereo is enabled, but stereoFrame was %i", stereoFrame );
425 }
426 }
427 else
428 {
429 if(r_anaglyphMode->integer)
430 {
431 if(r_anaglyphMode->modified)
432 {
433 // clear both, front and backbuffer.
434 qglColorMask(GL_TRUE, GL_TRUE, GL_TRUE, GL_TRUE);
435 backEnd.colorMask[0] = GL_FALSE;
436 backEnd.colorMask[1] = GL_FALSE;
437 backEnd.colorMask[2] = GL_FALSE;
438 backEnd.colorMask[3] = GL_FALSE;
439
440 if (glRefConfig.framebufferObject)
441 {
442 // clear all framebuffers
443 if (tr.msaaResolveFbo)
444 {
445 FBO_Bind(tr.msaaResolveFbo);
446 qglClear(GL_COLOR_BUFFER_BIT);
447 }
448
449 if (tr.renderFbo)
450 {
451 FBO_Bind(tr.renderFbo);
452 qglClear(GL_COLOR_BUFFER_BIT);
453 }
454
455 FBO_Bind(NULL);
456 }
457
458 qglDrawBuffer(GL_FRONT);
459 qglClear(GL_COLOR_BUFFER_BIT);
460 qglDrawBuffer(GL_BACK);
461 qglClear(GL_COLOR_BUFFER_BIT);
462
463 r_anaglyphMode->modified = qfalse;
464 }
465
466 if(stereoFrame == STEREO_LEFT)
467 {
468 if( !(cmd = R_GetCommandBuffer(sizeof(*cmd))) )
469 return;
470
471 if( !(colcmd = R_GetCommandBuffer(sizeof(*colcmd))) )
472 return;
473 }
474 else if(stereoFrame == STEREO_RIGHT)
475 {
476 clearDepthCommand_t *cldcmd;
477
478 if( !(cldcmd = R_GetCommandBuffer(sizeof(*cldcmd))) )
479 return;
480
481 cldcmd->commandId = RC_CLEARDEPTH;
482
483 if( !(colcmd = R_GetCommandBuffer(sizeof(*colcmd))) )
484 return;
485 }
486 else
487 ri.Error( ERR_FATAL, "RE_BeginFrame: Stereo is enabled, but stereoFrame was %i", stereoFrame );
488
489 R_SetColorMode(colcmd->rgba, stereoFrame, r_anaglyphMode->integer);
490 colcmd->commandId = RC_COLORMASK;
491 }
492 else
493 {
494 if(stereoFrame != STEREO_CENTER)
495 ri.Error( ERR_FATAL, "RE_BeginFrame: Stereo is disabled, but stereoFrame was %i", stereoFrame );
496
497 if( !(cmd = R_GetCommandBuffer(sizeof(*cmd))) )
498 return;
499 }
500
501 if(cmd)
502 {
503 cmd->commandId = RC_DRAW_BUFFER;
504
505 if(r_anaglyphMode->modified)
506 {
507 qglColorMask(GL_TRUE, GL_TRUE, GL_TRUE, GL_TRUE);
508 backEnd.colorMask[0] = 0;
509 backEnd.colorMask[1] = 0;
510 backEnd.colorMask[2] = 0;
511 backEnd.colorMask[3] = 0;
512 r_anaglyphMode->modified = qfalse;
513 }
514
515 if (!Q_stricmp(r_drawBuffer->string, "GL_FRONT"))
516 cmd->buffer = (int)GL_FRONT;
517 else
518 cmd->buffer = (int)GL_BACK;
519 }
520 }
521
522 tr.refdef.stereoFrame = stereoFrame;
523}

526/*
527=============
528RE_EndFrame
529
530Returns the number of msec spent in the back end
531=============
532*/
533void RE_EndFrame( int *frontEndMsec, int *backEndMsec ) {
534 swapBuffersCommand_t *cmd;
535
536 if ( !tr.registered ) {
537 return;
538 }
539 cmd = R_GetCommandBufferReserved( sizeof( *cmd ), 0 );
540 if ( !cmd ) {
541 return;
542 }
543 cmd->commandId = RC_SWAP_BUFFERS;
544
545 R_IssueRenderCommands( qtrue );
546
547 R_InitNextFrame();
548
549 if ( frontEndMsec ) {
550 *frontEndMsec = tr.frontEndMsec;
551 }
552 tr.frontEndMsec = 0;
553 if ( backEndMsec ) {
554 *backEndMsec = backEnd.pc.msec;
555 }
556 backEnd.pc.msec = 0;
557}

22// tr_extensions.c - extensions needed by the renderer not in sdl_glimp.c
23
24#ifdef USE_LOCAL_HEADERS
25# include "SDL.h"
26#else
27# include <SDL.h>
28#endif
29
30#include "tr_local.h"
31#include "tr_dsa.h"
32
33#define GLE(ret, name, ...) name##proc * qgl##name;
34QGL_1_3_PROCS;
35QGL_1_5_PROCS;
36QGL_2_0_PROCS;
37QGL_3_0_PROCS;
38QGL_ARB_framebuffer_object_PROCS;
39QGL_ARB_vertex_array_object_PROCS;
40QGL_EXT_direct_state_access_PROCS;
41#undef GLE
42
43void GLimp_InitExtraExtensions()
44{
45 char *extension;
46 const char* result[3] = { "...ignoring %s\n", "...using %s\n", "...%s not found\n" };
47 qboolean q_gl_version_at_least_3_0;
48 qboolean q_gl_version_at_least_3_2;
49
50 // Check OpenGL version
51 sscanf(glConfig.version_string, "%d.%d", &glRefConfig.openglMajorVersion, &glRefConfig.openglMinorVersion);
52 if (glRefConfig.openglMajorVersion < 2)
53 ri.Error(ERR_FATAL, "OpenGL 2.0 required!");
54 ri.Printf(PRINT_ALL, "...using OpenGL %s\n", glConfig.version_string);
55
56 q_gl_version_at_least_3_0 = (glRefConfig.openglMajorVersion >= 3);
57 q_gl_version_at_least_3_2 = (glRefConfig.openglMajorVersion > 3 || (glRefConfig.openglMajorVersion == 3 && glRefConfig.openglMinorVersion > 2));
58
59 // Check if we need Intel graphics specific fixes.
60 glRefConfig.intelGraphics = qfalse;
61 if (strstr((char *)qglGetString(GL_RENDERER), "Intel"))
62 glRefConfig.intelGraphics = qtrue;
63
64 // set DSA fallbacks
65#define GLE(ret, name, ...) qgl##name = GLDSA_##name;
66 QGL_EXT_direct_state_access_PROCS;
67#undef GLE
68
69 // GL function loader, based on https://gist.github.com/rygorous/16796a0c876cf8a5f542caddb55bce8a
70#define GLE(ret, name, ...) qgl##name = (name##proc *) SDL_GL_GetProcAddress("gl" #name);
71
72 // OpenGL 1.3, was GL_ARB_texture_compression
73 QGL_1_3_PROCS;
74
75 // OpenGL 1.5, was GL_ARB_vertex_buffer_object and GL_ARB_occlusion_query
76 QGL_1_5_PROCS;
77 glRefConfig.occlusionQuery = qtrue;
78
79 // OpenGL 2.0, was GL_ARB_shading_language_100, GL_ARB_vertex_program, GL_ARB_shader_objects, and GL_ARB_vertex_shader
80 QGL_2_0_PROCS;
81
82 // OpenGL 3.0 - no matching extension
83 // QGL_*_PROCS becomes several functions, do not remove {}
84 if (q_gl_version_at_least_3_0)
85 {
86 QGL_3_0_PROCS;
87 }
88
89 // OpenGL 3.0 - GL_ARB_framebuffer_object
90 extension = "GL_ARB_framebuffer_object";
91 glRefConfig.framebufferObject = qfalse;
92 glRefConfig.framebufferBlit = qfalse;
93 glRefConfig.framebufferMultisample = qfalse;
94 if (q_gl_version_at_least_3_0 || SDL_GL_ExtensionSupported(extension))
95 {
96 glRefConfig.framebufferObject = !!r_ext_framebuffer_object->integer;
97 glRefConfig.framebufferBlit = qtrue;
98 glRefConfig.framebufferMultisample = qtrue;
99
100 qglGetIntegerv(GL_MAX_RENDERBUFFER_SIZE, &glRefConfig.maxRenderbufferSize);
101 qglGetIntegerv(GL_MAX_COLOR_ATTACHMENTS, &glRefConfig.maxColorAttachments);
102
103 QGL_ARB_framebuffer_object_PROCS;
104
105 ri.Printf(PRINT_ALL, result[glRefConfig.framebufferObject], extension);
106 }
107 else
108 {
109 ri.Printf(PRINT_ALL, result[2], extension);
110 }
111
112 // OpenGL 3.0 - GL_ARB_vertex_array_object
113 extension = "GL_ARB_vertex_array_object";
114 glRefConfig.vertexArrayObject = qfalse;
115 if (q_gl_version_at_least_3_0 || SDL_GL_ExtensionSupported(extension))
116 {
117 if (q_gl_version_at_least_3_0)
118 {
119 // force VAO, core context requires it
120 glRefConfig.vertexArrayObject = qtrue;
121 }
122 else
123 {
124 glRefConfig.vertexArrayObject = !!r_arb_vertex_array_object->integer;
125 }
126
127 QGL_ARB_vertex_array_object_PROCS;
128
129 ri.Printf(PRINT_ALL, result[glRefConfig.vertexArrayObject], extension);
130 }
131 else
132 {
133 ri.Printf(PRINT_ALL, result[2], extension);
134 }
135
136 // OpenGL 3.0 - GL_ARB_texture_float
137 extension = "GL_ARB_texture_float";
138 glRefConfig.textureFloat = qfalse;
139 if (q_gl_version_at_least_3_0 || SDL_GL_ExtensionSupported(extension))
140 {
141 glRefConfig.textureFloat = !!r_ext_texture_float->integer;
142
143 ri.Printf(PRINT_ALL, result[glRefConfig.textureFloat], extension);
144 }
145 else
146 {
147 ri.Printf(PRINT_ALL, result[2], extension);
148 }
149
150 // OpenGL 3.2 - GL_ARB_depth_clamp
151 extension = "GL_ARB_depth_clamp";
152 glRefConfig.depthClamp = qfalse;
153 if (q_gl_version_at_least_3_2 || SDL_GL_ExtensionSupported(extension))
154 {
155 glRefConfig.depthClamp = qtrue;
156
157 ri.Printf(PRINT_ALL, result[glRefConfig.depthClamp], extension);
158 }
159 else
160 {
161 ri.Printf(PRINT_ALL, result[2], extension);
162 }
163
164 // OpenGL 3.2 - GL_ARB_seamless_cube_map
165 extension = "GL_ARB_seamless_cube_map";
166 glRefConfig.seamlessCubeMap = qfalse;
167 if (q_gl_version_at_least_3_2 || SDL_GL_ExtensionSupported(extension))
168 {
169 glRefConfig.seamlessCubeMap = !!r_arb_seamless_cube_map->integer;
170
171 ri.Printf(PRINT_ALL, result[glRefConfig.seamlessCubeMap], extension);
172 }
173 else
174 {
175 ri.Printf(PRINT_ALL, result[2], extension);
176 }
177
178 // Determine GLSL version
179 if (1)
180 {
181 char version[256];
182
183 Q_strncpyz(version, (char *)qglGetString(GL_SHADING_LANGUAGE_VERSION), sizeof(version));
184
185 sscanf(version, "%d.%d", &glRefConfig.glslMajorVersion, &glRefConfig.glslMinorVersion);
186
187 ri.Printf(PRINT_ALL, "...using GLSL version %s\n", version);
188 }
189
190 glRefConfig.memInfo = MI_NONE;
191
192 // GL_NVX_gpu_memory_info
193 extension = "GL_NVX_gpu_memory_info";
194 if( SDL_GL_ExtensionSupported( extension ) )
195 {
196 glRefConfig.memInfo = MI_NVX;
197
198 ri.Printf(PRINT_ALL, result[1], extension);
199 }
200 else
201 {
202 ri.Printf(PRINT_ALL, result[2], extension);
203 }
204
205 // GL_ATI_meminfo
206 extension = "GL_ATI_meminfo";
207 if( SDL_GL_ExtensionSupported( extension ) )
208 {
209 if (glRefConfig.memInfo == MI_NONE)
210 {
211 glRefConfig.memInfo = MI_ATI;
212
213 ri.Printf(PRINT_ALL, result[1], extension);
214 }
215 else
216 {
217 ri.Printf(PRINT_ALL, result[0], extension);
218 }
219 }
220 else
221 {
222 ri.Printf(PRINT_ALL, result[2], extension);
223 }
224
225 glRefConfig.textureCompression = TCR_NONE;
226
227 // GL_ARB_texture_compression_rgtc
228 extension = "GL_ARB_texture_compression_rgtc";
229 if (SDL_GL_ExtensionSupported(extension))
230 {
231 qboolean useRgtc = r_ext_compressed_textures->integer >= 1;
232
233 if (useRgtc)
234 glRefConfig.textureCompression |= TCR_RGTC;
235
236 ri.Printf(PRINT_ALL, result[useRgtc], extension);
237 }
238 else
239 {
240 ri.Printf(PRINT_ALL, result[2], extension);
241 }
242
243 glRefConfig.swizzleNormalmap = r_ext_compressed_textures->integer && !(glRefConfig.textureCompression & TCR_RGTC);
244
245 // GL_ARB_texture_compression_bptc
246 extension = "GL_ARB_texture_compression_bptc";
247 if (SDL_GL_ExtensionSupported(extension))
248 {
249 qboolean useBptc = r_ext_compressed_textures->integer >= 2;
250
251 if (useBptc)
252 glRefConfig.textureCompression |= TCR_BPTC;
253
254 ri.Printf(PRINT_ALL, result[useBptc], extension);
255 }
256 else
257 {
258 ri.Printf(PRINT_ALL, result[2], extension);
259 }
260
261 // GL_EXT_direct_state_access
262 extension = "GL_EXT_direct_state_access";
263 glRefConfig.directStateAccess = qfalse;
264 if (SDL_GL_ExtensionSupported(extension))
265 {
266 glRefConfig.directStateAccess = !!r_ext_direct_state_access->integer;
267
268 // QGL_*_PROCS becomes several functions, do not remove {}
269 if (glRefConfig.directStateAccess)
270 {
271 QGL_EXT_direct_state_access_PROCS;
272 }
273
274 ri.Printf(PRINT_ALL, result[glRefConfig.directStateAccess], extension);
275 }
276 else
277 {
278 ri.Printf(PRINT_ALL, result[2], extension);
279 }
280
281#undef GLE
282}

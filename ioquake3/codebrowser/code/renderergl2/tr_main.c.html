46/*
47================
48R_CompareVert
49================
50*/
51qboolean R_CompareVert(srfVert_t * v1, srfVert_t * v2, qboolean checkST)
52{
53 int i;
54
55 for(i = 0; i < 3; i++)
56 {
57 if(floor(v1->xyz[i] + 0.1) != floor(v2->xyz[i] + 0.1))
58 {
59 return qfalse;
60 }
61
62 if(checkST && ((v1->st[0] != v2->st[0]) || (v1->st[1] != v2->st[1])))
63 {
64 return qfalse;
65 }
66 }
67
68 return qtrue;
69}

71/*
72=============
73R_CalcTexDirs
74
75Lengyel, Eric. “Computing Tangent Space Basis Vectors for an Arbitrary Mesh”. Terathon Software 3D Graphics Library, 2001. http://www.terathon.com/code/tangent.html
76=============
77*/
78void R_CalcTexDirs(vec3_t sdir, vec3_t tdir, const vec3_t v1, const vec3_t v2,
79 const vec3_t v3, const vec2_t w1, const vec2_t w2, const vec2_t w3)
80{
81 float x1, x2, y1, y2, z1, z2;
82 float s1, s2, t1, t2, r;
83
84 x1 = v2[0] - v1[0];
85 x2 = v3[0] - v1[0];
86 y1 = v2[1] - v1[1];
87 y2 = v3[1] - v1[1];
88 z1 = v2[2] - v1[2];
89 z2 = v3[2] - v1[2];
90
91 s1 = w2[0] - w1[0];
92 s2 = w3[0] - w1[0];
93 t1 = w2[1] - w1[1];
94 t2 = w3[1] - w1[1];
95
96 r = s1 * t2 - s2 * t1;
97 if (r) r = 1.0f / r;
98
99 VectorSet(sdir, (t2 * x1 - t1 * x2) * r, (t2 * y1 - t1 * y2) * r, (t2 * z1 - t1 * z2) * r);
100 VectorSet(tdir, (s1 * x2 - s2 * x1) * r, (s1 * y2 - s2 * y1) * r, (s1 * z2 - s2 * z1) * r);
101}

103/*
104=============
105R_CalcTangentSpace
106
107Lengyel, Eric. “Computing Tangent Space Basis Vectors for an Arbitrary Mesh”. Terathon Software 3D Graphics Library, 2001. http://www.terathon.com/code/tangent.html
108=============
109*/
110vec_t R_CalcTangentSpace(vec3_t tangent, vec3_t bitangent, const vec3_t normal, const vec3_t sdir, const vec3_t tdir)
111{
112 vec3_t n_cross_t;
113 vec_t n_dot_t, handedness;
114
115 // Gram-Schmidt orthogonalize
116 n_dot_t = DotProduct(normal, sdir);
117 VectorMA(sdir, -n_dot_t, normal, tangent);
118 VectorNormalize(tangent);
119
120 // Calculate handedness
121 CrossProduct(normal, sdir, n_cross_t);
122 handedness = (DotProduct(n_cross_t, tdir) < 0.0f) ? -1.0f : 1.0f;
123
124 // Calculate orthogonal bitangent, if necessary
125 if (bitangent)
126 CrossProduct(normal, tangent, bitangent);
127
128 return handedness;
129}

131qboolean R_CalcTangentVectors(srfVert_t * dv[3])
132{
133 int i;
134 float bb, s, t;
135 vec3_t bary;
136
137
138 /* calculate barycentric basis for the triangle */
139 bb = (dv[1]->st[0] - dv[0]->st[0]) * (dv[2]->st[1] - dv[0]->st[1]) - (dv[2]->st[0] - dv[0]->st[0]) * (dv[1]->st[1] - dv[0]->st[1]);
140 if(fabs(bb) < 0.00000001f)
141 return qfalse;
142
143 /* do each vertex */
144 for(i = 0; i < 3; i++)
145 {
146 vec4_t tangent;
147 vec3_t normal, bitangent, nxt;
148
149 // calculate s tangent vector
150 s = dv[i]->st[0] + 10.0f;
151 t = dv[i]->st[1];
152 bary[0] = ((dv[1]->st[0] - s) * (dv[2]->st[1] - t) - (dv[2]->st[0] - s) * (dv[1]->st[1] - t)) / bb;
153 bary[1] = ((dv[2]->st[0] - s) * (dv[0]->st[1] - t) - (dv[0]->st[0] - s) * (dv[2]->st[1] - t)) / bb;
154 bary[2] = ((dv[0]->st[0] - s) * (dv[1]->st[1] - t) - (dv[1]->st[0] - s) * (dv[0]->st[1] - t)) / bb;
155
156 tangent[0] = bary[0] * dv[0]->xyz[0] + bary[1] * dv[1]->xyz[0] + bary[2] * dv[2]->xyz[0];
157 tangent[1] = bary[0] * dv[0]->xyz[1] + bary[1] * dv[1]->xyz[1] + bary[2] * dv[2]->xyz[1];
158 tangent[2] = bary[0] * dv[0]->xyz[2] + bary[1] * dv[1]->xyz[2] + bary[2] * dv[2]->xyz[2];
159
160 VectorSubtract(tangent, dv[i]->xyz, tangent);
161 VectorNormalize(tangent);
162
163 // calculate t tangent vector
164 s = dv[i]->st[0];
165 t = dv[i]->st[1] + 10.0f;
166 bary[0] = ((dv[1]->st[0] - s) * (dv[2]->st[1] - t) - (dv[2]->st[0] - s) * (dv[1]->st[1] - t)) / bb;
167 bary[1] = ((dv[2]->st[0] - s) * (dv[0]->st[1] - t) - (dv[0]->st[0] - s) * (dv[2]->st[1] - t)) / bb;
168 bary[2] = ((dv[0]->st[0] - s) * (dv[1]->st[1] - t) - (dv[1]->st[0] - s) * (dv[0]->st[1] - t)) / bb;
169
170 bitangent[0] = bary[0] * dv[0]->xyz[0] + bary[1] * dv[1]->xyz[0] + bary[2] * dv[2]->xyz[0];
171 bitangent[1] = bary[0] * dv[0]->xyz[1] + bary[1] * dv[1]->xyz[1] + bary[2] * dv[2]->xyz[1];
172 bitangent[2] = bary[0] * dv[0]->xyz[2] + bary[1] * dv[1]->xyz[2] + bary[2] * dv[2]->xyz[2];
173
174 VectorSubtract(bitangent, dv[i]->xyz, bitangent);
175 VectorNormalize(bitangent);
176
177 // store bitangent handedness
178 R_VaoUnpackNormal(normal, dv[i]->normal);
179 CrossProduct(normal, tangent, nxt);
180 tangent[3] = (DotProduct(nxt, bitangent) < 0.0f) ? -1.0f : 1.0f;
181
182 R_VaoPackTangent(dv[i]->tangent, tangent);
183
184 // debug code
185 //% Sys_FPrintf( SYS_VRB, "%d S: (%f %f %f) T: (%f %f %f)\n", i,
186 //% stv[ i ][ 0 ], stv[ i ][ 1 ], stv[ i ][ 2 ], ttv[ i ][ 0 ], ttv[ i ][ 1 ], ttv[ i ][ 2 ] );
187 }
188
189 return qtrue;
190}

284/*
285=================
286R_CullBox
287
288Returns CULL_IN, CULL_CLIP, or CULL_OUT
289=================
290*/
291int R_CullBox(vec3_t worldBounds[2]) {
292 int i;
293 cplane_t *frust;
294 qboolean anyClip;
295 int r, numPlanes;
296
297 numPlanes = (tr.viewParms.flags & VPF_FARPLANEFRUSTUM) ? 5 : 4;
298
299 // check against frustum planes
300 anyClip = qfalse;
301 for(i = 0; i < numPlanes; i++)
302 {
303 frust = &tr.viewParms.frustum[i];
304
305 r = BoxOnPlaneSide(worldBounds[0], worldBounds[1], frust);
306
307 if(r == 2)
308 {
309 // completely outside frustum
310 return CULL_OUT;
311 }
312 if(r == 3)
313 {
314 anyClip = qtrue;
315 }
316 }
317
318 if(!anyClip)
319 {
320 // completely inside frustum
321 return CULL_IN;
322 }
323
324 // partially clipped
325 return CULL_CLIP;
326}

340/*
341** R_CullPointAndRadius
342*/
343int R_CullPointAndRadiusEx( const vec3_t pt, float radius, const cplane_t* frustum, int numPlanes )
344{
345 int i;
346 float dist;
347 const cplane_t *frust;
348 qboolean mightBeClipped = qfalse;
349
350 if ( r_nocull->integer ) {
351 return CULL_CLIP;
352 }
353
354 // check against frustum planes
355 for (i = 0 ; i < numPlanes ; i++)
356 {
357 frust = &frustum[i];
358
359 dist = DotProduct( pt, frust->normal) - frust->dist;
360 if ( dist < -radius )
361 {
362 return CULL_OUT;
363 }
364 else if ( dist <= radius )
365 {
366 mightBeClipped = qtrue;
367 }
368 }
369
370 if ( mightBeClipped )
371 {
372 return CULL_CLIP;
373 }
374
375 return CULL_IN; // completely inside frustum
376}

423==========================
424R_TransformModelToClip
425
426==========================
427*/
428void R_TransformModelToClip( const vec3_t src, const float *modelMatrix, const float *projectionMatrix,
429 vec4_t eye, vec4_t dst ) {
430 int i;
431
432 for ( i = 0 ; i < 4 ; i++ ) {
433 eye[i] =
434 src[0] * modelMatrix[ i + 0 * 4 ] +
435 src[1] * modelMatrix[ i + 1 * 4 ] +
436 src[2] * modelMatrix[ i + 2 * 4 ] +
437 1 * modelMatrix[ i + 3 * 4 ];
438 }
439
440 for ( i = 0 ; i < 4 ; i++ ) {
441 dst[i] =
442 eye[0] * projectionMatrix[ i + 0 * 4 ] +
443 eye[1] * projectionMatrix[ i + 1 * 4 ] +
444 eye[2] * projectionMatrix[ i + 2 * 4 ] +
445 eye[3] * projectionMatrix[ i + 3 * 4 ];
446 }
447}

451R_TransformClipToWindow
452
453==========================
454*/
455void R_TransformClipToWindow( const vec4_t clip, const viewParms_t *view, vec4_t normalized, vec4_t window ) {
456 normalized[0] = clip[0] / clip[3];
457 normalized[1] = clip[1] / clip[3];
458 normalized[2] = ( clip[2] + clip[3] ) / ( 2 * clip[3] );
459
460 window[0] = 0.5f * ( 1.0f + normalized[0] ) * view->viewportWidth;
461 window[1] = 0.5f * ( 1.0f + normalized[1] ) * view->viewportHeight;
462 window[2] = normalized[2];
463
464 window[0] = (int) ( window[0] + 0.5 );
465 window[1] = (int) ( window[1] + 0.5 );
466}

469/*
470==========================
471myGlMultMatrix
472
473==========================
474*/
475void myGlMultMatrix( const float *a, const float *b, float *out ) {
476 int i, j;
477
478 for ( i = 0 ; i < 4 ; i++ ) {
479 for ( j = 0 ; j < 4 ; j++ ) {
480 out[ i * 4 + j ] =
481 a [ i * 4 + 0 ] * b [ 0 * 4 + j ]
482 + a [ i * 4 + 1 ] * b [ 1 * 4 + j ]
483 + a [ i * 4 + 2 ] * b [ 2 * 4 + j ]
484 + a [ i * 4 + 3 ] * b [ 3 * 4 + j ];
485 }
486 }
487}

489/*
490=================
491R_RotateForEntity
492
493Generates an orientation for an entity and viewParms
494Does NOT produce any GL calls
495Called by both the front end and the back end
496=================
497*/
498void R_RotateForEntity( const trRefEntity_t *ent, const viewParms_t *viewParms,
499 orientationr_t *or ) {
500 float glMatrix[16];
501 vec3_t delta;
502 float axisLength;
503
504 if ( ent->e.reType != RT_MODEL ) {
505 *or = viewParms->world;
506 return;
507 }
508
509 VectorCopy( ent->e.origin, or->origin );
510
511 VectorCopy( ent->e.axis[0], or->axis[0] );
512 VectorCopy( ent->e.axis[1], or->axis[1] );
513 VectorCopy( ent->e.axis[2], or->axis[2] );
514
515 glMatrix[0] = or->axis[0][0];
516 glMatrix[4] = or->axis[1][0];
517 glMatrix[8] = or->axis[2][0];
518 glMatrix[12] = or->origin[0];
519
520 glMatrix[1] = or->axis[0][1];
521 glMatrix[5] = or->axis[1][1];
522 glMatrix[9] = or->axis[2][1];
523 glMatrix[13] = or->origin[1];
524
525 glMatrix[2] = or->axis[0][2];
526 glMatrix[6] = or->axis[1][2];
527 glMatrix[10] = or->axis[2][2];
528 glMatrix[14] = or->origin[2];
529
530 glMatrix[3] = 0;
531 glMatrix[7] = 0;
532 glMatrix[11] = 0;
533 glMatrix[15] = 1;
534
535 Mat4Copy(glMatrix, or->transformMatrix);
536 myGlMultMatrix( glMatrix, viewParms->world.modelMatrix, or->modelMatrix );
537
538 // calculate the viewer origin in the model's space
539 // needed for fog, specular, and environment mapping
540 VectorSubtract( viewParms->or.origin, or->origin, delta );
541
542 // compensate for scale in the axes if necessary
543 if ( ent->e.nonNormalizedAxes ) {
544 axisLength = VectorLength( ent->e.axis[0] );
545 if ( !axisLength ) {
546 axisLength = 0;
547 } else {
548 axisLength = 1.0f / axisLength;
549 }
550 } else {
551 axisLength = 1.0f;
552 }
553
554 or->viewOrigin[0] = DotProduct( delta, or->axis[0] ) * axisLength;
555 or->viewOrigin[1] = DotProduct( delta, or->axis[1] ) * axisLength;
556 or->viewOrigin[2] = DotProduct( delta, or->axis[2] ) * axisLength;
557}

559/*
560=================
561R_RotateForViewer
562
563Sets up the modelview matrix for a given viewParm
564=================
565*/
566void R_RotateForViewer (void)
567{
568 float viewerMatrix[16];
569 vec3_t origin;
570
571 Com_Memset (&tr.or, 0, sizeof(tr.or));
572 tr.or.axis[0][0] = 1;
573 tr.or.axis[1][1] = 1;
574 tr.or.axis[2][2] = 1;
575 VectorCopy (tr.viewParms.or.origin, tr.or.viewOrigin);
576
577 // transform by the camera placement
578 VectorCopy( tr.viewParms.or.origin, origin );
579
580 viewerMatrix[0] = tr.viewParms.or.axis[0][0];
581 viewerMatrix[4] = tr.viewParms.or.axis[0][1];
582 viewerMatrix[8] = tr.viewParms.or.axis[0][2];
583 viewerMatrix[12] = -origin[0] * viewerMatrix[0] + -origin[1] * viewerMatrix[4] + -origin[2] * viewerMatrix[8];
584
585 viewerMatrix[1] = tr.viewParms.or.axis[1][0];
586 viewerMatrix[5] = tr.viewParms.or.axis[1][1];
587 viewerMatrix[9] = tr.viewParms.or.axis[1][2];
588 viewerMatrix[13] = -origin[0] * viewerMatrix[1] + -origin[1] * viewerMatrix[5] + -origin[2] * viewerMatrix[9];
589
590 viewerMatrix[2] = tr.viewParms.or.axis[2][0];
591 viewerMatrix[6] = tr.viewParms.or.axis[2][1];
592 viewerMatrix[10] = tr.viewParms.or.axis[2][2];
593 viewerMatrix[14] = -origin[0] * viewerMatrix[2] + -origin[1] * viewerMatrix[6] + -origin[2] * viewerMatrix[10];
594
595 viewerMatrix[3] = 0;
596 viewerMatrix[7] = 0;
597 viewerMatrix[11] = 0;
598 viewerMatrix[15] = 1;
599
600 // convert from our coordinate system (looking down X)
601 // to OpenGL's coordinate system (looking down -Z)
602 myGlMultMatrix( viewerMatrix, s_flipMatrix, tr.or.modelMatrix );
603
604 tr.viewParms.world = tr.or;
605
606}

608/*
609** SetFarClip
610*/
611static void R_SetFarClip( void )
612{
613 float farthestCornerDistance = 0;
614 int i;
615
616 // if not rendering the world (icons, menus, etc)
617 // set a 2k far clip plane
618 if ( tr.refdef.rdflags & RDF_NOWORLDMODEL ) {
619 tr.viewParms.zFar = 2048;
620 return;
621 }
622
623 //
624 // set far clipping planes dynamically
625 //
626 farthestCornerDistance = 0;
627 for ( i = 0; i < 8; i++ )
628 {
629 vec3_t v;
630 vec3_t vecTo;
631 float distance;
632
633 if ( i & 1 )
634 {
635 v[0] = tr.viewParms.visBounds[0][0];
636 }
637 else
638 {
639 v[0] = tr.viewParms.visBounds[1][0];
640 }
641
642 if ( i & 2 )
643 {
644 v[1] = tr.viewParms.visBounds[0][1];
645 }
646 else
647 {
648 v[1] = tr.viewParms.visBounds[1][1];
649 }
650
651 if ( i & 4 )
652 {
653 v[2] = tr.viewParms.visBounds[0][2];
654 }
655 else
656 {
657 v[2] = tr.viewParms.visBounds[1][2];
658 }
659
660 VectorSubtract( v, tr.viewParms.or.origin, vecTo );
661
662 distance = vecTo[0] * vecTo[0] + vecTo[1] * vecTo[1] + vecTo[2] * vecTo[2];
663
664 if ( distance > farthestCornerDistance )
665 {
666 farthestCornerDistance = distance;
667 }
668 }
669 tr.viewParms.zFar = sqrt( farthestCornerDistance );
670}

672/*
673=================
674R_SetupFrustum
675
676Set up the culling frustum planes for the current view using the results we got from computing the first two rows of
677the projection matrix.

679*/
680void R_SetupFrustum (viewParms_t *dest, float xmin, float xmax, float ymax, float zProj, float zFar, float stereoSep)
681{
682 vec3_t ofsorigin;
683 float oppleg, adjleg, length;
684 int i;
685
686 if(stereoSep == 0 && xmin == -xmax)
687 {
688 // symmetric case can be simplified
689 VectorCopy(dest->or.origin, ofsorigin);
690
691 length = sqrt(xmax * xmax + zProj * zProj);
692 oppleg = xmax / length;
693 adjleg = zProj / length;
694
695 VectorScale(dest->or.axis[0], oppleg, dest->frustum[0].normal);
696 VectorMA(dest->frustum[0].normal, adjleg, dest->or.axis[1], dest->frustum[0].normal);
697
698 VectorScale(dest->or.axis[0], oppleg, dest->frustum[1].normal);
699 VectorMA(dest->frustum[1].normal, -adjleg, dest->or.axis[1], dest->frustum[1].normal);
700 }
701 else
702 {
703 // In stereo rendering, due to the modification of the projection matrix, dest->or.origin is not the
704 // actual origin that we're rendering so offset the tip of the view pyramid.
705 VectorMA(dest->or.origin, stereoSep, dest->or.axis[1], ofsorigin);
706
707 oppleg = xmax + stereoSep;
708 length = sqrt(oppleg * oppleg + zProj * zProj);
709 VectorScale(dest->or.axis[0], oppleg / length, dest->frustum[0].normal);
710 VectorMA(dest->frustum[0].normal, zProj / length, dest->or.axis[1], dest->frustum[0].normal);
711
712 oppleg = xmin + stereoSep;
713 length = sqrt(oppleg * oppleg + zProj * zProj);
714 VectorScale(dest->or.axis[0], -oppleg / length, dest->frustum[1].normal);
715 VectorMA(dest->frustum[1].normal, -zProj / length, dest->or.axis[1], dest->frustum[1].normal);
716 }
717
718 length = sqrt(ymax * ymax + zProj * zProj);
719 oppleg = ymax / length;
720 adjleg = zProj / length;
721
722 VectorScale(dest->or.axis[0], oppleg, dest->frustum[2].normal);
723 VectorMA(dest->frustum[2].normal, adjleg, dest->or.axis[2], dest->frustum[2].normal);
724
725 VectorScale(dest->or.axis[0], oppleg, dest->frustum[3].normal);
726 VectorMA(dest->frustum[3].normal, -adjleg, dest->or.axis[2], dest->frustum[3].normal);
727
728 for (i=0 ; i<4 ; i++) {
729 dest->frustum[i].type = PLANE_NON_AXIAL;
730 dest->frustum[i].dist = DotProduct (ofsorigin, dest->frustum[i].normal);
731 SetPlaneSignbits( &dest->frustum[i] );
732 }
733
734 if (zFar != 0.0f)
735 {
736 vec3_t farpoint;
737
738 VectorMA(ofsorigin, zFar, dest->or.axis[0], farpoint);
739 VectorScale(dest->or.axis[0], -1.0f, dest->frustum[4].normal);
740
741 dest->frustum[4].type = PLANE_NON_AXIAL;
742 dest->frustum[4].dist = DotProduct (farpoint, dest->frustum[4].normal);
743 SetPlaneSignbits( &dest->frustum[4] );
744 dest->flags |= VPF_FARPLANEFRUSTUM;
745 }
746}

748/*
749===============
750R_SetupProjection
751===============
752*/
753void R_SetupProjection(viewParms_t *dest, float zProj, float zFar, qboolean computeFrustum)
754{
755 float xmin, xmax, ymin, ymax;
756 float width, height, stereoSep = r_stereoSeparation->value;
757
758 /*
759 * offset the view origin of the viewer for stereo rendering
760 * by setting the projection matrix appropriately.
761 */
762
763 if(stereoSep != 0)
764 {
765 if(dest->stereoFrame == STEREO_LEFT)
766 stereoSep = zProj / stereoSep;
767 else if(dest->stereoFrame == STEREO_RIGHT)
768 stereoSep = zProj / -stereoSep;
769 else
770 stereoSep = 0;
771 }
772
773 ymax = zProj * tan(dest->fovY * M_PI / 360.0f);
774 ymin = -ymax;
775
776 xmax = zProj * tan(dest->fovX * M_PI / 360.0f);
777 xmin = -xmax;
778
779 width = xmax - xmin;
780 height = ymax - ymin;
781
782 dest->projectionMatrix[0] = 2 * zProj / width;
783 dest->projectionMatrix[4] = 0;
784 dest->projectionMatrix[8] = (xmax + xmin + 2 * stereoSep) / width;
785 dest->projectionMatrix[12] = 2 * zProj * stereoSep / width;
786
787 dest->projectionMatrix[1] = 0;
788 dest->projectionMatrix[5] = 2 * zProj / height;
789 dest->projectionMatrix[9] = ( ymax + ymin ) / height; // normally 0
790 dest->projectionMatrix[13] = 0;
791
792 dest->projectionMatrix[3] = 0;
793 dest->projectionMatrix[7] = 0;
794 dest->projectionMatrix[11] = -1;
795 dest->projectionMatrix[15] = 0;
796
797 // Now that we have all the data for the projection matrix we can also setup the view frustum.
798 if(computeFrustum)
799 R_SetupFrustum(dest, xmin, xmax, ymax, zProj, zFar, stereoSep);
800}

802/*
803===============
804R_SetupProjectionZ
805
806Sets the z-component transformation part in the projection matrix
807===============
808*/
809void R_SetupProjectionZ(viewParms_t *dest)
810{
811 float zNear, zFar, depth;
812
813 zNear = r_znear->value;
814 zFar = dest->zFar;
815
816 depth = zFar - zNear;
817
818 dest->projectionMatrix[2] = 0;
819 dest->projectionMatrix[6] = 0;
820 dest->projectionMatrix[10] = -( zFar + zNear ) / depth;
821 dest->projectionMatrix[14] = -2 * zFar * zNear / depth;
822
823 if (dest->isPortal)
824 {
825 float plane[4];
826 float plane2[4];
827 vec4_t q, c;
828
829 // transform portal plane into camera space
830 plane[0] = dest->portalPlane.normal[0];
831 plane[1] = dest->portalPlane.normal[1];
832 plane[2] = dest->portalPlane.normal[2];
833 plane[3] = dest->portalPlane.dist;
834
835 plane2[0] = -DotProduct (dest->or.axis[1], plane);
836 plane2[1] = DotProduct (dest->or.axis[2], plane);
837 plane2[2] = -DotProduct (dest->or.axis[0], plane);
838 plane2[3] = DotProduct (plane, dest->or.origin) - plane[3];
839
840 // Lengyel, Eric. "Modifying the Projection Matrix to Perform Oblique Near-plane Clipping".
841 // Terathon Software 3D Graphics Library, 2004. http://www.terathon.com/code/oblique.html
842 q[0] = (SGN(plane2[0]) + dest->projectionMatrix[8]) / dest->projectionMatrix[0];
843 q[1] = (SGN(plane2[1]) + dest->projectionMatrix[9]) / dest->projectionMatrix[5];
844 q[2] = -1.0f;
845 q[3] = (1.0f + dest->projectionMatrix[10]) / dest->projectionMatrix[14];
846
847 VectorScale4(plane2, 2.0f / DotProduct4(plane2, q), c);
848
849 dest->projectionMatrix[2] = c[0];
850 dest->projectionMatrix[6] = c[1];
851 dest->projectionMatrix[10] = c[2] + 1.0f;
852 dest->projectionMatrix[14] = c[3];
853
854 }
855
856}

863void R_SetupProjectionOrtho(viewParms_t *dest, vec3_t viewBounds[2])

898
899 dest->projectionMatrix[3] = 0;
900 dest->projectionMatrix[7] = 0;
901 dest->projectionMatrix[11] = 0;
902 dest->projectionMatrix[15] = 1;
903
904 VectorScale(dest->or.axis[1], 1.0f, dest->frustum[0].normal);

915
916 VectorScale(dest->or.axis[2], -1.0f, dest->frustum[3].normal);
917 VectorMA(dest->or.origin, -viewBounds[1][2], dest->frustum[3].normal, pop);
918 dest->frustum[3].dist = DotProduct(pop, dest->frustum[3].normal);
919
920 VectorScale(dest->or.axis[0], -1.0f, dest->frustum[4].normal);
921 VectorMA(dest->or.origin, -viewBounds[1][0], dest->frustum[4].normal, pop);

968=============
969R_PlaneForSurface
970=============
971*/
972void R_PlaneForSurface (surfaceType_t *surfType, cplane_t *plane) {
973 srfBspSurface_t *tri;
974 srfPoly_t *poly;
975 srfVert_t *v1, *v2, *v3;
976 vec4_t plane4;
977
978 if (!surfType) {
979 Com_Memset (plane, 0, sizeof(*plane));
980 plane->normal[0] = 1;
981 return;
982 }
983 switch (*surfType) {
984 case SF_FACE:
985 *plane = ((srfBspSurface_t *)surfType)->cullPlane;
986 return;
987 case SF_TRIANGLES:
988 tri = (srfBspSurface_t *)surfType;
989 v1 = tri->verts + tri->indexes[0];
990 v2 = tri->verts + tri->indexes[1];
991 v3 = tri->verts + tri->indexes[2];
992 PlaneFromPoints( plane4, v1->xyz, v2->xyz, v3->xyz );
993 VectorCopy( plane4, plane->normal );
994 plane->dist = plane4[3];
995 return;
996 case SF_POLY:
997 poly = (srfPoly_t *)surfType;
998 PlaneFromPoints( plane4, poly->verts[0].xyz, poly->verts[1].xyz, poly->verts[2].xyz );
999 VectorCopy( plane4, plane->normal );
1000 plane->dist = plane4[3];
1001 return;
1002 default:
1003 Com_Memset (plane, 0, sizeof(*plane));
1004 plane->normal[0] = 1;
1005 return;
1006 }
1007}

1009/*
1010=================
1011R_GetPortalOrientation
1012
1013entityNum is the entity that the portal surface is a part of, which may
1014be moving and rotating.
1015
1016Returns qtrue if it should be mirrored
1017=================
1018*/
1019qboolean R_GetPortalOrientations( drawSurf_t *drawSurf, int entityNum,
1020 orientation_t *surface, orientation_t *camera,
1021 vec3_t pvsOrigin, qboolean *mirror ) {
1022 int i;
1023 cplane_t originalPlane, plane;
1024 trRefEntity_t *e;
1025 float d;
1026 vec3_t transformed;
1027
1028 // create plane axis for the portal we are seeing
1029 R_PlaneForSurface( drawSurf->surface, &originalPlane );
1030
1031 // rotate the plane if necessary
1032 if ( entityNum != REFENTITYNUM_WORLD ) {
1033 tr.currentEntityNum = entityNum;
1034 tr.currentEntity = &tr.refdef.entities[entityNum];
1035
1036 // get the orientation of the entity
1037 R_RotateForEntity( tr.currentEntity, &tr.viewParms, &tr.or );
1038
1039 // rotate the plane, but keep the non-rotated version for matching
1040 // against the portalSurface entities
1041 R_LocalNormalToWorld( originalPlane.normal, plane.normal );
1042 plane.dist = originalPlane.dist + DotProduct( plane.normal, tr.or.origin );
1043
1044 // translate the original plane
1045 originalPlane.dist = originalPlane.dist + DotProduct( originalPlane.normal, tr.or.origin );
1046 } else {
1047 plane = originalPlane;
1048 }
1049
1050 VectorCopy( plane.normal, surface->axis[0] );
1051 PerpendicularVector( surface->axis[1], surface->axis[0] );
1052 CrossProduct( surface->axis[0], surface->axis[1], surface->axis[2] );
1053
1054 // locate the portal entity closest to this plane.
1055 // origin will be the origin of the portal, origin2 will be
1056 // the origin of the camera
1057 for ( i = 0 ; i < tr.refdef.num_entities ; i++ ) {
1058 e = &tr.refdef.entities[i];
1059 if ( e->e.reType != RT_PORTALSURFACE ) {
1060 continue;
1061 }
1062
1063 d = DotProduct( e->e.origin, originalPlane.normal ) - originalPlane.dist;
1064 if ( d > 64 || d < -64) {
1065 continue;
1066 }
1067
1068 // get the pvsOrigin from the entity
1069 VectorCopy( e->e.oldorigin, pvsOrigin );
1070
1071 // if the entity is just a mirror, don't use as a camera point
1072 if ( e->e.oldorigin[0] == e->e.origin[0] &&
1073 e->e.oldorigin[1] == e->e.origin[1] &&
1074 e->e.oldorigin[2] == e->e.origin[2] ) {
1075 VectorScale( plane.normal, plane.dist, surface->origin );
1076 VectorCopy( surface->origin, camera->origin );
1077 VectorSubtract( vec3_origin, surface->axis[0], camera->axis[0] );
1078 VectorCopy( surface->axis[1], camera->axis[1] );
1079 VectorCopy( surface->axis[2], camera->axis[2] );
1080
1081 *mirror = qtrue;
1082 return qtrue;
1083 }
1084
1085 // project the origin onto the surface plane to get
1086 // an origin point we can rotate around
1087 d = DotProduct( e->e.origin, plane.normal ) - plane.dist;
1088 VectorMA( e->e.origin, -d, surface->axis[0], surface->origin );
1089
1090 // now get the camera origin and orientation
1091 VectorCopy( e->e.oldorigin, camera->origin );
1092 AxisCopy( e->e.axis, camera->axis );
1093 VectorSubtract( vec3_origin, camera->axis[0], camera->axis[0] );
1094 VectorSubtract( vec3_origin, camera->axis[1], camera->axis[1] );
1095
1096 // optionally rotate
1097 if ( e->e.oldframe ) {
1098 // if a speed is specified
1099 if ( e->e.frame ) {
1100 // continuous rotate
1101 d = (tr.refdef.time/1000.0f) * e->e.frame;
1102 VectorCopy( camera->axis[1], transformed );
1103 RotatePointAroundVector( camera->axis[1], camera->axis[0], transformed, d );
1104 CrossProduct( camera->axis[0], camera->axis[1], camera->axis[2] );
1105 } else {
1106 // bobbing rotate, with skinNum being the rotation offset
1107 d = sin( tr.refdef.time * 0.003f );
1108 d = e->e.skinNum + d * 4;
1109 VectorCopy( camera->axis[1], transformed );
1110 RotatePointAroundVector( camera->axis[1], camera->axis[0], transformed, d );
1111 CrossProduct( camera->axis[0], camera->axis[1], camera->axis[2] );
1112 }
1113 }
1114 else if ( e->e.skinNum ) {
1115 d = e->e.skinNum;
1116 VectorCopy( camera->axis[1], transformed );
1117 RotatePointAroundVector( camera->axis[1], camera->axis[0], transformed, d );
1118 CrossProduct( camera->axis[0], camera->axis[1], camera->axis[2] );
1119 }
1120 *mirror = qfalse;
1121 return qtrue;
1122 }
1123
1124 // if we didn't locate a portal entity, don't render anything.
1125 // We don't want to just treat it as a mirror, because without a
1126 // portal entity the server won't have communicated a proper entity set
1127 // in the snapshot
1128
1129 // unfortunately, with local movement prediction it is easily possible
1130 // to see a surface before the server has communicated the matching
1131 // portal surface entity, so we don't want to print anything here...
1132
1133 //ri.Printf( PRINT_ALL, "Portal surface without a portal entity\n" );
1134
1135 return qfalse;
1136}

1194/*
1195** SurfIsOffscreen
1196**
1197** Determines if a surface is completely offscreen.
1198*/
1199static qboolean SurfIsOffscreen( const drawSurf_t *drawSurf, vec4_t clipDest[128] ) {
1200 float shortest = 100000000;
1201 int entityNum;
1202 int numTriangles;
1203 shader_t *shader;
1204 int fogNum;
1205 int dlighted;
1206 int pshadowed;
1207 vec4_t clip, eye;
1208 int i;
1209 unsigned int pointOr = 0;
1210 unsigned int pointAnd = (unsigned int)~0;
1211
1212 R_RotateForViewer();
1213
1214 R_DecomposeSort( drawSurf->sort, &entityNum, &shader, &fogNum, &dlighted, &pshadowed );
1215 RB_BeginSurface( shader, fogNum, drawSurf->cubemapIndex);
1216 rb_surfaceTable[ *drawSurf->surface ]( drawSurf->surface );
1217
1218 assert( tess.numVertexes < 128 );
1219
1220 for ( i = 0; i < tess.numVertexes; i++ )
1221 {
1222 int j;
1223 unsigned int pointFlags = 0;
1224
1225 R_TransformModelToClip( tess.xyz[i], tr.or.modelMatrix, tr.viewParms.projectionMatrix, eye, clip );
1226
1227 for ( j = 0; j < 3; j++ )
1228 {
1229 if ( clip[j] >= clip[3] )
1230 {
1231 pointFlags |= (1 << (j*2));
1232 }
1233 else if ( clip[j] <= -clip[3] )
1234 {
1235 pointFlags |= ( 1 << (j*2+1));
1236 }
1237 }
1238 pointAnd &= pointFlags;
1239 pointOr |= pointFlags;
1240 }
1241
1242 // trivially reject
1243 if ( pointAnd )
1244 {
1245 return qtrue;
1246 }
1247
1248 // determine if this surface is backfaced and also determine the distance
1249 // to the nearest vertex so we can cull based on portal range. Culling
1250 // based on vertex distance isn't 100% correct (we should be checking for
1251 // range to the surface), but it's good enough for the types of portals
1252 // we have in the game right now.
1253 numTriangles = tess.numIndexes / 3;
1254
1255 for ( i = 0; i < tess.numIndexes; i += 3 )
1256 {
1257 vec3_t normal, tNormal;
1258
1259 float len;
1260
1261 VectorSubtract( tess.xyz[tess.indexes[i]], tr.viewParms.or.origin, normal );
1262
1263 len = VectorLengthSquared( normal ); // lose the sqrt
1264 if ( len < shortest )
1265 {
1266 shortest = len;
1267 }
1268
1269 R_VaoUnpackNormal(tNormal, tess.normal[tess.indexes[i]]);
1270
1271 if ( DotProduct( normal, tNormal ) >= 0 )
1272 {
1273 numTriangles--;
1274 }
1275 }
1276 if ( !numTriangles )
1277 {
1278 return qtrue;
1279 }
1280
1281 // mirrors can early out at this point, since we don't do a fade over distance
1282 // with them (although we could)
1283 if ( IsMirror( drawSurf, entityNum ) )
1284 {
1285 return qfalse;
1286 }
1287
1288 if ( shortest > (tess.shader->portalRange*tess.shader->portalRange) )
1289 {
1290 return qtrue;
1291 }
1292
1293 return qfalse;
1294}

1298R_MirrorViewBySurface
1299
1300Returns qtrue if another view has been rendered
1301========================
1302*/
1303qboolean R_MirrorViewBySurface (drawSurf_t *drawSurf, int entityNum) {
1304 vec4_t clipDest[128];
1305 viewParms_t newParms;
1306 viewParms_t oldParms;
1307 orientation_t surface, camera;
1308

1318
1319 // trivially reject portal/mirror
1320 if ( SurfIsOffscreen( drawSurf, clipDest ) ) {
1321 return qfalse;
1322 }
1323
1324 // save old viewParms so we can return to it after the mirror view
1325 oldParms = tr.viewParms;
1326
1327 newParms = tr.viewParms;
1328 newParms.isPortal = qtrue;
1329 newParms.zFar = 0.0f;
1330 newParms.flags &= ~VPF_FARPLANEFRUSTUM;
1331 if ( !R_GetPortalOrientations( drawSurf, entityNum, &surface, &camera,
1332 newParms.pvsOrigin, &newParms.isMirror ) ) {
1333 return qfalse; // bad portal, no portalentity
1334 }
1335
1336 // Never draw viewmodels in portal or mirror views.
1337 newParms.flags |= VPF_NOVIEWMODEL;
1338
1339 R_MirrorPoint (oldParms.or.origin, &surface, &camera, newParms.or.origin );
1340
1341 VectorSubtract( vec3_origin, camera.axis[0], newParms.portalPlane.normal );
1342 newParms.portalPlane.dist = DotProduct( camera.origin, newParms.portalPlane.normal );
1343
1344 R_MirrorVector (oldParms.or.axis[0], &surface, &camera, newParms.or.axis[0]);
1345 R_MirrorVector (oldParms.or.axis[1], &surface, &camera, newParms.or.axis[1]);
1346 R_MirrorVector (oldParms.or.axis[2], &surface, &camera, newParms.or.axis[2]);
1347
1348 // OPTIMIZE: restrict the viewport on the mirrored view
1349
1350 // render the mirror view
1351 R_RenderView (&newParms);
1352
1353 tr.viewParms = oldParms;
1354
1355 return qtrue;
1356}

1403/*
1404===============
1405R_Radix
1406===============
1407*/
1408static ID_INLINE void R_Radix( int byte, int size, drawSurf_t *source, drawSurf_t *dest )
1409{
1410 int count[ 256 ] = { 0 };
1411 int index[ 256 ];
1412 int i;
1413 unsigned char *sortKey = NULL;
1414 unsigned char *end = NULL;
1415
1416 sortKey = ( (unsigned char *)&source[ 0 ].sort ) + byte;
1417 end = sortKey + ( size * sizeof( drawSurf_t ) );
1418 for( ; sortKey < end; sortKey += sizeof( drawSurf_t ) )
1419 ++count[ *sortKey ];
1420
1421 index[ 0 ] = 0;
1422
1423 for( i = 1; i < 256; ++i )
1424 index[ i ] = index[ i - 1 ] + count[ i - 1 ];
1425
1426 sortKey = ( (unsigned char *)&source[ 0 ].sort ) + byte;
1427 for( i = 0; i < size; ++i, sortKey += sizeof( drawSurf_t ) )
1428 dest[ index[ *sortKey ]++ ] = source[ i ];
1429}

1432===============
1433R_RadixSort
1434
1435Radix sort with 4 byte size buckets
1436===============
1437*/
1438static void R_RadixSort( drawSurf_t *source, int size )
1439{
1440 static drawSurf_t scratch[ MAX_DRAWSURFS ];
1441#ifdef Q3_LITTLE_ENDIAN
1442 R_Radix( 0, size, source, scratch );
1443 R_Radix( 1, size, scratch, source );
1444 R_Radix( 2, size, source, scratch );
1445 R_Radix( 3, size, scratch, source );
1446#else
1447 R_Radix( 3, size, source, scratch );
1448 R_Radix( 2, size, scratch, source );
1449 R_Radix( 1, size, source, scratch );
1450 R_Radix( 0, size, scratch, source );
1451#endif //Q3_LITTLE_ENDIAN
1452}

1492/*
1493=================
1494R_SortDrawSurfs
1495=================
1496*/
1497void R_SortDrawSurfs( drawSurf_t *drawSurfs, int numDrawSurfs ) {
1498 shader_t *shader;
1499 int fogNum;
1500 int entityNum;
1501 int dlighted;
1502 int pshadowed;
1503 int i;
1504
1505 //ri.Printf(PRINT_ALL, "firstDrawSurf %d numDrawSurfs %d\n", (int)(drawSurfs - tr.refdef.drawSurfs), numDrawSurfs);
1506
1507 // it is possible for some views to not have any surfaces
1508 if ( numDrawSurfs < 1 ) {
1509 // we still need to add it for hyperspace cases
1510 R_AddDrawSurfCmd( drawSurfs, numDrawSurfs );
1511 return;
1512 }
1513
1514 // sort the drawsurfs by sort type, then orientation, then shader
1515 R_RadixSort( drawSurfs, numDrawSurfs );
1516
1517 // skip pass through drawing if rendering a shadow map
1518 if (tr.viewParms.flags & (VPF_SHADOWMAP | VPF_DEPTHSHADOW))
1519 {
1520 R_AddDrawSurfCmd( drawSurfs, numDrawSurfs );
1521 return;
1522 }
1523
1524 // check for any pass through drawing, which
1525 // may cause another view to be rendered first
1526 for ( i = 0 ; i < numDrawSurfs ; i++ ) {
1527 R_DecomposeSort( (drawSurfs+i)->sort, &entityNum, &shader, &fogNum, &dlighted, &pshadowed );
1528
1529 if ( shader->sort > SS_PORTAL ) {
1530 break;
1531 }
1532
1533 // no shader should ever have this sort type
1534 if ( shader->sort == SS_BAD ) {
1535 ri.Error (ERR_DROP, "Shader '%s'with sort == SS_BAD", shader->name );
1536 }
1537
1538 // if the mirror was completely clipped away, we may need to check another surface
1539 if ( R_MirrorViewBySurface( (drawSurfs+i), entityNum) ) {
1540 // this is a debug option to see exactly what is being mirrored
1541 if ( r_portalOnly->integer ) {
1542 return;
1543 }
1544 break; // only one mirror view at a time
1545 }
1546 }
1547
1548 R_AddDrawSurfCmd( drawSurfs, numDrawSurfs );
1549}

30#define WAVEVALUE( table, base, amplitude, phase, freq ) ((base) + table[ ( (int64_t) ( ( (phase) + tess.shaderTime * (freq) ) * FUNCTABLE_SIZE ) ) & FUNCTABLE_MASK ] * (amplitude))
31
32static float *TableForFunc( genFunc_t func )
33{
34 switch ( func )
35 {
36 case GF_SIN:
37 return tr.sinTable;
38 case GF_TRIANGLE:
39 return tr.triangleTable;
40 case GF_SQUARE:
41 return tr.squareTable;
42 case GF_SAWTOOTH:
43 return tr.sawToothTable;
44 case GF_INVERSE_SAWTOOTH:
45 return tr.inverseSawToothTable;
46 case GF_NONE:
47 default:
48 break;
49 }
50
51 ri.Error( ERR_DROP, "TableForFunc called with invalid function '%d' in shader '%s'", func, tess.shader->name );
52 return NULL;
53}

55/*
56** EvalWaveForm
57**
58** Evaluates a given waveForm_t, referencing backEnd.refdef.time directly
59*/
60static float EvalWaveForm( const waveForm_t *wf )
61{
62 float *table;
63
64 table = TableForFunc( wf->func );
65
66 return WAVEVALUE( table, wf->base, wf->amplitude, wf->phase, wf->frequency );
67}
68
69static float EvalWaveFormClamped( const waveForm_t *wf )
70{
71 float glow = EvalWaveForm( wf );
72
73 if ( glow < 0 )
74 {
75 return 0;
76 }
77
78 if ( glow > 1 )
79 {
80 return 1;
81 }
82
83 return glow;
84}

86/*
87** RB_CalcStretchTexMatrix
88*/
89void RB_CalcStretchTexMatrix( const waveForm_t *wf, float *matrix )
90{
91 float p;
92
93 p = 1.0f / EvalWaveForm( wf );
94
95 matrix[0] = p; matrix[2] = 0; matrix[4] = 0.5f - 0.5f * p;
96 matrix[1] = 0; matrix[3] = p; matrix[5] = 0.5f - 0.5f * p;
97}

107/*
108========================
109RB_CalcDeformVertexes
110
111========================
112*/
113void RB_CalcDeformVertexes( deformStage_t *ds )
114{
115 int i;
116 vec3_t offset;
117 float scale;
118 float *xyz = ( float * ) tess.xyz;
119 int16_t *normal = tess.normal[0];
120 float *table;
121
122 if ( ds->deformationWave.frequency == 0 )
123 {
124 scale = EvalWaveForm( &ds->deformationWave );
125
126 for ( i = 0; i < tess.numVertexes; i++, xyz += 4, normal += 4 )
127 {
128 R_VaoUnpackNormal(offset, normal);
129
130 xyz[0] += offset[0] * scale;
131 xyz[1] += offset[1] * scale;
132 xyz[2] += offset[2] * scale;
133 }
134 }
135 else
136 {
137 table = TableForFunc( ds->deformationWave.func );
138
139 for ( i = 0; i < tess.numVertexes; i++, xyz += 4, normal += 4 )
140 {
141 float off = ( xyz[0] + xyz[1] + xyz[2] ) * ds->deformationSpread;
142
143 scale = WAVEVALUE( table, ds->deformationWave.base,
144 ds->deformationWave.amplitude,
145 ds->deformationWave.phase + off,
146 ds->deformationWave.frequency );
147
148 R_VaoUnpackNormal(offset, normal);
149
150 xyz[0] += offset[0] * scale;
151 xyz[1] += offset[1] * scale;
152 xyz[2] += offset[2] * scale;
153 }
154 }
155}

157/*
158=========================
159RB_CalcDeformNormals
160
161Wiggle the normals for wavy environment mapping
162=========================
163*/
164void RB_CalcDeformNormals( deformStage_t *ds ) {
165 int i;
166 float scale;
167 float *xyz = ( float * ) tess.xyz;
168 int16_t *normal = tess.normal[0];
169
170 for ( i = 0; i < tess.numVertexes; i++, xyz += 4, normal += 4 ) {
171 vec3_t fNormal;
172
173 R_VaoUnpackNormal(fNormal, normal);
174
175 scale = 0.98f;
176 scale = R_NoiseGet4f( xyz[0] * scale, xyz[1] * scale, xyz[2] * scale,
177 tess.shaderTime * ds->deformationWave.frequency );
178 fNormal[ 0 ] += ds->deformationWave.amplitude * scale;
179
180 scale = 0.98f;
181 scale = R_NoiseGet4f( 100 + xyz[0] * scale, xyz[1] * scale, xyz[2] * scale,
182 tess.shaderTime * ds->deformationWave.frequency );
183 fNormal[ 1 ] += ds->deformationWave.amplitude * scale;
184
185 scale = 0.98f;
186 scale = R_NoiseGet4f( 200 + xyz[0] * scale, xyz[1] * scale, xyz[2] * scale,
187 tess.shaderTime * ds->deformationWave.frequency );
188 fNormal[ 2 ] += ds->deformationWave.amplitude * scale;
189
190 VectorNormalizeFast( fNormal );
191
192 R_VaoPackNormal(normal, fNormal);
193 }
194}

196/*
197========================
198RB_CalcBulgeVertexes
199
200========================
201*/
202void RB_CalcBulgeVertexes( deformStage_t *ds ) {
203 int i;
204 const float *st = ( const float * ) tess.texCoords[0];
205 float *xyz = ( float * ) tess.xyz;
206 int16_t *normal = tess.normal[0];
207 double now;
208
209 now = backEnd.refdef.time * 0.001 * ds->bulgeSpeed;
210
211 for ( i = 0; i < tess.numVertexes; i++, xyz += 4, st += 2, normal += 4 ) {
212 int64_t off;
213 float scale;
214 vec3_t fNormal;
215
216 R_VaoUnpackNormal(fNormal, normal);
217
218 off = (float)( FUNCTABLE_SIZE / (M_PI*2) ) * ( st[0] * ds->bulgeWidth + now );
219
220 scale = tr.sinTable[ off & FUNCTABLE_MASK ] * ds->bulgeHeight;
221
222 xyz[0] += fNormal[0] * scale;
223 xyz[1] += fNormal[1] * scale;
224 xyz[2] += fNormal[2] * scale;
225 }
226}

229/*
230======================
231RB_CalcMoveVertexes
232
233A deformation that can move an entire surface along a wave path
234======================
235*/
236void RB_CalcMoveVertexes( deformStage_t *ds ) {
237 int i;
238 float *xyz;
239 float *table;
240 float scale;
241 vec3_t offset;
242
243 table = TableForFunc( ds->deformationWave.func );
244
245 scale = WAVEVALUE( table, ds->deformationWave.base,
246 ds->deformationWave.amplitude,
247 ds->deformationWave.phase,
248 ds->deformationWave.frequency );
249
250 VectorScale( ds->moveVector, scale, offset );
251
252 xyz = ( float * ) tess.xyz;
253 for ( i = 0; i < tess.numVertexes; i++, xyz += 4 ) {
254 VectorAdd( xyz, offset, xyz );
255 }
256}

259/*
260=============
261DeformText
262
263Change a polygon into a bunch of text polygons
264=============
265*/
266void DeformText( const char *text ) {
267 int i;
268 vec3_t origin, width, height;
269 int len;
270 int ch;
271 float color[4];
272 float bottom, top;
273 vec3_t mid;
274 vec3_t fNormal;
275
276 height[0] = 0;
277 height[1] = 0;
278 height[2] = -1;
279
280 R_VaoUnpackNormal(fNormal, tess.normal[0]);
281 CrossProduct( fNormal, height, width );
282
283 // find the midpoint of the box
284 VectorClear( mid );
285 bottom = 999999;
286 top = -999999;
287 for ( i = 0 ; i < 4 ; i++ ) {
288 VectorAdd( tess.xyz[i], mid, mid );
289 if ( tess.xyz[i][2] < bottom ) {
290 bottom = tess.xyz[i][2];
291 }
292 if ( tess.xyz[i][2] > top ) {
293 top = tess.xyz[i][2];
294 }
295 }
296 VectorScale( mid, 0.25f, origin );
297
298 // determine the individual character size
299 height[0] = 0;
300 height[1] = 0;
301 height[2] = ( top - bottom ) * 0.5f;
302
303 VectorScale( width, height[2] * -0.75f, width );
304
305 // determine the starting position
306 len = strlen( text );
307 VectorMA( origin, (len-1), width, origin );
308
309 // clear the shader indexes
310 tess.numIndexes = 0;
311 tess.numVertexes = 0;
312 tess.firstIndex = 0;
313
314 color[0] = color[1] = color[2] = color[3] = 1.0f;
315
316 // draw each character
317 for ( i = 0 ; i < len ; i++ ) {
318 ch = text[i];
319 ch &= 255;
320
321 if ( ch != ' ' ) {
322 int row, col;
323 float frow, fcol, size;
324
325 row = ch>>4;
326 col = ch&15;
327
328 frow = row*0.0625f;
329 fcol = col*0.0625f;
330 size = 0.0625f;
331
332 RB_AddQuadStampExt( origin, width, height, color, fcol, frow, fcol + size, frow + size );
333 }
334 VectorMA( origin, -2, width, origin );
335 }
336}

343static void GlobalVectorToLocal( const vec3_t in, vec3_t out ) {
344 out[0] = DotProduct( in, backEnd.or.axis[0] );
345 out[1] = DotProduct( in, backEnd.or.axis[1] );
346 out[2] = DotProduct( in, backEnd.or.axis[2] );

349/*
350=====================
351AutospriteDeform
352
353Assuming all the triangles for this shader are independant
354quads, rebuild them as forward facing sprites
355=====================
356*/
357static void AutospriteDeform( void ) {
358 int i;
359 int oldVerts;
360 float *xyz;
361 vec3_t mid, delta;
362 float radius;
363 vec3_t left, up;
364 vec3_t leftDir, upDir;
365
366 if ( tess.numVertexes & 3 ) {
367 ri.Printf( PRINT_WARNING, "Autosprite shader %s had odd vertex count\n", tess.shader->name );
368 }
369 if ( tess.numIndexes != ( tess.numVertexes >> 2 ) * 6 ) {
370 ri.Printf( PRINT_WARNING, "Autosprite shader %s had odd index count\n", tess.shader->name );
371 }
372
373 oldVerts = tess.numVertexes;
374 tess.numVertexes = 0;
375 tess.numIndexes = 0;
376 tess.firstIndex = 0;
377
378 if ( backEnd.currentEntity != &tr.worldEntity ) {
379 GlobalVectorToLocal( backEnd.viewParms.or.axis[1], leftDir );
380 GlobalVectorToLocal( backEnd.viewParms.or.axis[2], upDir );
381 } else {
382 VectorCopy( backEnd.viewParms.or.axis[1], leftDir );
383 VectorCopy( backEnd.viewParms.or.axis[2], upDir );
384 }
385
386 for ( i = 0 ; i < oldVerts ; i+=4 ) {
387 vec4_t color;
388 // find the midpoint
389 xyz = tess.xyz[i];
390
391 mid[0] = 0.25f * (xyz[0] + xyz[4] + xyz[8] + xyz[12]);
392 mid[1] = 0.25f * (xyz[1] + xyz[5] + xyz[9] + xyz[13]);
393 mid[2] = 0.25f * (xyz[2] + xyz[6] + xyz[10] + xyz[14]);
394
395 VectorSubtract( xyz, mid, delta );
396 radius = VectorLength( delta ) * 0.707f; // / sqrt(2)
397
398 VectorScale( leftDir, radius, left );
399 VectorScale( upDir, radius, up );
400
401 if ( backEnd.viewParms.isMirror ) {
402 VectorSubtract( vec3_origin, left, left );
403 }
404
405 // compensate for scale in the axes if necessary
406 if ( backEnd.currentEntity->e.nonNormalizedAxes ) {
407 float axisLength;
408 axisLength = VectorLength( backEnd.currentEntity->e.axis[0] );
409 if ( !axisLength ) {
410 axisLength = 0;
411 } else {
412 axisLength = 1.0f / axisLength;
413 }
414 VectorScale(left, axisLength, left);
415 VectorScale(up, axisLength, up);
416 }
417
418 VectorScale4(tess.color[i], 1.0f / 65535.0f, color);
419 RB_AddQuadStamp( mid, left, up, color );
420 }
421}

424/*
425=====================
426Autosprite2Deform
427
428Autosprite2 will pivot a rectangular quad along the center of its long axis
429=====================
430*/
431int edgeVerts[6][2] = {
432 { 0, 1 },
433 { 0, 2 },
434 { 0, 3 },
435 { 1, 2 },
436 { 1, 3 },
437 { 2, 3 }
438};
439
440static void Autosprite2Deform( void ) {
441 int i, j, k;
442 int indexes;
443 float *xyz;
444 vec3_t forward;
445
446 if ( tess.numVertexes & 3 ) {
447 ri.Printf( PRINT_WARNING, "Autosprite2 shader %s had odd vertex count\n", tess.shader->name );
448 }
449 if ( tess.numIndexes != ( tess.numVertexes >> 2 ) * 6 ) {
450 ri.Printf( PRINT_WARNING, "Autosprite2 shader %s had odd index count\n", tess.shader->name );
451 }
452
453 if ( backEnd.currentEntity != &tr.worldEntity ) {
454 GlobalVectorToLocal( backEnd.viewParms.or.axis[0], forward );
455 } else {
456 VectorCopy( backEnd.viewParms.or.axis[0], forward );
457 }
458
459 // this is a lot of work for two triangles...
460 // we could precalculate a lot of it is an issue, but it would mess up
461 // the shader abstraction
462 for ( i = 0, indexes = 0 ; i < tess.numVertexes ; i+=4, indexes+=6 ) {
463 float lengths[2];
464 int nums[2];
465 vec3_t mid[2];
466 vec3_t major, minor;
467 float *v1, *v2;
468
469 // find the midpoint
470 xyz = tess.xyz[i];
471
472 // identify the two shortest edges
473 nums[0] = nums[1] = 0;
474 lengths[0] = lengths[1] = 999999;
475
476 for ( j = 0 ; j < 6 ; j++ ) {
477 float l;
478 vec3_t temp;
479
480 v1 = xyz + 4 * edgeVerts[j][0];
481 v2 = xyz + 4 * edgeVerts[j][1];
482
483 VectorSubtract( v1, v2, temp );
484
485 l = DotProduct( temp, temp );
486 if ( l < lengths[0] ) {
487 nums[1] = nums[0];
488 lengths[1] = lengths[0];
489 nums[0] = j;
490 lengths[0] = l;
491 } else if ( l < lengths[1] ) {
492 nums[1] = j;
493 lengths[1] = l;
494 }
495 }
496
497 for ( j = 0 ; j < 2 ; j++ ) {
498 v1 = xyz + 4 * edgeVerts[nums[j]][0];
499 v2 = xyz + 4 * edgeVerts[nums[j]][1];
500
501 mid[j][0] = 0.5f * (v1[0] + v2[0]);
502 mid[j][1] = 0.5f * (v1[1] + v2[1]);
503 mid[j][2] = 0.5f * (v1[2] + v2[2]);
504 }
505
506 // find the vector of the major axis
507 VectorSubtract( mid[1], mid[0], major );
508
509 // cross this with the view direction to get minor axis
510 CrossProduct( major, forward, minor );
511 VectorNormalize( minor );
512
513 // re-project the points
514 for ( j = 0 ; j < 2 ; j++ ) {
515 float l;
516
517 v1 = xyz + 4 * edgeVerts[nums[j]][0];
518 v2 = xyz + 4 * edgeVerts[nums[j]][1];
519
520 l = 0.5 * sqrt( lengths[j] );
521
522 // we need to see which direction this edge
523 // is used to determine direction of projection
524 for ( k = 0 ; k < 5 ; k++ ) {
525 if ( tess.indexes[ indexes + k ] == i + edgeVerts[nums[j]][0]
526 && tess.indexes[ indexes + k + 1 ] == i + edgeVerts[nums[j]][1] ) {
527 break;
528 }
529 }
530
531 if ( k == 5 ) {
532 VectorMA( mid[j], l, minor, v1 );
533 VectorMA( mid[j], -l, minor, v2 );
534 } else {
535 VectorMA( mid[j], -l, minor, v1 );
536 VectorMA( mid[j], l, minor, v2 );
537 }
538 }
539 }
540}

543/*
544=====================
545RB_DeformTessGeometry
546
547=====================
548*/
549void RB_DeformTessGeometry( void ) {
550 int i;
551 deformStage_t *ds;
552
553 if(!ShaderRequiresCPUDeforms(tess.shader))
554 {
555 // we don't need the following CPU deforms
556 return;
557 }
558
559 for ( i = 0 ; i < tess.shader->numDeforms ; i++ ) {
560 ds = &tess.shader->deforms[ i ];
561
562 switch ( ds->deformation ) {
563 case DEFORM_NONE:
564 break;
565 case DEFORM_NORMALS:
566 RB_CalcDeformNormals( ds );
567 break;
568 case DEFORM_WAVE:
569 RB_CalcDeformVertexes( ds );
570 break;
571 case DEFORM_BULGE:
572 RB_CalcBulgeVertexes( ds );
573 break;
574 case DEFORM_MOVE:
575 RB_CalcMoveVertexes( ds );
576 break;
577 case DEFORM_PROJECTION_SHADOW:
578 RB_ProjectionShadowDeform();
579 break;
580 case DEFORM_AUTOSPRITE:
581 AutospriteDeform();
582 break;
583 case DEFORM_AUTOSPRITE2:
584 Autosprite2Deform();
585 break;
586 case DEFORM_TEXT0:
587 case DEFORM_TEXT1:
588 case DEFORM_TEXT2:
589 case DEFORM_TEXT3:
590 case DEFORM_TEXT4:
591 case DEFORM_TEXT5:
592 case DEFORM_TEXT6:
593 case DEFORM_TEXT7:
594 DeformText( backEnd.refdef.text[ds->deformation - DEFORM_TEXT0] );
595 break;
596 }
597 }
598}

609/*
610** RB_CalcWaveColorSingle
611*/
612float RB_CalcWaveColorSingle( const waveForm_t *wf )
613{
614 float glow;
615
616 if ( wf->func == GF_NOISE ) {
617 glow = wf->base + R_NoiseGet4f( 0, 0, 0, ( tess.shaderTime + wf->phase ) * wf->frequency ) * wf->amplitude;
618 } else {
619 glow = EvalWaveForm( wf ) * tr.identityLight;
620 }
621
622 if ( glow < 0 ) {
623 glow = 0;
624 }
625 else if ( glow > 1 ) {
626 glow = 1;
627 }
628
629 return glow;
630}

640/*
641** RB_CalcModulateColorsByFog
642*/
643void RB_CalcModulateColorsByFog( unsigned char *colors ) {
644 int i;
645 float texCoords[SHADER_MAX_VERTEXES][2] = {{0.0f}};
646
647 // calculate texcoords so we can derive density
648 // this is not wasted, because it would only have
649 // been previously called if the surface was opaque
650 RB_CalcFogTexCoords( texCoords[0] );
651
652 for ( i = 0; i < tess.numVertexes; i++, colors += 4 ) {
653 float f = 1.0 - R_FogFactor( texCoords[i][0], texCoords[i][1] );
654 colors[0] *= f;
655 colors[1] *= f;
656 colors[2] *= f;
657 }
658}

669/*
670========================
671RB_CalcFogTexCoords
672
673To do the clipped fog plane really correctly, we should use
674projected textures, but I don't trust the drivers and it
675doesn't fit our shader data.
676========================
677*/
678void RB_CalcFogTexCoords( float *st ) {
679 int i;
680 float *v;
681 float s, t;
682 float eyeT;
683 qboolean eyeOutside;
684 fog_t *fog;
685 vec3_t local;
686 vec4_t fogDistanceVector, fogDepthVector = {0, 0, 0, 0};
687
688 fog = tr.world->fogs + tess.fogNum;
689
690 // all fogging distance is based on world Z units
691 VectorSubtract( backEnd.or.origin, backEnd.viewParms.or.origin, local );
692 fogDistanceVector[0] = -backEnd.or.modelMatrix[2];
693 fogDistanceVector[1] = -backEnd.or.modelMatrix[6];
694 fogDistanceVector[2] = -backEnd.or.modelMatrix[10];
695 fogDistanceVector[3] = DotProduct( local, backEnd.viewParms.or.axis[0] );
696
697 // scale the fog vectors based on the fog's thickness
698 fogDistanceVector[0] *= fog->tcScale;
699 fogDistanceVector[1] *= fog->tcScale;
700 fogDistanceVector[2] *= fog->tcScale;
701 fogDistanceVector[3] *= fog->tcScale;
702
703 // rotate the gradient vector for this orientation
704 if ( fog->hasSurface ) {
705 fogDepthVector[0] = fog->surface[0] * backEnd.or.axis[0][0] +
706 fog->surface[1] * backEnd.or.axis[0][1] + fog->surface[2] * backEnd.or.axis[0][2];
707 fogDepthVector[1] = fog->surface[0] * backEnd.or.axis[1][0] +
708 fog->surface[1] * backEnd.or.axis[1][1] + fog->surface[2] * backEnd.or.axis[1][2];
709 fogDepthVector[2] = fog->surface[0] * backEnd.or.axis[2][0] +
710 fog->surface[1] * backEnd.or.axis[2][1] + fog->surface[2] * backEnd.or.axis[2][2];
711 fogDepthVector[3] = -fog->surface[3] + DotProduct( backEnd.or.origin, fog->surface );
712
713 eyeT = DotProduct( backEnd.or.viewOrigin, fogDepthVector ) + fogDepthVector[3];
714 } else {
715 eyeT = 1; // non-surface fog always has eye inside
716 }
717
718 // see if the viewpoint is outside
719 // this is needed for clipping distance even for constant fog
720
721 if ( eyeT < 0 ) {
722 eyeOutside = qtrue;
723 } else {
724 eyeOutside = qfalse;
725 }
726
727 fogDistanceVector[3] += 1.0/512;
728
729 // calculate density for each point
730 for (i = 0, v = tess.xyz[0] ; i < tess.numVertexes ; i++, v += 4) {
731 // calculate the length in fog
732 s = DotProduct( v, fogDistanceVector ) + fogDistanceVector[3];
733 t = DotProduct( v, fogDepthVector ) + fogDepthVector[3];
734
735 // partially clipped fogs use the T axis
736 if ( eyeOutside ) {
737 if ( t < 1.0 ) {
738 t = 1.0/32; // point is outside, so no fogging
739 } else {
740 t = 1.0/32 + 30.0/32 * t / ( t - eyeT ); // cut the distance at the fog plane
741 }
742 } else {
743 if ( t < 0 ) {
744 t = 1.0/32; // point is outside, so no fogging
745 } else {
746 t = 31.0/32;
747 }
748 }
749
750 st[0] = s;
751 st[1] = t;
752 st += 2;
753 }
754}

765/*
766** RB_CalcScaleTexMatrix
767*/
768void RB_CalcScaleTexMatrix( const float scale[2], float *matrix )
769{

775** RB_CalcScrollTexMatrix
776*/
777void RB_CalcScrollTexMatrix( const float scrollSpeed[2], float *matrix )
778{
779 double timeScale = tess.shaderTime;
780 double adjustedScrollS, adjustedScrollT;
781
782 adjustedScrollS = scrollSpeed[0] * timeScale;
783 adjustedScrollT = scrollSpeed[1] * timeScale;
784
785 // clamp so coordinates don't continuously get larger, causing problems
786 // with hardware limits
787 adjustedScrollS = adjustedScrollS - floor( adjustedScrollS );
788 adjustedScrollT = adjustedScrollT - floor( adjustedScrollT );
789
790 matrix[0] = 1.0f; matrix[2] = 0.0f; matrix[4] = adjustedScrollS;
791 matrix[1] = 0.0f; matrix[3] = 1.0f; matrix[5] = adjustedScrollT;

803/*
804** RB_CalcRotateTexMatrix
805*/
806void RB_CalcRotateTexMatrix( float degsPerSecond, float *matrix )
807{
808 double timeScale = tess.shaderTime;
809 double degs;
810 int64_t index;
811 float sinValue, cosValue;
812
813 degs = -degsPerSecond * timeScale;
814 index = degs * ( FUNCTABLE_SIZE / 360.0f );
815
816 sinValue = tr.sinTable[ index & FUNCTABLE_MASK ];
817 cosValue = tr.sinTable[ ( index + FUNCTABLE_SIZE / 4 ) & FUNCTABLE_MASK ];
818
819 matrix[0] = cosValue; matrix[2] = -sinValue; matrix[4] = 0.5 - 0.5 * cosValue + 0.5 * sinValue;
820 matrix[1] = sinValue; matrix[3] = cosValue; matrix[5] = 0.5 - 0.5 * sinValue - 0.5 * cosValue;
821}

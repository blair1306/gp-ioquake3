235/*
236** InitOpenGL
237**
238** This function is responsible for initializing a valid OpenGL subsystem. This
239** is done by calling GLimp_Init (which gives us a working OGL subsystem) then
240** setting variables, checking GL constants, and reporting the gfx system config
241** to the user.
242*/
243static void InitOpenGL( void )
244{
245 char renderer_buffer[1024];
246
247 //
248 // initialize OS specific portions of the renderer
249 //
250 // GLimp_Init directly or indirectly references the following cvars:
251 // - r_fullscreen
252 // - r_mode
253 // - r_(color|depth|stencil)bits
254 // - r_ignorehwgamma
255 // - r_gamma
256 //
257
258 if ( glConfig.vidWidth == 0 )
259 {
260 GLint temp;
261
262 GLimp_Init( qtrue );
263 GLimp_InitExtraExtensions();
264
265 strcpy( renderer_buffer, glConfig.renderer_string );
266 Q_strlwr( renderer_buffer );
267
268 // OpenGL driver constants
269 qglGetIntegerv( GL_MAX_TEXTURE_SIZE, &temp );
270 glConfig.maxTextureSize = temp;
271
272 // stubbed or broken drivers may have reported 0...
273 if ( glConfig.maxTextureSize <= 0 )
274 {
275 glConfig.maxTextureSize = 0;
276 }
277 }
278
279 // set default state
280 GL_SetDefaultState();
281}

283/*
284==================
285GL_CheckErrors
286==================
287*/
288void GL_CheckErrs( char *file, int line ) {
289 int err;
290 char s[64];
291
292 err = qglGetError();
293 if ( err == GL_NO_ERROR ) {
294 return;
295 }
296 if ( r_ignoreGLErrors->integer ) {
297 return;
298 }
299 switch( err ) {
300 case GL_INVALID_ENUM:
301 strcpy( s, "GL_INVALID_ENUM" );
302 break;
303 case GL_INVALID_VALUE:
304 strcpy( s, "GL_INVALID_VALUE" );
305 break;
306 case GL_INVALID_OPERATION:
307 strcpy( s, "GL_INVALID_OPERATION" );
308 break;
309 case GL_STACK_OVERFLOW:
310 strcpy( s, "GL_STACK_OVERFLOW" );
311 break;
312 case GL_STACK_UNDERFLOW:
313 strcpy( s, "GL_STACK_UNDERFLOW" );
314 break;
315 case GL_OUT_OF_MEMORY:
316 strcpy( s, "GL_OUT_OF_MEMORY" );
317 break;
318 default:
319 Com_sprintf( s, sizeof(s), "%i", err);
320 break;
321 }
322
323 ri.Error( ERR_FATAL, "GL_CheckErrors: %s in %s at line %d", s , file, line);
324}

355 vidmode_t *vm;
356 float pixelAspect;
357
358 if ( mode < -1 ) {
359 return qfalse;
360 }
361 if ( mode >= s_numVidModes ) {
362 return qfalse;
363 }
364
365 if ( mode == -1 ) {
366 *width = r_customwidth->integer;
367 *height = r_customheight->integer;

370 vm = &r_vidModes[mode];
371
372 *width = vm->width;
373 *height = vm->height;
374 pixelAspect = vm->pixelAspect;
375 }
376
377 *windowAspect = (float)*width / ( *height * pixelAspect );
378
379 return qtrue;
380}

417/*
418==================
419RB_ReadPixels
420
421Reads an image but takes care of alignment issues for reading RGB images.
422
423Reads a minimum offset for where the RGB data starts in the image from
424integer stored at pointer offset. When the function has returned the actual
425offset was written back to address offset. This address will always have an
426alignment of packAlign to ensure efficient copying.
427
428Stores the length of padding after a line of pixels to address padlen
429
430Return value must be freed with ri.Hunk_FreeTempMemory()
431==================
432*/
433
434byte *RB_ReadPixels(int x, int y, int width, int height, size_t *offset, int *padlen)
435{
436 byte *buffer, *bufstart;
437 int padwidth, linelen;
438 GLint packAlign;
439
440 qglGetIntegerv(GL_PACK_ALIGNMENT, &packAlign);
441
442 linelen = width * 3;
443 padwidth = PAD(linelen, packAlign);
444
445 // Allocate a few more bytes so that we can choose an alignment we like
446 buffer = ri.Hunk_AllocateTempMemory(padwidth * height + *offset + packAlign - 1);
447
448 bufstart = PADP((intptr_t) buffer + *offset, packAlign);
449
450 qglReadPixels(x, y, width, height, GL_RGB, GL_UNSIGNED_BYTE, bufstart);
451
452 *offset = bufstart - buffer;
453 *padlen = padwidth - linelen;
454
455 return buffer;
456}

458/*
459==================
460RB_TakeScreenshot
461==================
462*/
463void RB_TakeScreenshot(int x, int y, int width, int height, char *fileName)
464{
465 byte *allbuf, *buffer;
466 byte *srcptr, *destptr;
467 byte *endline, *endmem;
468 byte temp;
469
470 int linelen, padlen;
471 size_t offset = 18, memcount;
472
473 allbuf = RB_ReadPixels(x, y, width, height, &offset, &padlen);
474 buffer = allbuf + offset - 18;
475
476 Com_Memset (buffer, 0, 18);
477 buffer[2] = 2; // uncompressed type
478 buffer[12] = width & 255;
479 buffer[13] = width >> 8;
480 buffer[14] = height & 255;
481 buffer[15] = height >> 8;
482 buffer[16] = 24; // pixel size
483
484 // swap rgb to bgr and remove padding from line endings
485 linelen = width * 3;
486
487 srcptr = destptr = allbuf + offset;
488 endmem = srcptr + (linelen + padlen) * height;
489
490 while(srcptr < endmem)
491 {
492 endline = srcptr + linelen;
493
494 while(srcptr < endline)
495 {
496 temp = srcptr[0];
497 *destptr++ = srcptr[2];
498 *destptr++ = srcptr[1];
499 *destptr++ = temp;
500
501 srcptr += 3;
502 }
503
504 // Skip the pad
505 srcptr += padlen;
506 }
507
508 memcount = linelen * height;
509
510 // gamma correct
511 if(glConfig.deviceSupportsGamma)
512 R_GammaCorrect(allbuf + offset, memcount);
513
514 ri.FS_WriteFile(fileName, buffer, memcount + 18);
515
516 ri.Hunk_FreeTempMemory(allbuf);
517}

638/*
639====================
640R_LevelShot
641
642levelshots are specialized 128*128 thumbnails for
643the menu system, sampled down from full screen distorted images
644====================
645*/
646void R_LevelShot( void ) {
647 char checkname[MAX_OSPATH];
648 byte *buffer;
649 byte *source, *allsource;
650 byte *src, *dst;
651 size_t offset = 0;
652 int padlen;
653 int x, y;
654 int r, g, b;
655 float xScale, yScale;
656 int xx, yy;
657
658 Com_sprintf(checkname, sizeof(checkname), "levelshots/%s.tga", tr.world->baseName);
659
660 allsource = RB_ReadPixels(0, 0, glConfig.vidWidth, glConfig.vidHeight, &offset, &padlen);
661 source = allsource + offset;
662
663 buffer = ri.Hunk_AllocateTempMemory(128 * 128*3 + 18);
664 Com_Memset (buffer, 0, 18);
665 buffer[2] = 2; // uncompressed type
666 buffer[12] = 128;
667 buffer[14] = 128;
668 buffer[16] = 24; // pixel size
669
670 // resample from source
671 xScale = glConfig.vidWidth / 512.0f;
672 yScale = glConfig.vidHeight / 384.0f;
673 for ( y = 0 ; y < 128 ; y++ ) {
674 for ( x = 0 ; x < 128 ; x++ ) {
675 r = g = b = 0;
676 for ( yy = 0 ; yy < 3 ; yy++ ) {
677 for ( xx = 0 ; xx < 4 ; xx++ ) {
678 src = source + (3 * glConfig.vidWidth + padlen) * (int)((y*3 + yy) * yScale) +
679 3 * (int) ((x*4 + xx) * xScale);
680 r += src[0];
681 g += src[1];
682 b += src[2];
683 }
684 }
685 dst = buffer + 18 + 3 * ( y * 128 + x );
686 dst[0] = b / 12;
687 dst[1] = g / 12;
688 dst[2] = r / 12;
689 }
690 }
691
692 // gamma correct
693 if ( glConfig.deviceSupportsGamma ) {
694 R_GammaCorrect( buffer + 18, 128 * 128 * 3 );
695 }
696
697 ri.FS_WriteFile( checkname, buffer, 128 * 128*3 + 18 );
698
699 ri.Hunk_FreeTempMemory(buffer);
700 ri.Hunk_FreeTempMemory(allsource);
701
702 ri.Printf( PRINT_ALL, "Wrote %s\n", checkname );
703}

705/*
706==================
707R_ScreenShot_f
708
709screenshot
710screenshot [silent]
711screenshot [levelshot]
712screenshot [filename]
713
714Doesn't print the pacifier message if there is a second arg
715==================
716*/
717void R_ScreenShot_f (void) {
718 char checkname[MAX_OSPATH];
719 static int lastNumber = -1;
720 qboolean silent;
721
722 if ( !strcmp( ri.Cmd_Argv(1), "levelshot" ) ) {
723 R_LevelShot();
724 return;
725 }
726
727 if ( !strcmp( ri.Cmd_Argv(1), "silent" ) ) {
728 silent = qtrue;
729 } else {
730 silent = qfalse;
731 }
732
733 if ( ri.Cmd_Argc() == 2 && !silent ) {
734 // explicit filename
735 Com_sprintf( checkname, MAX_OSPATH, "screenshots/%s.tga", ri.Cmd_Argv( 1 ) );
736 } else {
737 // scan for a free filename
738
739 // if we have saved a previous screenshot, don't scan
740 // again, because recording demo avis can involve
741 // thousands of shots
742 if ( lastNumber == -1 ) {
743 lastNumber = 0;
744 }
745 // scan for a free number
746 for ( ; lastNumber <= 9999 ; lastNumber++ ) {
747 R_ScreenshotFilename( lastNumber, checkname );
748
749 if (!ri.FS_FileExists( checkname ))
750 {
751 break; // file doesn't exist
752 }
753 }
754
755 if ( lastNumber >= 9999 ) {
756 ri.Printf (PRINT_ALL, "ScreenShot: Couldn't create a file\n");
757 return;
758 }
759
760 lastNumber++;
761 }
762
763 R_TakeScreenshot( 0, 0, glConfig.vidWidth, glConfig.vidHeight, checkname, qfalse );
764
765 if ( !silent ) {
766 ri.Printf (PRINT_ALL, "Wrote %s\n", checkname);
767 }
768}

778 }
779

782 } else {
783 silent = qfalse;
784 }
785
786 if ( ri.Cmd_Argc() == 2 && !silent ) {
787 // explicit filename
788 Com_sprintf( checkname, MAX_OSPATH, "screenshots/%s.jpg", ri.Cmd_Argv( 1 ) );
789 } else {
790 // scan for a free filename
791
792 // if we have saved a previous screenshot, don't scan
793 // again, because recording demo avis can involve
794 // thousands of shots
795 if ( lastNumber == -1 ) {
796 lastNumber = 0;
797 }
798 // scan for a free number
799 for ( ; lastNumber <= 9999 ; lastNumber++ ) {
800 R_ScreenshotFilenameJPEG( lastNumber, checkname );
801
802 if (!ri.FS_FileExists( checkname ))
803 {
804 break; // file doesn't exist
805 }
806 }
807
808 if ( lastNumber == 10000 ) {
809 ri.Printf (PRINT_ALL, "ScreenShot: Couldn't create a file\n");
810 return;
811 }
812
813 lastNumber++;
814 }
815
816 R_TakeScreenshot( 0, 0, glConfig.vidWidth, glConfig.vidHeight, checkname, qtrue );
817
818 if ( !silent ) {
819 ri.Printf (PRINT_ALL, "Wrote %s\n", checkname);
820 }
821}

854/*
855==================
856RB_TakeVideoFrameCmd
857==================
858*/
859const void *RB_TakeVideoFrameCmd( const void *data )
860{
861 const videoFrameCommand_t *cmd;
862 byte *cBuf;
863 size_t memcount, linelen;
864 int padwidth, avipadwidth, padlen, avipadlen;
865 GLint packAlign;
866
867 // finish any 2D drawing if needed
868 if(tess.numIndexes)
869 RB_EndSurface();
870
871 cmd = (const videoFrameCommand_t *)data;
872
873 qglGetIntegerv(GL_PACK_ALIGNMENT, &packAlign);
874
875 linelen = cmd->width * 3;
876
877 // Alignment stuff for glReadPixels
878 padwidth = PAD(linelen, packAlign);
879 padlen = padwidth - linelen;
880 // AVI line padding
881 avipadwidth = PAD(linelen, AVI_LINE_PADDING);
882 avipadlen = avipadwidth - linelen;
883
884 cBuf = PADP(cmd->captureBuffer, packAlign);
885
886 qglReadPixels(0, 0, cmd->width, cmd->height, GL_RGB,
887 GL_UNSIGNED_BYTE, cBuf);
888
889 memcount = padwidth * cmd->height;
890
891 // gamma correct
892 if(glConfig.deviceSupportsGamma)
893 R_GammaCorrect(cBuf, memcount);
894
895 if(cmd->motionJpeg)
896 {
897 memcount = RE_SaveJPGToBuffer(cmd->encodeBuffer, linelen * cmd->height,
898 r_aviMotionJpegQuality->integer,
899 cmd->width, cmd->height, cBuf, padlen);
900 ri.CL_WriteAVIVideoFrame(cmd->encodeBuffer, memcount);
901 }
902 else
903 {
904 byte *lineend, *memend;
905 byte *srcptr, *destptr;
906
907 srcptr = cBuf;
908 destptr = cmd->encodeBuffer;
909 memend = srcptr + memcount;
910
911 // swap R and B and remove line paddings
912 while(srcptr < memend)
913 {
914 lineend = srcptr + linelen;
915 while(srcptr < lineend)
916 {
917 *destptr++ = srcptr[2];
918 *destptr++ = srcptr[1];
919 *destptr++ = srcptr[0];
920 srcptr += 3;
921 }
922
923 Com_Memset(destptr, '\0', avipadlen);
924 destptr += avipadlen;
925
926 srcptr += padlen;
927 }
928
929 ri.CL_WriteAVIVideoFrame(cmd->encodeBuffer, avipadwidth * cmd->height);
930 }
931
932 return (const void *)(cmd + 1);
933}

1395void R_Init( void ) {
1396 int err;
1397 int i;
1398 byte *ptr;
1399
1400 ri.Printf( PRINT_ALL, "----- R_Init -----\n" );
1401
1402 // clear all our internal state
1403 Com_Memset( &tr, 0, sizeof( tr ) );
1404 Com_Memset( &backEnd, 0, sizeof( backEnd ) );
1405 Com_Memset( &tess, 0, sizeof( tess ) );
1406
1407 if(sizeof(glconfig_t) != 11332)
1408 ri.Error( ERR_FATAL, "Mod ABI incompatible: sizeof(glconfig_t) == %u != 11332", (unsigned int) sizeof(glconfig_t));
1409
1410// Swap_Init();
1411
1412 if ( (intptr_t)tess.xyz & 15 ) {
1413 ri.Printf( PRINT_WARNING, "tess.xyz not 16 byte aligned\n" );
1414 }
1415 //Com_Memset( tess.constantColor255, 255, sizeof( tess.constantColor255 ) );
1416
1417 //
1418 // init function tables
1419 //
1420 for ( i = 0; i < FUNCTABLE_SIZE; i++ )
1421 {
1422 tr.sinTable[i] = sin( DEG2RAD( i * 360.0f / ( ( float ) ( FUNCTABLE_SIZE - 1 ) ) ) );
1423 tr.squareTable[i] = ( i < FUNCTABLE_SIZE/2 ) ? 1.0f : -1.0f;
1424 tr.sawToothTable[i] = (float)i / FUNCTABLE_SIZE;
1425 tr.inverseSawToothTable[i] = 1.0f - tr.sawToothTable[i];
1426
1427 if ( i < FUNCTABLE_SIZE / 2 )
1428 {
1429 if ( i < FUNCTABLE_SIZE / 4 )
1430 {
1431 tr.triangleTable[i] = ( float ) i / ( FUNCTABLE_SIZE / 4 );
1432 }
1433 else
1434 {
1435 tr.triangleTable[i] = 1.0f - tr.triangleTable[i-FUNCTABLE_SIZE / 4];
1436 }
1437 }
1438 else
1439 {
1440 tr.triangleTable[i] = -tr.triangleTable[i-FUNCTABLE_SIZE/2];
1441 }
1442 }
1443
1444 R_InitFogTable();
1445
1446 R_NoiseInit();
1447
1448 R_Register();
1449
1450 max_polys = r_maxpolys->integer;
1451 if (max_polys < MAX_POLYS)
1452 max_polys = MAX_POLYS;
1453
1454 max_polyverts = r_maxpolyverts->integer;
1455 if (max_polyverts < MAX_POLYVERTS)
1456 max_polyverts = MAX_POLYVERTS;
1457
1458 ptr = ri.Hunk_Alloc( sizeof( *backEndData ) + sizeof(srfPoly_t) * max_polys + sizeof(polyVert_t) * max_polyverts, h_low);
1459 backEndData = (backEndData_t *) ptr;
1460 backEndData->polys = (srfPoly_t *) ((char *) ptr + sizeof( *backEndData ));
1461 backEndData->polyVerts = (polyVert_t *) ((char *) ptr + sizeof( *backEndData ) + sizeof(srfPoly_t) * max_polys);
1462 R_InitNextFrame();
1463
1464 InitOpenGL();

1469 FBO_Init();
1470
1471 GLSL_InitGPUShaders();

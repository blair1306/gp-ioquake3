66static long generateHashValue( const char *fname ) {
67 int i;
68 long hash;
69 char letter;
70
71 hash = 0;
72 i = 0;
73 while (fname[i] != '\0') {
74 letter = tolower(fname[i]);
75 if (letter =='.') break; // don't include extension
76 if (letter =='\\') letter = '/'; // damn path names
77 hash+=(long)(letter)*(i+119);
78 i++;
79 }
80 hash &= (FILE_HASH_SIZE-1);
81 return hash;
82}

84/*
85===============
86GL_TextureMode
87===============
88*/
89void GL_TextureMode( const char *string ) {
90 int i;
91 image_t *glt;
92
93 for ( i=0 ; i< 6 ; i++ ) {
94 if ( !Q_stricmp( modes[i].name, string ) ) {
95 break;
96 }
97 }
98
99 // hack to prevent trilinear from being set on voodoo,
100 // because their driver freaks...
101 if ( i == 5 && glConfig.hardwareType == GLHW_3DFX_2D3D ) {
102 ri.Printf( PRINT_ALL, "Refusing to set trilinear on a voodoo.\n" );
103 i = 3;
104 }
105
106
107 if ( i == 6 ) {
108 ri.Printf (PRINT_ALL, "bad filter name\n");
109 return;
110 }
111
112 gl_filter_min = modes[i].minimize;
113 gl_filter_max = modes[i].maximize;
114
115 // change all the existing mipmap texture objects
116 for ( i = 0 ; i < tr.numImages ; i++ ) {
117 glt = tr.images[ i ];
118 if ( glt->flags & IMGFLAG_MIPMAP && !(glt->flags & IMGFLAG_CUBEMAP)) {
119 qglTextureParameterfEXT(glt->texnum, GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, gl_filter_min);
120 qglTextureParameterfEXT(glt->texnum, GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, gl_filter_max);
121 }
122 }
123}

321/*
322================
323ResampleTexture
324
325Used to resample images in a more general than quartering fashion.
326
327This will only be filtered properly if the resampled size
328is greater than half the original size.
329
330If a larger shrinking is needed, use the mipmap function
331before or after.
332================
333*/
334static void ResampleTexture( byte *in, int inwidth, int inheight, byte *out,
335 int outwidth, int outheight ) {
336 int i, j;
337 byte *inrow, *inrow2;
338 int frac, fracstep;
339 int p1[2048], p2[2048];
340 byte *pix1, *pix2, *pix3, *pix4;
341
342 if (outwidth>2048)
343 ri.Error(ERR_DROP, "ResampleTexture: max width");
344
345 fracstep = inwidth*0x10000/outwidth;
346
347 frac = fracstep>>2;
348 for ( i=0 ; i<outwidth ; i++ ) {
349 p1[i] = 4*(frac>>16);
350 frac += fracstep;
351 }
352 frac = 3*(fracstep>>2);
353 for ( i=0 ; i<outwidth ; i++ ) {
354 p2[i] = 4*(frac>>16);
355 frac += fracstep;
356 }
357
358 for (i=0 ; i<outheight ; i++) {
359 inrow = in + 4*inwidth*(int)((i+0.25)*inheight/outheight);
360 inrow2 = in + 4*inwidth*(int)((i+0.75)*inheight/outheight);
361 for (j=0 ; j<outwidth ; j++) {
362 pix1 = inrow + p1[j];
363 pix2 = inrow + p2[j];
364 pix3 = inrow2 + p1[j];
365 pix4 = inrow2 + p2[j];
366 *out++ = (pix1[0] + pix2[0] + pix3[0] + pix4[0])>>2;
367 *out++ = (pix1[1] + pix2[1] + pix3[1] + pix4[1])>>2;
368 *out++ = (pix1[2] + pix2[2] + pix3[2] + pix4[2])>>2;
369 *out++ = (pix1[3] + pix2[3] + pix3[3] + pix4[3])>>2;
370 }
371 }
372}

374static void RGBAtoYCoCgA(const byte *in, byte *out, int width, int height)
375{
376 int x, y;
377
378 for (y = 0; y < height; y++)
379 {
380 const byte *inbyte = in + y * width * 4;
381 byte *outbyte = out + y * width * 4;
382
383 for (x = 0; x < width; x++)
384 {
385 byte r, g, b, a, rb2;
386
387 r = *inbyte++;
388 g = *inbyte++;
389 b = *inbyte++;
390 a = *inbyte++;
391 rb2 = (r + b) >> 1;
392
393 *outbyte++ = (g + rb2) >> 1; // Y = R/4 + G/2 + B/4
394 *outbyte++ = (r - b + 256) >> 1; // Co = R/2 - B/2
395 *outbyte++ = (g - rb2 + 256) >> 1; // Cg = -R/4 + G/2 - B/4
396 *outbyte++ = a;
397 }
398 }
399}

401static void YCoCgAtoRGBA(const byte *in, byte *out, int width, int height)
402{
403 int x, y;
404
405 for (y = 0; y < height; y++)
406 {
407 const byte *inbyte = in + y * width * 4;
408 byte *outbyte = out + y * width * 4;
409
410 for (x = 0; x < width; x++)
411 {
412 byte _Y, Co, Cg, a;
413
414 _Y = *inbyte++;
415 Co = *inbyte++;
416 Cg = *inbyte++;
417 a = *inbyte++;
418
419 *outbyte++ = CLAMP(_Y + Co - Cg, 0, 255); // R = Y + Co - Cg
420 *outbyte++ = CLAMP(_Y + Cg - 128, 0, 255); // G = Y + Cg
421 *outbyte++ = CLAMP(_Y - Co - Cg + 256, 0, 255); // B = Y - Co - Cg
422 *outbyte++ = a;
423 }
424 }

428// uses a sobel filter to change a texture to a normal map
429static void RGBAtoNormal(const byte *in, byte *out, int width, int height, qboolean clampToEdge)
430{
431 int x, y, max;
432
433 // convert to heightmap, storing in alpha
434 // same as converting to Y in YCoCg
435 max = 1;
436 for (y = 0; y < height; y++)
437 {
438 const byte *inbyte = in + y * width * 4;
439 byte *outbyte = out + y * width * 4 + 3;
440
441 for (x = 0; x < width; x++)
442 {
443 byte result = (inbyte[0] >> 2) + (inbyte[1] >> 1) + (inbyte[2] >> 2);
444 result = result * result / 255; // Make linear
445 *outbyte = result;
446 max = MAX(max, *outbyte);
447 outbyte += 4;
448 inbyte += 4;
449 }
450 }
451
452 // level out heights
453 if (max < 255)
454 {
455 for (y = 0; y < height; y++)
456 {
457 byte *outbyte = out + y * width * 4 + 3;
458
459 for (x = 0; x < width; x++)
460 {
461 *outbyte = *outbyte + (255 - max);
462 outbyte += 4;
463 }
464 }
465 }
466
467
468 // now run sobel filter over height values to generate X and Y
469 // then normalize
470 for (y = 0; y < height; y++)
471 {
472 byte *outbyte = out + y * width * 4;
473
474 for (x = 0; x < width; x++)
475 {
476 // 0 1 2
477 // 3 4 5
478 // 6 7 8
479
480 byte s[9];
481 int x2, y2, i;
482 vec3_t normal;
483
484 i = 0;
485 for (y2 = -1; y2 <= 1; y2++)
486 {
487 int src_y = y + y2;
488
489 if (clampToEdge)
490 {
491 src_y = CLAMP(src_y, 0, height - 1);
492 }
493 else
494 {
495 src_y = (src_y + height) % height;
496 }
497
498
499 for (x2 = -1; x2 <= 1; x2++)
500 {
501 int src_x = x + x2;
502
503 if (clampToEdge)
504 {
505 src_x = CLAMP(src_x, 0, width - 1);
506 }
507 else
508 {
509 src_x = (src_x + width) % width;
510 }
511
512 s[i++] = *(out + (src_y * width + src_x) * 4 + 3);
513 }
514 }
515
516 normal[0] = s[0] - s[2]
517 + 2 * s[3] - 2 * s[5]
518 + s[6] - s[8];
519
520 normal[1] = s[0] + 2 * s[1] + s[2]
521
522 - s[6] - 2 * s[7] - s[8];
523
524 normal[2] = s[4] * 4;
525
526 if (!VectorNormalize2(normal, normal))
527 {
528 VectorSet(normal, 0, 0, 1);
529 }
530
531 *outbyte++ = FloatToOffsetByte(normal[0]);
532 *outbyte++ = FloatToOffsetByte(normal[1]);
533 *outbyte++ = FloatToOffsetByte(normal[2]);
534 outbyte++;
535 }
536 }
537}

541// based on Fast Curve Based Interpolation
542// from Fast Artifacts-Free Image Interpolation (http://www.andreagiachetti.it/icbi/)
543// assumes data has a 2 pixel thick border of clamped or wrapped data
544// expects data to be a grid with even (0, 0), (2, 0), (0, 2), (2, 2) etc pixels filled
545// only performs FCBI on specified component
546static void DoFCBI(byte *in, byte *out, int width, int height, int component)
547{
548 int x, y;
549 byte *outbyte, *inbyte;
550
551 // copy in to out
552 for (y = 2; y < height - 2; y += 2)
553 {
554 inbyte = in + (y * width + 2) * 4 + component;
555 outbyte = out + (y * width + 2) * 4 + component;
556
557 for (x = 2; x < width - 2; x += 2)
558 {
559 *outbyte = *inbyte;
560 outbyte += 8;
561 inbyte += 8;
562 }
563 }
564
565 for (y = 3; y < height - 3; y += 2)
566 {
567 // diagonals
568 //
569 // NWp - northwest interpolated pixel
570 // NEp - northeast interpolated pixel
571 // NWd - northwest first derivative
572 // NEd - northeast first derivative
573 // NWdd - northwest second derivative
574 // NEdd - northeast second derivative
575 //
576 // Uses these samples:
577 //
578 // 0
579 // - - a - b - -
580 // - - - - - - -
581 // c - d - e - f
582 // 0 - - - - - - -
583 // g - h - i - j
584 // - - - - - - -
585 // - - k - l - -
586 //
587 // x+2 uses these samples:
588 //
589 // 0
590 // - - - - a - b - -
591 // - - - - - - - - -
592 // - - c - d - e - f
593 // 0 - - - - - - - - -
594 // - - g - h - i - j
595 // - - - - - - - - -
596 // - - - - k - l - -
597 //
598 // so we can reuse 8 of them on next iteration
599 //
600 // a=b, c=d, d=e, e=f, g=h, h=i, i=j, k=l
601 //
602 // only b, f, j, and l need to be sampled on next iteration
603
604 byte sa, sb, sc, sd, se, sf, sg, sh, si, sj, sk, sl;
605 byte *line1, *line2, *line3, *line4;
606
607 x = 3;
608
609 // optimization one
610 // SAMPLE2(sa, x-1, y-3);
611 //SAMPLE2(sc, x-3, y-1); SAMPLE2(sd, x-1, y-1); SAMPLE2(se, x+1, y-1);
612 //SAMPLE2(sg, x-3, y+1); SAMPLE2(sh, x-1, y+1); SAMPLE2(si, x+1, y+1);
613 // SAMPLE2(sk, x-1, y+3);
614
615 // optimization two
616 line1 = in + ((y - 3) * width + (x - 1)) * 4 + component;
617 line2 = in + ((y - 1) * width + (x - 3)) * 4 + component;
618 line3 = in + ((y + 1) * width + (x - 3)) * 4 + component;
619 line4 = in + ((y + 3) * width + (x - 1)) * 4 + component;
620
621 // COPYSAMPLE(sa, line1); line1 += 8;
622 //COPYSAMPLE(sc, line2); line2 += 8; COPYSAMPLE(sd, line2); line2 += 8; COPYSAMPLE(se, line2); line2 += 8;
623 //COPYSAMPLE(sg, line3); line3 += 8; COPYSAMPLE(sh, line3); line3 += 8; COPYSAMPLE(si, line3); line3 += 8;
624 // COPYSAMPLE(sk, line4); line4 += 8;
625
626 sa = *line1; line1 += 8;
627 sc = *line2; line2 += 8; sd = *line2; line2 += 8; se = *line2; line2 += 8;
628 sg = *line3; line3 += 8; sh = *line3; line3 += 8; si = *line3; line3 += 8;
629 sk = *line4; line4 += 8;
630
631 outbyte = out + (y * width + x) * 4 + component;
632
633 for ( ; x < width - 3; x += 2)
634 {
635 int NWd, NEd, NWp, NEp;
636
637 // original
638 // SAMPLE2(sa, x-1, y-3); SAMPLE2(sb, x+1, y-3);
639 //SAMPLE2(sc, x-3, y-1); SAMPLE2(sd, x-1, y-1); SAMPLE2(se, x+1, y-1); SAMPLE2(sf, x+3, y-1);
640 //SAMPLE2(sg, x-3, y+1); SAMPLE2(sh, x-1, y+1); SAMPLE2(si, x+1, y+1); SAMPLE2(sj, x+3, y+1);
641 // SAMPLE2(sk, x-1, y+3); SAMPLE2(sl, x+1, y+3);
642
643 // optimization one
644 //SAMPLE2(sb, x+1, y-3);
645 //SAMPLE2(sf, x+3, y-1);
646 //SAMPLE2(sj, x+3, y+1);
647 //SAMPLE2(sl, x+1, y+3);
648
649 // optimization two
650 //COPYSAMPLE(sb, line1); line1 += 8;
651 //COPYSAMPLE(sf, line2); line2 += 8;
652 //COPYSAMPLE(sj, line3); line3 += 8;
653 //COPYSAMPLE(sl, line4); line4 += 8;
654
655 sb = *line1; line1 += 8;
656 sf = *line2; line2 += 8;
657 sj = *line3; line3 += 8;
658 sl = *line4; line4 += 8;
659
660 NWp = sd + si;
661 NEp = se + sh;
662 NWd = abs(sd - si);
663 NEd = abs(se - sh);
664
665 if (NWd > 100 || NEd > 100 || abs(NWp-NEp) > 200)
666 {
667 if (NWd < NEd)
668 *outbyte = NWp >> 1;
669 else
670 *outbyte = NEp >> 1;
671 }
672 else
673 {
674 int NWdd, NEdd;
675
676 //NEdd = abs(sg + sd + sb - 3 * (se + sh) + sk + si + sf);
677 //NWdd = abs(sa + se + sj - 3 * (sd + si) + sc + sh + sl);
678 NEdd = abs(sg + sb - 3 * NEp + sk + sf + NWp);
679 NWdd = abs(sa + sj - 3 * NWp + sc + sl + NEp);
680
681 if (NWdd > NEdd)
682 *outbyte = NWp >> 1;
683 else
684 *outbyte = NEp >> 1;
685 }
686
687 outbyte += 8;
688
689 // COPYSAMPLE(sa, sb);
690 //COPYSAMPLE(sc, sd); COPYSAMPLE(sd, se); COPYSAMPLE(se, sf);
691 //COPYSAMPLE(sg, sh); COPYSAMPLE(sh, si); COPYSAMPLE(si, sj);
692 // COPYSAMPLE(sk, sl);
693
694 sa = sb;
695 sc = sd; sd = se; se = sf;
696 sg = sh; sh = si; si = sj;
697 sk = sl;
698 }
699 }
700
701 // hack: copy out to in again
702 for (y = 3; y < height - 3; y += 2)
703 {
704 inbyte = out + (y * width + 3) * 4 + component;
705 outbyte = in + (y * width + 3) * 4 + component;
706
707 for (x = 3; x < width - 3; x += 2)
708 {
709 *outbyte = *inbyte;
710 outbyte += 8;
711 inbyte += 8;
712 }
713 }
714
715 for (y = 2; y < height - 3; y++)
716 {
717 // horizontal & vertical
718 //
719 // hp - horizontally interpolated pixel
720 // vp - vertically interpolated pixel
721 // hd - horizontal first derivative
722 // vd - vertical first derivative
723 // hdd - horizontal second derivative
724 // vdd - vertical second derivative
725 // Uses these samples:
726 //
727 // 0
728 // - a - b -
729 // c - d - e
730 // 0 - f - g -
731 // h - i - j
732 // - k - l -
733 //
734 // x+2 uses these samples:
735 //
736 // 0
737 // - - - a - b -
738 // - - c - d - e
739 // 0 - - - f - g -

1802static void CompressMonoBlock(byte outdata[8], const byte indata[16])
1803{
1804 int hi, lo, diff, bias, outbyte, shift, i;
1805 byte *p = outdata;
1806
1807 hi = lo = indata[0];
1808 for (i = 1; i < 16; i++)
1809 {
1810 hi = MAX(indata[i], hi);
1811 lo = MIN(indata[i], lo);
1812 }
1813
1814 *p++ = hi;
1815 *p++ = lo;
1816
1817 diff = hi - lo;
1818
1819 if (diff == 0)
1820 {
1821 outbyte = (hi == 255) ? 255 : 0;
1822
1823 for (i = 0; i < 6; i++)
1824 *p++ = outbyte;
1825
1826 return;
1827 }
1828
1829 bias = diff / 2 - lo * 7;
1830 outbyte = shift = 0;
1831 for (i = 0; i < 16; i++)
1832 {
1833 const byte fixIndex[8] = { 1, 7, 6, 5, 4, 3, 2, 0 };
1834 byte index = fixIndex[(indata[i] * 7 + bias) / diff];
1835
1836 outbyte |= index << shift;
1837 shift += 3;
1838 if (shift >= 8)
1839 {
1840 *p++ = outbyte & 0xff;
1841 shift -= 8;
1842 outbyte >>= 8;
1843 }
1844 }
1845}

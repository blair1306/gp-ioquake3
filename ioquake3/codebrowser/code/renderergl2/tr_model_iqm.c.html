35static qboolean IQM_CheckRange( iqmHeader_t *header, int offset,
36 int count,int size ) {
37 // return true if the range specified by offset, count and size
38 // doesn't fit into the file
39 return ( count <= 0 ||
40 offset < 0 ||
41 offset > header->filesize ||
42 offset + count * size < 0 ||
43 offset + count * size > header->filesize );
44}
45// "multiply" 3x4 matrices, these are assumed to be the top 3 rows
46// of a 4x4 matrix with the last row = (0 0 0 1)
47static void Matrix34Multiply( float *a, float *b, float *out ) {
48 out[ 0] = a[0] * b[0] + a[1] * b[4] + a[ 2] * b[ 8];
49 out[ 1] = a[0] * b[1] + a[1] * b[5] + a[ 2] * b[ 9];
50 out[ 2] = a[0] * b[2] + a[1] * b[6] + a[ 2] * b[10];
51 out[ 3] = a[0] * b[3] + a[1] * b[7] + a[ 2] * b[11] + a[ 3];
52 out[ 4] = a[4] * b[0] + a[5] * b[4] + a[ 6] * b[ 8];
53 out[ 5] = a[4] * b[1] + a[5] * b[5] + a[ 6] * b[ 9];
54 out[ 6] = a[4] * b[2] + a[5] * b[6] + a[ 6] * b[10];
55 out[ 7] = a[4] * b[3] + a[5] * b[7] + a[ 6] * b[11] + a[ 7];
56 out[ 8] = a[8] * b[0] + a[9] * b[4] + a[10] * b[ 8];
57 out[ 9] = a[8] * b[1] + a[9] * b[5] + a[10] * b[ 9];
58 out[10] = a[8] * b[2] + a[9] * b[6] + a[10] * b[10];
59 out[11] = a[8] * b[3] + a[9] * b[7] + a[10] * b[11] + a[11];
60}
61static void Matrix34Multiply_OnlySetOrigin( float *a, float *b, float *out ) {
62 out[ 3] = a[0] * b[3] + a[1] * b[7] + a[ 2] * b[11] + a[ 3];
63 out[ 7] = a[4] * b[3] + a[5] * b[7] + a[ 6] * b[11] + a[ 7];
64 out[11] = a[8] * b[3] + a[9] * b[7] + a[10] * b[11] + a[11];
65}
66static void InterpolateMatrix( float *a, float *b, float lerp, float *mat ) {
67 float unLerp = 1.0f - lerp;
68
69 mat[ 0] = a[ 0] * unLerp + b[ 0] * lerp;
70 mat[ 1] = a[ 1] * unLerp + b[ 1] * lerp;
71 mat[ 2] = a[ 2] * unLerp + b[ 2] * lerp;
72 mat[ 3] = a[ 3] * unLerp + b[ 3] * lerp;
73 mat[ 4] = a[ 4] * unLerp + b[ 4] * lerp;
74 mat[ 5] = a[ 5] * unLerp + b[ 5] * lerp;
75 mat[ 6] = a[ 6] * unLerp + b[ 6] * lerp;
76 mat[ 7] = a[ 7] * unLerp + b[ 7] * lerp;
77 mat[ 8] = a[ 8] * unLerp + b[ 8] * lerp;
78 mat[ 9] = a[ 9] * unLerp + b[ 9] * lerp;
79 mat[10] = a[10] * unLerp + b[10] * lerp;
80 mat[11] = a[11] * unLerp + b[11] * lerp;
81}
82static void JointToMatrix( vec4_t rot, vec3_t scale, vec3_t trans,
83 float *mat ) {
84 float xx = 2.0f * rot[0] * rot[0];
85 float yy = 2.0f * rot[1] * rot[1];
86 float zz = 2.0f * rot[2] * rot[2];
87 float xy = 2.0f * rot[0] * rot[1];
88 float xz = 2.0f * rot[0] * rot[2];
89 float yz = 2.0f * rot[1] * rot[2];
90 float wx = 2.0f * rot[3] * rot[0];
91 float wy = 2.0f * rot[3] * rot[1];
92 float wz = 2.0f * rot[3] * rot[2];
93
94 mat[ 0] = scale[0] * (1.0f - (yy + zz));
95 mat[ 1] = scale[0] * (xy - wz);
96 mat[ 2] = scale[0] * (xz + wy);
97 mat[ 3] = trans[0];
98 mat[ 4] = scale[1] * (xy + wz);
99 mat[ 5] = scale[1] * (1.0f - (xx + zz));
100 mat[ 6] = scale[1] * (yz - wx);
101 mat[ 7] = trans[1];
102 mat[ 8] = scale[2] * (xz - wy);
103 mat[ 9] = scale[2] * (yz + wx);
104 mat[10] = scale[2] * (1.0f - (xx + yy));
105 mat[11] = trans[2];
106}
107static void Matrix34Invert( float *inMat, float *outMat )
108{
109 vec3_t trans;
110 float invSqrLen, *v;
111
112 outMat[ 0] = inMat[ 0]; outMat[ 1] = inMat[ 4]; outMat[ 2] = inMat[ 8];
113 outMat[ 4] = inMat[ 1]; outMat[ 5] = inMat[ 5]; outMat[ 6] = inMat[ 9];
114 outMat[ 8] = inMat[ 2]; outMat[ 9] = inMat[ 6]; outMat[10] = inMat[10];
115
116 v = outMat + 0; invSqrLen = 1.0f / DotProduct(v, v); VectorScale(v, invSqrLen, v);
117 v = outMat + 4; invSqrLen = 1.0f / DotProduct(v, v); VectorScale(v, invSqrLen, v);
118 v = outMat + 8; invSqrLen = 1.0f / DotProduct(v, v); VectorScale(v, invSqrLen, v);
119
120 trans[0] = inMat[ 3];
121 trans[1] = inMat[ 7];
122 trans[2] = inMat[11];
123
124 outMat[ 3] = -DotProduct(outMat + 0, trans);
125 outMat[ 7] = -DotProduct(outMat + 4, trans);
126 outMat[11] = -DotProduct(outMat + 8, trans);
127}

129/*
130=================
131R_LoadIQM
132
133Load an IQM model and compute the joint matrices for every frame.
134=================
135*/
136qboolean R_LoadIQM( model_t *mod, void *buffer, int filesize, const char *mod_name ) {
137 iqmHeader_t *header;
138 iqmVertexArray_t *vertexarray;
139 iqmTriangle_t *triangle;
140 iqmMesh_t *mesh;
141 iqmJoint_t *joint;
142 iqmPose_t *pose;
143 iqmBounds_t *bounds;
144 unsigned short *framedata;
145 char *str;
146 int i, j;
147 float jointInvMats[IQM_MAX_JOINTS * 12] = {0.0f};
148 float *mat, *matInv;
149 size_t size, joint_names;
150 iqmData_t *iqmData;
151 srfIQModel_t *surface;
152 char meshName[MAX_QPATH];
153 byte blendIndexesType, blendWeightsType;
154
155 if( filesize < sizeof(iqmHeader_t) ) {
156 return qfalse;
157 }
158
159 header = (iqmHeader_t *)buffer;
160 if( Q_strncmp( header->magic, IQM_MAGIC, sizeof(header->magic) ) ) {
161 return qfalse;
162 }
163
164 LL( header->version );
165 if( header->version != IQM_VERSION ) {
166 ri.Printf(PRINT_WARNING, "R_LoadIQM: %s is a unsupported IQM version (%d), only version %d is supported.\n",
167 mod_name, header->version, IQM_VERSION);
168 return qfalse;
169 }
170
171 LL( header->filesize );
172 if( header->filesize > filesize || header->filesize > 16<<20 ) {
173 return qfalse;
174 }
175
176 LL( header->flags );
177 LL( header->num_text );
178 LL( header->ofs_text );
179 LL( header->num_meshes );
180 LL( header->ofs_meshes );
181 LL( header->num_vertexarrays );
182 LL( header->num_vertexes );
183 LL( header->ofs_vertexarrays );
184 LL( header->num_triangles );
185 LL( header->ofs_triangles );
186 LL( header->ofs_adjacency );
187 LL( header->num_joints );
188 LL( header->ofs_joints );
189 LL( header->num_poses );
190 LL( header->ofs_poses );
191 LL( header->num_anims );
192 LL( header->ofs_anims );
193 LL( header->num_frames );
194 LL( header->num_framechannels );
195 LL( header->ofs_frames );
196 LL( header->ofs_bounds );
197 LL( header->num_comment );
198 LL( header->ofs_comment );
199 LL( header->num_extensions );
200 LL( header->ofs_extensions );
201
202 // check ioq3 joint limit
203 if ( header->num_joints > IQM_MAX_JOINTS ) {
204 ri.Printf(PRINT_WARNING, "R_LoadIQM: %s has more than %d joints (%d).\n",
205 mod_name, IQM_MAX_JOINTS, header->num_joints);
206 return qfalse;
207 }
208
209 blendIndexesType = blendWeightsType = IQM_UBYTE;
210
211 // check and swap vertex arrays
212 if( IQM_CheckRange( header, header->ofs_vertexarrays,
213 header->num_vertexarrays,
214 sizeof(iqmVertexArray_t) ) ) {
215 return qfalse;
216 }
217 vertexarray = (iqmVertexArray_t *)((byte *)header + header->ofs_vertexarrays);
218 for( i = 0; i < header->num_vertexarrays; i++, vertexarray++ ) {
219 int n, *intPtr;
220
221 if( vertexarray->size <= 0 || vertexarray->size > 4 ) {
222 return qfalse;
223 }
224
225 // total number of values
226 n = header->num_vertexes * vertexarray->size;
227
228 switch( vertexarray->format ) {
229 case IQM_BYTE:
230 case IQM_UBYTE:
231 // 1 byte, no swapping necessary
232 if( IQM_CheckRange( header, vertexarray->offset,
233 n, sizeof(byte) ) ) {
234 return qfalse;
235 }
236 break;
237 case IQM_INT:
238 case IQM_UINT:
239 case IQM_FLOAT:
240 // 4-byte swap
241 if( IQM_CheckRange( header, vertexarray->offset,
242 n, sizeof(float) ) ) {
243 return qfalse;
244 }
245 intPtr = (int *)((byte *)header + vertexarray->offset);
246 for( j = 0; j < n; j++, intPtr++ ) {
247 LL( *intPtr );
248 }
249 break;
250 default:
251 // not supported
252 return qfalse;
253 break;
254 }
255
256 switch( vertexarray->type ) {
257 case IQM_POSITION:
258 case IQM_NORMAL:
259 if( vertexarray->format != IQM_FLOAT ||
260 vertexarray->size != 3 ) {
261 return qfalse;
262 }
263 break;
264 case IQM_TANGENT:
265 if( vertexarray->format != IQM_FLOAT ||
266 vertexarray->size != 4 ) {
267 return qfalse;
268 }
269 break;
270 case IQM_TEXCOORD:
271 if( vertexarray->format != IQM_FLOAT ||
272 vertexarray->size != 2 ) {
273 return qfalse;
274 }
275 break;
276 case IQM_BLENDINDEXES:
277 if( (vertexarray->format != IQM_INT &&
278 vertexarray->format != IQM_UBYTE) ||
279 vertexarray->size != 4 ) {
280 return qfalse;
281 }
282 blendIndexesType = vertexarray->format;
283 break;
284 case IQM_BLENDWEIGHTS:
285 if( (vertexarray->format != IQM_FLOAT &&
286 vertexarray->format != IQM_UBYTE) ||
287 vertexarray->size != 4 ) {
288 return qfalse;
289 }
290 blendWeightsType = vertexarray->format;
291 break;
292 case IQM_COLOR:
293 if( vertexarray->format != IQM_UBYTE ||
294 vertexarray->size != 4 ) {
295 return qfalse;
296 }
297 break;
298 }
299 }
300
301 // check and swap triangles
302 if( IQM_CheckRange( header, header->ofs_triangles,
303 header->num_triangles, sizeof(iqmTriangle_t) ) ) {
304 return qfalse;
305 }
306 triangle = (iqmTriangle_t *)((byte *)header + header->ofs_triangles);
307 for( i = 0; i < header->num_triangles; i++, triangle++ ) {
308 LL( triangle->vertex[0] );
309 LL( triangle->vertex[1] );
310 LL( triangle->vertex[2] );
311
312 if( triangle->vertex[0] > header->num_vertexes ||
313 triangle->vertex[1] > header->num_vertexes ||
314 triangle->vertex[2] > header->num_vertexes ) {
315 return qfalse;
316 }
317 }
318
319 // check and swap meshes
320 if( IQM_CheckRange( header, header->ofs_meshes,
321 header->num_meshes, sizeof(iqmMesh_t) ) ) {
322 return qfalse;
323 }
324 mesh = (iqmMesh_t *)((byte *)header + header->ofs_meshes);
325 for( i = 0; i < header->num_meshes; i++, mesh++) {
326 LL( mesh->name );
327 LL( mesh->material );
328 LL( mesh->first_vertex );
329 LL( mesh->num_vertexes );
330 LL( mesh->first_triangle );
331 LL( mesh->num_triangles );
332
333 if ( mesh->name < header->num_text ) {
334 Q_strncpyz( meshName, (char*)header + header->ofs_text + mesh->name, sizeof (meshName) );
335 } else {
336 meshName[0] = '\0';
337 }
338
339 // check ioq3 limits
340 if ( mesh->num_vertexes >= SHADER_MAX_VERTEXES )
341 {
342 ri.Printf(PRINT_WARNING, "R_LoadIQM: %s has more than %i verts on %s (%i).\n",
343 mod_name, SHADER_MAX_VERTEXES - 1, meshName[0] ? meshName : "a surface",
344 mesh->num_vertexes );
345 return qfalse;
346 }
347 if ( mesh->num_triangles*3 >= SHADER_MAX_INDEXES )
348 {
349 ri.Printf(PRINT_WARNING, "R_LoadIQM: %s has more than %i triangles on %s (%i).\n",
350 mod_name, ( SHADER_MAX_INDEXES / 3 ) - 1, meshName[0] ? meshName : "a surface",
351 mesh->num_triangles );
352 return qfalse;
353 }
354
355 if( mesh->first_vertex >= header->num_vertexes ||
356 mesh->first_vertex + mesh->num_vertexes > header->num_vertexes ||
357 mesh->first_triangle >= header->num_triangles ||
358 mesh->first_triangle + mesh->num_triangles > header->num_triangles ||
359 mesh->name >= header->num_text ||
360 mesh->material >= header->num_text ) {
361 return qfalse;
362 }
363 }
364
365 if( header->num_poses != header->num_joints && header->num_poses != 0 ) {
366 ri.Printf(PRINT_WARNING, "R_LoadIQM: %s has %d poses and %d joints, must have the same number or 0 poses\n",
367 mod_name, header->num_poses, header->num_joints );
368 return qfalse;
369 }
370
371 joint_names = 0;
372
373 if ( header->num_joints )
374 {
375 // check and swap joints
376 if( IQM_CheckRange( header, header->ofs_joints,
377 header->num_joints, sizeof(iqmJoint_t) ) ) {
378 return qfalse;
379 }
380 joint = (iqmJoint_t *)((byte *)header + header->ofs_joints);
381 for( i = 0; i < header->num_joints; i++, joint++ ) {
382 LL( joint->name );
383 LL( joint->parent );
384 LL( joint->translate[0] );
385 LL( joint->translate[1] );
386 LL( joint->translate[2] );
387 LL( joint->rotate[0] );
388 LL( joint->rotate[1] );
389 LL( joint->rotate[2] );
390 LL( joint->rotate[3] );
391 LL( joint->scale[0] );
392 LL( joint->scale[1] );
393 LL( joint->scale[2] );
394
395 if( joint->parent < -1 ||
396 joint->parent >= (int)header->num_joints ||
397 joint->name >= (int)header->num_text ) {
398 return qfalse;
399 }
400 joint_names += strlen( (char *)header + header->ofs_text +
401 joint->name ) + 1;
402 }
403 }
404
405 if ( header->num_poses )
406 {
407 // check and swap poses
408 if( IQM_CheckRange( header, header->ofs_poses,
409 header->num_poses, sizeof(iqmPose_t) ) ) {
410 return qfalse;
411 }
412 pose = (iqmPose_t *)((byte *)header + header->ofs_poses);
413 for( i = 0; i < header->num_poses; i++, pose++ ) {
414 LL( pose->parent );
415 LL( pose->mask );
416 LL( pose->channeloffset[0] );
417 LL( pose->channeloffset[1] );
418 LL( pose->channeloffset[2] );
419 LL( pose->channeloffset[3] );
420 LL( pose->channeloffset[4] );
421 LL( pose->channeloffset[5] );
422 LL( pose->channeloffset[6] );
423 LL( pose->channeloffset[7] );
424 LL( pose->channeloffset[8] );
425 LL( pose->channeloffset[9] );
426 LL( pose->channelscale[0] );
427 LL( pose->channelscale[1] );
428 LL( pose->channelscale[2] );
429 LL( pose->channelscale[3] );
430 LL( pose->channelscale[4] );
431 LL( pose->channelscale[5] );
432 LL( pose->channelscale[6] );
433 LL( pose->channelscale[7] );
434 LL( pose->channelscale[8] );
435 LL( pose->channelscale[9] );
436 }
437 }
438
439 if (header->ofs_bounds)
440 {
441 // check and swap model bounds
442 if(IQM_CheckRange(header, header->ofs_bounds,
443 header->num_frames, sizeof(*bounds)))
444 {
445 return qfalse;
446 }
447 bounds = (iqmBounds_t *) ((byte *) header + header->ofs_bounds);
448 for(i = 0; i < header->num_frames; i++)
449 {
450 LL(bounds->bbmin[0]);
451 LL(bounds->bbmin[1]);
452 LL(bounds->bbmin[2]);
453 LL(bounds->bbmax[0]);
454 LL(bounds->bbmax[1]);
455 LL(bounds->bbmax[2]);
456
457 bounds++;
458 }
459 }
460
461 // allocate the model and copy the data
462 size = sizeof(iqmData_t);
463 size += header->num_meshes * sizeof( srfIQModel_t );
464 size += header->num_joints * 12 * sizeof( float ); // joint mats
465 size += header->num_poses * header->num_frames * 12 * sizeof( float ); // pose mats
466 if(header->ofs_bounds)
467 size += header->num_frames * 6 * sizeof(float); // model bounds
468 size += header->num_vertexes * 3 * sizeof(float); // positions
469 size += header->num_vertexes * 2 * sizeof(float); // texcoords
470 size += header->num_vertexes * 3 * sizeof(float); // normals
471 size += header->num_vertexes * 4 * sizeof(float); // tangents
472 size += header->num_vertexes * 4 * sizeof(byte); // blendIndexes
473 size += header->num_vertexes * 4 * sizeof(byte); // colors
474 size += header->num_joints * sizeof(int); // parents
475 size += header->num_triangles * 3 * sizeof(int); // triangles
476 size += joint_names; // joint names
477
478 // blendWeights
479 if (blendWeightsType == IQM_FLOAT) {
480 size += header->num_vertexes * 4 * sizeof(float);
481 } else {
482 size += header->num_vertexes * 4 * sizeof(byte);
483 }
484
485 mod->type = MOD_IQM;
486 iqmData = (iqmData_t *)ri.Hunk_Alloc( size, h_low );
487 mod->modelData = iqmData;
488
489 // fill header
490 iqmData->num_vertexes = header->num_vertexes;
491 iqmData->num_triangles = header->num_triangles;
492 iqmData->num_frames = header->num_frames;
493 iqmData->num_surfaces = header->num_meshes;
494 iqmData->num_joints = header->num_joints;
495 iqmData->num_poses = header->num_poses;
496 iqmData->blendWeightsType = blendWeightsType;
497 iqmData->surfaces = (srfIQModel_t *)(iqmData + 1);
498 iqmData->jointMats = (float *) (iqmData->surfaces + iqmData->num_surfaces);
499 iqmData->poseMats = iqmData->jointMats + 12 * header->num_joints;
500 if(header->ofs_bounds)
501 {
502 iqmData->bounds = iqmData->poseMats + 12 * header->num_poses * header->num_frames;
503 iqmData->positions = iqmData->bounds + 6 * header->num_frames;
504 }
505 else
506 iqmData->positions = iqmData->poseMats + 12 * header->num_poses * header->num_frames;
507 iqmData->texcoords = iqmData->positions + 3 * header->num_vertexes;
508 iqmData->normals = iqmData->texcoords + 2 * header->num_vertexes;
509 iqmData->tangents = iqmData->normals + 3 * header->num_vertexes;
510 iqmData->blendIndexes = (byte *)(iqmData->tangents + 4 * header->num_vertexes);
511
512 if(blendWeightsType == IQM_FLOAT) {
513 iqmData->blendWeights.f = (float *)(iqmData->blendIndexes + 4 * header->num_vertexes);
514 iqmData->colors = (byte *)(iqmData->blendWeights.f + 4 * header->num_vertexes);
515 } else {
516 iqmData->blendWeights.b = iqmData->blendIndexes + 4 * header->num_vertexes;
517 iqmData->colors = iqmData->blendWeights.b + 4 * header->num_vertexes;
518 }
519
520 iqmData->jointParents = (int *)(iqmData->colors + 4 * header->num_vertexes);
521 iqmData->triangles = iqmData->jointParents + header->num_joints;
522 iqmData->names = (char *)(iqmData->triangles + 3 * header->num_triangles);
523
524 if ( header->num_joints == 0 )
525 iqmData->jointMats = NULL;
526
527 if ( header->num_poses == 0 )
528 iqmData->poseMats = NULL;
529
530 // calculate joint matrices and their inverses
531 // joint inverses are needed only until the pose matrices are calculated
532 mat = iqmData->jointMats;
533 matInv = jointInvMats;
534 joint = (iqmJoint_t *)((byte *)header + header->ofs_joints);
535 for( i = 0; i < header->num_joints; i++, joint++ ) {
536 float baseFrame[12], invBaseFrame[12];
537
538 JointToMatrix( joint->rotate, joint->scale, joint->translate, baseFrame );
539 Matrix34Invert( baseFrame, invBaseFrame );
540
541 if ( joint->parent >= 0 )
542 {
543 Matrix34Multiply( iqmData->jointMats + 12 * joint->parent, baseFrame, mat );
544 mat += 12;
545 Matrix34Multiply( invBaseFrame, jointInvMats + 12 * joint->parent, matInv );
546 matInv += 12;
547 }
548 else
549 {
550 Com_Memcpy( mat, baseFrame, sizeof(baseFrame) );
551 mat += 12;
552 Com_Memcpy( matInv, invBaseFrame, sizeof(invBaseFrame) );
553 matInv += 12;
554 }
555 }
556
557 // calculate pose matrices
558 framedata = (unsigned short *)((byte *)header + header->ofs_frames);
559 mat = iqmData->poseMats;
560 for( i = 0; i < header->num_frames; i++ ) {
561 pose = (iqmPose_t *)((byte *)header + header->ofs_poses);
562 for( j = 0; j < header->num_poses; j++, pose++ ) {
563 vec3_t translate;
564 vec4_t rotate;
565 vec3_t scale;
566 float mat1[12], mat2[12];
567
568 translate[0] = pose->channeloffset[0];
569 if( pose->mask & 0x001)
570 translate[0] += *framedata++ * pose->channelscale[0];
571 translate[1] = pose->channeloffset[1];
572 if( pose->mask & 0x002)
573 translate[1] += *framedata++ * pose->channelscale[1];
574 translate[2] = pose->channeloffset[2];
575 if( pose->mask & 0x004)
576 translate[2] += *framedata++ * pose->channelscale[2];
577
578 rotate[0] = pose->channeloffset[3];
579 if( pose->mask & 0x008)
580 rotate[0] += *framedata++ * pose->channelscale[3];
581 rotate[1] = pose->channeloffset[4];
582 if( pose->mask & 0x010)
583 rotate[1] += *framedata++ * pose->channelscale[4];
584 rotate[2] = pose->channeloffset[5];
585 if( pose->mask & 0x020)
586 rotate[2] += *framedata++ * pose->channelscale[5];
587 rotate[3] = pose->channeloffset[6];
588 if( pose->mask & 0x040)
589 rotate[3] += *framedata++ * pose->channelscale[6];
590
591 scale[0] = pose->channeloffset[7];
592 if( pose->mask & 0x080)
593 scale[0] += *framedata++ * pose->channelscale[7];
594 scale[1] = pose->channeloffset[8];
595 if( pose->mask & 0x100)
596 scale[1] += *framedata++ * pose->channelscale[8];
597 scale[2] = pose->channeloffset[9];
598 if( pose->mask & 0x200)
599 scale[2] += *framedata++ * pose->channelscale[9];
600
601 // construct transformation matrix
602 JointToMatrix( rotate, scale, translate, mat1 );
603
604 if( pose->parent >= 0 ) {
605 Matrix34Multiply( iqmData->jointMats + 12 * pose->parent,
606 mat1, mat2 );
607 } else {
608 Com_Memcpy( mat2, mat1, sizeof(mat1) );
609 }
610
611 Matrix34Multiply( mat2, jointInvMats + 12 * j, mat );
612 mat += 12;
613 }
614 }
615
616 // register shaders
617 // overwrite the material offset with the shader index
618 mesh = (iqmMesh_t *)((byte *)header + header->ofs_meshes);
619 surface = iqmData->surfaces;
620 str = (char *)header + header->ofs_text;
621 for( i = 0; i < header->num_meshes; i++, mesh++, surface++ ) {
622 surface->surfaceType = SF_IQM;
623 Q_strncpyz(surface->name, str + mesh->name, sizeof (surface->name));
624 Q_strlwr(surface->name); // lowercase the surface name so skin compares are faster
625 surface->shader = R_FindShader( str + mesh->material, LIGHTMAP_NONE, qtrue );
626 if( surface->shader->defaultShader )
627 surface->shader = tr.defaultShader;
628 surface->data = iqmData;
629 surface->first_vertex = mesh->first_vertex;
630 surface->num_vertexes = mesh->num_vertexes;
631 surface->first_triangle = mesh->first_triangle;
632 surface->num_triangles = mesh->num_triangles;
633 }
634
635 // copy vertexarrays and indexes
636 vertexarray = (iqmVertexArray_t *)((byte *)header + header->ofs_vertexarrays);
637 for( i = 0; i < header->num_vertexarrays; i++, vertexarray++ ) {
638 int n;
639
640 // total number of values
641 n = header->num_vertexes * vertexarray->size;
642
643 switch( vertexarray->type ) {
644 case IQM_POSITION:
645 Com_Memcpy( iqmData->positions,
646 (byte *)header + vertexarray->offset,
647 n * sizeof(float) );
648 break;
649 case IQM_NORMAL:
650 Com_Memcpy( iqmData->normals,
651 (byte *)header + vertexarray->offset,
652 n * sizeof(float) );
653 break;
654 case IQM_TANGENT:
655 Com_Memcpy( iqmData->tangents,
656 (byte *)header + vertexarray->offset,
657 n * sizeof(float) );
658 break;
659 case IQM_TEXCOORD:
660 Com_Memcpy( iqmData->texcoords,
661 (byte *)header + vertexarray->offset,
662 n * sizeof(float) );
663 break;
664 case IQM_BLENDINDEXES:
665 if( blendIndexesType == IQM_INT ) {
666 int *data = (int*)((byte*)header + vertexarray->offset);
667 for ( j = 0; j < n; j++ ) {
668 iqmData->blendIndexes[j] = (byte)data[j];
669 }
670 } else {
671 Com_Memcpy( iqmData->blendIndexes,
672 (byte *)header + vertexarray->offset,
673 n * sizeof(byte) );
674 }
675 break;
676 case IQM_BLENDWEIGHTS:
677 if( blendWeightsType == IQM_FLOAT ) {
678 Com_Memcpy( iqmData->blendWeights.f,
679 (byte *)header + vertexarray->offset,
680 n * sizeof(float) );
681 } else {
682 Com_Memcpy( iqmData->blendWeights.b,
683 (byte *)header + vertexarray->offset,
684 n * sizeof(byte) );
685 }
686 break;
687 case IQM_COLOR:
688 Com_Memcpy( iqmData->colors,
689 (byte *)header + vertexarray->offset,
690 n * sizeof(byte) );
691 break;
692 }
693 }
694
695 // copy joint parents
696 joint = (iqmJoint_t *)((byte *)header + header->ofs_joints);
697 for( i = 0; i < header->num_joints; i++, joint++ ) {
698 iqmData->jointParents[i] = joint->parent;
699 }
700
701 // copy triangles
702 triangle = (iqmTriangle_t *)((byte *)header + header->ofs_triangles);
703 for( i = 0; i < header->num_triangles; i++, triangle++ ) {
704 iqmData->triangles[3*i+0] = triangle->vertex[0];
705 iqmData->triangles[3*i+1] = triangle->vertex[1];
706 iqmData->triangles[3*i+2] = triangle->vertex[2];
707 }
708
709 // copy joint names
710 str = iqmData->names;
711 joint = (iqmJoint_t *)((byte *)header + header->ofs_joints);
712 for( i = 0; i < header->num_joints; i++, joint++ ) {
713 char *name = (char *)header + header->ofs_text +
714 joint->name;
715 int len = strlen( name ) + 1;
716 Com_Memcpy( str, name, len );
717 str += len;
718 }
719
720 // copy model bounds
721 if(header->ofs_bounds)
722 {
723 mat = iqmData->bounds;
724 bounds = (iqmBounds_t *) ((byte *) header + header->ofs_bounds);
725 for(i = 0; i < header->num_frames; i++)
726 {
727 mat[0] = bounds->bbmin[0];
728 mat[1] = bounds->bbmin[1];
729 mat[2] = bounds->bbmin[2];
730 mat[3] = bounds->bbmax[0];
731 mat[4] = bounds->bbmax[1];
732 mat[5] = bounds->bbmax[2];
733
734 mat += 6;
735 bounds++;
736 }
737 }
738
739 return qtrue;
740}

742/*
743=============
744R_CullIQM
745=============
746*/
747static int R_CullIQM( iqmData_t *data, trRefEntity_t *ent ) {
748 vec3_t bounds[2];
749 vec_t *oldBounds, *newBounds;
750 int i;
751
752 if (!data->bounds) {
753 tr.pc.c_box_cull_md3_clip++;
754 return CULL_CLIP;
755 }
756
757 // compute bounds pointers
758 oldBounds = data->bounds + 6*ent->e.oldframe;
759 newBounds = data->bounds + 6*ent->e.frame;
760
761 // calculate a bounding box in the current coordinate system
762 for (i = 0 ; i < 3 ; i++) {
763 bounds[0][i] = oldBounds[i] < newBounds[i] ? oldBounds[i] : newBounds[i];
764 bounds[1][i] = oldBounds[i+3] > newBounds[i+3] ? oldBounds[i+3] : newBounds[i+3];
765 }
766
767 switch ( R_CullLocalBox( bounds ) )
768 {
769 case CULL_IN:
770 tr.pc.c_box_cull_md3_in++;
771 return CULL_IN;
772 case CULL_CLIP:
773 tr.pc.c_box_cull_md3_clip++;
774 return CULL_CLIP;
775 case CULL_OUT:
776 default:
777 tr.pc.c_box_cull_md3_out++;
778 return CULL_OUT;
779 }
780}

782/*
783=================
784R_ComputeIQMFogNum
785
786=================
787*/
788int R_ComputeIQMFogNum( iqmData_t *data, trRefEntity_t *ent ) {
789 int i, j;
790 fog_t *fog;
791 const vec_t *bounds;
792 const vec_t defaultBounds[6] = { -8, -8, -8, 8, 8, 8 };
793 vec3_t diag, center;
794 vec3_t localOrigin;
795 vec_t radius;
796
797 if ( tr.refdef.rdflags & RDF_NOWORLDMODEL ) {
798 return 0;
799 }
800
801 // FIXME: non-normalized axis issues
802 if (data->bounds) {
803 bounds = data->bounds + 6*ent->e.frame;
804 } else {
805 bounds = defaultBounds;
806 }
807 VectorSubtract( bounds+3, bounds, diag );
808 VectorMA( bounds, 0.5f, diag, center );
809 VectorAdd( ent->e.origin, center, localOrigin );
810 radius = 0.5f * VectorLength( diag );
811
812 for ( i = 1 ; i < tr.world->numfogs ; i++ ) {
813 fog = &tr.world->fogs[i];
814 for ( j = 0 ; j < 3 ; j++ ) {
815 if ( localOrigin[j] - radius >= fog->bounds[1][j] ) {
816 break;
817 }
818 if ( localOrigin[j] + radius <= fog->bounds[0][j] ) {
819 break;
820 }
821 }
822 if ( j == 3 ) {
823 return i;
824 }
825 }
826
827 return 0;
828}

830/*
831=================
832R_AddIQMSurfaces
833
834Add all surfaces of this model
835=================
836*/
837void R_AddIQMSurfaces( trRefEntity_t *ent ) {
838 iqmData_t *data;
839 srfIQModel_t *surface;
840 int i, j;
841 qboolean personalModel;
842 int cull;
843 int fogNum;
844 int cubemapIndex;
845 shader_t *shader;
846 skin_t *skin;
847
848 data = tr.currentModel->modelData;
849 surface = data->surfaces;
850
851 // don't add third_person objects if not in a portal
852 personalModel = (ent->e.renderfx & RF_THIRD_PERSON) && !(tr.viewParms.isPortal
853 || (tr.viewParms.flags & (VPF_SHADOWMAP | VPF_DEPTHSHADOW)));
854
855 if ( ent->e.renderfx & RF_WRAP_FRAMES ) {
856 ent->e.frame %= data->num_frames;
857 ent->e.oldframe %= data->num_frames;
858 }
859
860 //
861 // Validate the frames so there is no chance of a crash.
862 // This will write directly into the entity structure, so
863 // when the surfaces are rendered, they don't need to be
864 // range checked again.
865 //
866 if ( (ent->e.frame >= data->num_frames)
867 || (ent->e.frame < 0)
868 || (ent->e.oldframe >= data->num_frames)
869 || (ent->e.oldframe < 0) ) {
870 ri.Printf( PRINT_DEVELOPER, "R_AddIQMSurfaces: no such frame %d to %d for '%s'\n",
871 ent->e.oldframe, ent->e.frame,
872 tr.currentModel->name );
873 ent->e.frame = 0;
874 ent->e.oldframe = 0;
875 }
876
877 //
878 // cull the entire model if merged bounding box of both frames
879 // is outside the view frustum.
880 //
881 cull = R_CullIQM ( data, ent );
882 if ( cull == CULL_OUT ) {
883 return;
884 }
885
886 //
887 // set up lighting now that we know we aren't culled
888 //
889 if ( !personalModel || r_shadows->integer > 1 ) {
890 R_SetupEntityLighting( &tr.refdef, ent );
891 }
892
893 //
894 // see if we are in a fog volume
895 //
896 fogNum = R_ComputeIQMFogNum( data, ent );
897
898 cubemapIndex = R_CubemapForPoint(ent->e.origin);
899
900 for ( i = 0 ; i < data->num_surfaces ; i++ ) {
901 if(ent->e.customShader)
902 shader = R_GetShaderByHandle( ent->e.customShader );
903 else if(ent->e.customSkin > 0 && ent->e.customSkin < tr.numSkins)
904 {
905 skin = R_GetSkinByHandle(ent->e.customSkin);
906 shader = tr.defaultShader;
907
908 for(j = 0; j < skin->numSurfaces; j++)
909 {
910 if (!strcmp(skin->surfaces[j].name, surface->name))
911 {
912 shader = skin->surfaces[j].shader;
913 break;
914 }
915 }
916 } else {
917 shader = surface->shader;
918 }
919
920 // we will add shadows even if the main object isn't visible in the view
921
922 // stencil shadows can't do personal models unless I polyhedron clip
923 if ( !personalModel
924 && r_shadows->integer == 2
925 && fogNum == 0
926 && !(ent->e.renderfx & ( RF_NOSHADOW | RF_DEPTHHACK ) )
927 && shader->sort == SS_OPAQUE ) {
928 R_AddDrawSurf( (void *)surface, tr.shadowShader, 0, 0, 0, 0 );
929 }
930
931 // projection shadows work fine with personal models
932 if ( r_shadows->integer == 3
933 && fogNum == 0
934 && (ent->e.renderfx & RF_SHADOW_PLANE )
935 && shader->sort == SS_OPAQUE ) {
936 R_AddDrawSurf( (void *)surface, tr.projectionShadowShader, 0, 0, 0, 0 );
937 }
938
939 if( !personalModel ) {
940 R_AddDrawSurf( (void *)surface, shader, fogNum, 0, 0, cubemapIndex );
941 }
942
943 surface++;
944 }
945}

948static void ComputePoseMats( iqmData_t *data, int frame, int oldframe,
949 float backlerp, float *mat ) {
950 float *mat1, *mat2;
951 int *joint = data->jointParents;
952 int i;
953
954 if ( data->num_poses == 0 ) {
955 for( i = 0; i < data->num_joints; i++, joint++ ) {
956 if( *joint >= 0 ) {
957 Matrix34Multiply( mat + 12 * *joint,
958 identityMatrix, mat + 12*i );
959 } else {
960 Com_Memcpy( mat + 12*i, identityMatrix, 12 * sizeof(float) );
961 }
962 }
963 return;
964 }
965
966 if ( oldframe == frame ) {
967 mat1 = data->poseMats + 12 * data->num_poses * frame;
968 for( i = 0; i < data->num_poses; i++, joint++ ) {
969 if( *joint >= 0 ) {
970 Matrix34Multiply( mat + 12 * *joint,
971 mat1 + 12*i, mat + 12*i );
972 } else {
973 Com_Memcpy( mat + 12*i, mat1 + 12*i, 12 * sizeof(float) );
974 }
975 }
976 } else {
977 mat1 = data->poseMats + 12 * data->num_poses * frame;
978 mat2 = data->poseMats + 12 * data->num_poses * oldframe;
979
980 for( i = 0; i < data->num_poses; i++, joint++ ) {
981 if( *joint >= 0 ) {
982 float tmpMat[12];
983 InterpolateMatrix( mat1 + 12*i, mat2 + 12*i,
984 backlerp, tmpMat );
985 Matrix34Multiply( mat + 12 * *joint,
986 tmpMat, mat + 12*i );
987
988 } else {
989 InterpolateMatrix( mat1 + 12*i, mat2 + 12*i,
990 backlerp, mat );
991 }
992 }
993 }
994}
995
996static void ComputeJointMats( iqmData_t *data, int frame, int oldframe,
997 float backlerp, float *mat ) {
998 float *mat1;
999 int i;
1000
1001 ComputePoseMats( data, frame, oldframe, backlerp, mat );
1002
1003 for( i = 0; i < data->num_joints; i++ ) {
1004 float outmat[12];
1005 mat1 = mat + 12 * i;
1006
1007 Com_Memcpy(outmat, mat1, sizeof(outmat));
1008
1009 Matrix34Multiply_OnlySetOrigin( outmat, data->jointMats + 12 * i, mat1 );
1010 }
1011}

1014/*
1015=================
1016RB_AddIQMSurfaces
1017
1018Compute vertices for this model surface
1019=================
1020*/
1021void RB_IQMSurfaceAnim( surfaceType_t *surface ) {
1022 srfIQModel_t *surf = (srfIQModel_t *)surface;
1023 iqmData_t *data = surf->data;
1024 float jointMats[IQM_MAX_JOINTS * 12];
1025 int i;
1026
1027 vec4_t *outXYZ;
1028 int16_t *outNormal;
1029 int16_t *outTangent;
1030 vec2_t *outTexCoord;
1031 uint16_t *outColor;
1032
1033 int frame = data->num_frames ? backEnd.currentEntity->e.frame % data->num_frames : 0;
1034 int oldframe = data->num_frames ? backEnd.currentEntity->e.oldframe % data->num_frames : 0;
1035 float backlerp = backEnd.currentEntity->e.backlerp;
1036
1037 int *tri;
1038 glIndex_t *ptr;
1039 glIndex_t base;
1040
1041 RB_CHECKOVERFLOW( surf->num_vertexes, surf->num_triangles * 3 );
1042
1043 outXYZ = &tess.xyz[tess.numVertexes];
1044 outNormal = tess.normal[tess.numVertexes];
1045 outTangent = tess.tangent[tess.numVertexes];
1046 outTexCoord = &tess.texCoords[tess.numVertexes];
1047 outColor = tess.color[tess.numVertexes];
1048
1049 // compute interpolated joint matrices
1050 if ( data->num_poses > 0 ) {
1051 ComputePoseMats( data, frame, oldframe, backlerp, jointMats );
1052 }
1053
1054 // transform vertexes and fill other data
1055 for( i = 0; i < surf->num_vertexes;
1056 i++, outXYZ++, outNormal+=4, outTexCoord++, outColor+=4 ) {
1057 int j, k;
1058 float vtxMat[12];
1059 float nrmMat[9];
1060 int vtx = i + surf->first_vertex;
1061 float blendWeights[4];
1062 int numWeights;
1063
1064 for ( numWeights = 0; numWeights < 4; numWeights++ ) {
1065 if ( data->blendWeightsType == IQM_FLOAT )
1066 blendWeights[numWeights] = data->blendWeights.f[4*vtx + numWeights];
1067 else
1068 blendWeights[numWeights] = (float)data->blendWeights.b[4*vtx + numWeights] / 255.0f;
1069
1070 if ( blendWeights[numWeights] <= 0 )
1071 break;
1072 }
1073
1074 if ( data->num_poses == 0 || numWeights == 0 ) {
1075 // no blend joint, use identity matrix.
1076 Com_Memcpy( vtxMat, identityMatrix, 12 * sizeof (float) );
1077 } else {
1078 // compute the vertex matrix by blending the up to
1079 // four blend weights
1080 Com_Memset( vtxMat, 0, 12 * sizeof (float) );
1081 for( j = 0; j < numWeights; j++ ) {
1082 for( k = 0; k < 12; k++ ) {
1083 vtxMat[k] += blendWeights[j] * jointMats[12*data->blendIndexes[4*vtx + j] + k];
1084 }
1085 }
1086 }
1087
1088 // compute the normal matrix as transpose of the adjoint
1089 // of the vertex matrix
1090 nrmMat[ 0] = vtxMat[ 5]*vtxMat[10] - vtxMat[ 6]*vtxMat[ 9];
1091 nrmMat[ 1] = vtxMat[ 6]*vtxMat[ 8] - vtxMat[ 4]*vtxMat[10];
1092 nrmMat[ 2] = vtxMat[ 4]*vtxMat[ 9] - vtxMat[ 5]*vtxMat[ 8];
1093 nrmMat[ 3] = vtxMat[ 2]*vtxMat[ 9] - vtxMat[ 1]*vtxMat[10];
1094 nrmMat[ 4] = vtxMat[ 0]*vtxMat[10] - vtxMat[ 2]*vtxMat[ 8];
1095 nrmMat[ 5] = vtxMat[ 1]*vtxMat[ 8] - vtxMat[ 0]*vtxMat[ 9];
1096 nrmMat[ 6] = vtxMat[ 1]*vtxMat[ 6] - vtxMat[ 2]*vtxMat[ 5];
1097 nrmMat[ 7] = vtxMat[ 2]*vtxMat[ 4] - vtxMat[ 0]*vtxMat[ 6];
1098 nrmMat[ 8] = vtxMat[ 0]*vtxMat[ 5] - vtxMat[ 1]*vtxMat[ 4];
1099
1100 (*outTexCoord)[0] = data->texcoords[2*vtx + 0];
1101 (*outTexCoord)[1] = data->texcoords[2*vtx + 1];
1102
1103 (*outXYZ)[0] =
1104 vtxMat[ 0] * data->positions[3*vtx+0] +
1105 vtxMat[ 1] * data->positions[3*vtx+1] +
1106 vtxMat[ 2] * data->positions[3*vtx+2] +
1107 vtxMat[ 3];
1108 (*outXYZ)[1] =
1109 vtxMat[ 4] * data->positions[3*vtx+0] +
1110 vtxMat[ 5] * data->positions[3*vtx+1] +
1111 vtxMat[ 6] * data->positions[3*vtx+2] +
1112 vtxMat[ 7];
1113 (*outXYZ)[2] =
1114 vtxMat[ 8] * data->positions[3*vtx+0] +
1115 vtxMat[ 9] * data->positions[3*vtx+1] +
1116 vtxMat[10] * data->positions[3*vtx+2] +
1117 vtxMat[11];
1118 (*outXYZ)[3] = 1.0f;
1119
1120 {
1121 vec3_t normal;
1122 vec4_t tangent;
1123
1124 normal[0] = DotProduct(&nrmMat[0], &data->normals[3*vtx]);
1125 normal[1] = DotProduct(&nrmMat[3], &data->normals[3*vtx]);
1126 normal[2] = DotProduct(&nrmMat[6], &data->normals[3*vtx]);
1127
1128 R_VaoPackNormal(outNormal, normal);
1129
1130 tangent[0] = DotProduct(&nrmMat[0], &data->tangents[4*vtx]);
1131 tangent[1] = DotProduct(&nrmMat[3], &data->tangents[4*vtx]);
1132 tangent[2] = DotProduct(&nrmMat[6], &data->tangents[4*vtx]);
1133 tangent[3] = data->tangents[4*vtx+3];
1134
1135 R_VaoPackTangent(outTangent, tangent);
1136 outTangent+=4;
1137 }
1138
1139 outColor[0] = data->colors[4*vtx+0] * 257;
1140 outColor[1] = data->colors[4*vtx+1] * 257;
1141 outColor[2] = data->colors[4*vtx+2] * 257;
1142 outColor[3] = data->colors[4*vtx+3] * 257;
1143 }
1144
1145 tri = data->triangles + 3 * surf->first_triangle;
1146 ptr = &tess.indexes[tess.numIndexes];
1147 base = tess.numVertexes;
1148
1149 for( i = 0; i < surf->num_triangles; i++ ) {
1150 *ptr++ = base + (*tri++ - surf->first_vertex);
1151 *ptr++ = base + (*tri++ - surf->first_vertex);
1152 *ptr++ = base + (*tri++ - surf->first_vertex);
1153 }
1154
1155 tess.numIndexes += 3 * surf->num_triangles;
1156 tess.numVertexes += surf->num_vertexes;
1157}

1159int R_IQMLerpTag( orientation_t *tag, iqmData_t *data,
1160 int startFrame, int endFrame,
1161 float frac, const char *tagName ) {
1162 float jointMats[IQM_MAX_JOINTS * 12];
1163 int joint;
1164 char *names = data->names;
1165
1166 // get joint number by reading the joint names
1167 for( joint = 0; joint < data->num_joints; joint++ ) {
1168 if( !strcmp( tagName, names ) )
1169 break;
1170 names += strlen( names ) + 1;
1171 }
1172 if( joint >= data->num_joints ) {
1173 AxisClear( tag->axis );
1174 VectorClear( tag->origin );
1175 return qfalse;
1176 }
1177
1178 ComputeJointMats( data, startFrame, endFrame, frac, jointMats );
1179
1180 tag->axis[0][0] = jointMats[12 * joint + 0];
1181 tag->axis[1][0] = jointMats[12 * joint + 1];
1182 tag->axis[2][0] = jointMats[12 * joint + 2];
1183 tag->origin[0] = jointMats[12 * joint + 3];
1184 tag->axis[0][1] = jointMats[12 * joint + 4];
1185 tag->axis[1][1] = jointMats[12 * joint + 5];
1186 tag->axis[2][1] = jointMats[12 * joint + 6];
1187 tag->origin[1] = jointMats[12 * joint + 7];
1188 tag->axis[0][2] = jointMats[12 * joint + 8];
1189 tag->axis[1][2] = jointMats[12 * joint + 9];
1190 tag->axis[2][2] = jointMats[12 * joint + 10];
1191 tag->origin[2] = jointMats[12 * joint + 11];
1192
1193 return qtrue;

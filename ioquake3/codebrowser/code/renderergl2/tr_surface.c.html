46/*
47==============
48RB_CheckOverflow
49==============
50*/
51void RB_CheckOverflow( int verts, int indexes ) {
52 if (tess.numVertexes + verts < SHADER_MAX_VERTEXES
53 && tess.numIndexes + indexes < SHADER_MAX_INDEXES) {
54 return;
55 }
56
57 RB_EndSurface();
58
59 if ( verts >= SHADER_MAX_VERTEXES ) {
60 ri.Error(ERR_DROP, "RB_CheckOverflow: verts > MAX (%d > %d)", verts, SHADER_MAX_VERTEXES );
61 }
62 if ( indexes >= SHADER_MAX_INDEXES ) {
63 ri.Error(ERR_DROP, "RB_CheckOverflow: indices > MAX (%d > %d)", indexes, SHADER_MAX_INDEXES );
64 }
65
66 RB_BeginSurface(tess.shader, tess.fogNum, tess.cubemapIndex );
67}
68
69void RB_CheckVao(vao_t *vao)
70{
71 if (vao != glState.currentVao)
72 {
73 RB_EndSurface();
74 RB_BeginSurface(tess.shader, tess.fogNum, tess.cubemapIndex);
75
76 R_BindVao(vao);
77 }
78
79 if (vao != tess.vao)
80 tess.useInternalVao = qfalse;
81}

84/*
85==============
86RB_AddQuadStampExt
87==============
88*/
89void RB_AddQuadStampExt( vec3_t origin, vec3_t left, vec3_t up, float color[4], float s1, float t1, float s2, float t2 ) {
90 vec3_t normal;
91 int16_t iNormal[4];
92 uint16_t iColor[4];
93 int ndx;
94
95 RB_CheckVao(tess.vao);
96
97 RB_CHECKOVERFLOW( 4, 6 );
98
99 ndx = tess.numVertexes;
100
101 // triangle indexes for a simple quad
102 tess.indexes[ tess.numIndexes ] = ndx;
103 tess.indexes[ tess.numIndexes + 1 ] = ndx + 1;
104 tess.indexes[ tess.numIndexes + 2 ] = ndx + 3;
105
106 tess.indexes[ tess.numIndexes + 3 ] = ndx + 3;
107 tess.indexes[ tess.numIndexes + 4 ] = ndx + 1;
108 tess.indexes[ tess.numIndexes + 5 ] = ndx + 2;
109
110 tess.xyz[ndx][0] = origin[0] + left[0] + up[0];
111 tess.xyz[ndx][1] = origin[1] + left[1] + up[1];
112 tess.xyz[ndx][2] = origin[2] + left[2] + up[2];
113
114 tess.xyz[ndx+1][0] = origin[0] - left[0] + up[0];
115 tess.xyz[ndx+1][1] = origin[1] - left[1] + up[1];
116 tess.xyz[ndx+1][2] = origin[2] - left[2] + up[2];
117
118 tess.xyz[ndx+2][0] = origin[0] - left[0] - up[0];
119 tess.xyz[ndx+2][1] = origin[1] - left[1] - up[1];
120 tess.xyz[ndx+2][2] = origin[2] - left[2] - up[2];
121
122 tess.xyz[ndx+3][0] = origin[0] + left[0] - up[0];
123 tess.xyz[ndx+3][1] = origin[1] + left[1] - up[1];
124 tess.xyz[ndx+3][2] = origin[2] + left[2] - up[2];
125
126
127 // constant normal all the way around
128 VectorSubtract( vec3_origin, backEnd.viewParms.or.axis[0], normal );
129
130 R_VaoPackNormal(iNormal, normal);
131
132 VectorCopy4(iNormal, tess.normal[ndx]);
133 VectorCopy4(iNormal, tess.normal[ndx + 1]);
134 VectorCopy4(iNormal, tess.normal[ndx + 2]);
135 VectorCopy4(iNormal, tess.normal[ndx + 3]);
136
137 // standard square texture coordinates
138 VectorSet2(tess.texCoords[ndx], s1, t1);
139 VectorSet2(tess.lightCoords[ndx], s1, t1);
140
141 VectorSet2(tess.texCoords[ndx+1], s2, t1);
142 VectorSet2(tess.lightCoords[ndx+1], s2, t1);
143
144 VectorSet2(tess.texCoords[ndx+2], s2, t2);
145 VectorSet2(tess.lightCoords[ndx+2], s2, t2);
146
147 VectorSet2(tess.texCoords[ndx+3], s1, t2);
148 VectorSet2(tess.lightCoords[ndx+3], s1, t2);
149
150 // constant color all the way around
151 // should this be identity and let the shader specify from entity?
152
153 R_VaoPackColor(iColor, color);
154
155 VectorCopy4(iColor, tess.color[ndx]);
156 VectorCopy4(iColor, tess.color[ndx + 1]);
157 VectorCopy4(iColor, tess.color[ndx + 2]);
158 VectorCopy4(iColor, tess.color[ndx + 3]);
159
160 tess.numVertexes += 4;
161 tess.numIndexes += 6;
162}

174/*
175==============
176RB_InstantQuad
177
178based on Tess_InstantQuad from xreal
179==============
180*/
181void RB_InstantQuad2(vec4_t quadVerts[4], vec2_t texCoords[4])
182{
183 GLimp_LogComment("--- RB_InstantQuad2 ---\n");
184
185 tess.numVertexes = 0;
186 tess.numIndexes = 0;
187 tess.firstIndex = 0;
188
189 VectorCopy4(quadVerts[0], tess.xyz[tess.numVertexes]);
190 VectorCopy2(texCoords[0], tess.texCoords[tess.numVertexes]);
191 tess.numVertexes++;
192
193 VectorCopy4(quadVerts[1], tess.xyz[tess.numVertexes]);
194 VectorCopy2(texCoords[1], tess.texCoords[tess.numVertexes]);
195 tess.numVertexes++;
196
197 VectorCopy4(quadVerts[2], tess.xyz[tess.numVertexes]);
198 VectorCopy2(texCoords[2], tess.texCoords[tess.numVertexes]);
199 tess.numVertexes++;
200
201 VectorCopy4(quadVerts[3], tess.xyz[tess.numVertexes]);
202 VectorCopy2(texCoords[3], tess.texCoords[tess.numVertexes]);
203 tess.numVertexes++;
204
205 tess.indexes[tess.numIndexes++] = 0;
206 tess.indexes[tess.numIndexes++] = 1;
207 tess.indexes[tess.numIndexes++] = 2;
208 tess.indexes[tess.numIndexes++] = 0;
209 tess.indexes[tess.numIndexes++] = 2;
210 tess.indexes[tess.numIndexes++] = 3;
211
212 RB_UpdateTessVao(ATTR_POSITION | ATTR_TEXCOORD);
213
214 R_DrawElements(tess.numIndexes, tess.firstIndex);
215
216 tess.numIndexes = 0;
217 tess.numVertexes = 0;
218 tess.firstIndex = 0;
219}

222void RB_InstantQuad(vec4_t quadVerts[4])
223{
224 vec2_t texCoords[4];

226 VectorSet2(texCoords[0], 0.0f, 0.0f);

235
236 RB_InstantQuad2(quadVerts, texCoords);
237}

240/*
241==============
242RB_SurfaceSprite
243==============
244*/
245static void RB_SurfaceSprite( void ) {
246 vec3_t left, up;
247 float radius;
248 float colors[4];
249 trRefEntity_t *ent = backEnd.currentEntity;
250
251 // calculate the xyz locations for the four corners
252 radius = ent->e.radius;
253 if ( ent->e.rotation == 0 ) {
254 VectorScale( backEnd.viewParms.or.axis[1], radius, left );
255 VectorScale( backEnd.viewParms.or.axis[2], radius, up );
256 } else {
257 float s, c;
258 float ang;
259
260 ang = M_PI * ent->e.rotation / 180;
261 s = sin( ang );
262 c = cos( ang );
263
264 VectorScale( backEnd.viewParms.or.axis[1], c * radius, left );
265 VectorMA( left, -s * radius, backEnd.viewParms.or.axis[2], left );
266
267 VectorScale( backEnd.viewParms.or.axis[2], c * radius, up );
268 VectorMA( up, s * radius, backEnd.viewParms.or.axis[1], up );
269 }
270 if ( backEnd.viewParms.isMirror ) {
271 VectorSubtract( vec3_origin, left, left );
272 }
273
274 VectorScale4(ent->e.shaderRGBA, 1.0f / 255.0f, colors);
275
276 RB_AddQuadStamp( ent->e.origin, left, up, colors );
277}

280/*
281=============
282RB_SurfacePolychain
283=============
284*/
285static void RB_SurfacePolychain( srfPoly_t *p ) {
286 int i;
287 int numv;
288
289 RB_CheckVao(tess.vao);
290
291 RB_CHECKOVERFLOW( p->numVerts, 3*(p->numVerts - 2) );
292
293 // fan triangles into the tess array
294 numv = tess.numVertexes;
295 for ( i = 0; i < p->numVerts; i++ ) {
296 VectorCopy( p->verts[i].xyz, tess.xyz[numv] );
297 tess.texCoords[numv][0] = p->verts[i].st[0];
298 tess.texCoords[numv][1] = p->verts[i].st[1];
299 tess.color[numv][0] = (int)p->verts[i].modulate[0] * 257;
300 tess.color[numv][1] = (int)p->verts[i].modulate[1] * 257;
301 tess.color[numv][2] = (int)p->verts[i].modulate[2] * 257;
302 tess.color[numv][3] = (int)p->verts[i].modulate[3] * 257;
303
304 numv++;
305 }
306
307 // generate fan indexes into the tess array
308 for ( i = 0; i < p->numVerts-2; i++ ) {
309 tess.indexes[tess.numIndexes + 0] = tess.numVertexes;
310 tess.indexes[tess.numIndexes + 1] = tess.numVertexes + i + 1;
311 tess.indexes[tess.numIndexes + 2] = tess.numVertexes + i + 2;
312 tess.numIndexes += 3;
313 }
314
315 tess.numVertexes = numv;
316}

318static void RB_SurfaceVertsAndIndexes( int numVerts, srfVert_t *verts, int numIndexes, glIndex_t *indexes, int dlightBits, int pshadowBits)
319{
320 int i;
321 glIndex_t *inIndex;
322 srfVert_t *dv;
323 float *xyz, *texCoords, *lightCoords;
324 int16_t *lightdir;
325 int16_t *normal;
326 int16_t *tangent;
327 glIndex_t *outIndex;
328 uint16_t *color;
329
330 RB_CheckVao(tess.vao);
331
332 RB_CHECKOVERFLOW( numVerts, numIndexes );
333
334 inIndex = indexes;
335 outIndex = &tess.indexes[ tess.numIndexes ];
336 for ( i = 0 ; i < numIndexes ; i++ ) {
337 *outIndex++ = tess.numVertexes + *inIndex++;
338 }
339 tess.numIndexes += numIndexes;
340
341 if ( tess.shader->vertexAttribs & ATTR_POSITION )
342 {
343 dv = verts;
344 xyz = tess.xyz[ tess.numVertexes ];
345 for ( i = 0 ; i < numVerts ; i++, dv++, xyz+=4 )
346 VectorCopy(dv->xyz, xyz);
347 }
348
349 if ( tess.shader->vertexAttribs & ATTR_NORMAL )
350 {
351 dv = verts;
352 normal = tess.normal[ tess.numVertexes ];
353 for ( i = 0 ; i < numVerts ; i++, dv++, normal+=4 )
354 VectorCopy4(dv->normal, normal);
355 }
356
357 if ( tess.shader->vertexAttribs & ATTR_TANGENT )
358 {
359 dv = verts;
360 tangent = tess.tangent[ tess.numVertexes ];
361 for ( i = 0 ; i < numVerts ; i++, dv++, tangent+=4 )
362 VectorCopy4(dv->tangent, tangent);
363 }
364
365 if ( tess.shader->vertexAttribs & ATTR_TEXCOORD )
366 {
367 dv = verts;
368 texCoords = tess.texCoords[tess.numVertexes];
369 for ( i = 0 ; i < numVerts ; i++, dv++, texCoords+=2 )
370 VectorCopy2(dv->st, texCoords);
371 }
372
373 if ( tess.shader->vertexAttribs & ATTR_LIGHTCOORD )
374 {
375 dv = verts;
376 lightCoords = tess.lightCoords[ tess.numVertexes ];
377 for ( i = 0 ; i < numVerts ; i++, dv++, lightCoords+=2 )
378 VectorCopy2(dv->lightmap, lightCoords);
379 }
380
381 if ( tess.shader->vertexAttribs & ATTR_COLOR )
382 {
383 dv = verts;
384 color = tess.color[ tess.numVertexes ];
385 for ( i = 0 ; i < numVerts ; i++, dv++, color+=4 )
386 VectorCopy4(dv->color, color);
387 }
388
389 if ( tess.shader->vertexAttribs & ATTR_LIGHTDIRECTION )
390 {
391 dv = verts;
392 lightdir = tess.lightdir[ tess.numVertexes ];
393 for ( i = 0 ; i < numVerts ; i++, dv++, lightdir+=4 )
394 VectorCopy4(dv->lightdir, lightdir);
395 }
396
397#if 0 // nothing even uses vertex dlightbits
398 for ( i = 0 ; i < numVerts ; i++ ) {
399 tess.vertexDlightBits[ tess.numVertexes + i ] = dlightBits;
400 }
401#endif
402
403 tess.dlightBits |= dlightBits;
404 tess.pshadowBits |= pshadowBits;
405
406 tess.numVertexes += numVerts;
407}

409static qboolean RB_SurfaceVaoCached(int numVerts, srfVert_t *verts, int numIndexes, glIndex_t *indexes, int dlightBits, int pshadowBits)
410{
411 qboolean recycleVertexBuffer = qfalse;
412 qboolean recycleIndexBuffer = qfalse;
413 qboolean endSurface = qfalse;
414
415 if (!(!ShaderRequiresCPUDeforms(tess.shader) && !tess.shader->isSky && !tess.shader->isPortal))
416 return qfalse;
417
418 if (!numIndexes || !numVerts)
419 return qfalse;
420
421 VaoCache_BindVao();
422
423 tess.dlightBits |= dlightBits;
424 tess.pshadowBits |= pshadowBits;
425
426 VaoCache_CheckAdd(&endSurface, &recycleVertexBuffer, &recycleIndexBuffer, numVerts, numIndexes);
427
428 if (endSurface)
429 {
430 RB_EndSurface();
431 RB_BeginSurface(tess.shader, tess.fogNum, tess.cubemapIndex);
432 }
433
434 if (recycleVertexBuffer)
435 VaoCache_RecycleVertexBuffer();
436
437 if (recycleIndexBuffer)
438 VaoCache_RecycleIndexBuffer();
439
440 if (!tess.numVertexes)
441 VaoCache_InitQueue();
442
443 VaoCache_AddSurface(verts, numVerts, indexes, numIndexes);
444
445 tess.numIndexes += numIndexes;
446 tess.numVertexes += numVerts;
447 tess.useInternalVao = qfalse;
448 tess.useCacheVao = qtrue;
449
450 return qtrue;
451}

472/*
473==============
474RB_SurfaceBeam
475==============
476*/
477static void RB_SurfaceBeam( void )
478{
479#define NUM_BEAM_SEGS 6
480 refEntity_t *e;
481 shaderProgram_t *sp = &tr.textureColorShader;
482 int i;
483 vec3_t perpvec;
484 vec3_t direction, normalized_direction;
485 vec3_t start_points[NUM_BEAM_SEGS], end_points[NUM_BEAM_SEGS];
486 vec3_t oldorigin, origin;
487
488 e = &backEnd.currentEntity->e;
489
490 oldorigin[0] = e->oldorigin[0];
491 oldorigin[1] = e->oldorigin[1];
492 oldorigin[2] = e->oldorigin[2];
493
494 origin[0] = e->origin[0];
495 origin[1] = e->origin[1];
496 origin[2] = e->origin[2];
497
498 normalized_direction[0] = direction[0] = oldorigin[0] - origin[0];
499 normalized_direction[1] = direction[1] = oldorigin[1] - origin[1];
500 normalized_direction[2] = direction[2] = oldorigin[2] - origin[2];
501
502 if ( VectorNormalize( normalized_direction ) == 0 )
503 return;
504
505 PerpendicularVector( perpvec, normalized_direction );
506
507 VectorScale( perpvec, 4, perpvec );
508
509 for ( i = 0; i < NUM_BEAM_SEGS ; i++ )
510 {
511 RotatePointAroundVector( start_points[i], normalized_direction, perpvec, (360.0/NUM_BEAM_SEGS)*i );
512// VectorAdd( start_points[i], origin, start_points[i] );
513 VectorAdd( start_points[i], direction, end_points[i] );
514 }
515
516 GL_BindToTMU( tr.whiteImage, TB_COLORMAP );
517
518 GL_State( GLS_SRCBLEND_ONE | GLS_DSTBLEND_ONE );
519
520 // FIXME: Quake3 doesn't use this, so I never tested it
521 tess.numVertexes = 0;
522 tess.numIndexes = 0;
523 tess.firstIndex = 0;
524
525 for ( i = 0; i <= NUM_BEAM_SEGS; i++ ) {
526 VectorCopy(start_points[ i % NUM_BEAM_SEGS ], tess.xyz[tess.numVertexes++]);
527 VectorCopy(end_points [ i % NUM_BEAM_SEGS ], tess.xyz[tess.numVertexes++]);
528 }
529
530 for ( i = 0; i < NUM_BEAM_SEGS; i++ ) {
531 tess.indexes[tess.numIndexes++] = i * 2;
532 tess.indexes[tess.numIndexes++] = (i + 1) * 2;
533 tess.indexes[tess.numIndexes++] = 1 + i * 2;
534
535 tess.indexes[tess.numIndexes++] = 1 + i * 2;
536 tess.indexes[tess.numIndexes++] = (i + 1) * 2;
537 tess.indexes[tess.numIndexes++] = 1 + (i + 1) * 2;
538 }
539
540 // FIXME: A lot of this can probably be removed for speed, and refactored into a more convenient function
541 RB_UpdateTessVao(ATTR_POSITION);
542
543 GLSL_BindProgram(sp);
544
545 GLSL_SetUniformMat4(sp, UNIFORM_MODELVIEWPROJECTIONMATRIX, glState.modelviewProjection);
546
547 GLSL_SetUniformVec4(sp, UNIFORM_COLOR, colorRed);
548
549 GLSL_SetUniformInt(sp, UNIFORM_ALPHATEST, 0);
550
551 R_DrawElements(tess.numIndexes, tess.firstIndex);
552
553 tess.numIndexes = 0;
554 tess.numVertexes = 0;
555 tess.firstIndex = 0;
556}

560static void DoRailCore( const vec3_t start, const vec3_t end, const vec3_t up, float len, float spanWidth )
561{
562 float spanWidth2;
563 int vbase;
564 float t = len / 256.0f;
565
566 RB_CheckVao(tess.vao);
567
568 RB_CHECKOVERFLOW( 4, 6 );
569
570 vbase = tess.numVertexes;
571
572 spanWidth2 = -spanWidth;
573
574 // FIXME: use quad stamp?
575 VectorMA( start, spanWidth, up, tess.xyz[tess.numVertexes] );
576 tess.texCoords[tess.numVertexes][0] = 0;
577 tess.texCoords[tess.numVertexes][1] = 0;
578 tess.color[tess.numVertexes][0] = backEnd.currentEntity->e.shaderRGBA[0] * 0.25f * 257.0f;
579 tess.color[tess.numVertexes][1] = backEnd.currentEntity->e.shaderRGBA[1] * 0.25f * 257.0f;
580 tess.color[tess.numVertexes][2] = backEnd.currentEntity->e.shaderRGBA[2] * 0.25f * 257.0f;
581 tess.numVertexes++;
582
583 VectorMA( start, spanWidth2, up, tess.xyz[tess.numVertexes] );
584 tess.texCoords[tess.numVertexes][0] = 0;
585 tess.texCoords[tess.numVertexes][1] = 1;
586 tess.color[tess.numVertexes][0] = backEnd.currentEntity->e.shaderRGBA[0] * 257;
587 tess.color[tess.numVertexes][1] = backEnd.currentEntity->e.shaderRGBA[1] * 257;
588 tess.color[tess.numVertexes][2] = backEnd.currentEntity->e.shaderRGBA[2] * 257;
589 tess.numVertexes++;
590
591 VectorMA( end, spanWidth, up, tess.xyz[tess.numVertexes] );
592
593 tess.texCoords[tess.numVertexes][0] = t;
594 tess.texCoords[tess.numVertexes][1] = 0;
595 tess.color[tess.numVertexes][0] = backEnd.currentEntity->e.shaderRGBA[0] * 257;
596 tess.color[tess.numVertexes][1] = backEnd.currentEntity->e.shaderRGBA[1] * 257;
597 tess.color[tess.numVertexes][2] = backEnd.currentEntity->e.shaderRGBA[2] * 257;
598 tess.numVertexes++;
599
600 VectorMA( end, spanWidth2, up, tess.xyz[tess.numVertexes] );
601 tess.texCoords[tess.numVertexes][0] = t;
602 tess.texCoords[tess.numVertexes][1] = 1;
603 tess.color[tess.numVertexes][0] = backEnd.currentEntity->e.shaderRGBA[0] * 257;
604 tess.color[tess.numVertexes][1] = backEnd.currentEntity->e.shaderRGBA[1] * 257;
605 tess.color[tess.numVertexes][2] = backEnd.currentEntity->e.shaderRGBA[2] * 257;
606 tess.numVertexes++;
607
608 tess.indexes[tess.numIndexes++] = vbase;
609 tess.indexes[tess.numIndexes++] = vbase + 1;
610 tess.indexes[tess.numIndexes++] = vbase + 2;
611
612 tess.indexes[tess.numIndexes++] = vbase + 2;
613 tess.indexes[tess.numIndexes++] = vbase + 1;
614 tess.indexes[tess.numIndexes++] = vbase + 3;
615}

617static void DoRailDiscs( int numSegs, const vec3_t start, const vec3_t dir, const vec3_t right, const vec3_t up )
618{
619 int i;
620 vec3_t pos[4];
621 vec3_t v;
622 int spanWidth = r_railWidth->integer;
623 float c, s;
624 float scale;
625
626 if ( numSegs > 1 )
627 numSegs--;
628 if ( !numSegs )
629 return;
630
631 scale = 0.25;
632
633 for ( i = 0; i < 4; i++ )
634 {
635 c = cos( DEG2RAD( 45 + i * 90 ) );
636 s = sin( DEG2RAD( 45 + i * 90 ) );
637 v[0] = ( right[0] * c + up[0] * s ) * scale * spanWidth;
638 v[1] = ( right[1] * c + up[1] * s ) * scale * spanWidth;
639 v[2] = ( right[2] * c + up[2] * s ) * scale * spanWidth;
640 VectorAdd( start, v, pos[i] );
641
642 if ( numSegs > 1 )
643 {
644 // offset by 1 segment if we're doing a long distance shot
645 VectorAdd( pos[i], dir, pos[i] );
646 }
647 }
648
649 RB_CheckVao(tess.vao);
650
651 for ( i = 0; i < numSegs; i++ )
652 {
653 int j;
654
655 RB_CHECKOVERFLOW( 4, 6 );
656
657 for ( j = 0; j < 4; j++ )
658 {
659 VectorCopy( pos[j], tess.xyz[tess.numVertexes] );
660 tess.texCoords[tess.numVertexes][0] = (j < 2);
661 tess.texCoords[tess.numVertexes][1] = (j && j != 3);
662 tess.color[tess.numVertexes][0] = backEnd.currentEntity->e.shaderRGBA[0] * 257;
663 tess.color[tess.numVertexes][1] = backEnd.currentEntity->e.shaderRGBA[1] * 257;
664 tess.color[tess.numVertexes][2] = backEnd.currentEntity->e.shaderRGBA[2] * 257;
665 tess.numVertexes++;
666
667 VectorAdd( pos[j], dir, pos[j] );
668 }
669
670 tess.indexes[tess.numIndexes++] = tess.numVertexes - 4 + 0;
671 tess.indexes[tess.numIndexes++] = tess.numVertexes - 4 + 1;
672 tess.indexes[tess.numIndexes++] = tess.numVertexes - 4 + 3;
673 tess.indexes[tess.numIndexes++] = tess.numVertexes - 4 + 3;
674 tess.indexes[tess.numIndexes++] = tess.numVertexes - 4 + 1;
675 tess.indexes[tess.numIndexes++] = tess.numVertexes - 4 + 2;
676 }
677}

778static void LerpMeshVertexes(mdvSurface_t *surf, float backlerp)
779{
780 float *outXyz;
781 int16_t *outNormal, *outTangent;
782 mdvVertex_t *newVerts;
783 int vertNum;
784
785 newVerts = surf->verts + backEnd.currentEntity->e.frame * surf->numVerts;
786
787 outXyz = tess.xyz[tess.numVertexes];
788 outNormal = tess.normal[tess.numVertexes];
789 outTangent = tess.tangent[tess.numVertexes];
790
791 if (backlerp == 0)
792 {
793 //
794 // just copy the vertexes
795 //
796
797 for (vertNum=0 ; vertNum < surf->numVerts ; vertNum++)
798 {
799 VectorCopy(newVerts->xyz, outXyz);
800 VectorCopy4(newVerts->normal, outNormal);
801 VectorCopy4(newVerts->tangent, outTangent);
802
803 newVerts++;
804 outXyz += 4;
805 outNormal += 4;
806 outTangent += 4;
807 }
808 }
809 else
810 {
811 //
812 // interpolate and copy the vertex and normal
813 //
814
815 mdvVertex_t *oldVerts;
816
817 oldVerts = surf->verts + backEnd.currentEntity->e.oldframe * surf->numVerts;
818
819 for (vertNum=0 ; vertNum < surf->numVerts ; vertNum++)
820 {
821 VectorLerp(newVerts->xyz, oldVerts->xyz, backlerp, outXyz);
822
823 outNormal[0] = (int16_t)(newVerts->normal[0] * (1.0f - backlerp) + oldVerts->normal[0] * backlerp);
824 outNormal[1] = (int16_t)(newVerts->normal[1] * (1.0f - backlerp) + oldVerts->normal[1] * backlerp);
825 outNormal[2] = (int16_t)(newVerts->normal[2] * (1.0f - backlerp) + oldVerts->normal[2] * backlerp);
826 outNormal[3] = 0;
827
828 outTangent[0] = (int16_t)(newVerts->tangent[0] * (1.0f - backlerp) + oldVerts->tangent[0] * backlerp);
829 outTangent[1] = (int16_t)(newVerts->tangent[1] * (1.0f - backlerp) + oldVerts->tangent[1] * backlerp);
830 outTangent[2] = (int16_t)(newVerts->tangent[2] * (1.0f - backlerp) + oldVerts->tangent[2] * backlerp);
831 outTangent[3] = newVerts->tangent[3];
832
833 newVerts++;
834 oldVerts++;
835 outXyz += 4;
836 outNormal += 4;
837 outTangent += 4;
838 }
839 }

841}

906static float LodErrorForVolume( vec3_t local, float radius ) {
907 vec3_t world;
908 float d;
909
910 // never let it go negative
911 if ( r_lodCurveError->value < 0 ) {
912 return 0;
913 }
914
915 world[0] = local[0] * backEnd.or.axis[0][0] + local[1] * backEnd.or.axis[1][0] +
916 local[2] * backEnd.or.axis[2][0] + backEnd.or.origin[0];
917 world[1] = local[0] * backEnd.or.axis[0][1] + local[1] * backEnd.or.axis[1][1] +
918 local[2] * backEnd.or.axis[2][1] + backEnd.or.origin[1];
919 world[2] = local[0] * backEnd.or.axis[0][2] + local[1] * backEnd.or.axis[1][2] +
920 local[2] * backEnd.or.axis[2][2] + backEnd.or.origin[2];
921
922 VectorSubtract( world, backEnd.viewParms.or.origin, world );
923 d = DotProduct( world, backEnd.viewParms.or.axis[0] );
924
925 if ( d < 0 ) {
926 d = -d;
927 }
928 d -= radius;
929 if ( d < 1 ) {
930 d = 1;
931 }
932
933 return r_lodCurveError->value / d;
934}

936/*
937=============
938RB_SurfaceGrid
939
940Just copy the grid of points and triangulate
941=============
942*/
943static void RB_SurfaceGrid( srfBspSurface_t *srf ) {
944 int i, j;
945 float *xyz;
946 float *texCoords, *lightCoords;
947 int16_t *normal;
948 int16_t *tangent;
949 uint16_t *color;
950 int16_t *lightdir;
951 srfVert_t *dv;
952 int rows, irows, vrows;
953 int used;
954 int widthTable[MAX_GRID_SIZE];
955 int heightTable[MAX_GRID_SIZE];
956 float lodError;
957 int lodWidth, lodHeight;
958 int numVertexes;
959 int dlightBits;
960 int pshadowBits;
961 //int *vDlightBits;
962
963 if (RB_SurfaceVaoCached(srf->numVerts, srf->verts, srf->numIndexes,
964 srf->indexes, srf->dlightBits, srf->pshadowBits))
965 {
966 return;
967 }
968
969 RB_CheckVao(tess.vao);
970
971 dlightBits = srf->dlightBits;
972 tess.dlightBits |= dlightBits;
973
974 pshadowBits = srf->pshadowBits;
975 tess.pshadowBits |= pshadowBits;
976
977 // determine the allowable discrepance
978 lodError = LodErrorForVolume( srf->lodOrigin, srf->lodRadius );
979
980 // determine which rows and columns of the subdivision
981 // we are actually going to use
982 widthTable[0] = 0;
983 lodWidth = 1;
984 for ( i = 1 ; i < srf->width-1 ; i++ ) {
985 if ( srf->widthLodError[i] <= lodError ) {
986 widthTable[lodWidth] = i;
987 lodWidth++;
988 }
989 }
990 widthTable[lodWidth] = srf->width-1;
991 lodWidth++;
992
993 heightTable[0] = 0;
994 lodHeight = 1;
995 for ( i = 1 ; i < srf->height-1 ; i++ ) {
996 if ( srf->heightLodError[i] <= lodError ) {
997 heightTable[lodHeight] = i;
998 lodHeight++;
999 }
1000 }
1001 heightTable[lodHeight] = srf->height-1;
1002 lodHeight++;
1003
1004
1005 // very large grids may have more points or indexes than can be fit
1006 // in the tess structure, so we may have to issue it in multiple passes
1007
1008 used = 0;
1009 while ( used < lodHeight - 1 ) {
1010 // see how many rows of both verts and indexes we can add without overflowing
1011 do {
1012 vrows = ( SHADER_MAX_VERTEXES - tess.numVertexes ) / lodWidth;
1013 irows = ( SHADER_MAX_INDEXES - tess.numIndexes ) / ( lodWidth * 6 );
1014
1015 // if we don't have enough space for at least one strip, flush the buffer
1016 if ( vrows < 2 || irows < 1 ) {
1017 RB_EndSurface();
1018 RB_BeginSurface(tess.shader, tess.fogNum, tess.cubemapIndex );
1019 } else {
1020 break;
1021 }
1022 } while ( 1 );
1023
1024 rows = irows;
1025 if ( vrows < irows + 1 ) {
1026 rows = vrows - 1;
1027 }
1028 if ( used + rows > lodHeight ) {
1029 rows = lodHeight - used;
1030 }
1031
1032 numVertexes = tess.numVertexes;
1033
1034 xyz = tess.xyz[numVertexes];
1035 normal = tess.normal[numVertexes];
1036 tangent = tess.tangent[numVertexes];
1037 texCoords = tess.texCoords[numVertexes];
1038 lightCoords = tess.lightCoords[numVertexes];
1039 color = tess.color[numVertexes];
1040 lightdir = tess.lightdir[numVertexes];
1041 //vDlightBits = &tess.vertexDlightBits[numVertexes];
1042
1043 for ( i = 0 ; i < rows ; i++ ) {
1044 for ( j = 0 ; j < lodWidth ; j++ ) {
1045 dv = srf->verts + heightTable[ used + i ] * srf->width
1046 + widthTable[ j ];
1047
1048 if ( tess.shader->vertexAttribs & ATTR_POSITION )
1049 {
1050 VectorCopy(dv->xyz, xyz);
1051 xyz += 4;
1052 }
1053
1054 if ( tess.shader->vertexAttribs & ATTR_NORMAL )
1055 {
1056 VectorCopy4(dv->normal, normal);
1057 normal += 4;
1058 }
1059
1060 if ( tess.shader->vertexAttribs & ATTR_TANGENT )
1061 {
1062 VectorCopy4(dv->tangent, tangent);
1063 tangent += 4;
1064 }
1065
1066 if ( tess.shader->vertexAttribs & ATTR_TEXCOORD )
1067 {
1068 VectorCopy2(dv->st, texCoords);
1069 texCoords += 2;
1070 }
1071
1072 if ( tess.shader->vertexAttribs & ATTR_LIGHTCOORD )
1073 {
1074 VectorCopy2(dv->lightmap, lightCoords);
1075 lightCoords += 2;
1076 }
1077
1078 if ( tess.shader->vertexAttribs & ATTR_COLOR )
1079 {
1080 VectorCopy4(dv->color, color);
1081 color += 4;
1082 }
1083
1084 if ( tess.shader->vertexAttribs & ATTR_LIGHTDIRECTION )
1085 {
1086 VectorCopy4(dv->lightdir, lightdir);
1087 lightdir += 4;
1088 }
1089
1090 //*vDlightBits++ = dlightBits;
1091 }
1092 }
1093
1094
1095 // add the indexes
1096 {
1097 int numIndexes;
1098 int w, h;
1099
1100 h = rows - 1;
1101 w = lodWidth - 1;
1102 numIndexes = tess.numIndexes;
1103 for (i = 0 ; i < h ; i++) {
1104 for (j = 0 ; j < w ; j++) {
1105 int v1, v2, v3, v4;
1106
1107 // vertex order to be reckognized as tristrips
1108 v1 = numVertexes + i*lodWidth + j + 1;
1109 v2 = v1 - 1;
1110 v3 = v2 + lodWidth;
1111 v4 = v3 + 1;
1112
1113 tess.indexes[numIndexes] = v2;
1114 tess.indexes[numIndexes+1] = v3;
1115 tess.indexes[numIndexes+2] = v1;
1116
1117 tess.indexes[numIndexes+3] = v1;
1118 tess.indexes[numIndexes+4] = v3;
1119 tess.indexes[numIndexes+5] = v4;
1120 numIndexes += 6;
1121 }
1122 }
1123
1124 tess.numIndexes = numIndexes;
1125 }
1126
1127 tess.numVertexes += rows * lodWidth;
1128
1129 used += rows - 1;
1130 }
1131}

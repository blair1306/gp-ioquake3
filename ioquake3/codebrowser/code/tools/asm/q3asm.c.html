248
249static void hashtable_init (hashtable_t *H, int buckets)
250{
251 H->buckets = buckets;
252 H->table = calloc(H->buckets, sizeof(*(H->table)));
253}
254
255static hashtable_t *hashtable_new (int buckets)
256{
257 hashtable_t *H;
258
259 H = malloc(sizeof(hashtable_t));
260 hashtable_init(H, buckets);
261 return H;
262}
263
264/* No destroy/destructor. No need. */
265
266static void hashtable_add (hashtable_t *H, int hashvalue, void *datum)
267{
268 hashchain_t *hc, **hb;
269
270 hashvalue = (abs(hashvalue) % H->buckets);
271 hb = &(H->table[hashvalue]);
272 if (*hb == 0)
273 {
274 /* Empty bucket. Create new one. */
275 *hb = calloc(1, sizeof(**hb));
276 hc = *hb;
277 }
278 else
279 {
280 /* Get hc to point to last node in chain. */
281 for (hc = *hb; hc && hc->next; hc = hc->next);
282 hc->next = calloc(1, sizeof(*hc));
283 hc = hc->next;
284 }
285 hc->data = datum;
286 hc->next = 0;
287}
288
289static hashchain_t *hashtable_get (hashtable_t *H, int hashvalue)
290{
291 hashvalue = (abs(hashvalue) % H->buckets);
292 return (H->table[hashvalue]);
293}

295static void hashtable_stats (hashtable_t *H)
296{
297 int len, empties, longest, nodes;
298 int i;
299 float meanlen;
300 hashchain_t *hc;
301
302 report("Stats for hashtable %08X", H);
303 empties = 0;
304 longest = 0;
305 nodes = 0;
306 for (i = 0; i < H->buckets; i++)
307 {
308 if (H->table[i] == 0)
309 { empties++; continue; }
310 for (hc = H->table[i], len = 0; hc; hc = hc->next, len++);
311 if (len > longest) { longest = len; }
312 nodes += len;
313 }
314 meanlen = (float)(nodes) / (H->buckets - empties);
315#if 0
316/* Long stats display */
317 report(" Total buckets: %d\n", H->buckets);
318 report(" Total stored nodes: %d\n", nodes);
319 report(" Longest chain: %d\n", longest);
320 report(" Empty chains: %d\n", empties);
321 report(" Mean non-empty chain length: %f\n", meanlen);
322#else //0
323/* Short stats display */
324 report(", %d buckets, %d nodes", H->buckets, nodes);
325 report("\n");
326 report(" Longest chain: %d, empty chains: %d, mean non-empty: %f", longest, empties, meanlen);
327#endif //0
328 report("\n");
329}

332/* Kludge. */
333/* Check if symbol already exists. */
334/* Returns 0 if symbol does NOT already exist, non-zero otherwise. */
335static int hashtable_symbol_exists (hashtable_t *H, int hash, char *sym)
336{
337 hashchain_t *hc;
338 symbol_t *s;
339
340 hash = (abs(hash) % H->buckets);
341 hc = H->table[hash];
342 if (hc == 0)
343 {
344 /* Empty chain means this symbol has not yet been defined. */
345 return 0;
346 }
347 for (; hc; hc = hc->next)
348 {
349 s = (symbol_t*)hc->data;
350// if ((hash == s->hash) && (strcmp(sym, s->name) == 0))
351/* We _already_ know the hash is the same. That's why we're probing! */
352 if (strcmp(sym, s->name) == 0)
353 {
354 /* Symbol collisions -- symbol already exists. */
355 return 1;
356 }
357 }
358 return 0; /* Can't find collision. */
359}

375/*
376 Sort the symbols list by using QuickSort (qsort()).
377 This may take a LOT of memory (a few megabytes?), but memory is cheap these days.
378 However, qsort(3) already exists, and I'm really lazy.
379 -PH
380*/
381static void sort_symbols ()
382{
383 int i, elems;
384 symbol_t *s;
385 symbol_t **symlist;
386
387 if(!symbols)
388 return;
389
390//crumb("sort_symbols: Constructing symlist array\n");
391 for (elems = 0, s = symbols; s; s = s->next, elems++) /* nop */ ;
392
393 symlist = malloc(elems * sizeof(symbol_t*));
394 for (i = 0, s = symbols; s; s = s->next, i++)
395 {
396 symlist[i] = s;
397 }
398//crumbf("sort_symbols: Quick-sorting %d symbols\n", elems);
399 qsort(symlist, elems, sizeof(symbol_t*), symlist_cmp);
400//crumbf("sort_symbols: Reconstructing symbols list\n");
401 s = symbols = symlist[0];
402 for (i = 1; i < elems; i++)
403 {
404 s->next = symlist[i];
405 s = s->next;
406 }
407 lastSymbol = s;
408 s->next = 0;
409//crumbf("sort_symbols: verifying..."); fflush(stdout);
410 for (i = 0, s = symbols; s; s = s->next, i++) /*nop*/ ;
411//crumbf(" %d elements\n", i);
412 free(symlist); /* d'oh. no gc. */
413}

424/*
425 Problem:
426 BYTE values are specified as signed decimal string. A properly functional
427 atoip() will cap large signed values at 0x7FFFFFFF. Negative word values are
428 often specified as very large decimal values by lcc. Therefore, values that
429 should be between 0x7FFFFFFF and 0xFFFFFFFF come out as 0x7FFFFFFF when using
430 atoi(). Bad.
431
432 This function is one big evil hack to work around this problem.
433*/
434static int atoiNoCap (const char *s)
435{
436 INT64 l;
437 union {
438 unsigned int u;
439 signed int i;
440 } retval;
441
442 l = atoi64(s);
443 /* Now smash to signed 32 bits accordingly. */
444 if (l < 0) {
445 retval.i = (int)l;
446 } else {
447 retval.u = (unsigned int)l;
448 }
449 return retval.i; /* <- union hackage. I feel dirty with this. -PH */
450}

454/*
455=============
456HashString
457=============
458*/
459/* Default hash function of Kazlib 1.19, slightly modified. */
460static unsigned int HashString (const char *key)
461{
462 static unsigned long randbox[] = {
463 0x49848f1bU, 0xe6255dbaU, 0x36da5bdcU, 0x47bf94e9U,
464 0x8cbcce22U, 0x559fc06aU, 0xd268f536U, 0xe10af79aU,
465 0xc1af4d69U, 0x1d2917b5U, 0xec4c304dU, 0x9ee5016cU,
466 0x69232f74U, 0xfead7bb3U, 0xe9089ab6U, 0xf012f6aeU,
467 };
468
469 const char *str = key;
470 unsigned int acc = 0;
471
472 while (*str) {
473 acc ^= randbox[(*str + acc) & 0xf];
474 acc = (acc << 1) | (acc >> 31);
475 acc &= 0xffffffffU;
476 acc ^= randbox[((*str++ >> 4) + acc) & 0xf];
477 acc = (acc << 2) | (acc >> 30);
478 acc &= 0xffffffffU;
479 }
480 return acc;
481}

502============
503EmitByte
504============
505*/
506static void EmitByte( segment_t *seg, int v ) {
507 if ( seg->imageUsed >= MAX_IMAGE ) {
508 Error( "MAX_IMAGE" );
509 }
510 seg->image[ seg->imageUsed ] = v;
511 seg->imageUsed++;
512}

514/*
515============
516EmitInt
517============
518*/
519static void EmitInt( segment_t *seg, int v ) {
520 if ( seg->imageUsed >= MAX_IMAGE - 4) {
521 Error( "MAX_IMAGE" );
522 }
523 seg->image[ seg->imageUsed ] = v & 255;
524 seg->image[ seg->imageUsed + 1 ] = ( v >> 8 ) & 255;
525 seg->image[ seg->imageUsed + 2 ] = ( v >> 16 ) & 255;
526 seg->image[ seg->imageUsed + 3 ] = ( v >> 24 ) & 255;
527 seg->imageUsed += 4;
528}

530/*
531============
532DefineSymbol
533
534Symbols can only be defined on pass 0
535============
536*/
537static void DefineSymbol( char *sym, int value ) {
538 /* Hand optimization by PhaethonH */
539 symbol_t *s;
540 char expanded[MAX_LINE_LENGTH];
541 int hash;
542
543 if ( passNumber == 1 ) {
544 return;
545 }
546
547 // add the file prefix to local symbols to guarantee unique
548 if ( sym[0] == '$' ) {
549 sprintf( expanded, "%s_%i", sym, currentFileIndex );
550 sym = expanded;
551 }
552
553 hash = HashString( sym );
554
555 if (hashtable_symbol_exists(symtable, hash, sym)) {
556 CodeError( "Multiple definitions for %s\n", sym );
557 return;
558 }
559
560 s = malloc( sizeof( *s ) );
561 s->next = NULL;
562 s->name = copystring( sym );
563 s->hash = hash;
564 s->value = value;
565 s->segment = currentSegment;
566
567 hashtable_add(symtable, hash, s);
568
569/*
570 Hash table lookup already speeds up symbol lookup enormously.
571 We postpone sorting until end of pass 0.
572 Since we're not doing the insertion sort, lastSymbol should always
573 wind up pointing to the end of list.
574 This allows constant time for adding to the list.
575 -PH
576*/
577 if (symbols == 0) {
578 lastSymbol = symbols = s;
579 } else {
580 lastSymbol->next = s;
581 lastSymbol = s;
582 }
583}

586/*
587============
588LookupSymbol
589
590Symbols can only be evaluated on pass 1
591============
592*/
593static int LookupSymbol( char *sym ) {
594 symbol_t *s;
595 char expanded[MAX_LINE_LENGTH];
596 int hash;
597 hashchain_t *hc;
598
599 if ( passNumber == 0 ) {
600 return 0;
601 }
602
603 // add the file prefix to local symbols to guarantee unique
604 if ( sym[0] == '$' ) {
605 sprintf( expanded, "%s_%i", sym, currentFileIndex );
606 sym = expanded;
607 }
608
609 hash = HashString( sym );
610
611/*
612 Hand optimization by PhaethonH
613
614 Using a hash table with chain/bucket for lookups alone sped up q3asm by almost 3x for me.
615 -PH
616*/
617 for (hc = hashtable_get(symtable, hash); hc; hc = hc->next) {
618 s = (symbol_t*)hc->data; /* ugly typecasting, but it's fast! */
619 if ( (hash == s->hash) && !strcmp(sym, s->name) ) {
620 return s->segment->segmentBase + s->value;
621 }
622 }
623
624 CodeError( "error: symbol %s undefined\n", sym );
625 passNumber = 0;
626 DefineSymbol( sym, 0 ); // so more errors aren't printed
627 passNumber = 1;
628 return 0;
629}

632/*
633==============
634ExtractLine
635
636Extracts the next line from the given text block.
637If a full line isn't parsed, returns NULL
638Otherwise returns the updated parse pointer
639===============
640*/
641static char *ExtractLine( char *data ) {
642/* Goal:
643 Given a string `data', extract one text line into buffer `lineBuffer' that
644 is no longer than MAX_LINE_LENGTH characters long. Return value is
645 remainder of `data' that isn't part of `lineBuffer'.
646 -PH
647*/
648 /* Hand-optimized by PhaethonH */
649 char *p, *q;
650
651 currentFileLine++;
652
653 lineParseOffset = 0;
654 token[0] = 0;
655 *lineBuffer = 0;
656
657 p = q = data;
658 if (!*q) {
659 return NULL;
660 }
661
662 for ( ; !((*p == 0) || (*p == '\n')); p++) /* nop */ ;
663
664 if ((p - q) >= MAX_LINE_LENGTH) {
665 CodeError( "MAX_LINE_LENGTH" );
666 return data;
667 }
668
669 memcpy( lineBuffer, data, (p - data) );
670 lineBuffer[(p - data)] = 0;
671 p += (*p == '\n') ? 1 : 0; /* Skip over final newline. */
672 return p;
673}

676/*
677==============
678Parse
679
680Parse a token out of linebuffer
681==============
682*/
683static qboolean Parse( void ) {
684 /* Hand-optimized by PhaethonH */
685 const char *p, *q;
686
687 /* Because lineParseOffset is only updated just before exit, this makes this code version somewhat harder to debug under a symbolic debugger. */
688
689 *token = 0; /* Clear token. */
690
691 // skip whitespace
692 for (p = lineBuffer + lineParseOffset; *p && (*p <= ' '); p++) /* nop */ ;
693
694 // skip ; comments
695 /* die on end-of-string */
696 if ((*p == ';') || (*p == 0)) {
697 lineParseOffset = p - lineBuffer;
698 return qfalse;
699 }
700
701 q = p; /* Mark the start of token. */
702 /* Find separator first. */
703 for ( ; *p > 32; p++) /* nop */ ; /* XXX: unsafe assumptions. */
704 /* *p now sits on separator. Mangle other values accordingly. */
705 strncpy(token, q, p - q);
706 token[p - q] = 0;
707
708 lineParseOffset = p - lineBuffer;
709
710 return qtrue;
711}

725/*
726==============
727ParseExpression
728==============
729*/
730static int ParseExpression(void) {
731 /* Hand optimization, PhaethonH */
732 int i, j;
733 char sym[MAX_LINE_LENGTH];
734 int v;
735
736 /* Skip over a leading minus. */
737 for ( i = ((token[0] == '-') ? 1 : 0) ; i < MAX_LINE_LENGTH ; i++ ) {
738 if ( token[i] == '+' || token[i] == '-' || token[i] == 0 ) {
739 break;
740 }
741 }
742
743 memcpy( sym, token, i );
744 sym[i] = 0;
745
746 switch (*sym) { /* Resolve depending on first character. */
747/* Optimizing compilers can convert cases into "calculated jumps". I think these are faster. -PH */
748 case '-':
749 case '0': case '1': case '2': case '3': case '4':
750 case '5': case '6': case '7': case '8': case '9':
751 v = atoiNoCap(sym);
752 break;
753 default:
754 v = LookupSymbol(sym);
755 break;
756 }
757
758 // parse add / subtract offsets
759 while ( token[i] != 0 ) {
760 for ( j = i + 1 ; j < MAX_LINE_LENGTH ; j++ ) {
761 if ( token[j] == '+' || token[j] == '-' || token[j] == 0 ) {
762 break;
763 }
764 }
765
766 memcpy( sym, token+i+1, j-i-1 );
767 sym[j-i-1] = 0;
768
769 switch (token[i]) {
770 case '+':
771 v += atoiNoCap(sym);
772 break;
773 case '-':
774 v -= atoiNoCap(sym);
775 break;
776 }
777
778 i = j;
779 }
780
781 return v;
782}

785/*
786==============
787HackToSegment
788
789BIG HACK: I want to put all 32 bit values in the data
790segment so they can be byte swapped, and all char data in the lit
791segment, but switch jump tables are emited in the lit segment and
792initialized strng variables are put in the data segment.
793
794I can change segments here, but I also need to fixup the
795label that was just defined
796
797Note that the lit segment is read-write in the VM, so strings
798aren't read only as in some architectures.
799==============
800*/
801static void HackToSegment( segmentName_t seg ) {
802 if ( currentSegment == &segment[seg] ) {
803 return;
804 }
805
806 currentSegment = &segment[seg];
807 if ( passNumber == 0 ) {
808 lastSymbol->segment = currentSegment;
809 lastSymbol->value = currentSegment->imageUsed;
810 }
811}

824/*
825 These clauses were moved out from AssembleLine() to allow reordering of if's.
826 An optimizing compiler should reconstruct these back into inline code.
827 -PH
828*/
829
830 // call instructions reset currentArgOffset
831ASM(CALL)
832{
833 if ( !strncmp( token, "CALL", 4 ) ) {
834STAT("CALL");
835 EmitByte( &segment[CODESEG], OP_CALL );
836 instructionCount++;
837 currentArgOffset = 0;
838 return 1;
839 }
840 return 0;
841}
842
843 // arg is converted to a reversed store
844ASM(ARG)
845{
846 if ( !strncmp( token, "ARG", 3 ) ) {
847STAT("ARG");
848 EmitByte( &segment[CODESEG], OP_ARG );
849 instructionCount++;
850 if ( 8 + currentArgOffset >= 256 ) {
851 CodeError( "currentArgOffset >= 256" );
852 return 1;
853 }
854 EmitByte( &segment[CODESEG], 8 + currentArgOffset );
855 currentArgOffset += 4;
856 return 1;
857 }
858 return 0;
859}
860
861 // ret just leaves something on the op stack
862ASM(RET)
863{
864 if ( !strncmp( token, "RET", 3 ) ) {
865STAT("RET");
866 EmitByte( &segment[CODESEG], OP_LEAVE );
867 instructionCount++;
868 EmitInt( &segment[CODESEG], 8 + currentLocals + currentArgs );
869 return 1;
870 }
871 return 0;
872}
873
874 // pop is needed to discard the return value of
875 // a function
876ASM(POP)
877{
878 if ( !strncmp( token, "pop", 3 ) ) {
879STAT("POP");
880 EmitByte( &segment[CODESEG], OP_POP );
881 instructionCount++;
882 return 1;
883 }
884 return 0;
885}

887 // address of a parameter is converted to OP_LOCAL
888ASM(ADDRF)
889{
890 int v;
891 if ( !strncmp( token, "ADDRF", 5 ) ) {
892STAT("ADDRF");
893 instructionCount++;
894 Parse();
895 v = ParseExpression();
896 v = 16 + currentArgs + currentLocals + v;
897 EmitByte( &segment[CODESEG], OP_LOCAL );
898 EmitInt( &segment[CODESEG], v );
899 return 1;
900 }
901 return 0;
902}

906{
907 int v;
908 if ( !strncmp( token, "ADDRL", 5 ) ) {
909STAT("ADDRL");
910 instructionCount++;
911 Parse();
912 v = ParseExpression();

915 EmitInt( &segment[CODESEG], v );
916 return 1;

919}

921ASM(PROC)
922{
923 char name[1024];
924 if ( !strcmp( token, "proc" ) ) {
925STAT("PROC");
926 Parse(); // function name
927 strcpy( name, token );
928
929 DefineSymbol( token, instructionCount ); // segment[CODESEG].imageUsed );
930
931 currentLocals = ParseValue(); // locals
932 currentLocals = ( currentLocals + 3 ) & ~3;
933 currentArgs = ParseValue(); // arg marshalling
934 currentArgs = ( currentArgs + 3 ) & ~3;
935
936 if ( 8 + currentLocals + currentArgs >= 32767 ) {
937 CodeError( "Locals > 32k in %s\n", name );
938 }
939
940 instructionCount++;
941 EmitByte( &segment[CODESEG], OP_ENTER );
942 EmitInt( &segment[CODESEG], 8 + currentLocals + currentArgs );
943 return 1;
944 }
945 return 0;
946}

949ASM(ENDPROC)
950{
951 if ( !strcmp( token, "endproc" ) ) {
952STAT("ENDPROC");
953 Parse(); // skip the function name
954 ParseValue(); // locals
955 ParseValue(); // arg marshalling
956
957 // all functions must leave something on the opstack
958 instructionCount++;
959 EmitByte( &segment[CODESEG], OP_PUSH );
960
961 instructionCount++;
962 EmitByte( &segment[CODESEG], OP_LEAVE );
963 EmitInt( &segment[CODESEG], 8 + currentLocals + currentArgs );
964
965 return 1;
966 }
967 return 0;
968}

984 return 1;
985 }

1103ASM(BYTE)
1104{
1105 int i, v, v2;
1106 if ( !strcmp( token, "byte" ) ) {
1107STAT("BYTE");
1108 v = ParseValue();
1109 v2 = ParseValue();
1110
1111 if ( v == 1 ) {
1112/* Character (1-byte) values go into lit(eral) segment. */
1113 HackToSegment( LITSEG );
1114 } else if ( v == 4 ) {
1115/* 32-bit (4-byte) values go into data segment. */
1116 HackToSegment( DATASEG );
1117 } else if ( v == 2 ) {
1118/* and 16-bit (2-byte) values will cause q3asm to barf. */
1119 CodeError( "16 bit initialized data not supported" );
1120 }
1121
1122 // emit little endien
1123 for ( i = 0 ; i < v ; i++ ) {
1124 EmitByte( currentSegment, (v2 & 0xFF) ); /* paranoid ANDing -PH */
1125 v2 >>= 8;
1126 }
1127 return 1;
1128 }
1129 return 0;
1130}

1133 // because the physical size of the code will change with
1134 // different run time compilers and we want to minimize the

1137{

1142 DefineSymbol( token, instructionCount );
1143 } else {
1144 DefineSymbol( token, currentSegment->imageUsed );
1145 }
1146 return 1;
1147 }
1148 return 0;
1149}

1153/*
1154==============
1155AssembleLine
1156
1157==============
1158*/
1159static void AssembleLine( void ) {
1160 hashchain_t *hc;
1161 sourceOps_t *op;
1162 int i;
1163 int hash;
1164
1165 Parse();
1166 if ( !token[0] ) {
1167 return;
1168 }
1169
1170 hash = HashString( token );
1171
1172/*
1173 Opcode search using hash table.
1174 Since the opcodes stays mostly fixed, this may benefit even more from a tree.
1175 Always with the tree :)
1176 -PH
1177*/
1178 for (hc = hashtable_get(optable, hash); hc; hc = hc->next) {
1179 op = (sourceOps_t*)(hc->data);
1180 i = op - sourceOps;
1181 if ((hash == opcodesHash[i]) && (!strcmp(token, op->name))) {
1182 int opcode;
1183 int expression;
1184
1185 if ( op->opcode == OP_UNDEF ) {
1186 CodeError( "Undefined opcode: %s\n", token );
1187 }
1188 if ( op->opcode == OP_IGNORE ) {
1189 return; // we ignore most conversions
1190 }
1191
1192 // sign extensions need to check next parm
1193 opcode = op->opcode;
1194 if ( opcode == OP_SEX8 ) {
1195 Parse();
1196 if ( token[0] == '1' ) {
1197 opcode = OP_SEX8;
1198 } else if ( token[0] == '2' ) {
1199 opcode = OP_SEX16;
1200 } else {
1201 CodeError( "Bad sign extension: %s\n", token );
1202 return;
1203 }
1204 }
1205
1206 // check for expression
1207 Parse();
1208 if ( token[0] && op->opcode != OP_CVIF
1209 && op->opcode != OP_CVFI ) {
1210 expression = ParseExpression();
1211
1212 // code like this can generate non-dword block copies:
1213 // auto char buf[2] = " ";
1214 // we are just going to round up. This might conceivably
1215 // be incorrect if other initialized chars follow.
1216 if ( opcode == OP_BLOCK_COPY ) {
1217 expression = ( expression + 3 ) & ~3;
1218 }
1219
1220 EmitByte( &segment[CODESEG], opcode );
1221 EmitInt( &segment[CODESEG], expression );
1222 } else {
1223 EmitByte( &segment[CODESEG], opcode );
1224 }
1225
1226 instructionCount++;
1227 return;
1228 }
1229 }

1342===============
1343WriteVmFile
1344===============
1345*/
1346static void WriteVmFile( void ) {
1347 char imageName[MAX_OS_PATH];
1348 vmHeader_t header;
1349 FILE *f;
1350 int headerSize;
1351
1352 report( "%i total errors\n", errorCount );
1353
1354 strcpy( imageName, outputFilename );
1355 StripExtension( imageName );
1356 strcat( imageName, ".qvm" );
1357
1358 remove( imageName );
1359
1360 report( "code segment: %7i\n", segment[CODESEG].imageUsed );
1361 report( "data segment: %7i\n", segment[DATASEG].imageUsed );
1362 report( "lit segment: %7i\n", segment[LITSEG].imageUsed );
1363 report( "bss segment: %7i\n", segment[BSSSEG].imageUsed );
1364 report( "instruction count: %i\n", instructionCount );
1365
1366 if ( errorCount != 0 ) {
1367 report( "Not writing a file due to errors\n" );
1368 return;
1369 }
1370
1371 if( !options.vanillaQ3Compatibility ) {
1372 header.vmMagic = VM_MAGIC_VER2;
1373 headerSize = sizeof( header );
1374 } else {
1375 header.vmMagic = VM_MAGIC;
1376
1377 // Don't write the VM_MAGIC_VER2 bits when maintaining 1.32b compatibility.
1378 // (I know this isn't strictly correct due to padding, but then platforms
1379 // that pad wouldn't be able to write a correct header anyway). Note: if
1380 // vmHeader_t changes, this needs to be adjusted too.
1381 headerSize = sizeof( header ) - sizeof( header.jtrgLength );
1382 }
1383
1384 header.instructionCount = instructionCount;
1385 header.codeOffset = headerSize;
1386 header.codeLength = segment[CODESEG].imageUsed;
1387 header.dataOffset = header.codeOffset + segment[CODESEG].imageUsed;
1388 header.dataLength = segment[DATASEG].imageUsed;
1389 header.litLength = segment[LITSEG].imageUsed;
1390 header.bssLength = segment[BSSSEG].imageUsed;
1391 header.jtrgLength = segment[JTRGSEG].imageUsed;
1392
1393 report( "Writing to %s\n", imageName );
1394
1395#ifdef Q3_BIG_ENDIAN
1396 {
1397 int i;
1398
1399 // byte swap the header
1400 for ( i = 0 ; i < sizeof( vmHeader_t ) / 4 ; i++ ) {
1401 ((int *)&header)[i] = LittleLong( ((int *)&header)[i] );
1402 }
1403 }
1404#endif
1405
1406 CreatePath( imageName );
1407 f = SafeOpenWrite( imageName );
1408 SafeWrite( f, &header, headerSize );
1409 SafeWrite( f, &segment[CODESEG].image, segment[CODESEG].imageUsed );
1410 SafeWrite( f, &segment[DATASEG].image, segment[DATASEG].imageUsed );
1411 SafeWrite( f, &segment[LITSEG].image, segment[LITSEG].imageUsed );
1412
1413 if( !options.vanillaQ3Compatibility ) {
1414 SafeWrite( f, &segment[JTRGSEG].image, segment[JTRGSEG].imageUsed );
1415 }
1416
1417 fclose( f );
1418}

1420/*
1421===============
1422Assemble
1423===============
1424*/
1425static void Assemble( void ) {
1426 int i;
1427 char filename[MAX_OS_PATH];
1428 char *ptr;
1429
1430 report( "outputFilename: %s\n", outputFilename );
1431
1432 for ( i = 0 ; i < numAsmFiles ; i++ ) {
1433 strcpy( filename, asmFileNames[ i ] );
1434 DefaultExtension( filename, ".asm" );
1435 LoadFile( filename, (void **)&asmFiles[i] );
1436 }
1437
1438 // assemble
1439 for ( passNumber = 0 ; passNumber < 2 ; passNumber++ ) {
1440 segment[LITSEG].segmentBase = segment[DATASEG].imageUsed;
1441 segment[BSSSEG].segmentBase = segment[LITSEG].segmentBase + segment[LITSEG].imageUsed;
1442 segment[JTRGSEG].segmentBase = segment[BSSSEG].segmentBase + segment[BSSSEG].imageUsed;
1443 for ( i = 0 ; i < NUM_SEGMENTS ; i++ ) {
1444 segment[i].imageUsed = 0;
1445 }
1446 segment[DATASEG].imageUsed = 4; // skip the 0 byte, so NULL pointers are fixed up properly
1447 instructionCount = 0;
1448
1449 for ( i = 0 ; i < numAsmFiles ; i++ ) {
1450 currentFileIndex = i;
1451 currentFileName = asmFileNames[ i ];
1452 currentFileLine = 0;
1453 report("pass %i: %s\n", passNumber, currentFileName );
1454 fflush( NULL );
1455 ptr = asmFiles[i];
1456 while ( ptr ) {
1457 ptr = ExtractLine( ptr );
1458 AssembleLine();
1459 }
1460 }
1461
1462 // align all segment
1463 for ( i = 0 ; i < NUM_SEGMENTS ; i++ ) {
1464 segment[i].imageUsed = (segment[i].imageUsed + 3) & ~3;
1465 }
1466 if (passNumber == 0) {
1467 sort_symbols();
1468 }
1469 }
1470
1471 // reserve the stack in bss
1472 DefineSymbol( "_stackStart", segment[BSSSEG].imageUsed );
1473 segment[BSSSEG].imageUsed += stackSize;
1474 DefineSymbol( "_stackEnd", segment[BSSSEG].imageUsed );
1475
1476 // write the image
1477 WriteVmFile();
1478
1479 // write the map file even if there were errors
1480 if( options.writeMapFile ) {
1481 WriteMapFile();
1482 }
1483}

1534/*
1535==============
1536main
1537==============
1538*/
1539int main( int argc, char **argv ) {
1540 int i;
1541 double start, end;
1542
1543// _chdir( "/quake3/jccode/cgame/lccout" ); // hack for vc profiler
1544
1545 if ( argc < 2 ) {
1546 ShowHelp( argv[0] );
1547 }
1548
1549 start = I_FloatTime ();
1550
1551 // default filename is "q3asm"
1552 strcpy( outputFilename, "q3asm" );
1553 numAsmFiles = 0;
1554
1555 for ( i = 1 ; i < argc ; i++ ) {
1556 if ( argv[i][0] != '-' ) {
1557 break;
1558 }
1559 if( !strcmp( argv[ i ], "-h" ) ||
1560 !strcmp( argv[ i ], "--help" ) ||
1561 !strcmp( argv[ i ], "-?") ) {
1562 ShowHelp( argv[0] );
1563 }
1564
1565 if ( !strcmp( argv[i], "-o" ) ) {
1566 if ( i == argc - 1 ) {
1567 Error( "-o must preceed a filename" );
1568 }
1569/* Timbo of Tremulous pointed out -o not working; stock ID q3asm folded in the change. Yay. */
1570 strcpy( outputFilename, argv[ i+1 ] );
1571 i++;
1572 continue;
1573 }
1574
1575 if ( !strcmp( argv[i], "-f" ) ) {
1576 if ( i == argc - 1 ) {
1577 Error( "-f must preceed a filename" );
1578 }
1579 ParseOptionFile( argv[ i+1 ] );
1580 i++;
1581 continue;
1582 }
1583
1584 if (!strcmp(argv[i], "-b")) {
1585 if (i == argc - 1) {
1586 Error("-b requires an argument");
1587 }
1588 i++;
1589 symtablelen = atoiNoCap(argv[i]);
1590 continue;
1591 }
1592
1593 if( !strcmp( argv[ i ], "-v" ) ) {
1594/* Verbosity option added by Timbo, 2002.09.14.
1595By default (no -v option), q3asm remains silent except for critical errors.
1596Verbosity turns on all messages, error or not.
1597Motivation: not wanting to scrollback for pages to find asm error.
1598*/
1599 options.verbose = qtrue;
1600 continue;
1601 }
1602
1603 if( !strcmp( argv[ i ], "-m" ) ) {
1604 options.writeMapFile = qtrue;
1605 continue;
1606 }
1607
1608 if( !strcmp( argv[ i ], "-vq3" ) ) {
1609 options.vanillaQ3Compatibility = qtrue;
1610 continue;
1611 }
1612
1613 Error( "Unknown option: %s", argv[i] );
1614 }
1615
1616 // the rest of the command line args are asm files
1617 for ( ; i < argc ; i++ ) {
1618 asmFileNames[ numAsmFiles ] = copystring( argv[ i ] );
1619 numAsmFiles++;
1620 }
1621 // In some case it Segfault without this check
1622 if ( numAsmFiles == 0 ) {
1623 Error( "No file to assemble" );
1624 }
1625
1626 InitTables();
1627 Assemble();
1628
1629 {
1630 symbol_t *s;
1631
1632 for ( i = 0, s = symbols ; s ; s = s->next, i++ ) /* nop */ ;
1633
1634 if (options.verbose)
1635 {
1636 report("%d symbols defined\n", i);
1637 hashtable_stats(symtable);
1638 hashtable_stats(optable);
1639 }
1640 }
1641
1642 end = I_FloatTime ();
1643 report ("%5.0f seconds elapsed\n", end-start);
1644
1645 return errorCount;
1646}

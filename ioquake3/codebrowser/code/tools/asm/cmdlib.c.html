53ExpandWildcards
54
55Mimic unix command line expansion
56===================

58#define MAX_EX_ARGC 1024
59int ex_argc;
60char *ex_argv[MAX_EX_ARGC];
61#ifdef _WIN32
62#include "io.h"
63void ExpandWildcards( int *argc, char ***argv )
64{
65 struct _finddata_t fileinfo;
66 intptr_t handle;
67 int i;
68 char filename[1024];
69 char filebase[1024];
70 char *path;
71
72 ex_argc = 0;
73 for (i=0 ; i<*argc ; i++)
74 {
75 path = (*argv)[i];
76 if ( path[0] == '-'
77 || ( !strstr(path, "*") && !strstr(path, "?") ) )
78 {
79 ex_argv[ex_argc++] = path;
80 continue;
81 }
82
83 handle = _findfirst (path, &fileinfo);
84 if (handle == -1)
85 return;
86
87 ExtractFilePath (path, filebase);
88
89 do
90 {
91 sprintf (filename, "%s%s", filebase, fileinfo.name);
92 ex_argv[ex_argc++] = copystring (filename);
93 } while (_findnext( handle, &fileinfo ) != -1);
94
95 _findclose (handle);
96 }
97
98 *argc = ex_argc;
99 *argv = ex_argv;
100}

103{
104}

208qdir will hold the path up to the quake directory, including the slash
209
210 f:\quake\
211 /raid/quake/
212

214

217char qdir[1024];
218char gamedir[1024];
219char writedir[1024];
220
221void SetQdirFromPath( const char *path )
222{
223 char temp[1024];
224 const char *c;
225 const char *sep;
226 int len, count;
227
228 if (!(path[0] == '/' || path[0] == '\\' || path[1] == ':'))
229 { // path is partial
230 Q_getwd (temp);
231 strcat (temp, path);
232 path = temp;
233 }
234
235 // search for "quake2" in path
236
237 len = strlen(BASEDIRNAME);
238 for (c=path+strlen(path)-1 ; c != path ; c--)
239 {
240 int i;
241
242 if (!Q_strncasecmp (c, BASEDIRNAME, len))
243 {
244 //
245 //strncpy (qdir, path, c+len+2-path);
246 // the +2 assumes a 2 or 3 following quake which is not the
247 // case with a retail install
248 // so we need to add up how much to the next separator
249 sep = c + len;
250 count = 1;
251 while (*sep && *sep != '/' && *sep != '\\')
252 {
253 sep++;
254 count++;
255 }
256 strncpy (qdir, path, c+len+count-path);
257 qprintf ("qdir: %s\n", qdir);
258 for ( i = 0; i < strlen( qdir ); i++ )
259 {
260 if ( qdir[i] == '\\' )
261 qdir[i] = '/';
262 }
263
264 c += len+count;
265 while (*c)
266 {
267 if (*c == '/' || *c == '\\')
268 {
269 strncpy (gamedir, path, c+1-path);
270
271 for ( i = 0; i < strlen( gamedir ); i++ )
272 {
273 if ( gamedir[i] == '\\' )
274 gamedir[i] = '/';
275 }
276
277 qprintf ("gamedir: %s\n", gamedir);
278
279 if ( !writedir[0] )
280 strcpy( writedir, gamedir );
281 else if ( writedir[strlen( writedir )-1] != '/' )
282 {
283 writedir[strlen( writedir )] = '/';
284 writedir[strlen( writedir )+1] = 0;
285 }
286
287 return;
288 }
289 c++;
290 }
291 Error ("No gamedir in %s", path);
292 return;
293 }
294 }
295 Error ("SetQdirFromPath: no '%s' in %s", BASEDIRNAME, path);
296}

449/*
450==============
451COM_Parse
452
453Parse a token out of a string
454==============
455*/
456char *COM_Parse (char *data)
457{
458 int c;
459 int len;
460
461 len = 0;
462 com_token[0] = 0;
463
464 if (!data)
465 return NULL;
466
467// skip whitespace
468skipwhite:
469 while ( (c = *data) <= ' ')
470 {
471 if (c == 0)
472 {
473 com_eof = qtrue;
474 return NULL; // end of file;
475 }
476 data++;
477 }
478
479// skip // comments
480 if (c=='/' && data[1] == '/')
481 {
482 while (*data && *data != '\n')
483 data++;
484 goto skipwhite;
485 }
486
487
488// handle quoted strings specially
489 if (c == '\"')
490 {
491 data++;
492 do
493 {
494 c = *data++;
495 if (c=='\"')
496 {
497 com_token[len] = 0;
498 return data;
499 }
500 com_token[len] = c;
501 len++;
502 } while (1);
503 }
504
505// parse single characters
506 if (c=='{' || c=='}'|| c==')'|| c=='(' || c=='\'' || c==':')
507 {
508 com_token[len] = c;
509 len++;
510 com_token[len] = 0;
511 return data+1;
512 }
513
514// parse a regular word
515 do
516 {
517 com_token[len] = c;
518 data++;
519 len++;
520 c = *data;
521 if (c=='{' || c=='}'|| c==')'|| c=='(' || c=='\'' || c==':')
522 break;
523 } while (c>32);
524
525 com_token[len] = 0;
526 return data;
527}

530int Q_strncasecmp (const char *s1, const char *s2, int n)
531{
532 int c1, c2;
533
534 do
535 {
536 c1 = *s1++;
537 c2 = *s2++;
538
539 if (!n--)
540 return 0; // strings are equal until end point
541
542 if (c1 != c2)
543 {
544 if (c1 >= 'a' && c1 <= 'z')
545 c1 -= ('a' - 'A');
546 if (c2 >= 'a' && c2 <= 'z')
547 c2 -= ('a' - 'A');
548 if (c1 != c2)
549 return -1; // strings not equal
550 }
551 } while (c1);
552
553 return 0; // strings are equal
554}

556int Q_stricmp (const char *s1, const char *s2)
557{
558 return Q_strncasecmp (s1, s2, 99999);
559}

930/*
931==============
932ParseNum / ParseHex
933==============
934*/
935int ParseHex (const char *hex)
936{
937 const char *str;
938 int num;
939
940 num = 0;
941 str = hex;
942
943 while (*str)
944 {
945 num <<= 4;
946 if (*str >= '0' && *str <= '9')
947 num += *str-'0';
948 else if (*str >= 'a' && *str <= 'f')
949 num += 10 + *str-'a';
950 else if (*str >= 'A' && *str <= 'F')
951 num += 10 + *str-'A';
952 else
953 Error ("Bad hex number: %s",hex);
954 str++;
955 }
956
957 return num;
958}

966 return ParseHex (str+2);

980short ShortSwap (short l)
981{
982 byte b1,b2;
983
984 b1 = l&255;
985 b2 = (l>>8)&255;
986
987 return (b1<<8) + b2;
988}
989
990int LongSwap (int l)
991{
992 byte b1,b2,b3,b4;
993
994 b1 = l&255;
995 b2 = (l>>8)&255;
996 b3 = (l>>16)&255;
997 b4 = (l>>24)&255;
998
999 return ((int)b1<<24) + ((int)b2<<16) + ((int)b3<<8) + b4;
1000}

1009

1102 path += 2;

1105 {
1106 c = *ofs;

1108 { // create the directory

1113 }

1115#ifdef _WIN32

1117 {
1118 _chdrive( olddrive );

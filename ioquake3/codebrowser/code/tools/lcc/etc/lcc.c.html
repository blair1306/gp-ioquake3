83int main(int argc, char *argv[]) {
84 int i, j, nf;
85
86 progname = argv[0];
87
88 UpdatePaths( progname );
89
90 ac = argc + 50;
91 av = alloc(ac*sizeof(char *));
92 if (signal(SIGINT, SIG_IGN) != SIG_IGN)
93 signal(SIGINT, interrupt);
94 if (signal(SIGTERM, SIG_IGN) != SIG_IGN)
95 signal(SIGTERM, interrupt);
96#ifdef SIGHUP
97 if (signal(SIGHUP, SIG_IGN) != SIG_IGN)
98 signal(SIGHUP, interrupt);
99#endif
100 if (getenv("TMP"))
101 tempdir = getenv("TMP");
102 else if (getenv("TEMP"))
103 tempdir = getenv("TEMP");
104 else if (getenv("TMPDIR"))
105 tempdir = getenv("TMPDIR");
106 assert(tempdir);
107 i = strlen(tempdir);
108 for (; (i > 0 && tempdir[i-1] == '/') || tempdir[i-1] == '\\'; i--)
109 tempdir[i-1] = '\0';
110 if (argc <= 1) {
111 help();
112 exit(0);
113 }
114 plist = append("-D__LCC__", 0);
115 initinputs();
116 if (getenv("LCCDIR"))
117 option(stringf("-lccdir=%s", getenv("LCCDIR")));
118 for (nf = 0, i = j = 1; i < argc; i++) {
119 if (strcmp(argv[i], "-o") == 0) {
120 if (++i < argc) {
121 if (suffix(argv[i], suffixes, 2) >= 0) {
122 error("-o would overwrite %s", argv[i]);
123 exit(8);
124 }
125 outfile = argv[i];
126 continue;
127 } else {
128 error("unrecognized option `%s'", argv[i-1]);
129 exit(8);
130 }
131 } else if (strcmp(argv[i], "-target") == 0) {
132 if (argv[i+1] && *argv[i+1] != '-')
133 i++;
134 continue;
135 } else if (*argv[i] == '-' && argv[i][1] != 'l') {
136 opt(argv[i]);
137 continue;
138 } else if (*argv[i] != '-' && suffix(argv[i], suffixes, 3) >= 0)
139 nf++;
140 argv[j++] = argv[i];
141 }
142 if ((cflag || Sflag) && outfile && nf != 1) {
143 fprintf(stderr, "%s: -o %s ignored\n", progname, outfile);
144 outfile = 0;
145 }
146 argv[j] = 0;
147 for (i = 0; include[i]; i++)
148 plist = append(include[i], plist);
149 if (ilist) {
150 List b = ilist;
151 do {
152 b = b->link;
153 plist = append(b->str, plist);
154 } while (b != ilist);
155 }
156 ilist = 0;
157 for (i = 1; argv[i]; i++)
158 if (*argv[i] == '-')
159 opt(argv[i]);
160 else {
161 char *name = exists(argv[i]);
162 if (name) {
163 if (strcmp(name, argv[i]) != 0
164 || (nf > 1 && suffix(name, suffixes, 3) >= 0))
165 fprintf(stderr, "%s:\n", name);
166 filename(name, 0);
167 } else
168 error("can't find `%s'", argv[i]);
169 }
170 if (errcnt == 0 && !Eflag && !Sflag && !cflag && llist[1]) {
171 compose(ld, llist[0], llist[1],
172 append(outfile ? outfile : concat("a", first(suffixes[4])), 0));
173 if (callsys(av))
174 errcnt++;
175 }
176 rm(rmlist);
177 return errcnt ? EXIT_FAILURE : EXIT_SUCCESS;
178}

180/* alloc - allocate n bytes or die */
181static void *alloc(int n) {
182 static char *avail, *limit;
183
184 n = (n + sizeof(char *) - 1)&~(sizeof(char *) - 1);
185 if (n >= limit - avail) {
186 avail = malloc(n + 4*1024);
187 assert(avail);
188 limit = avail + n + 4*1024;
189 }
190 avail += n;
191 return avail - n;
192}

195static List append(char *str, List list) {

198 p->str = str;
199 if (list) {
200 p->link = list->link;
201 list->link = p;
202 } else
203 p->link = p;
204 return p;
205}

207/* basename - return base name for name, e.g. /usr/drh/foo.c => foo */
208char *basename(char *name) {
209 char *s, *b, *t = 0;
210
211 for (b = s = name; *s; s++)
212 if (*s == '/' || *s == '\\') {
213 b = s + 1;
214 t = 0;
215 } else if (*s == '.')
216 t = s;
217 s = strsave(b);
218 if (t)
219 s[t-b] = 0;
220 return s;
221}

226static char *escapeDoubleQuotes(const char *string) {
227 int stringLength = strlen(string);
228 int bufferSize = stringLength + 1;
229 int i, j;
230 char *newString;
231
232 if (string == NULL)
233 return NULL;
234
235 for (i = 0; i < stringLength; i++) {
236 if (string[i] == '"')
237 bufferSize++;
238 }
239
240 newString = (char*)malloc(bufferSize);
241
242 if (newString == NULL)
243 return NULL;
244
245 for (i = 0, j = 0; i < stringLength; i++) {
246 if (string[i] == '"')
247 newString[j++] = '\\';
248
249 newString[j++] = string[i];
250 }
251
252 newString[j] = '\0';
253
254 return newString;
255}

257static int spawn(const char *cmdname, char **argv) {
258 int argc = 0;
259 char **newArgv = argv;
260 int i;
261 intptr_t exitStatus;
262
263 // _spawnvp removes double quotes from arguments, so we
264 // have to escape them manually
265 while (*newArgv++ != NULL)
266 argc++;
267
268 newArgv = (char **)malloc(sizeof(char*) * (argc + 1));
269
270 for (i = 0; i < argc; i++)
271 newArgv[i] = escapeDoubleQuotes(argv[i]);
272
273 newArgv[argc] = NULL;
274
275 exitStatus = _spawnvp(_P_WAIT, cmdname, (const char *const *)newArgv);
276
277 for (i = 0; i < argc; i++)
278 free(newArgv[i]);
279
280 free(newArgv);
281 return exitStatus;
282}

292static int spawn(const char *cmdname, char **argv) {
293 int pid, n, status;
294
295 switch (pid = fork()) {
296 case -1:
297 fprintf(stderr, "%s: no more processes\n", progname);
298 return 100;
299 case 0:
300 // TTimo removing hardcoded paths, searching in $PATH
301 execvp(cmdname, argv);
302 fprintf(stderr, "%s: ", progname);
303 perror(cmdname);
304 fflush(stdout);
305 exit(100);
306 }
307 while ((n = wait(&status)) != pid && n != -1)
308 ;
309 if (n == -1)
310 status = -1;
311 if (status&0377) {
312 fprintf(stderr, "%s: fatal error in %s\n", progname, cmdname);
313 status |= 0400;
314 }
315 return (status>>8)&0377;
316}

319/* callsys - execute the command described by av[0...], return status */
320static int callsys(char **av) {
321 int i, status = 0;
322 static char **argv;
323 static int argc;
324 char *executable;
325
326 for (i = 0; av[i] != NULL; i++)
327 ;
328 if (i + 1 > argc) {
329 argc = i + 1;
330 if (argv == NULL)
331 argv = malloc(argc*sizeof *argv);
332 else
333 argv = realloc(argv, argc*sizeof *argv);
334 assert(argv);
335 }
336 for (i = 0; status == 0 && av[i] != NULL; ) {
337 int j = 0;
338 char *s = NULL;
339 for ( ; av[i] != NULL && (s = strchr(av[i], '\n')) == NULL; i++)
340 argv[j++] = av[i];
341 if (s != NULL) {
342 if (s > av[i])
343 argv[j++] = stringf("%.*s", s - av[i], av[i]);
344 if (s[1] != '\0')
345 av[i] = s + 1;
346 else
347 i++;
348 }
349 argv[j] = NULL;
350 executable = strsave( argv[0] );
351 argv[0] = stringf( "\"%s\"", argv[0] );
352 if (verbose > 0) {
353 int k;
354 fprintf(stderr, "%s", argv[0]);
355 for (k = 1; argv[k] != NULL; k++)
356 fprintf(stderr, " %s", argv[k]);
357 fprintf(stderr, "\n");
358 }
359 if (verbose < 2)
360 status = spawn(executable, argv);
361 if (status == -1) {
362 fprintf(stderr, "%s: ", progname);
363 perror(argv[0]);
364 }
365 }
366 return status;
367}

385/* compose - compose cmd into av substituting a, b, c for $1, $2, $3, resp. */
386static void compose(char *cmd[], List a, List b, List c) {
387 int i, j;
388 List lists[3];
389
390 lists[0] = a;
391 lists[1] = b;
392 lists[2] = c;
393 for (i = j = 0; cmd[i]; i++) {
394 char *s = strchr(cmd[i], '$');
395 if (s && isdigit(s[1])) {
396 int k = s[1] - '0';
397 assert(k >=1 && k <= 3);
398 if ((b = lists[k-1])) {
399 b = b->link;
400 av[j] = alloc(strlen(cmd[i]) + strlen(b->str) - 1);
401 strncpy(av[j], cmd[i], s - cmd[i]);
402 av[j][s-cmd[i]] = '\0';
403 strcat(av[j], b->str);
404 strcat(av[j++], s + 2);
405 while (b != lists[k-1]) {
406 b = b->link;
407 assert(j < ac);
408 av[j++] = b->str;
409 };
410 }
411 } else if (*cmd[i]) {
412 assert(j < ac);
413 av[j++] = cmd[i];
414 }
415 }
416 av[j] = NULL;
417}

427/* exists - if `name' readable return its path name or return null */
428static char *exists(char *name) {
429 List b;
430
431 if ( (name[0] == '/' || name[0] == '\\' || name[2] == ':')
432 && access(name, 4) == 0)
433 return name;
434 if (!(name[0] == '/' || name[0] == '\\' || name[2] == ':')
435 && (b = lccinputs))
436 do {
437 b = b->link;
438 if (b->str[0]) {
439 char buf[1024];
440 sprintf(buf, "%s/%s", b->str, name);
441 if (access(buf, 4) == 0)
442 return strsave(buf);
443 } else if (access(name, 4) == 0)
444 return name;
445 } while (b != lccinputs);
446 if (verbose > 1)
447 return name;
448 return 0;
449}

464/* filename - process file name argument `name', return status */
465static int filename(char *name, char *base) {
466 int status = 0;
467 static char *stemp, *itemp;
468
469 if (base == 0)
470 base = basename(name);
471 switch (suffix(name, suffixes, 4)) {
472 case 0: /* C source files */
473 compose(cpp, plist, append(name, 0), 0);
474 if (Eflag) {
475 status = callsys(av);
476 break;
477 }
478 if (itemp == NULL)
479 itemp = tempname(first(suffixes[1]));
480 compose(cpp, plist, append(name, 0), append(itemp, 0));
481 status = callsys(av);
482 if (status == 0)
483 return filename(itemp, base);
484 break;
485 case 1: /* preprocessed source files */
486 if (Eflag)
487 break;
488 if (Sflag)
489 status = compile(name, outfile ? outfile : concat(base, first(suffixes[2])));
490 else if ((status = compile(name, stemp?stemp:(stemp=tempname(first(suffixes[2]))))) == 0)
491 return filename(stemp, base);
492 break;
493 case 2: /* assembly language files */
494 if (Eflag)
495 break;
496 if (!Sflag) {
497 char *ofile;
498 if (cflag && outfile)
499 ofile = outfile;
500 else if (cflag)
501 ofile = concat(base, first(suffixes[3]));
502 else
503 ofile = tempname(first(suffixes[3]));
504 compose(as, alist, append(name, 0), append(ofile, 0));
505 status = callsys(av);
506 if (!find(ofile, llist[1]))
507 llist[1] = append(ofile, llist[1]);
508 }
509 break;
510 case 3: /* object files */
511 if (!find(name, llist[1]))
512 llist[1] = append(name, llist[1]);
513 break;
514 default:
515 if (Eflag) {
516 compose(cpp, plist, append(name, 0), 0);
517 status = callsys(av);
518 }
519 llist[1] = append(name, llist[1]);
520 break;
521 }
522 if (status)
523 errcnt++;
524 return status;
525}

527/* find - find 1st occurrence of str in list, return list node or 0 */
528static List find(char *str, List list) {
529 List b;
530
531 if ((b = list))
532 do {
533 if (strcmp(str, b->str) == 0)
534 return b;
535 } while ((b = b->link) != list);
536 return 0;
537}

593/* initinputs - if LCCINPUTS or include is defined, use them to initialize various lists */
594static void initinputs(void) {
595 char *s = getenv("LCCINPUTS");
596 List b;
597
598 if (s == 0 || (s = inputs)[0] == 0)
599 s = ".";
600 if (s) {
601 lccinputs = path2list(s);
602 if ((b = lccinputs))
603 do {
604 b = b->link;
605 if (strcmp(b->str, ".") != 0) {
606 ilist = append(concat("-I", b->str), ilist);
607 if (strstr(com[1], "win32") == NULL)
608 llist[0] = append(concat("-L", b->str), llist[0]);
609 } else
610 b->str = "";
611 } while (b != lccinputs);
612 }
613}

621/* opt - process option in arg */
622static void opt(char *arg) {
623 switch (arg[1]) { /* multi-character options */
624 case 'W': /* -Wxarg */
625 if (arg[2] && arg[3])
626 switch (arg[2]) {
627 case 'o':
628 if (option(&arg[3]))
629 return;
630 break;
631 case 'p':
632 plist = append(&arg[3], plist);
633 return;
634 case 'f':
635 if (strcmp(&arg[3], "-C") || option("-b")) {
636 clist = append(&arg[3], clist);
637 return;
638 }
639 break; /* and fall thru */
640 case 'a':
641 alist = append(&arg[3], alist);
642 return;
643 case 'l':
644 llist[0] = append(&arg[3], llist[0]);
645 return;
646 }
647 fprintf(stderr, "%s: %s ignored\n", progname, arg);
648 return;
649 case 'd': /* -dn */
650 arg[1] = 's';
651 clist = append(arg, clist);
652 return;
653 case 't': /* -t -tname -tempdir=dir */
654 if (strncmp(arg, "-tempdir=", 9) == 0)
655 tempdir = arg + 9;
656 else
657 clist = append(arg, clist);
658 return;
659 case 'p': /* -p -pg */
660 if (option(arg))
661 clist = append(arg, clist);
662 else
663 fprintf(stderr, "%s: %s ignored\n", progname, arg);
664 return;
665 case 'D': /* -Dname -Dname=def */
666 case 'U': /* -Uname */
667 case 'I': /* -Idir */
668 plist = append(arg, plist);
669 return;
670 case 'B': /* -Bdir -Bstatic -Bdynamic */
671#ifdef sparc
672 if (strcmp(arg, "-Bstatic") == 0 || strcmp(arg, "-Bdynamic") == 0)
673 llist[1] = append(arg, llist[1]);
674 else
675#endif
676 {
677 static char *path;
678 if (path)
679 error("-B overwrites earlier option", 0);
680 path = arg + 2;
681 if (strstr(com[1], "win32") != NULL)
682 com[0] = concat(replace(path, '/', '\\'), concat("rcc", first(suffixes[4])));
683 else
684 com[0] = concat(path, "rcc");
685 if (path[0] == 0)
686 error("missing directory in -B option", 0);
687 }
688 return;
689 case 'h':
690 if (strcmp(arg, "-help") == 0) {
691 static int printed = 0;
692 case '?':
693 if (!printed)
694 help();
695 printed = 1;
696 return;
697 }
698#ifdef linux
699 case 's':
700 if (strcmp(arg,"-static") == 0) {
701 if (!option(arg))
702 fprintf(stderr, "%s: %s ignored\n", progname, arg);
703 return;
704 }
705#endif
706 }
707 if (arg[2] == 0)
708 switch (arg[1]) { /* single-character options */
709 case 'S':
710 Sflag++;
711 return;
712 case 'O':
713 fprintf(stderr, "%s: %s ignored\n", progname, arg);
714 return;
715 case 'A': case 'n': case 'w': case 'P':
716 clist = append(arg, clist);
717 return;
718 case 'g': case 'b':
719 if (option(arg))
720 clist = append(arg[1] == 'g' ? "-g2" : arg, clist);
721 else
722 fprintf(stderr, "%s: %s ignored\n", progname, arg);
723 return;
724 case 'G':
725 if (option(arg)) {
726 clist = append("-g3", clist);
727 llist[0] = append("-N", llist[0]);
728 } else
729 fprintf(stderr, "%s: %s ignored\n", progname, arg);
730 return;
731 case 'E':
732 Eflag++;
733 return;
734 case 'c':
735 cflag++;
736 return;
737 case 'N':
738 if (strcmp(basename(cpp[0]), "gcc-cpp") == 0)
739 plist = append("-nostdinc", plist);
740 include[0] = 0;
741 ilist = 0;
742 return;
743 case 'v':
744 if (verbose++ == 0) {
745 if (strcmp(basename(cpp[0]), "gcc-cpp") == 0)
746 plist = append(arg, plist);
747 clist = append(arg, clist);
748 fprintf(stderr, "%s %s\n", progname, rcsid);
749 }
750 return;
751 }
752 if (cflag || Sflag || Eflag)
753 fprintf(stderr, "%s: %s ignored\n", progname, arg);
754 else

758/* path2list - convert a colon- or semicolon-separated list to a list */
759static List path2list(const char *path) {
760 List list = NULL;
761 char sep = ':';
762
763 if (path == NULL)
764 return NULL;
765 if (strchr(path, ';'))
766 sep = ';';
767 while (*path) {
768 char *p, buf[512];
769 if ((p = strchr(path, sep))) {
770 assert(p - path < sizeof buf);
771 strncpy(buf, path, p - path);
772 buf[p-path] = '\0';
773 } else {
774 assert(strlen(path) < sizeof buf);
775 strcpy(buf, path);
776 }
777 if (!find(buf, list))
778 list = append(strsave(buf), list);
779 if (p == 0)
780 break;
781 path = p + 1;
782 }
783 return list;
784}

828/* suffix - if one of tails[0..n-1] holds a proper suffix of name, return its index */
829int suffix(char *name, char *tails[], int n) {
830 int i, len = strlen(name);
831
832 for (i = 0; i < n; i++) {
833 char *s = tails[i], *t;
834 for ( ; (t = strchr(s, ';')); s = t + 1) {
835 int m = t - s;
836 if (len > m && strncmp(&name[len-m], s, m) == 0)
837 return i;
838 }
839 if (*s) {
840 int m = strlen(s);
841 if (len > m && strncmp(&name[len-m], s, m) == 0)
842 return i;
843 }
844 }
845 return -1;
846}

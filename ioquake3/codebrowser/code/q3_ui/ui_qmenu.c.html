105/*
106=================
107Text_Draw
108=================
109*/
110static void Text_Draw( menutext_s *t )
111{
112 int x;
113 int y;
114 char buff[512];
115 float* color;
116
117 x = t->generic.x;
118 y = t->generic.y;
119
120 buff[0] = '\0';
121
122 // possible label
123 if (t->generic.name)
124 strcpy(buff,t->generic.name);
125
126 // possible value
127 if (t->string)
128 strcat(buff,t->string);
129
130 if (t->generic.flags & QMF_GRAYED)
131 color = text_color_disabled;
132 else
133 color = t->color;
134
135 UI_DrawString( x, y, buff, t->style, color );
136}

170/*
171=================
172PText_Init
173=================
174*/
175static void PText_Init( menutext_s *t )
176{
177 int x;
178 int y;
179 int w;
180 int h;
181 float sizeScale;
182
183 sizeScale = UI_ProportionalSizeScale( t->style );
184
185 x = t->generic.x;
186 y = t->generic.y;
187 w = UI_ProportionalStringWidth( t->string ) * sizeScale;
188 h = PROP_HEIGHT * sizeScale;
189
190 if( t->generic.flags & QMF_RIGHT_JUSTIFY ) {
191 x -= w;
192 }
193 else if( t->generic.flags & QMF_CENTER_JUSTIFY ) {
194 x -= w / 2;
195 }
196
197 t->generic.left = x - PROP_GAP_WIDTH * sizeScale;
198 t->generic.right = x + w + PROP_GAP_WIDTH * sizeScale;
199 t->generic.top = y;
200 t->generic.bottom = y + h;
201}

237=================
238Bitmap_Init
239=================
240*/
241void Bitmap_Init( menubitmap_s *b )
242{
243 int x;
244 int y;
245 int w;
246 int h;
247
248 x = b->generic.x;
249 y = b->generic.y;
250 w = b->width;
251 h = b->height;
252 if( w < 0 ) {
253 w = -w;
254 }
255 if( h < 0 ) {
256 h = -h;
257 }
258
259 if (b->generic.flags & QMF_RIGHT_JUSTIFY)
260 {
261 x = x - w;
262 }
263 else if (b->generic.flags & QMF_CENTER_JUSTIFY)
264 {
265 x = x - w/2;
266 }
267
268 b->generic.left = x;
269 b->generic.right = x + w;
270 b->generic.top = y;
271 b->generic.bottom = y + h;
272
273 b->shader = 0;
274 b->focusshader = 0;
275}

277/*
278=================
279Bitmap_Draw
280=================
281*/
282void Bitmap_Draw( menubitmap_s *b )
283{
284 float x;
285 float y;
286 float w;
287 float h;
288 vec4_t tempcolor;
289 float* color;
290
291 x = b->generic.x;
292 y = b->generic.y;
293 w = b->width;
294 h = b->height;
295
296 if (b->generic.flags & QMF_RIGHT_JUSTIFY)
297 {
298 x = x - w;
299 }
300 else if (b->generic.flags & QMF_CENTER_JUSTIFY)
301 {
302 x = x - w/2;
303 }
304
305 // used to refresh shader
306 if (b->generic.name && !b->shader)
307 {
308 b->shader = trap_R_RegisterShaderNoMip( b->generic.name );
309 if (!b->shader && b->errorpic)
310 b->shader = trap_R_RegisterShaderNoMip( b->errorpic );
311 }
312
313 if (b->focuspic && !b->focusshader)
314 b->focusshader = trap_R_RegisterShaderNoMip( b->focuspic );
315
316 if (b->generic.flags & QMF_GRAYED)
317 {
318 if (b->shader)
319 {
320 trap_R_SetColor( colorMdGrey );
321 UI_DrawHandlePic( x, y, w, h, b->shader );
322 trap_R_SetColor( NULL );
323 }
324 }
325 else
326 {
327 if (b->shader)
328 UI_DrawHandlePic( x, y, w, h, b->shader );
329
330 if ( ( (b->generic.flags & QMF_PULSE)
331 || (b->generic.flags & QMF_PULSEIFFOCUS) )
332 && (Menu_ItemAtCursor( b->generic.parent ) == b))
333 {
334 if (b->focuscolor)
335 {
336 tempcolor[0] = b->focuscolor[0];
337 tempcolor[1] = b->focuscolor[1];
338 tempcolor[2] = b->focuscolor[2];
339 color = tempcolor;
340 }
341 else
342 color = pulse_color;
343 color[3] = 0.5+0.5*sin(uis.realtime/PULSE_DIVISOR);
344
345 trap_R_SetColor( color );
346 UI_DrawHandlePic( x, y, w, h, b->focusshader );
347 trap_R_SetColor( NULL );
348 }
349 else if ((b->generic.flags & QMF_HIGHLIGHT) || ((b->generic.flags & QMF_HIGHLIGHT_IF_FOCUS) && (Menu_ItemAtCursor( b->generic.parent ) == b)))
350 {
351 if (b->focuscolor)
352 {
353 trap_R_SetColor( b->focuscolor );
354 UI_DrawHandlePic( x, y, w, h, b->focusshader );
355 trap_R_SetColor( NULL );
356 }
357 else
358 UI_DrawHandlePic( x, y, w, h, b->focusshader );
359 }
360 }
361}

385/*
386=================
387Action_Draw
388=================
389*/
390static void Action_Draw( menuaction_s *a )
391{
392 int x, y;
393 int style;
394 float* color;
395
396 style = 0;
397 color = menu_text_color;
398 if ( a->generic.flags & QMF_GRAYED )
399 {
400 color = text_color_disabled;
401 }
402 else if (( a->generic.flags & QMF_PULSEIFFOCUS ) && ( a->generic.parent->cursor == a->generic.menuPosition ))
403 {
404 color = text_color_highlight;
405 style = UI_PULSE;
406 }
407 else if (( a->generic.flags & QMF_HIGHLIGHT_IF_FOCUS ) && ( a->generic.parent->cursor == a->generic.menuPosition ))
408 {
409 color = text_color_highlight;
410 }
411 else if ( a->generic.flags & QMF_BLINK )
412 {
413 style = UI_BLINK;
414 color = text_color_highlight;
415 }
416
417 x = a->generic.x;
418 y = a->generic.y;
419
420 UI_DrawString( x, y, a->generic.name, UI_LEFT|style, color );
421
422 if ( a->generic.parent->cursor == a->generic.menuPosition )
423 {
424 // draw cursor
425 UI_DrawChar( x - BIGCHAR_WIDTH, y, 13, UI_LEFT|UI_BLINK, color);
426 }
427}

450/*
451=================
452RadioButton_Key
453=================
454*/
455static sfxHandle_t RadioButton_Key( menuradiobutton_s *rb, int key )
456{
457 switch (key)
458 {
459 case K_MOUSE1:
460 if (!(rb->generic.flags & QMF_HASMOUSEFOCUS))
461 break;
462
463 case K_JOY1:
464 case K_JOY2:
465 case K_JOY3:
466 case K_JOY4:
467 case K_ENTER:
468 case K_KP_ENTER:
469 case K_KP_LEFTARROW:
470 case K_LEFTARROW:
471 case K_KP_RIGHTARROW:
472 case K_RIGHTARROW:
473 rb->curvalue = !rb->curvalue;
474 if ( rb->generic.callback )
475 rb->generic.callback( rb, QM_ACTIVATED );
476
477 return (menu_move_sound);
478 }
479
480 // key not handled
481 return 0;
482}

561/*
562=================
563Slider_Key
564=================
565*/
566static sfxHandle_t Slider_Key( menuslider_s *s, int key )
567{
568 sfxHandle_t sound;
569 int x;
570 int oldvalue;
571
572 switch (key)
573 {
574 case K_MOUSE1:
575 x = uis.cursorx - s->generic.x - 2*SMALLCHAR_WIDTH;
576 oldvalue = s->curvalue;
577 s->curvalue = (x/(float)(SLIDER_RANGE*SMALLCHAR_WIDTH)) * (s->maxvalue-s->minvalue) + s->minvalue;
578
579 if (s->curvalue < s->minvalue)
580 s->curvalue = s->minvalue;
581 else if (s->curvalue > s->maxvalue)
582 s->curvalue = s->maxvalue;
583 if (s->curvalue != oldvalue)
584 sound = menu_move_sound;
585 else
586 sound = 0;
587 break;
588
589 case K_KP_LEFTARROW:
590 case K_LEFTARROW:
591 if (s->curvalue > s->minvalue)
592 {
593 s->curvalue--;
594 sound = menu_move_sound;
595 }
596 else
597 sound = menu_buzz_sound;
598 break;
599
600 case K_KP_RIGHTARROW:
601 case K_RIGHTARROW:
602 if (s->curvalue < s->maxvalue)
603 {
604 s->curvalue++;
605 sound = menu_move_sound;
606 }
607 else
608 sound = menu_buzz_sound;
609 break;
610
611 default:
612 // key not handled
613 sound = 0;
614 break;
615 }
616
617 if ( sound && s->generic.callback )
618 s->generic.callback( s, QM_ACTIVATED );
619
620 return (sound);
621}

757/*
758=================
759SpinControl_Init
760=================
761*/
762static void SpinControl_Init( menulist_s *s ) {
763 int len;
764 int l;
765 const char* str;
766
767 if (s->generic.name)
768 len = strlen(s->generic.name) * SMALLCHAR_WIDTH;
769 else
770 len = 0;
771
772 s->generic.left = s->generic.x - SMALLCHAR_WIDTH - len;
773
774 len = s->numitems = 0;
775 while ( (str = s->itemnames[s->numitems]) != 0 )
776 {
777 l = strlen(str);
778 if (l > len)
779 len = l;
780
781 s->numitems++;
782 }
783
784 s->generic.top = s->generic.y;
785 s->generic.right = s->generic.x + (len+1)*SMALLCHAR_WIDTH;
786 s->generic.bottom = s->generic.y + SMALLCHAR_HEIGHT;
787}

789/*
790=================
791SpinControl_Key
792=================
793*/
794static sfxHandle_t SpinControl_Key( menulist_s *s, int key )
795{
796 sfxHandle_t sound;
797
798 sound = 0;
799 switch (key)
800 {
801 case K_KP_RIGHTARROW:
802 case K_RIGHTARROW:
803 case K_MOUSE1:
804 s->curvalue++;
805 if (s->curvalue >= s->numitems)
806 s->curvalue = 0;
807 sound = menu_move_sound;
808 break;
809
810 case K_KP_LEFTARROW:
811 case K_LEFTARROW:
812 s->curvalue--;
813 if (s->curvalue < 0)
814 s->curvalue = s->numitems-1;
815 sound = menu_move_sound;
816 break;
817 }
818
819 if ( sound && s->generic.callback )
820 s->generic.callback( s, QM_ACTIVATED );
821
822 return (sound);
823}

903/*
904=================
905ScrollList_Key
906=================
907*/
908sfxHandle_t ScrollList_Key( menulist_s *l, int key )
909{
910 int x;
911 int y;
912 int w;
913 int i;
914 int j;
915 int c;
916 int cursorx;
917 int cursory;
918 int column;
919 int index;
920
921 switch (key)
922 {
923 case K_MOUSE1:
924 if (l->generic.flags & QMF_HASMOUSEFOCUS)
925 {
926 // check scroll region
927 x = l->generic.x;
928 y = l->generic.y;
929 w = ( (l->width + l->seperation) * l->columns - l->seperation) * SMALLCHAR_WIDTH;
930 if( l->generic.flags & QMF_CENTER_JUSTIFY ) {
931 x -= w / 2;
932 }
933 if (UI_CursorInRect( x, y, w, l->height*SMALLCHAR_HEIGHT ))
934 {
935 cursorx = (uis.cursorx - x)/SMALLCHAR_WIDTH;
936 column = cursorx / (l->width + l->seperation);
937 cursory = (uis.cursory - y)/SMALLCHAR_HEIGHT;
938 index = column * l->height + cursory;
939 if (l->top + index < l->numitems)
940 {
941 l->oldvalue = l->curvalue;
942 l->curvalue = l->top + index;
943
944 if (l->oldvalue != l->curvalue && l->generic.callback)
945 {
946 l->generic.callback( l, QM_GOTFOCUS );
947 return (menu_move_sound);
948 }
949 }
950 }
951
952 // absorbed, silent sound effect
953 return (menu_null_sound);
954 }
955 break;
956
957 case K_KP_HOME:
958 case K_HOME:
959 l->oldvalue = l->curvalue;
960 l->curvalue = 0;
961 l->top = 0;
962
963 if (l->oldvalue != l->curvalue && l->generic.callback)
964 {
965 l->generic.callback( l, QM_GOTFOCUS );
966 return (menu_move_sound);
967 }
968 return (menu_buzz_sound);
969
970 case K_KP_END:
971 case K_END:
972 l->oldvalue = l->curvalue;
973 l->curvalue = l->numitems-1;
974 if( l->columns > 1 ) {
975 c = (l->curvalue / l->height + 1) * l->height;
976 l->top = c - (l->columns * l->height);
977 }
978 else {
979 l->top = l->curvalue - (l->height - 1);
980 }
981 if (l->top < 0)
982 l->top = 0;
983
984 if (l->oldvalue != l->curvalue && l->generic.callback)
985 {
986 l->generic.callback( l, QM_GOTFOCUS );
987 return (menu_move_sound);
988 }
989 return (menu_buzz_sound);
990
991 case K_PGUP:
992 case K_KP_PGUP:
993 if( l->columns > 1 ) {
994 return menu_null_sound;
995 }
996
997 if (l->curvalue > 0)
998 {
999 l->oldvalue = l->curvalue;
1000 l->curvalue -= l->height-1;
1001 if (l->curvalue < 0)
1002 l->curvalue = 0;
1003 l->top = l->curvalue;
1004 if (l->top < 0)
1005 l->top = 0;
1006
1007 if (l->generic.callback)
1008 l->generic.callback( l, QM_GOTFOCUS );
1009
1010 return (menu_move_sound);
1011 }
1012 return (menu_buzz_sound);
1013
1014 case K_PGDN:
1015 case K_KP_PGDN:
1016 if( l->columns > 1 ) {
1017 return menu_null_sound;
1018 }
1019
1020 if (l->curvalue < l->numitems-1)
1021 {
1022 l->oldvalue = l->curvalue;
1023 l->curvalue += l->height-1;
1024 if (l->curvalue > l->numitems-1)
1025 l->curvalue = l->numitems-1;
1026 l->top = l->curvalue - (l->height-1);
1027 if (l->top < 0)
1028 l->top = 0;
1029
1030 if (l->generic.callback)
1031 l->generic.callback( l, QM_GOTFOCUS );
1032
1033 return (menu_move_sound);
1034 }
1035 return (menu_buzz_sound);
1036
1037 case K_MWHEELUP:
1038 if( l->columns > 1 ) {
1039 return menu_null_sound;
1040 }
1041
1042 if (l->top > 0)
1043 {
1044 // if scrolling 3 lines would replace over half of the
1045 // displayed items, only scroll 1 item at a time.
1046 int scroll = l->height < 6 ? 1 : 3;
1047 l->top -= scroll;
1048 if (l->top < 0)
1049 l->top = 0;
1050
1051 if (l->generic.callback)
1052 l->generic.callback( l, QM_GOTFOCUS );
1053
1054 // make scrolling silent
1055 return (menu_null_sound);
1056 }
1057 return (menu_buzz_sound);
1058
1059 case K_MWHEELDOWN:
1060 if( l->columns > 1 ) {
1061 return menu_null_sound;
1062 }
1063
1064 if (l->top < l->numitems-l->height)
1065 {
1066 // if scrolling 3 items would replace over half of the
1067 // displayed items, only scroll 1 item at a time.
1068 int scroll = l->height < 6 ? 1 : 3;
1069 l->top += scroll;
1070 if (l->top > l->numitems-l->height)
1071 l->top = l->numitems-l->height;
1072
1073 if (l->generic.callback)
1074 l->generic.callback( l, QM_GOTFOCUS );
1075
1076 // make scrolling silent
1077 return (menu_null_sound);
1078 }
1079 return (menu_buzz_sound);
1080
1081 case K_KP_UPARROW:
1082 case K_UPARROW:
1083 if( l->curvalue == 0 ) {
1084 return menu_buzz_sound;
1085 }
1086
1087 l->oldvalue = l->curvalue;
1088 l->curvalue--;
1089
1090 if( l->curvalue < l->top ) {
1091 if( l->columns == 1 ) {
1092 l->top--;
1093 }
1094 else {
1095 l->top -= l->height;
1096 }
1097 }
1098
1099 if( l->generic.callback ) {
1100 l->generic.callback( l, QM_GOTFOCUS );
1101 }
1102
1103 return (menu_move_sound);
1104
1105 case K_KP_DOWNARROW:
1106 case K_DOWNARROW:
1107 if( l->curvalue == l->numitems - 1 ) {
1108 return menu_buzz_sound;
1109 }
1110
1111 l->oldvalue = l->curvalue;
1112 l->curvalue++;
1113
1114 if( l->curvalue >= l->top + l->columns * l->height ) {
1115 if( l->columns == 1 ) {
1116 l->top++;
1117 }
1118 else {
1119 l->top += l->height;
1120 }
1121 }
1122
1123 if( l->generic.callback ) {
1124 l->generic.callback( l, QM_GOTFOCUS );
1125 }
1126
1127 return menu_move_sound;
1128
1129 case K_KP_LEFTARROW:
1130 case K_LEFTARROW:
1131 if( l->columns == 1 ) {
1132 return menu_null_sound;
1133 }
1134
1135 if( l->curvalue < l->height ) {
1136 return menu_buzz_sound;
1137 }
1138
1139 l->oldvalue = l->curvalue;
1140 l->curvalue -= l->height;
1141
1142 if( l->curvalue < l->top ) {
1143 l->top -= l->height;
1144 }
1145
1146 if( l->generic.callback ) {
1147 l->generic.callback( l, QM_GOTFOCUS );
1148 }
1149
1150 return menu_move_sound;
1151
1152 case K_KP_RIGHTARROW:
1153 case K_RIGHTARROW:
1154 if( l->columns == 1 ) {
1155 return menu_null_sound;
1156 }
1157
1158 c = l->curvalue + l->height;
1159
1160 if( c >= l->numitems ) {
1161 return menu_buzz_sound;
1162 }
1163
1164 l->oldvalue = l->curvalue;
1165 l->curvalue = c;
1166
1167 if( l->curvalue > l->top + l->columns * l->height - 1 ) {
1168 l->top += l->height;
1169 }
1170
1171 if( l->generic.callback ) {
1172 l->generic.callback( l, QM_GOTFOCUS );
1173 }
1174
1175 return menu_move_sound;
1176 }
1177
1178 // cycle look for ascii key inside list items
1179 if ( !Q_isprint( key ) )
1180 return (0);
1181
1182 // force to lower for case insensitive compare
1183 if ( Q_isupper( key ) )
1184 {
1185 key -= 'A' - 'a';
1186 }
1187
1188 // iterate list items
1189 for (i=1; i<=l->numitems; i++)
1190 {
1191 j = (l->curvalue + i) % l->numitems;
1192 c = l->itemnames[j][0];
1193 if ( Q_isupper( c ) )
1194 {
1195 c -= 'A' - 'a';
1196 }
1197
1198 if (c == key)
1199 {
1200 // set current item, mimic windows listbox scroll behavior
1201 if (j < l->top)
1202 {
1203 // behind top most item, set this as new top
1204 l->top = j;
1205 }
1206 else if (j > l->top+l->height-1)
1207 {
1208 // past end of list box, do page down
1209 l->top = (j+1) - l->height;
1210 }
1211
1212 if (l->curvalue != j)
1213 {
1214 l->oldvalue = l->curvalue;
1215 l->curvalue = j;
1216 if (l->generic.callback)
1217 l->generic.callback( l, QM_GOTFOCUS );
1218 return ( menu_move_sound );
1219 }
1220
1221 return (menu_buzz_sound);
1222 }
1223 }
1224
1225 return (menu_buzz_sound);
1226}

1428/*
1429** Menu_AdjustCursor
1430**
1431** This function takes the given menu, the direction, and attempts
1432** to adjust the menu's cursor so that it's at the next available
1433** slot.
1434*/
1435void Menu_AdjustCursor( menuframework_s *m, int dir ) {
1436 menucommon_s *item = NULL;
1437 qboolean wrapped = qfalse;
1438
1439wrap:
1440 while ( m->cursor >= 0 && m->cursor < m->nitems ) {
1441 item = ( menucommon_s * ) m->items[m->cursor];
1442 if (( item->flags & (QMF_GRAYED|QMF_MOUSEONLY|QMF_INACTIVE) ) ) {
1443 m->cursor += dir;
1444 }
1445 else {
1446 break;
1447 }
1448 }
1449
1450 if ( dir == 1 ) {
1451 if ( m->cursor >= m->nitems ) {
1452 if ( m->wrapAround ) {
1453 if ( wrapped ) {
1454 m->cursor = m->cursor_prev;
1455 return;
1456 }
1457 m->cursor = 0;
1458 wrapped = qtrue;
1459 goto wrap;
1460 }
1461 m->cursor = m->cursor_prev;
1462 }
1463 }
1464 else {
1465 if ( m->cursor < 0 ) {
1466 if ( m->wrapAround ) {
1467 if ( wrapped ) {
1468 m->cursor = m->cursor_prev;
1469 return;
1470 }
1471 m->cursor = m->nitems - 1;
1472 wrapped = qtrue;
1473 goto wrap;
1474 }
1475 m->cursor = m->cursor_prev;
1476 }
1477 }
1478}

1609/*
1610=================
1611Menu_DefaultKey
1612=================
1613*/
1614sfxHandle_t Menu_DefaultKey( menuframework_s *m, int key )
1615{
1616 sfxHandle_t sound = 0;
1617 menucommon_s *item;
1618 int cursor_prev;
1619
1620 // menu system keys
1621 switch ( key )
1622 {
1623 case K_MOUSE2:
1624 case K_ESCAPE:
1625 UI_PopMenu();
1626 return menu_out_sound;
1627 }
1628
1629 if (!m || !m->nitems)
1630 return 0;
1631
1632 // route key stimulus to widget
1633 item = Menu_ItemAtCursor( m );
1634 if (item && !(item->flags & (QMF_GRAYED|QMF_INACTIVE)))
1635 {
1636 switch (item->type)
1637 {
1638 case MTYPE_SPINCONTROL:
1639 sound = SpinControl_Key( (menulist_s*)item, key );
1640 break;
1641
1642 case MTYPE_RADIOBUTTON:
1643 sound = RadioButton_Key( (menuradiobutton_s*)item, key );
1644 break;
1645
1646 case MTYPE_SLIDER:
1647 sound = Slider_Key( (menuslider_s*)item, key );
1648 break;
1649
1650 case MTYPE_SCROLLLIST:
1651 sound = ScrollList_Key( (menulist_s*)item, key );
1652 break;
1653
1654 case MTYPE_FIELD:
1655 sound = MenuField_Key( (menufield_s*)item, &key );
1656 break;
1657 }
1658
1659 if (sound) {
1660 // key was handled
1661 return sound;
1662 }
1663 }
1664
1665 // default handling
1666 switch ( key )
1667 {
1668#ifndef NDEBUG
1669 case K_F11:
1670 uis.debug ^= 1;
1671 break;
1672
1673 case K_F12:

1675 break;
1676#endif
1677 case K_KP_UPARROW:
1678 case K_UPARROW:
1679 cursor_prev = m->cursor;
1680 m->cursor_prev = m->cursor;
1681 m->cursor--;
1682 Menu_AdjustCursor( m, -1 );
1683 if ( cursor_prev != m->cursor ) {
1684 Menu_CursorMoved( m );

1689 case K_TAB:
1690 case K_KP_DOWNARROW:

1694 m->cursor++;
1695 Menu_AdjustCursor( m, 1 );
1696 if ( cursor_prev != m->cursor ) {
1697 Menu_CursorMoved( m );

1699 }
1700 break;
1701
1702 case K_MOUSE1:
1703 case K_MOUSE3:
1704 if (item)
1705 if ((item->flags & QMF_HASMOUSEFOCUS) && !(item->flags & (QMF_GRAYED|QMF_INACTIVE)))
1706 return (Menu_ActivateItem( m, item ));
1707 break;
1708
1709 case K_JOY1:
1710 case K_JOY2:
1711 case K_JOY3:
1712 case K_JOY4:
1713 case K_AUX1:
1714 case K_AUX2:
1715 case K_AUX3:
1716 case K_AUX4:
1717 case K_AUX5:
1718 case K_AUX6:
1719 case K_AUX7:
1720 case K_AUX8:
1721 case K_AUX9:
1722 case K_AUX10:
1723 case K_AUX11:
1724 case K_AUX12:
1725 case K_AUX13:
1726 case K_AUX14:
1727 case K_AUX15:
1728 case K_AUX16:
1729 case K_KP_ENTER:
1730 case K_ENTER:
1731 if (item)

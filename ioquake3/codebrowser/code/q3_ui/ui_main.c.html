35/*
36================
37vmMain
38
39This is the only way control passes into the module.
40This must be the very first function compiled into the .qvm file
41================
42*/
43Q_EXPORT intptr_t vmMain( int command, int arg0, int arg1, int arg2, int arg3, int arg4, int arg5, int arg6, int arg7, int arg8, int arg9, int arg10, int arg11 ) {
44 switch ( command ) {
45 case UI_GETAPIVERSION:
46 return UI_API_VERSION;
47
48 case UI_INIT:
49 UI_Init();
50 return 0;
51
52 case UI_SHUTDOWN:
53 UI_Shutdown();
54 return 0;
55
56 case UI_KEY_EVENT:
57 UI_KeyEvent( arg0, arg1 );
58 return 0;
59
60 case UI_MOUSE_EVENT:
61 UI_MouseEvent( arg0, arg1 );
62 return 0;
63
64 case UI_REFRESH:
65 UI_Refresh( arg0 );
66 return 0;
67
68 case UI_IS_FULLSCREEN:
69 return UI_IsFullscreen();
70
71 case UI_SET_ACTIVE_MENU:
72 UI_SetActiveMenu( arg0 );
73 return 0;
74
75 case UI_CONSOLE_COMMAND:
76 return UI_ConsoleCommand(arg0);
77
78 case UI_DRAW_CONNECT_SCREEN:
79 UI_DrawConnectScreen( arg0 );
80 return 0;
81 case UI_HASUNIQUECDKEY: // mod authors need to observe this
82 return qtrue; // change this to qfalse for mods!
83 }
84
85 return -1;
86}

224/*
225=================
226UI_RegisterCvars
227=================
228*/
229void UI_RegisterCvars( void ) {
230 int i;
231 cvarTable_t *cv;
232
233 for ( i = 0, cv = cvarTable ; i < cvarTableSize ; i++, cv++ ) {
234 trap_Cvar_Register( cv->vmCvar, cv->cvarName, cv->defaultString, cv->cvarFlags );
235 }
236}
237
238/*
239=================
240UI_UpdateCvars
241=================
242*/
243void UI_UpdateCvars( void ) {
244 int i;
245 cvarTable_t *cv;
246
247 for ( i = 0, cv = cvarTable ; i < cvarTableSize ; i++, cv++ ) {
248 if ( !cv->vmCvar ) {
249 continue;
250 }
251
252 trap_Cvar_Update( cv->vmCvar );
253 }

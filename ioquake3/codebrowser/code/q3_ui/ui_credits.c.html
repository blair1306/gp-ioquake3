43/*
44===============
45UI_CreditMenu_Draw_ioq3
46===============
47*/
48static void UI_CreditMenu_Draw_ioq3( void ) {
49 int y;
50 int i;
51
52 // These are all people that have made commits to Subversion, and thus
53 // probably incomplete.
54 // (These are in alphabetical order, for the defense of everyone's egos.)
55 static const char *names[] = {
56 "Tim Angus",
57 "James Canete",
58 "Vincent Cojot",
59 "Ryan C. Gordon",
60 "Aaron Gyes",
61 "Zack Middleton",
62 "Ludwig Nussel",
63 "Julian Priestley",
64 "Scirocco Six",
65 "Thilo Schulz",
66 "Zachary J. Slater",
67 "Tony J. White",
68 "...and many, many others!", // keep this one last.
69 NULL
70 };
71
72 // Center text vertically on the screen
73 y = (SCREEN_HEIGHT - ARRAY_LEN(names) * (1.42 * PROP_HEIGHT * PROP_SMALL_SIZE_SCALE)) / 2;
74
75 UI_DrawProportionalString( 320, y, "ioquake3 contributors:", UI_CENTER|UI_SMALLFONT, color_white );
76 y += 1.42 * PROP_HEIGHT * PROP_SMALL_SIZE_SCALE;
77
78 for (i = 0; names[i]; i++) {
79 UI_DrawProportionalString( 320, y, names[i], UI_CENTER|UI_SMALLFONT, color_white );
80 y += 1.42 * PROP_HEIGHT * PROP_SMALL_SIZE_SCALE;
81 }
82
83 UI_DrawString( 320, 459, "http://www.ioquake3.org/", UI_CENTER|UI_SMALLFONT, color_red );
84}

87/*
88=================
89UI_CreditMenu_Key
90=================
91*/
92static sfxHandle_t UI_CreditMenu_Key( int key ) {
93 if( key & K_CHAR_FLAG ) {
94 return 0;
95 }
96
97 s_credits.frame++;
98 if (s_credits.frame == 1) {
99 s_credits.menu.draw = UI_CreditMenu_Draw_ioq3;
100 } else {
101 trap_Cmd_ExecuteText( EXEC_APPEND, "quit\n" );
102 }
103 return 0;
104}

167/*
168===============
169UI_CreditMenu
170===============
171*/
172void UI_CreditMenu( void ) {
173 memset( &s_credits, 0 ,sizeof(s_credits) );
174
175 s_credits.menu.draw = UI_CreditMenu_Draw;
176 s_credits.menu.key = UI_CreditMenu_Key;
177 s_credits.menu.fullscreen = qtrue;
178 UI_PushMenu ( &s_credits.menu );
179}

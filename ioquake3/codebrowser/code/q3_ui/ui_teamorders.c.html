119/*
120===============
121UI_TeamOrdersMenu_SetList
122===============
123*/
124static void UI_TeamOrdersMenu_SetList( int id ) {
125 switch( id ) {
126 default:
127 case ID_LIST_BOTS:
128 teamOrdersMenuInfo.list.generic.id = id;
129 teamOrdersMenuInfo.list.numitems = teamOrdersMenuInfo.numBots;
130 teamOrdersMenuInfo.list.itemnames = (const char **)teamOrdersMenuInfo.bots;
131 break;
132
133 case ID_LIST_CTF_ORDERS:
134 teamOrdersMenuInfo.list.generic.id = id;
135 teamOrdersMenuInfo.list.numitems = NUM_CTF_ORDERS;
136 teamOrdersMenuInfo.list.itemnames = ctfOrders;
137 break;
138
139 case ID_LIST_TEAM_ORDERS:
140 teamOrdersMenuInfo.list.generic.id = id;
141 teamOrdersMenuInfo.list.numitems = NUM_TEAM_ORDERS;
142 teamOrdersMenuInfo.list.itemnames = teamOrders;
143 break;
144 }
145
146 teamOrdersMenuInfo.list.generic.bottom = teamOrdersMenuInfo.list.generic.top + teamOrdersMenuInfo.list.numitems * PROP_HEIGHT;
147}

150/*
151=================
152UI_TeamOrdersMenu_Key
153=================
154*/
155sfxHandle_t UI_TeamOrdersMenu_Key( int key ) {
156 menulist_s *l;
157 int x;
158 int y;
159 int index;
160
161 l = (menulist_s *)Menu_ItemAtCursor( &teamOrdersMenuInfo.menu );
162 if( l != &teamOrdersMenuInfo.list ) {
163 return Menu_DefaultKey( &teamOrdersMenuInfo.menu, key );
164 }
165
166 switch( key ) {
167 case K_MOUSE1:
168 x = l->generic.left;
169 y = l->generic.top;
170 if( UI_CursorInRect( x, y, l->generic.right - x, l->generic.bottom - y ) ) {
171 index = (uis.cursory - y) / PROP_HEIGHT;
172 l->oldvalue = l->curvalue;
173 l->curvalue = index;
174
175 if( l->generic.callback ) {
176 l->generic.callback( l, QM_ACTIVATED );
177 return menu_move_sound;
178 }
179 }
180 return menu_null_sound;
181
182 case K_KP_UPARROW:
183 case K_UPARROW:
184 l->oldvalue = l->curvalue;
185
186 if( l->curvalue == 0 ) {
187 l->curvalue = l->numitems - 1;
188 }
189 else {
190 l->curvalue--;
191 }
192 return menu_move_sound;
193
194 case K_KP_DOWNARROW:
195 case K_DOWNARROW:
196 l->oldvalue = l->curvalue;
197
198 if( l->curvalue == l->numitems - 1 ) {
199 l->curvalue = 0;;
200 }
201 else {
202 l->curvalue++;
203 }
204 return menu_move_sound;
205 }
206
207 return Menu_DefaultKey( &teamOrdersMenuInfo.menu, key );
208}

211/*
212=================
213UI_TeamOrdersMenu_ListDraw
214=================
215*/
216static void UI_TeamOrdersMenu_ListDraw( void *self ) {
217 menulist_s *l;
218 int x;
219 int y;
220 int i;
221 float *color;
222 qboolean hasfocus;
223 int style;
224
225 l = (menulist_s *)self;
226
227 hasfocus = (l->generic.parent->cursor == l->generic.menuPosition);
228
229 x = 320;//l->generic.x;
230 y = l->generic.y;
231 for( i = 0; i < l->numitems; i++ ) {
232 style = UI_LEFT|UI_SMALLFONT|UI_CENTER;
233 if( i == l->curvalue ) {
234 color = color_yellow;
235 if( hasfocus ) {
236 style |= UI_PULSE;
237 }
238 }
239 else {
240 color = color_orange;
241 }
242
243 UI_DrawProportionalString( x, y, l->itemnames[i], style, color );
244 y += PROP_HEIGHT;
245 }
246}

249/*
250===============
251UI_TeamOrdersMenu_ListEvent
252===============
253*/
254static void UI_TeamOrdersMenu_ListEvent( void *ptr, int event ) {
255 int id;
256 int selection;
257 char message[256];
258
259 if (event != QM_ACTIVATED)
260 return;
261
262 id = ((menulist_s *)ptr)->generic.id;
263 selection = ((menulist_s *)ptr)->curvalue;
264
265 if( id == ID_LIST_BOTS ) {
266 teamOrdersMenuInfo.selectedBot = selection;
267 if( teamOrdersMenuInfo.gametype == GT_CTF ) {
268 UI_TeamOrdersMenu_SetList( ID_LIST_CTF_ORDERS );
269 }
270 else {
271 UI_TeamOrdersMenu_SetList( ID_LIST_TEAM_ORDERS );
272 }
273 return;
274 }
275
276 if( id == ID_LIST_CTF_ORDERS ) {
277 Com_sprintf( message, sizeof(message), ctfMessages[selection], teamOrdersMenuInfo.botNames[teamOrdersMenuInfo.selectedBot] );
278 }
279 else {
280 Com_sprintf( message, sizeof(message), teamMessages[selection], teamOrdersMenuInfo.botNames[teamOrdersMenuInfo.selectedBot] );
281 }
282
283 trap_Cmd_ExecuteText( EXEC_APPEND, va( "say_team \"%s\"\n", message ) );
284 UI_PopMenu();
285}

288/*
289===============
290UI_TeamOrdersMenu_BuildBotList
291===============
292*/
293static void UI_TeamOrdersMenu_BuildBotList( void ) {
294 uiClientState_t cs;
295 int numPlayers;
296 int isBot;
297 int n;
298 char playerTeam = '3';
299 char botTeam;
300 char info[MAX_INFO_STRING];
301
302 for( n = 0; n < 9; n++ ) {
303 teamOrdersMenuInfo.bots[n] = teamOrdersMenuInfo.botNames[n];
304 }
305
306 trap_GetClientState( &cs );
307
308 Q_strncpyz( teamOrdersMenuInfo.botNames[0], "Everyone", 16 );
309 teamOrdersMenuInfo.numBots = 1;
310
311 trap_GetConfigString( CS_SERVERINFO, info, sizeof(info) );
312 numPlayers = atoi( Info_ValueForKey( info, "sv_maxclients" ) );
313 teamOrdersMenuInfo.gametype = atoi( Info_ValueForKey( info, "g_gametype" ) );
314
315 for( n = 0; n < numPlayers && teamOrdersMenuInfo.numBots < 9; n++ ) {
316 trap_GetConfigString( CS_PLAYERS + n, info, MAX_INFO_STRING );
317
318 if( n == cs.clientNum ) {
319 playerTeam = *Info_ValueForKey( info, "t" );
320 continue;
321 }
322
323 isBot = atoi( Info_ValueForKey( info, "skill" ) );
324 if( !isBot ) {
325 continue;
326 }
327
328 botTeam = *Info_ValueForKey( info, "t" );
329 if( botTeam != playerTeam ) {
330 continue;
331 }
332
333 Q_strncpyz( teamOrdersMenuInfo.botNames[teamOrdersMenuInfo.numBots], Info_ValueForKey( info, "n" ), 16 );
334 Q_CleanStr( teamOrdersMenuInfo.botNames[teamOrdersMenuInfo.numBots] );
335 teamOrdersMenuInfo.numBots++;
336 }
337}

345static void UI_TeamOrdersMenu_Init( void ) {
346 UI_TeamOrdersMenu_Cache();
347
348 memset( &teamOrdersMenuInfo, 0, sizeof(teamOrdersMenuInfo) );
349 teamOrdersMenuInfo.menu.fullscreen = qfalse;
350 teamOrdersMenuInfo.menu.key = UI_TeamOrdersMenu_Key;
351
352 UI_TeamOrdersMenu_BuildBotList();
353
354 teamOrdersMenuInfo.banner.generic.type = MTYPE_BTEXT;
355 teamOrdersMenuInfo.banner.generic.x = 320;
356 teamOrdersMenuInfo.banner.generic.y = 16;
357 teamOrdersMenuInfo.banner.string = "TEAM ORDERS";
358 teamOrdersMenuInfo.banner.color = color_white;
359 teamOrdersMenuInfo.banner.style = UI_CENTER;
360
361 teamOrdersMenuInfo.frame.generic.type = MTYPE_BITMAP;
362 teamOrdersMenuInfo.frame.generic.flags = QMF_INACTIVE;
363 teamOrdersMenuInfo.frame.generic.name = ART_FRAME;
364 teamOrdersMenuInfo.frame.generic.x = 320-233;
365 teamOrdersMenuInfo.frame.generic.y = 240-166;
366 teamOrdersMenuInfo.frame.width = 466;
367 teamOrdersMenuInfo.frame.height = 332;
368
369 teamOrdersMenuInfo.list.generic.type = MTYPE_SCROLLLIST;
370 teamOrdersMenuInfo.list.generic.flags = QMF_PULSEIFFOCUS;
371 teamOrdersMenuInfo.list.generic.ownerdraw = UI_TeamOrdersMenu_ListDraw;
372 teamOrdersMenuInfo.list.generic.callback = UI_TeamOrdersMenu_ListEvent;
373 teamOrdersMenuInfo.list.generic.x = 320-64;
374 teamOrdersMenuInfo.list.generic.y = 120;
375
376 teamOrdersMenuInfo.back.generic.type = MTYPE_BITMAP;
377 teamOrdersMenuInfo.back.generic.name = ART_BACK0;
378 teamOrdersMenuInfo.back.generic.flags = QMF_LEFT_JUSTIFY|QMF_PULSEIFFOCUS;
379 teamOrdersMenuInfo.back.generic.callback = UI_TeamOrdersMenu_BackEvent;
380 teamOrdersMenuInfo.back.generic.x = 0;
381 teamOrdersMenuInfo.back.generic.y = 480-64;
382 teamOrdersMenuInfo.back.width = 128;
383 teamOrdersMenuInfo.back.height = 64;
384 teamOrdersMenuInfo.back.focuspic = ART_BACK1;
385
386 Menu_AddItem( &teamOrdersMenuInfo.menu, &teamOrdersMenuInfo.banner );
387 Menu_AddItem( &teamOrdersMenuInfo.menu, &teamOrdersMenuInfo.frame );
388 Menu_AddItem( &teamOrdersMenuInfo.menu, &teamOrdersMenuInfo.list );
389 Menu_AddItem( &teamOrdersMenuInfo.menu, &teamOrdersMenuInfo.back );
390
391 teamOrdersMenuInfo.list.generic.left = 220;
392 teamOrdersMenuInfo.list.generic.top = teamOrdersMenuInfo.list.generic.y;
393 teamOrdersMenuInfo.list.generic.right = 420;
394 UI_TeamOrdersMenu_SetList( ID_LIST_BOTS );
395}

421/*
422===============
423UI_TeamOrdersMenu_f
424===============
425*/
426void UI_TeamOrdersMenu_f( void ) {
427 uiClientState_t cs;
428 char info[MAX_INFO_STRING];
429 int team;
430
431 // make sure it's a team game
432 trap_GetConfigString( CS_SERVERINFO, info, sizeof(info) );
433 teamOrdersMenuInfo.gametype = atoi( Info_ValueForKey( info, "g_gametype" ) );
434 if( teamOrdersMenuInfo.gametype < GT_TEAM ) {
435 return;
436 }
437
438 // not available to spectators
439 trap_GetClientState( &cs );
440 trap_GetConfigString( CS_PLAYERS + cs.clientNum, info, MAX_INFO_STRING );
441 team = atoi( Info_ValueForKey( info, "t" ) );
442 if( team == TEAM_SPECTATOR ) {
443 return;
444 }
445
446 UI_TeamOrdersMenu();
447}

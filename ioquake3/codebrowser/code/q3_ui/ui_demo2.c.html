82/*
83===============
84Demos_MenuEvent
85===============
86*/
87static void Demos_MenuEvent( void *ptr, int event ) {
88 if( event != QM_ACTIVATED ) {
89 return;
90 }
91
92 switch( ((menucommon_s*)ptr)->id ) {
93 case ID_GO:
94 UI_ForceMenuOff ();
95 trap_Cmd_ExecuteText( EXEC_APPEND, va( "demo %s\n",
96 s_demos.list.itemnames[s_demos.list.curvalue]) );
97 break;
98
99 case ID_BACK:
100 UI_PopMenu();
101 break;
102
103 case ID_LEFT:
104 ScrollList_Key( &s_demos.list, K_LEFTARROW );
105 break;
106
107 case ID_RIGHT:
108 ScrollList_Key( &s_demos.list, K_RIGHTARROW );
109 break;
110 }
111}

114/*
115===============
116Demos_MenuInit
117===============
118*/
119static void Demos_MenuInit( void ) {
120 int i, j;
121 int len;
122 char *demoname, extension[32];
123 int protocol, protocolLegacy;
124
125 memset( &s_demos, 0 ,sizeof(demos_t) );
126
127 Demos_Cache();
128
129 s_demos.menu.fullscreen = qtrue;
130 s_demos.menu.wrapAround = qtrue;
131
132 s_demos.banner.generic.type = MTYPE_BTEXT;
133 s_demos.banner.generic.x = 320;
134 s_demos.banner.generic.y = 16;
135 s_demos.banner.string = "DEMOS";
136 s_demos.banner.color = color_white;
137 s_demos.banner.style = UI_CENTER;
138
139 s_demos.framel.generic.type = MTYPE_BITMAP;
140 s_demos.framel.generic.name = ART_FRAMEL;
141 s_demos.framel.generic.flags = QMF_INACTIVE;
142 s_demos.framel.generic.x = 0;
143 s_demos.framel.generic.y = 78;
144 s_demos.framel.width = 256;
145 s_demos.framel.height = 329;
146
147 s_demos.framer.generic.type = MTYPE_BITMAP;
148 s_demos.framer.generic.name = ART_FRAMER;
149 s_demos.framer.generic.flags = QMF_INACTIVE;
150 s_demos.framer.generic.x = 376;
151 s_demos.framer.generic.y = 76;
152 s_demos.framer.width = 256;
153 s_demos.framer.height = 334;
154
155 s_demos.arrows.generic.type = MTYPE_BITMAP;
156 s_demos.arrows.generic.name = ART_ARROWS;
157 s_demos.arrows.generic.flags = QMF_INACTIVE;
158 s_demos.arrows.generic.x = 320-ARROWS_WIDTH/2;
159 s_demos.arrows.generic.y = 400;
160 s_demos.arrows.width = ARROWS_WIDTH;
161 s_demos.arrows.height = ARROWS_HEIGHT;
162
163 s_demos.left.generic.type = MTYPE_BITMAP;
164 s_demos.left.generic.flags = QMF_LEFT_JUSTIFY|QMF_PULSEIFFOCUS|QMF_MOUSEONLY;
165 s_demos.left.generic.x = 320-ARROWS_WIDTH/2;
166 s_demos.left.generic.y = 400;
167 s_demos.left.generic.id = ID_LEFT;
168 s_demos.left.generic.callback = Demos_MenuEvent;
169 s_demos.left.width = ARROWS_WIDTH/2;
170 s_demos.left.height = ARROWS_HEIGHT;
171 s_demos.left.focuspic = ART_ARROWLEFT;
172
173 s_demos.right.generic.type = MTYPE_BITMAP;
174 s_demos.right.generic.flags = QMF_LEFT_JUSTIFY|QMF_PULSEIFFOCUS|QMF_MOUSEONLY;
175 s_demos.right.generic.x = 320;
176 s_demos.right.generic.y = 400;
177 s_demos.right.generic.id = ID_RIGHT;
178 s_demos.right.generic.callback = Demos_MenuEvent;
179 s_demos.right.width = ARROWS_WIDTH/2;
180 s_demos.right.height = ARROWS_HEIGHT;
181 s_demos.right.focuspic = ART_ARROWRIGHT;
182
183 s_demos.back.generic.type = MTYPE_BITMAP;
184 s_demos.back.generic.name = ART_BACK0;
185 s_demos.back.generic.flags = QMF_LEFT_JUSTIFY|QMF_PULSEIFFOCUS;
186 s_demos.back.generic.id = ID_BACK;
187 s_demos.back.generic.callback = Demos_MenuEvent;
188 s_demos.back.generic.x = 0;
189 s_demos.back.generic.y = 480-64;
190 s_demos.back.width = 128;
191 s_demos.back.height = 64;
192 s_demos.back.focuspic = ART_BACK1;
193
194 s_demos.go.generic.type = MTYPE_BITMAP;
195 s_demos.go.generic.name = ART_GO0;
196 s_demos.go.generic.flags = QMF_RIGHT_JUSTIFY|QMF_PULSEIFFOCUS;
197 s_demos.go.generic.id = ID_GO;
198 s_demos.go.generic.callback = Demos_MenuEvent;
199 s_demos.go.generic.x = 640;
200 s_demos.go.generic.y = 480-64;
201 s_demos.go.width = 128;
202 s_demos.go.height = 64;
203 s_demos.go.focuspic = ART_GO1;
204
205 s_demos.list.generic.type = MTYPE_SCROLLLIST;
206 s_demos.list.generic.flags = QMF_PULSEIFFOCUS;
207 s_demos.list.generic.callback = Demos_MenuEvent;
208 s_demos.list.generic.id = ID_LIST;
209 s_demos.list.generic.x = 118;
210 s_demos.list.generic.y = 130;
211 s_demos.list.width = 16;
212 s_demos.list.height = 14;
213 s_demos.list.itemnames = (const char **)s_demos.demolist;
214 s_demos.list.columns = 3;
215
216 protocolLegacy = trap_Cvar_VariableValue("com_legacyprotocol");
217 protocol = trap_Cvar_VariableValue("com_protocol");
218
219 if(!protocol)
220 protocol = trap_Cvar_VariableValue("protocol");
221 if(protocolLegacy == protocol)
222 protocolLegacy = 0;
223
224 Com_sprintf(extension, sizeof(extension), ".%s%d", DEMOEXT, protocol);
225 s_demos.numDemos = trap_FS_GetFileList("demos", extension, s_demos.names, ARRAY_LEN(s_demos.names));
226
227 demoname = s_demos.names;
228 i = 0;
229
230 for(j = 0; j < 2; j++)
231 {
232 if(s_demos.numDemos > MAX_DEMOS)
233 s_demos.numDemos = MAX_DEMOS;
234
235 for(; i < s_demos.numDemos; i++)
236 {
237 s_demos.list.itemnames[i] = demoname;
238
239 len = strlen(demoname);
240
241 demoname += len + 1;
242 }
243
244 if(!j)
245 {
246 if(protocolLegacy > 0 && s_demos.numDemos < MAX_DEMOS)
247 {
248 Com_sprintf(extension, sizeof(extension), ".%s%d", DEMOEXT, protocolLegacy);
249 s_demos.numDemos += trap_FS_GetFileList("demos", extension, demoname,
250 ARRAY_LEN(s_demos.names) - (demoname - s_demos.names));
251 }
252 else
253 break;
254 }
255 }
256
257 s_demos.list.numitems = s_demos.numDemos;
258
259 if(!s_demos.numDemos)
260 {
261 s_demos.list.itemnames[0] = "No Demos Found.";
262 s_demos.list.numitems = 1;
263
264 //degenerate case, not selectable
265 s_demos.go.generic.flags |= (QMF_INACTIVE|QMF_HIDDEN);
266 }
267
268 Menu_AddItem( &s_demos.menu, &s_demos.banner );
269 Menu_AddItem( &s_demos.menu, &s_demos.framel );
270 Menu_AddItem( &s_demos.menu, &s_demos.framer );
271 Menu_AddItem( &s_demos.menu, &s_demos.list );
272 Menu_AddItem( &s_demos.menu, &s_demos.arrows );
273 Menu_AddItem( &s_demos.menu, &s_demos.left );
274 Menu_AddItem( &s_demos.menu, &s_demos.right );
275 Menu_AddItem( &s_demos.menu, &s_demos.back );
276 Menu_AddItem( &s_demos.menu, &s_demos.go );
277}

279/*
280=================
281Demos_Cache
282=================
283*/
284void Demos_Cache( void ) {
285 trap_R_RegisterShaderNoMip( ART_BACK0 );
286 trap_R_RegisterShaderNoMip( ART_BACK1 );
287 trap_R_RegisterShaderNoMip( ART_GO0 );
288 trap_R_RegisterShaderNoMip( ART_GO1 );
289 trap_R_RegisterShaderNoMip( ART_FRAMEL );
290 trap_R_RegisterShaderNoMip( ART_FRAMER );
291 trap_R_RegisterShaderNoMip( ART_ARROWS );
292 trap_R_RegisterShaderNoMip( ART_ARROWLEFT );
293 trap_R_RegisterShaderNoMip( ART_ARROWRIGHT );
294}

301void UI_DemosMenu( void ) {
302 Demos_MenuInit();
303 UI_PushMenu( &s_demos.menu );

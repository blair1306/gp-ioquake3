96/*
97=================
98UI_SoundOptionsMenu_Event
99=================
100*/
101static void UI_SoundOptionsMenu_Event( void* ptr, int event ) {
102 if( event != QM_ACTIVATED ) {
103 return;
104 }
105
106 switch( ((menucommon_s*)ptr)->id ) {
107 case ID_GRAPHICS:
108 UI_PopMenu();
109 UI_GraphicsOptionsMenu();
110 break;
111
112 case ID_DISPLAY:
113 UI_PopMenu();
114 UI_DisplayOptionsMenu();
115 break;
116
117 case ID_SOUND:
118 break;
119
120 case ID_NETWORK:
121 UI_PopMenu();
122 UI_NetworkOptionsMenu();
123 break;
124/*
125 case ID_A3D:
126 if( soundOptionsInfo.a3d.curvalue ) {
127 trap_Cmd_ExecuteText( EXEC_NOW, "s_enable_a3d\n" );
128 }
129 else {
130 trap_Cmd_ExecuteText( EXEC_NOW, "s_disable_a3d\n" );
131 }
132 soundOptionsInfo.a3d.curvalue = (int)trap_Cvar_VariableValue( "s_usingA3D" );
133 break;
134*/
135 case ID_BACK:
136 UI_PopMenu();
137 break;
138
139 case ID_APPLY:
140 trap_Cvar_SetValue( "s_volume", soundOptionsInfo.sfxvolume.curvalue / 10 );
141 soundOptionsInfo.sfxvolume_original = soundOptionsInfo.sfxvolume.curvalue;
142
143 trap_Cvar_SetValue( "s_musicvolume", soundOptionsInfo.musicvolume.curvalue / 10 );
144 soundOptionsInfo.musicvolume_original = soundOptionsInfo.musicvolume.curvalue;
145
146 // Check if something changed that requires the sound system to be restarted.
147 if (soundOptionsInfo.quality_original != soundOptionsInfo.quality.curvalue
148 || soundOptionsInfo.soundSystem_original != soundOptionsInfo.soundSystem.curvalue)
149 {
150 int speed;
151
152 switch ( soundOptionsInfo.quality.curvalue )
153 {
154 default:
155 case 0:
156 speed = 11025;
157 break;
158 case 1:
159 speed = 22050;
160 break;
161 case 2:
162 speed = 44100;
163 break;
164 }
165
166 if (speed == DEFAULT_SDL_SND_SPEED)
167 speed = 0;
168
169 trap_Cvar_SetValue( "s_sdlSpeed", speed );
170 soundOptionsInfo.quality_original = soundOptionsInfo.quality.curvalue;
171
172 trap_Cvar_SetValue( "s_useOpenAL", (soundOptionsInfo.soundSystem.curvalue == UISND_OPENAL) );
173 soundOptionsInfo.soundSystem_original = soundOptionsInfo.soundSystem.curvalue;
174
175 UI_ForceMenuOff();
176 trap_Cmd_ExecuteText( EXEC_APPEND, "snd_restart\n" );
177 }
178 break;
179 }
180}

182/*
183=================
184SoundOptions_UpdateMenuItems
185=================
186*/
187static void SoundOptions_UpdateMenuItems( void )
188{
189 if ( soundOptionsInfo.soundSystem.curvalue == UISND_SDL )
190 {
191 soundOptionsInfo.quality.generic.flags &= ~QMF_GRAYED;
192 }
193 else
194 {
195 soundOptionsInfo.quality.generic.flags |= QMF_GRAYED;
196 }
197
198 soundOptionsInfo.apply.generic.flags |= QMF_HIDDEN|QMF_INACTIVE;
199
200 if ( soundOptionsInfo.sfxvolume_original != soundOptionsInfo.sfxvolume.curvalue )
201 {
202 soundOptionsInfo.apply.generic.flags &= ~(QMF_HIDDEN|QMF_INACTIVE);
203 }
204 if ( soundOptionsInfo.musicvolume_original != soundOptionsInfo.musicvolume.curvalue )
205 {
206 soundOptionsInfo.apply.generic.flags &= ~(QMF_HIDDEN|QMF_INACTIVE);
207 }
208 if ( soundOptionsInfo.soundSystem_original != soundOptionsInfo.soundSystem.curvalue )
209 {
210 soundOptionsInfo.apply.generic.flags &= ~(QMF_HIDDEN|QMF_INACTIVE);
211 }
212 if ( soundOptionsInfo.quality_original != soundOptionsInfo.quality.curvalue )
213 {
214 soundOptionsInfo.apply.generic.flags &= ~(QMF_HIDDEN|QMF_INACTIVE);
215 }
216}

278/*
279=================
280ArenaServers_Compare
281=================
282*/
283static int QDECL ArenaServers_Compare( const void *arg1, const void *arg2 ) {
284 float f1;
285 float f2;
286 servernode_t* t1;
287 servernode_t* t2;
288
289 t1 = (servernode_t *)arg1;
290 t2 = (servernode_t *)arg2;
291
292 switch( g_sortkey ) {
293 case SORT_HOST:
294 return Q_stricmp( t1->hostname, t2->hostname );
295
296 case SORT_MAP:
297 return Q_stricmp( t1->mapname, t2->mapname );
298
299 case SORT_CLIENTS:
300 f1 = t1->maxclients - t1->numclients;
301 if( f1 < 0 ) {
302 f1 = 0;
303 }
304
305 f2 = t2->maxclients - t2->numclients;
306 if( f2 < 0 ) {
307 f2 = 0;
308 }
309
310 if( f1 < f2 ) {
311 return 1;
312 }
313 if( f1 == f2 ) {
314 return 0;
315 }
316 return -1;
317
318 case SORT_GAME:
319 if( t1->gametype < t2->gametype ) {
320 return -1;
321 }
322 if( t1->gametype == t2->gametype ) {
323 return 0;
324 }
325 return 1;
326
327 case SORT_PING:
328 if( t1->pingtime < t2->pingtime ) {
329 return -1;
330 }
331 if( t1->pingtime > t2->pingtime ) {
332 return 1;
333 }
334 return Q_stricmp( t1->hostname, t2->hostname );
335 }
336
337 return 0;
338}

427 g_arenaservers.gametype.generic.flags &= ~QMF_GRAYED;
428 g_arenaservers.sortkey.generic.flags &= ~QMF_GRAYED;
429 g_arenaservers.showempty.generic.flags &= ~QMF_GRAYED;

432 g_arenaservers.refresh.generic.flags &= ~QMF_GRAYED;
433 g_arenaservers.go.generic.flags &= ~QMF_GRAYED;
434 g_arenaservers.punkbuster.generic.flags &= ~QMF_GRAYED;
435
436 // update status bar
437 if( g_servertype >= UIAS_GLOBAL0 && g_servertype <= UIAS_GLOBAL5 ) {
438 g_arenaservers.statusbar.string = quake3worldMessage;
439 }
440 else {
441 g_arenaservers.statusbar.string = "";
442 }
443
444 }
445 }
446 else {
447 // no servers found
448 if( g_arenaservers.refreshservers ) {
449 strcpy( g_arenaservers.status.string,"Scanning For Servers." );
450 g_arenaservers.statusbar.string = "Press SPACE to stop";
451
452 // disable controls during refresh
453 g_arenaservers.gametype.generic.flags |= QMF_GRAYED;
454 g_arenaservers.sortkey.generic.flags |= QMF_GRAYED;
455 g_arenaservers.showempty.generic.flags |= QMF_GRAYED;
456 g_arenaservers.showfull.generic.flags |= QMF_GRAYED;

458 g_arenaservers.refresh.generic.flags |= QMF_GRAYED;
459 g_arenaservers.go.generic.flags |= QMF_GRAYED;
460 g_arenaservers.punkbuster.generic.flags |= QMF_GRAYED;
461 }
462 else {
463 if( g_arenaservers.numqueriedservers < 0 ) {
464 strcpy(g_arenaservers.status.string,"No Response From Master Server." );
465 }
466 else {
467 strcpy(g_arenaservers.status.string,"No Servers Found." );
468 }
469
470 // update status bar
471 if( g_servertype >= UIAS_GLOBAL0 && g_servertype <= UIAS_GLOBAL5 ) {
472 g_arenaservers.statusbar.string = quake3worldMessage;
473 }
474 else {
475 g_arenaservers.statusbar.string = "";
476 }
477
478 // end of refresh - set control state
479 g_arenaservers.master.generic.flags &= ~QMF_GRAYED;
480 g_arenaservers.gametype.generic.flags &= ~QMF_GRAYED;
481 g_arenaservers.sortkey.generic.flags &= ~QMF_GRAYED;

492 g_arenaservers.list.curvalue = 0;
493 g_arenaservers.list.top = 0;
494
495 // update picture
496 ArenaServers_UpdatePicture();

498 }
499
500 // build list box strings - apply culling filters
501 servernodeptr = g_arenaservers.serverlist;
502 count = *g_arenaservers.numservers;
503 for( i = 0, j = 0; i < count; i++, servernodeptr++ ) {
504 tableptr = &g_arenaservers.table[j];
505 tableptr->servernode = servernodeptr;
506 buff = tableptr->buff;
507
508 // can only cull valid results
509 if( !g_emptyservers && !servernodeptr->numclients ) {
510 continue;
511 }
512
513 if( !g_fullservers && ( servernodeptr->numclients == servernodeptr->maxclients ) ) {
514 continue;
515 }
516
517 switch( g_gametype ) {
518 case GAMES_ALL:
519 break;
520
521 case GAMES_FFA:
522 if( servernodeptr->gametype != GT_FFA ) {
523 continue;
524 }
525 break;
526
527 case GAMES_TEAMPLAY:
528 if( servernodeptr->gametype != GT_TEAM ) {
529 continue;
530 }
531 break;
532
533 case GAMES_TOURNEY:
534 if( servernodeptr->gametype != GT_TOURNAMENT ) {
535 continue;
536 }
537 break;
538
539 case GAMES_CTF:
540 if( servernodeptr->gametype != GT_CTF ) {
541 continue;
542 }
543 break;
544 }
545
546 if( servernodeptr->pingtime < servernodeptr->minPing ) {
547 pingColor = S_COLOR_BLUE;
548 }
549 else if( servernodeptr->maxPing && servernodeptr->pingtime > servernodeptr->maxPing ) {
550 pingColor = S_COLOR_BLUE;
551 }
552 else if( servernodeptr->pingtime < 200 ) {
553 pingColor = S_COLOR_GREEN;
554 }
555 else if( servernodeptr->pingtime < 400 ) {
556 pingColor = S_COLOR_YELLOW;
557 }
558 else {
559 pingColor = S_COLOR_RED;
560 }
561
562 Com_sprintf( buff, MAX_LISTBOXWIDTH, "%-20.20s %-12.12s %2d/%2d %-8.8s %4s%s%3d " S_COLOR_YELLOW "%s",
563 servernodeptr->hostname, servernodeptr->mapname, servernodeptr->numclients,
564 servernodeptr->maxclients, servernodeptr->gamename,
565 netnames[servernodeptr->nettype], pingColor, servernodeptr->pingtime, servernodeptr->bPB ? "Yes" : "No" );
566 j++;
567 }
568
569 g_arenaservers.list.numitems = j;
570 g_arenaservers.list.curvalue = 0;
571 g_arenaservers.list.top = 0;
572
573 // update picture
574 ArenaServers_UpdatePicture();
575}

578/*
579=================
580ArenaServers_Remove
581=================
582*/
583static void ArenaServers_Remove( void )
584{
585 int i;
586 servernode_t* servernodeptr;
587 table_t* tableptr;
588
589 if (!g_arenaservers.list.numitems)
590 return;
591
592 // remove selected item from display list
593 // items are in scattered order due to sort and cull
594 // perform delete on list box contents, resync all lists
595
596 tableptr = &g_arenaservers.table[g_arenaservers.list.curvalue];
597 servernodeptr = tableptr->servernode;
598
599 // find address in master list
600 for (i=0; i<g_arenaservers.numfavoriteaddresses; i++)
601 {
602 if (!Q_stricmp(g_arenaservers.favoriteaddresses[i],servernodeptr->adrstr))
603 {
604 // delete address from master list
605 if (i < g_arenaservers.numfavoriteaddresses-1)
606 {
607 // shift items up
608 memcpy( &g_arenaservers.favoriteaddresses[i], &g_arenaservers.favoriteaddresses[i+1], (g_arenaservers.numfavoriteaddresses - i - 1)* MAX_ADDRESSLENGTH );
609 }
610 g_arenaservers.numfavoriteaddresses--;
611 memset( &g_arenaservers.favoriteaddresses[g_arenaservers.numfavoriteaddresses], 0, MAX_ADDRESSLENGTH );
612 break;
613 }
614 }
615
616 // find address in server list
617 for (i=0; i<g_numfavoriteservers; i++)
618 {
619 if (&g_favoriteserverlist[i] == servernodeptr)
620 {
621
622 // delete address from server list
623 if (i < g_numfavoriteservers-1)
624 {
625 // shift items up
626 memcpy( &g_favoriteserverlist[i], &g_favoriteserverlist[i+1], (g_numfavoriteservers - i - 1)*sizeof(servernode_t));
627 }
628 g_numfavoriteservers--;
629 memset( &g_favoriteserverlist[ g_numfavoriteservers ], 0, sizeof(servernode_t));
630 break;
631 }
632 }
633
634 g_arenaservers.numqueriedservers = g_arenaservers.numfavoriteaddresses;
635 g_arenaservers.currentping = g_arenaservers.numfavoriteaddresses;
636}

644static void ArenaServers_Insert( char* adrstr, char* info, int pingtime )
645{
646 servernode_t* servernodeptr;
647 char* s;
648 int i;
649
650
651 if ((pingtime >= ArenaServers_MaxPing()) && (g_servertype != UIAS_FAVORITES))
652 {
653 // slow global or local servers do not get entered
654 return;
655 }
656
657 if (*g_arenaservers.numservers >= g_arenaservers.maxservers) {
658 // list full;
659 servernodeptr = g_arenaservers.serverlist+(*g_arenaservers.numservers)-1;
660 } else {
661 // next slot
662 servernodeptr = g_arenaservers.serverlist+(*g_arenaservers.numservers);
663 (*g_arenaservers.numservers)++;
664 }
665
666 Q_strncpyz( servernodeptr->adrstr, adrstr, MAX_ADDRESSLENGTH );
667
668 Q_strncpyz( servernodeptr->hostname, Info_ValueForKey( info, "hostname"), MAX_HOSTNAMELENGTH );
669 Q_CleanStr( servernodeptr->hostname );
670 Q_strupr( servernodeptr->hostname );
671
672 Q_strncpyz( servernodeptr->mapname, Info_ValueForKey( info, "mapname"), MAX_MAPNAMELENGTH );
673 Q_CleanStr( servernodeptr->mapname );
674 Q_strupr( servernodeptr->mapname );
675
676 servernodeptr->numclients = atoi( Info_ValueForKey( info, "clients") );
677 servernodeptr->maxclients = atoi( Info_ValueForKey( info, "sv_maxclients") );
678 servernodeptr->pingtime = pingtime;
679 servernodeptr->minPing = atoi( Info_ValueForKey( info, "minPing") );
680 servernodeptr->maxPing = atoi( Info_ValueForKey( info, "maxPing") );
681 servernodeptr->bPB = atoi( Info_ValueForKey( info, "punkbuster") );
682
683 /*
684 s = Info_ValueForKey( info, "nettype" );
685 for (i=0; ;i++)
686 {
687 if (!netnames[i])
688 {
689 servernodeptr->nettype = 0;
690 break;
691 }
692 else if (!Q_stricmp( netnames[i], s ))
693 {
694 servernodeptr->nettype = i;
695 break;
696 }
697 }
698 */
699 servernodeptr->nettype = atoi(Info_ValueForKey(info, "nettype"));
700 if (servernodeptr->nettype < 0 || servernodeptr->nettype >= ARRAY_LEN(netnames) - 1) {
701 servernodeptr->nettype = 0;
702 }
703
704 s = Info_ValueForKey( info, "game");
705 i = atoi( Info_ValueForKey( info, "gametype") );
706 if( i < 0 ) {
707 i = 0;
708 }
709 else if( i > 11 ) {
710 i = 12;
711 }
712 if( *s ) {
713 servernodeptr->gametype = i;//-1;
714 Q_strncpyz( servernodeptr->gamename, s, sizeof(servernodeptr->gamename) );
715 }
716 else {
717 servernodeptr->gametype = i;
718 Q_strncpyz( servernodeptr->gamename, gamenames[i], sizeof(servernodeptr->gamename) );
719 }
720}

723/*
724=================
725ArenaServers_LoadFavorites
726
727Load cvar address book entries into local lists.
728=================
729*/
730void ArenaServers_LoadFavorites( void )
731{
732 int i;
733 int j;
734 int numtempitems;
735 char adrstr[MAX_ADDRESSLENGTH];
736 servernode_t templist[MAX_FAVORITESERVERS];
737 qboolean found;
738
739 found = qfalse;
740
741 // copy the old
742 memcpy( templist, g_favoriteserverlist, sizeof(servernode_t)*MAX_FAVORITESERVERS );
743 numtempitems = g_numfavoriteservers;
744
745 // clear the current for sync
746 memset( g_favoriteserverlist, 0, sizeof(servernode_t)*MAX_FAVORITESERVERS );
747 g_numfavoriteservers = 0;
748
749 // resync existing results with new or deleted cvars
750 for (i=0; i<MAX_FAVORITESERVERS; i++)
751 {
752 trap_Cvar_VariableStringBuffer( va("server%d",i+1), adrstr, MAX_ADDRESSLENGTH );
753 if (!adrstr[0])
754 continue;
755
756 // favorite server addresses must be maintained outside refresh list
757 // this mimics local and global netadr's stored in client
758 // these can be fetched to fill ping list
759 strcpy( g_arenaservers.favoriteaddresses[g_numfavoriteservers], adrstr );
760
761 // find this server in the old list
762 for (j=0; j<numtempitems; j++)
763 if (!Q_stricmp( templist[j].adrstr, adrstr ))
764 break;
765
766 if (j < numtempitems)
767 {
768 // found server - add exisiting results
769 memcpy( &g_favoriteserverlist[g_numfavoriteservers], &templist[j], sizeof(servernode_t) );
770 found = qtrue;
771 }
772 else
773 {
774 // add new server
775 Q_strncpyz( g_favoriteserverlist[g_numfavoriteservers].adrstr, adrstr, MAX_ADDRESSLENGTH );
776 g_favoriteserverlist[g_numfavoriteservers].pingtime = ArenaServers_MaxPing();
777 }
778
779 g_numfavoriteservers++;
780 }
781
782 g_arenaservers.numfavoriteaddresses = g_numfavoriteservers;
783
784 if (!found)
785 {
786 // no results were found, reset server list
787 // list will be automatically refreshed when selected
788 g_numfavoriteservers = 0;
789 }
790}

820/*
821=================
822ArenaServers_DoRefresh
823=================
824*/
825static void ArenaServers_DoRefresh( void )
826{
827 int i;
828 int j;
829 int time;
830 int maxPing;
831 char adrstr[MAX_ADDRESSLENGTH];
832 char info[MAX_INFO_STRING];
833
834 if (uis.realtime < g_arenaservers.refreshtime)
835 {
836 if (g_servertype != UIAS_FAVORITES) {
837 if (g_servertype == UIAS_LOCAL) {
838 if (!trap_LAN_GetServerCount(AS_LOCAL)) {
839 return;
840 }
841 }
842 if (trap_LAN_GetServerCount(ArenaServers_SourceForLAN()) < 0) {
843 // still waiting for response
844 return;
845 }
846 }
847 } else if (g_servertype == UIAS_LOCAL) {
848 if (!trap_LAN_GetServerCount(AS_LOCAL)) {
849 // no local servers found, check again
850 trap_Cmd_ExecuteText( EXEC_APPEND, "localservers\n" );
851 g_arenaservers.refreshtime = uis.realtime + 5000;
852 return;
853 }
854 }
855
856 if (uis.realtime < g_arenaservers.nextpingtime)
857 {
858 // wait for time trigger
859 return;
860 }
861
862 // trigger at 10Hz intervals
863 g_arenaservers.nextpingtime = uis.realtime + 10;
864
865 // process ping results
866 maxPing = ArenaServers_MaxPing();
867 for (i=0; i<MAX_PINGREQUESTS; i++)
868 {
869 trap_LAN_GetPing( i, adrstr, MAX_ADDRESSLENGTH, &time );
870 if (!adrstr[0])
871 {
872 // ignore empty or pending pings
873 continue;
874 }
875
876 // find ping result in our local list
877 for (j=0; j<MAX_PINGREQUESTS; j++)
878 if (!Q_stricmp( adrstr, g_arenaservers.pinglist[j].adrstr ))
879 break;
880
881 if (j < MAX_PINGREQUESTS)
882 {
883 // found it
884 if (!time)
885 {
886 time = uis.realtime - g_arenaservers.pinglist[j].start;
887 if (time < maxPing)
888 {
889 // still waiting
890 continue;
891 }
892 }
893
894 if (time > maxPing)
895 {
896 // stale it out
897 info[0] = '\0';
898 time = maxPing;
899
900 // set hostname for nonresponsive favorite server
901 if (g_servertype == UIAS_FAVORITES) {
902 Info_SetValueForKey( info, "hostname", adrstr );
903 Info_SetValueForKey( info, "game", "???" );
904 }
905 }
906 else
907 {
908 trap_LAN_GetPingInfo( i, info, MAX_INFO_STRING );
909 }
910
911 // insert ping results
912 ArenaServers_Insert( adrstr, info, time );
913
914 // clear this query from internal list
915 g_arenaservers.pinglist[j].adrstr[0] = '\0';
916 }
917
918 // clear this query from external list
919 trap_LAN_ClearPing( i );
920 }
921
922 // get results of servers query
923 // counts can increase as servers respond
924 if (g_servertype == UIAS_FAVORITES) {
925 g_arenaservers.numqueriedservers = g_arenaservers.numfavoriteaddresses;
926 } else {
927 g_arenaservers.numqueriedservers = trap_LAN_GetServerCount(ArenaServers_SourceForLAN());
928 }
929
930// if (g_arenaservers.numqueriedservers > g_arenaservers.maxservers)
931// g_arenaservers.numqueriedservers = g_arenaservers.maxservers;
932
933 // send ping requests in reasonable bursts
934 // iterate ping through all found servers
935 for (i=0; i<MAX_PINGREQUESTS && g_arenaservers.currentping < g_arenaservers.numqueriedservers; i++)
936 {
937 if (trap_LAN_GetPingQueueCount() >= MAX_PINGREQUESTS)
938 {
939 // ping queue is full
940 break;
941 }
942
943 // find empty slot
944 for (j=0; j<MAX_PINGREQUESTS; j++)
945 if (!g_arenaservers.pinglist[j].adrstr[0])
946 break;
947
948 if (j >= MAX_PINGREQUESTS)
949 // no empty slots available yet - wait for timeout
950 break;
951
952 // get an address to ping
953
954 if (g_servertype == UIAS_FAVORITES) {
955 strcpy( adrstr, g_arenaservers.favoriteaddresses[g_arenaservers.currentping] );
956 } else {
957 trap_LAN_GetServerAddressString(ArenaServers_SourceForLAN(), g_arenaservers.currentping, adrstr, MAX_ADDRESSLENGTH );
958 }
959
960 strcpy( g_arenaservers.pinglist[j].adrstr, adrstr );
961 g_arenaservers.pinglist[j].start = uis.realtime;
962
963 trap_Cmd_ExecuteText( EXEC_NOW, va( "ping %s\n", adrstr ) );
964
965 // advance to next server
966 g_arenaservers.currentping++;
967 }
968
969 if (!trap_LAN_GetPingQueueCount())
970 {
971 // all pings completed
972 ArenaServers_StopRefresh();
973 return;
974 }
975
976 // update the user interface with ping status
977 ArenaServers_UpdateMenu();
978}

1156 ArenaServers_UpdateMenu();
1157 strcpy(g_arenaservers.status.string,"hit refresh to update");

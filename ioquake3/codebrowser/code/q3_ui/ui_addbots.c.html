88/*
89=================
90UI_AddBotsMenu_FightEvent
91=================
92*/
93static void UI_AddBotsMenu_FightEvent( void* ptr, int event ) {
94 const char *team;
95 int skill;
96
97 if (event != QM_ACTIVATED) {
98 return;
99 }
100
101 team = addBotsMenuInfo.team.itemnames[addBotsMenuInfo.team.curvalue];
102 skill = addBotsMenuInfo.skill.curvalue + 1;
103
104 trap_Cmd_ExecuteText( EXEC_APPEND, va("addbot %s %i %s %i\n",
105 addBotsMenuInfo.botnames[addBotsMenuInfo.selectedBotNum], skill, team, addBotsMenuInfo.delay) );
106
107 addBotsMenuInfo.delay += 1500;
108}

112=================
113UI_AddBotsMenu_BotEvent
114=================
115*/
116static void UI_AddBotsMenu_BotEvent( void* ptr, int event ) {
117 if (event != QM_ACTIVATED) {
118 return;
119 }
120
121 addBotsMenuInfo.bots[addBotsMenuInfo.selectedBotNum].color = color_orange;
122 addBotsMenuInfo.selectedBotNum = ((menucommon_s*)ptr)->id - ID_BOTNAME0;
123 addBotsMenuInfo.bots[addBotsMenuInfo.selectedBotNum].color = color_white;
124}

140/*
141=================
142UI_AddBotsMenu_SetBotNames
143=================
144*/
145static void UI_AddBotsMenu_SetBotNames( void ) {
146 int n;
147 const char *info;
148
149 for ( n = 0; n < 7; n++ ) {
150 info = UI_GetBotInfoByNumber( addBotsMenuInfo.sortedBotNums[addBotsMenuInfo.baseBotNum + n] );
151 Q_strncpyz( addBotsMenuInfo.botnames[n], Info_ValueForKey( info, "name" ), sizeof(addBotsMenuInfo.botnames[n]) );
152 }
153
154}

191/*
192=================
193UI_AddBotsMenu_GetSortedBotNums
194=================
195*/
196static int QDECL UI_AddBotsMenu_SortCompare( const void *arg1, const void *arg2 ) {
197 int num1, num2;
198 const char *info1, *info2;
199 const char *name1, *name2;
200
201 num1 = *(int *)arg1;
202 num2 = *(int *)arg2;
203
204 info1 = UI_GetBotInfoByNumber( num1 );
205 info2 = UI_GetBotInfoByNumber( num2 );
206
207 name1 = Info_ValueForKey( info1, "name" );
208 name2 = Info_ValueForKey( info2, "name" );
209
210 return Q_stricmp( name1, name2 );
211}
212
213static void UI_AddBotsMenu_GetSortedBotNums( void ) {
214 int n;
215
216 // initialize the array
217 for( n = 0; n < addBotsMenuInfo.numBots; n++ ) {
218 addBotsMenuInfo.sortedBotNums[n] = n;
219 }
220
221 qsort( addBotsMenuInfo.sortedBotNums, addBotsMenuInfo.numBots, sizeof(addBotsMenuInfo.sortedBotNums[0]), UI_AddBotsMenu_SortCompare );
222}

249static void UI_AddBotsMenu_Init( void ) {
250 int n;
251 int y;
252 int gametype;
253 int count;
254 char info[MAX_INFO_STRING];
255
256 trap_GetConfigString(CS_SERVERINFO, info, MAX_INFO_STRING);
257 gametype = atoi( Info_ValueForKey( info,"g_gametype" ) );
258
259 memset( &addBotsMenuInfo, 0 ,sizeof(addBotsMenuInfo) );
260 addBotsMenuInfo.menu.fullscreen = qfalse;
261 addBotsMenuInfo.menu.wrapAround = qtrue;
262 addBotsMenuInfo.delay = 1000;
263
264 UI_AddBots_Cache();
265
266 addBotsMenuInfo.numBots = UI_GetNumBots();
267 count = addBotsMenuInfo.numBots < 7 ? addBotsMenuInfo.numBots : 7;
268
269 addBotsMenuInfo.banner.generic.type = MTYPE_BTEXT;
270 addBotsMenuInfo.banner.generic.x = 320;
271 addBotsMenuInfo.banner.generic.y = 16;
272 addBotsMenuInfo.banner.string = "ADD BOTS";
273 addBotsMenuInfo.banner.color = color_white;
274 addBotsMenuInfo.banner.style = UI_CENTER;
275
276 addBotsMenuInfo.background.generic.type = MTYPE_BITMAP;
277 addBotsMenuInfo.background.generic.name = ART_BACKGROUND;
278 addBotsMenuInfo.background.generic.flags = QMF_INACTIVE;
279 addBotsMenuInfo.background.generic.x = 320-233;
280 addBotsMenuInfo.background.generic.y = 240-166;
281 addBotsMenuInfo.background.width = 466;
282 addBotsMenuInfo.background.height = 332;
283
284 addBotsMenuInfo.arrows.generic.type = MTYPE_BITMAP;
285 addBotsMenuInfo.arrows.generic.name = ART_ARROWS;
286 addBotsMenuInfo.arrows.generic.flags = QMF_INACTIVE;
287 addBotsMenuInfo.arrows.generic.x = 200;
288 addBotsMenuInfo.arrows.generic.y = 128;
289 addBotsMenuInfo.arrows.width = 64;
290 addBotsMenuInfo.arrows.height = 128;
291
292 addBotsMenuInfo.up.generic.type = MTYPE_BITMAP;
293 addBotsMenuInfo.up.generic.flags = QMF_LEFT_JUSTIFY|QMF_PULSEIFFOCUS;
294 addBotsMenuInfo.up.generic.x = 200;
295 addBotsMenuInfo.up.generic.y = 128;
296 addBotsMenuInfo.up.generic.id = ID_UP;
297 addBotsMenuInfo.up.generic.callback = UI_AddBotsMenu_UpEvent;
298 addBotsMenuInfo.up.width = 64;
299 addBotsMenuInfo.up.height = 64;
300 addBotsMenuInfo.up.focuspic = ART_ARROWUP;
301
302 addBotsMenuInfo.down.generic.type = MTYPE_BITMAP;
303 addBotsMenuInfo.down.generic.flags = QMF_LEFT_JUSTIFY|QMF_PULSEIFFOCUS;
304 addBotsMenuInfo.down.generic.x = 200;
305 addBotsMenuInfo.down.generic.y = 128+64;
306 addBotsMenuInfo.down.generic.id = ID_DOWN;
307 addBotsMenuInfo.down.generic.callback = UI_AddBotsMenu_DownEvent;
308 addBotsMenuInfo.down.width = 64;
309 addBotsMenuInfo.down.height = 64;
310 addBotsMenuInfo.down.focuspic = ART_ARROWDOWN;
311
312 for( n = 0, y = 120; n < count; n++, y += 20 ) {
313 addBotsMenuInfo.bots[n].generic.type = MTYPE_PTEXT;
314 addBotsMenuInfo.bots[n].generic.flags = QMF_LEFT_JUSTIFY|QMF_PULSEIFFOCUS;
315 addBotsMenuInfo.bots[n].generic.id = ID_BOTNAME0 + n;
316 addBotsMenuInfo.bots[n].generic.x = 320 - 56;
317 addBotsMenuInfo.bots[n].generic.y = y;
318 addBotsMenuInfo.bots[n].generic.callback = UI_AddBotsMenu_BotEvent;
319 addBotsMenuInfo.bots[n].string = addBotsMenuInfo.botnames[n];
320 addBotsMenuInfo.bots[n].color = color_orange;
321 addBotsMenuInfo.bots[n].style = UI_LEFT|UI_SMALLFONT;
322 }
323
324 y += 12;
325 addBotsMenuInfo.skill.generic.type = MTYPE_SPINCONTROL;
326 addBotsMenuInfo.skill.generic.flags = QMF_PULSEIFFOCUS|QMF_SMALLFONT;
327 addBotsMenuInfo.skill.generic.x = 320;
328 addBotsMenuInfo.skill.generic.y = y;
329 addBotsMenuInfo.skill.generic.name = "Skill:";
330 addBotsMenuInfo.skill.generic.id = ID_SKILL;
331 addBotsMenuInfo.skill.itemnames = skillNames;
332 addBotsMenuInfo.skill.curvalue = Com_Clamp( 0, 4, (int)trap_Cvar_VariableValue( "g_spSkill" ) - 1 );
333
334 y += SMALLCHAR_HEIGHT;
335 addBotsMenuInfo.team.generic.type = MTYPE_SPINCONTROL;
336 addBotsMenuInfo.team.generic.flags = QMF_PULSEIFFOCUS|QMF_SMALLFONT;
337 addBotsMenuInfo.team.generic.x = 320;
338 addBotsMenuInfo.team.generic.y = y;
339 addBotsMenuInfo.team.generic.name = "Team: ";
340 addBotsMenuInfo.team.generic.id = ID_TEAM;
341 if( gametype >= GT_TEAM ) {
342 addBotsMenuInfo.team.itemnames = teamNames2;
343 }
344 else {
345 addBotsMenuInfo.team.itemnames = teamNames1;
346 addBotsMenuInfo.team.generic.flags = QMF_GRAYED;
347 }
348
349 addBotsMenuInfo.go.generic.type = MTYPE_BITMAP;
350 addBotsMenuInfo.go.generic.name = ART_FIGHT0;

352 addBotsMenuInfo.go.generic.id = ID_GO;
353 addBotsMenuInfo.go.generic.callback = UI_AddBotsMenu_FightEvent;
354 addBotsMenuInfo.go.generic.x = 320+128-128;
355 addBotsMenuInfo.go.generic.y = 256+128-64;
356 addBotsMenuInfo.go.width = 128;
357 addBotsMenuInfo.go.height = 64;
358 addBotsMenuInfo.go.focuspic = ART_FIGHT1;
359
360 addBotsMenuInfo.back.generic.type = MTYPE_BITMAP;
361 addBotsMenuInfo.back.generic.name = ART_BACK0;
362 addBotsMenuInfo.back.generic.flags = QMF_LEFT_JUSTIFY|QMF_PULSEIFFOCUS;
363 addBotsMenuInfo.back.generic.id = ID_BACK;
364 addBotsMenuInfo.back.generic.callback = UI_AddBotsMenu_BackEvent;
365 addBotsMenuInfo.back.generic.x = 320-128;
366 addBotsMenuInfo.back.generic.y = 256+128-64;
367 addBotsMenuInfo.back.width = 128;
368 addBotsMenuInfo.back.height = 64;
369 addBotsMenuInfo.back.focuspic = ART_BACK1;
370
371 addBotsMenuInfo.baseBotNum = 0;
372 addBotsMenuInfo.selectedBotNum = 0;
373 addBotsMenuInfo.bots[0].color = color_white;
374
375 UI_AddBotsMenu_GetSortedBotNums();
376 UI_AddBotsMenu_SetBotNames();
377
378 Menu_AddItem( &addBotsMenuInfo.menu, &addBotsMenuInfo.background );
379 Menu_AddItem( &addBotsMenuInfo.menu, &addBotsMenuInfo.banner );
380 Menu_AddItem( &addBotsMenuInfo.menu, &addBotsMenuInfo.arrows );
381 Menu_AddItem( &addBotsMenuInfo.menu, &addBotsMenuInfo.up );
382 Menu_AddItem( &addBotsMenuInfo.menu, &addBotsMenuInfo.down );
383 for( n = 0; n < count; n++ ) {
384 Menu_AddItem( &addBotsMenuInfo.menu, &addBotsMenuInfo.bots[n] );
385 }
386 Menu_AddItem( &addBotsMenuInfo.menu, &addBotsMenuInfo.skill );
387 Menu_AddItem( &addBotsMenuInfo.menu, &addBotsMenuInfo.team );
388 Menu_AddItem( &addBotsMenuInfo.menu, &addBotsMenuInfo.go );
389 Menu_AddItem( &addBotsMenuInfo.menu, &addBotsMenuInfo.back );
390}

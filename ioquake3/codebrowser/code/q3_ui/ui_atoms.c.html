76/*
77=================
78UI_PushMenu
79=================
80*/
81void UI_PushMenu( menuframework_s *menu )
82{
83 int i;
84 menucommon_s* item;
85
86 // avoid stacking menus invoked by hotkeys
87 for (i=0 ; i<uis.menusp ; i++)
88 {
89 if (uis.stack[i] == menu)
90 {
91 uis.menusp = i;
92 break;
93 }
94 }
95
96 if (i == uis.menusp)
97 {
98 if (uis.menusp >= MAX_MENUDEPTH)
99 trap_Error("UI_PushMenu: menu stack overflow");
100
101 uis.stack[uis.menusp++] = menu;
102 }
103
104 uis.activemenu = menu;
105
106 // default cursor position
107 menu->cursor = 0;
108 menu->cursor_prev = 0;
109
110 m_entersound = qtrue;
111
112 trap_Key_SetCatcher( KEYCATCH_UI );
113
114 // force first available item to have focus
115 for (i=0; i<menu->nitems; i++)
116 {
117 item = (menucommon_s *)menu->items[i];
118 if (!(item->flags & (QMF_GRAYED|QMF_MOUSEONLY|QMF_INACTIVE)))
119 {
120 menu->cursor_prev = -1;
121 Menu_SetCursor( menu, i );
122 break;
123 }
124 }
125
126 uis.firstdraw = qtrue;
127}

129/*
130=================
131UI_PopMenu
132=================
133*/
134void UI_PopMenu (void)
135{
136 trap_S_StartLocalSound( menu_out_sound, CHAN_LOCAL_SOUND );
137
138 uis.menusp--;
139
140 if (uis.menusp < 0)
141 trap_Error ("UI_PopMenu: menu stack underflow");
142
143 if (uis.menusp) {
144 uis.activemenu = uis.stack[uis.menusp-1];
145 uis.firstdraw = qtrue;
146 }
147 else {
148 UI_ForceMenuOff ();
149 }

162/*
163=================
164UI_LerpColor
165=================
166*/
167void UI_LerpColor(vec4_t a, vec4_t b, vec4_t c, float t)
168{
169 int i;
170
171 // lerp and clamp each component
172 for (i=0; i<4; i++)
173 {
174 c[i] = a[i] + t*(b[i]-a[i]);
175 if (c[i] < 0)
176 c[i] = 0;
177 else if (c[i] > 1.0)
178 c[i] = 1.0;
179 }
180}

333/*
334=================
335UI_DrawBannerString
336=================
337*/
338static void UI_DrawBannerString2( int x, int y, const char* str, vec4_t color )
339{
340 const char* s;
341 unsigned char ch;
342 float ax;
343 float ay;
344 float aw;
345 float ah;
346 float frow;
347 float fcol;
348 float fwidth;
349 float fheight;
350
351 // draw the colored text
352 trap_R_SetColor( color );
353
354 ax = x * uis.xscale + uis.bias;
355 ay = y * uis.yscale;
356
357 s = str;
358 while ( *s )
359 {
360 ch = *s & 127;
361 if ( ch == ' ' ) {
362 ax += ((float)PROPB_SPACE_WIDTH + (float)PROPB_GAP_WIDTH)* uis.xscale;
363 }
364 else if ( ch >= 'A' && ch <= 'Z' ) {
365 ch -= 'A';
366 fcol = (float)propMapB[ch][0] / 256.0f;
367 frow = (float)propMapB[ch][1] / 256.0f;
368 fwidth = (float)propMapB[ch][2] / 256.0f;
369 fheight = (float)PROPB_HEIGHT / 256.0f;
370 aw = (float)propMapB[ch][2] * uis.xscale;
371 ah = (float)PROPB_HEIGHT * uis.yscale;
372 trap_R_DrawStretchPic( ax, ay, aw, ah, fcol, frow, fcol+fwidth, frow+fheight, uis.charsetPropB );
373 ax += (aw + (float)PROPB_GAP_WIDTH * uis.xscale);
374 }
375 s++;
376 }
377
378 trap_R_SetColor( NULL );
379}

381void UI_DrawBannerString( int x, int y, const char* str, int style, vec4_t color ) {
382 const char * s;
383 int ch;
384 int width;
385 vec4_t drawcolor;
386
387 // find the width of the drawn text
388 s = str;
389 width = 0;
390 while ( *s ) {
391 ch = *s;
392 if ( ch == ' ' ) {
393 width += PROPB_SPACE_WIDTH;
394 }
395 else if ( ch >= 'A' && ch <= 'Z' ) {
396 width += propMapB[ch - 'A'][2] + PROPB_GAP_WIDTH;
397 }
398 s++;
399 }
400 width -= PROPB_GAP_WIDTH;
401
402 switch( style & UI_FORMATMASK ) {
403 case UI_CENTER:
404 x -= width / 2;
405 break;
406
407 case UI_RIGHT:
408 x -= width;
409 break;
410
411 case UI_LEFT:
412 default:
413 break;
414 }
415
416 if ( style & UI_DROPSHADOW ) {
417 drawcolor[0] = drawcolor[1] = drawcolor[2] = 0;
418 drawcolor[3] = color[3];
419 UI_DrawBannerString2( x+2, y+2, str, drawcolor );
420 }
421
422 UI_DrawBannerString2( x, y, str, color );
423}

426int UI_ProportionalStringWidth( const char* str ) {
427 const char * s;
428 int ch;
429 int charWidth;
430 int width;
431
432 s = str;
433 width = 0;
434 while ( *s ) {
435 ch = *s & 127;
436 charWidth = propMap[ch][2];
437 if ( charWidth != -1 ) {
438 width += charWidth;
439 width += PROP_GAP_WIDTH;
440 }
441 s++;
442 }
443
444 width -= PROP_GAP_WIDTH;
445 return width;
446}

505/*
506=================
507UI_DrawProportionalString
508=================
509*/
510void UI_DrawProportionalString( int x, int y, const char* str, int style, vec4_t color ) {
511 vec4_t drawcolor;
512 int width;
513 float sizeScale;
514
515 sizeScale = UI_ProportionalSizeScale( style );
516
517 switch( style & UI_FORMATMASK ) {
518 case UI_CENTER:
519 width = UI_ProportionalStringWidth( str ) * sizeScale;
520 x -= width / 2;
521 break;
522
523 case UI_RIGHT:
524 width = UI_ProportionalStringWidth( str ) * sizeScale;
525 x -= width;
526 break;
527
528 case UI_LEFT:
529 default:
530 break;
531 }
532
533 if ( style & UI_DROPSHADOW ) {
534 drawcolor[0] = drawcolor[1] = drawcolor[2] = 0;
535 drawcolor[3] = color[3];
536 UI_DrawProportionalString2( x+2, y+2, str, drawcolor, sizeScale, uis.charsetProp );
537 }
538
539 if ( style & UI_INVERSE ) {
540 drawcolor[0] = color[0] * 0.7;
541 drawcolor[1] = color[1] * 0.7;
542 drawcolor[2] = color[2] * 0.7;
543 drawcolor[3] = color[3];
544 UI_DrawProportionalString2( x, y, str, drawcolor, sizeScale, uis.charsetProp );
545 return;
546 }
547
548 if ( style & UI_PULSE ) {
549 drawcolor[0] = color[0] * 0.7;
550 drawcolor[1] = color[1] * 0.7;
551 drawcolor[2] = color[2] * 0.7;
552 drawcolor[3] = color[3];
553 UI_DrawProportionalString2( x, y, str, color, sizeScale, uis.charsetProp );
554
555 drawcolor[0] = color[0];
556 drawcolor[1] = color[1];
557 drawcolor[2] = color[2];
558 drawcolor[3] = 0.5 + 0.5 * sin( uis.realtime / PULSE_DIVISOR );
559 UI_DrawProportionalString2( x, y, str, drawcolor, sizeScale, uis.charsetPropGlow );
560 return;
561 }
562
563 UI_DrawProportionalString2( x, y, str, color, sizeScale, uis.charsetProp );
564}

566/*
567=================
568UI_DrawProportionalString_Wrapped
569=================
570*/
571void UI_DrawProportionalString_AutoWrapped( int x, int y, int xmax, int ystep, const char* str, int style, vec4_t color ) {
572 int width;
573 char *s1,*s2,*s3;
574 char c_bcp;
575 char buf[1024];
576 float sizeScale;
577
578 if (!str || str[0]=='\0')
579 return;
580
581 sizeScale = UI_ProportionalSizeScale( style );
582
583 Q_strncpyz(buf, str, sizeof(buf));
584 s1 = s2 = s3 = buf;
585
586 while (1) {
587 do {
588 s3++;
589 } while (*s3!=' ' && *s3!='\0');
590 c_bcp = *s3;
591 *s3 = '\0';
592 width = UI_ProportionalStringWidth(s1) * sizeScale;
593 *s3 = c_bcp;
594 if (width > xmax) {
595 if (s1==s2)
596 {
597 // fuck, don't have a clean cut, we'll overflow
598 s2 = s3;
599 }
600 *s2 = '\0';
601 UI_DrawProportionalString(x, y, s1, style, color);
602 y += ystep;
603 if (c_bcp == '\0')
604 {
605 // that was the last word
606 // we could start a new loop, but that wouldn't be much use
607 // even if the word is too long, we would overflow it (see above)
608 // so just print it now if needed
609 s2++;
610 if (*s2 != '\0') // if we are printing an overflowing line we have s2 == s3
611 UI_DrawProportionalString(x, y, s2, style, color);
612 break;
613 }
614 s2++;
615 s1 = s2;
616 s3 = s2;
617 }
618 else
619 {
620 s2 = s3;
621 if (c_bcp == '\0') // we reached the end
622 {
623 UI_DrawProportionalString(x, y, s1, style, color);
624 break;
625 }
626 }
627 }
628}

630/*
631=================
632UI_DrawString2
633=================
634*/
635static void UI_DrawString2( int x, int y, const char* str, vec4_t color, int charw, int charh )
636{
637 const char* s;
638 char ch;
639 int forceColor = qfalse; //APSFIXME;
640 vec4_t tempcolor;
641 float ax;
642 float ay;
643 float aw;
644 float ah;
645 float frow;
646 float fcol;
647
648 if (y < -charh)
649 // offscreen
650 return;
651
652 // draw the colored text
653 trap_R_SetColor( color );
654
655 ax = x * uis.xscale + uis.bias;
656 ay = y * uis.yscale;
657 aw = charw * uis.xscale;
658 ah = charh * uis.yscale;
659
660 s = str;
661 while ( *s )
662 {
663 if ( Q_IsColorString( s ) )
664 {
665 if ( !forceColor )
666 {
667 memcpy( tempcolor, g_color_table[ColorIndex(s[1])], sizeof( tempcolor ) );
668 tempcolor[3] = color[3];
669 trap_R_SetColor( tempcolor );
670 }
671 s += 2;
672 continue;
673 }
674
675 ch = *s & 255;
676 if (ch != ' ')
677 {
678 frow = (ch>>4)*0.0625;
679 fcol = (ch&15)*0.0625;
680 trap_R_DrawStretchPic( ax, ay, aw, ah, fcol, frow, fcol + 0.0625, frow + 0.0625, uis.charset );
681 }
682
683 ax += aw;
684 s++;
685 }
686
687 trap_R_SetColor( NULL );
688}

690/*
691=================
692UI_DrawString
693=================
694*/
695void UI_DrawString( int x, int y, const char* str, int style, vec4_t color )
696{
697 int len;
698 int charw;
699 int charh;
700 vec4_t newcolor;
701 vec4_t lowlight;
702 float *drawcolor;
703 vec4_t dropcolor;
704
705 if( !str ) {
706 return;
707 }
708
709 if ((style & UI_BLINK) && ((uis.realtime/BLINK_DIVISOR) & 1))
710 return;
711
712 if (style & UI_SMALLFONT)
713 {
714 charw = SMALLCHAR_WIDTH;
715 charh = SMALLCHAR_HEIGHT;
716 }
717 else if (style & UI_GIANTFONT)
718 {
719 charw = GIANTCHAR_WIDTH;
720 charh = GIANTCHAR_HEIGHT;
721 }
722 else
723 {
724 charw = BIGCHAR_WIDTH;
725 charh = BIGCHAR_HEIGHT;
726 }
727
728 if (style & UI_PULSE)
729 {
730 lowlight[0] = 0.8*color[0];
731 lowlight[1] = 0.8*color[1];
732 lowlight[2] = 0.8*color[2];
733 lowlight[3] = 0.8*color[3];
734 UI_LerpColor(color,lowlight,newcolor,0.5+0.5*sin(uis.realtime/PULSE_DIVISOR));
735 drawcolor = newcolor;
736 }
737 else
738 drawcolor = color;
739
740 switch (style & UI_FORMATMASK)
741 {
742 case UI_CENTER:
743 // center justify at x
744 len = strlen(str);
745 x = x - len*charw/2;
746 break;
747
748 case UI_RIGHT:
749 // right justify at x
750 len = strlen(str);
751 x = x - len*charw;
752 break;
753
754 default:
755 // left justify at x
756 break;
757 }
758
759 if ( style & UI_DROPSHADOW )
760 {
761 dropcolor[0] = dropcolor[1] = dropcolor[2] = 0;
762 dropcolor[3] = drawcolor[3];
763 UI_DrawString2(x+2,y+2,str,dropcolor,charw,charh);
764 }
765
766 UI_DrawString2(x,y,str,drawcolor,charw,charh);
767}

804void UI_SetActiveMenu( uiMenuCommand_t menu ) {
805 // this should be the ONLY way the menu system is brought up
806 // enusure minumum menu data is cached
807 Menu_Cache();
808
809 switch ( menu ) {
810 case UIMENU_NONE:
811 UI_ForceMenuOff();

867/*
868=================
869UI_MouseEvent
870=================
871*/
872void UI_MouseEvent( int dx, int dy )
873{
874 int i;
875 int bias;
876 menucommon_s* m;
877
878 if (!uis.activemenu)
879 return;
880
881 // convert X bias to 640 coords
882 bias = uis.bias / uis.xscale;
883
884 // update mouse screen position
885 uis.cursorx += dx;
886 if (uis.cursorx < -bias)
887 uis.cursorx = -bias;
888 else if (uis.cursorx > SCREEN_WIDTH+bias)
889 uis.cursorx = SCREEN_WIDTH+bias;
890
891 uis.cursory += dy;
892 if (uis.cursory < 0)
893 uis.cursory = 0;
894 else if (uis.cursory > SCREEN_HEIGHT)
895 uis.cursory = SCREEN_HEIGHT;
896
897 // region test the active menu items
898 for (i=0; i<uis.activemenu->nitems; i++)
899 {
900 m = (menucommon_s*)uis.activemenu->items[i];
901
902 if (m->flags & (QMF_GRAYED|QMF_INACTIVE))
903 continue;
904
905 if ((uis.cursorx < m->left) ||
906 (uis.cursorx > m->right) ||
907 (uis.cursory < m->top) ||
908 (uis.cursory > m->bottom))
909 {
910 // cursor out of item bounds
911 continue;
912 }
913
914 // set focus to item at cursor
915 if (uis.activemenu->cursor != i)
916 {
917 Menu_SetCursor( uis.activemenu, i );
918 ((menucommon_s*)(uis.activemenu->items[uis.activemenu->cursor_prev]))->flags &= ~QMF_HASMOUSEFOCUS;
919
920 if ( !(((menucommon_s*)(uis.activemenu->items[uis.activemenu->cursor]))->flags & QMF_SILENT ) ) {
921 trap_S_StartLocalSound( menu_move_sound, CHAN_LOCAL_SOUND );
922 }
923 }
924
925 ((menucommon_s*)(uis.activemenu->items[uis.activemenu->cursor]))->flags |= QMF_HASMOUSEFOCUS;
926 return;
927 }
928
929 if (uis.activemenu->nitems > 0) {
930 // out of any region
931 ((menucommon_s*)(uis.activemenu->items[uis.activemenu->cursor]))->flags &= ~QMF_HASMOUSEFOCUS;
932 }
933}

1061/*
1062=================
1063UI_Init
1064=================
1065*/
1066void UI_Init( void ) {
1067 UI_RegisterCvars();
1068
1069 UI_InitGameinfo();
1070
1071 // cache redundant calulations
1072 trap_GetGlconfig( &uis.glconfig );
1073
1074 // for 640x480 virtualized screen
1075 uis.xscale = uis.glconfig.vidWidth * (1.0/640.0);
1076 uis.yscale = uis.glconfig.vidHeight * (1.0/480.0);
1077 if ( uis.glconfig.vidWidth * 480 > uis.glconfig.vidHeight * 640 ) {
1078 // wide screen
1079 uis.bias = 0.5 * ( uis.glconfig.vidWidth - ( uis.glconfig.vidHeight * (640.0/480.0) ) );
1080 uis.xscale = uis.yscale;
1081 }
1082 else {
1083 // no wide screen
1084 uis.bias = 0;
1085 }
1086
1087 // initialize the menu system
1088 Menu_Cache();
1089
1090 uis.activemenu = NULL;
1091 uis.menusp = 0;
1092}

1222 else
1223 Menu_Draw( uis.activemenu );
1224
1225 if( uis.firstdraw ) {

123/*
124=================
125PlayerIcon
126=================
127*/
128static void PlayerIcon( const char *modelAndSkin, char *iconName, int iconNameMaxSize ) {
129 char *skin;
130 char model[MAX_QPATH];
131
132 Q_strncpyz( model, modelAndSkin, sizeof(model));
133 skin = strrchr( model, '/' );
134 if ( skin ) {
135 *skin++ = '\0';
136 }
137 else {
138 skin = "default";
139 }
140
141 Com_sprintf(iconName, iconNameMaxSize, "models/players/%s/icon_%s.tga", model, skin );
142
143 if( !trap_R_RegisterShaderNoMip( iconName ) && Q_stricmp( skin, "default" ) != 0 ) {
144 Com_sprintf(iconName, iconNameMaxSize, "models/players/%s/icon_default.tga", model );
145 }
146}

162/*
163=================
164UI_SPLevelMenu_SetBots
165=================
166*/
167static void UI_SPLevelMenu_SetBots( void ) {
168 char *p;
169 char *bot;
170 char *botInfo;
171 char bots[MAX_INFO_STRING];
172
173 levelMenuInfo.numBots = 0;
174 if ( selectedArenaSet > currentSet ) {
175 return;
176 }
177
178 Q_strncpyz( bots, Info_ValueForKey( levelMenuInfo.selectedArenaInfo, "bots" ), sizeof(bots) );
179
180 p = &bots[0];
181 while( *p && levelMenuInfo.numBots < 7 ) {
182 //skip spaces
183 while( *p && *p == ' ' ) {
184 p++;
185 }
186 if( !*p ) {
187 break;
188 }
189
190 // mark start of bot name
191 bot = p;
192
193 // skip until space of null
194 while( *p && *p != ' ' ) {
195 p++;
196 }
197 if( *p ) {
198 *p++ = 0;
199 }
200
201 botInfo = UI_GetBotInfoByName( bot );
202 if(!botInfo)
203 {
204 botInfo = UI_GetBotInfoByNumber( levelMenuInfo.numBots );
205 }
206
207 if( botInfo ) {
208 levelMenuInfo.botPics[levelMenuInfo.numBots] = PlayerIconHandle( Info_ValueForKey( botInfo, "model" ) );
209 Q_strncpyz( levelMenuInfo.botNames[levelMenuInfo.numBots], Info_ValueForKey( botInfo, "name" ), 10 );
210 }
211 else {
212 levelMenuInfo.botPics[levelMenuInfo.numBots] = 0;
213 Q_strncpyz( levelMenuInfo.botNames[levelMenuInfo.numBots], bot, 10 );
214 }
215 Q_CleanStr( levelMenuInfo.botNames[levelMenuInfo.numBots] );
216 levelMenuInfo.numBots++;
217 }
218}

226static void UI_SPLevelMenu_SetMenuArena( int n, int level, const char *arenaInfo ) {
227 char map[MAX_QPATH];
228
229 Q_strncpyz( map, Info_ValueForKey( arenaInfo, "map" ), sizeof(map) );
230
231 Q_strncpyz( levelMenuInfo.levelNames[n], map, sizeof(levelMenuInfo.levelNames[n]) );
232 Q_strupr( levelMenuInfo.levelNames[n] );
233
234 UI_GetBestScore( level, &levelMenuInfo.levelScores[n], &levelMenuInfo.levelScoresSkill[n] );
235 if( levelMenuInfo.levelScores[n] > 8 ) {
236 levelMenuInfo.levelScores[n] = 8;
237 }
238
239 Com_sprintf( levelMenuInfo.levelPicNames[n], sizeof(levelMenuInfo.levelPicNames[n]), "levelshots/%s.tga", map );
240 if( !trap_R_RegisterShaderNoMip( levelMenuInfo.levelPicNames[n] ) ) {
241 strcpy( levelMenuInfo.levelPicNames[n], ART_MAP_UNKNOWN );
242 }
243 levelMenuInfo.item_maps[n].shader = 0;
244 if ( selectedArenaSet > currentSet ) {
245 levelMenuInfo.item_maps[n].generic.flags |= QMF_GRAYED;
246 }
247 else {
248 levelMenuInfo.item_maps[n].generic.flags &= ~QMF_GRAYED;
249 }
250
251 levelMenuInfo.item_maps[n].generic.flags &= ~QMF_INACTIVE;
252}

254static void UI_SPLevelMenu_SetMenuItems( void ) {
255 int n;
256 int level;
257 const char *arenaInfo;
258
259 if ( selectedArenaSet > currentSet ) {
260 selectedArena = -1;
261 }
262 else if ( selectedArena == -1 ) {
263 selectedArena = 0;
264 }
265
266 if( selectedArenaSet == trainingTier || selectedArenaSet == finalTier ) {
267 selectedArena = 0;
268 }
269
270 if( selectedArena != -1 ) {
271 trap_Cvar_SetValue( "ui_spSelection", selectedArenaSet * ARENAS_PER_TIER + selectedArena );
272 }
273
274 if( selectedArenaSet == trainingTier ) {
275 arenaInfo = UI_GetSpecialArenaInfo( "training" );
276 level = atoi( Info_ValueForKey( arenaInfo, "num" ) );
277 UI_SPLevelMenu_SetMenuArena( 0, level, arenaInfo );
278 levelMenuInfo.selectedArenaInfo = arenaInfo;
279
280 levelMenuInfo.item_maps[0].generic.x = 256;
281 Bitmap_Init( &levelMenuInfo.item_maps[0] );
282 levelMenuInfo.item_maps[0].generic.bottom += 32;
283 levelMenuInfo.numMaps = 1;
284
285 levelMenuInfo.item_maps[1].generic.flags |= QMF_INACTIVE;
286 levelMenuInfo.item_maps[2].generic.flags |= QMF_INACTIVE;
287 levelMenuInfo.item_maps[3].generic.flags |= QMF_INACTIVE;
288 levelMenuInfo.levelPicNames[1][0] = 0;
289 levelMenuInfo.levelPicNames[2][0] = 0;
290 levelMenuInfo.levelPicNames[3][0] = 0;
291 levelMenuInfo.item_maps[1].shader = 0;
292 levelMenuInfo.item_maps[2].shader = 0;
293 levelMenuInfo.item_maps[3].shader = 0;
294 }
295 else if( selectedArenaSet == finalTier ) {
296 arenaInfo = UI_GetSpecialArenaInfo( "final" );
297 level = atoi( Info_ValueForKey( arenaInfo, "num" ) );
298 UI_SPLevelMenu_SetMenuArena( 0, level, arenaInfo );
299 levelMenuInfo.selectedArenaInfo = arenaInfo;
300
301 levelMenuInfo.item_maps[0].generic.x = 256;
302 Bitmap_Init( &levelMenuInfo.item_maps[0] );
303 levelMenuInfo.item_maps[0].generic.bottom += 32;
304 levelMenuInfo.numMaps = 1;
305
306 levelMenuInfo.item_maps[1].generic.flags |= QMF_INACTIVE;
307 levelMenuInfo.item_maps[2].generic.flags |= QMF_INACTIVE;
308 levelMenuInfo.item_maps[3].generic.flags |= QMF_INACTIVE;
309 levelMenuInfo.levelPicNames[1][0] = 0;
310 levelMenuInfo.levelPicNames[2][0] = 0;
311 levelMenuInfo.levelPicNames[3][0] = 0;
312 levelMenuInfo.item_maps[1].shader = 0;
313 levelMenuInfo.item_maps[2].shader = 0;
314 levelMenuInfo.item_maps[3].shader = 0;
315 }
316 else {
317 levelMenuInfo.item_maps[0].generic.x = 46;
318 Bitmap_Init( &levelMenuInfo.item_maps[0] );
319 levelMenuInfo.item_maps[0].generic.bottom += 18;
320 levelMenuInfo.numMaps = 4;
321
322 for ( n = 0; n < 4; n++ ) {
323 level = selectedArenaSet * ARENAS_PER_TIER + n;
324 arenaInfo = UI_GetArenaInfoByNumber( level );
325 UI_SPLevelMenu_SetMenuArena( n, level, arenaInfo );
326 }
327
328 if( selectedArena != -1 ) {
329 levelMenuInfo.selectedArenaInfo = UI_GetArenaInfoByNumber( selectedArenaSet * ARENAS_PER_TIER + selectedArena );
330 }
331 }
332
333 // enable/disable arrows when they are valid/invalid
334 if ( selectedArenaSet == minTier ) {
335 levelMenuInfo.item_leftarrow.generic.flags |= ( QMF_INACTIVE | QMF_HIDDEN );
336 }
337 else {
338 levelMenuInfo.item_leftarrow.generic.flags &= ~( QMF_INACTIVE | QMF_HIDDEN );
339 }
340
341 if ( selectedArenaSet == maxTier ) {
342 levelMenuInfo.item_rightarrow.generic.flags |= ( QMF_INACTIVE | QMF_HIDDEN );
343 }
344 else {
345 levelMenuInfo.item_rightarrow.generic.flags &= ~( QMF_INACTIVE | QMF_HIDDEN );
346 }
347
348 UI_SPLevelMenu_SetBots();
349}

364static void UI_SPLevelMenu_ResetAction( qboolean result ) {
365 if( !result ) {
366 return;
367 }
368
369 // clear game variables
370 UI_NewGame();
371 if ( UI_GetSpecialArenaInfo( "training" ) ) {
372 trap_Cvar_SetValue( "ui_spSelection", -4 );
373 } else {
374 trap_Cvar_SetValue( "ui_spSelection", 0 );
375 }
376
377 // make the level select menu re-initialize
378 UI_PopMenu();
379 UI_SPLevelMenu();
380}

386 }

538=================
539UI_SPLevelMenu_MenuDraw
540=================
541*/
542#define LEVEL_DESC_LEFT_MARGIN 332
543
544static void UI_SPLevelMenu_MenuDraw( void ) {
545 int n, i;
546 int x, y;
547 vec4_t color;
548 int level;
549// int fraglimit;
550 int pad;
551 char buf[MAX_INFO_VALUE];
552 char string[64];
553
554 if( levelMenuInfo.reinit ) {
555 UI_PopMenu();
556 UI_SPLevelMenu();
557 return;
558 }
559
560 // draw player name
561 trap_Cvar_VariableStringBuffer( "name", string, 32 );
562 Q_CleanStr( string );
563 UI_DrawProportionalString( 320, PLAYER_Y, string, UI_CENTER|UI_SMALLFONT, color_orange );
564
565 // check for model changes
566 trap_Cvar_VariableStringBuffer( "model", buf, sizeof(buf) );
567 if( Q_stricmp( buf, levelMenuInfo.playerModel ) != 0 ) {
568 Q_strncpyz( levelMenuInfo.playerModel, buf, sizeof(levelMenuInfo.playerModel) );
569 PlayerIcon( levelMenuInfo.playerModel, levelMenuInfo.playerPicName, sizeof(levelMenuInfo.playerPicName) );
570 levelMenuInfo.item_player.shader = 0;
571 }
572
573 // standard menu drawing
574 Menu_Draw( &levelMenuInfo.menu );
575
576 // draw player award levels
577 y = AWARDS_Y;
578 i = 0;
579 for( n = 0; n < 6; n++ ) {
580 level = levelMenuInfo.awardLevels[n];
581 if( level > 0 ) {
582 if( i & 1 ) {
583 x = 224 - (i - 1 ) / 2 * (48 + 16);
584 }
585 else {
586 x = 368 + i / 2 * (48 + 16);
587 }
588 i++;
589
590 if( level == 1 ) {
591 continue;
592 }
593
594 if( level >= 1000000 ) {
595 Com_sprintf( string, sizeof(string), "%im", level / 1000000 );
596 }
597 else if( level >= 1000 ) {
598 Com_sprintf( string, sizeof(string), "%ik", level / 1000 );
599 }
600 else {
601 Com_sprintf( string, sizeof(string), "%i", level );
602 }
603
604 UI_DrawString( x + 24, y + 48, string, UI_CENTER, color_yellow );
605 }
606 }
607
608 UI_DrawProportionalString( 18, 38, va( "Tier %i", selectedArenaSet + 1 ), UI_LEFT|UI_SMALLFONT, color_orange );
609
610 for ( n = 0; n < levelMenuInfo.numMaps; n++ ) {
611 x = levelMenuInfo.item_maps[n].generic.x;
612 y = levelMenuInfo.item_maps[n].generic.y;
613 UI_FillRect( x, y + 96, 128, 18, color_black );
614 }
615
616 if ( selectedArenaSet > currentSet ) {
617 UI_DrawProportionalString( 320, 216, "ACCESS DENIED", UI_CENTER|UI_BIGFONT, color_red );
618 return;
619 }
620
621 // show levelshots for levels of current tier
622 Vector4Copy( color_white, color );
623 color[3] = 0.5+0.5*sin(uis.realtime/PULSE_DIVISOR);
624 for ( n = 0; n < levelMenuInfo.numMaps; n++ ) {
625 x = levelMenuInfo.item_maps[n].generic.x;
626 y = levelMenuInfo.item_maps[n].generic.y;
627
628 UI_DrawString( x + 64, y + 96, levelMenuInfo.levelNames[n], UI_CENTER|UI_SMALLFONT, color_orange );
629
630 if( levelMenuInfo.levelScores[n] == 1 ) {
631 UI_DrawHandlePic( x, y, 128, 96, levelMenuInfo.levelCompletePic[levelMenuInfo.levelScoresSkill[n] - 1] );
632 }
633
634 if ( n == selectedArena ) {
635 if( Menu_ItemAtCursor( &levelMenuInfo.menu ) == &levelMenuInfo.item_maps[n] ) {
636 trap_R_SetColor( color );
637 }
638 UI_DrawHandlePic( x-1, y-1, 130, 130 - 14, levelMenuInfo.levelSelectedPic );
639 trap_R_SetColor( NULL );
640 }
641 else if( Menu_ItemAtCursor( &levelMenuInfo.menu ) == &levelMenuInfo.item_maps[n] ) {
642 trap_R_SetColor( color );
643 UI_DrawHandlePic( x-31, y-30, 256, 256-27, levelMenuInfo.levelFocusPic);
644 trap_R_SetColor( NULL );
645 }
646 }
647
648 // show map name and long name of selected level
649 y = 192;
650 Q_strncpyz( buf, Info_ValueForKey( levelMenuInfo.selectedArenaInfo, "map" ), 20 );
651 Q_strupr( buf );
652 Com_sprintf( string, sizeof(string), "%s: %s", buf, Info_ValueForKey( levelMenuInfo.selectedArenaInfo, "longname" ) );
653 UI_DrawProportionalString( 320, y, string, UI_CENTER|UI_SMALLFONT, color_orange );
654
655// fraglimit = atoi( Info_ValueForKey( levelMenuInfo.selectedArenaInfo, "fraglimit" ) );
656// UI_DrawString( 18, 212, va("Frags %i", fraglimit) , UI_LEFT|UI_SMALLFONT, color_orange );
657
658 // draw bot opponents
659 y += 24;
660 pad = (7 - levelMenuInfo.numBots) * (64 + 26) / 2;
661 for( n = 0; n < levelMenuInfo.numBots; n++ ) {
662 x = 18 + pad + (64 + 26) * n;
663 if( levelMenuInfo.botPics[n] ) {
664 UI_DrawHandlePic( x, y, 64, 64, levelMenuInfo.botPics[n]);
665 }
666 else {
667 UI_FillRect( x, y, 64, 64, color_black );
668 UI_DrawProportionalString( x+22, y+18, "?", UI_BIGFONT, color_orange );
669 }
670 UI_DrawString( x, y + 64, levelMenuInfo.botNames[n], UI_SMALLFONT|UI_LEFT, color_orange );
671 }
672}

682
683 trap_R_RegisterShaderNoMip( ART_LEVELFRAME_FOCUS );

691 trap_R_RegisterShaderNoMip( ART_MAP_COMPLETE4 );
692 trap_R_RegisterShaderNoMip( ART_MAP_COMPLETE5 );
693 trap_R_RegisterShaderNoMip( ART_BACK0 );
694 trap_R_RegisterShaderNoMip( ART_BACK1 );
695 trap_R_RegisterShaderNoMip( ART_FIGHT0 );

697 trap_R_RegisterShaderNoMip( ART_RESET0 );
698 trap_R_RegisterShaderNoMip( ART_RESET1 );
699 trap_R_RegisterShaderNoMip( ART_CUSTOM0 );
700 trap_R_RegisterShaderNoMip( ART_CUSTOM1 );

702 for( n = 0; n < 6; n++ ) {

941/*
942=================
943UI_SPLevelMenu
944=================
945*/
946void UI_SPLevelMenu( void ) {
947 int level;
948 int trainingLevel;
949 const char *arenaInfo;
950
951 trainingTier = -1;
952 arenaInfo = UI_GetSpecialArenaInfo( "training" );
953 if( arenaInfo ) {
954 minTier = trainingTier;
955 trainingLevel = atoi( Info_ValueForKey( arenaInfo, "num" ) );
956 }
957 else {
958 minTier = 0;
959 trainingLevel = -2;
960 }
961
962 finalTier = UI_GetNumSPTiers();
963 arenaInfo = UI_GetSpecialArenaInfo( "final" );
964 if( arenaInfo ) {
965 maxTier = finalTier;
966 }
967 else {
968 maxTier = finalTier - 1;
969 if( maxTier < minTier ) {
970 maxTier = minTier;
971 }
972 }
973
974 level = UI_GetCurrentGame();
975 if ( level == -1 ) {
976 level = UI_GetNumSPArenas() - 1;
977 if( maxTier == finalTier ) {
978 level++;
979 }
980 }
981
982 if( level == trainingLevel ) {
983 currentSet = -1;
984 currentGame = 0;
985 }
986 else {
987 currentSet = level / ARENAS_PER_TIER;
988 currentGame = level % ARENAS_PER_TIER;
989 }
990
991 UI_SPLevelMenu_Init();
992 UI_PushMenu( &levelMenuInfo.menu );
993 Menu_SetCursorToItem( &levelMenuInfo.menu, &levelMenuInfo.item_next );
994}

55static char *op_string_range_dup(const char *_start,const char *_end){
56 size_t len;
57 char *ret;
58 OP_ASSERT(_start<=_end);
59 len=_end-_start;
60 /*This is to help avoid overflow elsewhere, later.*/
61 if(OP_UNLIKELY(len>=INT_MAX))return NULL;
62 ret=(char *)_ogg_malloc(sizeof(*ret)*(len+1));
63 if(OP_LIKELY(ret!=NULL)){
64 ret=(char *)memcpy(ret,_start,sizeof(*ret)*(len));
65 ret[len]='\0';
66 }
67 return ret;
68}

74static char *op_string_tolower(char *_s){
75 int i;
76 for(i=0;_s[i]!='\0';i++){
77 int c;
78 c=_s[i];
79 if(c>='A'&&c<='Z')c+='a'-'A';
80 _s[i]=(char)c;
81 }
82 return _s;
83}

112/*Check the <% HEXDIG HEXDIG> escapes of a URL for validity.
113 Return: 0 if valid, or a negative value on failure.*/
114static int op_validate_url_escapes(const char *_s){
115 int i;
116 for(i=0;_s[i];i++){
117 if(_s[i]=='%'){
118 if(OP_UNLIKELY(!isxdigit(_s[i+1]))
119 ||OP_UNLIKELY(!isxdigit(_s[i+2]))
120 /*RFC 3986 says %00 "should be rejected if the application is not
121 expecting to receive raw data within a component."*/
122 ||OP_UNLIKELY(_s[i+1]=='0'&&_s[i+2]=='0')){
123 return OP_FALSE;
124 }
125 i+=2;
126 }
127 }
128 return 0;
129}
130
131/*Convert a hex digit to its actual value.
132 _c: The hex digit to convert.
133 Presumed to be valid ('0'...'9', 'A'...'F', or 'a'...'f').
134 Return: The value of the digit, in the range [0,15].*/
135static int op_hex_value(int _c){
136 return _c>='a'?_c-'a'+10:_c>='A'?_c-'A'+10:_c-'0';
137}

139/*Unescape all the <% HEXDIG HEXDIG> sequences in a string in-place.
140 This does no validity checking.*/
141static char *op_unescape_url_component(char *_s){
142 int i;
143 int j;
144 for(i=j=0;_s[i];i++,j++){
145 if(_s[i]=='%'){
146 _s[i]=(char)(op_hex_value(_s[i+1])<<4|op_hex_value(_s[i+2]));
147 i+=2;
148 }
149 }
150 return _s;
151}

153/*Parse a file: URL.
154 This code is not meant to be fast: strspn() with large sets is likely to be
155 slow, but it is very convenient.
156 It is meant to be RFC 1738-compliant (as updated by RFC 3986).*/
157static const char *op_parse_file_url(const char *_src){
158 const char *scheme_end;
159 const char *path;
160 const char *path_end;
161 scheme_end=_src+strspn(_src,OP_URL_SCHEME);
162 if(OP_UNLIKELY(*scheme_end!=':')
163 ||scheme_end-_src!=4||op_strncasecmp(_src,"file",4)!=0){
164 /*Unsupported protocol.*/
165 return NULL;
166 }
167 /*Make sure all escape sequences are valid to simplify unescaping later.*/
168 if(OP_UNLIKELY(op_validate_url_escapes(scheme_end+1)<0))return NULL;
169 if(scheme_end[1]=='/'&&scheme_end[2]=='/'){
170 const char *host;
171 /*file: URLs can have a host!
172 Yeah, I was surprised, too, but that's what RFC 1738 says.
173 It also says, "The file URL scheme is unusual in that it does not specify
174 an Internet protocol or access method for such files; as such, its
175 utility in network protocols between hosts is limited," which is a mild
176 understatement.*/
177 host=scheme_end+3;
178 /*The empty host is what we expect.*/
179 if(OP_LIKELY(*host=='/'))path=host;
180 else{
181 const char *host_end;
182 char host_buf[28];
183 /*RFC 1738 says localhost "is interpreted as `the machine from which the
184 URL is being interpreted,'" so let's check for it.*/
185 host_end=host+strspn(host,OP_URL_PCHAR_BASE);
186 /*No <port> allowed.
187 This also rejects IP-Literals.*/
188 if(*host_end!='/')return NULL;
189 /*An escaped "localhost" can take at most 27 characters.*/
190 if(OP_UNLIKELY(host_end-host>27))return NULL;
191 memcpy(host_buf,host,sizeof(*host_buf)*(host_end-host));
192 host_buf[host_end-host]='\0';
193 op_unescape_url_component(host_buf);
194 op_string_tolower(host_buf);
195 /*Some other host: give up.*/
196 if(OP_UNLIKELY(strcmp(host_buf,"localhost")!=0))return NULL;
197 path=host_end;
198 }
199 }
200 else path=scheme_end+1;
201 path_end=path+strspn(path,OP_URL_PATH);
202 /*This will reject a <query> or <fragment> component, too.
203 I don't know what to do with queries, but a temporal fragment would at
204 least make sense.
205 RFC 1738 pretty clearly defines a <searchpart> that's equivalent to the
206 RFC 3986 <query> component for other schemes, but not the file: scheme,
207 so I'm going to just reject it.*/
208 if(*path_end!='\0')return NULL;
209 return path;
210}

262/*The usage of FD_SET() below is O(N^2).
263 This is okay because select() is limited to 64 sockets in Winsock, anyway.
264 In practice, we only ever call it with one or two sockets.*/
265static int op_poll_win32(struct pollfd *_fds,nfds_t _nfds,int _timeout){
266 struct timeval tv;
267 fd_set ifds;
268 fd_set ofds;
269 fd_set efds;
270 nfds_t i;
271 int ret;
272 FD_ZERO(&ifds);
273 FD_ZERO(&ofds);
274 FD_ZERO(&efds);
275 for(i=0;i<_nfds;i++){
276 _fds[i].revents=0;
277 if(_fds[i].events&POLLIN)FD_SET(_fds[i].fd,&ifds);
278 if(_fds[i].events&POLLOUT)FD_SET(_fds[i].fd,&ofds);
279 FD_SET(_fds[i].fd,&efds);
280 }
281 if(_timeout>=0){
282 tv.tv_sec=_timeout/1000;
283 tv.tv_usec=(_timeout%1000)*1000;
284 }
285 ret=select(-1,&ifds,&ofds,&efds,_timeout<0?NULL:&tv);
286 if(ret>0){
287 for(i=0;i<_nfds;i++){
288 if(FD_ISSET(_fds[i].fd,&ifds))_fds[i].revents|=POLLIN;
289 if(FD_ISSET(_fds[i].fd,&ofds))_fds[i].revents|=POLLOUT;
290 /*This isn't correct: there are several different things that might have
291 happened to a fd in efds, but I don't know a good way to distinguish
292 them without more context from the caller.
293 It's okay, because we don't actually check any of these bits, we just
294 need _some_ bit set.*/
295 if(FD_ISSET(_fds[i].fd,&efds))_fds[i].revents|=POLLHUP;
296 }
297 }
298 return ret;
299}

398/*The amount of data to request after a seek.
399 This is a trade-off between read throughput after a seek vs. the the ability
400 to quickly perform another seek with the same connection.*/
401# define OP_PIPELINE_CHUNK_SIZE (32*(opus_int32)1024)
402/*Subsequent chunks are requested with larger and larger sizes until they pass
403 this threshold, after which we just ask for the rest of the resource.*/
404# define OP_PIPELINE_CHUNK_SIZE_MAX (1024*(opus_int32)1024)
405/*This is the maximum number of requests we'll make with a single connection.
406 Many servers will simply disconnect after we attempt some number of requests,
407 possibly without sending a Connection: close header, meaning we won't
408 discover it until we try to read beyond the end of the current chunk.
409 We can reconnect when that happens, but this is slow.
410 Instead, we impose a limit ourselves (set to the default for Apache
411 installations and thus likely the most common value in use).*/

414 OP_PIPELINE_CHUNK_SIZE and doubling each time, until we exceed

417 requests left, to reduce the chances we'll have to open a new connection
418 while reading forward afterwards.*/
419# define OP_PIPELINE_MIN_REQUESTS (7)

448/*Parse a URL.
449 This code is not meant to be fast: strspn() with large sets is likely to be
450 slow, but it is very convenient.
451 It is meant to be RFC 3986-compliant.
452 We currently do not support IRIs (Internationalized Resource Identifiers,
453 RFC 3987).
454 Callers should translate them to URIs first.*/
455static int op_parse_url_impl(OpusParsedURL *_dst,const char *_src){
456 const char *scheme_end;
457 const char *authority;
458 const char *userinfo_end;
459 const char *user;
460 const char *user_end;
461 const char *pass;
462 const char *hostport;
463 const char *hostport_end;
464 const char *host_end;
465 const char *port;
466 opus_int32 port_num;
467 const char *port_end;
468 const char *path;
469 const char *path_end;
470 const char *uri_end;
471 scheme_end=_src+strspn(_src,OP_URL_SCHEME);
472 if(OP_UNLIKELY(*scheme_end!=':')
473 ||OP_UNLIKELY(scheme_end-_src<4)||OP_UNLIKELY(scheme_end-_src>5)
474 ||OP_UNLIKELY(op_strncasecmp(_src,"https",scheme_end-_src)!=0)){
475 /*Unsupported protocol.*/
476 return OP_EIMPL;
477 }
478 if(OP_UNLIKELY(scheme_end[1]!='/')||OP_UNLIKELY(scheme_end[2]!='/')){
479 /*We require an <authority> component.*/
480 return OP_EINVAL;
481 }
482 authority=scheme_end+3;
483 /*Make sure all escape sequences are valid to simplify unescaping later.*/
484 if(OP_UNLIKELY(op_validate_url_escapes(authority)<0))return OP_EINVAL;
485 /*Look for a <userinfo> component.*/
486 userinfo_end=authority+strspn(authority,OP_URL_PCHAR_NA);
487 if(*userinfo_end=='@'){
488 /*Found one.*/
489 user=authority;
490 /*Look for a password (yes, clear-text passwords are deprecated, I know,
491 but what else are people supposed to use? use SSL if you care).*/
492 user_end=authority+strspn(authority,OP_URL_PCHAR_BASE);
493 if(*user_end==':')pass=user_end+1;
494 else pass=NULL;
495 hostport=userinfo_end+1;
496 }
497 else{
498 /*We shouldn't have to initialize user_end, but gcc is too dumb to figure
499 out that user!=NULL below means we didn't take this else branch.*/
500 user=user_end=NULL;
501 pass=NULL;
502 hostport=authority;
503 }
504 /*Try to figure out where the <host> component ends.*/
505 if(hostport[0]=='['){
506 hostport++;
507 /*We have an <IP-literal>, which can contain colons.*/
508 hostport_end=host_end=hostport+strspn(hostport,OP_URL_PCHAR_NA);
509 if(OP_UNLIKELY(*hostport_end++!=']'))return OP_EINVAL;
510 }
511 /*Currently we don't support IDNA (RFC 5894), because I don't want to deal
512 with the policy about which domains should not be internationalized to
513 avoid confusing similarities.
514 Give this API Punycode (RFC 3492) domain names instead.*/
515 else hostport_end=host_end=hostport+strspn(hostport,OP_URL_PCHAR_BASE);
516 /*TODO: Validate host.*/
517 /*Is there a port number?*/
518 port_num=-1;
519 if(*hostport_end==':'){
520 int i;
521 port=hostport_end+1;
522 port_end=port+strspn(port,OP_URL_DIGIT);
523 path=port_end;
524 /*Not part of RFC 3986, but require port numbers in the range 0...65535.*/
525 if(OP_LIKELY(port_end-port>0)){
526 while(*port=='0')port++;
527 if(OP_UNLIKELY(port_end-port>5))return OP_EINVAL;
528 port_num=0;
529 for(i=0;i<port_end-port;i++)port_num=port_num*10+port[i]-'0';
530 if(OP_UNLIKELY(port_num>65535))return OP_EINVAL;
531 }
532 }
533 else path=hostport_end;
534 path_end=path+strspn(path,OP_URL_PATH);
535 /*If the path is not empty, it must begin with a '/'.*/
536 if(OP_LIKELY(path_end>path)&&OP_UNLIKELY(path[0]!='/'))return OP_EINVAL;
537 /*Consume the <query> component, if any (right now we don't split this out
538 from the <path> component).*/
539 if(*path_end=='?')path_end=path_end+strspn(path_end,OP_URL_QUERY_FRAG);
540 /*Discard the <fragment> component, if any.
541 This doesn't get sent to the server.
542 Some day we should add support for Media Fragment URIs
543 <http://www.w3.org/TR/media-frags/>.*/
544 if(*path_end=='#')uri_end=path_end+1+strspn(path_end+1,OP_URL_QUERY_FRAG);
545 else uri_end=path_end;
546 /*If there's anything left, this was not a valid URL.*/
547 if(OP_UNLIKELY(*uri_end!='\0'))return OP_EINVAL;
548 _dst->scheme=op_string_range_dup(_src,scheme_end);
549 if(OP_UNLIKELY(_dst->scheme==NULL))return OP_EFAULT;
550 op_string_tolower(_dst->scheme);
551 if(user!=NULL){
552 _dst->user=op_string_range_dup(user,user_end);
553 if(OP_UNLIKELY(_dst->user==NULL))return OP_EFAULT;
554 op_unescape_url_component(_dst->user);
555 /*Unescaping might have created a ':' in the username.
556 That's not allowed by RFC 2617's Basic Authentication Scheme.*/
557 if(OP_UNLIKELY(strchr(_dst->user,':')!=NULL))return OP_EINVAL;
558 }
559 else _dst->user=NULL;
560 if(pass!=NULL){
561 _dst->pass=op_string_range_dup(pass,userinfo_end);
562 if(OP_UNLIKELY(_dst->pass==NULL))return OP_EFAULT;
563 op_unescape_url_component(_dst->pass);
564 }
565 else _dst->pass=NULL;
566 _dst->host=op_string_range_dup(hostport,host_end);
567 if(OP_UNLIKELY(_dst->host==NULL))return OP_EFAULT;
568 if(port_num<0){

580 _dst->path[0]='/';
581 return 0;
582}

660static int op_sb_append(OpusStringBuf *_sb,const char *_s,int _len){

716}

894/*Close the connection and move it to the free list.
895 _stream: The stream containing the free list.
896 _conn: The connection to close.
897 _pnext: The linked-list pointer currently pointing to this connection.
898 _gracefully: Whether or not to shut down cleanly.*/
899static void op_http_conn_close(OpusHTTPStream *_stream,OpusHTTPConn *_conn,
900 OpusHTTPConn **_pnext,int _gracefully){
901 /*If we don't shut down gracefully, the server MUST NOT re-use our session
902 according to RFC 2246, because it can't tell the difference between an
903 abrupt close and a truncation attack.
904 So we shut down gracefully if we can.
905 However, we will not wait if this would block (it's not worth the savings
906 from session resumption to do so).
907 Clients (that's us) MAY resume a TLS session that ended with an incomplete
908 close, according to RFC 2818, so there's no reason to make sure the server
909 shut things down gracefully.*/
910 if(_gracefully&&_conn->ssl_conn!=NULL)SSL_shutdown(_conn->ssl_conn);
911 op_http_conn_clear(_conn);
912 _conn->next_pos=-1;
913 _conn->ssl_conn=NULL;
914 _conn->fd=OP_INVALID_SOCKET;
915 OP_ASSERT(*_pnext==_conn);
916 *_pnext=_conn->next;
917 _conn->next=_stream->free_head;
918 _stream->free_head=_conn;
919}

932}

934static int op_http_conn_write_fully(OpusHTTPConn *_conn,
935 const char *_buf,int _buf_size){
936 struct pollfd fd;
937 SSL *ssl_conn;
938 fd.fd=_conn->fd;
939 ssl_conn=_conn->ssl_conn;
940 while(_buf_size>0){
941 int err;
942 if(ssl_conn!=NULL){
943 int ret;
944 ret=SSL_write(ssl_conn,_buf,_buf_size);
945 if(ret>0){
946 /*Wrote some data.*/
947 _buf+=ret;
948 _buf_size-=ret;
949 continue;
950 }
951 /*Connection closed.*/
952 else if(ret==0)return OP_FALSE;
953 err=SSL_get_error(ssl_conn,ret);
954 /*Yes, renegotiations can cause SSL_write() to block for reading.*/
955 if(err==SSL_ERROR_WANT_READ)fd.events=POLLIN;
956 else if(err==SSL_ERROR_WANT_WRITE)fd.events=POLLOUT;
957 else return OP_FALSE;
958 }
959 else{
960 ssize_t ret;
961 op_reset_errno();
962 ret=send(fd.fd,_buf,_buf_size,0);
963 if(ret>0){
964 _buf+=ret;
965 _buf_size-=ret;
966 continue;
967 }
968 err=op_errno();
969 if(err!=EAGAIN&&err!=EWOULDBLOCK)return OP_FALSE;
970 fd.events=POLLOUT;
971 }
972 if(poll(&fd,1,OP_POLL_TIMEOUT_MS)<=0)return OP_FALSE;
973 }
974 return 0;
975}

977static int op_http_conn_estimate_available(OpusHTTPConn *_conn){
978 int available;
979 int ret;
980 ret=ioctl(_conn->fd,FIONREAD,&available);
981 if(ret<0)available=0;
982 /*This requires the SSL read_ahead flag to be unset to work.
983 We ignore partial records as well as the protocol overhead for any pending
984 bytes.
985 This means we might return somewhat less than can truly be read without
986 blocking (if there's a partial record).
987 This is okay, because we're using this value to estimate network transfer
988 time, and we _have_ already received those bytes.
989 We also might return slightly more (due to protocol overhead), but that's
990 small enough that it probably doesn't matter.*/
991 if(_conn->ssl_conn!=NULL)available+=SSL_pending(_conn->ssl_conn);
992 return available;
993}

995static opus_int32 op_time_diff_ms(const struct timeb *_end,
996 const struct timeb *_start){
997 opus_int64 dtime;
998 dtime=_end->time-(opus_int64)_start->time;
999 OP_ASSERT(_end->millitm<1000);
1000 OP_ASSERT(_start->millitm<1000);
1001 if(OP_UNLIKELY(dtime>(OP_INT32_MAX-1000)/1000))return OP_INT32_MAX;
1002 if(OP_UNLIKELY(dtime<(OP_INT32_MIN+1000)/1000))return OP_INT32_MIN;
1003 return (opus_int32)dtime*1000+_end->millitm-_start->millitm;
1004}

1006/*Update the read rate estimate for this connection.*/
1007static void op_http_conn_read_rate_update(OpusHTTPConn *_conn){
1008 struct timeb read_time;
1009 opus_int32 read_delta_ms;
1010 opus_int64 read_delta_bytes;
1011 opus_int64 read_rate;
1012 read_delta_bytes=_conn->read_bytes;
1013 if(read_delta_bytes<=0)return;
1014 ftime(&read_time);
1015 read_delta_ms=op_time_diff_ms(&read_time,&_conn->read_time);
1016 read_rate=_conn->read_rate;
1017 read_delta_ms=OP_MAX(read_delta_ms,1);
1018 read_rate+=read_delta_bytes*1000/read_delta_ms-read_rate+4>>3;
1019 *&_conn->read_time=*&read_time;
1020 _conn->read_bytes=0;
1021 _conn->read_rate=read_rate;
1022}

1024/*Tries to read from the given connection.
1025 [out] _buf: Returns the data read.
1026 _buf_size: The size of the buffer.
1027 _blocking: Whether or not to block until some data is retrieved.
1028 Return: A positive number of bytes read on success.
1029 0: The read would block, or the connection was closed.
1030 OP_EREAD: There was a fatal read error.*/
1031static int op_http_conn_read(OpusHTTPConn *_conn,
1032 char *_buf,int _buf_size,int _blocking){
1033 struct pollfd fd;
1034 SSL *ssl_conn;
1035 int nread;
1036 int nread_unblocked;
1037 fd.fd=_conn->fd;
1038 ssl_conn=_conn->ssl_conn;
1039 nread=nread_unblocked=0;
1040 /*RFC 2818 says "client implementations MUST treat any premature closes as
1041 errors and the data received as potentially truncated," so we make very
1042 sure to report read errors upwards.*/
1043 do{
1044 int err;
1045 if(ssl_conn!=NULL){
1046 int ret;
1047 ret=SSL_read(ssl_conn,_buf+nread,_buf_size-nread);
1048 OP_ASSERT(ret<=_buf_size-nread);
1049 if(ret>0){
1050 /*Read some data.
1051 Keep going to see if there's more.*/
1052 nread+=ret;
1053 nread_unblocked+=ret;
1054 continue;
1055 }
1056 /*If we already read some data, return it right now.*/
1057 if(nread>0)break;
1058 err=SSL_get_error(ssl_conn,ret);
1059 if(ret==0){
1060 /*Connection close.
1061 Check for a clean shutdown to prevent truncation attacks.
1062 This check always succeeds for SSLv2, as it has no "close notify"
1063 message and thus can't verify an orderly shutdown.*/
1064 return err==SSL_ERROR_ZERO_RETURN?0:OP_EREAD;
1065 }
1066 if(err==SSL_ERROR_WANT_READ)fd.events=POLLIN;
1067 /*Yes, renegotiations can cause SSL_read() to block for writing.*/
1068 else if(err==SSL_ERROR_WANT_WRITE)fd.events=POLLOUT;
1069 /*Some other error.*/
1070 else return OP_EREAD;
1071 }
1072 else{
1073 ssize_t ret;
1074 op_reset_errno();
1075 ret=recv(fd.fd,_buf+nread,_buf_size-nread,0);
1076 OP_ASSERT(ret<=_buf_size-nread);
1077 if(ret>0){
1078 /*Read some data.
1079 Keep going to see if there's more.*/
1080 nread+=ret;
1081 nread_unblocked+=ret;
1082 continue;
1083 }
1084 /*If we already read some data or the connection was closed, return
1085 right now.*/
1086 if(ret==0||nread>0)break;
1087 err=op_errno();
1088 if(err!=EAGAIN&&err!=EWOULDBLOCK)return OP_EREAD;
1089 fd.events=POLLIN;
1090 }
1091 _conn->read_bytes+=nread_unblocked;
1092 op_http_conn_read_rate_update(_conn);
1093 nread_unblocked=0;
1094 if(!_blocking)break;
1095 /*Need to wait to get any data at all.*/
1096 if(poll(&fd,1,OP_POLL_TIMEOUT_MS)<=0)return OP_EREAD;
1097 }
1098 while(nread<_buf_size);
1099 _conn->read_bytes+=nread_unblocked;
1100 return nread;
1101}

1103/*Tries to look at the pending data for a connection without consuming it.
1104 [out] _buf: Returns the data at which we're peeking.
1105 _buf_size: The size of the buffer.*/
1106static int op_http_conn_peek(OpusHTTPConn *_conn,char *_buf,int _buf_size){
1107 struct pollfd fd;
1108 SSL *ssl_conn;
1109 int ret;
1110 fd.fd=_conn->fd;
1111 ssl_conn=_conn->ssl_conn;
1112 for(;;){
1113 int err;
1114 if(ssl_conn!=NULL){
1115 ret=SSL_peek(ssl_conn,_buf,_buf_size);
1116 /*Either saw some data or the connection was closed.*/
1117 if(ret>=0)return ret;
1118 err=SSL_get_error(ssl_conn,ret);
1119 if(err==SSL_ERROR_WANT_READ)fd.events=POLLIN;
1120 /*Yes, renegotiations can cause SSL_peek() to block for writing.*/
1121 else if(err==SSL_ERROR_WANT_WRITE)fd.events=POLLOUT;
1122 else return 0;
1123 }
1124 else{
1125 op_reset_errno();
1126 ret=(int)recv(fd.fd,_buf,_buf_size,MSG_PEEK);
1127 /*Either saw some data or the connection was closed.*/
1128 if(ret>=0)return ret;
1129 err=op_errno();
1130 if(err!=EAGAIN&&err!=EWOULDBLOCK)return 0;
1131 fd.events=POLLIN;
1132 }
1133 /*Need to wait to get any data at all.*/
1134 if(poll(&fd,1,OP_POLL_TIMEOUT_MS)<=0)return 0;
1135 }
1136}

1138/*When parsing response headers, RFC 2616 mandates that all lines end in CR LF.
1139 However, even in the year 2012, I have seen broken servers use just a LF.
1140 This is the evil that Postel's advice from RFC 761 breeds.*/
1141
1142/*Reads the entirety of a response to an HTTP request into the response buffer.
1143 Actual parsing and validation is done later.
1144 Return: The number of bytes in the response on success, OP_EREAD if the
1145 connection was closed before reading any data, or another negative
1146 value on any other error.*/
1147static int op_http_conn_read_response(OpusHTTPConn *_conn,
1148 OpusStringBuf *_response){
1149 int ret;
1150 _response->nbuf=0;
1151 ret=op_sb_ensure_capacity(_response,OP_RESPONSE_SIZE_MIN);
1152 if(OP_UNLIKELY(ret<0))return ret;
1153 for(;;){
1154 char *buf;
1155 int size;
1156 int capacity;
1157 int read_limit;
1158 int terminated;
1159 size=_response->nbuf;
1160 capacity=_response->cbuf-1;
1161 if(OP_UNLIKELY(size>=capacity)){
1162 ret=op_sb_grow(_response,OP_RESPONSE_SIZE_MAX);
1163 if(OP_UNLIKELY(ret<0))return ret;
1164 capacity=_response->cbuf-1;
1165 /*The response was too large.
1166 This prevents a bad server from running us out of memory.*/
1167 if(OP_UNLIKELY(size>=capacity))return OP_EIMPL;
1168 }
1169 buf=_response->buf;
1170 ret=op_http_conn_peek(_conn,buf+size,capacity-size);
1171 if(OP_UNLIKELY(ret<=0))return size<=0?OP_EREAD:OP_FALSE;
1172 /*We read some data.*/
1173 /*Make sure the starting characters are "HTTP".
1174 Otherwise we could wind up waiting forever for a response from
1175 something that is not an HTTP server.*/
1176 if(size<4&&op_strncasecmp(buf,"HTTP",OP_MIN(size+ret,4))!=0){
1177 return OP_FALSE;
1178 }
1179 /*How far can we read without passing the "\r\n\r\n" terminator?*/
1180 buf[size+ret]='\0';
1181 terminated=0;
1182 for(read_limit=OP_MAX(size-3,0);read_limit<size+ret;read_limit++){
1183 /*We don't look for the leading '\r' thanks to broken servers.*/
1184 if(buf[read_limit]=='\n'){
1185 if(buf[read_limit+1]=='\r'&&OP_LIKELY(buf[read_limit+2]=='\n')){
1186 terminated=3;
1187 break;
1188 }
1189 /*This case is for broken servers.*/
1190 else if(OP_UNLIKELY(buf[read_limit+1]=='\n')){
1191 terminated=2;
1192 break;
1193 }
1194 }
1195 }
1196 read_limit+=terminated;
1197 OP_ASSERT(size<=read_limit);
1198 OP_ASSERT(read_limit<=size+ret);
1199 /*Actually consume that data.*/
1200 ret=op_http_conn_read(_conn,buf+size,read_limit-size,1);
1201 if(OP_UNLIKELY(ret<=0))return OP_FALSE;
1202 size+=ret;
1203 buf[size]='\0';
1204 _response->nbuf=size;
1205 /*We found the terminator and read all the data up to and including it.*/
1206 if(terminated&&OP_LIKELY(size>=read_limit))return size;
1207 }
1208 return OP_EIMPL;
1209}

1230/*Return: The amount of linear white space (LWS) at the start of _s.*/
1231static int op_http_lwsspn(const char *_s){
1232 int i;
1233 for(i=0;;){
1234 if(_s[0]=='\r'&&_s[1]=='\n'&&(_s[2]=='\t'||_s[2]==' '))i+=3;
1235 /*This case is for broken servers.*/
1236 else if(_s[0]=='\n'&&(_s[1]=='\t'||_s[1]==' '))i+=2;
1237 else if(_s[i]=='\t'||_s[i]==' ')i++;
1238 else return i;
1239 }
1240}

1242static char *op_http_parse_status_line(int *_v1_1_compat,
1243 char **_status_code,char *_response){
1244 char *next;
1245 char *status_code;
1246 int v1_1_compat;
1247 size_t d;
1248 /*RFC 2616 Section 6.1 does not say that the tokens in the Status-Line cannot
1249 be separated by optional LWS, but since it specifically calls out where
1250 spaces are to be placed and that CR and LF are not allowed except at the
1251 end, I am assuming this to be true.*/
1252 /*We already validated that this starts with "HTTP"*/
1253 OP_ASSERT(op_strncasecmp(_response,"HTTP",4)==0);
1254 next=_response+4;
1255 if(OP_UNLIKELY(*next++!='/'))return NULL;
1256 d=strspn(next,OP_HTTP_DIGIT);
1257 /*"Leading zeros MUST be ignored by recipients."*/
1258 while(*next=='0'){
1259 next++;
1260 OP_ASSERT(d>0);
1261 d--;
1262 }
1263 /*We only support version 1.x*/
1264 if(OP_UNLIKELY(d!=1)||OP_UNLIKELY(*next++!='1'))return NULL;
1265 if(OP_UNLIKELY(*next++!='.'))return NULL;
1266 d=strspn(next,OP_HTTP_DIGIT);
1267 if(OP_UNLIKELY(d<=0))return NULL;
1268 /*"Leading zeros MUST be ignored by recipients."*/
1269 while(*next=='0'){
1270 next++;
1271 OP_ASSERT(d>0);
1272 d--;
1273 }
1274 /*We don't need to parse the version number.
1275 Any non-zero digit means it's greater than 1.*/
1276 v1_1_compat=d>0;
1277 next+=d;
1278 if(OP_UNLIKELY(*next++!=' '))return NULL;
1279 status_code=next;
1280 d=strspn(next,OP_HTTP_DIGIT);
1281 if(OP_UNLIKELY(d!=3))return NULL;
1282 next+=d;
1283 /*The Reason-Phrase can be empty, but the space must be here.*/
1284 if(OP_UNLIKELY(*next++!=' '))return NULL;
1285 next+=strcspn(next,OP_HTTP_CREASON_PHRASE);
1286 /*We are not mandating this be present thanks to broken servers.*/
1287 if(OP_LIKELY(*next=='\r'))next++;
1288 if(OP_UNLIKELY(*next++!='\n'))return NULL;
1289 if(_v1_1_compat!=NULL)*_v1_1_compat=v1_1_compat;
1290 *_status_code=status_code;
1291 return next;
1292}

1294/*Get the next response header.
1295 [out] _header: The header token, NUL-terminated, with leading and trailing
1296 whitespace stripped, and converted to lower case (to simplify
1297 case-insensitive comparisons), or NULL if there are no more
1298 response headers.
1299 [out] _cdr: The remaining contents of the header, excluding the initial
1300 colon (':') and the terminating CRLF ("\r\n"),
1301 NUL-terminated, and with leading and trailing whitespace
1302 stripped, or NULL if there are no more response headers.
1303 [inout] _s: On input, this points to the start of the current line of the
1304 response headers.
1305 On output, it points to the start of the first line following
1306 this header, or NULL if there are no more response headers.
1307 Return: 0 on success, or a negative value on failure.*/
1308static int op_http_get_next_header(char **_header,char **_cdr,char **_s){
1309 char *header;
1310 char *header_end;
1311 char *cdr;
1312 char *cdr_end;
1313 char *next;
1314 size_t d;
1315 next=*_s;
1316 /*The second case is for broken servers.*/
1317 if(next[0]=='\r'&&next[1]=='\n'||OP_UNLIKELY(next[0]=='\n')){
1318 /*No more headers.*/
1319 *_header=NULL;
1320 *_cdr=NULL;
1321 *_s=NULL;
1322 return 0;
1323 }
1324 header=next+op_http_lwsspn(next);
1325 d=strcspn(header,OP_HTTP_CTOKEN);
1326 if(OP_UNLIKELY(d<=0))return OP_FALSE;
1327 header_end=header+d;
1328 next=header_end+op_http_lwsspn(header_end);
1329 if(OP_UNLIKELY(*next++!=':'))return OP_FALSE;
1330 next+=op_http_lwsspn(next);
1331 cdr=next;
1332 do{
1333 cdr_end=next+strcspn(next,OP_HTTP_CTLS);
1334 next=cdr_end+op_http_lwsspn(cdr_end);
1335 }
1336 while(next>cdr_end);
1337 /*We are not mandating this be present thanks to broken servers.*/
1338 if(OP_LIKELY(*next=='\r'))next++;
1339 if(OP_UNLIKELY(*next++!='\n'))return OP_FALSE;
1340 *header_end='\0';
1341 *cdr_end='\0';
1342 /*Field names are case-insensitive.*/
1343 op_string_tolower(header);
1344 *_header=header;
1345 *_cdr=cdr;
1346 *_s=next;
1347 return 0;
1348}

1350static opus_int64 op_http_parse_nonnegative_int64(const char **_next,
1351 const char *_cdr){
1352 const char *next;
1353 opus_int64 ret;
1354 int i;
1355 next=_cdr+strspn(_cdr,OP_HTTP_DIGIT);
1356 *_next=next;
1357 if(OP_UNLIKELY(next<=_cdr))return OP_FALSE;
1358 while(*_cdr=='0')_cdr++;
1359 if(OP_UNLIKELY(next-_cdr>19))return OP_EIMPL;
1360 ret=0;
1361 for(i=0;i<next-_cdr;i++){
1362 int digit;
1363 digit=_cdr[i]-'0';
1364 /*Check for overflow.*/
1365 if(OP_UNLIKELY(ret>(OP_INT64_MAX-9)/10+(digit<=7)))return OP_EIMPL;
1366 ret=ret*10+digit;
1367 }
1368 return ret;
1369}

1379static int op_http_parse_content_range(opus_int64 *_first,opus_int64 *_last,
1380 opus_int64 *_length,const char *_cdr){
1381 opus_int64 first;
1382 opus_int64 last;
1383 opus_int64 length;
1384 size_t d;
1385 if(OP_UNLIKELY(op_strncasecmp(_cdr,"bytes",5)!=0))return OP_FALSE;
1386 _cdr+=5;
1387 d=op_http_lwsspn(_cdr);
1388 if(OP_UNLIKELY(d<=0))return OP_FALSE;
1389 _cdr+=d;
1390 if(*_cdr!='*'){
1391 first=op_http_parse_nonnegative_int64(&_cdr,_cdr);
1392 if(OP_UNLIKELY(first<0))return (int)first;
1393 _cdr+=op_http_lwsspn(_cdr);
1394 if(*_cdr++!='-')return OP_FALSE;
1395 _cdr+=op_http_lwsspn(_cdr);
1396 last=op_http_parse_nonnegative_int64(&_cdr,_cdr);
1397 if(OP_UNLIKELY(last<0))return (int)last;
1398 _cdr+=op_http_lwsspn(_cdr);
1399 }
1400 else{
1401 /*This is for a 416 response (Requested range not satisfiable).*/
1402 first=last=-1;
1403 _cdr++;
1404 }
1405 if(OP_UNLIKELY(*_cdr++!='/'))return OP_FALSE;
1406 if(*_cdr!='*'){
1407 length=op_http_parse_nonnegative_int64(&_cdr,_cdr);
1408 if(OP_UNLIKELY(length<0))return (int)length;
1409 }
1410 else{
1411 /*The total length is unspecified.*/
1412 _cdr++;
1413 length=-1;
1414 }
1415 if(OP_UNLIKELY(*_cdr!='\0'))return OP_FALSE;
1416 if(OP_UNLIKELY(last<first))return OP_FALSE;
1417 if(length>=0&&OP_UNLIKELY(last>=length))return OP_FALSE;
1418 *_first=first;
1419 *_last=last;
1420 *_length=length;
1421 return 0;
1422}

1424/*Parse the Connection response header and look for a "close" token.
1425 Return: 1 if a "close" token is found, 0 if it's not found, and a negative
1426 value on error.*/
1427static int op_http_parse_connection(char *_cdr){
1428 size_t d;
1429 int ret;
1430 ret=0;
1431 for(;;){
1432 d=strcspn(_cdr,OP_HTTP_CTOKEN);
1433 if(OP_UNLIKELY(d<=0))return OP_FALSE;
1434 if(op_strncasecmp(_cdr,"close",(int)d)==0)ret=1;
1435 /*We're supposed to strip and ignore any headers mentioned in the
1436 Connection header if this response is from an HTTP/1.0 server (to
1437 work around forwarding of hop-by-hop headers by old proxies), but the
1438 only hop-by-hop header we look at is Connection itself.
1439 Everything else is a well-defined end-to-end header, and going back and
1440 undoing the things we did based on already-examined headers would be
1441 hard (since we only scan them once, in a destructive manner).
1442 Therefore we just ignore all the other tokens.*/
1443 _cdr+=d;
1444 d=op_http_lwsspn(_cdr);
1445 if(d<=0)break;
1446 _cdr+=d;
1447 }
1448 return OP_UNLIKELY(*_cdr!='\0')?OP_FALSE:ret;
1449}

1611/*Match a host name against a host with a possible wildcard pattern according
1612 to the rules of RFC 6125 Section 6.4.3.
1613 Return: 0 if the pattern doesn't match, and a non-zero value if it does.*/
1614static int op_http_hostname_match(const char *_host,size_t _host_len,
1615 ASN1_STRING *_pattern){
1616 const char *pattern;
1617 size_t host_label_len;
1618 size_t host_suffix_len;
1619 size_t pattern_len;
1620 size_t pattern_label_len;
1621 size_t pattern_prefix_len;
1622 size_t pattern_suffix_len;
1623 pattern=(const char *)ASN1_STRING_data(_pattern);
1624 pattern_len=strlen(pattern);
1625 /*Check the pattern for embedded NULs.*/
1626 if(OP_UNLIKELY(pattern_len!=(size_t)ASN1_STRING_length(_pattern)))return 0;
1627 pattern_label_len=strcspn(pattern,".");
1628 OP_ASSERT(pattern_label_len<=pattern_len);
1629 pattern_prefix_len=strcspn(pattern,"*");
1630 if(pattern_prefix_len>=pattern_label_len){
1631 /*"The client SHOULD NOT attempt to match a presented identifier in which
1632 the wildcard character comprises a label other than the left-most label
1633 (e.g., do not match bar.*.example.net)." [RFC 6125 Section 6.4.3]*/
1634 if(pattern_prefix_len<pattern_len)return 0;
1635 /*If the pattern does not contain a wildcard in the first element, do an
1636 exact match.
1637 Don't use the system strcasecmp here, as that uses the locale and
1638 RFC 4343 makes clear that DNS's case-insensitivity only applies to
1639 the ASCII range.*/
1640 return _host_len==pattern_len&&op_strncasecmp(_host,pattern,_host_len)==0;
1641 }
1642 /*"However, the client SHOULD NOT attempt to match a presented identifier
1643 where the wildcard character is embedded within an A-label or U-label of
1644 an internationalized domain name." [RFC 6125 Section 6.4.3]*/
1645 if(op_strncasecmp(pattern,"xn--",4)==0)return 0;
1646 host_label_len=strcspn(_host,".");
1647 /*Make sure the host has at least two dots, to prevent the wildcard match
1648 from being ridiculously wide.
1649 We should have already checked to ensure it had at least one.*/
1650 if(OP_UNLIKELY(_host[host_label_len]!='.')
1651 ||strchr(_host+host_label_len+1,'.')==NULL){
1652 return 0;
1653 }
1654 OP_ASSERT(host_label_len<_host_len);
1655 /*"If the wildcard character is the only character of the left-most label in
1656 the presented identifier, the client SHOULD NOT compare against anything
1657 but the left-most label of the reference identifier (e.g., *.example.com
1658 would match foo.example.com but not bar.foo.example.com)." [RFC 6125
1659 Section 6.4.3]
1660 This is really confusingly worded, as we check this by actually comparing
1661 the rest of the pattern for an exact match.
1662 We also use the fact that the wildcard must match at least one character,
1663 so the left-most label of the hostname must be at least as large as the
1664 left-most label of the pattern.*/
1665 if(host_label_len<pattern_label_len)return 0;
1666 OP_ASSERT(pattern[pattern_prefix_len]=='*');
1667 /*"The client MAY match a presented identifier in which the wildcard
1668 character is not the only character of the label (e.g., baz*.example.net
1669 and *baz.example.net and b*z.example.net would be taken to match
1670 baz1.example.net and foobaz.example.net and buzz.example.net,
1671 respectively)." [RFC 6125 Section 6.4.3]*/
1672 pattern_suffix_len=pattern_len-pattern_prefix_len-1;
1673 host_suffix_len=_host_len-host_label_len
1674 +pattern_label_len-pattern_prefix_len-1;
1675 return pattern_suffix_len==host_suffix_len
1676 &&op_strncasecmp(_host,pattern,pattern_prefix_len)==0
1677 &&op_strncasecmp(_host+_host_len-host_suffix_len,
1678 pattern+pattern_prefix_len+1,host_suffix_len)==0;
1679}

1694/*Verify the server's hostname matches the certificate they presented using
1695 the procedure from Section 6 of RFC 6125.
1696 Return: 0 if the certificate doesn't match, and a non-zero value if it does.*/
1697static int op_http_verify_hostname(OpusHTTPStream *_stream,SSL *_ssl_conn){
1698 X509 *peer_cert;
1699 STACK_OF(GENERAL_NAME) *san_names;
1700 char *host;
1701 size_t host_len;
1702 int ret;
1703 host=_stream->url.host;
1704 host_len=strlen(host);
1705 peer_cert=SSL_get_peer_certificate(_ssl_conn);
1706 /*We set VERIFY_PEER, so we shouldn't get here without a certificate.*/
1707 if(OP_UNLIKELY(peer_cert==NULL))return 0;
1708 ret=0;
1709 OP_ASSERT(host_len<INT_MAX);
1710 /*RFC 2818 says (after correcting for Eratta 1077): "If a subjectAltName
1711 extension of type dNSName is present, that MUST be used as the identity.
1712 Otherwise, the (most specific) Common Name field in the Subject field of
1713 the certificate MUST be used.

1716 instead."
1717 "Matching is performed using the matching rules specified by RFC 2459.
1718 If more than one identity of a given type is present in the certificate
1719 (e.g., more than one dNSName name), a match in any one of the set is
1720 considered acceptable.
1721 Names may contain the wildcard character * which is condered to match any
1722 single domain name component or component fragment.
1723 E.g., *.a.com matches foo.a.com but not bar.foo.a.com.
1724 f*.com matches foo.com but not bar.com."
1725 "In some cases, the URI is specified as an IP address rather than a
1726 hostname.
1727 In this case, the iPAddress subjectAltName must be present in the
1728 certificate and must exactly match the IP in the URI."*/
1729 san_names=X509_get_ext_d2i(peer_cert,NID_subject_alt_name,NULL,NULL);
1730 if(san_names!=NULL){
1731 struct addrinfo *addr;
1732 unsigned char *ip;
1733 int ip_len;
1734 int nsan_names;
1735 int sni;
1736 /*Check to see if the host was specified as a simple IP address.*/
1737 addr=op_inet_pton(host);
1738 ip=NULL;
1739 ip_len=0;
1740 if(addr!=NULL){
1741 switch(addr->ai_family){
1742 case AF_INET:{
1743 struct sockaddr_in *s;
1744 s=(struct sockaddr_in *)addr->ai_addr;
1745 OP_ASSERT(addr->ai_addrlen>=sizeof(*s));
1746 ip=(unsigned char *)&s->sin_addr;
1747 ip_len=sizeof(s->sin_addr);
1748 }break;
1749 case AF_INET6:{
1750 struct sockaddr_in6 *s;
1751 s=(struct sockaddr_in6 *)addr->ai_addr;
1752 OP_ASSERT(addr->ai_addrlen>=sizeof(*s));
1753 ip=(unsigned char *)&s->sin6_addr;
1754 ip_len=sizeof(s->sin6_addr);
1755 }break;
1756 }
1757 }
1758 /*We can only verify fully-qualified domain names.
1759 To quote RFC 6125: "The extracted data MUST include only information that
1760 can be securely parsed out of the inputs (e.g., parsing the fully
1761 qualified DNS domain name out of the "host" component (or its
1762 equivalent) of a URI or deriving the application service type from the
1763 scheme of a URI) ..."
1764 We don't have a way to check (without relying on DNS records, which might
1765 be subverted) if this address is fully-qualified.
1766 This is particularly problematic when using a CONNECT tunnel, as it is
1767 the server that does DNS lookup, not us.
1768 However, we are certain that if the hostname has no '.', it is definitely
1769 not a fully-qualified domain name (with the exception of crazy TLDs that
1770 actually resolve, like "uz", but I am willing to ignore those).
1771 RFC 1535 says "...in any event where a '.' exists in a specified name it
1772 should be assumed to be a fully qualified domain name (FQDN) and SHOULD
1773 be tried as a rooted name first."
1774 That doesn't give us any security guarantees, of course (a subverted DNS
1775 could fail the original query and our resolver might still retry with a
1776 local domain appended).
1777 If we don't have a FQDN, just set the number of names to 0, so we'll fail
1778 and clean up any resources we allocated.*/
1779 if(ip==NULL&&strchr(host,'.')==NULL)nsan_names=0;
1780 /*RFC 2459 says there MUST be at least one, but we don't depend on it.*/
1781 else nsan_names=sk_GENERAL_NAME_num(san_names);
1782 for(sni=0;sni<nsan_names;sni++){
1783 const GENERAL_NAME *name;
1784 name=sk_GENERAL_NAME_value(san_names,sni);
1785 if(ip==NULL){
1786 if(name->type==GEN_DNS
1787 &&op_http_hostname_match(host,host_len,name->d.dNSName)){
1788 ret=1;
1789 break;
1790 }
1791 }
1792 else if(name->type==GEN_IPADD){
1793 unsigned char *cert_ip;
1794 /*If we do have an IP address, compare it directly.
1795 RFC 6125: "When the reference identity is an IP address, the identity
1796 MUST be converted to the 'network byte order' octet string
1797 representation.
1798 For IP Version 4, as specified in RFC 791, the octet string will
1799 contain exactly four octets.
1800 For IP Version 6, as specified in RFC 2460, the octet string will
1801 contain exactly sixteen octets.
1802 This octet string is then compared against subjectAltName values of
1803 type iPAddress.
1804 A match occurs if the reference identity octet string and the value
1805 octet strings are identical."*/
1806 cert_ip=ASN1_STRING_data(name->d.iPAddress);
1807 if(ip_len==ASN1_STRING_length(name->d.iPAddress)
1808 &&memcmp(ip,cert_ip,ip_len)==0){
1809 ret=1;
1810 break;
1811 }
1812 }
1813 }
1814 sk_GENERAL_NAME_pop_free(san_names,GENERAL_NAME_free);
1815 if(addr!=NULL)freeaddrinfo(addr);
1816 }
1817 /*Do the same FQDN check we did above.
1818 We don't do this once in advance for both cases, because in the
1819 subjectAltName case we might have an IPv6 address without a dot.*/
1820 else if(strchr(host,'.')!=NULL){
1821 int last_cn_loc;
1822 int cn_loc;
1823 /*If there is no subjectAltName, match against commonName.
1824 RFC 6125 says that at least one significant CA is known to issue certs
1825 with multiple CNs, although it SHOULD NOT.
1826 It also says: "The server's identity may also be verified by comparing
1827 the reference identity to the Common Name (CN) value in the last
1828 Relative Distinguished Name (RDN) of the subject field of the server's
1829 certificate (where "last" refers to the DER-encoded order...)."
1830 So find the last one and check it.*/
1831 cn_loc=-1;
1832 do{
1833 last_cn_loc=cn_loc;
1834 cn_loc=X509_NAME_get_index_by_NID(X509_get_subject_name(peer_cert),
1835 NID_commonName,last_cn_loc);
1836 }
1837 while(cn_loc>=0);
1838 ret=last_cn_loc>=0
1839 &&op_http_hostname_match(host,host_len,
1840 X509_NAME_ENTRY_get_data(
1841 X509_NAME_get_entry(X509_get_subject_name(peer_cert),last_cn_loc)));
1842 }
1843 X509_free(peer_cert);
1844 return ret;
1845}

1901/*Try to start a connection to the next address in the given list of a given
1902 type.
1903 _fd: The socket to connect with.
1904 [inout] _addr: A pointer to the list of addresses.
1905 This will be advanced to the first one that matches the given
1906 address family (possibly the current one).
1907 _ai_family: The address family to connect to.
1908 Return: 1 If the connection was successful.
1909 0 If the connection is in progress.
1910 OP_FALSE If the connection failed and there were no more addresses
1911 left to try.
1912 *_addr will be set to NULL in this case.*/
1913static int op_sock_connect_next(op_sock _fd,
1914 const struct addrinfo **_addr,int _ai_family){
1915 const struct addrinfo *addr;
1916 int err;
1917 addr=*_addr;
1918 for(;;){
1919 /*Move to the next address of the requested type.*/
1920 for(;addr!=NULL&&addr->ai_family!=_ai_family;addr=addr->ai_next);
1921 *_addr=addr;
1922 /*No more: failure.*/
1923 if(addr==NULL)return OP_FALSE;
1924 if(connect(_fd,addr->ai_addr,addr->ai_addrlen)>=0)return 1;
1925 err=op_errno();
1926 /*Winsock will set WSAEWOULDBLOCK.*/
1927 if(OP_LIKELY(err==EINPROGRESS||err==EWOULDBLOCK))return 0;
1928 addr=addr->ai_next;
1929 }
1930}

1932/*The number of address families to try connecting to simultaneously.*/
1933# define OP_NPROTOS (2)

1935static int op_http_connect_impl(OpusHTTPStream *_stream,OpusHTTPConn *_conn,
1936 const struct addrinfo *_addrs,struct timeb *_start_time){
1937 const struct addrinfo *addr;
1938 const struct addrinfo *addrs[OP_NPROTOS];
1939 struct pollfd fds[OP_NPROTOS];
1940 int ai_family;
1941 int nprotos;
1942 int ret;
1943 int pi;
1944 int pj;
1945 for(pi=0;pi<OP_NPROTOS;pi++)addrs[pi]=NULL;
1946 /*Try connecting via both IPv4 and IPv6 simultaneously, and keep the first
1947 one that succeeds.
1948 Start by finding the first address from each family.
1949 We order the first connection attempts in the same order the address
1950 families were returned in the DNS records in accordance with RFC 6555.*/
1951 for(addr=_addrs,nprotos=0;addr!=NULL&&nprotos<OP_NPROTOS;addr=addr->ai_next){
1952 if(addr->ai_family==AF_INET6||addr->ai_family==AF_INET){
1953 OP_ASSERT(addr->ai_addrlen<=sizeof(struct sockaddr_in6));
1954 OP_ASSERT(addr->ai_addrlen<=sizeof(struct sockaddr_in));
1955 /*If we've seen this address family before, skip this address for now.*/
1956 for(pi=0;pi<nprotos;pi++)if(addrs[pi]->ai_family==addr->ai_family)break;
1957 if(pi<nprotos)continue;
1958 addrs[nprotos++]=addr;
1959 }
1960 }
1961 /*Pop the connection off the free list and put it on the LRU list.*/
1962 OP_ASSERT(_stream->free_head==_conn);
1963 _stream->free_head=_conn->next;
1964 _conn->next=_stream->lru_head;
1965 _stream->lru_head=_conn;
1966 ftime(_start_time);
1967 *&_conn->read_time=*_start_time;
1968 _conn->read_bytes=0;
1969 _conn->read_rate=0;
1970 /*Try to start a connection to each protocol.
1971 RFC 6555 says it is RECOMMENDED that connection attempts be paced
1972 150...250 ms apart "to balance human factors against network load", but
1973 that "stateful algorithms" (that's us) "are expected to be more
1974 aggressive".
1975 We are definitely more aggressive: we don't pace at all.*/
1976 for(pi=0;pi<nprotos;pi++){
1977 ai_family=addrs[pi]->ai_family;
1978 fds[pi].fd=socket(ai_family,SOCK_STREAM,addrs[pi]->ai_protocol);
1979 fds[pi].events=POLLOUT;
1980 if(OP_LIKELY(fds[pi].fd!=OP_INVALID_SOCKET)){
1981 if(OP_LIKELY(op_sock_set_nonblocking(fds[pi].fd,1)>=0)){
1982 ret=op_sock_connect_next(fds[pi].fd,addrs+pi,ai_family);
1983 if(OP_UNLIKELY(ret>0)){
1984 /*It succeeded right away (technically possible), so stop.*/
1985 nprotos=pi+1;
1986 break;
1987 }
1988 /*Otherwise go on to the next protocol, and skip the clean-up below.*/
1989 else if(ret==0)continue;
1990 /*Tried all the addresses for this protocol.*/
1991 }
1992 /*Clean up the socket.*/
1993 close(fds[pi].fd);
1994 }
1995 /*Remove this protocol from the list.*/
1996 memmove(addrs+pi,addrs+pi+1,sizeof(*addrs)*(nprotos-pi-1));
1997 nprotos--;
1998 pi--;
1999 }
2000 /*Wait for one of the connections to finish.*/
2001 while(pi>=nprotos&&nprotos>0&&poll(fds,nprotos,OP_POLL_TIMEOUT_MS)>0){
2002 for(pi=0;pi<nprotos;pi++){
2003 socklen_t errlen;
2004 int err;
2005 /*Still waiting...*/
2006 if(!fds[pi].revents)continue;
2007 errlen=sizeof(err);
2008 /*Some platforms will return the pending error in &err and return 0.
2009 Others will put it in errno and return -1.*/
2010 ret=getsockopt(fds[pi].fd,SOL_SOCKET,SO_ERROR,&err,&errlen);
2011 if(ret<0)err=op_errno();
2012 /*Success!*/
2013 if(err==0||err==EISCONN)break;
2014 /*Move on to the next address for this protocol.*/
2015 ai_family=addrs[pi]->ai_family;
2016 addrs[pi]=addrs[pi]->ai_next;
2017 ret=op_sock_connect_next(fds[pi].fd,addrs+pi,ai_family);
2018 /*It succeeded right away, so stop.*/
2019 if(ret>0)break;
2020 /*Otherwise go on to the next protocol, and skip the clean-up below.*/
2021 else if(ret==0)continue;
2022 /*Tried all the addresses for this protocol.
2023 Remove it from the list.*/
2024 close(fds[pi].fd);
2025 memmove(fds+pi,fds+pi+1,sizeof(*fds)*(nprotos-pi-1));
2026 memmove(addrs+pi,addrs+pi+1,sizeof(*addrs)*(nprotos-pi-1));
2027 nprotos--;
2028 pi--;
2029 }
2030 }
2031 /*Close all the other sockets.*/
2032 for(pj=0;pj<nprotos;pj++)if(pi!=pj)close(fds[pj].fd);
2033 /*If none of them succeeded, we're done.*/
2034 if(pi>=nprotos)return OP_FALSE;
2035 /*Save this address for future connection attempts.*/
2036 if(addrs[pi]!=&_stream->addr_info){
2037 memcpy(&_stream->addr_info,addrs[pi],sizeof(_stream->addr_info));
2038 _stream->addr_info.ai_addr=&_stream->addr.s;
2039 _stream->addr_info.ai_next=NULL;
2040 memcpy(&_stream->addr,addrs[pi]->ai_addr,addrs[pi]->ai_addrlen);
2041 }
2042 if(OP_URL_IS_SSL(&_stream->url)){
2043 SSL *ssl_conn;
2044 /*Start the SSL connection.*/
2045 OP_ASSERT(_stream->ssl_ctx!=NULL);
2046 ssl_conn=SSL_new(_stream->ssl_ctx);
2047 if(OP_LIKELY(ssl_conn!=NULL)){
2048 ret=op_http_conn_start_tls(_stream,_conn,fds[pi].fd,ssl_conn);
2049 if(OP_LIKELY(ret>=0))return ret;
2050 SSL_free(ssl_conn);
2051 }
2052 close(fds[pi].fd);
2053 _conn->fd=OP_INVALID_SOCKET;
2054 return OP_FALSE;
2055 }
2056 /*Just a normal non-SSL connection.*/
2057 _conn->ssl_conn=NULL;
2058 _conn->fd=fds[pi].fd;
2059 _conn->nrequests_left=OP_PIPELINE_MAX_REQUESTS;
2060 /*Disable write coalescing.
2061 We always send whole requests at once and always parse the response headers
2062 before sending another one.*/
2063 op_sock_set_tcp_nodelay(fds[pi].fd,1);
2064 return 0;
2065}

2607static int op_http_conn_send_request(OpusHTTPStream *_stream,
2608 OpusHTTPConn *_conn,opus_int64 _pos,opus_int32 _chunk_size,
2609 int _try_not_to_block){
2610 opus_int64 next_end;
2611 int ret;
2612 /*We shouldn't have another request outstanding.*/
2613 OP_ASSERT(_conn->next_pos<0);
2614 /*Build the request to send.*/
2615 OP_ASSERT(_stream->request.nbuf>=_stream->request_tail);
2616 _stream->request.nbuf=_stream->request_tail;
2617 ret=op_sb_append_nonnegative_int64(&_stream->request,_pos);
2618 ret|=op_sb_append(&_stream->request,"-",1);
2619 if(_chunk_size>0&&OP_ADV_OFFSET(_pos,2*_chunk_size)<_stream->content_length){
2620 /*We shouldn't be pipelining requests with non-HTTP/1.1 servers.*/
2621 OP_ASSERT(_stream->pipeline);
2622 next_end=_pos+_chunk_size;
2623 ret|=op_sb_append_nonnegative_int64(&_stream->request,next_end-1);
2624 /*Use a larger chunk size for our next request.*/
2625 _chunk_size<<=1;
2626 /*But after a while, just request the rest of the resource.*/
2627 if(_chunk_size>OP_PIPELINE_CHUNK_SIZE_MAX)_chunk_size=-1;
2628 }
2629 else{
2630 /*Either this was a non-pipelined request or we were close enough to the
2631 end to just ask for the rest.*/
2632 next_end=-1;
2633 _chunk_size=-1;
2634 }
2635 ret|=op_sb_append(&_stream->request,"\r\n\r\n",4);
2636 if(OP_UNLIKELY(ret<0))return ret;
2637 /*If we don't want to block, check to see if there's enough space in the send
2638 queue.
2639 There's still a chance we might block, even if there is enough space, but
2640 it's a much slimmer one.
2641 Blocking at all is pretty unlikely, as we won't have any requests queued
2642 when _try_not_to_block is set, so if FIONSPACE isn't available (e.g., on
2643 Linux), just skip the test.*/
2644 if(_try_not_to_block){
2645# if defined(FIONSPACE)
2646 int available;
2647 ret=ioctl(_conn->fd,FIONSPACE,&available);
2648 if(ret<0||available<_stream->request.nbuf)return 1;
2649# endif
2650 }
2651 ret=op_http_conn_write_fully(_conn,
2652 _stream->request.buf,_stream->request.nbuf);
2653 if(OP_UNLIKELY(ret<0))return ret;
2654 _conn->next_pos=_pos;
2655 _conn->next_end=next_end;
2656 /*Save the chunk size to use for the next request.*/
2657 _conn->chunk_size=_chunk_size;
2658 _conn->nrequests_left--;
2659 return ret;
2660}

2662/*Handles the response to all requests after the first one.
2663 Return: 1 if the connection was closed or timed out, 0 on success, or a
2664 negative value on any other error.*/
2665static int op_http_conn_handle_response(OpusHTTPStream *_stream,
2666 OpusHTTPConn *_conn){
2667 char *next;
2668 char *status_code;
2669 opus_int64 range_length;
2670 opus_int64 next_pos;
2671 opus_int64 next_end;
2672 int ret;
2673 ret=op_http_conn_read_response(_conn,&_stream->response);
2674 /*If the server just closed the connection on us, we may have just hit a
2675 connection re-use limit, so we might want to retry.*/
2676 if(OP_UNLIKELY(ret<0))return ret==OP_EREAD?1:ret;
2677 next=op_http_parse_status_line(NULL,&status_code,_stream->response.buf);
2678 if(OP_UNLIKELY(next==NULL))return OP_FALSE;
2679 /*We _need_ a 206 Partial Content response.
2680 Nothing else will do.*/
2681 if(strncmp(status_code,"206",3)!=0){
2682 /*But on a 408 Request Timeout, we might want to re-try.*/
2683 return strncmp(status_code,"408",3)==0?1:OP_FALSE;
2684 }
2685 next_pos=_conn->next_pos;
2686 next_end=_conn->next_end;
2687 range_length=-1;
2688 for(;;){
2689 char *header;
2690 char *cdr;
2691 ret=op_http_get_next_header(&header,&cdr,&next);
2692 if(OP_UNLIKELY(ret<0))return ret;
2693 if(header==NULL)break;
2694 if(strcmp(header,"content-range")==0){
2695 opus_int64 range_first;
2696 opus_int64 range_last;
2697 /*Two Content-Range headers?*/
2698 if(OP_UNLIKELY(range_length>=0))return OP_FALSE;
2699 ret=op_http_parse_content_range(&range_first,&range_last,
2700 &range_length,cdr);
2701 if(OP_UNLIKELY(ret<0))return ret;
2702 /*"A response with satus code 206 (Partial Content) MUST NOT
2703 include a Content-Range field with a byte-range-resp-spec of
2704 '*'."*/
2705 if(OP_UNLIKELY(range_first<0)||OP_UNLIKELY(range_last<0))return OP_FALSE;
2706 /*We also don't want range_last to overflow.*/
2707 if(OP_UNLIKELY(range_last>=OP_INT64_MAX))return OP_FALSE;
2708 range_last++;
2709 /*Quit if we didn't get the offset we asked for.*/
2710 if(range_first!=next_pos)return OP_FALSE;
2711 if(next_end<0){
2712 /*We asked for the rest of the resource.*/
2713 if(range_length>=0){
2714 /*Quit if we didn't get it.*/
2715 if(OP_UNLIKELY(range_last!=range_length))return OP_FALSE;
2716 }
2717 /*If there was no length, use the end of the range.*/
2718 else range_length=range_last;
2719 next_end=range_last;
2720 }
2721 else{
2722 if(range_last!=next_end)return OP_FALSE;
2723 /*If there was no length, use the larger of the content length or the
2724 end of this chunk.*/
2725 if(range_length<0){
2726 range_length=OP_MAX(range_last,_stream->content_length);
2727 }
2728 }
2729 }
2730 else if(strcmp(header,"content-length")==0){
2731 opus_int64 content_length;
2732 /*Validate the Content-Length header, if present, against the request we
2733 made.*/
2734 content_length=op_http_parse_content_length(cdr);
2735 if(OP_UNLIKELY(content_length<0))return (int)content_length;
2736 if(next_end<0){
2737 /*If we haven't seen the Content-Range header yet and we asked for the
2738 rest of the resource, set next_end, so we can make sure they match
2739 when we do find the Content-Range header.*/
2740 if(OP_UNLIKELY(next_pos>OP_INT64_MAX-content_length))return OP_FALSE;
2741 next_end=next_pos+content_length;
2742 }
2743 /*Otherwise, make sure they match now.*/
2744 else if(OP_UNLIKELY(next_end-next_pos!=content_length))return OP_FALSE;
2745 }
2746 else if(strcmp(header,"connection")==0){
2747 ret=op_http_parse_connection(cdr);
2748 if(OP_UNLIKELY(ret<0))return ret;
2749 /*If the server told us it was going to close the connection, don't make
2750 any more requests.*/
2751 if(OP_UNLIKELY(ret>0))_conn->nrequests_left=0;
2752 }
2753 }
2754 /*No Content-Range header.*/
2755 if(OP_UNLIKELY(range_length<0))return OP_FALSE;
2756 /*Update the content_length if necessary.*/
2757 _stream->content_length=range_length;
2758 _conn->pos=next_pos;
2759 _conn->end_pos=next_end;
2760 _conn->next_pos=-1;
2761 return 0;
2762}

2764/*Open a new connection that will start reading at byte offset _pos.
2765 _pos: The byte offset to start reading from.
2766 _chunk_size: The number of bytes to ask for in the initial request, or -1 to
2767 request the rest of the resource.
2768 This may be more bytes than remain, in which case it will be
2769 converted into a request for the rest.*/
2770static int op_http_conn_open_pos(OpusHTTPStream *_stream,
2771 OpusHTTPConn *_conn,opus_int64 _pos,opus_int32 _chunk_size){
2772 struct timeb start_time;
2773 struct timeb end_time;
2774 opus_int32 connect_rate;
2775 opus_int32 connect_time;
2776 int ret;
2777 ret=op_http_connect(_stream,_conn,&_stream->addr_info,&start_time);
2778 if(OP_UNLIKELY(ret<0))return ret;
2779 ret=op_http_conn_send_request(_stream,_conn,_pos,_chunk_size,0);
2780 if(OP_UNLIKELY(ret<0))return ret;
2781 ret=op_http_conn_handle_response(_stream,_conn);
2782 if(OP_UNLIKELY(ret!=0))return OP_FALSE;
2783 ftime(&end_time);
2784 _stream->cur_conni=_conn-_stream->conns;
2785 OP_ASSERT(_stream->cur_conni>=0&&_stream->cur_conni<OP_NCONNS_MAX);
2786 /*The connection has been successfully opened.
2787 Update the connection time estimate.*/
2788 connect_time=op_time_diff_ms(&end_time,&start_time);
2789 connect_rate=_stream->connect_rate;
2790 connect_rate+=OP_MAX(connect_time,1)-connect_rate+8>>4;
2791 _stream->connect_rate=connect_rate;
2792 return 0;
2793}

2795/*Read data from the current response body.
2796 If we're pipelining and we get close to the end of this response, queue
2797 another request.
2798 If we've reached the end of this response body, parse the next response and
2799 keep going.
2800 [out] _buf: Returns the data read.
2801 _buf_size: The size of the buffer.
2802 Return: A positive number of bytes read on success.
2803 0: The connection was closed.
2804 OP_EREAD: There was a fatal read error.*/
2805static int op_http_conn_read_body(OpusHTTPStream *_stream,
2806 OpusHTTPConn *_conn,unsigned char *_buf,int _buf_size){
2807 opus_int64 pos;
2808 opus_int64 end_pos;
2809 opus_int64 next_pos;
2810 opus_int64 content_length;
2811 int nread;
2812 int pipeline;
2813 int ret;
2814 /*Currently this function can only be called on the LRU head.
2815 Otherwise, we'd need a _pnext pointer if we needed to close the connection,
2816 and re-opening it would re-organize the lists.*/
2817 OP_ASSERT(_stream->lru_head==_conn);
2818 /*We should have filtered out empty reads by this point.*/
2819 OP_ASSERT(_buf_size>0);
2820 pos=_conn->pos;
2821 end_pos=_conn->end_pos;
2822 next_pos=_conn->next_pos;
2823 pipeline=_stream->pipeline;
2824 content_length=_stream->content_length;
2825 if(end_pos>=0){
2826 /*Have we reached the end of the current response body?*/
2827 if(pos>=end_pos){
2828 OP_ASSERT(content_length>=0);
2829 /*If this was the end of the stream, we're done.
2830 Also return early if a non-blocking read was requested (regardless of
2831 whether we might be able to parse the next response without
2832 blocking).*/
2833 if(content_length<=end_pos)return 0;
2834 /*Otherwise, start on the next response.*/
2835 if(next_pos<0){
2836 /*We haven't issued another request yet.*/
2837 if(!pipeline||_conn->nrequests_left<=0){
2838 /*There are two ways to get here: either the server told us it was
2839 going to close the connection after the last request, or we
2840 thought we were reading the whole resource, but it grew while we
2841 were reading it.
2842 The only way the latter could have happened is if content_length
2843 changed while seeking.
2844 Open a new request to read the rest.*/
2845 OP_ASSERT(_stream->seekable);
2846 /*Try to open a new connection to read another chunk.*/
2847 op_http_conn_close(_stream,_conn,&_stream->lru_head,1);
2848 /*If we're not pipelining, we should be requesting the rest.*/
2849 OP_ASSERT(pipeline||_conn->chunk_size==-1);
2850 ret=op_http_conn_open_pos(_stream,_conn,end_pos,_conn->chunk_size);
2851 if(OP_UNLIKELY(ret<0))return OP_EREAD;
2852 }
2853 else{
2854 /*Issue the request now (better late than never).*/
2855 ret=op_http_conn_send_request(_stream,_conn,pos,_conn->chunk_size,0);
2856 if(OP_UNLIKELY(ret<0))return OP_EREAD;
2857 next_pos=_conn->next_pos;
2858 OP_ASSERT(next_pos>=0);
2859 }
2860 }
2861 if(next_pos>=0){
2862 /*We shouldn't be trying to read past the current request body if we're
2863 seeking somewhere else.*/
2864 OP_ASSERT(next_pos==end_pos);
2865 ret=op_http_conn_handle_response(_stream,_conn);
2866 if(OP_UNLIKELY(ret<0))return OP_EREAD;
2867 if(OP_UNLIKELY(ret>0)&&pipeline){
2868 opus_int64 next_end;
2869 next_end=_conn->next_end;
2870 /*Our request timed out or the server closed the connection.
2871 Try re-connecting.*/
2872 op_http_conn_close(_stream,_conn,&_stream->lru_head,1);
2873 /*Unless there's a bug, we should be able to convert
2874 (next_pos,next_end) into valid (_pos,_chunk_size) parameters.*/
2875 OP_ASSERT(next_end<0
2876 ||next_end-next_pos>=0&&next_end-next_pos<=OP_INT32_MAX);
2877 ret=op_http_conn_open_pos(_stream,_conn,next_pos,
2878 next_end<0?-1:(opus_int32)(next_end-next_pos));
2879 if(OP_UNLIKELY(ret<0))return OP_EREAD;
2880 }
2881 else if(OP_UNLIKELY(ret!=0))return OP_EREAD;
2882 }
2883 pos=_conn->pos;
2884 end_pos=_conn->end_pos;
2885 content_length=_stream->content_length;
2886 }
2887 OP_ASSERT(end_pos>pos);
2888 _buf_size=OP_MIN(_buf_size,end_pos-pos);
2889 }
2890 nread=op_http_conn_read(_conn,(char *)_buf,_buf_size,1);
2891 if(OP_UNLIKELY(nread<0))return nread;
2892 pos+=nread;
2893 _conn->pos=pos;
2894 OP_ASSERT(end_pos<0||content_length>=0);
2895 /*TODO: If nrequests_left<=0, we can't make a new request, and there will be
2896 a big pause after we hit the end of the chunk while we open a new
2897 connection.
2898 It would be nice to be able to start that process now, but we have no way
2899 to do it in the background without blocking (even if we could start it, we
2900 have no guarantee the application will return control to us in a
2901 sufficiently timely manner to allow us to complete it, and this is
2902 uncommon enough that it's not worth using threads just for this).*/
2903 if(end_pos>=0&&end_pos<content_length&&next_pos<0
2904 &&pipeline&&OP_LIKELY(_conn->nrequests_left>0)){
2905 opus_int64 request_thresh;
2906 opus_int32 chunk_size;
2907 /*Are we getting close to the end of the current response body?
2908 If so, we should request more data.*/
2909 request_thresh=_stream->connect_rate*_conn->read_rate>>12;
2910 /*But don't commit ourselves too quickly.*/
2911 chunk_size=_conn->chunk_size;
2912 if(chunk_size>=0)request_thresh=OP_MIN(chunk_size>>2,request_thresh);
2913 if(end_pos-pos<request_thresh){
2914 ret=op_http_conn_send_request(_stream,_conn,end_pos,_conn->chunk_size,1);
2915 if(OP_UNLIKELY(ret<0))return OP_EREAD;
2916 }
2917 }
2918 return nread;
2919}

2921static int op_http_stream_read(void *_stream,
2922 unsigned char *_ptr,int _buf_size){
2923 OpusHTTPStream *stream;
2924 ptrdiff_t nread;
2925 opus_int64 size;
2926 opus_int64 pos;
2927 int ci;
2928 stream=(OpusHTTPStream *)_stream;
2929 /*Check for an empty read.*/
2930 if(_buf_size<=0)return 0;
2931 ci=stream->cur_conni;
2932 /*No current connection => EOF.*/
2933 if(ci<0)return 0;
2934 pos=stream->conns[ci].pos;
2935 size=stream->content_length;
2936 /*Check for EOF.*/
2937 if(size>=0){
2938 if(pos>=size)return 0;
2939 /*Check for a short read.*/
2940 if(_buf_size>size-pos)_buf_size=(int)(size-pos);
2941 }
2942 nread=op_http_conn_read_body(stream,stream->conns+ci,_ptr,_buf_size);
2943 if(OP_UNLIKELY(nread<=0)){
2944 /*We hit an error or EOF.
2945 Either way, we're done with this connection.*/
2946 op_http_conn_close(stream,stream->conns+ci,&stream->lru_head,1);
2947 stream->cur_conni=-1;
2948 stream->pos=pos;
2949 }
2950 return nread;
2951}

2953/*Discard data until we reach the _target position.
2954 This destroys the contents of _stream->response.buf, as we need somewhere to
2955 read this data, and that is a convenient place.
2956 _just_read_ahead: Whether or not this is a plain fast-forward.
2957 If 0, we need to issue a new request for a chunk at _target
2958 and discard all the data from our current request(s).
2959 Otherwise, we should be able to reach _target without
2960 issuing any new requests.
2961 _target: The stream position to which to read ahead.*/
2962static int op_http_conn_read_ahead(OpusHTTPStream *_stream,
2963 OpusHTTPConn *_conn,int _just_read_ahead,opus_int64 _target){
2964 opus_int64 pos;
2965 opus_int64 end_pos;
2966 opus_int64 next_pos;
2967 opus_int64 next_end;
2968 ptrdiff_t nread;
2969 int ret;
2970 pos=_conn->pos;
2971 end_pos=_conn->end_pos;
2972 next_pos=_conn->next_pos;
2973 next_end=_conn->next_end;
2974 if(!_just_read_ahead){
2975 /*We need to issue a new pipelined request.
2976 This is the only case where we allow more than one outstanding request
2977 at a time, so we need to reset next_pos (we'll restore it below if we
2978 did have an outstanding request).*/
2979 OP_ASSERT(_stream->pipeline);
2980 _conn->next_pos=-1;
2981 ret=op_http_conn_send_request(_stream,_conn,_target,
2982 OP_PIPELINE_CHUNK_SIZE,0);
2983 if(OP_UNLIKELY(ret<0))return ret;
2984 }
2985 /*We can reach the target position by reading forward in the current chunk.*/
2986 if(_just_read_ahead&&(end_pos<0||_target<end_pos))end_pos=_target;
2987 else if(next_pos>=0){
2988 opus_int64 next_next_pos;
2989 opus_int64 next_next_end;
2990 /*We already have a request outstanding.
2991 Finish off the current chunk.*/
2992 while(pos<end_pos){
2993 nread=op_http_conn_read(_conn,_stream->response.buf,
2994 (int)OP_MIN(end_pos-pos,_stream->response.cbuf),1);
2995 /*We failed to read ahead.*/
2996 if(nread<=0)return OP_FALSE;
2997 pos+=nread;
2998 }
2999 OP_ASSERT(pos==end_pos);
3000 if(_just_read_ahead){
3001 next_next_pos=next_next_end=-1;
3002 end_pos=_target;
3003 }
3004 else{
3005 OP_ASSERT(_conn->next_pos==_target);
3006 next_next_pos=_target;
3007 next_next_end=_conn->next_end;
3008 _conn->next_pos=next_pos;
3009 _conn->next_end=next_end;
3010 end_pos=next_end;
3011 }
3012 ret=op_http_conn_handle_response(_stream,_conn);
3013 if(OP_UNLIKELY(ret!=0))return OP_FALSE;
3014 _conn->next_pos=next_next_pos;
3015 _conn->next_end=next_next_end;
3016 }
3017 while(pos<end_pos){
3018 nread=op_http_conn_read(_conn,_stream->response.buf,
3019 (int)OP_MIN(end_pos-pos,_stream->response.cbuf),1);
3020 /*We failed to read ahead.*/
3021 if(nread<=0)return OP_FALSE;
3022 pos+=nread;
3023 }
3024 OP_ASSERT(pos==end_pos);
3025 if(!_just_read_ahead){
3026 ret=op_http_conn_handle_response(_stream,_conn);
3027 if(OP_UNLIKELY(ret!=0))return OP_FALSE;
3028 }
3029 else _conn->pos=end_pos;
3030 OP_ASSERT(_conn->pos==_target);
3031 return 0;
3032}

3034static int op_http_stream_seek(void *_stream,opus_int64 _offset,int _whence){
3035 struct timeb seek_time;
3036 OpusHTTPStream *stream;
3037 OpusHTTPConn *conn;
3038 OpusHTTPConn **pnext;
3039 OpusHTTPConn *close_conn;
3040 OpusHTTPConn **close_pnext;
3041 opus_int64 content_length;
3042 opus_int64 pos;
3043 int pipeline;
3044 int ci;
3045 int ret;
3046 stream=(OpusHTTPStream *)_stream;
3047 if(!stream->seekable)return -1;
3048 content_length=stream->content_length;
3049 /*If we're seekable, we should have gotten a Content-Length.*/
3050 OP_ASSERT(content_length>=0);
3051 ci=stream->cur_conni;
3052 pos=ci<0?content_length:stream->conns[ci].pos;
3053 switch(_whence){
3054 case SEEK_SET:{
3055 /*Check for overflow:*/
3056 if(_offset<0)return -1;
3057 pos=_offset;
3058 }break;
3059 case SEEK_CUR:{
3060 /*Check for overflow:*/
3061 if(_offset<-pos||_offset>OP_INT64_MAX-pos)return -1;
3062 pos+=_offset;
3063 }break;
3064 case SEEK_END:{
3065 /*Check for overflow:*/
3066 if(_offset>content_length||_offset<content_length-OP_INT64_MAX)return -1;
3067 pos=content_length-_offset;
3068 }break;
3069 default:return -1;
3070 }
3071 /*Mark when we deactivated the active connection.*/
3072 if(ci>=0){
3073 op_http_conn_read_rate_update(stream->conns+ci);
3074 *&seek_time=*&stream->conns[ci].read_time;
3075 }
3076 else ftime(&seek_time);
3077 /*If we seeked past the end of the stream, just disable the active
3078 connection.*/
3079 if(pos>=content_length){
3080 stream->cur_conni=-1;
3081 stream->pos=pos;
3082 return 0;
3083 }
3084 /*First try to find a connection we can use without waiting.*/
3085 pnext=&stream->lru_head;
3086 conn=stream->lru_head;
3087 while(conn!=NULL){
3088 opus_int64 conn_pos;
3089 opus_int64 end_pos;
3090 int available;
3091 /*If this connection has been dormant too long or has made too many
3092 requests, close it.
3093 This is to prevent us from hitting server limits/firewall timeouts.*/
3094 if(op_time_diff_ms(&seek_time,&conn->read_time)>
3095 OP_CONNECTION_IDLE_TIMEOUT_MS
3096 ||conn->nrequests_left<OP_PIPELINE_MIN_REQUESTS){
3097 op_http_conn_close(stream,conn,pnext,1);
3098 conn=*pnext;
3099 continue;
3100 }
3101 available=op_http_conn_estimate_available(conn);
3102 conn_pos=conn->pos;
3103 end_pos=conn->end_pos;
3104 if(conn->next_pos>=0){
3105 OP_ASSERT(end_pos>=0);
3106 OP_ASSERT(conn->next_pos==end_pos);
3107 end_pos=conn->next_end;
3108 }
3109 OP_ASSERT(end_pos<0||conn_pos<=end_pos);
3110 /*Can we quickly read ahead without issuing a new request or waiting for
3111 any more data?
3112 If we have an oustanding request, we'll over-estimate the amount of data
3113 it has available (because we'll count the response headers, too), but
3114 that probably doesn't matter.*/
3115 if(conn_pos<=pos&&pos-conn_pos<=available&&(end_pos<0||pos<end_pos)){
3116 /*Found a suitable connection to re-use.*/
3117 ret=op_http_conn_read_ahead(stream,conn,1,pos);
3118 if(OP_UNLIKELY(ret<0)){
3119 /*The connection might have become stale, so close it and keep going.*/
3120 op_http_conn_close(stream,conn,pnext,1);
3121 conn=*pnext;
3122 continue;
3123 }
3124 /*Sucessfully resurrected this connection.*/
3125 *pnext=conn->next;
3126 conn->next=stream->lru_head;
3127 stream->lru_head=conn;
3128 stream->cur_conni=conn-stream->conns;
3129 return 0;
3130 }
3131 pnext=&conn->next;
3132 conn=conn->next;
3133 }
3134 /*Chances are that didn't work, so now try to find one we can use by reading
3135 ahead a reasonable amount and/or by issuing a new request.*/
3136 close_pnext=NULL;
3137 close_conn=NULL;
3138 pnext=&stream->lru_head;
3139 conn=stream->lru_head;
3140 pipeline=stream->pipeline;
3141 while(conn!=NULL){
3142 opus_int64 conn_pos;
3143 opus_int64 end_pos;
3144 opus_int64 read_ahead_thresh;
3145 int available;
3146 int just_read_ahead;
3147 /*Dividing by 2048 instead of 1000 scales this by nearly 1/2, biasing away
3148 from connection re-use (and roughly compensating for the lag required to
3149 reopen the TCP window of a connection that's been idle).
3150 There's no overflow checking here, because it's vanishingly unlikely, and
3151 all it would do is cause us to make poor decisions.*/
3152 read_ahead_thresh=OP_MAX(OP_READAHEAD_THRESH_MIN,
3153 stream->connect_rate*conn->read_rate>>11);
3154 available=op_http_conn_estimate_available(conn);
3155 conn_pos=conn->pos;
3156 end_pos=conn->end_pos;
3157 if(conn->next_pos>=0){
3158 OP_ASSERT(end_pos>=0);
3159 OP_ASSERT(conn->next_pos==end_pos);
3160 end_pos=conn->next_end;
3161 }
3162 OP_ASSERT(end_pos<0||conn_pos<=end_pos);
3163 /*Can we quickly read ahead without issuing a new request?*/
3164 just_read_ahead=conn_pos<=pos&&pos-conn_pos-available<=read_ahead_thresh
3165 &&(end_pos<0||pos<end_pos);
3166 if(just_read_ahead||pipeline&&end_pos>=0
3167 &&end_pos-conn_pos-available<=read_ahead_thresh){
3168 /*Found a suitable connection to re-use.*/
3169 ret=op_http_conn_read_ahead(stream,conn,just_read_ahead,pos);
3170 if(OP_UNLIKELY(ret<0)){
3171 /*The connection might have become stale, so close it and keep going.*/
3172 op_http_conn_close(stream,conn,pnext,1);
3173 conn=*pnext;
3174 continue;
3175 }
3176 /*Sucessfully resurrected this connection.*/
3177 *pnext=conn->next;
3178 conn->next=stream->lru_head;
3179 stream->lru_head=conn;
3180 stream->cur_conni=conn-stream->conns;
3181 return 0;
3182 }
3183 close_pnext=pnext;
3184 close_conn=conn;
3185 pnext=&conn->next;
3186 conn=conn->next;
3187 }
3188 /*No suitable connections.
3189 Open a new one.*/
3190 if(stream->free_head==NULL){
3191 /*All connections in use.
3192 Expire one of them (we should have already picked which one when scanning
3193 the list).*/
3194 OP_ASSERT(close_conn!=NULL);
3195 OP_ASSERT(close_pnext!=NULL);
3196 op_http_conn_close(stream,close_conn,close_pnext,1);
3197 }
3198 OP_ASSERT(stream->free_head!=NULL);
3199 conn=stream->free_head;
3200 /*If we can pipeline, only request a chunk of data.
3201 If we're seeking now, there's a good chance we will want to seek again
3202 soon, and this avoids committing this connection to reading the rest of
3203 the stream.
3204 Particularly with SSL or proxies, issuing a new request on the same
3205 connection can be substantially faster than opening a new one.
3206 This also limits the amount of data the server will blast at us on this
3207 connection if we later seek elsewhere and start reading from a different
3208 connection.*/
3209 ret=op_http_conn_open_pos(stream,conn,pos,
3210 pipeline?OP_PIPELINE_CHUNK_SIZE:-1);
3211 if(OP_UNLIKELY(ret<0)){
3212 op_http_conn_close(stream,conn,&stream->lru_head,1);
3213 return -1;
3214 }
3215 return 0;
3216}

3325 succeeds, or for clearing *_info if it ultimately fails.*/
3326void *op_url_stream_vcreate_impl(OpusFileCallbacks *_cb,
3327 const char *_url,OpusServerInfo *_info,OpusServerInfo **_pinfo,va_list _ap){
3328 int skip_certificate_check;
3329 const char *proxy_host;
3330 opus_int32 proxy_port;
3331 const char *proxy_user;
3332 const char *proxy_pass;
3333 OpusServerInfo *pinfo;
3334 skip_certificate_check=0;
3335 proxy_host=NULL;
3336 proxy_port=8080;
3337 proxy_user=NULL;
3338 proxy_pass=NULL;
3339 pinfo=NULL;
3340 for(;;){
3341 ptrdiff_t request;
3342 request=va_arg(_ap,char *)-(char *)NULL;
3343 /*If we hit NULL, we're done processing options.*/
3344 if(!request)break;
3345 switch(request){
3346 case OP_SSL_SKIP_CERTIFICATE_CHECK_REQUEST:{
3347 skip_certificate_check=!!va_arg(_ap,opus_int32);
3348 }break;
3349 case OP_HTTP_PROXY_HOST_REQUEST:{
3350 proxy_host=va_arg(_ap,const char *);
3351 }break;
3352 case OP_HTTP_PROXY_PORT_REQUEST:{
3353 proxy_port=va_arg(_ap,opus_int32);
3354 if(proxy_port<0||proxy_port>(opus_int32)65535)return NULL;
3355 }break;
3356 case OP_HTTP_PROXY_USER_REQUEST:{
3357 proxy_user=va_arg(_ap,const char *);
3358 }break;
3359 case OP_HTTP_PROXY_PASS_REQUEST:{
3360 proxy_pass=va_arg(_ap,const char *);
3361 }break;
3362 case OP_GET_SERVER_INFO_REQUEST:{
3363 pinfo=va_arg(_ap,OpusServerInfo *);
3364 }break;
3365 /*Some unknown option.*/
3366 default:return NULL;
3367 }
3368 }
3369 /*If the caller has requested server information, proxy it to a local copy to
3370 simplify error handling.*/
3371 *_pinfo=NULL;
3372 if(pinfo!=NULL){
3373 void *ret;
3374 opus_server_info_init(_info);
3375 ret=op_url_stream_create_impl(_cb,_url,skip_certificate_check,
3376 proxy_host,proxy_port,proxy_user,proxy_pass,_info);
3377 if(ret!=NULL)*_pinfo=pinfo;
3378 else opus_server_info_clear(_info);
3379 return ret;
3380 }
3381 return op_url_stream_create_impl(_cb,_url,skip_certificate_check,
3382 proxy_host,proxy_port,proxy_user,proxy_pass,NULL);
3383}

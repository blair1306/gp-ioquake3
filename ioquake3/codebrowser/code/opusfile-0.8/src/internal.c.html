26/*A version of strncasecmp() that is guaranteed to only ignore the case of
27 ASCII characters.*/
28int op_strncasecmp(const char *_a,const char *_b,int _n){
29 int i;
30 for(i=0;i<_n;i++){
31 int a;
32 int b;
33 int d;
34 a=_a[i];
35 b=_b[i];
36 if(a>='a'&&a<='z')a-='a'-'A';
37 if(b>='a'&&b<='z')b-='a'-'A';
38 d=a-b;
39 if(d)return d;
40 }
41 return 0;
42}
